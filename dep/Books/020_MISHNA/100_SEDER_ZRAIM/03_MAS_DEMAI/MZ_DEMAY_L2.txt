<b>הַקַּלִּין שֶׁבַּדְּמַאי.</b> שֶׁהֵקֵלּוּ חֲכָמִים עַל הַפֵּרוֹת הַלָּלוּ הַנִּזְכָּרִים בְּמַתְנִיתִין שֶׁלֹּא לְעַשְּׂרָן דְּמַאי, לְפִי שֶׁחֶזְקָתָן בָּאִים מִן הַהֶפְקֵר, מִשּׁוּם דְּלֹא חֲשִׁיבֵי, וְאִית בְּהוּ תְּרֵי סְפֵקֵי, סָפֵק בָּאִים מִן הַהֶפְקֵר וּפְטוּרִים מִן הַמַּעֲשֵׂר, וַאֲפִלּוּ אִם תִּמְצָא לוֹמַר שֶׁבָּאִים מִן הַשָּׁמוּר וְחַיָּבִים בְּמַעֲשֵׂר, שֶׁמָּא נִתְעַשְּׂרוּ. וּדְמַאי הֵם הַתְּבוּאָה וְהַפֵּרוֹת הַנִּלְקָחִים מֵעַמֵּי הָאָרֶץ שֶׁהֵם חֲשׁוּדִים עַל הַמַּעַשְׂרוֹת. וְהֶחָבֵר הַלּוֹקֵחַ פֵּרוֹת מֵעַמֵּי הָאָרֶץ צָרִיךְ לְהַפְרִישׁ מֵהֶם תְּרוּמַת מַעֲשֵׂר וּמַעֲשֵׂר שֵׁנִי בִּלְבָד אִם הִיא שְׁנַת מַעֲשֵׂר שֵׁנִי {א}, אֲבָל תְּרוּמָה גְדוֹלָה לֹא נֶחְשְׁדוּ עַמֵּי הָאָרֶץ עָלֶיהָ לְפִי שֶׁהִיא בְּמִיתָה, וְאֵין לָהּ שִׁעוּר שֶׁחִטָּה אַחַת פּוֹטֶרֶת כָּל הַכְּרִי, וְהַכֹּל נִזְהָרִים בָּהּ. וּמַעֲשֵׂר רִאשׁוֹן וּמַעֲשַׂר עָנִי אֵינוֹ צָרִיךְ לְהוֹצִיא מִן הַדְּמַאי אַף עַל פִּי שֶׁעַמֵּי הָאָרֶץ חֲשׁוּדִים עֲלֵיהֶם, לְפִי שֶׁיָּכוֹל בַּעַל הַפֵּרוֹת לוֹמַר לַלֵּוִי אוֹ לֶעָנִי הַבָּאִים לִטֹּל הַמַּעַשְׂרוֹת, הָבֵא רְאָיָה שֶׁאֵין פֵּרוֹת הַלָּלוּ מְעֻשָּׂרִים וְטֹּל, דְּקַיְמָא לָן בְּכָל דּוּכְתָּא הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה. אֲבָל בִּתְרוּמַת מַעֲשֵׂר לֵיכָּא לְמֵימַר הָכִי, שֶׁהָאוֹכֵל טֶבֶל שֶׁלֹּא הוּרְמוּ מִמֶּנּוּ מַתְּנוֹת כְּהֻנָּה חַיָּב מִיתָה וּמִשּׁוּם אִסּוּרָא מַפְרִישִׁינַן לַהּ לִתְרוּמַת מַעֲשֵׂר {ב}, וּמַעֲשֵׂר שֵׁנִי נַמִּי כְּדֵי שֶׁלֹּא לְאָכְלוֹ בְּטֻמְאָה אוֹ שֶׁלֹּא לְאָכְלוֹ חוּץ לִירוּשָׁלַיִם: <B>הַשִּׁיתִין.</B> תְּאֵנִים מִדְבָּרִיּוֹת: <B>וְהָרִימִין.</B> פֵּרֵשׁ הֶעָרוּךְ פולצרא"קי בְּלַעַ"ז: <B>וְעֻזְרָדִין.</B> בַּעֲרָבִי זערו"ד וּבְלַעַ"ז סורבי"ש: <B>וּבְנוֹת שׁוּחַ.</B> תְּאֵנִים לְבָנוֹת הַמִּתְגַּדְּלוֹת מִשָּׁלֹשׁ שָׁנִים לְשָׁלֹשׁ שָׁנִים, וּמִתְגַּדְּלִים בַּיְעָרִים: <B>וּבְנוֹת שִׁקְמָה.</B> תְּאֵנָה הַמֻּרְכֶּבֶת בְּעַרְמוֹן: <B>וְנוֹבְלוֹת תְּמָרָה.</B> תְּמָרִים שֶׁאֵינָם מִתְבַּשְּׁלִין בָּאִילָן וְתוֹלְשִׁין אוֹתָן וּמַנִּיחִים אוֹתָם זֶה עַל זֶה עַד שֶׁיִּתְבַּשְּׁלוּ. וְיֵשׁ אוֹמְרִים נוֹבְלוֹת תְּמָרָה תְּמָרִים שֶׁהִפִּילָתָן הָרוּחַ קֹדֶם בִּשּׁוּלָן: <B>גֻּפְנִים.</B> עֲנָבִים שֶׁמַּנִּיחִים אוֹתָן בַּגֶּפֶן בְּסוֹף הַבָּצִיר וּבְקֹשִׁי מִתְבַּשְּׁלִים: <B>הַנִּצְפָּה.</B> צָלָף שֶׁקּוֹרִין קאיפר"י בְּלַעַ"ז: <B>הָאוֹג.</B> פְּרִי אָדֹם שֶׁקּוֹרִין לוֹ בְּלַעַ"ז קורניאול"י. וְרַמְבַּ"ם אוֹמֵר שֶׁהוּא אִילָן שֶׁעוֹשֶׂה כְּמִין אֶשְׁכּוֹלוֹת אֲדֻמּוֹת אֲדֻמּוֹת וְקוֹרִין לוֹ בַּעֲרָבִי סומא"ק. וְלֹא הָיָה זֶה הַפְּרִי חָשׁוּב בִּיהוּדָה: <B>וְהַחֹמֶץ.</B> בִּתְחִלָּה לֹא הָיָה הַיַּיִן שֶׁבִּיהוּדָה מַחְמִיץ לְפִי שֶׁהָיוּ מְבִיאִים מִמֶּנּוּ נְסָכִים, וְכָל הַחֹמֶץ שֶׁבִּיהוּדָה לֹא הָיָה בָּא אֶלָּא מִן הַתֶּמֶד, לְפִיכָךְ הַחֹמֶץ שֶׁבִּיהוּדָה פָּטוּר. אֲבָל לְאַחַר שֶׁבָּטְלוּ נְסָכִים וְהַחֹמֶץ בָּא מִן הַיַּיִן, הַחֹמֶץ חַיָּב בִּדְמַאי אַף בִּיהוּדָה כְּמוֹ בִשְׁאָר מְקוֹמוֹת: <B>וְהַכֻּסְבָּר.</B> כִּזְרַע גַּד (שְׁמוֹת טז) תַּרְגּוּם יְרוּשַׁלְמִי כֻּסְבָּר, כוליאנדר"ו בְּלַעַ"ז: <B>דְּיוּפְרָא.</B> שֶׁטּוֹעֵן פֵּרוֹתָיו שְׁנֵי פְעָמִים בַּשָּׁנָה. דְּיוּ בִּלְשׁוֹן יָוָן שְׁנַיִם: <B>שִׁקְמוֹנָה.</B> שֵׁם מָקוֹם: <B>הַמֻּסְטָפוֹס.</B> שֶׁנִּתְבַּשְּׁלוּ בָּאִילָן עַד שֶׁנִּתְבַּקְּעוּ מֵאֲלֵיהֶן. שֶׁכָּל אֵלּוּ חֲשׁוּבִים הֵן וְחֶזְקָתָן שֶׁאֵין בָּאִין מִן הַהֶפְקֵר. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

<B>הַדְּמַאי אֵין לוֹ חֹמֶשׁ.</B> הַפּוֹדֶה {ד} מַעֲשֵׂר שֵׁנִי שֶׁל דְּמַאי אֵינוֹ נוֹתֵן אֶת הַחֹמֶשׁ, מִשּׁוּם דְּרֹב עַמֵּי הָאָרֶץ מְעַשְּׂרִין הֵם. הִלְכָּךְ קֶרֶן דִּמְעַכֵּב בִּדְאוֹרַיְתָא אִיתָא בִּדְרַבָּנָן, חֹמֶשׁ דְּלֹא מְעַכֵּב בִּדְאוֹרַיְתָא יֵאָכֵל: <B>וְאֵין לוֹ בִּעוּר.</B> בְּסוֹף שָׁלֹשׁ שָׁנִים, שֶׁחַיָּב לְבַעֵר כָּל מַעַשְׂרוֹתָיו כְּדִכְתִיב (דְּבָרִים כו) בִּעַרְתִּי הַקֹּדֶשׁ מִן הַבַּיִת, וְלֹא תִקְּנוּהוּ רַבָּנָן בִּדְמַאי: <B>וְנֶאֱכָל לְאוֹנֵן.</B> דְּמַעֲשֵׂר שֵׁנִי שֶׁל וַדַּאי אָסוּר לָאוֹנֵן כְּדִכְתִיב (שָׁם) לֹא אָכַלְתִּי בְאוֹנִי מִמֶּנּוּ, וּמַעֲשֵׂר שֵׁנִי שֶׁל דְּמַאי לֹא גָזְרוּ בוֹ: <B>וְנִכְנָס לִירוּשָׁלַיִם וְיוֹצֵא.</B> נִפְדֶּה וְנֶאֱכָל חוּץ לִירוּשָׁלַיִם, מַה שֶּׁאֵין כֵּן בְּוַדַּאי שֶׁהַמְּחִצּוֹת קוֹלְטוֹת וְאֵין יְכוֹלִין לִפְדּוֹתוֹ וּלְהוֹצִיאוֹ חוּץ לַחוֹמָה לְאַחַר שֶׁנִכְנַס, וּבִדְמַאי לֹא גָזְרוּ: <B>וּמְאַבְּדִים מִעוּטוֹ בַּדְּרָכִים.</B> אִם הָיָה מַעֲשֵׂר שֵׁנִי שֶׁל דְּמַאי בַּדְּרָכִים בִּמְקוֹם גְּדוּדֵי חַיָּה וְלִסְטִים, אֲפִלּוּ הָיָה דָּבָר מֻעָט וְיָכוֹל לַהֲבִיאוֹ בְּלֹא טֹרַח וּבְלֹא הֶפְסֵד, אֵינוֹ חוֹשֵׁשׁ לוֹ וּמַנִּיחַ אוֹתוֹ לֵאָבֵד בַּדֶּרֶךְ בַּמָּקוֹם שֶׁהוּא שָׁם. וְרַמְבַּ"ם פֵּרֵשׁ שֶׁאִם בְּשָׁעָה שֶׁמּוֹלִיךְ מַעֲשֵׂר שֵׁנִי שֶׁל דְּמַאי בִּירוּשָׁלַיִם נֶאֱבַד מִמֶּנּוּ מְעַט בַּדֶּרֶךְ אֵינוֹ חוֹשֵׁשׁ לוֹ, מַה שֶּׁאֵין כֵּן בְּוַדַּאי שֶׁצָּרִיךְ שֶׁיַּעֲלֵהוּ כֻּלּוֹ אוֹ דָמָיו וְלֹא יִפָּסֵד מִמֶּנּוּ כְּלוּם בַּדְּרָכִים: <B>וְנוֹתְנִים אוֹתוֹ לְעַם הָאָרֶץ.</B> לְאָכְלוֹ בִּירוּשָׁלַיִם וְאַף עַל פִּי שֶׁהוּא חָשׁוּד לְאָכְלוֹ בְּטֻמְאָה, מַה שֶּׁאֵין כֵּן בְּוַדַּאי שֶׁאֵין מוֹסְרִים אוֹתוֹ לְעַם הָאָרֶץ לְפִי שֶׁאֵינוֹ נִזְהָר לְאָכְלוֹ בְּטָהֳרָה: <B>וְיֹאכַל כְּנֶגְדוֹ.</B> כְּנֶגֶד מַה שֶּׁנָּתַן לְעַם הָאָרֶץ לוֹקֵחַ מִנְּכָסָיו וְאוֹכֵל בְּטָהֳרַת מַעֲשֵׂר בִּירוּשָׁלַיִם: <B>וּמְחַלְּלִים אוֹתוֹ כֶּסֶף עַל כֶּסֶף.</B> אִם יֵשׁ לוֹ כֶּסֶף מַעֲשֵׂר שֵׁנִי שֶׁל דְּמַאי וְהוּא צָרִיךְ לוֹ יָכוֹל לְחַלְּלוֹ עַל כֶּסֶף חֻלִּין, מַה שֶּׁאֵין כֵּן בְּוַדַּאי, שֶׁאֵין מְחַלְּלִין כֶּסֶף עַל כֶּסֶף אֶלָּא בִּסְלָעִים שֶׁל שְׁנֵי מְלָכִים שֶׁיָּכוֹל לְחַלֵּל זֶה עַל זֶה אִם הַשֵּׁנִי חָרִיף, אֲבָל שֶׁל מֶלֶךְ אֶחָד לֹא דְּלָאו דֶּרֶךְ חִלּוּל הוּא, וּבִדְמַאי שָׁרֵי: <B>כֶּסֶף עַל נְחֹשֶׁת.</B> אֲפִלּוּ שֶׁלֹּא מִדֹּחַק. אֲבָל שֶׁל וַדַּאי מִדֹּחַק אִין, שֶׁלֹּא מִדֹּחַק לֹא: <B>וְיַחֲזֹר וְיִפְדֶּה אֶת הַפֵּרוֹת.</B> גָּרְסִינַן, וְלֹא גָרְסִינַן וּבִלְבָד שֶׁיַּחֲזֹר וְיִפְדֶּה {ה}. וְהָכִי קָאָמַר וְיָכוֹל לַחֲזֹר וְלִפְדּוֹת אֶת הַפֵּרוֹת אִם יִרְצֶה, דִּבְרֵי רַבִּי מֵאִיר, וַחֲכָמִים אוֹמְרִים אֵינוֹ יָכוֹל <B>לִהְיוֹת חָבֵר.</B> לְעִנְיַן טָהֳרוֹת, דְּהַיְנוּ פָּרוּשׁ, וְיִהְיוּ בְגָדָיו וּמַשְׁקִים שֶׁלּוֹ טְהוֹרִים. וַאֲפִלּוּ תַּלְמִיד חָכָם אֵינוֹ נֶאֱמָן לְעִנְיַן טָהֳרוֹת עַד שֶׁיְּקַבֵּל עָלָיו דִּבְרֵי חֲבֵרוּת, אֶלָּא אִם כֵּן הָיָה זָקֵן וְיוֹשֵׁב בִּישִׁיבָה. וְהַמְקַבֵּל עָלָיו דִּבְרֵי חֲבֵרוּת צָרִיךְ לְהַרְגִּיל עַצְמוֹ שְׁלֹשִׁים יוֹם וְאַחַר כָּךְ יִהְיוּ בְּגָדָיו וּמַשְׁקִים שֶׁלּוֹ טְהוֹרִים. וְאֵין קַבָּלַת דִּבְרֵי חֲבֵרוּת בְּפָחוֹת מִשְּׁלֹשָׁה חֲבֵרִים, אֶלָּא אִם כֵּן הָיָה הַמְקַבֵּל תַּלְמִיד חָכָם שֶׁאֵין צָרִיךְ בִּפְנֵי שְׁלֹשָׁה {ו}, וְלֹא עוֹד אֶלָּא שֶׁאֲחֵרִים מְקַבְּלִים לְפָנָיו: <B>לַח וְיָבֵשׁ.</B> דְּאֵין מוֹסְרִים טָהֳרוֹת לְעַם הָאָרֶץ, שֶׁאָסוּר לִגְרֹם טֻמְאָה לְחֻלִּין שֶׁבְּאֶרֶץ יִשְׂרָאֵל: <B>וְאֵין לוֹקֵחַ מִמֶּנּוּ לַח.</B> אֲבָל יָבֵשׁ לוֹקֵחַ מִמֶּנּוּ, שֶׁלֹּא הֻכְשַׁר לְקַבֵּל טֻמְאָה כָּל זְמַן שֶׁלֹּא בָא עָלָיו מַשְׁקֶה. וְנֶאֱמָן עַם הָאָרֶץ לוֹמַר הַפֵּרוֹת הַלָּלוּ לֹא הֻכְשְׁרוּ {ז}, אֲבָל אֵינוֹ נֶאֱמָן לוֹמַר הֻכְשְׁרוּ אֲבָל לֹא נִטְמְאוּ: <B>וְלֹא יִתְאָרֵחַ אֵצֶל עַם הָאָרֶץ {ח}.</B> שֶׁלֹּא יִטָּמֵּא וְיָבֹא וִיטַמֵּא טָהֳרוֹת: <B>וְלֹא מְאָרְחוֹ.</B> לְעַם הָאָרֶץ: <B>אֶצְלוֹ בִּכְסוּתוֹ.</B> דִּכְסוּת עַם הָאָרֶץ טֻמְאָתוֹ חֲמוּרָה מִטֻּמְאַת עַם הָאָרֶץ עַצְמוֹ, דְּחַיְשִׁינַן שֶׁמָּא יָשְׁבָה עֲלֵיהֶן אִשְׁתּוֹ נִדָּה, וּבִגְדֵי עַם הָאָרֶץ מִדְרָס לַפְּרוּשִׁים. אִי נַמִּי לְכָךְ אָמְרוּ לֹא מְאָרְחוֹ אֶצְלוֹ בִּכְסוּתוֹ, דְּמִמַּגַּע עַצְמוֹ יָכוֹל לִזָּהֵר טְפֵי מִמַּגַּע כְּסוּתוֹ: <B>אַף לֹא יְגַדֵּל בְּהֵמָה דַקָּה.</B> שֶׁאָסוּר לְגַדֵּל בְּהֵמָה [דַקָּה] בְּאֶרֶץ יִשְׂרָאֵל, שֶׁלֹּא יִרְעוּ בִּשְׂדוֹת אֲחֵרִים: <B>וְלֹא יְהֵא פָרוּץ בִּנְדָרִים.</B> שֶׁסּוֹפוֹ לָבֹא לִידֵי חִלּוּל: <B>וְלֹא יְהֵא פָרוּץ בִּשְׂחוֹק.</B> דִּשְׂחוֹק וְקַלּוּת רֹאשׁ מַרְגִּילִין אֶת הָאָדָם לְעֶרְוָה: <B>וּמְשַׁמֵּשׁ.</B> תַּלְמִידֵי חֲכָמִים בְּבֵית הַמִּדְרָשׁ: <B>לֹא בָאוּ אֵלּוּ לַכְּלָל.</B> דְּאֵין עִנְיָנָם נוֹגֵעַ לְטָהֳרוֹת. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

<B>מַאֲכִילִין אֶת הָעֲנִיִּים דְּמָאי.</B> אֲפִלּוּ הֵן חֲבֵרִים. וְצָרִיךְ לְהוֹדִיעָן, וְהָרוֹצֶה לְעַשְּׂרָן יְעַשְּׂרָן: <B>וְאֶת הָאַכְסַנְיָא.</B> חַיִל שֶׁל מֶלֶךְ יִשְׂרָאֵל הָעוֹבֵר מִמָּקוֹם לְמָקוֹם וּמֻטָּל עַל בְּנֵי הַמָּקוֹם לְפַרְנְסָם, מַאֲכִילִין אוֹתָם דְּמַאי, בִּזְמַן שֶׁהִיא עוֹבֶרֶת, אֲבָל לָנוּ שָׁם בַּלַּיְלָה חַיָּבִין לְתַקֵּן {א}: <B>רַבָּן גַּמְלִיאֵל הָיָה מַאֲכִיל אֶת פּוֹעֲלָיו דְּמַאי.</B> עֲנִיִּים הָיוּ. וְאֵין הֲלָכָה כְּרַבָּן גַּמְלִיאֵל, דְּכֵיוָן דְּחַיָּב בִּמְזוֹנוֹתֵיהֶן נִמְצָא פּוֹרֵעַ אֶת חוֹבוֹ בִּדְמַאי: <B>וְאֶת שֶׁאֵינוֹ מְעֻשָּׂר לִמְעַשֵּׂר.</B> וְנוֹתְנִין לוֹ בְּיוֹתֵר מִכְּדֵי מַעַשְׂרוֹתָיו, וְנִמְצְאוּ כָל הָאָדָם אוֹכְלִים מְתֻקָּן. וּבֵית שַׁמַּאי לְטַעֲמַיְהוּ דְּאָמְרֵי אֵין מַאֲכִילִין אֶת הָעֲנִיִּים דְּמָאי {ב}: <B>וְהָרוֹצֶה לְתַקֵּן יְתַקֵּן.</B> שֶׁלֹּא הֻתַּר הַדְּמַאי לָעֲנִיִּים אֶלָּא כְּשֶׁאוֹכְלִין אֲכִילָה אַחַת אֵצֶל בַּעַל הַבַּיִת, אֲבָל הַפֵּרוֹת שֶׁיָּבֹאוּ לְיָדָם חַיָּבִים הֵם לְעַשְּׂרָן דְּמַאי. דַּיְקָא נַמִּי דְּקָתָנֵי מַאֲכִילִין אֶת הָעֲנִיִּים דְּמַאי וְלֹא תָנֵי הָעֲנִיִּים אוֹכְלִין דְּמַאי. כָּךְ כָּתַב הָרַמְבַּ"ם {ג}:

<B>לַחֲזֹם.</B> לִכְרֹת וּלְחַתֵּךְ. וְאֵין לוֹ דּוֹמֶה בַּמִּשְׁנָה: <B>לְהָקֵל מִמַּשָּׂאוֹ.</B> שֶׁלֹּא יִכְבַּד עָלָיו: <B>לֹא יַשְׁלִיךְ עַד שֶׁיְּעַשֵּׂר.</B> שֶׁמָּא יִמְצָאֵם עַם הָאָרֶץ וְיֹאכַל אוֹתָם כְּשֶׁאֵינָם מְתֻקָּנִים, וְנִמְצָא נוֹתֵן מִכְשׁוֹל. וְהַנֵּי מִילֵי בְּיָרָק הַנֶּאֱגַד, דְּאִי בְּשֶׁלֹּא נֶאֱגַד אַכַּתִּי לֹא אִתְחַיֵּב בְּמַעֲשֵׂר {ד}: <B>הַלּוֹקֵחַ יָרָק מִן הַשּׁוּק.</B> וְנָטַל אֲגֻדּוֹת יָרָק בְּיָדוֹ שֶׁשּׁוּמָתָן יְדוּעָה כָּךְ וְכָךְ אֲגֻדּוֹת בִּפְרוּטָה, וּלְאַחַר שֶׁנְּטָלָן נִמְלַךְ עֲלֵיהֶן לְהַחֲזִירָן לַמּוֹכֵר: <B>לֹא יַחֲזִיר עַד שֶׁיְּעַשֵּׂר.</B> וְיִתֵּן לוֹ דְּמֵי הַמַּעֲשֵׂר שֶׁהוּא מַפְרִישׁ, דִּמְכִי אַגְבְּהִינְהוּ {ו} נַעֲשׂוּ שֶׁלּוֹ וְנִתְחַיֵּב עֲלֵיהֶן לְעַשְּׂרָן: <B>שֶׁאֵינוֹ מְעֻשָּׂר אֶלָּא מִנְיָן.</B> הַאי מְעֻשָּׂר הֲוֵי פֵּרוּשָׁא מְחֻסָּר, כְּלוֹמַר כְּבָר קָנָה אוֹתָם בְּהַגְבָּהָה וְלֹא הָיָה מְחֻסָּר אֶלָּא לִמְנוֹתָן, שֶׁכְּבָר הָיָה יָדוּעַ כָּךְ וְכָךְ אֲגֻדּוֹת בִּפְרוּטָה: <B>הָיָה עוֹמֵד וְלוֹקֵחַ.</B> וְלֹא נְטָלָם בְּיָדוֹ וְלֹא הִגְבִּיהָם: <B>טֹעַן אַחֵר.</B> מַשְּׂאוֹי אַחֵר שֶׁל יְרָקוֹת: 

~ פרק ג - משנה ג
<B>לֹא יַצְנִיעַ עַד שֶׁיְּעַשֵּׂר.</B> דְּאָסוּר לְהוֹצִיא מִיָּדוֹ דָּבָר שֶׁאֵינוֹ מְתֻקָּן: <B>שֶׁלֹּא יֹאבֵדוּ.</B> וְלֹא נִתְכַּוֵּן לִזְכּוֹת בָּהֶן: <B>שֶׁאֵין אָדָם רַשַּׁאי לְמָכְרוֹ דְּמָאי.</B> כְּגוֹן בְּמִדָּה דַקָּה דְּאָמְרִינַן בְּפִרְקִין דִּלְעֵיל: <B>לֹא יִשְׁלַח לַחֲבֵרוֹ דְּמָאי.</B> בְּדַקָּה, אֲבָל שׁוֹלֵחַ לוֹ בְּגַסָּה כֵּיוָן שֶׁרַשַּׁאי לְמָכְרוֹ בְּגַסָּה: <B>רַבִּי יוֹסֵי מַתִּיר.</B> לִשְׁלֹחַ לַחֲבֵרוֹ וַדַּאי אֲפִלּוּ בְּדַקָּה, וּבִלְבָד שֶׁיּוֹדִיעוֹ. וּמוֹדֶה רַבִּי יוֹסֵי שֶׁבִּדְמַאי אָסוּר בְּדַקָּה. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: 

~ פרק ג - משנה ד
<B>לְטוֹחֵן כּוּתִי.</B> לְכוּתִי שֶׁטּוֹחֵן הַתְּבוּאָה: <B>הֲרֵי הֵן בְּחֶזְקָתָן.</B> דְּלֹא חֲשִׁידֵי שֶׁמָּא הֶחְלִיפוּ אֵלּוּ הַמְתֻקָּנִים בַּאֲחֵרִים שֶׁאֵינָן מְתֻקָּנִין: <B>לְטוֹחֵן עוֹבֵד כּוֹכָבִים דְּמָאי.</B> דְּעוֹבֵד כּוֹכָבִים חָשׁוּד לְהַחְלִיף. וּלְבָתַר דְּאַשְׁכְּחוּ יוֹנָה בְּרֹאשׁ הַר גְּרִיזִים לַכּוּתִים שֶׁהָיוּ עוֹבְדִים אוֹתָהּ, כּוּתִי הֲרֵי הוּא כְּעוֹבֵד כּוֹכָבִים: <B>אֵצֶל הָעוֹבֵד כּוֹכָבִים כְּפֵרוֹתָיו.</B> וּפֵרוֹת שֶׁל עוֹבֵד כּוֹכָבִים פְּטוּרִים {יב} מִן הַמַּעֲשֵׂר. וְרֵישָׁא מִפְּנֵי שֶׁדֶּרֶךְ בְּנֵי אָדָם לְהוֹלִיךְ קֻפּוֹת רַבּוֹת שֶׁל חִטִּים וּלְהַנִּיחָן אֵצֶל הַטּוֹחֵן חַיְשִׁינַן שֶׁמָּא נִתְחַלְּפָה קֻפָּה זוֹ בְּקֻפּוֹת אֲחֵרוֹת שֶׁל יִשְׂרָאֵל שֶׁהֵן דְּמַאי, אֲבָל פֵּרוֹת לֵיכָּא לְמֵימַר שֶׁמָּא הֶחְלִיפָן בְּפֵרוֹת שֶׁל יִשְׂרָאֵל אֶלָּא שֶׁמָּא הֶחְלִיפָן הָעוֹבֵד כּוֹכָבִים בְּפֵרוֹתָיו, לְפִיכָךְ הֲרֵי הֵן כְּפֵרוֹתָיו שֶׁל עוֹבֵד כּוֹכָבִים. וְרַבִּי שִׁמְעוֹן סָבַר דְּאַף בְּפֵרוֹת חַיְשִׁינַן שֶׁמָּא נִתְחַלְּפוּ בְּבֵית הָעוֹבֵד כּוֹכָבִים בְּפֵרוֹת שֶׁל יִשְׂרָאֵל אַחֵר, שֶׁכְּשֵׁם שֶׁיִּשְׂרָאֵל זֶה הִפְקִיד פֵּרוֹתָיו בְּיַד הָעוֹבֵד כּוֹכָבִים כָּךְ יֵשׁ לָחוּשׁ שֶׁמָּא יִשְׂרָאֵל אַחֵר מִן הַחֲשׁוּדִים עַל הַמַּעַשְׂרוֹת הִפְקִיד גַּם כֵּן בְּיָדוֹ וְנִתְחַלְּפוּ אֵלּוּ בָאֵלּוּ. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: 

~ פרק ג - משנה ה
<B>פֻּנְדָּקִית.</B> שׁוֹמֶרֶת הַפֻּנְדָּק וְהוֹלְכֵי דְרָכִים מִתְאַכְסְנִים אֶצְלָהּ: <B>מְעַשֵּׂר אֶת שֶׁהוּא נוֹתֵן לָהּ.</B> דְּאֵין חָבֵר מוֹצִיא מִתַּחַת יָדוֹ דָּבָר שֶׁאֵינוֹ מְתֻקָּן {יג}: <B>וְאֶת שֶׁהוּא נוֹטֵל מִמֶּנָּה.</B> דְּאַף עַל גַּב דְּעַם הָאָרֶץ אֵינוֹ חָשׁוּד לְהַחְלִיף כִּדְאָמְרָן לְעֵיל, הַפֻּנְדָּקִית חֲשׁוּדָה לְהַחְלִיף שֶׁהִיא מִתְכַּוֶּנֶת לְטוֹבָה וְנוֹתֶנֶת מִשֶּׁלָּהּ שֶׁהוּא יוֹתֵר יָפֶה לֶחָבֵר, וְאוֹמֶרֶת בְּלִבָּהּ רָאוּי שֶׁאַאֲכִיל מִשֶּׁלִּי שֶׁהוּא חַם וְיָפֶה לֶחָבֵר וַאֲנִי אֶטֹּל אֶת שֶׁלּוֹ הַקַּר וְהַגָּרוּעַ: <B>אֵין אָנוּ אַחֲרָאִין לָרַמָּאִים.</B> כְּלוֹמַר אֵין אַחְרָיוּת הָרַמָּאִים עָלֵינוּ לְשָׁמְרָם שֶׁלֹּא יֹאכְלוּ דָּבָר שֶׁאֵינוֹ מְעֻשָּׂר. לְפִיכָךְ אֵינוֹ מְעַשֵּׂר מַה שֶּׁהוּא נוֹתֵן לָהּ, וְאִם תִּקָּחֶנּוּ הַפֻּנְדָּקִית לְעַצְמָהּ וְתֹאכְלֶנּוּ כְּשֶׁאֵינוֹ מְתֻקָּן אֵין לֶחָבֵר מִזֶּה כְּלוּם, וְאֵינוֹ מְעַשֵּׂר אֶלָּא מַה שֶּׁהוּא נוֹטֵל בִּלְבָד. וְרַבִּי יוֹסֵי סָבַר דְּלִגְזֹל הִיא מִתְכַּוֶּנֶת {יד}, וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: 

~ פרק ג - משנה ו
<B>הַמִּתְקַלְקֵל.</B> פַּת אוֹ תַּבְשִׁיל שֶׁנִּתְקַלְקֵל: <B>אָמַר רַבִּי יְהוּדָה.</B> בַּיְרוּשַׁלְמִי מְפָרֵשׁ דְּקָמַיְתָא נַמִּי מִלְּתֵיהּ דְּרַבִּי יְהוּדָה הִיא. וְהָכִי קָאָמַר מִפְּנֵי שֶׁהִיא חֲשׁוּדָה לְהַחְלִיף אֶת הַמִּתְקַלְקֵל, דִּבְרֵי רַבִּי יְהוּדָה, שֶׁרַבִּי יְהוּדָה אוֹמֵר רוֹצָה הִיא בְּתַקָּנַת בִּתָּהּ וְכוּ': <B>בְּנוֹתֵן לַחֲמוֹתוֹ שְׁבִיעִית.</B> בְּנוֹתֵן לָהּ לֶאֱפוֹת וּלְבַשֵּׁל בַּשָּׁנָה הַשְּׁבִיעִית: <B>שֶׁאֵינָהּ חֲשׁוּדָה לְהַאֲכִיל אֶת בִּתָּהּ שְׁבִיעִית.</B> דַּחֲמִירָא לְהוּ שְׁבִיעִית וַאֲפִלּוּ אִם יִתְקַלְקֵל לֹא תַחְלִיפֶנּוּ בְּפֵרוֹת שְׁבִיעִית. וַהֲלָכָה כְּרַבִּי יְהוּדָה:

~ פרק ד - משנה א
<B>הַלּוֹקֵחַ פֵּרוֹת.</B> בְּשַׁבָּת יֹאכַל עַל פִּיו. דְּאֵימַת שַׁבָּת עָלָיו, שֶׁמִּתְיָרְאִים לַעֲבֹר עֲבֵרָה וּלְשַׁקֵּר בְּשַׁבָּת יוֹתֵר מִבְּחֹל: <B>תְּרוּמַת מַעֲשֵׂר שֶׁל דְּמַאי.</B> לְהָכִי נָקַט תְּרוּמַת מַעֲשֵׂר, דְּעַמֵּי הָאָרֶץ נֶחְשְׁדוּ עָלֶיהָ דְּסָבְרֵי אֵינָהּ בְּמִיתָה כָּל זְמַן שֶׁלֹּא הֻפְרַשׁ מַעֲשֵׂר רִאשׁוֹן, אֲבָל תְּרוּמָה גְדוֹלָה לֹא נֶחְשְׁדוּ עָלֶיהָ. וּתְרוּמַת מַעֲשֵׂר הַוְיָא אֶחָד מִמֵּאָה, וּכְשֶׁחָזְרָה לִמְקוֹמָהּ לְפָחוֹת מִמֵּאָה חָזְרָה וּמְדַמַּעַת: <B>שׁוֹאֲלוֹ וְאוֹכֵל עַל פִּיו.</B> דְּמֵאַחַר שֶׁנִּקְרָא לָהּ שֵׁם תְּרוּמָה אֵימַת דִּמּוּעַ עַל עַם הָאָרֶץ וְיָרֵא לְשַׁקֵּר. וַהֲלָכָה כְּרַבִּי שִׁמְעוֹן שְׁזוּרִי: 

~ פרק ד - משנה ב
<B>הַמַּדִּיר אֶת חֲבֵרוֹ שֶׁיֹּאכַל עִמּוֹ.</B> דְּאָמַר לֵיהּ קוֹנָם מַה שֶּׁאַתָּה נֶהֱנֶה לִי {ז} אִם אֵין אַתָּה סוֹעֵד אֶצְלִי: <B>שַׁבָּת הָרִאשׁוֹנָה.</B> שֶׁל סְעֻדַּת נִשּׂוּאִין לְבָחוּר שֶׁנָּשָׂא בְתוּלָה הִתִּירוּ לוֹ לֶאֱכֹל עִמּוֹ מִשּׁוּם אֵיבָה: 

~ פרק ד - משנה ג
<B>אֵין צָרִיךְ לִקְרוֹת שֵׁם לְמַעְשַׂר עָנִי שֶׁל דְּמַאי.</B> דְּלֹא נֶחְשְׁדוּ עַמֵּי הָאָרֶץ עַל מַעְשַׂר עָנִי, שֶׁהֵם יוֹדְעִים שֶׁפֵּרוֹת הַטְּבוּלִים לְמַעְשַׂר עָנִי הֵם בְּמִיתָה, וּמַפְרִישִׁים אוֹתוֹ וְלוֹקְחִים אוֹתוֹ לְעַצְמָן: <B>וַחֲכָמִים אוֹמְרִים.</B> אַף עַל פִּי כֵן קוֹרֵא לוֹ שֵׁם, שֶׁהֲרֵי אֵינוֹ מַפְסִיד כְּלוּם בָּזֶה {ט}, דְּאֵינוֹ צָרִיךְ לְהַפְרִישׁ וְלִתֵּן לֶעָנִי מִשּׁוּם דְּעָנִי הֲוֵי בָּזֶה מוֹצִיא מֵחֲבֵרוֹ וְהַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה: 

~ פרק ד - משנה ד
<B>מִי שֶׁקָּרָא שֵׁם לִתְרוּמַת מַעֲשֵׂר שֶׁל דְּמַאי.</B> שֶׁאָמַר תְּרוּמַת מַעֲשֵׂר שֶׁאֲנִי חַיָּב לְהַפְרִישׁ מִכֶּרִי זֶה יְהֵא מֻנָּח בִּצְפוֹנוֹ אוֹ בִדְרוֹמוֹ, וְלֹא הִפְרִישָׁהּ. וּתְרוּמַת מַעֲשֵׂר שֶׁל דְּמַאי דְּנָקַט לְפִי שֶׁהַיִּשְׂרָאֵל קוֹרֵא לָהּ שֵׁם וְנוֹתְנָהּ לַכֹּהֵן וְלוֹקֵחַ הַמַּעֲשֵׂר לְעַצְמוֹ, אֲבָל תְּרוּמַת מַעֲשֵׂר שֶׁל וַדַּאי אֵין הַיִּשְׂרָאֵל קוֹרֵא לָהּ שֵׁם אֶלָּא הוּא נוֹתֵן מַעֲשֵׂר לַלֵּוִי וְהַלֵּוִי מַפְרִישׁ תְּרוּמַת מַעֲשֵׂר וְנוֹתְנָהּ לַכֹּהֵן: <B>לֹא יִטְּלֵם.</B> כְּדֵי לִתְּנָם לַכֹּהֵן {יב} אוֹ לֶעָנִי שֶׁבֶּחָצֵר וְשֶׁבַּמָּבוֹי, שֶׁאָסוּר לִתֵּן מַתָּנוֹת לַכֹּהֵן וְלֶעָנִי בְּשַׁבָּת. אֲבָל בִּלְמוּדִים לֶאֱכֹל אֶצְלוֹ כְּלוֹמַר הָרְגִילִין לֶאֱכֹל עַל שֻׁלְחָנוֹ שָׁרֵי, וּבִלְבָד שֶׁיּוֹדִיעֵם שֶׁהֵם שֶׁל תְּרוּמַת מַעֲשֵׂר אוֹ שֶׁל מַעְשַׂר עָנִי, שֶׁאִם אֵינוֹ מוֹדִיעָם וְהֵם סְבוּרִים שֶׁמִּשֶּׁלּוֹ הוּא מַאֲכִילָם הֲוָה לֵיהּ כְּמַאֲכִיל אוֹרְחָיו תְּרוּמָה וּמַעַשְׂרוֹת, שֶׁהוּא אָסוּר: 

~ פרק ד - משנה ה
<B>אֵינוֹ נֶאֱמָן.</B> דַּאֲפִלּוּ יִמָּצֵא שֶׁלָּקַח מִמִּי שֶׁאֵינוֹ נֶאֱמָן מָצֵי לְאִשְׁתַּמּוּטֵי וְלוֹמַר בְּעֵינַי הָיָה נֶאֱמָן. אֲבָל כִּי אָמַר לֵיהּ מֵאִישׁ פְּלוֹנִי לֹא מָצֵי לְאִשְׁתַּמוֹטֵי דְּהָא אֵינוֹ רַשַּׁאי לִקַּח מֵאַחֵר: 

~ פרק ד - משנה ו
<B>אִישׁ פְּלוֹנִי נֶאֱמָן הֲרֵי זֶה נֶאֱמָן.</B> אַף עַל גַּב דְּלֹא הָיָה רָאוּי לְהַאֲמִינוֹ כֵּיוָן שֶׁהוּא עַצְמוֹ חָשׁוּד, קֻלָּא הִיא שֶׁהֵקֵלּוּ בְּאַכְסְנַאי מִשּׁוּם חַיֵּי נֶפֶשׁ. וְדַוְקָא כְּשֶׁאֵין מַכִּיר אָדָם שָׁם אֲבָל אִם מַכִּיר אָדָם שָׁם, לֹא יִטֹּל אֶלָּא מִן הַמֻּמְחֶה: <B>מִי כָּאן מוֹכֵר יָשָׁן.</B> שֶׁיָּרֵא שֶׁמָּא יַאֲכִילוּהוּ חָדָשׁ קֹדֶם שֶׁיִּקְרַב הָעֹמֶר. וְרֹב עַמֵּי הָאָרֶץ אֵין חֲשׁוּדִים עַל הֶחָדָשׁ וַהֲוֵי כִּדְמַאי דְּרֹב עַמֵּי הָאָרֶץ מְעַשְּׂרִים הֵן, הִלְכָּךְ לֹא הֶחְמִירוּ עֲלֵיהֶן כָּל כָּךְ דְּנֵימָא שֶׁהֵם גּוֹמְלִים חֶסֶד זֶה לָזֶה, תָּעִיד אַתָּה עָלַי וַאֲנִי אָעִיד עָלֶיךָ: 

~ פרק ד - משנה ז
<B>הַחַמָּרִים.</B> הַמְּבִיאִים תְּבוּאָה מִמְּקוֹם הַזּוֹל לִמְקוֹם הַיֹּקֶר: <B>אֵין נֶאֱמָנִים.</B> דְּוַדַּאי גּוֹמְלִים זֶה אֶת זֶה וּמְשַׁבֵּחַ שֶׁל חֲבֵרוֹ בְּעִיר זוֹ כְּדֵי שֶׁיְּשַׁבַּח חֲבֵרוֹ אֶת שֶׁלּוֹ בְּעִיר אַחֶרֶת: <B>רַבִּי יְהוּדָה אוֹמֵר נֶאֱמָנִים.</B> הוֹאִיל וְרֹב עַמֵּי הָאָרֶץ מְעַשְּׂרִים הֵן, בִּדְמַאי הֵקֵלּוּ מִשּׁוּם חַיֵּיהֶן שֶׁל בְּנֵי הָעִיר, שֶׁיִּהְיוּ מוֹכְרֵי תְבוּאָה וּפֵרוֹת רְגִילִין לָבֹא שָׁם {טז}. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

~ פרק ה - משנה א
<B>הַלּוֹקֵחַ מִן הַנַּחְתּוֹם.</B> כְּשֶׁהַנַּחְתּוֹם מוֹכֵר בְּמִדָּה גַסָּה הַלּוֹקֵחַ מִמֶּנּוּ חַיָּב לְתַקֵּן מַה שֶּׁהוּא לוֹקֵחַ, וְהַנַּחְתּוֹם פָּטוּר. וְהָא דִתְנַן לְעֵיל בְּפֶרֶק ב' שֶׁהַנַּחְתּוֹמִים חַיָּבִים לְהַפְרִישׁ כְּדֵי שִׁעוּר תְּרוּמַת מַעֲשֵׂר וְחַלָּה, הָתָם מַיְרֵי בְּמוֹכֵר בְּמִדָּה דַקָּה, שֶׁמִּפְּנֵי שֶׁהוּא מִשְׂתַּכֵּר הַרְבֵּה הִטִּילוּ עָלָיו לְעַשֵּׂר, אִי נַמִּי מִפְּנֵי הַתִּינוֹקוֹת שֶׁלּוֹקְחִים מִמִּי שֶׁמּוֹכֵר בְּדַקָּה כְּדֵי שֶׁלֹּא יֹאכְלוּ טְבָלִים הִטִּילוּ עָלָיו לְתַקֵּן. מִדָּה גַסָּה וְדַקָּה מְפֹרָשִׁים לְעֵיל סוֹף פֶּרֶק ב': <B>נוֹטֵל כְּדֵי תְּרוּמַת מַעֲשֵׂר.</B> דְּהַיְנוּ אֶחָד מִמֵּאָה, וּכְדֵי חַלָּה אֶחָד מֵאַרְבָּעִים וּשְׁמֹנָה. וְלֹא שֶׁיַּפְרִישָׁם מִן הַכִּכָּר שֶׁלָּקַח מִן הַנַּחְתּוֹם, אֶלָּא קוֹבֵעַ לָהֶם מָקוֹם וּמַנִּיחָם מְחֻבָּרִים בִּמְקוֹמָם: <B>אֶחָד מִמֵּאָה מִמַּה שֶׁיֶּשׁ כָּאן.</B> מִכָּל מַה שֶׁיֶּשׁ כָּאן מַשְׁמַע, אַף עִם כְּדֵי שִׁעוּר חַלָּה שֶׁהִפְרִישׁ, עָשׂוּי מַעֲשֵׂר, וְנִמְצָא מַפְרִישׁ מַעֲשֵׂר אַף מִן הַחַלָּה: <B>וּשְׁאָר מַעֲשֵׂר.</B> דְּהַיְנוּ עוֹד תִּשְׁעָה סָמוּךְ לוֹ, הֲרֵי קָבַע מָקוֹם לְכָל הַמַּעֲשֵׂר, וְאוֹתוֹ הָאֶחָד שֶׁקָּרָאתִי לוֹ שֵׁם מַעֲשֵׂר תְּחִלָּה עָשׂוּי תְּרוּמַת מַעֲשֵׂר עַל הַתִּשְׁעָה הַסְּמוּכִים לוֹ: <B>וְהַשְּׁאָר חַלָּה.</B> אֶחָד מֵאַרְבָּעִים וּשְׁמֹנָה שֶׁהִפְרַשְׁתִּי חַלָּה: <B>וּמַעֲשֵׂר שֵׁנִי בִּצְפוֹנוֹ אוֹ בִדְרוֹמוֹ.</B> קוֹבֵעַ לוֹ מָקוֹם וְאַחַר כָּךְ מְחַלְּלוֹ, וְאֵינוֹ צָרִיךְ לְהוֹסִיף חֹמֶשׁ: 

~ פרק ה - משנה ב
<B>נוֹטֵל אֶחָד מִשְּׁלֹשִׁים וְשָׁלֹשׁ וּשְׁלִישׁ.</B> הַיְנוּ שְׁלִישׁ מִמֵּאָה. וּסְאָה שֶׁנּוֹטֵל יֵשׁ בָּהּ שְׁלֹשָׁה שְׁלִישִׁים, שְׁנַיִם מֵהֶן לִתְרוּמָה שֶׁהִיא תְּרֵי מִמֵּאָה וְאֶחָד לִתְרוּמַת מַעֲשֵׂר שֶׁהִיא אֶחָד מִמֵּאָה. וְאוֹמֵר עַל אוֹתָהּ סְאָה שֶׁנָּטַל, אֶחָד מִמֵּאָה שֶׁיֶּשׁ כָּאן בֵּין הַכֹּל, דְּהַיְנוּ שְׁלִישׁ אוֹתָהּ סְאָה: <B>הֲרֵי הוּא בְּצַד זֶה חֻלִּין.</B> פֵּרוּשׁ יִשָּׁאֵר חֻלִּין כְּמוֹ שֶׁהוּא עַתָּה בְּטִבְלוֹ בְּצַד שְׁנֵי שְׁלִישִׁים שֶׁעִמּוֹ, וְהַשְּׁאָר דְּהַיְנוּ שְׁנֵי שְׁלִישִׁים יִהְיוּ תְּרוּמָה עַל הַכֹּל: <B>מֵהַחֻלִּין שֶׁיֶּשׁ כָּאן.</B> דְּהַיְנוּ הַשְּׁלִישׁ שֶׁאָמַרְתִּי עָלָיו שֶׁיִּשָּׁאֵר חֻלִּין. הֲרֵי זֶה בְּצַד זֶה מַעֲשֵׂר: <B>וּשְׁאָר מַעֲשֵׂר.</B> דְּהַיְנוּ עוֹד תִּשְׁעָה סָמוּךְ לִשְׁלִישׁ זֶה, אוֹתָם עֲשָׂרָה שְׁלִישִׁים יִהְיוּ מַעֲשֵׂר עַל הַמֵּאָה שְׁלִישִׁים. וְלָאו דַּוְקָא עֲשָׂרָה אֶלָּא מְעַט פָּחוּת כְּפִי הַמַּגִּיעַ לִשְׁנַיִם שֶׁל תְּרוּמָה: <B>זֶה שֶׁעָשִׂיתִי מַעֲשֵׂר.</B> הַיְנוּ הַשְּׁלִישׁ שֶׁקָּרָא עָלָיו שֵׁם מַעֲשֵׂר תְּחִלָּה עָשׂוּי תְּרוּמַת מַעֲשֵׂר עַל תִּשְׁעָה הַשְּׁלִישִׁים. וְלֹא שְׁלִישׁ שָׁלֵם אֶלָּא חָסֵר דָּבָר מֻעָט, וּלְהָכִי קָאָמַר מֵהַחֻלִּין שֶׁיֶּשׁ כָּאן דְּמַשְׁמַע וְלֹא כָל הַחֻלִּין. וּמַתְנִיתִין אַתְיָא כְּאַבָּא אֶלְעָזָר בֶּן גּוֹמֵל דְּאָמַר בְּסוֹף פֶּרֶק כָּל הַגֵּט (גִּטִּין לא.) כְּשֵׁם שֶׁיֵּשׁ לוֹ לְבַעַל הַבַּיִת רְשׁוּת לִתְרֹם תְּרוּמָה גְדוֹלָה כָּךְ יֵשׁ לוֹ רְשׁוּת לִתְרֹם תְּרוּמַת מַעֲשֵׂר. וְאִיהוּ נַמִּי שָׁרֵי לְהַפְרִישׁ תְּרוּמַת מַעֲשֵׂר בְּאֹמֶד {ה} בְּעַיִן יָפָה בְּפֶרֶק כָּל הַמְּנָחוֹת בָּאוֹת מַצָּה (מְנָחוֹת דַּף נד: <B>נה.</B>). הִלְכָּךְ אוֹתוֹ שְׁלִישׁ שֶׁהוּא מַפְרִישׁ מִשָּׁלֵם, אַף עַל פִּי שֶׁהָיָה רָאוּי לִהְיוֹת חָסֵר מְעַט, בְּעַיִן יָפָה הוּא מַפְרִישׁ וְשָׁרֵי:

<B>וּמִן הַצּוֹנֶנֶת עַל הַחַמָּה.</B> אַף עַל פִּי שֶּׁהוּא מְעַשֵּׂר מִן הָרַע עַל הַיָּפֶה, סָבַר לַהּ כְּרַבִּי אִלָּעִאי דְּאָמַר (קִדּוּשִׁין מו:) הַתּוֹרֵם מִן הָרַע עַל הַיָּפֶה תְּרוּמָתוֹ תְּרוּמָה, וּבִדְמַאי הִתִּירוּ אֲפִלּוּ לְכַתְּחִלָּה: <B>וַאֲפִלּוּ מִדְּפוּסִים הַרְבֵּה.</B> וְלֹא חַיְשִׁינַן שֶׁמָּא אֶתְמוֹל קָנָה הַנַּחְתּוֹם תְּבוּאָה מִמִּי שֶׁהוּא מְעַשֵּׂר וְהַיּוֹם מִמִּי שֶׁאֵינוֹ מְעַשֵּׂר וְנִמְצָא מַפְרִישׁ מִן הַפָּטוּר עַל הַחִיּוּב, דְּנַחְתּוֹם מֵחַד גַּבְרָא זְבִין, אַף עַל פִּי שֶׁהוּא עוֹשֶׂה מִדְּפוּסִים הַרְבֵּה: <B>רַבִּי יְהוּדָה אוֹסֵר.</B> כְּשֶׁשְּׁנֵיהֶם שֶׁל אֶמֶשׁ אוֹ שְׁנֵיהֶם שֶׁל הַיּוֹם מוֹדֶה רַבִּי יְהוּדָה לְרַבִּי מֵאִיר דַּאֲפִלּוּ מִדְּפוּסִים הַרְבֵּה שָׁרֵי, דְּלֹא חַיֵּשׁ לִדְפוּסִים {ז}. אֲבָל אֶחָד שֶׁל אֶמֶשׁ וְאֶחָד שֶׁל הַיּוֹם אֲפִלּוּ מִדְּפוּס אֶחָד אָסַר, שֶׁאֲנִי אוֹמֵר וְכוּ': <B>רַבִּי שִׁמְעוֹן אוֹסֵר בִּתְרוּמַת מַעֲשֵׂר.</B> דַּאֲפִלּוּ שְׁנֵיהֶם שֶׁל הַיּוֹם אוֹ שֶׁל אֶמֶשׁ, כֵּיוָן דְּבִשְׁנֵי דְפוּסִים הוּא אָמְרִינַן מִתְּרֵי גַבְרֵי זְבַן אֶת הַתְּבוּאָה, וְשֶׁמָּא הָאֶחָד עִשֵּׂר וְהָאֶחָד לֹא, וְנִמְצָא מַפְרִישׁ מִן הַפָּטוּר עַל הַחִיּוּב: <B>וּמַתִּיר בְּחַלָּה.</B> דִּבְחַלָּה כֻּלֵּי עַלְמָא מוֹדוּ אֲפִלּוּ שֶׁל אֶמֶשׁ בְּשֶׁל הַיּוֹם וַאֲפִלּוּ מִשְּׁנֵי דְפוּסִים, דְּאִי נַמִּי מִתְּרֵי גַבְרֵי זְבַן לֹא חַיְשִׁינַן, שֶׁהֲרֵי אֵצֶל הַנַּחְתּוֹם מִתְחַיְּבָה בְּחַלָּה, שֶׁהוּא גִלְגֵּל אֶת הָעִסָּה. וַהֲלָכָה כְּרַבִּי שִׁמְעוֹן:

<B>פַּלְטֵר.</B> הוּא הַלּוֹקֵחַ מִן הַנַּחְתּוֹמִים כִּכָּרוֹת הַרְבֵּה בְּבַת אַחַת וּמוֹכְרָן אַחֵר כָּךְ בַּשּׁוּק אֶחָד אֶחָד: <B>מְעַשֵּׂר מִכָּל דְּפוּס וּדְפוּס דִּבְרֵי רַבִּי מֵאִיר.</B> דִּסְבִירָא לֵיהּ לְרַבִּי מֵאִיר פַּלְטֵר מִתְּרֵי תְּלָתָא נַחְתּוֹמֵי זְבַן, וְאָנוּ אוֹמְרִים דְּפוּס זֶה קָנָה מִנַּחְתּוֹם אֶחָד וְהַשֵּׁנִי מִנַּחְתּוֹם אַחֵר. וְרַבִּי יְהוּדָה סָבַר פַּלְטֵר מֵחַד נַחְתּוֹם זְבַן, וְאַף עַל פִּי שֶׁהֵם מִשְּׁנֵי דְפוּסִים כֻּלָּן מִנַּחְתּוֹם אֶחָד הֵן, הִלְכָּךְ אִם כֻּלָּן מִשֶּׁל אֶמֶשׁ אוֹ מִשֶּׁל הַיּוֹם מְעַשֵּׂר מֵאֶחָד עַל הַכֹּל, כְּדִין נַחְתּוֹם: <B>מַנְפּוֹל.</B> קוֹנֶה מִן פַּלְטֵרִין הַרְבֵּה וּמוֹכֵר לַאֲחֵרִים, וְכֵיוָן שֶׁהוּא רָגִיל לִקְנוֹת מֵאֲנָשִׁים הַרְבֵּה הַלּוֹקֵחַ מִמֶּנּוּ מְעַשֵּׂר עַל כָּל אֶחָד וְאֶחָד. וַהֲלָכָה כְּרַבִּי יְהוּדָה:

<B>הַלּוֹקֵחַ מִן הֶעָנִי.</B> הַמְחַזֵּר עַל הַפְּתָחִים: <B>פִּלְחֵי דְבֵלָה.</B> חֲתִיכוֹת שֶׁל דְּבֵלָה: <B>בּוֹלֵל וְנוֹטֵל.</B> דּוֹרֵס הַתְּמָרִים אוֹ הַגְּרוֹגָרוֹת וּמְעָרְבָן יַחַד, וְהֵן נִבְלָלִים וּמִתְעָרְבִים, וְנִמְצָא מַפְרִישׁ מִן הַחִיּוּב, דְּקָסָבַר יֵשׁ בְּלִילָה בְּדָבָר הַיָּבֵשׁ. אֲבָל פְּרוּסוֹת שֶׁל לֶחֶם וּפִלְחֵי דְבֵלָה אֵין נִבְלָלִין, וְאָתֵי לְאַפְרוּשֵׁי מִן הַפָּטוּר עַל הַחִיּוּב: <B>בִּזְמַן שֶׁהַמַּתָּנָה מְרֻבָּה.</B> בִּזְמַן שֶׁהַכֹּל נוֹתְנִין מַתָּנָה מְרֻבָּה, שֶׁכָּל הַמַּתָּנוֹת שָׁווֹת, אָז מוֹעֶלֶת בְּלִילָה: <B>אֲבָל בִּזְמַן שֶׁהַמַּתָּנָה מֻעֶטֶת.</B> כְּלוֹמַר שֶׁמַּתָּנָה אַחַת מְעוּטָה מֵחֲבֶרְתָּהּ, לֹא מְהַנְיָא בְּלִילָה, דְּדִלְמָא כָּל חַד וְחַד לֹא מִתְרְמֵי דִידֵיהּ, וְחַיְשִׁינַן שֶׁמָּא מִן הָרֹב מַפְרִישׁ עַל הַמֻּעָט. כָּךְ נִרְאֶה פֵּרוּשׁ מִשְׁנָה זוֹ בַּיְרוּשַׁלְמִי {י}: 

~ פרק ה - משנה ו
<B>סִיטוֹן.</B> סוֹחֵר הַלּוֹקֵחַ תְּבוּאָה מִבַּעֲלֵי בָתִּים הַרְבֵּה בְּיַחַד וּמוֹכֵר לַאֲחֵרִים בְּמִדָּה גַסָּה: <B>לֹא יְעַשֵּׂר מִזֶּה עַל זֶה.</B> דְּשֶׁמָּא מַה שֶּׁלָּקַח מִמֶּנּוּ בַּשְּׁנִיָּה מֵאָדָם אַחֵר קְנָאוֹ, וְאָתֵי לְאַפְרוּשֵׁי מִן הַפָּטוּר עַל הַחִיּוּב: <B>וַאֲפִלּוּ מֵאוֹתוֹ הַסּוּג.</B> כְּלוֹמַר מֵאוֹתָהּ קֻפָּה. וְדוֹמֶה לוֹ בְּמַסֶּכֶת כֵּלִים (פֶּרֶק טז מִשְׁנָה ג) הַסּוּגִים הַגְּדוֹלִים: <B>וַאֲפִלּוּ מֵאוֹתוֹ הַמִּין.</B> דִּשְׁנֵיהֶם שְׁחַמְתִּית אוֹ שְׁנֵיהֶם לְבָנָה. וְאִם אָמַר הַסִּיטוֹן דְּמֵאָדָם אֶחָד קָנָה, נֶאֱמָן: 

~ פרק ה - משנה ז
<B>מְעַשֵּׂר מִזֶּה עַל זֶה.</B> דְּכֻלּוֹ מְעֻשָּׂר אוֹ כֻּלּוֹ אֵינוֹ מְעֻשָּׂר. וּבִלְבָד שֶׁיִּהְיֶה מִמִּין אֶחָד וּבְתוֹךְ שְׁנָתוֹ, שֶׁאֵין תּוֹרְמִין מִשֶּׁל שָׁנָה זוֹ עַל [שֶׁל] שָׁנָה אַחֶרֶת: <B>וּמִגַּנּוֹת אֲחֵרוֹת מְעַשֵּׂר מִכָּל אַחַת וְאַחַת.</B> דְּבַעֲלֵי הַגַּנּוֹת שֶׁמָּא זֶה עִשֵּׂר וְזֶה לֹא עִשֵּׂר: 

~ פרק ה - משנה ח
<B>מִשְּׁנֵי מְקוֹמוֹת.</B> שֶׁשְּׁנֵי הַמּוֹכְרִים הוֹדִיעוּהוּ שֶׁהוּא טֶבֶל: <B>אֶלָּא לְצֹרֶךְ.</B> כְּגוֹן שֶׁנִּתְעָרֵב לַחֲבֵרוֹ חֻלִּין מֻעָטִין בְּטֶבֶל מְרֻבֶּה, שֶׁאֵין לְטֶבֶל זֶה תַּקָּנָה עַד שֶׁיְּעַשֵּׂר עָלָיו מִמָּקוֹם אַחֵר {יג}. וְדַוְקָא לְחָבֵר {יד} מֻתָּר לִמְכֹּר טֶבֶל, אֲבָל לֹא לְעַם הָאָרֶץ. וְעַם הָאָרֶץ שֶׁנִּתְעָרֵב לוֹ חֻלִּין בְּטֶבֶל כֵּיצַד הוּא עוֹשֶׂה, הוֹלֵךְ אֵצֶל חָבֵר וְהוּא לוֹקֵחַ לוֹ טֶבֶל מִמָּקוֹם אַחֵר וּמְעַשֵּׂר לוֹ: 

~ פרק ה - משנה ט
<B>מִשֶּׁל יִשְׂרָאֵל עַל שֶׁל עוֹבֵד כּוֹכָבִים.</B> הַאי תַּנָּא סָבַר אֵין קִנְיָן לְעוֹבֵד כּוֹכָבִים בְּאֶרֶץ יִשְׂרָאֵל לְהַפְקִיעַ מִן הַמַּעֲשֵׂר, וְרַבִּי מֵאִיר הִיא, וְאֵינָהּ הֲלָכָה, דְּקַיְמָא לָן יֵשׁ קִנְיָן לְעוֹבֵד כּוֹכָבִים בְּאֶרֶץ יִשְׂרָאֵל לְהַפְקִיעַ מִן הַמַּעֲשֵׂר דִּכְתִיב (וַיִּקְרָא כה) וְהָאָרֶץ לֹא תִמָּכֵר לִצְמִיתוּת, הָא אִם נִמְכְּרָה צְמוּתָה הִיא {טו}: <B>עַל שֶׁל כּוּתִים.</B> פֵּרוֹת הַכּוּתִים וַדַּאי טֶבֶל מוֹכְרִין אוֹתָן. דְּאַף עַל פִּי דִּכְשֶׁאוֹכְלִין פֵּרוֹתֵיהֶן עִשּׂוּרֵי מְעַשְּׂרֵי, כְּשֶׁמּוֹכְרִין אֵינָן מְעַשְּׂרִין, דְּלֹא חַיְשֵׁי עַל לִפְנֵי עִוֵּר לֹא תִתֵּן מִכְשׁוֹל {טז}: <B>רַבִּי אֶלְעָזָר אוֹסֵר מִשֶּׁל כּוּתִים עַל שֶׁל כּוּתִים.</B> דְּזִמְנִין דִּמְעַשֵּׂר לְנַפְשֵׁיהּ וְנִמְלַךְ וּמוֹכְרוֹ, וְשֶׁמָּא הָאֶחָד מְעֻשָּׂר וְהָאֶחָד לֹא, וְנִמְצָא מַפְרִישׁ מִן הַפָּטוּר עַל הַחִיּוּב {יז}. וּבַזְּמַן הַזֶּה הַכּוּתִים עֲשָׂאוּם כְּעוֹבְדֵי כוֹכָבִים גְּמוּרִים לְכָל דִּבְרֵיהֶם: 

~ פרק ה - משנה י
<B>עָצִיץ נָקוּב.</B> כְּלִי שֶׁמְּשִׂימִין בְּתוֹכוֹ עָפָר וְזוֹרְעִים בּוֹ, אִם נָקוּב כְּדֵי שֹׁרֶשׁ קָטָן שֶׁהוּא פָּחוֹת מִכַּזַּיִת הֲרֵי הוּא כָּאָרֶץ וּפֵרוֹתָיו טֶבֶל גָּמוּר דְּאוֹרַיְתָא: <B>תָּרַם מִן הָאָרֶץ.</B> אֲפִלּוּ לְכַתְּחִלָּה נַמִּי מָצֵי לִתְרֹם, וּמִשּׁוּם סֵיפָא נָקַט לְשׁוֹן דִּיעֲבַד: <B>תְּרוּמָה וְיַחֲזֹר וְיִתְרֹם.</B> דְּמִן הַפָּטוּר עַל הַחִיּוּב הִפְרִישׁ וְלָאו תְּרוּמָה הִיא, הִלְכָּךְ יַחֲזֹר וְיִתְרֹם, וּמִכָּל מָקוֹם כֵּיוָן שֶׁקָּרָא לָהּ שֵׁם תְּרוּמָה נוֹתְנָהּ לַכֹּהֵן דְּלֹא לֵיתֵי לְזִלְזוּלֵי בִּתְרוּמָה. וְאֵין צָרִיךְ לְהוֹצִיא עָלֶיהָ תְּרוּמָה וּמַעֲשֵׂר כְּמוֹ בְּסֵיפָא, מִשּׁוּם דְּמִדְּאוֹרַיְתָא לָאו טֶבֶל הוּא, אֲבָל מִן הַנָּקוּב עַל שֶׁאֵינוֹ נָקוּב דְּמִן הַחִיּוּב עַל הַפָּטוּר הִפְרִישׁ, הַתְּרוּמָה הַוְיָא טֶבֶל דְּאוֹרַיְתָא {יח}, הִלְכָּךְ לֹא יֹאכַל עַד שֶׁיּוֹצִיא עָלֶיהָ תְּרוּמָה וּמַעַשְׂרוֹת:

<B>הַמְקַבֵּל שָׂדֶה.</B> לְמֶחֱצָה לִשְׁלִישׁ וְלִרְבִיעַ כְּדֶרֶךְ הָאֲרִיסִין: <B>חוֹלֵק לִפְנֵיהֶם.</B> וְאֵינוֹ צָרִיךְ לְעַשֵּׂר {א} חֵלֶק בַּעַל הַשָּׂדֶה, אֲבָל מַנִּיחַ לִפְנֵיהֶם חֵלֶק שְׂכַר הַקַּרְקַע כְּשֶׁאֵינוֹ מְעֻשָּׂר, וְלֹא חִיְּבוּ חֲכָמִים לַמְּקַבֵּל שֶׁיְּעַשֵּׂר עַל חֶלְקוֹ שֶׁל בַּעַל הַשָּׂדֶה וְיִתֵּן לוֹ מַה שֶּׁפָּסַק עִמּוֹ כְּשֶׁהוּא מְעֻשָּׂר, מִפְּנֵי יִשּׁוּב אֶרֶץ יִשְׂרָאֵל, כְּדֵי שֶׁלֹּא יִמָּנְעוּ מִלְּקַבֵּל הַשָּׂדוֹת בַּאֲרִיסוּת: <B>הַחוֹכֵר.</B> מְקַבֵּל הַשָּׂדֶה בְּדָבָר קָצוּב, כָּךְ וְכָךְ כּוֹרִין לְשָׁנָה, בֵּין שֶׁהִיא עוֹשָׂה הַרְבֵּה בֵּין שֶׁהִיא עוֹשָׂה מְעַט: <B>תּוֹרֵם וְנוֹתֵן לוֹ.</B> אֲבָל לֹא מְעַשֵּׂר, דְּכָל חוֹכֵר עַל מְנַת כֵּן יוֹרֵד לַשָּׂדֶה שֶׁיִּתֵּן לוֹ חֲכִירוּתוֹ מִפֵּרוֹת הַגְּדֵלִים בַּשָּׂדֶה בְּטִבְלָם, וּמִיהוּ מִשּׁוּם דְּאֵין לְהַשְׁהוֹת הַתְּרוּמָה, דְּאִי אֶפְשָׁר לַגֹּרֶן שֶׁתֵּעָקֵר אֶלָּא אִם כֵּן נִתְרְמָה, מִשּׁוּם הָכִי תּוֹרֵם וְנוֹתֵן לוֹ, וּמִסְתַּבֵּר דִּמְנַכֶּה לוֹ הַתְּרוּמָה מֵחֲכִירוּתוֹ, דְּמַאי שְׁנָא תְּרוּמָה מִמַּעֲשֵׂר: <B>אָמַר רַבִּי יְהוּדָה.</B> אֵימָתַי סַגִּי לַחוֹכֵר שֶׁיִּתְרֹם לְבָד וְאֵינוֹ מְעַשֵּׂר, כְּשֶׁנּוֹתֵן לוֹ חֲכִירוּתוֹ מֵאוֹתָהּ שָׂדֶה וּמֵאוֹתוֹ הַמִּין {ג}. אֲבָל אִם נָתַן לוֹ מִשָּׂדֶה אַחֶרֶת אַף מֵאוֹתוֹ הַמִּין, אוֹ מִמִּין אַחֵר אַף מֵאוֹתָהּ שָׂדֶה, כְּגוֹן שֶׁזָּרַע מִקְצַת הַשָּׂדֶה מִין אַחֵר כְּשִׁעוּר חֲכִירוּתוֹ, חַיָּב אַף לְעַשֵּׂר, דְּדָמִי לְפוֹרֵעַ חוֹבוֹ. אִי נַמִּי אֵין דַּעַת [הַמַּחְכִּיר] לְקַבֵּל פֵּרוֹת טְבוּלִים אֶלָּא כְּשֶׁפּוֹרֵעַ לוֹ מֵאוֹתָהּ שָׂדֶה וּמֵאוֹתוֹ הַמִּין:

<B>הַחוֹכֵר שָׂדֶה מִן הָעוֹבֵד כּוֹכָבִים.</B> קְנָס קָנְסוּ חֲכָמִים לְחוֹכֵר שָׂדֶה מִן הָעוֹבֵד כּוֹכָבִים שֶׁיְּהֵא מְעַשֵּׂר מַה שֶּׁנּוֹתֵן לוֹ, כְּדֵי שֶׁלֹּא יְהֵא שׁוּם יִשְׂרָאֵל חוֹכֵר שָׂדֶה מִמֶּנּוּ וְתִשָּׁאֵר בּוּרָה בְּיָדוֹ, וּמִתּוֹךְ כָּךְ יִצְטָרֵךְ לְמָכְרָהּ לְיִשְׂרָאֵל בְּדָמִים מֻעָטִים {ד}. וּבִמְקַבֵּל בַּאֲרִיסוּת לֹא קָנְסוּ, שֶׁיּוֹתֵר חָפֵץ הָעוֹבֵד כּוֹכָבִים בַּחֲכִירוּת מִבַּאֲרִיסוּת: <B>שָׂדֶה שֶׁל אֲבוֹתָיו.</B> שֶׁגְּזָלָהּ הָעוֹבֵד כּוֹכָבִים מֵאֲבוֹתָיו, וּקְנָסוּהוּ שֶׁיִּצְטָרֵךְ לְעַשֵּׂר כְּדֵי שֶׁיִּדְחַק עַצְמוֹ וְיִקְנֶנָּה מִן הָעוֹבֵד כּוֹכָבִים, שֶׁמִּתּוֹךְ שֶׁהִיא חֲבִיבָה עָלָיו שֶׁהָיְתָה שֶׁל אֲבוֹתָיו לֹא יַנִּיחַ לְקַבְּלָהּ מִן הָעוֹבֵד כּוֹכָבִים בְּיוֹתֵר מִן הָרָאוּי לָהּ, וּכְשֶׁיִּכְבַּד עָלָיו הַמַּעֲשֵׂר יִקְנֶנָּה. וּכְוָתֵיהּ דְּרַבִּי יְהוּדָה קַיְמָא לָן, דְּלֹא פָּלֵיג תַּנָּא קַמָּא עָלֵיהּ:

<B>כְּשֵׁם שֶׁחוֹלְקִין בַּחֻלִּין כָּךְ חוֹלְקִים בַּתְּרוּמָה.</B> כְּשֶׁנּוֹטֵל בַּעַל הַשָּׂדֶה מֶחֱצָה וּשְׁלִישׁ וּרְבִיעַ מִמַּה שֶּׁהוֹצִיאָה הַשָּׂדֶה נוֹטֵל גַּם כֵּן מֶחֱצָה וּשְׁלִישׁ וּרְבִיעַ מִן הַתְּרוּמָה וְהַמַּעַשְׂרוֹת הַמּוּטָלוֹת עָלֶיהָ וְנוֹתְנָן הוּא לְכָל כֹּהֵן וְלֵוִי שֶׁיִּרְצֶה: <B>אַף הַמַּעַשְׂרוֹת שֶׁלָּהֶן.</B> אִם הַמְקַבֵּל כֹּהֵן כָּל הַתְּרוּמָה שֶׁלּוֹ, וְאִם לֵוִי הוּא כָּל הַמַּעֲשֵׂר שֶׁלּוֹ: <B>שֶׁעַל מְנָת כֵּן.</B> יָרְדוּ לְקַבֵּל שָׂדֶה זוֹ בַּאֲרִיסוּת. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר, דְּאַף עַל גַּב דְּעַל מְנָת כֵּן יָרְדוּ בְּמַה קָּנוּ {ה}:

<B>הַמַּעַשְׂרוֹת לַבְּעָלִים.</B> בְּהָא אֲפִלּוּ רַבָּנָן דִּפְלִיגֵי עֲלֵיה דְּרַבִּי אֱלִיעֶזֶר מוֹדוּ, דְּכֵיוָן דְּשָׂדֶה שֶׁל בְּעָלִים מְקוֹם מַעֲשֵׂר שִׁיּוּרֵי שַׁיְרֵיהּ: <B>הַקַּרְתָּנִי.</B> לְשׁוֹן קִרְיָה, כְּלוֹמַר הָאִישׁ הַיּוֹשֵׁב בַּקִּרְיָה בַּכְּפָרִים. שֶׁכֵּן דֶּרֶךְ בְּנֵי הַכְּפָרִים לְקַבֵּל שָׂדֶה מִבְּנֵי הַכְּרָכִים: <B>יָכוֹל הוּא הַקַּרְתָּנִי וְכוּ'.</B> הִלְכָּךְ כָּל אֶחָד נוֹטֵל חֶלְקוֹ בְּמַעֲשֵׂר שֵׁנִי. וְכֵן הֲלָכָה:

<B>הַמְקַבֵּל זֵיתִים לְשֶׁמֶן.</B> יִשְׂרָאֵל שֶׁקִּבֵּל זֵיתִים מִכֹּהֵן אוֹ מִלֵּוִי {ז}: <B>כְּשֵׁם שֶׁחוֹלְקִין בַּחֻלִּין כָּךְ חוֹלְקִים בַּתְּרוּמָה.</B> וְאַף עַל גַּב דִּלְעֵיל לַבְּעָלִים, הָכָא שָׁאנֵי דְּלֹא קִבֵּל קַרְקַע אֶלָּא אִילָנוֹת, וְרַבָּנָן לֹא עָבְדֵי זֵיתִים {ח} כְּקַרְקַע: <B>אוֹ לְמַחֲצִית שָׂכָר.</B> שֶׁיִּמְכֹּר הַשֶּׁמֶן וִיחַלְּקוּ הָרֶוַח לָאֶמְצַע: <B>הַמַּעַשְׂרוֹת לַבְּעָלִים.</B> דְּרַבִּי יְהוּדָה עָבֵיד זֵיתִים כְּקַרְקַע. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

<B>לֹא יִמְכֹּר אָדָם זֵיתָיו.</B> זֵיתִים שֶׁנִּתְלְשׁוּ מִן הָאִילָן וְלֹא הֻכְשְׁרוּ לְקַבֵּל טֻמְאָה, כְּגוֹן שֶׁעֲדַיִן לֹא הִזִּיעוּ זֵעַת הַמַּעֲטָן שֶׁהִיא מַכְשִׁירָתָן לְקַבֵּל טֻמְאָה: <B>אֶלָּא לְחָבֵר.</B> לְפָרוּשׁ הַמֻּחְזָק לְטָהֳרוֹת, וְלֹא יִמְכְּרֵם לְמִי שֶׁאֵינוֹ חָבֵר, שֶׁלֹּא יִדְרֹךְ אוֹתָם בְּטֻמְאָה: <B>אַף לִמְעַשֵּׂר.</B> אַף לְמִי שֶׁלֹּא קִבֵּל עָלָיו אֶלָּא לְעַשֵּׂר אֲבָל לֹא קִבֵּל עָלָיו לִהְיוֹת מֻחְזָק לְטָהֳרוֹת יָכוֹל לְמָכְרָם, דְּכֵיוָן דְּאַכָּתֵי לֹא הֻכְשְׁרוּ אֵמוּר שֶׁמָּא יֹאכְלֵם קֹדֶם שֶׁיַּזִּיעוּ. וְאַף עַל גַּב דְּרֹב זֵיתִים אֵין עוֹמְדִין לְאָכְלָן כָּךְ, בַּעֲלִילָה כָּל דְּהוּ דְּמָצִינַן לִתְלוֹת תָּלִינַן. וְרַמְבַּ"ם מוֹקֵי פְּלוּגְתַּיְהוּ דְּבֵית שַׁמַּאי וּבֵית הִלֵּל, דְּבֵית שַׁמַּאי סָבְרֵי אָסוּר לִגְרֹם טֻמְאָה לְחֻלִּין שֶׁבְּאֶרֶץ יִשְׂרָאֵל, וּבֵית הִלֵּל סָבְרֵי מֻתָּר לִגְרֹם טֻמְאָה לְחֻלִּין שֶׁבְּאֶרֶץ יִשְׂרָאֵל: <B>וּצְנוּעֵי בֵית הִלֵּל.</B> וָתִיקִין הַמְדַקְדְּקִים בְּמִצְוֹת שֶׁהָיוּ בְּבֵית הִלֵּל:

<B>אֶחָד מְעַשֵּׂר.</B> נֶאֱמָן עַל הַמַּעַשְׂרוֹת: <B>וְאֶחָד אֵינוֹ מְעַשֵּׂר.</B> אֵינוֹ נֶאֱמָן עַל הַמַּעַשְׂרוֹת: <B>הַמְּעַשֵּׂר מְעַשֵּׂר אֶת שֶׁלּוֹ.</B> מְתַקֵּן הַחֵצִי מִכָּל מַה שֶּׁגָּדֵל בַּכֶּרֶם שֶׁל שְׁנֵיהֶם: <B>וְחֶלְקוֹ בְּכָל מָקוֹם שֶׁהוּא.</B> בַּיְרוּשַׁלְמִי מְפָרֵשׁ דְּהָכִי קָאָמַר, וְחֶלְקוֹ שֶׁל חֲבֵרוֹ דְּהַיְנוּ שֶׁל זֶה שֶׁאֵינוֹ מְעַשֵּׂר בְּכָל מָקוֹם שֶׁהוּא דְּמַאי הוּא. וּמִשּׁוּם דְּאֵין בְּרֵירָה {ט} חַיְשִׁינַן שֶׁמָּא חֲצִי חֶלְקוֹ שֶׁל הַמְּעַשֵּׂר בְּיַד חֲבֵרוֹ שֶׁאֵינוֹ מְעַשֵּׂר וַחֲצִי חֵלֶק חֲבֵרוֹ בְּיָדוֹ, הִלְכָּךְ אַף עַל פִּי שֶׁכְּבָר תִּקֵּן הַחֵצִי מִכָּל מַה שֶּׁגָּדֵל בַּכֶּרֶם חַיָּב לְתַקֵּן בְּתוֹרַת דְּמַאי חֲצִי חֵלֶק חֲבֵרוֹ שֶׁבְּיָדוֹ:

<B>טֹל אַתָּה חִטִּין שֶׁבְּמָקוֹם פְּלוֹנִי.</B> הָכָא מַיְרֵי שֶׁחִלְּקוּ שָׂדֶה בְּקָמוֹתֶיהָ וַעֲנָבִים הַמְחֻבָּרִים בַּכֶּרֶם הִלְכָּךְ יֵשׁ בְּרֵירָה וְאָמְרִינַן זֶהוּ חֶלְקוֹ הַמַּגִּיעוֹ וְהַמְּעַשֵּׂר מְעַשֵּׂר אֶת שֶׁלּוֹ וְדַיּוֹ. וּלְעֵיל אַיְרֵי שֶׁחִלְּקוּ בַּתָּלוּשׁ וְהָתָם אֵין בְּרֵירָה דְּכָל קֶלַח וָקֶלַח שֶׁל שֻׁתָּפִים הוּא וְצָרִיךְ לְעַשֵּׂר דְּמַאי עַל חֵלֶק חֲבֵרוֹ שֶׁבְּיָדוֹ {י}: <B>אֲבָל לֹא יֹאמַר לוֹ טֹל אַתָּה חִטִּין וַאֲנִי שְׂעוֹרִים.</B> דְּלֵיכָּא לְמֵימַר הָכָא זֶהוּ חֶלְקוֹ הַמַּגִּיעוֹ אֶלָּא הֲווֹ כְּמַחְלִיפִים זֶה לָזֶה, וְנִמְצָא זֶה הַמְּעַשֵּׂר מוֹכֵר חֶלְקוֹ לְמִי שֶׁאֵינוֹ מְעַשֵּׂר:

<B>חָבֵר.</B> הַנֶּאֱמָן לְטָהֳרוֹת וְאֵין צָרִיךְ לוֹמַר לְמַעַשְׂרוֹת: <B>שֶׁיָּרְשׁוּ אֶת אֲבִיהֶן עַם הָאָרֶץ.</B> וּפֵרוֹת שֶׁהִנִּיחַ הֲווֹ דְּמַאי וּבְחֶזְקַת טְמֵאִים: <B>טֹל אַתָּה בַּלַּח.</B> שֶׁהֻכְשַׁר לְקַבֵּל טֻמְאָה וַאֲנִי בַּיָּבֵשׁ שֶׁלֹּא הֻכְשַׁר לְקַבֵּל טֻמְאָה, לְפִי שֶׁאָסוּר לְחָבֵר לִמְכֹּר לְעַם הָאָרֶץ לַח וְיָבֵשׁ כְּדִתְנַן לְעֵיל בְּפֶרֶק שֵׁנִי:

<B>טֹל אַתָּה עֲבוֹדַת כּוֹכָבִים וַאֲנִי מָעוֹת.</B> אַף עַל גַּב דַּחֲלוּפֵי עֲבוֹדַת כּוֹכָבִים וְיֵין נֶסֶךְ אֲסוּרִים בַּהֲנָאָה, הָכָא שָׁרֵי עַד שֶׁלֹּא בָאוּ לְיָדוֹ, מִשּׁוּם דִּירוּשַׁת גֵּר אֶת אָבִיו אֵינָהּ מִן הַתּוֹרָה אֶלָּא מִדִּבְרֵי סוֹפְרִים {יג}. וְלֹא דָמִי לְחָבֵר וְעַם הָאָרֶץ שֶׁיָּרְשׁוּ אֶת אֲבִיהֶם עַם הָאָרֶץ דַּאֲפִלּוּ קֹדֶם שֶׁבָּאוּ לְיָדוֹ אָסוּר לוֹמַר לוֹ טֹל אַתָּה חִטִּים וַאֲנִי שְׂעוֹרִים, דְּהָתָם הַוְיָא יְרוּשָׁה דְּאוֹרַיְתָא וַהֲוֵי כְּאִלּוּ בָא לְיָדוֹ:

<B>סוּרְיָא.</B> הֵם אֲרָצוֹת שֶׁכָּבַשׁ דָּוִד שֶׁלֹּא הָיוּ מֵאֶרֶץ יִשְׂרָאֵל כְּגוֹן אֲרַם נַהֲרַיִם וַאֲרַם צוֹבָה, וּלְפִי שֶׁעֲדַיִן לֹא נִכְבְּשָׁה כָּל אֶרֶץ יִשְׂרָאֵל לְפִיכָךְ לֹא קָדְשׁוּ כִּקְדֻשַּׁת הָאָרֶץ אַף עַל פִּי שֶׁהָיָה כִּבּוּשׁ שֶׁל רַבִּים, וּבְמִקְצַת דִּינִים הִיא כְּאֶרֶץ יִשְׂרָאֵל וּבְמִקְצַת דִּינִים הִיא כְּחוּצָה לָאָרֶץ. וְהַקּוֹנֶה פֵּרוֹת בְּסוּרְיָא אֵינוֹ חַיָּב לְהַפְרִישׁ מֵהֶן דְּמַאי לְפִי שֶׁרֹב פֵּרוֹת הַנִּמְכָּרִים בְּסוּרְיָא מִחוּץ לָאָרֶץ הֵן בָּאִים, הִלְכָּךְ כִּי אָמַר מְעֻשָּׂרִין הֵן נֶאֱמָן מִגּוֹ דְּאִי בָעֵי אָמַר שֶׁל חוּץ לָאָרֶץ הֵן וְהָיָה נֶאֱמָן כְּדִתְנַן בְּפֶרֶק קַמָּא מִכְּזִיב וּלְהַלָּן נֶאֱמָן, הַשְׁתָּא נַמִּי כִּי אָמַר שֶׁל אֶרֶץ יִשְׂרָאֵל הֵם וְעִשַּׂרְתִּים נֶאֱמָן: <B>שֶׁהַפֶּה שֶׁאָסַר.</B> כְּשֶׁאָמַר שֶׁל אֶרֶץ יִשְׂרָאֵל הֵם וְחַיָּבִים בְּמַעֲשֵׂר, הוּא הַפֶּה שֶׁהִתִּיר כְּשֶׁאָמַר עִשַּׂרְתִּים, וְכִי הֵיכִי דִּמְהֵמְנִינַן לֵיהּ בְּמַאי דְאָסַר כְּשֶׁאָמַר שֶׁל אֶרֶץ יִשְׂרָאֵל הֵם נְהַמְנֵיהּ נַמִּי בְּמַה שֶּׁהִתִּיר כְּשֶׁאָמַר עִשַּׂרְתִּים: <B>מִשֶּׁלִּי הֵן.</B> מִן הַשָּׂדֶה שֶׁיֵּשׁ לִי כָּאן בְּסוּרְיָא, חַיָּבִים לְעַשֵּׂר, דִּבְפֵרוֹת שֶׁל סוּרְיָא גָּזְרוּ עַל הַדְּמַאי כְּשֶׁיָּדוּעַ שֶׁגָּדְלוּ שָׁם: <B>וְאִם יָדוּעַ שֶׁיֶּשׁ לוֹ שָׂדֶה.</B> אֵין כָּאן הַפֶּה שֶׁאָסַר הוּא הַפֶּה שֶׁהִתִּיר, דְּכֵיוָן שֶׁיָּדוּעַ שֶׁיֵּשׁ לוֹ שָׂדֶה מִסְּתָמָא מִשָּׂדֵהוּ הֵבִיא הַפֵּרוֹת, וְלֹא הָיָה נֶאֱמָן לוֹמַר מִחוּץ לָאָרֶץ הֵם בָּאִים, הִלְכָּךְ כִּי אָמַר עִשַּׂרְתִּים אֵינוֹ נֶאֱמָן דְּלֵיכָּא מִגּוֹ:

<B>עַם הָאָרֶץ שֶׁאָמַר לְחָבֵר קַח לִי.</B> כְּגוֹן שֶׁהָיָה הֶחָבֵר הוֹלֵךְ לַשּׁוּק לִקְנוֹת יָרָק לְעַצְמוֹ וְאָמַר לוֹ עַם הָאָרֶץ קַח לִי גַּם כֵּן אֲגֻדָּה שֶׁל יָרָק, וְלָקַח חָבֵר שְׁתֵּי אֲגֻדּוֹת סְתָם וְלֹא פֵּרֵשׁ זוֹ לִי וְזוֹ לְעַם הָאָרֶץ, פָּטוּר מִלְּעַשֵּׂר אוֹתָהּ שֶׁנּוֹתֵן לְעַם הָאָרֶץ, דְּיֵשׁ בְּרֵירָה כְּשֶׁנּוֹתְנָהּ לְעַם הָאָרֶץ שֶׁזּוֹ הִיא אוֹתָהּ שֶׁלָּקַח מִתְּחִלָּה לְצָרְכּוֹ וְאֵין חָבֵר מַחֲלִיפָהּ בְּשֶׁלּוֹ {טו}. פֵּרוּשׁ אַחֵר מִן הַיְרוּשַׁלְמִי, עַם הָאָרֶץ שֶׁאָמַר לְחָבֵר קַח לִי אֲגֻדַּת יָרָק, וְהַמּוֹכֵר חָבֵר הוּא וְיוֹדֵעַ שֶׁזֶּה הַלּוֹקֵחַ בִּשְׁבִיל עַם הָאָרֶץ הוּא לוֹקֵחַ, וְלֹא לָקַח הֶחָבֵר לְעַצְמוֹ כְּלוּם, פְּטוּרָה הָאֲגֻדָּה הַזֹּאת מִמַּעֲשֵׂר דְּמַאי שֶׁכֵּיוָן שֶׁהַמּוֹכֵר חָבֵר וְיוֹדֵעַ שֶׁלְּצֹרֶךְ עַם הָאָרֶץ הוּא קוֹנֶה אֵינוֹ מוֹכֵר לוֹ אֶלָּא אִם כֵּן עִשֵּׂר, דְּהָכִי אָמְרִינַן לְעֵיל פֶּרֶק ב' שֶׁאָסוּר לִמְכֹּר לְעַם הָאָרֶץ אֶלָּא אִם כֵּן עִשֵּׂר תְּחִלָּה. אֲבָל אִם קָנָה הֶחָבֵר אַחַת לְעַצְמוֹ וְאַחַת לְעַם הָאָרֶץ וְנִתְעָרְבוּ, חַיָּב לְעַשֵּׂר דְּשֶׁמָּא מַה שֶּׁמָּכַר לוֹ לְעַצְמוֹ אֵינוֹ מְעֻשָּׂר, שֶׁהַמּוֹכֵר לְחָבֵר אֵינוֹ צָרִיךְ לְעַשֵּׂר וְהַקּוֹנֶה עָלָיו מֻטָּל לְעַשֵּׂר כִּדְאָמְרָן לְעֵיל פֶּרֶק ב': <B>וַאֲפִלּוּ הֵן מֵאָה.</B> וַאֲפִלּוּ נִתְעָרְבָה אַחַת שֶׁלּוֹ בְּמֵאָה שֶׁל עַם הָאָרֶץ: <B>גְּלֻסְקָן.</B> כִּכַּר לֶחֶם יָפֶה:

~ פרק ז - משנה א
<B>הַמַּזְמִין אֶת חֲבֵרוֹ.</B> אוֹמֵר מֵעֶרֶב שַׁבָּת וְכוּ'. וְהֵיכָא דְהִתְנָה מֵעֶרֶב שַׁבָּת מֻתָּר לְהַפְרִישׁ בְּשַׁבָּת, אֲבָל לֹא הִתְנָה אָסוּר דִּתְנַן סָפֵק חֲשֵׁכָה מְעַשְּׂרִים אֶת הַדְּמַאי, הָא וַדַּאי חֲשֵׁכָה לֹא. וְהָכָא מַיְרֵי בְּהִזְמִינוֹ חֲבֵרוֹ וְלֹא הִדִּירוֹ, דְּאִי הִדִּירוֹ {א} הָא אָמְרִינַן לְעֵיל פֶּרֶק ד' אוֹכֵל עִמּוֹ בְּשַׁבָּת רִאשׁוֹנָה אַף עַל פִּי שֶׁאֵינוֹ מַאֲמִינוֹ עַל הַמַּעַשְׂרוֹת. אִי נַמִּי הָתָם בְּשַׁבָּת רִאשׁוֹנָה שֶׁל בָּחוּר שֶׁנָּשָׂא בְתוּלָה וְהָכָא בִּשְׁאָר שַׁבָּתוֹת: <B>הֲרֵי הוּא מַעֲשֵׂר.</B> וּתְרוּמָה גְדוֹלָה אֵינוֹ צָרִיךְ לְהַפְרִישׁ דְלֹא נֶחְשְׁדוּ עַמֵּי הָאָרֶץ עָלֶיהָ כִּדְפָרְשִׁינַן בְּפֶרֶק קַמָּא. וְעַל אֶחָד מִמֵּאָה שֶׁמַּפְרִישׁ לְמָחָר אוֹמֵר הֲרֵי הוּא מַעֲשֵׂר רִאשׁוֹן, וּשְׁאָר מַעֲשֵׂר דְּהַיְנוּ תִּשְׁעָה שֶׁהוּא חַיָּב עֲדַיִן כְּדֵי שֶׁיְּהוּ עֲשָׂרָה לְמֵאָה מוֹדֵד {ג} לוֹ הָאֶחָד, וְאַחַר כָּךְ אוֹמֵר זֶה הָאֶחָד שֶׁעָשִׂיתִי תְּחִלָּה מַעֲשֵׂר עָשׂוּי תְּרוּמַת מַעֲשֵׂר עַל תִּשְׁעָה הַסְּמוּכִים לוֹ: <B>מַעֲשֵׂר שֵׁנִי.</B> יִהְיֶה בִּצְפוֹנוֹ אוֹ בִדְרוֹמוֹ וּמְחֻלָּל יִהְיֶה עַל הַמָּעוֹת, וְנִמְצָא לְמָחָר שֶׁאֵין צָרִיךְ לְהַפְרִישׁ כִּי אִם תְּרוּמַת מַעֲשֵׂר בִּלְבָד, וְאוֹכֵל וְשׁוֹתֶה אֶת הַשְּׁאָר: 

~ פרק ז - משנה ב
<B>מָזְגוּ לוֹ אֶת הַכּוֹס.</B> בְּיוֹם הַשַּׁבָּת: <B>אוֹמֵר וְכוּ'.</B> וְאֵין דַּי לוֹ בְּמַה שֶּׁהִתְנָה מֵעֶרֶב שַׁבָּת, אֲבָל צָרִיךְ לַחֲזֹר וְלוֹמַר בְּשַׁבָּת כְּשֶׁבָּא לֶאֱכֹל וְלִשְׁתּוֹת מַה שֶּׁאֲנִי עָתִיד לְשַׁיֵּר מִן הַכּוֹס {ד} וְכוּ'. בַּיְרוּשַׁלְמִי פָּרֵיךְ אִם בְּאוֹמֵר מֵעַכְשָׁיו תְּהֵא תְּרוּמָה הֲרֵי הַכֹּל מִתְעָרֵב יַחַד וְנַעֲשֶׂה מְדֻמָּע, וְאִם בְּאוֹמֵר לִכְשֶׁאֶשְׁתֶּה הָא לֹא הֲוֵי תְּרוּמָה אֶלָּא לְאַחַר שֶׁשָּׁתָה וְנִמְצָא שׁוֹתֶה טֶבֶל, וּמְשָׁנֵי בְּאוֹמֵר מֵעַכְשָׁיו וְלִכְשֶׁאֶשְׁתֶּה תְּהֵא תְּרוּמָה, וְנִמְצָא שֶׁלֹּא שָׁתָה טֶבֶל וְגַם לֹא הָיָה מְדֻמָּע: <B>בְּפִיו.</B> בְּפִי הַכּוֹס: 

~ פרק ז - משנה ג
<B>שֶׁאֵינוֹ מַאֲמִין לְבַעַל הַבַּיִת.</B> עַל הַמַּעַשְׂרוֹת: <B>וְחוֹשֵׂךְ גְּרוֹגֶרֶת אֶחָת.</B> נִמְנָע מִלֶּאֱכֹל גְּרוֹגֶרֶת אַחַת כְּנֶגֶד אוֹתָהּ שֶׁל תְּרוּמַת מַעֲשֵׂר שֶׁהִפְרִישׁ, שֶׁלֹּא יְהֵא גּוֹזֵל אֶת בַּעַל הַבַּיִת: <B>לֹא יַחְשֹׂךְ.</B> וְיִקְנֶה גְּרוֹגֶרֶת אַחַת וְיֹאכַל, שֶׁאִם לֹא יֹאכַל מַרְעִיב עַצְמוֹ וּמְמַעֵט מִסְּעֻדָּתוֹ וְנִמְצָא מְמַעֵט בְּכָךְ מְלַאכְתּוֹ שֶׁל בַּעַל הַבַּיִת, שֶׁלֹּא יוּכַל לַעֲשׂוֹת מְלָאכָה: <B>לֹא יַחְשֹׂךְ מִפְּנֵי שֶׁהוּא תְנַאי בֵּית דִּין.</B> שֶׁתְּהֵא תְּרוּמַת מַעֲשֵׂר מִשֶּׁל בַּעַל הַבַּיִת וּמַעֲשֵׂר שֵׁנִי מִשֶּׁל פּוֹעֵל, הִלְכָּךְ חַיָּב בַּעַל הַבַּיִת לִתֵּן לוֹ {ז} הַגְּרוֹגֶרֶת שֶׁהִפְרִישׁ לִתְרוּמַת מַעֲשֵׂר: 

~ פרק ז - משנה ד
<B>הַלּוֹקֵחַ יַיִן מִבֵּין הַכּוּתִים.</B> קֹדֶם שֶׁגָּזְרוּ עַל יֵינָן, וְהָיָה יֵינָן וַדַּאי טֶבֶל. וְהָכָא אַיְרֵי בְּלוֹקֵחַ עֶרֶב שַׁבָּת בֵּין הַשְּׁמָשׁוֹת אוֹ שֶׁלָּקַח מִבְּעוֹד יוֹם וְשָׁכַח לְהַפְרִישׁ עַד בֵּין הַשְּׁמָשׁוֹת, וְאָז אָסוּר לְהַפְרִישׁ דִּתְנַן סָפֵק חֲשֵׁכָה אֵין מְעַשְּׂרִין אֶת הַוַּדַּאי, הִלְכָּךְ אָסוּר לִקְבֹּעַ לוֹ מָקוֹם, דִּקְבִיעוּת מָקוֹם חָשׁוּב כְּהַפְרָשָׁה וְאָסוּר בֵּין הַשְּׁמָשׁוֹת, וּמִשּׁוּם הָכִי לֹא קָתָנֵי בְּסֵיפָא וּמַעֲשֵׂר שֵׁנִי מְחֻלָּל עַל הַמָּעוֹת, דְּכֵיוָן דְּאֵינוֹ רַשַּׁאי לִקְבֹּעַ לוֹ מָקוֹם בִּצְפוֹנוֹ אוֹ בִדְרוֹמוֹ אִי אֶפְשָׁר לְחַלְּלוֹ, אֲבָל לִקְרוֹת שֵׁם דְּאָמַר שֶׁאֲנִי עָתִיד לְהַפְרִישׁ שָׁרֵי, דְּכֻלֵּי הַאי לֹא גָזוּר בֵּין הַשְּׁמָשׁוֹת: <B>שֶׁאֲנִי עָתִיד לְהַפְרִישׁ.</B> הָכָא לֹא קָתָנֵי שֶׁאֲנִי עָתִיד לְהַפְרִישׁ לְמָחָר כִּדְקָתָנֵי לְעֵיל גַּבֵּי הַמַּזְמִין אֶת חֲבֵרוֹ, דְּדַוְקָא כְּשֶׁאֵינוֹ מְתַקֵּן אֶלָּא מַה שֶּׁאוֹכֵל וְשׁוֹתֶה הִתִּירוּ לְהַפְרִישׁ לְמָחָר מִמּוֹתַר אֲכִילָתוֹ וּשְׁתִיָּתוֹ הַנִּשְׁאַר בְּשׁוּלֵי הַכּוֹס, אֲבָל לְתַקֵּן אֶת כָּל הַנֹּאד בְּשַׁבָּת לֹא הִתִּירוּ. וְהָכָא אֵינוֹ קוֹרֵא שֵׁם לִתְרוּמַת מַעֲשֵׂר כְּמוֹ בַּבָּבָא שֶׁל דְּמַאי וּכְדִלְעֵיל גַּבֵּי הַמַּזְמִין אֶת חֲבֵרוֹ, מִשּׁוּם דְּגַבֵּי דְּמַאי דִּמְעַשְּׂרָן וְהֵן שֶׁלּוֹ קוֹרֵא שֵׁם לִתְרוּמַת מַעֲשֵׂר כְּדֵי לְתַקֵּן אֶת הַמַּעֲשֵׂר וּלְאָכְלוֹ, אֲבָל כָּאן שֶׁהוּא טֶבֶל וַדַּאי צָרִיךְ לִתֵּן הַמַּעֲשֵׂר לַלֵּוִי וְהַלֵּוִי יִקְרָא שֵׁם לִתְרוּמַת מַעֲשֵׂר: <B>וַעֲשָׂרָה מַעֲשֵׂר רִאשׁוֹן וְתִשְׁעָה מַעֲשֵׂר שֵׁנִי.</B> לָאו דַּוְקָא, שֶׁהֲרֵי לְאַחַר שֶׁהִפְרִישׁ מִמֵּאָה שְׁנֵי לֻגִּין לִתְרוּמָה גְדוֹלָה וְנִשְׁאַר תִּשְׁעִים וּשְׁמֹנָה לֹא יִהְיֶה הַמַּעֲשֵׂר אֶלָּא עֲשָׂרָה לֻגִּין פָּחוֹת חֹמֶשׁ וּמַעֲשֵׂר שֵׁנִי שְׁמֹנָה לֻגִּין וּשְׁמֹנֶה עֲשִׂירִיּוֹת וְחֹמֶשׁ עֲשִׂירִית: <B>מֵחֵל וְשׁוֹתֶה.</B> מַתְחִיל וְשׁוֹתֶה. פֵּרוּשׁ אַחֵר מוֹזֵג וְשׁוֹתֶה, מֵחֵל כְּמוֹ מֵהֵל, לְשׁוֹן סָבְאֵךְ מָהוּל בַּמָּיִם (יְשַׁעְיָה א). וּסְתָם מַתְנִיתִין אַתְיָא כְּרַבִּי מֵאִיר דְּאָמַר אַף בִּדְאוֹרַיְתָא יֵשׁ בְּרֵירָה וְזֶה שֶׁנִּשְׁאַר בְּשׁוּלֵי הַכּוֹס הֲוֵי כְּאִלּוּ הֻפְרַשׁ מִתְּחִלָּה. וְאֵינָהּ הֲלָכָה אֶלָּא קַיְמָא לָן דְּבִדְרַבָּנָן יֵשׁ בְּרֵירָה אֲבָל בִּדְאוֹרַיְתָא אֵין בְּרֵירָה, הִלְכָּךְ בְּוַדַּאי טֶבֶל לֹא יֹאכַל וְלֹא יִשְׁתֶּה עַד שֶׁיַּפְרִישׁ: 

~ פרק ז - משנה ה
<B>וְהוּא בְּבֵית הַמִּדְרָשׁ.</B> בְּעֶרֶב שַׁבָּת וּמִתְיָרֵא שֶׁמָּא יִקְדָּשׁ עָלָיו הַיּוֹם וְלֹא יוּכַל לְעַשֵּׂר:

<B>כַּלְכָּלוֹת.</B> סַלִּים: <B>מַעַשְׂרוֹת זוֹ בְזוֹ.</B> מַעַשְׂרוֹת שֶׁאֲנִי צָרִיךְ לְהַפְרִישׁ מִכַּלְכָּלָה זוֹ יִהְיוּ מֻנָּחִים בַּחֲבֶרְתָּהּ: <B>הָרִאשׁוֹנָה מְעֻשָּׂרֶת.</B> וּמַפְרִישׁ מִן הַשְּׁנִיָּה הַמַּעַשְׂרוֹת שֶׁל שְׁתֵּיהֶן: <B>שֶׁל זוֹ בָזוֹ וְשֶׁל זוֹ בָזוֹ הָרִאשׁוֹנָה מְעֻשֶּׂרֶת.</B> אֲבָל הַשְּׁנִיָּה אֵינָהּ מְעֻשֶּׂרֶת, דְּמִיָּד כְּשֶׁאָמַר שֶׁל זוֹ בְזוֹ נִתְקְנָה הָרִאשׁוֹנָה וְנִפְטְרָה מִן הַמַּעַשְׂרוֹת {יב} וְכִי הָדַר וְאָמַר שֶׁל זוֹ בְזוֹ נִמְצָא מַפְרִישׁ מִזּוֹ שֶׁכְּבָר נִפְטֶרֶת עַל זוֹ שֶׁהִיא חַיֶּבֶת, וְאֵין מַפְרִישִׁין מִן הַפָּטוּר עַל הַחִיּוּב וְלֹא אָמַר כְּלוּם {יג}, וְנוֹטֵל מִן הַשְּׁנִיָּה הַמַּעַשְׂרוֹת שֶׁל שְׁתֵּיהֶן: <B>מַעַשְׂרוֹתֵיהֶן מַעַשְׂרוֹת כַּלְכָּלָה בַחֲבֶרְתָּהּ.</B> הַמַּעַשְׂרוֹת הַמְחֻיָּבוֹת לִשְׁתֵּי כַלְכָּלוֹת הַלָּלוּ יִהְיוּ שֶׁל כַּלְכָּלָה בַחֲבֶרְתָּהּ, הֲרֵי קָרָא שֵׁם מַעַשְׂרוֹת שְׁתֵּיהֶן כְּאַחַת, וּמַפְרִישׁ מִזּוֹ עַל זוֹ וּמִזּוֹ עַל זוֹ, דִּבְכָל אַחַת שִׁיֵּר טֶבֶל כְּדֵי מַעַשְׂרוֹת חֲבֶרְתָּהּ. אֲבָל אֵינוֹ מַפְרִישׁ עֲלֵיהֶן מִמָּקוֹם אַחֵר, שֶׁהֲרֵי קָרָא שֵׁם וְקָבַע מַעַשְׂרוֹת כָּל אַחַת מֵהֶן בַּחֲבֶרְתָּהּ. פֵּרוּשׁ אַחֵר וְעִקָּר, מַעַשְׂרוֹתֵיהֶן מַעַשְׂרוֹת כַּלְכָּלָה בַחֲבֶרְתָּהּ קָרָא שֵׁם וּמַפְרִישׁ מֵאֵיזוֹ מֵהֶן שֶׁיִּרְצֶה, דְּהָכִי מְפָרְשִׁינַן לְמִלְּתֵיהּ, מַעַשְׂרוֹתֵיהֶן מֵאֵיזֶה מֵהֶן שֶׁאֶרְצֶה יִהְיוּ מַעַשְׂרוֹת כַּלְכָּלָה בַחֲבֶרְתָּהּ: 

~ פרק ז - משנה ז
<B>מֵאָה טֶבֶל וּמֵאָה חֻלִּין.</B> שֶׁנִּתְעָרְבוּ: <B>נוֹטֵל מֵאָה וְאֶחָד.</B> נוֹטֵל מֵאָה לְהַפְרִישׁ מֵהֶן מַעֲשֵׂר רִאשׁוֹן וּמַעֲשֵׂר שֵׁנִי כְּדִין כָּל טֶבֶל, וְאֶחָד נוֹטֵל מִן הַמֵּאָה (דִּלְשֵׁם) [שֶׁל] חֻלִּין לִתְרוּמַת מַעֲשֵׂר כְּאִלּוּ הָיוּ טֶבֶל, שֶׁהַחֻלִּין נַעֲשִׂים כַּטֶּבֶל עַצְמוֹ {יד} לְעִנְיַן תְּרוּמַת מַעֲשֵׂר בִּלְבָד אֲבָל לֹא לְעִנְיַן מַעֲשֵׂר רִאשׁוֹן וּמַעֲשֵׂר שֵׁנִי, וְנִמְצָא זֶה מַפְסִיד עַל יְדֵי תַּעֲרֹבֶת זוֹ אַחַת בִּלְבָד שֶׁמַּפְרִישׁ מִן הַחֻלִּין. וְטֶבֶל דְּאַיְרֵי בֵיהּ הָכָא הוּא שֶׁהוּרְמָה מִמֶּנּוּ תְּרוּמָה גְדוֹלָה אֲבָל לֹא הוּרְמוּ מִמֶּנּוּ מַעַשְׂרוֹת וְלֹא תְּרוּמַת מַעֲשֵׂר: <B>מֵאָה טֶבֶל וּמֵאָה מַעֲשֵׂר.</B> שֶׁלֹּא הוּרְמָה מִן הַמַּעֲשֵׂר הַזֶּה תְּרוּמַת מַעֲשֵׂר, וְנִתְעָרְבוּ, נוֹטֵל מֵאָה שֶׁל טֶבֶל לְהַפְרִישׁ מֵהֶן הַמַּעַשְׂרוֹת כְּדִפְרִישִׁית לְעֵיל, וְנוֹטֵל אַחַת מִן הַמֵּאָה שֶׁל מַעֲשֵׂר שֶׁהֵם נַעֲשִׂים כַּטֶּבֶל עַצְמוֹ לְהִתְחַיֵּב בִּתְרוּמַת מַעֲשֵׂר, וְנִשְׁאֲרוּ תִּשְׁעִים וָתֵשַׁע, מַפְרִישׁ מֵהֶן תְּרוּמַת מַעֲשֵׂר לְפִי חֶשְׁבּוֹן שֶׁהֵם עֶשֶׂר פָּחוֹת עֲשִׂירִית שֶׁל אֶחָד. וְאִם לֹא נִתְעָרֵב הַמַּעֲשֵׂר בַּטֶּבֶל לֹא הָיָה מַפְרִישׁ אֶלָּא עֲשָׂרָה לִתְרוּמַת מַעֲשֵׂר שֶׁבָּהֶן, הַשְׁתָּא מִפְּנֵי שֶׁנִּתְעָרְבוּ בַּטֶּבֶל מַפְרִישׁ אַחַד עָשָׂר פָּחוֹת עֲשִׂירִית נִמְצָא מַפְסִיד אֶחָד פָּחוֹת עֲשִׂירִית: <B>מֵאָה חֻלִּין מְתֻקָּנִין וּמֵאָה מַעֲשֵׂר.</B> רִאשׁוֹן שֶׁלֹּא נִטְּלָה תְּרוּמָתוֹ, שֶׁנִּתְעָרְבוּ, נוֹטֵל מֵאָה לְשֵׁם מַעֲשֵׂר רִאשׁוֹן כְּדֵי לְהַפְרִישׁ מֵהֶן עֲשָׂרָה לִתְרוּמַת מַעֲשֵׂר שֶׁבָּהֶן, וְעֶשֶׂר נוֹטֵל מִן הַמֵּאָה חֻלִּין מְתֻקָּנִין כְּאִלּוּ הָיוּ הֵם מַעֲשֵׂר רִאשׁוֹן דַּעֲשָׂרָה מֵהֶם לִתְרוּמַת מַעֲשֵׂר, וְנִמְצָא מַפְסִיד עֲשָׂרָה. וְהָא דְלֹא מַחְמְרִינַן הָכִי כְּשֶׁנִּתְעָרֵב עִם טֶבֶל לְהַזְקִיק הַטֶּבֶל לְהַפְרִישׁ מִמֶּנּוּ עֲשָׂרָה לִתְרוּמַת מַעֲשֵׂר כְּמֵאָה שֶׁל מַעֲשֵׂר, מִשּׁוּם שֶׁכָּךְ הוּא הַדִּין שֶׁכָּל דָּבָר הַמִּתְעָרֵב חוֹזֵר לִהְיוֹת כִּתְחִלָּתוֹ, הִלְכָּךְ הַמַּעֲשֵׂר חוֹזֵר לִהְיוֹת כַּטֶּבֶל וְאֵין הַטֶּבֶל חוֹזֵר לִהְיוֹת כַּמַּעֲשֵׂר: <B>מֵאָה טֶבֶל וְתִשְׁעִים מַעֲשֵׂר וְכוּ' לֹא הִפְסִיד כְּלוּם.</B> דְּכֵיוָן דְּאִיכָּא בַּטֶּבֶל עֲשָׂרָה יְתֵרִים דַּיְנִינַן לֵיהּ כְּאִלּוּ יֵשׁ לוֹ פַּרְנָסָה מִמָּקוֹם אַחֵר {טז}. וְכָל טֶבֶל שֶׁנִּתְעָרֵב בְּחֻלִּין, אִם יֵשׁ לוֹ טְבָלִים אֲחֵרִים שֶׁיָּכוֹל לְהַפְרִישׁ עַל זֶה הַתַּעֲרֹבֶת מִמָּקוֹם אַחֵר, אֵינוֹ מַפְרִישׁ אֶלָּא לְפִי חֶשְׁבּוֹן שֶׁל הַטֶּבֶל בִּלְבָד, וְהָכָא שֶׁהַטֶּבֶל יָתֵר עֶשֶׂר עַל [הַמַּעֲשֵׂר] חָשְׁבִינַן כְּאִלּוּ הֵם מִמָּקוֹם אַחֵר וּמַפְרִישׁ מֵהֶן לְפִי חֶשְׁבּוֹן שֶׁל טֶבֶל בִּלְבָד וְאֵינוֹ מַפְסִיד כְּלוּם: <B>כָּל זְמַן שֶׁהַטֶּבֶל מְרֻבֶּה.</B> וְהוּא שֶׁיִּהְיֶה מְרֻבֶּה עֶשֶׂר {יז}, וְהָכִי אִיתָא בַּיְרוּשַׁלְמִי: 

~ פרק ז - משנה ח
<B>עֶשֶׂר שׁוּרוֹת שֶׁל עֶשֶׂר עֶשֶׂר כַּדֵּי יַיִן.</B> מְסֻדָּרִים בָּאָרֶץ, שֶׁבְּכָל עִנְיָן שֶׁתִּמְנֶה בֵּין מִמִּזְרָח לְמַעֲרָב בֵּין מִצָּפוֹן לְדָרוֹם תִּמְצָא עֶשֶׂר שׁוּרוֹת שֶׁל עֶשֶׂר עֶשֶׂר שֶׁהֵן מֵאָה כַּדּוֹת: <B>שׁוּרָה הַחִיצוֹנָה אַחַת מַעֲשֵׂר.</B> חָבִית אַחַת שֶׁבַּשּׁוּרָה הַחִיצוֹנָה קָבַעְתִּי מַעֲשֵׂר עַל כַּדּוֹת אֲחֵרוֹת שֶׁיֵּשׁ לִי: <B>וְאֵין יָדוּעַ אֵיזוֹ.</B> חִיצוֹנָה, שֶׁהֲרֵי יֵשׁ כָּאן אַרְבַּע שׁוּרוֹת חִיצוֹנוֹת לְאַרְבַּע רוּחוֹת: <B>נוֹטֵל שְׁתֵּי חָבִיּוֹת בַּאֲלַכְסוֹן.</B> אַחַת לְמִקְצוֹעַ דְּרוֹמִית מִזְרָחִית וּשְׁנִיָּה לָהּ בְּמִקְצוֹעַ מַעֲרָבִית צְפוֹנִית, שֶׁכָּל חָבִית שֶׁבַּמִּקְצוֹעַ נִמְנֵית לִשְׁתֵּי רוּחוֹת, וִימַלֵּא מִשְּׁתֵּיהֶן חָבִית אַחַת וְתִהְיֶה מַעֲשֵׂר {יט}: <B>חֲצִי שׁוּרָה הַחִיצוֹנָה אַחַת מַעֲשֵׂר.</B> וְאִם אָמַר חָבִית אַחַת שֶׁבַּחֲצִי שׁוּרָה הַחִיצוֹנָה בָּהּ קָבַעְתִּי מַעֲשֵׂר עַל כַּדּוֹת אֲחֵרוֹת שֶׁיֵּשׁ לִי: <B>וְאֵין יָדוּעַ אֵיזוֹ.</B> חֲצִי שׁוּרָה מֵאַרְבַּע שׁוּרוֹת הַחִיצוֹנוֹת: <B>נוֹטֵל אַרְבַּע חָבִיּוֹת מֵאַרְבַּע זָוִיּוֹת.</B> מִפְּנֵי שֶׁיֵּשׁ בְּכָל זָוִית מִשְּׁתֵּי חֲצָאִים שֶׁבִּשְׁתֵּי שׁוּרוֹת הַחִיצוֹנוֹת הַסְּמוּכוֹת לָהּ, וּמְמַלֵּא מֵאַרְבַּעְתָּן חָבִית אַחַת וְתִהְיֶה מַעֲשֵׂר: <B>שׁוּרָה אַחַת מַעֲשֵׂר וְכוּ'.</B> וְאִם אָמַר חָבִית אַחַת שֶׁבְּשׁוּרָה אַחַת מֵאֵלּוּ עֶשֶׂר שׁוּרוֹת בָּהּ קָבַעְתִּי מַעֲשֵׂר וְאֵין יָדוּעַ בְּאֵיזוֹ שׁוּרָה הִיא: <B>נוֹטֵל שׁוּרָה אַחַת בַּאֲלַכְסוֹן.</B> כְּלוֹמַר נוֹטֵל עֶשֶׂר כַּדּוֹת מִשּׁוּרָה אַחַת מִמִּקְצוֹעַ דְּרוֹמִית מִזְרָחִית עַד מִקְצוֹעַ מַעֲרָבִית צְפוֹנִית, אוֹ מִמִּקְצוֹעַ דְּרוֹמִית מַעֲרָבִית עַד מִקְצוֹעַ מִזְרָחִית צְפוֹנִית וּמְמַלֵּא מִכָּל הָעֶשֶׂר חָבִיּוֹת חָבִית אַחַת וְתִהְיֶה מַעֲשֵׂר. וּמִפְּנֵי מַה נוֹטֵל בַּאֲלַכְסוֹן וְאֵינוֹ נוֹטֵל שׁוּרָה אַחַת כֻּלָּהּ בְּיֹשֶׁר מִמִּזְרָח לְמַעֲרָב אוֹ מִצָּפוֹן לְדָרוֹם, דִּכְשֶׁהוּא נוֹטֵל בַּאֲלַכְסוֹן בֵּין שֶׁתִּמְנֶה הַשּׁוּרוֹת כֻּלָּן מִמִּזְרָח לְמַעֲרָב בֵּין שֶׁתִּמְנֶה אוֹתָן מִצָּפוֹן לְדָרוֹם נִמְצָא נוֹטֵל חָבִית אַחַת מִכָּל שׁוּרָה וְשׁוּרָה, מַה שֶּׁאֵין כֵּן אִם הָיָה נוֹטֵל שׁוּרָה אַחַת בָּאֹרֶךְ מִמִּזְרָח לְמַעֲרָב אוֹ מִצָּפוֹן לְדָרוֹם:
