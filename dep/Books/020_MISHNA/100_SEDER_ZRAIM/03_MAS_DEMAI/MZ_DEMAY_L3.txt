{א} מַסְקָנָא דִּלְכֻלֵּי עַלְמָא רֹב עַמֵּי הָאָרֶץ מְעַשְּׂרִין הֵן הִלְכָּךְ לֹא גָזְרוּ בְּלָאו. וַאֲפִלּוּ הָכִי חָשׁוּ לְמִעוּט הָכָא דִּשְׁכִיחֵי טוּבָא וְהָיוּ הַרְבֵּה נִכְשָׁלִים. תּוֹסָפוֹת. וּדְמַאי, דָּא מַאי: {ב} וְלֹא פֵרֵשׁ מַה יַּעֲשֶׂה בָהּ, אִם יִתְּנֶנָּה לַכֹּהֵן אוֹ יִמְכְּרֶנָּה לוֹ. וְדַעַת הָרַמְבַּ"ם דְּנוֹתְנָן לַכֹּהֵן, אֲבָל הָרַ"שׁ כָּתַב אוֹ נוֹתֵן לוֹ אוֹ מוֹכֵר לוֹ הַיֶּתֶר לְפִי חֶשְׁבּוֹן: {ג} <B> וְהַכֻּסְבָּר.</B> מִדְּאִפְסִיק שֶׁבִּיהוּדָה בֵּין חֹמֶץ לְכֻסְבָּר שְׁמַע מִנָּהּ דְּכֻסְבָּר לָאו בִּיהוּדָה הוּא אֶלָּא בְּכָל מָקוֹם. וְכֵן הוּא בָּרַ"מ וּבָרַ"שׁ בְּשֵׁם הַיְרוּשַׁלְמִי דְּגַם כֻּסְבָּר בִּיהוּדָה:

{ו} וּבַיְרוּשַׁלְמִי פְּלִיגֵי רַבִּי אוֹשַׁעְיָא וְרַבִּי יוֹחָנָן. דְּרַבִּי יוֹחָנָן אוֹמֵר בְּשָׁעָה שֶׁגָּזְרוּ עַל הַדְּמַאי לֹא גָזְרוּ עַל דְּבָרִים הַלָּלוּ, וּלְרַבִּי אוֹשַׁעְיָא אֵימַת קָדָשִׁים עָלָיו וְאֵינוֹ נוֹתֵן לַכֹּהֵן דָּבָר שֶׁאֵינוֹ מְתֻקָּן: {ז} <B> וְהַמְדֻמָּע.</B> הַתְּרוּמָה שְׁמָהּ דִּמְעָה מִלְּשׁוֹן הַפָּסוּק מְלֵאָתְךָ וְדִמְעֲךָ: {ח} הַיְנוּ אַלִּיבָּא דְרַבִּי אוֹשַׁעְיָא. אֲבָל לְרַבִּי יוֹחָנָן לֹא גָזְרוּ וְאֵין צָרִיךְ לְטַעַם לֹא חַיְשִׁינַן וְכוּ', וְטַעֲמָא דְרַבִּי יוֹחָנָן דְּכֹל לַכֹּהֵן לֹא גָזְרוּ, וּכְעֵין דְּאִיתָא סוֹף פֶּרֶק ה'. תּוֹסְפוֹת יוֹם טוֹב: {ט} <B> שֶׁמֶן עָרֵב.</B> וּלְשׁוֹן הָרַ"שׁ דְּעִקָּרוֹ לִמְשִׁיחָה וְלֹא לַאֲכִילָה דַּחֲמִירָא, כְּלוֹמַר אַף עַל גַּב דְּסִיכָה נַמִּי כִּשְׁתִיָּה כְּדִלְקַמָּן מִכָּל מָקוֹם לֹא חָמוּר כְּמוֹ אֲכִילָה וּשְׁתִיָּה עַצְמָהּ:

{א} לְשׁוֹן הָרַ"מ, בֵּין בְּאֶרֶץ יִשְׂרָאֵל בֵּין בְּחוּצָה לָאָרֶץ, וְאִם כֵּן פֵּרוֹת אֶרֶץ יִשְׂרָאֵל שֶׁיָּצְאוּ לְחוּץ לָאָרֶץ חַיָּבִין בִּתְרוּמָה וּבְמַעֲשֵׂר: {ב} וְאֵין רְצוֹנוֹ בְּאֵלּוּ הַמִּינִים הַמִּין כֻּלּוֹ, אֶלָּא חֵלֶק יָדוּעַ מֵאוֹתוֹ הַמִּין שֶׁלֹּא יִמָּצֵא כָמוֹהוּ אֶלָּא בְּאֶרֶץ יִשְׂרָאֵל בִּלְבָד, וְעַל כֵּן יִתְחַיֵּב בּוֹ דְּמַאי בְּכָל מָקוֹם שֶׁהוּא. הָרַ"מ:

{א} <B> הֲרֵי זֶה.</B> פֵּרוּשׁ זֶה הָאֶחָד מִמֵּאָה יִהְיֶה בְּצַד זֶה מַעֲשֵׂר וְהַיְנוּ קְבִיעַת מָקוֹם: {ב} <B> תְּרוּמַת מַעֲשֵׂר.</B> לְפִי שֶׁהָהַפְרָשָׁה הִיא לִתְרוּמַת מַעֲשֵׂר בִּלְבָד, אֲבָל הַמַּעֲשֵׂר אֵין צָרִיךְ לְהַפְרִישׁ דְּכֵיוָן שֶׁקָּרָא שֵׁם וְנָטַל תְּרוּמָתוֹ הֲרֵי הַלֵּוִי הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה, הִלְכָּךְ קֹבֵעַ מָקוֹם וּמְסַיֵּם לְאֶחָד מִמֵּאָה שֶׁזֶּה יַפְרִישׁ אַחַר כָּךְ, וְאֵין צָרִיךְ יוֹתֵר. תּוֹסְפוֹת יוֹם טוֹב: {ג} <B> בִּצְפוֹנוֹ.</B> שֶׁכֵּן דֶּרֶךְ בְּנֵי אָדָם בְּהוֹשָׁטַת יַד יָמִין נוֹטֵל מַה שֶּׁכְּנֶגֶד יְמִינוֹ, וְהִיא שְׂמֹאלוֹ שֶׁל אוֹתוֹ דָבָר, לְכָךְ מַקְדִּים צְפוֹנוֹ. תּוֹסְפוֹת יוֹם טוֹב:

~ פרק ה - משנה ב

{ד} מִשְּׁלֹשִׁים וְשָׁלֹשׁ וּשְׁלִישׁ שֶׁחוֹלֵק הַמֵּאָה לִשְׁלֹשָׁה חֲלָקִים וּמִכָּל חֵלֶק נוֹטֵל עֲשִׂירִית הָעֲשִׂירִית וְשָׁלֹשׁ הָעֲשִׂירִיּוֹת הֵן אֶחָד מִשְּׁלֹשִׁים וְשָׁלֹשׁ וּשְׁלִישׁ. וְעַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב: {ה} כְּלוֹמַר דְּהַשְׁתָּא אֵין אָנוּ צְרִיכִין לְפָרֵשׁ חָסֵר דָּבָר מֻעָט כֵּיוָן דִּבְלָאו הָכִי מַתְנִיתִין כְּאַבָּא אֶלְעָזָר: {ו} וְהַשְּׁאָר חַלָּה. בַּיְרוּשַׁלְמִי הָרוֹצֶה לְהַפְרִישׁ תְּרוּמָה תְּרוּמַת מַעֲשֵׂר וְחַלָּה כְּאַחַת כַּמָּה הֵן אֶחָד מֵעֶשְׂרִים וּפֵרֵשׁ הָרַ"שׁ אֶחָד מֵעֶשְׂרִים הַיְנוּ חָמֵשׁ סְאָה מִמֵּאָה. שְׁתַּיִם מֵהֶן תְּרוּמָה גְדוֹלָה וְאַחַת תְּרוּמַת מַעֲשֵׂר וּשְׁתַּיִם לְחַלָּה שֶׁהִיא אֶחָד מֵאַרְבָּעִים וּשְׁמֹנָה, אֶלָּא דְכָךְ אֵין הַחֶשְׁבּוֹן מְכֻוָּן:

{ז} כִּי הַנַּחְתּוֹם בְּעַצְמוֹ אֶפְשָׁר שֶׁיַּעֲשֶׂה מֵעִסָּה אַחַת מִינִים הַרְבֵּה מִן הַלֶּחֶם בִּדְפוּסִים מִשְׁתַּנִּים זֶה מִזֶּה. הָרַ"מ: 

~ פרק ה - משנה ד

{ח} <B> וּטְפוּס.</B> כְּמוֹ דְּפוּס, לְפִי שֶׁהָיוּ עוֹשִׂין הַלֶּחֶם בִּדְפוּס. הָרַ"מ: 

~ פרק ה - משנה ה

{ט} <B> פִּלְחֵי דְבֵלָה.</B> כְּשֶׁנִּתְיַבְּשׁוּ וּדְרוּסוֹת בְּעִגּוּל מִתְעַגֵּל לִפְלָחִים שֶׁיֵּשׁ בְּפֶלַח כַּמָּה דְבֵלוֹת. הָרַ"שׁ. וּפֶלַח מִלְּשׁוֹן פּוֹלֵחַ וּבוֹקֵעַ: {י} הָרַמְבַּ"ם מְפָרֵשׁ בְּדֶרֶךְ אַחֵר.</B> לְפִי שֶׁנֹּאמַר שֶׁזֶּה שֶׁנָּתַן מַתָּנָה גְדוֹלָה אִישׁ נָדִיב הוּא וְעַל כָּל פָּנִים הוֹצִיא הַמַּעַשְׂרוֹת, וְזֶה שֶׁנָּתַן מַתָּנָה מֻעֶטֶת אִישׁ כִּילַי הוּא וְלֹא עִשֵּׂר, לְכָךְ לֹא יִתְעָרֵב הַכֹּל, שֶׁמָּא יַפְרִישׁ מִדְּמַאי עַל הַוַּדַּאי:

{יא} <B> וּמִגַּנּוֹת אֲחֵרוֹת.</B> אִם לָקַח מִמֶּנּוּ וְחָזַר וְלָקַח מִמֶּנּוּ פַּעַם שְׁנִיָּה. הָרַ"מ: 

~ פרק ה - משנה ח

{יב} <B> אֵין אָדָם רַשַּׁאי.</B> שֶׁאֵין רַשַּׁאי לְהוֹצִיא מִיָּדוֹ בִּלְתִּי מְתֻקָּן, דְּשֶׁמָּא לֹא יוֹדִיעַ לַלּוֹקֵחַ: {יג} דְּאִי אִפְּכָא טֶבֶל מֻעָט בְּחֻלִּין מְרֻבִּין נִתְבַּטֵּל בָּרֹב מִדְּאוֹרַיְתָא וְלֹא נִתְחַיֵּב אֶלָּא מִדְּרַבָּנָן. וְאִם יְתַקֵּן בְּטֶבֶל וַדַּאי מִמָּקוֹם אַחֵר יִהְיֶה מִן הַחִיּוּב עַל הַפָּטוּר: {יד} הוּא הַדִּין לְנֶאֱמָן עַל הַמַּעַשְׂרוֹת: 

~ פרק ה - משנה ט

{טו} וּלְרַבִּי מֵאִיר קָא מַשְׁמַע לָן קְרָא שֶׁלֹּא תוּכַל לְמָכְרָם לַחֲלוּטִין לְהַפְקִיעַ מִן הַמַּעַשְׂרוֹת: {טז} דְּדָרְשֵׁי לֵיהּ כְּפַשְׁטֵיהּ. וּמִשּׁוּם גֶּזֶל הַכֹּהֵן נַמִּי לֹא חַיְשֵׁי דַּהֲוֵי מָמוֹן שֶׁאֵין לוֹ תּוֹבְעִים וְעוֹד דְּלֹא הֻבְרַר חֶלְקָן. הָרַ"שׁ: {יז} וְכָל שֶׁכֵּן דְּמֵהַאי טַעֲמָא פָּלֵיג עַל מִשֶּׁל יִשְׂרָאֵל עַל שֶׁל כּוּתִים.</B> תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ה - משנה י

{יח} וְאָסוּר אַף לַכֹּהֲנִים כְּדִתְנַן בְּמִשְׁנָה ב' פֶּרֶק ח' דִּתְרוּמוֹת, וְנִדְרָשׁ מִפָּסוּק כֵּן תָּרִימוּ גַּם אַתֶּם לְרַבּוֹת כֹּהֲנִים וּלְוִיִּם: 

~ פרק ה - משנה יא

{יט} <B>מִן הַדְּמַאי.</B> תְּרוּמַת מַעֲשֵׂר בִּדְמַאי, וּבְוַדַּאי אַף תְּרוּמָה גְדוֹלָה בִּכְלַל, לְהָכִי תְּנַן סְתָמָא: {כ} <B> וְיַחֲזֹר כוּ'.</B> שֶׁהֲרֵי אֶפְשָׁר שֶׁאֶחָד מֵהֶן טֶבֶל. אֲבָל אִם תָּרַם מֵעָצִיץ שֶׁאֵינוֹ נָקוּב עַל שֶׁאֵינוֹ נָקוּב תְּרוּמָתוֹ תְּרוּמָה, לְפִי שֶׁשְּׁנֵיהֶם בְּעֵרֶךְ אֶחָד וְחִיּוּבָן מִדְּרַבָּנָן. הָרַ"מ: {כא} <B> עַד שֶׁיּוֹצִיא.</B> הַבַּעַל הַבַּיִת וְאַחַר כָּךְ יִתְּנֶנָּה לַכֹּהֵן. הָרַ"מ:

~ פרק ו - משנה א

{א} אֵינוֹ רוֹצֶה לוֹמַר אֲבָל לִתְרֹם צָרִיךְ, דְּאִם כֵּן אֵין בֵּין מְקַבֵּל לְחוֹכֵר, אֶלָּא הֻשְׁאַל שֵׁם מַעֲשֵׂר לִכְלָל הַמַּתָּנוֹת: {ב} <B>לִפְנֵיהֶם.</B> וְהַחֲלוּקָה יִהְיֶה בִּפְנֵי בַּעַל הַשָּׂדֶה שֶׁיּוֹדֵעַ שֶׁמַּה שֶּׁנָּתַן לוֹ הוּא טֶבֶל. הָרַ"מ: {ג} דְּאַדַּעְתָּא דְהָכִי יוֹרְדִים הַחוֹכְרִים וּמֵעִקָּרָא בְּשֶׁל רְשׁוּת הַמַּחְכִּיר גְּדֵלִים אֵין זֶה פּוֹרֵעַ, וּמֵהַאי טַעֲמָא נַמִּי לֹא הֲוֵי כְּמוֹכֵר טֶבֶל לְעַם הָאָרֶץ: 

~ פרק ו - משנה ב
{ד} כָּתַב הָרַ"מ וְזֶה הָעִקָּר בְּיָדֵינוּ מַה שֶׁאָמַר הַכָּתוּב (דְּבָרִים ז) לֹא תִכְרֹת לָהֶם בְּרִית וְלֹא תְחָנֵּם, וּבָאָה הַקַּבָּלָה לֹא תִתֵּן לָהֶם חֲנִיָּה בַקַּרְקַע, וּלְפִיכָך נִדְחֹק אֹתָם שְׁלֹא יִשְׂכְּרוּ אֹתָהּ מִמֶנּוּ עַד שֶׁיִּצְטָרֵךְ לְמֹכְרָהּ:

~ פרק ו - משנה ג

{ה} וּלְרַבִּי אֶלִיעֶזֶר סְתָמוֹ בְּהִתְנָה שֶׁהַמַּעַשְׂרוֹת שֶׁלָּהֶם דַּהֲוָה לֵיהּ כְּאוֹמֵר לוֹ תְּלֹשׁ מִקַּרְקַע זֶה שֶׁתִּקָּנֶה לְךָ, וּבְהָכִי קָנֵי:

{ז} וְכָל שֶׁכֵּן אִיפְּכָא וּסְתָמָא תְּנַן: {ח} מַשְׁמַע דְּהוּא הַדִּין אִם קִבֵּל לְמָסְקָן, וּמֵהָרַמְבַּ"ם מַשְׁמַע דְּדַוְקָא לְשֶׁמֶן תְּנַן:

{ט} וְאַף דִּלְקַמָּן בְּפֶרֶק ז' מִשְׁנָה ד' קַיְמָא לָן דְּבִדְרַבָּנָן יֵשׁ בְּרֵירָה, הַיְנוּ לְרַבִּי אוֹשַׁעְיָה דִּסְבִירָא לֵיהּ הָכִי. אֲבָל הָרַ"מ מְפָרֵשׁ דְּמַיְרֵי שֶׁחִלְּקוּ הָעֲנָבִים וְהִפְרִישׁ מֵחֶלְקוֹ מַעַשְׂרוֹתָיו וְאַחַר כָּךְ נִתְעָרְבוּ הָעֲנָבִים וְדָרְכוּ בְּיַחַד, אִם כֵּן אֵין זֶה עִנְיָן לְדִין בְּרֵירָה:

{י} וּלְפֵרוּשׁ הָרַ"מ אֵין צְרִיכִין לְחַלֵּק בָּזֶה, אֶלָּא דִּלְעֵיל מַיְרֵי שֶׁנִּתְעָרֵב אַחֲרֵי שֶׁכְּבָר חִלְּקוּ, וְהָכָא מַיְרֵי שֶׁעֲדַיִן לֹא חִלְּקוּ וְלֹא הִפְרִישׁ הַמְּעַשֵּׂר כְּלוּם:

{יא} <b>טֹל אַתָּה.</b> וּפֵרוּשׁ הָרַ"ב הָכָא נַמִּי בְּקָמוֹתֶיהָ וּבַעֲנָבִים הַמְחֻבָּרִים דַּוְקָא: {יב} <B> טֹל כוּ' הַיָּבֵשׁ.</B> וְאִיפְּכָא פְּשִׁיטָא שֶׁלֹּא יֹאמַר, שֶׁכָּל שֶׁהֻכְשַׁר לְטֻמְאָה הוּא בְּחֶזְקַת טָמֵא בְּיַד עַם הָאָרֶץ. וְעַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב:

~ פרק ז - משנה א

{א} וְאֵין לְהַקְשׁוֹת דִּבְלֹא הִדִּירוֹ אֵיךְ אוֹכֵל אֶצְלוֹ וְהָתְנַן בְּפֶרֶק ג' שֶׁאֵינוֹ מִתְאָרֵחַ אֵצֶל עַם הָאָרֶץ, דְּיֵשׁ לוֹמַר דְּהָכָא בָּעֵי תַּקָּנָה לְמִי שֶׁרוֹצֶה לֶאֱכֹל מְתֻקָּן וְאַף עַל פִּי שֶׁלֹּא קִבֵּל עָלָיו לִהְיוֹת נֶאֱמָן. אִי נַמִּי מַיְרֵי בְּיִחֵד לוֹ עֵדִים. תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> אוֹמֵר מֵעֶרֶב שַׁבָּת.</B>  כִּי בִדְמַאי אָדָם מַתְנֶה עַל הַדָּבָר שֶׁאֵינוֹ בִרְשׁוּתוֹ הָרַ"מ: {ג} מְשַׁעֵר בְּאֹמֶד כַּמָּה שִׁעוּר אֲכִילָתוֹ וּמְחַשֵּׁב בְּלִבּוֹ לְהַתְנוֹת עַכְשָׁיו וְשֶׁיַּפְרִישׁ כָּזֶה לְמָחָר: 

~ פרק ז - משנה ב

{ד} וְהוּא הַדִּין לְמַה שֶּׁיֹּאכַל שֶׁצָּרִיךְ שֶׁיַּחֲזֹר וְיִרְחֹשׁ בִּשְׂפָתָיו בְּשַׁבָּת בְּשָׁעָה שֶׁמַּפְרִישׁ זוֹ אַחַר זוֹ. יְרוּשַׁלְמִי: {ה} <B> וּמַעֲשֵׂר שֵׁנִי בְּפִיו.</B> בְּכוֹס לֹא שַׁיָּךְ כֻּלֵּי הַאי בִּצְפוֹנוֹ כוּ' וְהַוְיָא יוֹתֵר קְבִיעַת מָקוֹם בְּפִיו: 

~ פרק ז - משנה ג

{ו} <B> וְחוֹשֵׂךְ.</B> מִלְּשׁוֹן וְלֹא חָשַׂכְתָּ אֶת בִּנְךָ: {ז} וּלְכָךְ רַשַּׁאי לֶאֱכֹל, וְאֵין רְצוֹנוֹ לוֹמַר שֶׁנְּחַיְּבֵהוּ לְבַעַל הַבַּיִת שֶׁיִּתֵּן לוֹ כוּ' דְּמִכֵּיוָן שֶׁהוּא עַם הָאָרֶץ לֹא יִשְׁמַע לָנוּ לְעַשֵּׂר פֵּרוֹתָיו דְּמַאי:

{ח} <B>מִבֵּין הַכּוּתִים.</B> דְּאִלּוּ יִשְׂרָאֵל אֵינוֹ רַשַּׁאי לִמְכֹּר טֶבֶל אֶלָּא לְצֹרֶךְ, וְכֵיוָן דְּלֹא שְׁכִיחַ לֹא בָעֵי לְמַתְנֵיהּ הָכָא: 

~ פרק ז - משנה ה

{ט} <B> לְהַפְרִישׁ.</B> לֹא גָרְסִינַן לְמָחָר לְפִי מַה שֶּׁכָּתַב הָרַ"ב לְעֵיל לְחַלֵּק בֵּין שֶׁאֵינוֹ מְתַקֵּן אֶלָּא מַה שֶּׁאוֹכֵל לַמְתַקֵּן אֶת כָּל הַנֹּאד. דְּאִי גָרַס אֵין צָרִיךְ לְחַלֵּק בְּהָכִי אֶלָּא הֲוָה לֵיהּ לְחַלֵּק בֵּין וַדַּאי לִדְמַאי. וְעַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב: 

~ פרק ז - משנה ו

{י} <B> מַעַשְׂרוֹת.</B> הֻשְׁאַל שֵׁם מַעַשְׂרוֹת לִכְלַל הַמַּתָּנוֹת וּבִכְלָלָן גַּם הַתְּרוּמָה: {יא} <B> מְעֻשֶּׂרֶת.</B> בְּהַךְ קָמַיְתָא לֵיכָּא חִדּוּשׁ שֶׁהָרִאשׁוֹנָה מְעֻשֶּׂרֶת, וּמִשּׁוּם סֵיפָא נָקְטָהּ. הָרַ"שׁ: {יב} דְּרַבִּי מֵאִיר הִיא דְּאָמַר אֵינוֹ תּוֹפֵס אֶלָּא לָשׁוֹן רִאשׁוֹן כִּדְאִיתָא בְּמַסֶּכֶת תְּמוּרָה: {יג} וְאֵין לְהַקְשׁוֹת דְּהָא בְּמַפְרִישׁ מִפָּטוּר עַל הַחִיּוּב תְּרוּמָתוֹ תְּרוּמָה אֶלָּא דְּיַחֲזֹר וְיִתְרֹם, וְאִם כֵּן אֵיךְ קָאָמַר וְלֹא אָמַר כְּלוּם. דְּשָׁאנֵי הָתָם שֶׁהִפְרִישׁ וְהַתְּרוּמָה נִכֶּרֶת לְכָךְ תְּרוּמָתוֹ תְּרוּמָה כְּדֵי שֶׁלֹּא לְזַלְזוֹלֵי בִּתְרוּמָה, אֲבָל הָכָא שֶׁלֹּא נִתְרְמָה וְלֹא נִכֶּרֶת אֵין כָּאן בֵּית מֵחוּשׁ כְּלָל: 

~ פרק ז - משנה ז

{יד} וְאַמַּתְנִיתִין דִּשְׁתֵּי כַלְכָּלוֹת נַמִּי קָאֵי מִדְּלֹא תָנֵי בְּהָא אֶלָּא מַעַשְׂרוֹת: {טו} <B>חֻלִּין מְתֻקָּנִים.</B> הַטֶּבֶל כְּשֶׁיּוֹצִיא מִמֶּנּוּ תְּרוּמָה גְדוֹלָה וּתְרוּמַת מַעֲשֵׂר וְלֹא שְׁאָר מַעַשְׂרוֹת יִקָּרֵא חֻלִּין מִפְּנֵי שֶׁהַקְּדֻשָּׁה נִטְּלָה מִמֶּנּוּ, וְחֻלִּין מְתֻקָּנִים הַיְנוּ שֶׁהוֹצִיא מִמֶּנּוּ כָּל הַתְּרוּמוֹת וּמַעַשְׂרוֹת. הָרַ"מ: {טז} וְכֻלָּהּ מַתְנִיתִין כְּשֶׁאֵין לוֹ פַּרְנָסָה מִמָּקוֹם אַחֵר.</B> וְעַיֵּן בְּפֶרֶק ג' דְּחַלָּה מִשְׁנָה ט': {יז} וּלְהָכִי בָּעִינַן עֲשָׂרָה.</B> דְּהָא מִשּׁוּם דְּדַיְנִינַן לֵיהּ כְּאִלּוּ יֵשׁ לוֹ פַּרְנָסָה מִמָּקוֹם אַחֵר הוּא, וְכָל שֶׁיֵּשׁ לוֹ עֲשָׂרָה חֲלָקִים יְתֵרִים. שֶׁיֵּשׁ לָנוּ לְהַפְרִישׁ מֵהֶן אֶחָת לִתְרוּמַת מַעֲשֵׂר הָרְאוּיָה לֵחַלֵּק עֲשִׂירִית מֵעֲשָׂרָה חֶלְקֵי הַטֶּבֶל אוֹ הַמַּעֲשֵׁר, הִלְכָּךְ דַּיְנִינַן לֵיהּ כְּפַרְנָסָה מִמָּקוֹם אַחֵר, וּבְפָחוֹת מִזֶּה לֹא נוּכַל לְדַיְנוֹ כָּךְ. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ז - משנה ח

{יח} <B> שְׁתֵּי חָבִיּוֹת.</B> פָּתַח בְּכַד וְסִיֵּם בְּחָבִית, לוֹמַר לְךָ הַיְנוּ כַּד הַיְנוּ חָבִית, כִּדְאִיתָא בְּבָבָא קַמָּא (כז.): {יט} וְכָל הַחֲרָדָה הַזֹּאת בִּשְׁבִיל תְּרוּמַת מַעֲשֵׂר שֶׁיֵּשׁ בּוֹ.</B> תּוֹסְפוֹת יוֹם טוֹב: {כ} <B> חֲצִי שׁוּרָה אַחַת מַעֲשֵׂר.</B> לְשׁוֹן הָרַ"מ וְאִם אָמַר חָבִית אַחַת מֵחֲצִי שׁוּרָה אַחַת מַעֲשֵׂר וְלֹא יִחֵד אֵיזוֹ חֲצִי שׁוּרָה יִקַּח עֶשְׂרִים חָבִיּוֹת עֶשֶׂר עַל אֲלַכְסוֹן אֶחָד וְעֶשֶׂר עַל אֲלַכְסוֹן אַחֵר הָעוֹבֵר עָלָיו וְנִמְצָא לוֹקֵחַ חָבִית מִכָּל חֲצִי שׁוּרָה מִן הַשּׁוּרוֹת וְאַחַר כָּךְ יוֹצִיא מִכֻּלָּם חָבִית אַחַת:
