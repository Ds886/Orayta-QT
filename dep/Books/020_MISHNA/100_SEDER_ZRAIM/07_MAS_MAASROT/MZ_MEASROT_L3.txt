{א} דְּאִלּוּ רָאוּ פְּנֵי הַבַּיִת עַל יְדֵי בַעַל הַבַּיִת וְהֻקְבְּעוּ לְמַעֲשֵׂר טָבֵיל לְהוּ וַאֲסוּרִים לְכָל אָדָם: {ב} <B>לְבָתֵּיכֶם.</B> דַּוְקָא לְבָתֵּיכֶם, דְּאִלּוּ לְבֵיתִי אֵין בַּיִת אַחֵר טוֹבֵל כְּדִלְקַמָּן. תּוֹסְפוֹת יוֹם טוֹב:

{ג} וְצָרִיךְ לוֹמַר שֶׁהַפֵּרוֹת מֻנָּחִים בַּשּׁוּק, דְּאִלּוּ הַפֵּרוֹת הֵמָּה גַם כֵּן בַּשַּׁעַר אוֹ בַחֲנוּת, אִם כֵּן טְבוּלִים וְהֻקְבְּעוּ לַמַּעֲשֵׂר וַאֲסוּרִין גַּם לַאֲחֵרִים. וְהוּא הַדִּין דַּהֲוָה מָצֵי לְמִתְנֵי הָיוּ יוֹשְׁבִים בַּבַּיִת, אֶלָּא שֶׁדֶּרֶךְ הַמּוֹכֵר פֵּרוֹת לֵישֵׁב בַּשַּׁעַר אוֹ בַחֲנוּת: {ד} דִּבְרֵי רַבִּי נְחֶמְיָה בְּמִשְׁנָה ה' פֶּרֶק ג'.</B> וּבַיְרוּשַׁלְמִי, רַבִּי יְהוּדָה וְרַבִּי נְחֶמְיָה אָמְרוּ דָּבָר אֶחָד: 

~ פרק ב - משנה ג

{ה} דְּאִלּוּ הִגִּיעַ הַשַּׁבָּת טָבְלוּ אַף עַל פִּי שֶׁלֹּא יִחֲדָם לְצֹרֶךְ הַשַּׁבָּת.</B> יְרוּשַׁלְמִי: {ו} <B> הָרִאשׁוֹן.</B> כְּלוֹמַר כֵּיוָן שֶׁהִגִּיעַ לַבַּיִת הָרִאשׁוֹן שֶׁבָּעִיר נִטְבְּלוּ הַפֵּרוֹת אַף עַל פִּי שֶׁלֹּא לָן שָׁם. הָרַ"שׁ: 

~ פרק ב - משנה ד

{ז} <B> וַחֲכָמִים אוֹסְרִים.</B> טַעֲמַיְהוּ, דְּסָבְרֵי דִּכְשֶׁכּוֹנְסִין תְּאֵנִים בַּכַּלְכָּלָה יֵשׁ שֶׁמַּנִּיחִין אוֹתָן כָּךְ וַהֲוָה גְמַר מְלָאכָה, אַף עַל גַּב דְּלָאו כֻּלֵּי עַלְמָא עָבְדֵי הָכִי אֶלָּא מַחֲלִיקִים אוֹתָם כִּדְאִיתָא סוֹף פֶּרֶק דִּלְעֵיל. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ב - משנה ה

{ח} בַּגְּמָרָא פֶּרֶק ז' דְּבָבָא מְצִיעָא אַסְמְכִינַן אַקְּרָא כִּי קִבְּצָם כֶּעָמִיר גֹּרְנָה (מִיכָה ד), וּפֵרֵשׁ רַשִׁ"י אֵין גֹּרֶן בְּלֹא קִבּוּץ, וְאֵין קִבּוּץ פָּחוֹת מִשְּׁתַּיִם: 

~ פרק ב - משנה ו

{ט} <B>וְאוֹכֵל.</B> פֵּרוּשׁ בְּעוֹדוֹ בִמְחֻבָּר כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם, שֶׁלֹּא יִכְרֹת הָאֶשְׁכּוֹל כֻּלּוֹ כְּדֵי שֶׁלֹּא יִתְחַיֵּב בַּמַּעַשְׂרוֹת, שֶׁהֲרֵי לֹא קָנָה אֶלָּא הַנִּתְלַשׁ. מַה שֶּׁאֵין כֵּן בְּסֵיפָא דְּאָמְרִינַן אֵלּוּ בִמְחֻבָּר קָנָה, וּתְנַן בְּרֵישׁ פֶּרֶק ה' דְּהַקּוֹנֶה בִמְחֻבָּר לֹא נִקְבַּע לַמַּעֲשֵׂר: {י} <B> בִּמְחֻבָּר.</B> וְנַעֲשָׂה כְּבַעַל הַגִּנָּה דִּתְנַן לְעֵיל דְּאוֹכֵל מֵהֶן עֲרַאי, וְעוֹד תְּנַן בִּמְחֻבָּר לַקַּרְקַע פָּטוּר. הָרַ"שׁ: 

~ פרק ב - משנה ז

{יא} <B> בְּנִי בִּשְׂכָרִי.</B> פֵּרֵשׁ רַשִׁ"י בְּמַסֶּכֶת בָּבָא מְצִיעָא דַּף צ"ב בַּשָּׂכָר שֶׁקָּצַצְתָּ לִי. וְהַתּוֹסָפוֹת מְפָרְשִׁים בִּשְׂכָרִי הַיְנוּ תַּחְתַּי. וְחַיָּב אֲפִלּוּ עַל מַה שֶּׁהָיָה אוֹכֵל הָאָב, לֹא בִלְבָד מַה שֶּׁיֹּאכַל הוּא יוֹתֵר מִן הָאָב, מִשּׁוּם דְּמֵחֲזֵי כְּמִקָּח: {יב} <B> אוֹכֵל וְחַיָּב.</B> סְתָמָא דְּלֹא כְרַבִּי יְהוּדָה דְּסוֹף פִּרְקִין שֶׁסּוֹבֵר דְּכָל מִקָּח דְּלִקְצוֹת פָּטוּר (וּבְסֵפֶר תּוֹסָפוֹת חֲדָשִׁים מוֹכִיחַ מִלְּשׁוֹן הָרַמְבַּ"ם דְּלִקְצוֹת דְּרַבִּי יְהוּדָה אֵינוֹ דָמִי לִדְהָכָא דְּרוֹצֶה לֶאֱכֹל מִיָּד וַהֲוָה אֶצְלוֹ דָּבָר הַגָּמוּר) וְעַיֵּן עוֹד אַחֲרֵי זֶה: 

~ פרק ב - משנה ח

{יג} <B> בִּלְבָסִים.</B> מַתְנִיתִין כְּשֶׁלְּבָסִים וּתְאֵנִים מֻרְכָּבִים בְּיִחוּר אֶחָד, וַאֲפִלּוּ הָכִי לֹא יֹאכַל, דְּאִי לֹא תֵימָא הָכִי הָא תָּנֵינָא בָּבָא מְצִיעָא (דַּף צא) הָיָה עוֹשֶׂה בְּיִחוּר זֶה לֹא יֹאכַל בְּיִחוּר אַחֵר. הָרַ"שׁ בְּשֵׁם הַיְרוּשַׁלְמִי: {יד} וְטַעֲמָא דְמַתְנִיתִין דִּלְעֵיל סְתָמָא וְאַחַר כָּךְ מַחֲלוֹקֶת דְּהָכָא, וּבְכִי הָא לֵית הִלְכְתָא כִּסְתָם כוּ'. וּמִיהוּ נִרְאֶה לִי דְּאַף לְרַבִּי יְהוּדָה עַד כָּאן לֹא פָטַר בְּלִקְצוֹת אֶלָּא כְּשֶׁאוֹכֵל אַחַת אַחַת וְלֹא צֵרַף שְׁתַּיִם, דְּצֵרוּף קוֹבֵעַ לְמִקָּח כְּדִלְעֵיל מִשְׁנָה ה'. תּוֹסְפוֹת יוֹם טוֹב:

~ פרק ג - משנה א

{א} <B> בָּנָיו וּבְנֵי בֵיתוֹ.</B> בַּיְרוּשַׁלְמִי מְבֹאָר דְּאַף דִּמְזוֹנוֹת אִשְׁתּוֹ עָלָיו, הוֹאִיל וַאֲכִילָתָהּ אֵינָהּ אֶלָּא מִתְּנַאי בֵית דִּין, לֹא דָמִי לִתְנַאי דְפוֹעֲלִים. לְפִי שֶׁבֵּית דִּין שֶׁהִתְנוּ וְתִקְּנוּ מְזוֹנוֹת לָאִשָּׁה, לֹא תִקְּנוּ אֶלָּא שֶׁלְּעִנְיַן מַעַשְׂרוֹת לֹא תְהֵא אֶלָּא כְמַתָּנָה: {ב} <B> אוֹכְלִין.</B> אִם נָתַן לָהֶם: {ג} דְּהָא דְּפֵרֵשׁ סוֹף פִּרְקִין דִּלְעֵיל דְּפוֹעֵל אוֹכֵל מִן הַתּוֹרָה בְּתָלוּשׁ שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ, הַיְנוּ שֶׁלֹּא נִגְמְרָה לַמַּעֲשֵׂר. וּמִיהוּ בִּשְׁעַת גְּמַר מְלָאכָה בָּעִינַן וְלֹא לְהַעֲבִירָן כוּ'. וְעַיֵּן מִשְׁנָה ב' פֶּרֶק ז' דְּבָבָא מְצִיעָא: {ד} כְּלוֹמַר וְלֹא תוֹלֶה עֵינָיו בְּדָבָר אַחֵר לִגְמָרוֹ שֶׁל הַמְּלָאכָה, אֶלָּא אוֹכְלוֹ כָּךְ כְּשֶׁהוּא לוֹקְחוֹ: 

~ פרק ג - משנה ב

{ז} <B> רַבִּי יְהוּדָה מְחַיֵּב.</B> וְהַיְנוּ בַּמַּעֲשֵׂר, אֲבָל בַּתְּרוּמָה לֹא, כִּדְאָמַר לְעֵיל מִשְׁנָה ה' הַתּוֹרֵם בְּלִבּוֹ עַל הַכֹּל. הָרַ"שׁ בְּשֵׁם הַיְרוּשַׁלְמִי. עוֹד כָּתַב דְּעַד כָּאן לֹא מְחַיֵּב רַבִּי יְהוּדָה אֶלָּא מִנֵּיהּ וּבֵיהּ, אֲבָל מֵעָלְמָא לֹא, דִּלְמָא אָתֵי לְאַפְרוּשֵׁי מִן הַחִיּוּב עַל הַפָּטוּר כוּ': 

~ פרק ה - משנה ח

{ח} <B> שֶׁאֲבִיהֶן תְּרוּמָה.</B> כְּהַהִיא דִתְנַן בְּרֵישׁ פִּרְקִין הָעוֹקֵר לֶפֶת וּצְנוֹנוֹת מִתּוֹךְ שֶׁלּוֹ וְנוֹטֵעַ לְתוֹךְ שֶׁלּוֹ לְזֶרַע, חַיָּב מִפְּנֵי שֶׁהוּא גָּרְנָן. וְאִם עֲשָׂאָן תְּרוּמָה וּנְטָעָן לְזֶרַע, הַזֶּרַע נֶאֱכָל לְזָרִים. תּוֹסְפוֹת יוֹם טוֹב:
