~ פרק א - משנה א
<b>עַד אֵימָתַי חוֹרְשִׁים בִּשְׂדֵה הָאִילָן.</b> לְקַמָּן מְפָרֵשׁ אֵיזֶהוּ שְׂדֵה אִילָן: <B>עֶרֶב שְׁבִיעִית.</B> שֶׁצָּרִיךְ לְהוֹסִיף מִשָּׁנָה שִׁשִּׁית עַל הַשְּׁבִיעִית בְּאִסּוּר עֲבוֹדַת הָאָרֶץ, וּלְקַמָּן יָלְפִינַן לַהּ מִקְּרָאֵי: <B>כָּל זְמַן שֶׁהוּא יָפֶה לַפֶּרִי.</B> כְּלוֹמַר שֶׁהַחֲרִישָׁה יָפָה לַפְּרִי, הֲוֵי מְתַקֵּן הַקַּרְקַע לְצֹרֶךְ הַפְּרִי שֶׁל שִׁשִּׁית וְלֹא לְצֹרֶךְ שְׁבִיעִית: <B>עַד הָעֲצֶרֶת.</B> דְּעַד הָעֲצֶרֶת יָפֶה לַפְּרִי וְתוּ לֹא {א}, וּמִן הָעֲצֶרֶת וְאֵילָךְ נִרְאֶה כִּמְתַקֵּן לְצֹרֶךְ שְׁבִיעִית. וּבִשְׂדֵה לָבָן שֶׁאֵין בּוֹ אִילָן עַד הַפֶּסַח. וּמִשְׁנָה זוֹ הִיא מִשְׁנָה רִאשׁוֹנָה וְאֵינָהּ הֲלָכָה, דְּרַבָּן גַּמְלִיאֵל וּבֵית דִּינוֹ נִמְנוּ עַל שְׁנֵי פְרָקִים הַלָּלוּ שֶׁהֵם פֶּסַח בִּשְׂדֵה לָבָן וַעֲצֶרֶת בִּשְׂדֵה אִילָן וְהִתִּירוּם, וּמֻתָּר לַחֲרֹשׁ עַד רֹאשׁ הַשָּׁנָה שֶׁל שְׁבִיעִית {ב}: 

~ פרק א - משנה ב
<B>לְבֵית סְאָה.</B> קַרְקַע שֶׁהִיא חֲמִשִּׁים אַמָּה עַל חֲמִשִּׁים אַמָּה: <B>כִּכַּר דְּבֵלָה.</B> הַתְּאֵנִים הַיְבֵשִׁים עֲשׂוּיִין בְּמַעֲגֵּלָה כְּעֵין כִּכַּר לֶחֶם קְרוּיִם כִּכַּר דְּבֵלָה: <B>בָּאִיטַלְקִי.</B> בְּמָנֶה שֶׁל אִיטַלְיָא שֶׁל יָוָן, וּלְפִי שֶׁהָיָה כָּךְ בִּימֵי מֹשֶׁה מְשַׁעֲרִים בּוֹ, וְהַמָּנֶה מֵאָה דִינָרִים וְהַדִּינָר שֵׁשׁ מָעוֹת וּמִשְׁקַל כָּל מָעָה שֵׁשׁ עֶשְׂרֵה שְׂעוֹרוֹת בֵּינוֹנִיּוֹת: <B>פָּחוֹת מִכָּאן.</B> מִכִּכָּר שֶׁל שִׁשִּׁים מָנֶה אֵין חוֹרְשִׁים כָּל בֵּית סְאָה, אֶלָּא תַּחַת הָאִילָן לְבָד וְחוּצָה לוֹ: <B>כִּמְלֹא אוֹרֶה וְסַלּוֹ.</B> הַמְּלַקֵּט תְּאֵנִים קָרוּי אוֹרֶה כְּמוֹ גּוֹדֵר בִּתְמָרִים וּבוֹצֵר בַּעֲנָבִים. וְסַלּוֹ, הַסַּל שֶׁנּוֹתֵן בּוֹ מַה שֶּׁהוּא לוֹקֵט. וּשְׁאָר הַשָּׂדֶה חָשׁוּב שְׂדֵה הַלָּבָן וְאֵין חוֹרְשִׁין אוֹתוֹ אֶלָּא עַד הַפֶּסַח: 

~ פרק א - משנה ג
<B>אֶחָד אִילַן סְרָק.</B> שֶׁאֵינָהּ עוֹשָׂה פֵּרוֹת וּנְטוּעִין שְׁלֹשָׁה לְבֵית סְאָה: <B>אִם רְאוּיִן לַעֲשׂוֹת.</B> רוֹאִין כָּל אִילָן וְאִילָן בִּפְנֵי עַצְמוֹ אִם אִילָן שֶׁל תְּאֵנָה כַּיּוֹצֵא בוֹ הָיָה עוֹשֶׂה כִּכַּר דְּבֵלָה. וְלָמָּה שִׁעֲרוּ בִּתְאֵנָה, עַל שֶׁפֵּרוֹתֶיהָ גַּסִּין וְעוֹשָׂה הַרְבֵּה {ז}: <B>לְצָרְכָּן.</B> מְלֹא אוֹרֶה וְסַלּוֹ חוּצָה לוֹ: 

~ פרק א - משנה ד
<B>עַד שֶׁיִּהְיוּ מִשְּׁלֹשָׁה וְעַד תִּשְׁעָה.</B> כְּלוֹמַר דִּין זֶה שֶׁאָמַרְנוּ הַיְנוּ מִשְּׁלֹשָׁה וְעַד תִּשְׁעָה {ח}, אֲבָל אִם הָיוּ עֲשָׂרָה אוֹ יוֹתֵר אַף עַל פִּי שֶׁאֵינָן עוֹשִׂין חוֹרְשִׁין: <B>שֶׁנֶּאֱמַר בֶּחָרִישׁ וּבַקָּצִיר תִּשְׁבֹּת.</B> אַרֵישָׁא קָאֵי דִּתְנַן אֵין חוֹרְשִׁין לָהֶן אֶלָּא לְצָרְכָּן. וּמִנַּיִן שֶׁאֵין חוֹרְשִׁין כָּל הַבֵּית סְאָה, שֶׁנֶּאֱמַר בֶּחָרִישׁ וּבַקָּצִיר תִּשְׁבֹּת, וְאִם אֵינוֹ עִנְיָן לְשַׁבָּת בְּרֵאשִׁית שֶׁכְּבָר נֶאֱמַר (שְׁמוֹת כ) שֵׁשֶׁת יָמִים תַּעֲבֹד וְעָשִׂיתָ כָּל מְלַאכְתֶּךָ, וְלֹא לִשְׁמִטָּה שֶׁכְּבָר נֶאֱמַר (וַיִּקְרָא כה) שֵׁשׁ שָׁנִים תִּזְרַע שָׂדֶךָ וְשֵׁשׁ שָׁנִים תִּזְמֹר כַּרְמֶךָ, תְּנֵהוּ עִנְיָן לֶאֱסֹר שְׁנֵי פְרָקִים שֶׁל עֶרֶב שְׁבִיעִית, מִפֶּסַח לִשְׂדֵה לָבָן, וּמֵעֲצֶרֶת לִשְׂדֵה אִילָן. וְהָכִי קָאָמַר קְרָא, בֶּחָרִישׁ שֶׁקְּצִירוֹ אָסוּר וְאֵיזֶה זֶה חָרִישׁ שֶׁל עֶרֶב שְׁבִיעִית שֶׁנִּכְנָס לַשְּׁבִיעִית, תִּשְׁבֹּת. וּבַקָּצִיר שֶׁחֲרִישׁוֹ אָסוּר וְאֵיזֶה זֶה קָצִיר שֶׁל שְׁבִיעִית שֶׁיּוֹצֵא לְמוֹצָאֵי שְׁבִיעִית, נַמִּי תִּשְׁבֹּת, שֶׁסְּפִיחִים שֶׁגָּדְלוּ בַּשְּׁבִיעִית אוֹ שֶׁעָבַר וְזָרַע בַּשְּׁבִיעִית אָסוּר לְקָצְרָם בַּשְּׁמִינִית: <B>רַבִּי יִשְׁמָעֵאל אוֹמֵר.</B> לְעוֹלָם בְּשַׁבָּת מִשְׁתָּעֵי קְרָא כְּפַשְׁטֵיהּ, וַאֲתָא לְאַשְׁמְעִינַן מֶה חָרִישׁ רְשׁוּת, שֶׁאֵין לְךָ חָרִישׁ שֶׁל מִצְוָה, אַף קָצִיר דְּאָמַר רַחֲמָנָא תִּשְׁבֹּת בְּקָצִיר שֶׁל רְשׁוּת, יָצָא קְצִיר הָעֹמֶר שֶׁהוּא מִצְוָה דְּבָעִינַן קְצִירָה לִשְׁמָהּ, שֶׁזֶּה דּוֹחֶה אֶת הַשַּׁבָּת: 

~ פרק א - משנה ה
<B>הֲרֵי אֵלּוּ מִצְטָרְפִין.</B> וְנֶחְשָׁבִין כְּאִלּוּ הֵן שֶׁל אִישׁ אֶחָד וְחוֹרְשִׁין כָּל בֵּית סְאָה בִּשְׁבִילָן: <B>וְכַמָּה יְהֵא בֵינֵיהֶן.</B> בֵּין אִילָן לְאִילָן: <B>כִּמְלֹא בָקָר וְכֵלָיו.</B> צֶמֶד בָּקָר עִם הָעֹל, וְהַיְנוּ אַרְבַּע אַמּוֹת. אֲבָל פָּחוֹת מִכָּאן לְמֵעֲקָר קַיְמֵי וְאֵין חוֹרְשִׁין בִּשְׁבִילָן: 

~ פרק א - משנה ו
<B>נְטִיעוֹת.</B> יְלָדוֹת, וְלִזְקֵנוֹת קָרֵי אִילָן, וְהָכָא קָא מַשְׁמַע לָן דְּאַף עַל גַּב דִּשְׁלֹשָׁה אִילָנוֹת לְבֵית סְאָה קָרוּי שְׂדֵה אִילָן וְחוֹרְשִׁין כָּל בֵּית סְאָה בִּשְׁבִילָן, בִּנְטִיעוֹת יְלָדוֹת אֵין חוֹרְשִׁין עַד שֶׁיִּהְיוּ עֲשָׂרָה: <B>שׁוּרוֹת.</B> שֶׁנְּטוּעוֹת בְּמָקוֹם אֶחָד כְּחוֹמָה וְאֵינָן מְפֻזָּרוֹת: <B>אֵין חוֹרְשִׁין אֶלָּא לְצָרְכָּן.</B> מִשּׁוּם דִּכְשֶׁמְּפֻזָּרוֹת נִרְאֶה כְּחוֹרֵשׁ לְצֹרֶךְ הָאִילָן, אֲבָל הֵיכָא דְקַיְמֵי בְּמָקוֹם אֶחָד נִרְאֶה כִּמְתַקֵּן אֶת הָאָרֶץ לְצֹרֶךְ שְׁבִיעִית וְלֹא לְצֹרֶךְ הָאִילָנוֹת: <B>שׁוּרָה.</B> תַּרְגּוּם חוֹמָה שׁוּרָה: <B>עֲטָרָה.</B> גָּדֵר סָבִיב: 

~ פרק א - משנה ז
<B>הַנְּטִיעוֹת וְהַדְּלוּעִים מִצְטָרְפִין לְבֵית סְאָה.</B> בִּדְלַעַת יְוָנִית דַּוְקָא אַיְרֵי בְּמַתְנִיתִין שֶׁהִיא גְדוֹלָה כְּאִילָן וְלֹא בִשְׁאָר דְּלוּעִין. וְצָרִיךְ שֶׁיִּהְיוּ הַנְּטִיעוֹת רַבּוֹת עַל הַדְּלוּעִים, כְּגוֹן שֶׁהָיוּ הַנְּטִיעוֹת שִׁשָּׁה וְהַדְּלוּעִים אַרְבָּעָה, מִצְטָרְפִין לַעֲשָׂרָה וְחוֹרְשִׁין כָּל בֵּית סְאָה בִּשְׁבִילָן: <B>רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר.</B> אֲפִלּוּ הָיוּ כָּל הָעֲשָׂרָה דְּלוּעִין חוֹרְשִׁין כָּל בֵּית סְאָה בִּשְׁבִילָן, לְפִי שֶׁהֵן כִּנְטִיעוֹת. וַהֲלָכָה כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל: 

~ פרק א - משנה ח
<B>עַד שֶׁיָּחֹלּוּ.</B> עַד שֶׁיֵּצְאוּ לְחֻלִּין בַּשָּׁנָה הָרְבִיעִית עַל יְדֵי פִּדְיוֹן, וְאִם לֹא פְדָאָן בַּשָּׁנָה הָרְבִיעִית, עַד שֶׁיֵּעָשׂוּ חֻלִּין מֵאֲלֵיהֶן בַּשָּׁנָה הַחֲמִשִּׁית: <B>נְטִיעָה כִּשְׁמָהּ.</B> בַּת שְׁנָתָהּ קְרוּיָה נְטִיעָה וְתוּ לֹא. פֵּרוּשׁ אַחֵר כִּשְׁמָהּ כָּל זְמַן שֶׁבְּנֵי אָדָם קוֹרְאִין לָהּ נְטִיעָה וְאֵין קוֹרְאִין לָהּ אִילָן. וַהֲלָכָה כְּרַבִּי עֲקִיבָא: <B>שֶׁנִּגְמַם.</B> שֶׁנֶּחְתַּךְ: <B>חֲלִיפִין.</B> גִּדּוּלִים, לְשׁוֹן גִּזְעוֹ מַחֲלִיף: <B>כִּנְטִיעָה.</B> בֵּין לְעִנְיַן חֲרִישָׁה בֵּין לְעִנְיַן עָרְלָה. וַהֲלָכָה כְּרַבִּי שִׁמְעוֹן:

~ פרק ב - משנה א
<B>עַד אֵימָתַי, שְׂדֵה הַלָּבָן.</B> שָׂדֶה שֶׁל תְּבוּאָה וְקִטְנִית שֶׁאֵין בּוֹ אִילָן {א}: <B>הַלֵּחָה.</B> לַחְלוּחִית הַקַּרְקַע מֵחֲמַת הַגְּשָׁמִים: <B>מִקְשָׁאוֹת וּמִדְלָעוֹת.</B> קִשּׁוּאִים {ג} וּדְלוּעִים. אֲבָל מִשָּׁם וְאֵילֵךְ מֵחֲזֵי כִּמְתַקֵּן שָׂדֵהוּ לְצֹרֶךְ שְׁבִיעִית: <B>נָתַתָּ תּוֹרַת כָּל אֶחָד בְּיָדוֹ.</B> זֶה אוֹמֵר כָּלְתָה לֵחָה בְּתוֹךְ שֶׁלִּי וְזֶה אוֹמֵר לֹא כָלְתָה לֵחָה בְּתוֹךְ שֶׁלִּי: <B>אֶלָּא בִּשְׂדֵה לָבָן.</B> שֶׁעָתִיד לְזָרְעָהּ אַחַר הַחֲרִישָׁה וְצָרִיךְ שֶׁתְּהֵא רֹב לֵחָה קַיֶּמֶת, אֵין חוֹרְשִׁים אֶלָּא עַד הַפֶּסַח: <B>וּשְׂדֵה אִילָן.</B> שֶׁהִיא נְטוּעָה כְּבָר אֵין צָרִיךְ שֶׁתְּהֵא רֹב לֵחָה קַיֶּמֶת שֶׁאֵין חוֹרְשִׁין אוֹתָהּ אֶלָּא כְּדֵי שֶׁיֵּרְדוּ הַגְּשָׁמִים בְּעֹמֶק הַקַּרְקַע, הִלְכָּךְ חוֹרְשִׁים בָּהּ עַד הָעֲצֶרֶת. וְכָל הַמִּשְׁנָיוֹת הַלָּלוּ דְּחוּיוֹת הֵן כִּדְאָמְרִינַן לְעֵיל בְּפֶרֶק קַמָּא שֶׁרַבָּן גַּמְלִיאֵל וּבֵית דִּינוֹ נִמְנוּ עַל שְׁנֵי פְרָקִים הַלָּלוּ שֶׁהֵם פֶּסַח וַעֲצֶרֶת וּבִטְּלוּם, וּמֻתָּר לַחֲרֹשׁ עַד רֹאשׁ הַשָּׁנָה שֶׁל שְׁבִיעִית: 

~ פרק ב - משנה ב
<B>מְזַבְּלִין.</B> מַכְנִיסִין לָהֶם זֶבֶל: <B>מְעַדְּרִים.</B> חוֹפְרִים בְּעִקְּרֵי הָאִילָנוֹת {ד}, שֶׁכָּל דָּבָר שֶׁהוּא לְצֹרֶךְ פֵּרוֹת שֶׁל שִׁשִּׁית מֻתָּר בְּתוֹסֶפֶת שְׁבִיעִית, וְדָבָר שֶׁאֵינוֹ אֶלָּא לְצֹרֶךְ תִּקּוּן הָאִילָן אָסוּר, אֶלָּא אִם כֵּן הוּא דָּבָר שֶׁאֲפִלּוּ בַּשְּׁבִיעִית עַצְמָהּ אֵין אִסּוּרוֹ אֶלָּא מִדִּבְרֵי סוֹפְרִים, וּבְתוֹסֶפֶת שְׁבִיעִית לֹא גָזוּר: <B>בֵּית הַשְּׁלָחִין.</B> שֶׁאֵין דַּי לָהּ בְּמֵי גְשָׁמִים וְצָרִיךְ לְהַשְׁקוֹתָהּ, תַּרְגּוּם וְהוּא עָיֵף (בְּרֵאשִׁית כה) וְהוּא מְשַׁלְהֵי: <B>מְיַבְּלִין.</B> חוֹתְכִין הַיַּבֶּלֶת, מוּמִין הַנּוֹלָדִין בָּאִילָן: <B>מְפָרְקִין.</B> הֶעָלִין מֵעַל הָאִילָן לְהָקֵל מֵעָלָיו: <B>מְאַבְּקִין.</B> שָׁרָשִׁים הַמְּגֻלִּים מְכַסִּים אוֹתָן בְּאָבָק: <B>מְעַשְּׁנִין.</B> תַּחַת הָאִילָן לְהָמִית הַתּוֹלָעִים הַגְּדֵלִים בּוֹ: <B>אַף נוֹטֵל אֶת הֶעָלִין.</B> בַּשְּׁבִיעִית עַצְמָהּ, וְתַנָּא קַמָּא לֹא אָמַר אֶלָּא מְפָרְקִין אֶת הֶעָלִין בְּתוֹסֶפֶת שְׁבִיעִית לְבָד. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: 

~ פרק ב - משנה ג
<B>מְסַקְּלִין.</B> מְסִירִין אֶת הָאֲבָנִים: <B>מְקַרְסְמִין.</B> כְּמוֹ מְכַרְסְמִין בְּכַ"ף, לְשׁוֹן יְכַרְסְמֶנָּה חֲזִיר מִיָּעַר (תְּהִלִּים פ), כְּלוֹמַר כּוֹרְתִים וְחוֹתְכִין הָעֲנָפִים הַיְבֵשִׁים מִן הָאִילָן: <B>מְזָרְדִין.</B> הָעֲנָפִים הַלַּחִים כְּשֶׁהֵם מְרֻבִּין רְגִילִים לְקָצְצָן וּמַנִּיחִין מִקְצָת מֵהֶן, וְזֶהוּ זֵרוּד: <B>מְפַסְּלִין.</B> שֶׁנּוֹטֵל אֶת הַפְּסֹלֶת. וְיֵשׁ מְפָרְשִׁים מְפַסְּלִין לְשׁוֹן פְּסָל לְךָ (שְׁמוֹת לד), שֶׁחוֹתְכִים כָּל הָעֲנָפִים שֶׁבָּאִילָן כְּדֵי שֶׁיִּתְעַבֶּה: <B>כְּזֵרוּדָהּ וּכְפִסּוּלָהּ שֶׁל חֲמִשִּׁית וְכוּ'.</B> תַּנָּא קַמָּא לֹא שָׁרֵי אֶלָּא עַד רֹאשׁ הַשָּׁנָה שֶׁל שְׁבִיעִית. וְרַבִּי יְהוֹשֻׁעַ סָבַר כִּשְׁנַת חֲמִשִּׁית כֵּן שִׁשִּׁית, מַה חֲמִשִּׁית נִכְנֶסֶת לַשִּׁשִּׁית כָּךְ שִׁשִּׁית נִכְנֶסֶת לַשְּׁבִיעִית, וּכְמוֹ שֶׁמְּזָרְדִין שָׁל חֲמִשִּׁית בַּשִּׁשִּׁית כָּךְ מְזָרְדִין שֶׁל שִׁשִּׁית בַּשְּׁבִיעִית. וְרַבִּי שִׁמְעוֹן אוֹמֵר מֵעֲצֶרֶת וְאֵילָךְ שֶׁהוּא נֶאֱסַר בַּעֲבוֹדַת הָאִילָן נֶאֱסַר נַמִּי בְּזֵרוּד וּפִסּוּל. וְשָׁלֹשׁ מַחְלוֹקוֹת בַּדָּבָר, וַהֲלָכָה כְּתַנָּא קַמָּא: 

~ פרק ב - משנה ד
<B>מְזַהֲמִין אֶת הַנְּטִיעוֹת.</B> כְּשֶׁקְּלִפַּת הָאִילָן נוֹשֶׁרֶת מַדְבִּיקִים שָׁם זֶבֶל לְהַבְרִיאוֹ כְּדֵי שֶׁלֹּא יָמוּת הָאִילָן. וְיֵשׁ מְפָרְשִׁים מְזַהֲמִין שֶׁמּוֹשְׁחִים הָאִילָנוֹת בְּדָבָר שֶׁרֵיחוֹ רַע כְּדֵי שֶׁיִּבְרְחוּ הַתּוֹלָעִים {ה} מִן הָרֵיחַ אוֹ שֶׁיָּמוּתוּ: <B>וְכוֹרְכִים.</B> יֵשׁ מְפָרְשִׁים שֶׁכּוֹרְכִים דָּבָר סְבִיב הָאִילָן בַּקַּיִץ מִפְּנֵי הַחַמָּה וּבַחֹרֶף מִפְּנֵי הַצִּנָּה. וְיֵשׁ מְפָרְשִׁים כּוֹרְכִים הָעֲנָפִים זוֹ עִם זוֹ וְקוֹשְׁרִין אוֹתָם לְמַעְלָה בְּרָאשֵׁיהֶם שֶׁלֹּא יִהְיוּ נְטוּשִׁים עַל הָאָרֶץ: <B>וְקוֹטְמִין אוֹתָן.</B> יֵשׁ מְפָרְשִׁים לָשִׁים בְּשָׁרְשֵׁיהֶם אֵפֶר, תַּרְגּוּם אֵפֶר קִטְמָא. וְיֵשׁ מְפָרְשִׁים שׁוֹבֵר הָרָאשִׁים, כְּמוֹ נִקְטַם רֹאשׁוֹ [סֻכָּה כט:]: <B>וְעוֹשִׂים לָהֶם בָּתִּים.</B> רְגִילִים לַעֲשׂוֹת לָאִילָן גָּדֵר אַמָּה סְבִיבוֹתָיו וּמְמַלְּאִים אוֹתוֹ עָפָר. וְיֵשׁ מְפָרְשִׁים שֶׁעוֹשִׂים לוֹ סֻכָּה מִלְּמַעְלָה לְהָגֵן עָלָיו מִן הַחַמָּה אוֹ מִן הַצִּנָּה: <B>עַל הַנּוֹף.</B> שֶׁל הָאִילָן, אֲבָל לֹא עַל הָעִקָּר, שֶׁלֹּא יַעֲשֶׂה בַּשְּׁבִיעִית כְּדֶרֶךְ שֶׁעוֹשֶׂה בִּשְׁאָר הַשָּׁנִים. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר בְּרַבִּי צָדוֹק, אֶלָּא אָסוּר לְהַשְׁקוֹת הָאִילָן בַּשְּׁבִיעִית בֵּין עַל הַנּוֹף בֵּין עַל הָעִקָּר: 

~ פרק ב - משנה ה
<B>סָכִין אֶת הַפַּגִּים.</B> פֵּרוֹת שֶׁלֹּא נִגְמַר בִּשּׁוּלָן סָכִין אוֹתָן בְּשֶׁמֶן כְּשֶׁהֵן מְחֻבָּרִים לָאִילָן לְמַהֵר בִּשּׁוּלָן. וּפְעָמִים שֶׁמְּנַקְּבִים אוֹתָן וּמְשִׂימִים שֶׁמֶן בְּתוֹךְ הַנֶּקֶב, אִי נַמִּי מְנַקְּבִים הַפַּגִּים כְּדֵי שֶׁיִּכָּנְסוּ בָהֶם גְּשָׁמִים וְיִתְבַּשְּׁלוּ מְהֵרָה: <B>פַּגֵּי עֶרֶב שְׁבִיעִית.</B> פֵּרוֹת שֶׁאֵין מִתְבַּשְּׁלִים עַד תִּשְׁרֵי שֶׁל שְׁבִיעִית {ז}, אֵין סָכִין וְאֵין מְנַקְּבִין אוֹתָן בַּשִּׁשִּׁית, מֵאַחַר שֶׁאֵין נִגְמַר בִּשּׁוּלָן עַד שֶׁתִּכָּנֵס שְׁבִיעִית. וַאֲפִלּוּ בְּמָקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לָסוּךְ אָסַר תַּנָּא קַמָּא. וְרַבִּי יְהוּדָה מַפְלִיג בֵּין מָקוֹם שֶׁנָּהֲגוּ לָסוּךְ לְמָקוֹם שֶׁלֹּא נָהֲגוּ לָסוּךְ, דִּבְמָקוֹם שֶׁלֹּא נָהֲגוּ לָסוּךְ לֹא מְחַזְיָא סִיכָה עֲבוֹדָה וְשָׁרֵי: <B>רַבִּי שִׁמְעוֹן מַתִּיר בָּאִילָן.</B> שֶׁהוּא מָלֵא פֵּרוֹת שְׁבִיעִית, לַעֲשׂוֹת בּוֹ מְלָאכָה בְּמוֹצָאֵי שְׁבִיעִית {ח}, מִפְּנֵי שֶׁהוּא רַשַּׁאי בַּעֲבוֹדַת הָאִילָן אַף עַל פִּי שֶׁהַפֵּרוֹת קְדוֹשִׁים בִּקְדֻשַּׁת שְׁבִיעִית. וַהֲלָכָה כְּתַנָּא קַמָּא:

~ פרק ג - משנה א
<B>מֵאֵימָתַי מוֹצִיאִים זְבָלִים לְאַשְׁפּוֹת.</B> דֶּרֶךְ עוֹבְדֵי אֲדָמָה לְכַנֵּס הַזֶּבֶל לְמָקוֹם אֶחָד בַּשָּׂדֶה וְעוֹשֶׂה שָׁם אַשְׁפָּה גְדוֹלָה עַד שֶׁהוּא מְפַזְּרוֹ בַּשָּׂדֶה לְזַבְּלָהּ: <B>מִשֶׁיִּפְסְקוּ עוֹבְרֵי עֲבֵרָה.</B> לַעֲבֹד אֶת הָאָרֶץ בַּשָּׁנָה הַשְּׁבִיעִית, וְקֹדֶם שֶׁיִּפְסְקוּ עוֹבְרֵי עֲבֵרָה אָסוּר, שֶׁלֹּא יֹאמְרוּ מֵעוֹבְרֵי עֲבֵרָה הוּא {א} וּלְזַבֵּל שָׂדֵהוּ הוּא מוֹצִיא. וְיֵשׁ סְפָרִים שֶׁכָּתוּב בָּהֶם עוֹבְדֵי עֲבוֹדָה, וְהִיא הִיא: <B>מִשֶּׁיִּיבַשׁ הַמָּתוֹק.</B> מִשֶּׁיִּהְיֶה יָבֵשׁ הַזֶּבֶל שֶׁנּוֹתֵן מְתִיקָה {ב} לַפֵּרוֹת: <B>מִשֶּׁיִּקָּשֵׁר.</B> כְּשֶׁמַּתְחִיל לְהִתְיַבֵּשׁ מִתְעַבֶּה וְנַעֲשֶׂה קְשָׁרִים, וּקְרוֹבִים דִּבְרֵיהֶם לִהְיוֹת שָׁוִים. וּבַיְרוּשַׁלְמִי מְפָרֵשׁ מִשֶּׁיִּיבַשׁ הַמָּתוֹק סְדָקִין שֶׁבַּבִּקְעָה כְּשֶׁהַמָּטָר יוֹרֵד מִתְמַלְּאִים מַיִם וְאֵינָם מִתְיַבְּשִׁים אֶלָּא לְאַחַר זְמַן, וְנִקְרָאִים מָתוֹק עַל שֵׁם מָתְקוּ לוֹ רִגְבֵי נָחַל (אִיּוֹב כא): 

~ פרק ג - משנה ב
<B>מַשְׁפֵּלוֹת.</B> קֻפּוֹת שֶׁמּוֹצִיאִין בָּהֶם זְבָלִים, שֶׁכָּל מַשְׁפֵּל מַחֲזִיק לֶתֶךְ שֶׁהוּא מַשְּׂאוֹי שֶׁל חֲמוֹר, וְהַלֶּתֶךְ חֲצִי כּוֹר, חֲמֵשׁ עֶשְׂרֵה סְאִין: <B>מוֹסִיפִין עַל הַמַּשְׁפֵּלוֹת.</B> נוֹתֵן בְּאַשְׁפָּה אַחַת מַשְׁפֵּלוֹת כְּמוֹ שֶׁהוּא רוֹצֶה: <B>וְאֵין מוֹסִיפִין עַל הָאַשְׁפּוֹת.</B> לַעֲשׂוֹת יוֹתֵר מִשָּׁלֹשׁ אַשְׁפּוֹת בְּבֵית סְאָה, שֶׁלֹּא יֵרָאֶה כִּמְזַבֵּל אֶת הַשָּׂדֶה: <B>רַבִּי שִׁמְעוֹן אוֹמֵר אַף עַל הָאַשְׁפּוֹת.</B> מוֹסִיפִין, כֵּיוָן שֶׁיֵּשׁ צִבּוּר גָּדוֹל בְּמָקוֹם אֶחָד לֹא מֵחֲזֵי כִּמְזַבֵּל. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: 

~ פרק ג - משנה ג
<B>יָתֵר מִכָּאן.</B> מִשָּׁלֹשׁ [אַשְׁפּוֹת] מֻתָּר. וְהַיְנוּ דְּרַבִּי שִׁמְעוֹן דְּרֵישָׁא, וְהַשְׁתָּא אָתָא לְאַשְׁמְעִינַן דְּשָׁרֵי רַבִּי שִׁמְעוֹן אֲפִלּוּ אֵין עֶשֶׂר מַשְׁפֵּלוֹת, וּבִלְבָד שֶׁלֹּא יִהְיֶה בָהּ פָּחוֹת מִשָּׁלֹשׁ, דְּאַף עַל פִּי שֶׁיֵּשׁ מְעַט זֶבֶל בָּאַשְׁפָּה וְסַלְקָא דַעְתָּךְ אֲמֵינָא דְּמוֹדֶה רַבִּי שִׁמְעוֹן דְּאִם מוֹסִיף הָאַשְׁפּוֹת נִרְאֶה כִּמְזַבֵּל, קָא מַשְׁמַע לָן דְּלֹא מֵחֲזֵי כִּמְזַבֵּל הוֹאִיל וְאֵינוֹ מְפַזְּרוֹ: <B>עַד שֶׁיַּעֲמִיק שְׁלֹשָׁה.</B> שֶׁיַּעֲמִיק מְקוֹם הָאַשְׁפָּה אוֹ יַגְבִּיהַּ מִשְּׁאָר הַקַּרְקַע שְׁלֹשָׁה טְפָחִים, שֶׁיִּהְיֶה נִכָּר דְּלִכְנֹס הַזֶּבֶל לְמָקוֹם אֶחָד הוּא עוֹשֶׂה וְלֹא לְזַבֵּל אֶת הַשָּׂדֶה: <B>עוֹשֶׂה אָדָם אֶת זִבְלוֹ אוֹצָר.</B> אוֹתָם שָׁלֹשׁ אַשְׁפּוֹת שֶׁל שְׁלֹשִׁים מַשְׁפֵּלוֹת יָכוֹל לַעֲשׂוֹת אוֹתָם אַשְׁפָּה אַחַת וּלְהוֹסִיף עֲלֵיהֶם כְּמוֹ שֶׁיִּרְצֶה, דְּלֹא תֵימָא כִּי הֵיכִי דְאָסוּר יוֹתֵר מִשָּׁלֹשׁ אַשְׁפּוֹת לְבֵית סְאָה הָכִי נַמִּי אָסוּר יוֹתֵר מִשִּׁעוּר שָׁלֹשׁ אַשְׁפּוֹת בְּמָקוֹם אֶחָד: <B>רַבִּי מֵאִיר אוֹסֵר.</B> בַּיְרוּשַׁלְמִי מוּכָח דִּבְיוֹתֵר מִכַּשִּׁעוּר בִּלְבָד אָסַר רַבִּי מֵאִיר. אֲבָל בְּכַשִּׁעוּר, הַשְׁתָּא בִּשְׁלֹשָׁה מְקוֹמוֹת מֻתָּר, בְּמָקוֹם אֶחָד לֹא כָּל שֶׁכֵּן. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר: <B>הָיָה לוֹ דָּבָר מֻעָט.</B> שֶׁאֵין לוֹ לְהוֹצִיא בְּפַעַם אַחַת כַּשִּׁעוּר הַמְפֹרָשׁ לְעֵיל, מוֹצִיא אוֹתוֹ מְעַט וּמוֹסִיף עָלָיו וְהוֹלֵךְ {ו}: <B>רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה אוֹסֵר.</B> שֶׁמָּא לֹא יִהְיֶה לוֹ זֶבֶל יוֹתֵר וְנִרְאֶה כִּמְזַבֵּל שָׂדֵהוּ, עַד שֶׁיּוֹצִיא עֶשֶׂר מַשְׁפֵּלוֹת כְּאַחַת אוֹ עַד שֶׁיַּעֲמִיק שְׁלֹשָׁה אוֹ יַגְבִּיהַּ שְׁלֹשָׁה. שֶׁכֵּיוָן שֶׁאוֹתוֹ מָקוֹם מְשֻׁנֶּה מִשְּׁאָר הַשָּׂדֶה לֹא מֵחֲזֵי כִּמְזַבֵּל, וְכֵן אִם נְתָנוֹ עַל הַסֶּלַע שֶׁאֵינוֹ רָאוּי לִזְרִיעָה: 

~ פרק ג - משנה ד
<B>הַמְדַיֵּר שָׂדֵהוּ.</B> לְשׁוֹן דִּיר שֶׁל בְּהֵמוֹת, שֶׁאֵין לוֹ מָקוֹם פָּנוּי לְהַעֲמִיד בֶּהֱמוֹתָיו אֶלָּא שָׁם, וְאֵינוֹ מִתְכַּוֵּן לְזַבֵּל שָׂדֵהוּ בְּעִנְיָן זֶה, וְדַעְתּוֹ לְהַכְנִיס הַזֶּבֶל לָאַשְׁפּוֹת אַחַר כָּךְ: <B>סַהַר.</B> כְּמוֹ סַחַר {ז}, הֶקֵּף שֶׁל מְחִצָּה סְבִיב הַצֹּאן: <B>עוֹקֵר שָׁלֹשׁ רוּחוֹת.</B> כְּשֶׁנִּתְמַלְּאָה הַסַּהַר זֶבֶל עוֹקֵר מִן הַסַּחַר שָׁלֹשׁ רוּחוֹת וְזוֹקְפָן לְצַד [אַחֵר] שֶׁל אֶמְצָעִית וּמְדַיֵּר שָׁם כְּמוֹ כֵן בֵּית סָאתַיִם, וְעוֹשֶׂה אַשְׁפּוֹת בְּתוֹךְ שָׂדֵהוּ שָׁלֹשׁ אַשְׁפּוֹת לְבֵית סְאָה כִּדְאָמְרָן: <B>נִמְצָא מְדַיֵּר בֵּית אַרְבַּעַת סְאִים.</B> סָאתַיִם מִצַּד זֶה שֶׁל מְחִצָּה אֶמְצָעִית וְסָאתַיִם מִצַּד אַחֵר. וְאַשְׁמְעִינַן מַתְנִיתִין דְּשָׁרֵי לְדַיֵּר בֵּית אַרְבַּעַת סְאִין קֹדֶם שֶׁיַּעֲשֶׂה מִמֶּנָּה אַשְׁפּוֹת, אֲבָל טְפֵי לֹא, שֶׁלֹּא יֹאמְרוּ לְדַיֵּר שָׂדֵהוּ מִתְכַּוֵּן: <B>מִפְּנֵי מַרְאִית הָעַיִן.</B> שֶׁלֹּא יְהֵא נִרְאֶה כְּמִתְכַּוֵּן לְזַבֵּל שָׂדֵהוּ אִם עוֹשֶׂה אֶת כֻּלָּהּ סַחַר: <B>כְּדֶרֶךְ הַמְּזַבְּלִין.</B> שָׁלֹשׁ אַשְׁפּוֹת לְכָל בֵּית סְאָה. וְאֵין הֲלָכָה כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל: 

~ פרק ג - משנה ה
<B>מַחְצֵב.</B> מָקוֹם בְּתוֹךְ שָׂדֵהוּ שֶׁחוֹצְבִין מִמֶּנּוּ אֲבָנִים לְבִנְיָן. אִם הָיָה מְכֻסֶּה עָפָר וְלֹא הָיָה נִרְאֶה אֵין פּוֹתְחִין אוֹתוֹ בַּתְּחִלָּה בַּשְּׁבִיעִית, שֶׁהָרוֹאֶה אוֹמֵר לְתַקֵּן שָׂדֵהוּ הוּא עוֹשֶׂה כְּדֵי לִזְרֹעַ בְּתוֹכוֹ וְלֹא לְהוֹצִיא הָאֲבָנִים לְבִנְיָן: <B>עַד שֶׁיְּהֵא בוֹ שָׁלֹשׁ מוּרְבִּיּוֹת.</B> שֶׁיִּהְיוּ נִגְלִים וְנִרְאִים מֵאוֹתוֹ מַחֲצֵב קֹדֶם שְׁבִיעִית שָׁלֹשׁ שׁוּרוֹת שֶׁל אֲבָנִים, שֶׁכָּל שׁוּרָה מֵהֶם שָׁלֹשׁ אַמּוֹת אֹרֶךְ וְשָׁלֹשׁ אַמּוֹת רֹחַב בְּרוּם שָׁלֹשׁ אַמּוֹת, שֶׁנִּמְצְאוּ שָׁם תֵּשַׁע אֲבָנִים שֶׁכָּל אֶבֶן הוּא אַמָּה עַל אַמָּה בְּרוּם שָׁלֹשׁ, וְכֵן בְּכָל שׁוּרָה וְשׁוּרָה, הֲרֵי עֶשְׂרִים וְשֶׁבַע אֲבָנִים לְשָׁלֹשׁ שׁוּרוֹת, וּבְהָכִי נִכָּר שֶׁלְּצֹרֶךְ אֲבָנִים לְבִנְיָן הוּא מוֹצִיאָם וְלֹא לְתַקֵּן שָׂדֵהוּ: 

~ פרק ג - משנה ו
<B>שֶׁל מַשְּׂאוֹי שְׁנַיִם שְׁנַיִם.</B> שֶׁיֵּשׁ בְּכָל אַחַת מֵהֶן מַשְּׂאוֹי שְׁנֵי בְנֵי אָדָם: <B>הֲרֵי אֵלּוּ יִנָּטֵלוּ.</B> כֻּלָּן וַאֲפִלּוּ הַקְּטַנּוֹת שֶׁבָּהֶן {יא}, וְלֹא מִתְחֲזֵי כִּמְפַנֶּה הַגָּדֵר לַעֲשׂוֹת מְקוֹמוֹ שָׂדֶה: <B>פָּחוֹת מִכָּאן מְחַצֵּב.</B> וְאֵינוֹ נוֹטְלוֹ כֻּלּוֹ אֶלָּא גוֹמְמוֹ וּמַנִּיחַ מִן הַסֶּלַע בָּאָרֶץ טֶפַח כְּדֵי שֶׁלֹּא יְהֵא רָאוּי לִזְרִיעָה: <B>בַּמֶּה דְבָרִים אֲמוּרִים.</B> הַאי דְלֹא יִפְתַּח מַחְצֵב וְהַאי דְגוֹמְמוֹ: <B>מַה שֶּׁהוּא רוֹצֶה יִטֹּל.</B> דְּלֹא מֵחֲזֵי כִּמְתַקֵּן שָׂדֶה כֵּיוָן דְּלָאו דִּידֵיהּ הוּא, וַאֲפִלּוּ בְּשֶׁלּוֹ נַמִּי לֹא מֵחֲזֵי כִּמְתַקֵּן שָׂדֶה אִם הִתְחִיל לִטֹּל מֵעֶרֶב שְׁבִיעִית: 

~ פרק ג - משנה ז

<B>אֵין בּוֹנִין מַדְרֵגוֹת.</B> מַעֲלוֹת עַל פִּי הַגֵּאָיוֹת, שֶׁלֹּא יַחֲלִיקוּ בִּירִידָתָן לָקַחַת הַמַּיִם שֶׁנִּתְקַבְּצוּ בָּעֵמֶק בְּאוֹתָן הַגֻּמּוֹת, דְּנִרְאֶה כִּמְתַקְּנָן לְהַשְׁקוֹת בָּהֶן שְׂדוֹתָיו בַּשְּׁבִיעִית: <B>לֹא יִסְמֹךְ בְּעָפָר.</B> אִם בָּא לִהְיוֹת סוֹכֵר מְקוֹם יְצִיאַת הַמַּיִם בַּאֲבָנִים, לֹא יִתֵּן עָפָר וְטִיט בֵּין הָאֲבָנִים, מִשּׁוּם דְמֵחֲזֵי טְפֵי שֶׁהוּא מִתְכַּוֵּן לְהַשְׁקוֹת מֵהֶן שְׂדוֹתָיו: <B>אֲבָל עוֹשֶׂה הוּא חַיִץ.</B> אֲבָנִים סְדוּרוֹת זוֹ עַל זוֹ כְּעֵין גָּדֵר בְּלֹא עָפָר וְטִיט: <B>כָּל אֶבֶן.</B> שֶׁבְּשָׂדֵהוּ אֲפִלּוּ קְטַנָּה, אִם כְּשֶׁהוּא בּוֹנֶה הַגָּדֵר אוֹ הַבִּנְיָן יָכוֹל לִפְשֹׁט יָדוֹ וְלִטְּלָהּ וְלִבְנוֹת בָּהּ: <B>הֲרֵי זוֹ תִּנָּטֵל.</B> שֶׁהַגָּדֵר מוֹכַחַת עָלָיו דְּלִבְנוֹת גִּדְרוֹ הוּא נוֹטְלָהּ וְלֹא לְתַקֵּן הַשָּׂדֶה לִזְרִיעָה:

<B>בָּאוֹת מִכָּל מָקוֹם.</B> אֲפִלּוּ מִתּוֹךְ שֶׁלּוֹ, דְּמֵידַע יְדִיעַ דִּלְבִנְיָן קָבָעֵי לְהוּ וְלֹא לְתַקֵּן אֶת הַשָּׂדֶה: <B>וְהַקַּבְּלָן.</B> שֶׁקִּבֵּל שָׂדֶה בַּאֲרִיסוּת מֵבִיא מִכָּל מָקוֹם, וַאֲפִלּוּ מֵאוֹתָהּ שָׂדֶה שֶׁהוּא אָרִיס בָּהּ, אֲפִלּוּ אֲבָנִים קְטַנּוֹת, שֶׁאֵין [שָׂדֶה זוֹ חֲשׁוּבָה כְּשֶׁלּוֹ. וְיֵשׁ מְפָרְשִׁים וְהַקַּבְּלָן שֶׁדַּרְכּוֹ לִבְנוֹת בִּנְיָנִים בְּקִבּוֹלֶת, שֶׁאֵין] כַּוָּנָתוֹ אֶלָּא לְקַבֵּץ אֲבָנִים, מֵבִיא [מִכָּל מָקוֹם {טו} וַאֲפִלּוּ מִשָּׂדֶה] שֶׁלּוֹ: <B>רַבִּי יוֹסֵי אוֹמֵר כוּ'.</B> וַהֲלָכָה כְּרַבִּי יוֹסֵי:

<B>לְהַעֲמִיק עַד הַסֶּלַע.</B> וְלֹא חַיְשִׁינַן דִּלְמָא מִמְלָּךְ וְזָרַע לֵיהּ {טז}, שֶׁאֵין דֶּרֶךְ בְּנֵי אָדָם לִזְרֹעַ סָמוּךְ לִרְשׁוּת הָרַבִּים. אֲבָל בֵּינוֹ לְבֵין חֲבֵרוֹ חַיְשִׁינַן דִּלְמָא מִמְלָּךְ וְזָרַע לֵיהּ, וְאָסוּר: <B>צוֹבְרוֹ בִּרְשׁוּת הָרַבִּים.</B> וְאַחַר כָּךְ נוֹטְלוֹ מִשָּׁם וּמְתַקֵּן אֶת רְשׁוּת הָרַבִּים כְּמוֹ שֶׁהָיָה: <B>כְּשֵׁם שֶׁאֵין מְקַלְקְלִים רְשׁוּת הָרַבִּים כָּךְ לֹא יְתַקֵּן.</B> כְּלוֹמַר לֹא מִבַּעְיָא לִצְבֹּר עֲפָרוֹ לִרְשׁוּת הָרַבִּים שֶׁהוּא אָסוּר אַף עַל פִּי שֶׁיְּסִירֶנּוּ אַחַר כָּךְ, שֶׁהֲרֵי מְקַלְקֵל הוּא לְשָׁעָה, אֶלָּא אֲפִלּוּ לְתַקֵּן אָסוּר, כְּגוֹן אִם הָיוּ אֲבָנִים אוֹ עָפָר מְפֻזָּרִים בִּרְשׁוּת הָרַבִּים לֹא יִצְבְּרֵם בְּמָקוֹם אֶחָד, אֶלָּא יְסַלְּקָם לְגַמְרֵי מֵרְשׁוּת הָרַבִּים, וְהַיְנוּ דְקָאָמַר כְּשֵׁם שֶׁאֵין מְקַלְקְלִין רְשׁוּת הָרַבִּים לִצְבֹּר עֲפָרוֹ לִרְשׁוּת הָרַבִּים אַף עַל פִּי שֶׁדַּעְתּוֹ לְסַלְּקוֹ מִיָּד, כָּךְ לֹא יְתַקֵּן לִצְבֹּר אֲבָנִים אוֹ עָפָר הַמְפֻזָּרִים בִּרְשׁוּת הָרַבִּים וְלָשׂוּם אוֹתָם בְּצִדֵּי רְשׁוּת הָרַבִּים לְפִי שָׁעָה, אֶלָּא יְסַלְּקָם מִיָּד לְגַמְרֵי מִכָּל רְשׁוּת הָרַבִּים וְיוֹצִיאֵם לַיָּם אוֹ לַנָּהָר: <B>מַה יַּעֲשֶׂה בֶּעָפָר.</B> אַלִּיבָּא דְרַבִּי עֲקִיבָא: <B>צוֹבְרוֹ בְּתוֹךְ שָׂדֵהוּ כְּדֶרֶךְ הַמְזַבְּלִים.</B> שָׁלֹשׁ אַשְׁפּוֹת לְבֵית סְאָה: <B>וְכֵן הַחוֹפֵר בּוֹר שִׁיחַ וּמְעָרָה.</B> בִּשְׁאָר יְמוֹת הַשָּׁנָה, פְּלִיגֵי בָהּ רַבִּי יְהוֹשֻׁעַ וְרַבִּי עֲקִיבָא. לְרַבִּי יְהוֹשֻׁעַ צוֹבְרוֹ בִּרְשׁוּת הָרַבִּים, וּבִלְבָד שֶׁיְּסַלְּקֶנּוּ אַחַר כָּךְ וִיתַקֵּן רְשׁוּת הָרַבִּים כְּמוֹ שֶׁהָיָה, וּלְרַבִּי עֲקִיבָא אַף עַל פִּי שֶׁדַּעְתּוֹ לְתַקֵּן אָסוּר, אֶלָּא צוֹבְרוֹ בִּרְשׁוּתוֹ. וַהֲלָכָה כְּרַבִּי עֲקִיבָא:

<B>שֶׁנִּתְקַוְּצָה.</B> שֶׁנִּטְּלוּ קוֹצֶיהָ בַּשְּׁבִיעִית: <B>שֶׁנִּטַּיְּבָה.</B> שֶׁנֶּחְרְשָׁה הֵיטֵב, שֶׁכָּל הָעוֹלָם חוֹרְשִׁים פַּעַם אַחַת וְהוּא חָרַשׁ שְׁתֵּי פְּעָמִים. וּמַתְנִיתִין אַיְרֵי בִּשְׁעַת הַסַּכָּנָה שֶׁהָיְתָה הַמַּלְכוּת אוֹנֶסֶת עַל הַמַּס הַמּוּטָל עַל הַקַּרְקָעוֹת וְהִתִּירוּ לַחֲרֹשׁ בַּשְּׁבִיעִית כְּדֵי לִתֵּן הַמַּס הַקָּצוּב מִתְּבוּאָתָהּ לַמֶּלֶךְ {ג}, וְאִם חָרַשׁ שְׁתֵּי פְּעָמִים קְנָסוּהוּ רַבָּנָן, דַּחֲרִישָׁה אַחַת הִתִּירוּ וְלֹא שְׁתֵּי חֲרִישׁוֹת: <B>אוֹ שֶׁנִּדַּיְּרָה.</B> שֶׁעָשׂוּ מִמֶּנָּה דִּיר לַבְּהֵמוֹת שֶׁעָשָׂה בָהּ סַהַר כִּדְאָמְרִינַן בְּפִרְקִין דִּלְעֵיל, וְלֹא עָשָׂה מִמֶּנּוּ אַשְׁפּוֹת אֶלָּא הִנִּיחוֹ כְּדֵי לְזַבֵּל בּוֹ הַשָּׂדֶה: <B>לֹא תִזָּרַע לְמוֹצָאֵי שְׁבִיעִית.</B> דְּקָנְסִינַן לֵיהּ בַּהֲנֵי יוֹתֵר מִבְּנִתְקַוְּצָה: <B>בְּטוֹבָה.</B> אָסוּר, דְּאֵין מַחֲזִיקִין טוֹבָה לַבְּעָלִים, דְּרַחֲמָנָא אַפְקְרִינְהוּ: <B>חִלּוּף הַדְּבָרִים.</B> דְּבֵית שַׁמַּאי לְקֻלָּא. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: 

~ פרק ד - משנה ג
<B>חוֹכְרִין נִירִין.</B> יָכוֹל יִשְׂרָאֵל לְקַבֵּל עָלָיו בְּכָךְ וְכָךְ כּוֹרִים לְשָׁנָה שְׂדֵה הַנָּכְרִי שֶׁחֲרָשָׁהּ בַּשְּׁבִיעִית לְזָרְעָהּ לְמוֹצָאֵי שְׁבִיעִית, וְאַף עַל פִּי שֶׁגּוֹרֵם לַנָּכְרִי לַחֲרֹשׁ בַּשְּׁבִיעִית. נִירִין לְשׁוֹן נִירוּ לָכֶם נִיר (יִרְמְיָה ד): <B>וּמַחֲזִיקִים יְדֵי נָכְרִים.</B> אִם מְצָאוֹ חוֹרֵשׁ יָכוֹל לוֹמַר לוֹ יִישַׁר כֹּחֲךָ וְכַיּוֹצֵא בָזֶה: <B>וְשׁוֹאֲלִים בִּשְׁלוֹמָן.</B> וַאֲפִלּוּ בְּיוֹם חַגָּם: 

~ פרק ד - משנה ד
<B>הַמֵּדֵל בַּזֵּיתִים.</B> כְּשֶׁהַזֵּיתִים מְקוֹרָבִים זֶה לָזֶה יוֹתֵר מִדַּאי נוֹטֵל אֶחָד וּמַנִּיחַ שְׁנַיִם אוֹ נוֹטֵל שְׁנַיִם וּמַנִּיחַ אֶחָד כְּדֵי שֶׁיִּגְדַּל וְיִתְעַבֶּה {ד}. מֵדֵל לְשׁוֹן שְׁלִיפָה וַהֲסָרָה, וְיֵשׁ מְפָרְשִׁים לְשׁוֹן דִּלְדּוּל וְדַלּוּת, שֶׁמְּדַלְדֵּל שָׂדֵהוּ מִן הַזֵּיתִים: <B>יָגוֹם.</B> גּוֹמֵם וּמְקַצֵּץ הָאִילָן וּמַשְׁאִיר הַשֹּׁרֶשׁ בָּאָרֶץ, וְלֹא יְשָׁרֵשׁ לְגַמְרֵי, שֶׁלֹּא יֵרָאֶה כְּעוֹבֵד אֶת הָאֲדָמָה: <B>וּמוֹדִים בְּמַחֲלִיק.</B> כְּגוֹן אִם נוֹטֵל שְׁלֹשָׁה זֵיתִים זֶה בְּצַד זֶה, שֶׁמַּחֲלִיק פְּנֵי הַקַּרְקַע יוֹתֵר מִדַּאי, יֹאמְרוּ לַעֲבוֹדַת קַרְקַע הוּא מִתְכַּוֵּן וְאָסוּר: <B>מִתּוֹךְ שֶׁל חֲבֵרוֹ אַף הַמַּחֲלִיק יְשָׁרֵשׁ.</B> דְּלֹא עָבֵיד אִינָשׁ שֶׁיְּתַקֵּן קַרְקַע חֲבֵרוֹ, וְיָדוּעַ דְּלָעֵצִים הוּא מְכַוֵּן:

<B>לֹא יְחַפֶּה.</B> הַבִּקּוּעַ בְּעָפָר, שֶׁהֲרֵי הוּא מְתַקֵּן בְּכָךְ, שֶׁהֶעָפָר נַעֲשֶׂה טִיט וּמַשְׁבִּיחַ הָאִילָן: <B>בַּאֲבָנִים וּבְקַשׁ.</B> דַּהֲוֵי כְּמוֹשִׁיב שׁוֹמֵר שֶׁלֹּא יִיבַשׁ: <B>בְּתוּלַת שִׁקְמָה.</B> שֶׁלֹּא נִקְצְצָה מֵעוֹלָם: <B>מִפְּנֵי שֶׁהִיא עֲבוֹדָה.</B> שֶׁהִיא מֻשְׁבַּחַת בְּכָךְ: <B>כְּדַרְכָּהּ אָסוּר.</B> שֶׁהִיא מִתְגַּדֶּלֶת וּמִתְעַבָּה בְּכָךְ וַהֲוֵי כְּעֵין זוֹמֵר לַגְּפָנִים: <B>אֶלָּא מַגְבִּיהַּ עֲשָׂרָה.</B> שֶׁאֵין דֶּרֶךְ לְעוֹלָם לְכָרְתָהּ אֶלָּא לְמַטָּה מֵעֲשָׂרָה: <B>אוֹ גוֹמֵם מֵעַל הָאָרֶץ.</B> שֶׁאֵין דֶּרֶךְ לִכְרֹת אוֹתָהּ כָּךְ. וַהֲלָכָה כְּרַבִּי יְהוּדָה: 

~ פרק ד - משנה ו
<B>הַמְזַנֵּב בִּגְפָנִים.</B> מְקַטֵּעַ זַנְבוֹת הַגְּפָנִים כְּדֵי שֶׁיְּעַבֶּה הַגֶּזַע וְיִגְדַּל וְיִגְבַּר כֹּחוֹ, וְכֵן הַקּוֹצֵץ קָנִים כְּדֵי שֶׁיִּתְעַבּוּ וְיוֹסִיף כֹּחָן: <B>יַרְחִיק טֶפַח.</B> מִן הַקַּרְקַע, וְשׁוּב לֹא מֵחֲזֵי כַּעֲבוֹדָה {ח}: <B>שֶׁנִּפְשַׁח.</B> נִסְדַּק, כְּמוֹ (אֵיכָה ג) וַיְפַשְּׁחֵנִי: <B>לֹא שֶׁיַּעֲלֶה.</B> שֶׁיִּתְחַבְּרוּ סְדָקָיו: <B>אֶלָּא שֶׁלֹּא יוֹסִיף.</B> לִסָּדֵק: 

~ פרק ד - משנה ז
<B>מֵאֵימָתַי אוֹכְלִים פֵּרוֹת הָאִילָן בַּשְּׁבִיעִית.</B> מִשּׁוּם דְּדָרְשִׁינַן לְאָכְלָה וְלֹא לְהֶפְסֵד, וְאִם אוֹכְלָן קֹדֶם בִּשּׁוּלָן הַיְנוּ הֶפְסֵד: <B>הַפַּגִּים.</B> פַּגֵּי תְאֵנָה כְּדִכְתִיב (שִׁיר הַשִּׁירִים ב) הַתְּאֵנָה חָנְטָה פַגֶּיהָ: <B>מִשֶּׁהִזְרִיחוּ.</B> הֶאֱדִימוּ, וְזֶהוּ הַתְחָלַת בִּשּׁוּלָן: <B>אוֹכֵל בָּהֶם פִּתּוֹ בַּשָּׂדֶה.</B> אֲבָל לְלַקֵּט וּלְהַכְנִיס לַבַּיִת אֵינוֹ רַשַּׁאי עַד שֶׁיִּגָּמְרוּ: <B>בִּחֲלוּ.</B> שֶׁגָּדַל הַרְבֵּה, וְדוֹמֶה לוֹ וְגַם נַפְשָׁם בָּחֲלָה בִי (זְכַרְיָה יא). וּבַיְרוּשַׁלְמִי יָלִיף לַהּ מִדִּכְתִיב (וַיִּקְרָא כה) תִּהְיֶה כָל תְּבוּאָתָהּ לֶאֱכֹל וּכְתִיב (שָׁם) מִן הַשָּׂדֶה תֹּאכְלוּ אֶת תְּבוּאָתָהּ, בִּשְׁתֵּי תְבוּאוֹת הַכָּתוּב מְדַבֵּר, אַחַת מִן הַבַּיִת וְאַחַת מִן הַשָּׂדֶה:

<B>לוּף שֶׁעָבְרָה עָלָיו שְׁבִיעִית.</B> וְנִכְנְסָה עָלָיו שְׁמִינִית, שֶׁהוּא מִתְקַיֵּם זְמַן מְרֻבֶּה בָּאָרֶץ בִּמְחֻבָּר: <B>אִם לָקְטוּ עֲנִיִּים אֶת עָלָיו לָקָטוּ.</B> דַּעֲלֵי הַלּוּף יֵשׁ לָהֶן בִּעוּר וְעִקָּרוֹ אֵין לוֹ בִּעוּר כְּדִתְנַן לְקַמָּן בְּפֶרֶק ז', הִלְכָּךְ אִם לָקְטוּ עֲנִיִּים אֶת עָלָיו בַּשְּׁבִיעִית לָקָטוּ, וְאִם לָאו וְנִכְנְסוּ לַשְּׁמִינִית וְגָדְלוּ הֶעָלִין, נִמְצְאוּ גִּדּוּלֵי שְׁבִיעִית וּשְׁמִינִית מְעֹרָבִין יַחַד: <B>יַעֲשֶׂה חֶשְׁבּוֹן.</B> וִישַׁעֵר כַּמָּה גָּדַל מֵהֶן בַּשְּׁבִיעִית, וְיִתֵּן לָעֲנִיִּים, דְּרַבִּי אֱלִיעֶזֶר סָבַר כְּרַבִּי יְהוּדָה דְּאָמַר לְקַמָּן עֲנִיִּים אוֹכְלִים אַחַר בִּעוּר וְלֹא עֲשִׁירִים: <B>רַבִּי יְהוֹשֻׁעַ אוֹמֵר אֵין לָעֲנִיִּים עִמּוֹ חֶשְׁבּוֹן.</B> דְּסָבַר כְּרַבִּי יוֹסֵי דְּאָמַר לְקַמָּן אֶחָד עֲנִיִּים וְאֶחָד עֲשִׁירִים אוֹכְלִים אַחַר הַבִּעוּר. וַהֲלָכָה כְּרַבִּי יְהוֹשֻׁעַ:

<B>סוּרְיָא.</B> אֲרַם נַהֲרַיִם וַאֲרַם צוֹבָה שֶׁכָּבַשׁ דָּוִד, וְאֵינָהּ כְּאֶרֶץ יִשְׂרָאֵל לְכָל דָּבָר, מִפְּנֵי שֶׁכְּבָשָׁהּ קֹדֶם שֶׁכָּבַשׁ אֶרֶץ יִשְׂרָאֵל כֻּלָּהּ, וּבִקְצַת דִּינִים עֲשָׂאוּהָ כְּאֶרֶץ יִשְׂרָאֵל וּבִקְצָתָן כְּחוּצָה לָאָרֶץ: <B>עוֹשִׂים בְּתָלוּשׁ.</B> אֲפִלּוּ אֵצֶל הַחֲשׁוּדִים, וְאִלּוּ בְאֶרֶץ יִשְׂרָאֵל תְּנַן לֹא תָבוֹר וְלֹא תִטְחַן עִמָּהּ: <B>אֲבָל לֹא בַמְחֻבָּר.</B> כְּגוֹן מְסִיקָה קְצִירָה וּבְצִירָה, וְדַוְקָא בַּשָּׁמוּר, אֲבָל בַּמֻּפְקָר אֲפִלּוּ בָּאָרֶץ מֻתָּר, דְּתַנְיָא בְּתוֹרַת כֹּהֲנִים מִן הַשָּׁמוּר אִי אַתָּה בּוֹצֵר אֲבָל אַתָּה בּוֹצֵר מִן הַמֻּפְקָר. וְטַעֲמָא דְגָזוּר בְּסוּרְיָא בִּמְחֻבָּר כְּדֵי שֶׁלֹּא יִהְיוּ מַנִּיחִים אֶת אֶרֶץ יִשְׂרָאֵל וְהוֹלְכִים וּמִשְׁתַּקְּעִים בְּסוּרְיָא, וּבְתָלוּשׁ הִתִּירוּ, כְּדֵי שֶׁיִּמְצְאוּ הָעֲנִיִּים שֶׁבְּאֶרֶץ יִשְׂרָאֵל קְצַת רֶוַח בְּסוּרְיָא שֶׁהִיא קְרוֹבָה לָהֶם, וְלֹא יִצְטָרְכוּ לָצֵאת מֵאֶרֶץ יִשְׂרָאֵל לְחוּצָה לָאָרֶץ: <B>כָּל שֶׁכַּיּוֹצֵא בוֹ מֻתָּר בְּאֶרֶץ יִשְׂרָאֵל.</B> מִדְּאוֹרַיְתָא, כְּגוֹן תָּלוּשׁ, אַף עַל גַּב דְּמִדְּרַבָּנָן אָסוּר בְּלֹא שִׁנּוּי, עוֹשִׂין אוֹתוֹ בְּסוּרְיָא כִּי אוֹרְחֵיהּ וְאֵינוֹ צָרִיךְ לְשַׁנּוֹת. אִי נַמִּי הָכִי קָאָמַר, כָּל שֶׁכַּיּוֹצֵא בוֹ מֻתָּר בָּאָרֶץ בַּמֻּפְקָר עוֹשִׂים אוֹתוֹ בְּסוּרְיָא אֲפִלּוּ בַּשָּׁמוּר, אֲבָל בִּמְחֻבָּר אָסוּר מִדְּרַבָּנָן בָּאָרֶץ אֲפִלּוּ בַּמֻּפְקָר:

<B>כְּלָל גָּדוֹל.</B> וּלְדָמָיו שְׁבִיעִית. אִם מְכָרָן וְקִבֵּל הַדָּמִים שֶׁלָּהֶם, יֵשׁ בִּדְמֵיהֶן קְדֻשַּׁת שְׁבִיעִית: <B>יֶשׁ לוֹ בִּעוּר.</B> כְּשֶׁכָּלָה לַחַיָּה חַיָּב לְבַעֵר: <B>עֲלֵה לוּף שׁוֹטֶה.</B> אִצְטָרִיךְ לְמִתְנֵי לוּף שׁוֹטֶה מִשּׁוּם דִּתְנַן בְּפֶרֶק בַּתְרָא דְּעֻקְּצִים עֲלֵה לוּף שׁוֹטֶה אֵין מִטַּמֵּא טֻמְאַת אֳכָלִים, סַלְקָא דַעְתָּךְ אֲמֵינָא הָכִי נַמִּי לֹא לֵחוֹל עֲלַיְהוּ קְדֻשַּׁת שְׁבִיעִית, קָמַשְׁמַע לָן: <B>הַדַּנְדַּנָּה.</B> בַּעֲרָבִי נענ"ע וּבְלַעַ"ז מינטו"א: <B>הָעֻלְשִׁין.</B> תְּרֵי גַוְנֵי עֻלְשִׁין הֵן, עֻלְשֵׁי גִנָּה וְעֻלְשֵׁי שָׂדֶה, וּבִשְׁאָר שָׁנִים דִּשְׁכִיחֵי עֻלְשֵׁי גִנָּה, עֻלְשֵׁי שָׂדֶה לֹא חֲשִׁיבֵי אֹכֶל וְאֵין מִטַּמֵּא טֻמְאַת אָכֳלִים, וְהָכָא קָא מַשְׁמַע לָן דְּבַשְּׁבִיעִית דְּאֵין עֻלְשֵׁי גִנָּה מְצוּיִין, עֻלְשֵׁי שָׂדֶה חֲשִׁיבֵי וּמִטַּמְּאִין טֻמְאַת אָכֳלִים וּשְׁבִיעִית נוֹהֶגֶת בָּהֶן: <B>וּכְרֵשִׁין.</B> נַמִּי תְּרֵי גַוְנֵי נִינְהוּ, כְּרֵשֵׁי גִנָּה וּכְרֵשֵׁי שָׂדֶה, וּכְרֵשֵׁי שָׂדֶה אִצְטָרִיךְ לְאַשְׁמוּעִינַן: <B>הָרְגִילָה.</B> הֵם חֲלוֹגְלוֹגוֹת וּבְלַעַ"ז בירדולגי"ש: <B>וְנֵץ הֶחָלָב.</B> פְּרָחִים לְבָנִים כְּחָלָב. פֵּרוּשׁ אַחֵר עֵשֶׂב הוּא כְּשֶׁנֶּחְתָּךְ יוֹצֵא מִמֶּנּוּ חָלָב: <B>הַחוֹחִים וְהַדַּרְדָּרִים.</B> מִינֵי קוֹצִים הֵן: <B>אִסְטִיס.</B> הוּא שֶׁצּוֹבְעִין בּוֹ כְּמִין תְּכֵלֶת, וְקוֹרִים לוֹ בַּעֲרָבִי ני"ל, וְקוֹצְרִים אוֹתוֹ וְהוּא צוֹמֵחַ וְדוֹמֶה לְסָפִיחַ, לְכָךְ קוֹרֵהוּ הַתַּנָּא סְפִיחֵי אִסְטִיס: <B>וְקוֹצָה.</B> מִין מִמִּינֵי הַצְּבָעִים, יֵשׁ אוֹמְרִים שֶׁהוּא אלעצפו"ר בַּעֲרָבִי: <B>יֶשׁ לָהֶן שְׁבִיעִית.</B> קְדֻשַּׁת שְׁבִיעִית נוֹהֶגֶת בָּהֶן שֶׁלֹּא לְהַפְסִידָם וְשֶׁלֹּא לַעֲשׂוֹת בָּהֶם סְחוֹרָה: <B>וְיֶשׁ לָהֶם בִּעוּר.</B> כְּשֶׁיַּגִּיעַ זְמַן הַבִּעוּר צָרִיךְ לְבַעֲרָם, וּלְקַמָּן בִּפְנִים מְפֹרָשִׁים דִּינֵי הַבִּעוּר: 

~ פרק ז - משנה ב
עיקר. שרש
<B>עִקָּר.</B> שֹׁרֶשׁ: <B>עַרְקַבְנִין.</B> רַבּוֹתַי מְפָרְשִׁים עֵשֶׂב הַגָּדֵל אֵצֶל הַדֶּקֶל וְעוֹלֶה וְנִכְרָךְ סְבִיבוֹת הַדֶּקֶל. וַאֲנִי שָׁמַעְתִּי שֶׁהוּא עֵשֶׂב שֶׁעָלִין שֶׁלּוֹ דּוֹמוֹת לְעַקְרָב: <B>וְהַחַלְבִצִין.</B> הַשָּׁרָשִׁים שֶׁל נֵץ חָלָב הָאָמוּר לְמַעְלָה: <B>הַבֻּכְרִיָּה.</B> קוֹרִין לוֹ בַּעֲרָבִי כנג"ר: <B>הַפּוּאָה.</B> בַּעֲרָבִי פו"א וּבְלַעַ"ז זרויי"א: <B>הָרִכְפָּה.</B> שָׁרָשִׁים שֶׁתַּחַת הַקַּרְקַע, וְשׁוֹלְקִין אוֹתָן וְשׁוֹתִין מֵימֵיהֶן, וּמִי שֶׁיֵּשׁ לוֹ תּוֹלָעִים בְּבִטְנוֹ יְמִיתֵם: <B>אֵין לָהֶם בִּעוּר.</B> לְפִי שֶׁמִּתְקַיְּמִים בָּאָרֶץ וְלֹא כָלֶה לַחַיָּה מִן הַשָּׂדֶה לְפִיכָךְ אֵין צָרִיךְ לְבַעֲרוֹ מִן הַבַּיִת: <B>דְּמֵיהֶם מִתְבַּעֲרִים עַד רֹאשׁ הַשָּׁנָה.</B> הָכִי קָאָמַר דְּמֵיהֶם צָרִיךְ שֶׁיִּתְבַּעֲרוּ מִן הָעוֹלָם קֹדֶם רֹאשׁ הַשָּׁנָה שֶׁל שְׁמִינִית. וְהוּא הַדִּין דַּהֲוֵי מָצֵי לְמִתְנֵי מִתְבַּעֲרִים אַחַר רֹאשׁ הַשָּׁנָה, שֶׁאִם יֶשְׁנָן לְאוֹתָן דָּמִים בָּעוֹלָם אַחַר רֹאשׁ הַשָּׁנָה שֶׁל שְׁמִינִית חַיָּב לְבַעֲרָם בִּקְדֻשַּׁת שְׁבִיעִית: 

~ פרק ז - משנה ג
<B>קַל וָחֹמֶר לִדְמֵיהֶן.</B> וְרַבִּי מֵאִיר סָבַר מַחְמִיר אֲנִי בִּדְמֵיהֶן יוֹתֵר מִבָּהֶן עַצְמָן, דִּפְרִי עַצְמוֹ מִנְכָּר וְלֹא אָתֵי לְמֵעֲבֵד בֵּיהּ סְחוֹרָה, אֲבָל דָּמִים לֹא מִנְכְּרֵי וְאָתֵי לְמֵעֲבֵד בֵּיהּ סְחוֹרָה. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר: <B>וְהַנֵּץ שֶׁלּוֹ.</B> פֶּרַח שֶׁעַל הַפִּטְמָא: <B>וּקְלִפֵּי אֱגוֹזִים.</B> כָּל הַנֵּי צוֹבְעִים בָּהֶן: <B>וְהַגַּלְעִינִים.</B> גַּרְעִינֵי פֵרוֹת, כְּגוֹן הַגַּרְעִינִים שֶׁבַּזֵּיתִים שֶׁמּוֹצִיאִים מֵהֶן שֶׁמֶן, וְגַרְעִינֵי תְמָרִים שֶׁרְאוּיִן לְמַאֲכַל בְּהֵמָה: <B>צוֹבֵעַ לְעַצְמוֹ.</B> דִּכְתִיב לָכֶם, לְכָל צָרְכֵיכֶם: <B>שֶׁאֵין עוֹשִׂים סְחוֹרָה בְּפֵרוֹת שְׁבִיעִית.</B> וּצְבִיעָה בְּשָׂכָר הַיְנוּ סְחוֹרָה: <B>לֹא בִבְכוֹרוֹת וְלֹא בִתְרוּמוֹת.</B> דִּלְמָא מַשְׁהֵי לְהוּ לְגַבֵּיהּ וְאָתֵי בְהוּ לִידֵי תַקָּלָה. וְדַוְקָא בִּבְכוֹר חַי גָּזְרוּ, אֲבָל שָׁחוּט שָׁרֵי לְמָכְרוֹ אֲפִלּוּ מִדְּרַבָּנָן, וּבִלְבָד שֶׁלֹּא יִמְכְּרֶנּוּ בְאִטְלִיז: <B>לֹא בִנְבֵלוֹת וְלֹא בִטְרֵפוֹת.</B> דְּמֵאַחַר דְּכָתַב רַחֲמָנָא טְמֵאִים הֵם, מַה צָּרִיךְ לוֹמַר תּוּ וּטְמֵאִים יִהְיוּ לָכֶם, אֶלָּא אֶחָד לְאִסּוּר אֲכִילָה וְאֶחָד לְאִסּוּר הֲנָאָה. וְאִם תֹּאמַר הֲרֵי חֲמוֹר הֲרֵי גָמָל וְסוּס וָפֶרֶד, שָׁאנֵי הַנֵּי דְּלִמְלַאכְתָּן הֵן גְּדֵלִים, וְלֹא אָסְרוּ {ו} לַעֲשׂוֹת סְחוֹרָה אֶלָּא בְדָבָר הָעוֹמֵד לַאֲכִילָה. וְכָל דָּבָר שֶׁאִסּוּרוֹ מִן הַתּוֹרָה אָסוּר לַעֲשׂוֹת בּוֹ סְחוֹרָה, אֲבָל דָּבָר שֶׁאִסּוּרוֹ מִדִּבְרֵיהֶם אֵין אָסוּר לַעֲשׂוֹת בּוֹ סְחוֹרָה אֶלָּא סְתַם יֵינָן, דְּאַף עַל גַּב דְּאִסּוּרוֹ מִדִּבְרֵיהֶם אָסוּר לַעֲשׂוֹת בּוֹ סְחוֹרָה: <B>וְלֹא יְהֵא לוֹקֵחַ.</B> כְּלוֹמַר לוֹקֵט וּמוֹכֵר בַּשּׁוּק, דְּנִרְאֶה כִּסְחוֹרָה. אֲבָל כְּשֶׁבְּנוֹ מוֹכֵר עַל יָדוֹ, פֵּרוּשׁ בִּשְׁבִילוֹ, אַף עַל פִּי שֶׁהוּא לְקָטָן לְמָכְרָן לֹא מֵחֲזֵי כִּסְחוֹרָה: 

~ פרק ז - משנה ד
<B>לָקַח לְעַצְמוֹ.</B> לָקַט לְעַצְמוֹ לֶאֱכֹל וְהוֹתִיר, מֻתָּר לְמָכְרָן אֲפִלּוּ הוּא עַצְמוֹ: <B>לָקַח בְּכוֹר.</B> בַּעַל מוּם, דְּמֻתָּר לְזָרִים: <B>מֻתָּר לְמָכְרוֹ.</B> וּבִלְבָד שֶׁלֹּא יִמְכְּרֶנּוּ אֶלָּא בַדָּמִים שֶׁקָּנָה אוֹתוֹ, שֶׁלֹּא יִשְׂתַּכֵּר בּוֹ: <B>מֻתָּרִין לְמָכְרָן.</B> דְּהָא כְּתִיב גַּבֵּי נְבֵלָה (דְּבָרִים יד) אוֹ מָכֹר לְנָכְרִי {ח}, הָא לֹא אָסְרָה תּוֹרָה אֶלָּא לְקַיְּמָן לַעֲשׂוֹת בָּהֶן סְחוֹרָה, אֲבָל בְּנִזְדַּמֵּן מֻתָּר: <B>רַבִּי יְהוּדָה אוֹמֵר אַף מִי שֶׁנִּתְמַנָּה.</B> אַף עַל פִּי שֶׁאֵינוֹ צַיָּד אֶלָּא שֶׁמָּצָא עוֹפוֹת טְמֵאִים, כְּמוֹ כִּי יִקָּרֵא קַן צִפּוֹר (דְּבָרִים כב). וּפָלֵיג אַתַּנָּא קַמָּא דְּאָמַר צַיָּדֵי דַּוְקָא: <B>נִתְמַנָּה.</B> נִזְדַּמֵּן, לְשׁוֹן וַיְמַן הַמֶּלֶךְ (דָּנִיֵּאל א):

<B>וַחֲכָמִים אוֹסְרִים.</B> לְמִי שֶׁאֵינוֹ צַיָּד, וַאֲפִלּוּ אֵין אֻמָּנוּתוֹ בְּכָךְ, וְטַעֲמַיְהוּ דְרַבָּנָן לְפִי שֶׁהַצַּיָּדִים שֶׁצְּרִיכִים לִתֵּן מַס לַמֶּלֶךְ מֵאֻמָּנוּתָם {י} הִתִּירוּ לָהֶם לִמְכֹּר חַיָּה עוֹפוֹת וְדָגִים טְמֵאִים כְּדֵי שֶׁיּוּכְלוּ לִפְרֹעַ לַמֶּלֶךְ הַמַּס שֶׁעֲלֵיהֶם, אֲבָל שְׁאָר כָּל אָדָם לֹא. וַהֲלָכָה כְּרַבִּי יְהוּדָה: <B>לוּלְבֵי זְרָדִין.</B> לוּלְבֵי אִילָן שֶׁשְּׁמוֹ זַרְדָּתָא: <B>לוּלְבֵי הָאֵלָה.</B> לְשׁוֹן כָּאֵלָה וְכָאַלּוֹן (יְשַׁעְיָה ו) בַּעֲרָבִי באלו"ט וּבְלַעַ"ז גלנד"י: <B>בָּטְנָה.</B> מִלְּשׁוֹן בָּטְנִים וּשְׁקֵדִים, בִּלְשׁוֹן עֲרָבִי פאסת"ק: <B>וַאֲטָדִים.</B> מִין מִמִּינֵי הַקּוֹצִים שֶׁצּוֹמְחִים בּוֹ גַּרְעִינִים שְׁחוֹרִים קָשִׁים וְהֵן נֶאֱכָלִים: <B>אֲבָל לֶעָלִים.</B> שֶׁל כָּל אֵלּוּ יֵשׁ בִּעוּר, דְּאַף עַל גַּב דְּלוּלָבִים אֵין נוֹשְׁרִים וְאֵין אֲנִי קוֹרֵא בָהֶם וְלִבְהֶמְתֶּךָ וְלַחַיָּה, כָּלָה לַחַיָּה מִן הַשָּׂדֶה כַּלֶּה לִבְהֶמְתְּךָ מִן הַבַּיִת, מִכָּל מָקוֹם הֶעָלִים נוֹשְׁרִים הֵן מִמְּקוֹם חִבּוּרָן, וְלָעֲנָפִים שֶׁהֵם מְחֻבָּרִים בּוֹ קָרֵי אֲבִיהֶן, וְכֵיוָן שֶׁהֵם נוֹשְׁרִים יֵשׁ לָהֶם בִּעוּר:

<B>וֶרֶד.</B> שׁוֹשָׁן: <B>וְהַכֹּפֶר.</B> הוּא שֶׁקּוֹרִים לוֹ בַּעֲרָבִי אלחנ"א. וְיֵשׁ אוֹמְרִים שֶׁהוּא בֹּשֶׂם שֶׁקּוֹרִין גרופול"י: <B>וְהַקְּטָף.</B> אִילַן אֲפַרְסְמוֹן {יב}: <B>לֹטֶם.</B> תַּרְגּוּם לוֹט וַלְטוֹס. יֵשׁ אוֹמְרִים שֶׁהוּא צינוב"ר בַּעֲרָבִי, פניולי"ש בְּלַעַ"ז: <B>יֵשׁ לָהֶם שְׁבִיעִית וְלִדְמֵיהֶן שְׁבִיעִית.</B> וְהוּא הַדִּין דְּיֵשׁ לָהֶן בִּעוּר {יג} וְלִדְמֵיהֶן בִּעוּר, וְאַף עַל פִּי שֶׁלֹּא נִזְכַּר בַּמִּשְׁנָה: <B>מִפְּנֵי שֶׁאֵינוֹ פֶרִי.</B> וְדִינָן כְּעֵצִים דְּאֵין לָהֶם קְדֻשַּׁת שְׁבִיעִית מִשּׁוּם דַּהֲנָאָתָן אַחַר בִּעוּרָן, וְהַתּוֹרָה אָמְרָה לְאָכְלָה, דּוּמְיָא דַאֲכִילָה, שֶׁהֲנָאָתָן וּבִעוּרָן שָׁוֶה, יָצְאוּ עֵצִים דַּהֲנָאָתָן אַחַר בִּעוּרָן {יד}. וְתַנָּא קַמָּא סֵבֶר דִּקְטָפוֹ זֶהוּ פִּרְיוֹ. וַהֲלָכָה כְּרַבִּי שִׁמְעוֹן: 

~ פרק ז - משנה ז
<B>וֶרֶד חָדָשׁ שֶׁכְּבָשׁוֹ בְּשֶׁמֶן יָשָׁן.</B> וֶרֶד שֶׁל שְׁבִיעִית שֶׁכְּבָשׁוֹ בְּשֶׁמֶן שֶׁל שִׁשִּׁית, יִלְקֹט אֶת הַוֶּרֶד קֹדֶם שֶׁיַּגִּיעַ זְמַן הַבִּעוּר וְהַכֹּל מֻתָּר. אֲבָל וֶרֶד שֶׁל שְׁבִיעִית שֶׁכְּבָשׁוֹ בְּשֶׁמֶן שֶׁל שְׁמִינִית וּכְבָר הִגִּיעַ זְמַן הַבִּעוּר שֶׁל הַוֶּרֶד, חַיָּב לְבַעֵר אֶת הַכֹּל, כִּדְקָתָנֵי סֵיפָא דִשְׁבִיעִית אוֹסֶרֶת מִין בְּשֶׁאֵינוֹ מִינוֹ בְּנוֹתֵן טַעַם {טו}: <B>וְחָרוּבִין חֲדָשִׁים וְכוּ'.</B> הָכָא מַיְרֵי בְּבָא לִשְׁתּוֹת הַיַּיִן אַחַר זְמַן הַבִּעוּר, דִּבְכָל עִנְיָן חַיָּב לְבַעֵר. וְאִם לֹא נִשְׁתַּהוּ הֶחָרוּבִים בַּיַּיִן עַד זְמַן הַבִּעוּר יִלְקֹט הֶחָרוּבִין וְהַיַּיִן מֻתָּר כְּמוֹ בְּוֶרֶד {טז}, וְהָכִי תַּנְיָא בְּהֶדְיָא בַּתּוֹסֶפְתָּא: <B>זֶה הַכְּלָל וְכוּ'.</B> בְּכָל אִסּוּרִים {יז} שֶׁבַּתּוֹרָה מַיְרֵי, דִּבְכֻלָּן בְּמִינוֹ בְּמַשֶּׁהוּ וּמִין בְּשֶׁאֵינוֹ מִינוֹ בְּנוֹתֵן טַעַם. וְאֵינָהּ הֲלָכָה: <B>הַשְּׁבִיעִית אוֹסֶרֶת בְּכָל שֶׁהוּא בְּמִינָהּ.</B> לְאַחַר הַבִּעוּר, אֲבָל קֹדֶם הַבִּעוּר אֵין בָּהֶם קְדֻשַּׁת שְׁבִיעִית שֶׁיְּהֵא אָסוּר לְהַפְסִידָהּ וְלַעֲשׂוֹת בָּהֶם סְחוֹרָה עַד שֶׁיְּהֵא בָהֶם בְּנוֹתֵן טַעַם, בֵּין בְּמִינָהּ בֵּין שֶׁלֹּא בְמִינָהּ:

~ פרק ח - משנה א
<B>וְשֶׁלֹּא בְמִינָהּ.</B> אֲפִלּוּ לְאַחַר הַבִּעוּר בְּנוֹתֵן טַעַם: <B>כְּלָל גָּדוֹל.</B> מְלוּגְמָא. תַּחְבֹּשֶׁת וּרְטִיָּה, וּפֵרוּשׁ מְלוּגְמָא מְלֹא לוּגְמָא, שֶׁדֶּרֶךְ אָדָם לִלְעֹס מְלֹא לוּגְמָיו חִטִּים אוֹ תְאֵנִים וְנוֹתֵן עַל גַּבֵּי מַכָּתוֹ: <B>אֵין עוֹשִׂים מִמֶּנּוּ מְלוּגְמָא.</B> דִּכְתִיב (וַיִּקְרָא כה) וְהָיְתָה שַׁבַּת הָאָרֶץ לָכֶם לְאָכְלָה, מַה שֶּׁהוּא מְיֻחָד לָכֶם דְּהַיְנוּ לְמַאֲכַל אָדָם יִהְיֶה לְאָכְלָה וְלֹא לִמְלוּגְמָא, אֲבָל מַה שֶּׁאֵינוֹ מְיֻחָד לָכֶם יִהְיֶה בֵּין לְאָכְלָה בֵּין לִמְלוּגְמָא, הִלְכָּךְ כָּל שֶׁאֵינוֹ מַאֲכַל אָדָם עוֹשִׂין מִמֶּנוּ מְלוּגְמָא לְאָדָם: <B>עוֹשִׂין מִמֶּנוּ מְלוּגְמָא לְאָדָם אֲבָל לֹא לִבְהֵמָה.</B> דִּכְתִיב (שָׁם) וְלִבְהֶמְתְּךָ וְלַחַיָּה אֲשֶׁר בְּאַרְצֶךָ תִּהְיֶה כָּל תְּבוּאָתָהּ לֶאֱכֹל, דָּבָר הָרָאוּי לַחַיָּה וְלַבְּהֵמָה תִּהְיֶה לֶאֱכֹל לַבְּהֵמָה וְלֹא לַעֲשׂוֹת לָהּ מְלוּגְמָא: <B>חֻמְרֵי אָדָם.</B> שֶׁאֵין עוֹשִׂין מֵהֶם מְלוּגְמָא: <B>וְחֻמְרֵי בְהֵמָה.</B> שֶׁאָסוּר לְשָׁלְקָן {ב}: <B>הַסִּיאָה.</B> עֵשֶׂב שֶׁקּוֹרִין לוֹ בַּעֲרָבִי צידר"ג: <B>וְאֵזוֹב.</B> צעת"ר בַּעֲרָבִי וּבְלַעַ"ז איספ"ו: <B>וְקוּרְנִית.</B> בַּעֲרָבִי חש"א וּבְלַעַ"ז שדריא"ע:

לְהַפְקִירָן, וְהַלּוֹקֵחַ מִן הַמְשֻׁמָּר אֲפִלּוּ כַּחֲצִי אִסָּר {א} אָסוּר. וְכָל הֲנָךְ דְּמַתְנִיתִין שֶׁחֶזְקָתָן שֶׁאֵין מְשַׁמְּרִין אוֹתָן מֻתָּר לִקַּח מֵהֶן מֵעַם הָאָרֶץ כְּדֵי מָזוֹן שָׁלֹשׁ סְעֻדּוֹת בִּלְבָד {ב} וְלֹא יוֹתֵר, שֶׁאֵין מוֹסְרִין דְּמֵי פֵּרוֹת שְׁבִיעִית לְעַם הָאָרֶץ, אַף עַל פִּי שֶׁהַפֵּרוֹת הֵן מִן הַהֶפְקֵר, יוֹתֵר מִמָּזוֹן שָׁלֹשׁ סְעֻדּוֹת, מִפְּנֵי שֶׁהוּא חָשׁוּד לֶאֱצֹר הַדָּמִים וְאֵינוֹ נוֹהֵג בָּהֶם קְדֻשַּׁת שְׁבִיעִית: <B>שֶׁלֹּא נֶחְשְׁדוּ עֲלֵיהֶם עוֹבְרֵי עֲבֵרָה.</B> לַהֲבִיאָם מִן הַשָּׁמוּר. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: <B>כָּל הַסְּפִיחִין מֻתָּרִין.</B> לָקַחַת מִכָּל אָדָם, דְּמַחְזְקִינַן לְהוּ שֶׁאֵינָם מִן הַשָּׁמוּר: <B>חוּץ מִסְּפִיחֵי כְרוּב.</B> שֶׁחֶזְקָתָן מִן הַשָּׁמוּר: <B>לְפִי שֶׁאֵין כַּיּוֹצֵא בָהֶם בְּיַרְקוֹת שָׂדֶה.</B> שֶׁאֵין גְּדֵלִים כְּמוֹתָן בַּמִּדְבָּר {ג}. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן:

<B>שָׁלֹשׁ אֲרָצוֹת.</B> בְּאֶרֶץ יִשְׂרָאֵל חֲלוּקוֹת זוֹ מִזּוֹ לְעִנְיַן בִּעוּר. וּבִעוּר הָאָמוּר בְּכָל מָקוֹם בַּשְּׁבִיעִית הוּא שֶׁכָּל זְמַן שֶׁאֵין נִמְצָא בַּשָּׂדֶה מֵאוֹתוֹ הַמִּין שֶׁאָסַף אֵלָיו בַּבַּיִת, חַיָּב לְבַעֵר אוֹתוֹ הַמִּין מִן הַבַּיִת מִיָּד. כְּגוֹן אִם לָקַט תְּאֵנִים וְעָשָׂה מֵהֶן גְּרוֹגָרוֹת, אֵינוֹ יָכוֹל לֶאֱכֹל מֵאוֹתָם גְּרוֹגָרוֹת אֲפִלּוּ בַּשָּׁנָה הַשְּׁבִיעִית אֶלָּא כָּל זְמַן שֶׁנִּמְצָאִים תְּאֵנִים מְחֻבָּרוֹת לָאִילָן, וּכְשֶׁכָּלוּ הַתְּאֵנִים מִן הָאִילָנוֹת חַיָּב לְבַעֵר {ד} הַגְּרוֹגָרוֹת שֶׁבְּבֵיתוֹ, אוֹ לְאָכְלָן כֻּלָּן מִיָּד, אוֹ לְהַפְקִירָן עַד שֶׁלֹּא יִשָּׁאֵר מֵהֶם בְּבֵיתוֹ, דִּכְתִיב (וַיִּקְרָא כה) וְלִבְהֶמְתְּךָ וְלַחַיָּה, כָּל זְמַן שֶׁחַיָּה אוֹכֶלֶת בַּשָּׂדֶה הַאֲכֵל לִבְהֶמְתְּךָ מִן הַבַּיִת, כָּלָה לַחַיָּה מִן הַשָּׂדֶה כַּלֵּה לִבְהֶמְתְּךָ מִן הַבַּיִת. וְהַשְׁתָּא קָאָמַר שֶׁאֶרֶץ יִשְׂרָאֵל כֻּלָּהּ נֶחְלֶקֶת לִשְׁלֹשָׁה חֲלָקִים לְעִנְיַן בִּעוּר, וְכָל חֵלֶק וְחֵלֶק מִמֶּנָּה נֶחְלָק לִשְׁלֹשָׁה חֲלָקִים: <B>כָּל שֶׁאֵינוֹ מְגַדֵּל שִׁקְמִים גָּלִיל הָעֶלְיוֹן.</B> שֶׁהַשִּׁקְמִים אֵין גְּדֵלִים אֶלָּא בַּשְּׁפֵלָה, וְסִימָן לַדָּבָר כַּשִּׁקְמִים אֲשֶׁר בַּשְּׁפֵלָה (מְלָכִים א י): <B>וּתְחוּם טְבֶרְיָה.</B> הוּא הָעֵמֶק. אֵלּוּ הַשְּׁלֹשָׁה הֵם מֵהַגָּלִיל: <B>הָהָר וְהַשְּׁפֵלָה וְהָעֵמֶק.</B> הֵם שְׁלֹשָׁה חֲלָקִים שֶׁל אֶרֶץ יְהוּדָה: <B>שְׁפֵלַת לוּד.</B> וְהַר [שֶׁלָּהּ] לוּד. וּמִבֵּית חוֹרוֹן עַד הַיָּם. הֵם שְׁלֹשָׁה חֲלוּקוֹת שֶׁל עֵבֶר הַיַּרְדֵּן, וּשְׁפֵלַת לוּד כִּשְׁפֵלַת הַדָּרוֹם, דְּכָל זְמַן שֶׁאוֹכֵל בִּשְׁפֵלַת [הַדָּרוֹם] אוֹכֵל בִּשְׁפֵלַת [לוּד], וְהָהָר שֶׁל שְׁפֵלַת לוּד אוֹכְלִים בּוֹ כָּל זְמַן שֶׁאוֹכְלִים בְּהַר הַמֶּלֶךְ:

<B>מִבֵּית חוֹרוֹן וְעַד הַיָּם מְדִינָה אֶחָת.</B> וְכָל זְמַן שֶׁלֹּא כָלָה לַחַיָּה מֵאַחַת מִן הַמְּקוֹמוֹת שֶׁבֵּינֵיהֶם, יְכוֹלִים כָּל אַנְשֵׁי הַגְּבוּל הַהוּא לֶאֱכֹל עַל יְדֵי הַמָּקוֹם הָאֶחָד שֶׁלֹּא כָלָה בּוֹ לַחַיָּה מִן הַשָּׂדֶה: <B>וְלָמָּה אָמְרוּ שָׁלֹשׁ אֲרָצוֹת.</B> הוֹאִיל וְשָׁלֹשׁ אֲרָצוֹת בְּכָל אַחַת הֲרֵי כָּאן תֵּשַׁע אֲרָצוֹת, לְמַאי הִלְכְתָא אָמְרוּ שָׁלֹשׁ אֲרָצוֹת לַבִּעוּר וְתוּ לֹא: <B>שֶׁיִּהְיוּ אוֹכְלִים.</B> אוֹכְלִים בִּיהוּדָה עַד שֶׁיִּכְלֶה הָאַחֲרוֹן שֶׁבִּשְׁלֹשׁ אַרְצוֹתֶיהָ, וְכֵן בַּגָּלִיל, וְכֵן בְּעֵבֶר הַיַּרְדֵּן. אֲבָל אֵין אוֹכְלִים בִּיהוּדָה עַל יְדֵי אוֹתָן שֶׁבַּגָּלִיל, וְלֹא בַגָּלִיל עַל יְדֵי אוֹתָן שֶׁבִּיהוּדָה, דִּגְמִירֵי דְאֵין חַיָּה שֶׁבִּיהוּדָה מִתְרַחֶקֶת לָצֵאת מִיהוּדָה לַגָּלִיל, וּמִגָּלִיל לִיהוּדָה, אֲבָל יוֹצְאָה הִיא מִגָּלִיל לַגָּלִיל וּמִיְּהוּדָה לִיהוּדָה: <B>וּשְׁאָר כָּל הָאֲרָצוֹת כְּהַר הַמֶּלֶךְ.</B> שְׁאָר כָּל הָאֲרָצוֹת שֶׁבְּאֶרֶץ יִשְׂרָאֵל אוֹכְלִים עַל יְדֵי הַפֵּרוֹת הַנִּמְצָאִים בְּהַר הַמֶּלֶךְ, וְהוּא הָיָה רַב הַפֵּרוֹת וְאַחֲרוֹן מִכָּל שְׁאָר אֲרָצוֹת שֶׁבְּאֶרֶץ יִשְׂרָאֵל:

<B>וְכָל הָאֲרָצוֹת כְּאַחַת לַזֵּיתִים וְלַתְּמָרִים.</B> כָּל הָאֲרָצוֹת שֶׁבְּאֶרֶץ יִשְׂרָאֵל אוֹכְלִים זֵיתִים וּתְמָרִים עַל יְדֵי אֶרֶץ אַחַת שֶׁלֹּא כָלוּ בָהּ הַזֵּיתִים וְהַתְּמָרִים לַחַיָּה מִן הַשָּׂדֶה, בֵּין שֶׁתִּהְיֶה הָאָרֶץ הַהִיא בִּיהוּדָה אוֹ בְעֵבֶר הַיַּרְדֵּן אוֹ בַגָּלִיל. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: <B>אוֹכְלִים עַל הַמֻּפְקָר.</B> אוֹכְלִים מִמַּה שֶּׁבַּבַּיִת עַל יְדֵי שֶׁמָּצוּי מֵאוֹתוֹ הַמִּין בַּשָּׂדֶה בְּמָקוֹם מֻפְקָר. אֲבָל אִם פָּסַק וְכָלָה אוֹתוֹ הַמִּין מִמָּקוֹם הַמֻּפְקָר, אַף עַל פִּי שֶׁנִּמְצָא מִמֶּנּוּ בְּמָקוֹם הַשָּׁמוּר בַּבָּתִּים אוֹ בַגִּנּוֹת חַיָּב לְבַעֵר, כִּדְאָמְרָן כָּלָה לַחַיָּה מִן הַשָּׂדֶה כַּלֵּה לִבְהֶמְתְּךָ מִן הַבַּיִת: <B>רַבִּי יוֹסֵי מַתִּיר אַף עַל הַשָּׁמוּר.</B> וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: <B>אוֹכְלִים עַל הַטְּפִיחִים.</B> עַל מַה שֶּׁמְּשִׂימִים הָעוֹפוֹת בַּטְּפִיחִים, שֶׁהֵן כְּלֵי חֶרֶס הַבְּנוּיִים בַּכֹּתֶל אֲשֶׁר שָׁם צִפֳּרִים יְקַנֵּנוּ. וְאוֹכְלִים מִמַּה שֶּׁבַּבַּיִת כָּל זְמַן שֶׁמָּצוּי מֵאוֹתוֹ הַמִּין בַּטְּפִיחִים: <B>וְעַל הַדּוּפְרָא.</B> אִילָן שֶׁעוֹשֶׂה פֵּרוֹת שְׁנֵי פְעָמִים בַּשָּׁנָה, אוֹכְלִים בָּרִאשׁוֹן עַד שֶׁיִּכְלֶה הָאַחֲרוֹן שֶׁבְּמִינָם: <B>אֲבָל לֹא עַל הַסִּתְוָנִיּוֹת.</B> אֵין אוֹכְלִין בָּעֲנָבִים עַל יְדֵי הָעֲנָבִים הַגְּדֵלִים בִּימוֹת הַסְּתָיו:

<B>פְּרוֹזְבּוּל.</B> פְּרוֹז תַּקָּנָה, בּוּל עֲשִׁירִים, דִּכְתִיב (וַיִּקְרָא כו) וְשָׁבַרְתִּי אֶת גְּאוֹן עֻזְּכֶם, וְאָמַר רַב יוֹסֵף אֵלּוּ בּוּלָאוֹת שֶׁבִּיהוּדָה. תַּקָּנָה לָעֲשִׁירִים שֶׁלֹּא יַעַבְרוּ עַל הִשָּׁמֶר לְךָ פֶּן יִהְיֶה דָבָר עִם לְבָבְךָ בְּלִיַּעַל, וְתַקָּנָה לָעֲנִיִּים שֶׁיִּמְצְאוּ מִי שֶׁיַּלְוֶה לָהֶם. וְהַשְׁמָטַת כְּסָפִים בַּזְּמַן הַזֶּה דְּרַבָּנָן, דִּכְתִיב (דְּבָרִים טו) וְזֶה דְּבַר הַשְּׁמִטָּה שָׁמוֹט כָּל בַּעַל מַשֵּׁה יָדוֹ, בִּשְׁתֵּי שְׁמִטּוֹת הַכָּתוּב מְדַבֵּר, אַחַת שְׁמִטַּת קַרְקָעוֹת וְאַחַת שְׁמִטַּת כְּסָפִים, בִּזְמַן שֶׁאַתָּה מְשַׁמֵּט קַרְקַע {ד} אַתָּה מְשַׁמֵּט כְּסָפִים, וּבִזְמַן שֶׁאִי אַתָּה מְשַׁמֵּט קַרְקַע אִי אַתָּה מְשַׁמֵּט כְּסָפִים, וְרַבָּנָן הוּא דִּתְקִינוּ שְׁמִטַּת כְּסָפִים בַּזְּמַן הַזֶּה, וּמִשּׁוּם הָכִי הָיָה כֹּחַ בְּיַד הִלֵּל לְתַקֵּן פְּרוֹזְבּוּל לְהַפְקִיעַ דִּין הַשְּׁמִטָּה, הוֹאִיל וְאֵין שְׁמִטַּת כְּסָפִים בַּזְּמַן הַזֶּה אֶלָּא מִדִּבְרֵי סוֹפְרִים: <B>זֶהוּ גוּפוֹ שֶׁל פְּרוֹזְבּוּל.</B> עִקָּרוֹ וְשָׁרְשׁוֹ:

<B>וְהַדַּיָּנִים חוֹתְמִים לְמַטָּה אוֹ הָעֵדִים.</B> לְהָכִי קָתָנֵי הַדַּיָּנִים אוֹ הָעֵדִים, לְלַמֶּדְךָ שֶּׁיָּכוֹל הָעֵד לִהְיוֹת דַּיָּן {ז} אוֹ הַדַּיָּן עֵד, דְּבִדְרַבָּנָן עֵד נַעֲשֶׂה דַיָּן, וּפְרוֹזְבּוּל דְּרַבָּנָן הוּא: <B>פְּרוֹזְבּוּל הַמֻּקְדָּם.</B> שֶׁזְּמַנּוֹ מֻקְדָּם, שֶׁנִּכְתַּב בְּכִסְלֵו וּזְמַנּוֹ הַכָּתוּב בְּתוֹכוֹ מִתִּשְׁרֵי קוֹדֵם, כָּשֵׁר, מִפְּנֵי שֶׁהוּרַע כֹּחוֹ שֶׁל מַלְוֶה בְּכָךְ, דִּפְרוֹזְבּוּל אֵינוֹ מוֹעִיל אֶלָּא לַהַלְוָאוֹת שֶׁלְּפָנָיו, דְּמִשֶּׁנִּמְסְרוּ לְבֵית דִּין הֲרֵי הֵן כִּגְבוּיוֹת וְשׁוּב אֵין מַשְׁמִיטוֹת, הִלְכָּךְ הַלְוָאוֹת הַבָּאוֹת אַחֲרָיו שֶׁלֹּא נִמְסְרוּ לְבֵית דִּין מַשְׁמִיטוֹת, נִמְצָא דְמַפְסִיד בְּמַה שֶּׁהִקְדִּים זְמַנּוֹ. אֲבָל פְּרוֹזְבּוּל הַמְאֻחָר פָּסוּל, שֶׁאִחֵר זְמַנּוֹ וְהוּא הִלְוָה בֵּינְתַּיִם, וּכְשֶׁמּוֹצִיא פְּרוֹזְבּוּל זֶה וְרוֹאִים בֵּית דִּין הַזְּמַן סְבוּרִים שֶׁאוֹתָן הַלְוָאוֹת קָדְמוּ לַפְּרוֹזְבּוּל, וְנִמְצָא גּוֹבֶה שֶׁלֹּא כַדִּין: <B>שִׁטְרֵי חוֹב הַמֻּקְדָּמִים פְּסוּלִים.</B> כְּגוֹן דְּקָאֵי בְּנִיסָן וְכָתַב בַּשְּׁטָר זְמַן תִּשְׁרֵי שֶׁלְּפָנָיו, פְּסוּלִים מִשּׁוּם דְּגוֹבֶה שֶׁלֹּא כַדִּין מִן הַלָּקוֹחוֹת שֶׁבֵּין תִּשְׁרֵי לְנִיסָן. וְקָנְסִינְהוּ רַבָּנָן דְּלֹא לִיגְבּוּ בִּשְׁטָר זֶה מִן הַלָּקוֹחוֹת כְּלָל, וַאֲפִלּוּ מִנִּיסָן וּלְהַלָּן, גְּזֵרָה שֶׁמָּא יִגְבֶּה מֵאוֹתָן שֶׁבֵּין תִּשְׁרֵי לְנִיסָן שֶׁלֹּא כַדִּין. וּמִיהוּ לֹא הֲוֵי הַשְּׁטַר פָּסוּל מֵחֲמַת שֶׁהָעֵדִים שֶׁחָתְמוּ בוֹ הֲווֹ רְשָׁעִים שֶׁחָתְמוּ עַל שְׁטָר מֻקְדָּם בְּשֶׁקֶר, דְּמָצוּ הָעֵדִים לְמֵימַר אֲנַחְנוּ עַל עִקַּר הַמִּלְוָה חָתַמְנוּ וְלֹא עַל הַזְּמַן, שֶׁלֹּא בָאנוּ לְהָעִיד שֶׁנִּכְתַּב וְנֶחְתַּם בְּיוֹמוֹ {ט}: <B>וְהַמְאֻחָרִים כְּשֵׁרִים.</B> דְהוּרַע כֹּחוֹ שֶׁל מַלְוֶה, שֶׁאֵינוֹ טוֹרֵף לָקוֹחוֹת אֶלָּא מִזְּמַן שֶׁכָּתוּב בַּשְּׁטָר:

<B>כּוֹתְבִין פְּרוֹזְבּוּל לְכָל אֶחָד וְאֶחָד.</B> שֶׁכָּל מַלְוֶה וּמַלְוֶה צָרִיךְ שֶׁיִּמְסֹר שְׁטָרוֹתָיו לְבֵית דִּין וְיִכְתֹּב לוֹ פְּרוֹזְבּוּל, הִלְכָּךְ כְּשֶׁחֲמִשָּׁה לֹוִין מֵאֶחָד אֵינוֹ כּוֹתֵב אֶלָּא פְּרוֹזְבּוּל אֶחָד, שֶׁאֵין כָּאן אֶלָּא מַלְוֶה אֶחָד, וְאַף עַל פִּי שֶׁהַלֹּוִין מְרֻבִּין, כְּשֶׁיִּהְיֶה לַמַּלְוֶה פְּרוֹזְבּוּל אֶחָד עַל כֻּלָּן דַּיּוֹ: <B>אֶלָּא עַל הַקַּרְקַע.</B> אֶלָּא אִם כֵּן יֵשׁ לוֹ קַרְקַע לַלֹּוֶה, שֶׁאָז חָשׁוּב הַחוֹב כְּגָבוּי בְּיַד בֵּית דִּין וְלֹא קָרִינַן בֵּיהּ לֹא יִגֹּשׂ {י}: <B>אִם אֵין לוֹ.</B> לַלֹּוֶה, וּמַלְוֶה יֵשׁ לוֹ, מְזַכֶּה לוֹ מַלְוֶה לַלֹּוֶה בְּתוֹךְ שָׂדֵהוּ קַרְקַע כָּל דְּהוּ, וְכוֹתְבִים עָלָיו פְּרוֹזְבּוּל, וַאֲפִלּוּ הַחוֹב מֵאָה מָנֶה, מִשּׁוּם דְּקַרְקַע כָּל שֶׁהוּא רָאוּי לִגְבּוֹת בָּהּ כָּל חוֹבוֹ, כְּגוֹן דְּגָבֵי וְהָדָר גָּבֵי עַד דְּמַשְׁלִים כָּל חוֹבוֹ: <B>הָיְתָה לוֹ.</B> לַלֹּוֶה שָׂדֶה מְמֻשְׁכֶּנֶת בָּעִיר, בֵּין שֶׁאֲחֵרִים מִשְׁכְּנוּ אוֹתָהּ לוֹ בֵּין שֶׁמִּשְׁכֵּן הוּא לַאֲחֵרִים, כּוֹתְבִין פְּרוֹזְבּוּל: <B>כּוֹתְבִין לָאִישׁ עַל נִכְסֵי אִשְׁתּוֹ.</B> כְּגוֹן קַרְקַע שֶׁל נִכְסֵי מְלוֹג שֶׁהוּא אוֹכֵל פֵּרוֹת {יא}:

<B>וְהָרוֹדֶה.</B> מֵחַלּוֹת דְּבַשׁ שֶׁבְּתוֹכָהּ בְּשַׁבָּת חַיָּב חַטָּאת כְּתוֹלֵשׁ דָּבָר מִמְּקוֹם חִבּוּרוֹ, דִּכְתִיב (שְׁמוּאֵל א יד) וַיִּטְבֹּל אוֹתָהּ בְּיַעְרַת הַדְּבָשׁ, וְכִי מַה עִנְיַן יַעַר אֵצֶל דְּבַשׁ, אֶלָּא מַה יַּעַר הַתּוֹלֵשׁ מִמֶּנּוּ בְּשַׁבָּת חַיָּב חַטָּאת, אַף דְּבַשׁ הָרוֹדֶה מִמֶּנּוּ בְּשַׁבָּת חַיָּב חַטָּאת. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: <B>הַמַּחֲזִיר חוֹב בַּשְּׁבִיעִית.</B> בְּסוֹף שְׁבִיעִית, דְּהַיְנוּ בִּשְׁמִיטָתָהּ, שֶׁאֵין שְׁבִיעִית מְשַׁמֶּטֶת אֶלָּא בְּסוֹפָהּ, שֶׁנֶּאֱמַר (דְּבָרִים טו) מִקֵּץ שֶׁבַע שָׁנִים {טז}: <B>אָמַר לוֹ אַף עַל פִּי כֵן.</B> אָמַר לוֹ הַלֹּוֶה אַף עַל פִּי כֵן אֲנִי רוֹצֶה לִפְרֹעַ לְךָ {יז}: <B>יְקַבֵּל מִמֶּנּוּ.</B> וְלֹא עוֹד אֶלָּא שֶׁמֻּתָּר לוֹ שֶׁתְּהֵא יָדוֹ פְּשׁוּטָה לְקַבֵּל בְּשָׁעָה שֶׁאוֹמֵר לוֹ מְשַׁמֵּט אֲנִי:

<B>שֶׁנֶּאֱמַר וְזֶה דְבַר הַשְּׁמִטָּה.</B> כְּלוֹמַר דִּבּוּר בְּעַלְמָא שֶׁמּוֹצִיא מִפִּיו שֶׁהוּא מְשַׁמֵּט דַּיּוֹ, וּבְדִבּוּר רִאשׁוֹן שֶׁאָמַר מְשַׁמֵּט אֲנִי סַגִּי, וְאֵין צָרִיךְ שֶׁיֹּאמַר וְיִשְׁנֶה, שֶׁנֶּאֱמַר וְזֶה דְבַר, אֵין לְךָ אֶלָּא דִבּוּר רִאשׁוֹן, וְכֵן וְזֶה דְבַר הָרוֹצֵחַ: <B>הַמַּחֲזִיר חוֹב בַּשְּׁבִיעִית.</B> לֹוֶה שֶׁאוֹמֵר לַמַּלְוֶה אַף עַל פִּי כֵן, וּמַחֲזִיר לוֹ מָעוֹתָיו, רוּחַ חֲכָמִים נוֹחָה הֵימֶנּוּ: <B>לֹא יַחֲזִיר לְבָנָיו.</B> אֵינוֹ חַיָּב לְהַחֲזִיר לְבָנָיו אִם מֵת הַגֵּר: <B>וְאִם הֶחֱזִיר רוּחַ חֲכָמִים נוֹחָה הֵימֶנּוּ.</B> וְדַוְקָא לְבָנָיו שֶׁנִּתְגַּיְּרוּ עִמּוֹ, מִשּׁוּם דְּעוֹבֵד כּוֹכָבִים יוֹרֵשׁ אֶת אָבִיו מִן הַתּוֹרָה דִּכְתִיב (דְּבָרִים ב) כִּי יְרֻשָּׁה לְעֵשָׂו נָתַתִּי אֶת הַר שֵׂעִיר, וְאִם לֹא יַחֲזִיר לְבָנָיו יֵשׁ לָחוּשׁ שֶׁמָּא יַחְזְרוּ לְסוּרָם, שֶׁיֹּאמְרוּ אִם הָיוּ עוֹבְדֵי כוֹכָבִים הָיוּ יוֹרְשִׁים אֲבִיהֶם וְהָיוּ צְרִיכִין לְהַחֲזִיר לָהֶם. אֲבָל אִם הָיְתָה הוֹרָתָם שֶׁלֹּא בִקְדֻשָּׁה וְלֵדָתָם בִּקְדֻשָּׁה, דְּמֵעוֹלָם לֹא יָרְשׁוּ אֶת אֲבִיהֶם מִן הַתּוֹרָה וְלֹא הָיְתָה יְרֻשָּׁה זוֹ רְאוּיָה לָהֶם לְעוֹלָם, אֵין לָחוּשׁ בָּזֶה שֶׁמָּא יַחְזְרוּ לְסוּרָם, וְהַמַּחֲזִיר לָהֶם אֵין רוּחַ חֲכָמִים נוֹחָה הֵימֶנּוּ:
