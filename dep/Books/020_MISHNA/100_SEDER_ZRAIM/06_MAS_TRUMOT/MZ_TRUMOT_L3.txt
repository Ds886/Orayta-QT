~ פרק א - משנה א

{א} וּמִיהוּ תֵּימַהּ לְפֵרוּשֵׁי מַתְנִיתִין בִּכְהַאי גַוְנָא, דְּמִמָּה נַפְשָׁךְ אִי קָאֵי מִנְיָנָא עַל לֹא יִתְרֹמוּ, אִם כֵּן נֵימָא דְּדַוְקָא חֲמִשָּׁה לֹא יִתְרֹמוּ, הָא אֲחֵרִים אֲפִלּוּ לְכַתְּחִלָּה תּוֹרְמִים, וְאִי קָאֵי עַל וְאִם תָּרְמוּ אֵין תְּרוּמָתָן תְּרוּמָה, מַשְׁמַע נַמִּי הָא אַחֲרִינֵי אִם תָּרְמוּ בְּדִיעֲבַד מִיהָא תְּרוּמָתָן תְּרוּמָה. תּוֹסְפוֹת יוֹם טוֹב: {ב} יְרוּשַׁלְמִי, וְהַךְ קְרָא בִּתְרוּמַת הַמִּשְׁכָּן כְּתִיב, וְיָלְפִינַן מִנֵּיהּ תְּרוּמַת כֹּהֲנִים, מִדְּאִקְרֵי נַמִּי תְּרוּמָה כְּדִכְתִיב וַאֲנִי נָתַתִּי לָכֶם אֶת מִשְׁמֶרֶת תְּרוּמֹתָי. הָרַמְבַּ"ם: {ג} וְכֵן הַיּוֹצֵא יְחִידִי בַּלַּיְלָה וְהַלָּן בְּבֵית הַקְּבָרוֹת וְהַמְקָרֵעַ אֶת כְּסוּתוֹ, הָעוֹשֶׂה אַחַת מֵאֵלּוּ הוּא שׁוֹטֶה. יְרוּשַׁלְמִי: {ד} <B> אֵין תְּרוּמָתוֹ תְרוּמָה.</B> מִשּׁוּם דְּהַנָּכְרִי אֵינוֹ בִּכְלָל עִם הַחַיָּבִים לִתְרֹם, לְפִיכָךְ הוֹצִיאוֹ מִן הַכְּלָל וּשְׁנָאוֹ בְּבָבָא בִּפְנֵי עַצְמָהּ:

{ה} וְאַתְיָא כְּתַנָּא קַמָּא דְּמִשְׁנָה ג' פֶּרֶק ב' דִּבְרָכוֹת דְּפָסַק הֲלָכָה כְּמוֹתוֹ, דְּהָתָם נַמִּי לְכַתְּחִלָּה מוֹדֶה דְּצָרִיךְ לְהַשְׁמִיעַ לְאָזְנָיו: 

~ פרק א - משנה ג

{ו} <b>רַבִּי יְהוּדָה אוֹמֵר כוּ'.</b> בְּפֶרֶק ב' דִּמְגִלָּה מִשְׁנָה ד' כָּתַבְתִּי בְּשֵׁם הַתּוֹסָפוֹת בְּשֶׁהִגִּיעַ לְחִנּוּךְ, וְהָכָא נַמִּי דִּכְוָתֵיהּ, וּמִשּׁוּם הָכִי תְּנַן שֶׁלֹּא הֵבִיא שְׁתֵּי שְׂעָרוֹת, דְּמַשְׁמַע דִּבְסָמוּךְ לָהֶן מַיְרֵי. אֲבָל הָרַ"שׁ מוֹכִיחַ דְּהָכָא בְּכָל קָטָן קָאָמַר, וּמִשּׁוּם רַבִּי יוֹסֵי נָקַט. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ה

{ז} וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם.</B> וּדְבַר תֵּימַהּ הוּא דְּמַאי קָא מַשְׁמַע לָן, שֶׁאֵין פּוֹטֵר עַצְמוֹ מֵחוֹבוֹ בְּשֶׁל אֲחֵרִים, וְנִרְאֶה כְּפֵרוּשׁ הַיְרוּשַׁלְמִי דִּבְעָנִי עַצְמוֹ שֶׁיֵּשׁ לוֹ פֵּרוֹת שֶׁל טֶבֶל מַיְרֵי, וְאֵין תּוֹרְמִין מֵהֶן מִשּׁוּם דַּהֲוֵי מִפָּטוּר עַל הַחִיּוּב: {ח} וְנוּסְחָא אַחֲרִינָא שֶׁל הָרַמְבַּ"ם, שֶׁלֹּא נִטְּלָה תְּרוּמָתוֹ.</B> וּמְפָרְשָׁהּ שֶׁלֹּא יַעֲשֶׂה אוֹתוֹ מַעֲשֵׂר רִאשׁוֹן תְּרוּמָה גְדוֹלָה אֶלָּא אִם כֵּן חוֹשֵׁב כַּמָּה יֵשׁ בּוֹ מִן הַתְּרוּמַת מַעֲשֵׂר, וּמַה שֶּׁיִּשָּׁאֵר הוּא שֶׁרַשַּׁאי לַעֲשׂוֹת תְּרוּמָה גְדוֹלָה עַל כִּרְיוֹ. וְהוּא תֵימַהּ דְּמַאי קָא מַשְׁמַע לָן, פְּשִׁיטָא שֶׁאִי אֶפְשָׁר לְבֶן לֵוִי לְשַׁלֵּם חוֹב תְּרוּמָה גְדוֹלָה שֶׁעָלָיו מִתְּרוּמַת מַעֲשֵׂר שֶׁכְּבָר נִתְחַיֵּב לְתִתָּהּ לַכֹּהֵן. וְאֶפְשָׁר דְּקָא מַשְׁמַע לָן דְּכֵיוָן שֶׁבְּשָׁעָה שֶׁתָּרַם לֹא חִשֵּׁב בְּדַעְתּוֹ שֶׁיַּפְרִישׁ תְּחִלָּה הַתְּרוּמַת מַעֲשֵׂר שֶׁבּוֹ, לְפִיכָךְ אֵין תְּרוּמָתוֹ תְרוּמָה כְּלָל, לְפִי שֶׁנִּמְצָא תּוֹרֵם מִן הַתְּרוּמַת מַעֲשֵׂר שֶׁבּוֹ עַל הַטֶּבֶל, וּבְמֵזִיד מַיְרֵי כְּמוֹ שֶׁאֶכְתֹּב בְּפֶרֶק ב' מִשְׁנָה ב'. תּוֹסְפוֹת יוֹם טוֹב: {ט} וְתֵימַהּ לִי מִנַּיִן, דְּהָא בְּמַעֲשֵׂר רִאשׁוֹן קְרָאֵי דָּרְשִׁינַן בְּפֶרֶק ז' דִּבְרָכוֹת, וְאוּלַי מִסְּבָרָא מִפְּנֵי שֶׁשֵּׁם מַעֲשֵׂר אֶחָד הוּא, וַאֲפִלּוּ תְּרוּמָה יָלְפִינַן מִמַּעֲשֵׂר כְּמוֹ שֶׁאֶכְתֹּב לְקַמָּן. תּוֹסְפוֹת יוֹם טוֹב: {י} וּמִנַּיִן יוֹדֵעַ שֶׁהֵבִיאָה שְׁלִישׁ, כָּל שֶׁזּוֹרְעָהּ וּמַצְמַחַת בְּיָדוּעַ שֶׁהֵבִיאָה שְׁלִישׁ. בְּשֵׁם הַיְרוּשַׁלְמִי, וְעַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב: {יא} וּלְפִי זֶה מִן הַמְחֻבָּר עַל הַמְחֻבָּר נַמִּי לֹא, וְקַשְׁיָא אַמַּאי לֹא תָנֵי לַהּ.</B> תּוֹסְפוֹת יוֹם טוֹב: {יב} לְפִי שֶׁעִנְיָן אֶחָד כּוֹלֵל אוֹתָם, וְהוּא שֶׁיּוֹצִיא מִזֶּה מַה שֶּׁהוּא חַיָּב בּוֹ מִן הַחֹק. הָרַמְבַּ"ם: {יג} <B> מִפֵּרוֹת חוּצָה לָאָרֶץ.</B> שֶׁנִּתְחַיְּבוּ בְּתַקָּנַת נְבִיאִים אוֹ תַקָּנַת חֲכָמִים, בְּמִשְׁנָה ג' סוֹף מַסֶּכֶת יָדַיִם: {יד} מִשּׁוּם דַּהֲוֵי אֲמֵינָא דִּלְעֵיל מִשְׁנָה א' לֹא אַמְעִיט אֶלָּא דְרַבִּי יוֹסֵי, אִיצְטְרִיךְ תּוּ הָכָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ו

{טו} <B> הָאִלֵּם.</B> וְאִם תֹּאמַר אַמַּאי לֹא כָלֵיל בַּהֲדֵי הֲנָךְ חֲמִשָּׁה חֵרֵשׁ הַמְדַבֵּר וְאֵינוֹ שׁוֹמֵעַ, וְלִתְנֵי שִׁשָּׁה, וְאַמַּאי תָּנֵי לַהּ בְּמִשְׁנָה ב' בְּאַפֵּי נַפְשֵׁיהּ. וְיֵשׁ לוֹמַר דְּהוֹאִיל וּלְעֵיל בָּעֵי לְפֵרוּשֵׁי פְּלֻגְתָּא דְרַבִּי יְהוּדָה וְרַבִּי יוֹסֵי בְּקָטָן, תָּנֵי נַמִּי פֵּרוּשָׁא דְּחֵרֵשׁ שֶׁנִּשְׁנָה קֹדֶם לְקָטָן. תּוֹסְפוֹת יוֹם טוֹב: {טז} <B> וְאִם תָּרְמוּ תְּרוּמָתָן תְּרוּמָה.</B> לְפִי שֶׁאֵין הַבְּרָכוֹת מְעַכְּבוֹת, וְכֵן אִם תָּרַם בְּדִיעֲבַד מִן הָרַע עַל הַיָּפֶה. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ז

{יז} <B> לֹא בְמִדָּה כוּ'.</B> וְאַף עַל גַּב דַּחֲכָמִים נָתְנוּ בוֹ שִׁעוּר כְּדִתְנַן בְּמִשְׁנָה ג' פֶּרֶק ד', מִכָּל מָקוֹם יַעֲשֶׂה כְּדִין תּוֹרָה וְלֹא יִתְרֹם בְּמִדָּה כוּ' אֶלָּא אוֹמְדוֹ וּמַפְרִישׁ בְּדַעְתּוֹ כַּשִּׁעוּר דִּלְקַמָּן. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ח

{יח} <B> הַנִּכְתָּשִׁין וְכוּ'.</B> שֶׁזֶּה דוֹמֶה לְתוֹרֵם מִדָּבָר שֶׁנִּגְמְרָה מְלַאכְתּוֹ עַל דָּבָר שֶׁלֹּא נִגְמְרָה מְלַאכְתּוֹ: {יט} <B> וְיַחֲזֹר וְיִתְרֹם.</B> לֹא אִתְפָּרֵשׁ אִם יִתְּנֵם שְׁתֵּיהֶן לַכֹּהֵן בְּלֹא דָמִים וּמִשּׁוּם קְנָסָא, אוֹ שֶׁיִּתֵּן לַכֹּהֵן הַשְּׁנִיָּה בְּדָמִים כְּמוֹ בְרֵישׁ פֶּרֶק ג': {כ} חוּץ מִדְּמֵי תְרוּמָה שֶׁבָּהּ, כְּדִין מְדֻמָּע דְּרֵישׁ פֶּרֶק ה'.</B> וְעַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב: {כא} רוֹצֶה לוֹמַר תְּרוּמַת מַעֲשֵׂר, דְּמִדְּאוֹרַיְתָא כְּבָר נִתְקְנָה מִתְּרוּמָה גְדוֹלָה, שֶׁהָרִאשׁוֹנָה הִיא תְּרוּמָה מִדְּאוֹרַיְתָא: 

~ פרק א - משנה י

{כב} וְהָכִי נַמִּי הַתּוֹרֵם שֶׁמֶן עַל הַזֵּיתִים, שֶׁאַף עַל פִּי שֶׁתְּרוּמָתוֹ תְרוּמָה, מִכָּל מָקוֹם צָרִיךְ שֶׁיַּחֲזֹר וְיִתְרֹם כְּדִלְעֵיל מִשְׁנָה ח', וְחַד מִתְּרֵי נָקַט:

~ פרק ב - משנה א

{יא} <B> מֻתָּרִים.</B> וּתְמִיהָה לִי דִּבְסוֹף פֶּרֶק ז' דִּשְׁבִיעִית תְּנַן וֶרֶד חָדָשׁ שֶׁכְּבָשׁוֹ בְשֶׁמֶן יָשָׁן כוּ' שְׁמַע מִנָּהּ דְּכִבּוּשׁ אוֹסֵר. וְעַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב שֶׁהֶאֱרִיךְ: 

~ פרק י - משנה יא

{יב} <B> כָּל הַנִּשְׁלָקִים.</B> שְׁלִיקָה בִּשּׁוּל יוֹתֵר מִדַּאי: 

^ פרק יא

~ פרק יא - משנה א

{א} <B> מַשְׁבִּיחוֹ.</B> וְלֹא קַשְׁיָא אַדְּסוֹף פֶּרֶק ב' דְּאֵין תּוֹרְמִין מִן הַמְבֻשָּׁל עַל שֶׁאֵינוֹ מְבֻשָּׁל מִשּׁוּם דַּהֲוָה מִן הָרַע עַל הַיָּפֶה. דִּמְשָׁנֵי בַּיְרוּשַׁלְמִי תָּמָן תָּנִינַן בִּבְעָלִים וְהָכָא בְּכֹהֵן, דִּלְכֹהֵן הִתִּירוּ לְבַשֵּׁל, שֶׁמַּשְׁבִּיחוֹ שֶׁמִּתְקַיֵּם לוֹ, אֲבָל בְּעָלִים צְרִיכִין לִתְרֹם מִן הַיָּפֶה, וְשֶׁאֵינוֹ מְבֻשָּׁל יָפֶה לִשְׁתִיָּה, וְלֹא חַיְשִׁינַן בְּמִתְקַיֵּם: 

~ פרק יא - משנה ב

{ב} <B> דְּבַשׁ תְּמָרִים.</B> אִם עָבַר וְעָשָׂה דְבַשׁ מִתְּמָרִים. וְעַיֵּן מַתְנִיתִין דִּלְקַמָּן. תּוֹסְפוֹת יוֹם טוֹב: {ג} <B> פְּטָמִים.</B> שֵׁם פַּטָּם הוּא לְשׁוֹן אֲרָמִית, כְּתַרְגּוּם יְרוּשַׁלְמִי אֲשֶׁר יִרְקַח, דִּי יְפַטֵּם: 

~ פרק יא - משנה ג

{ד} <B> אֵין עוֹשִׂין כוּ' דְּבַשׁ.</B> מִשּׁוּם דַּחֲשִׁיבֵי אִיצְטְרִיךְ לְמִיתְנִינְהוּ. הָרַ"שׁ: {ה} <B> אֵין סוֹפְגִין כוּ'.</B> הַךְ וְהַהִיא דְּבִכּוּרִים מוֹקֵי לַהּ בְּחֻלִּין דַּף ק"כ כְּרַבִּי יְהוֹשֻׁעַ: {ו} <B> וְאֵינוֹ מִטַּמֵּא מִשׁוּם מַשְׁקֶה.</B> מַשְׁקִים הַיּוֹצְאִים מִן הַפֵּרוֹת. הָרַמְבַּ"ם: 

~ פרק יא - משנה ד

{ז} <B>עֻקְצֵי תְאֵנִים.</B> בַּיְרוּשַׁלְמִי, מַתְנִיתָא בְּמֻבְלָעִין בָּאֹכֶל, וְהָרַמְבַּ"ם כָּתַב אַף עַל פִּי שֶׁאֵין בָּהֶן אֹכֶל: 

~ פרק יא - משנה ו

{ח} <B>מְכַבֵּד כְּדַרְכּוֹ. </B>דְּכֵיוָן שֶׁמְּכַבֵּד כְּדַרְכּוֹ לֹא מַטְרְחִינַן לֵיהּ טְפֵי, הוֹאִיל וְאֵינוֹ מִתְכַּוֵּן לְבַטֵּל. תּוֹסְפוֹת יוֹם טוֹב בְּשֵׁם הַפֵּרוּשׁ: 

~ פרק יא - משנה ח

{ט} <B> חֻלִּין.</B> אַף כִּי עֲדַיִן יֵשׁ טִפּוֹת שֶׁל תְּרוּמָה אֵין לָחוּשׁ, הוֹאִיל וְאֵינוֹ מִתְכַּוֵּן לְבַטֵּל. מֵהַנֵּי שְׁנֵי בָּבֵי מַשְׁמַע דְּדָבָר שֶׁאֵינוֹ נֶחְשָׁב בְּעֵינֵי הַמְבַטֵּל וְשֶׁאֵין כַּוָּנָתוֹ לְבַטֵּל לְכַתְּחִלָּה מֻתָּר. תּוֹסְפוֹת יוֹם טוֹב בְּשֵׁם הַפֵּרוּשׁ: {י} תַּרְגּוּם הַטִּי נָא כַדֵּךְ (בְּרֵאשִׁית כד) אַרְכִּינִי כְעַן, וּמַיְרֵי שֶׁעֲדַיִן לֹא נָתַן בְּתוֹכָהּ חֻלִּין, אֶלָּא דַּהֲוֵי אֲמֵינָא שֶׁהֵן חֻלִּין, מִפְּנֵי שֶׁהִסִּיחַ דַּעְתּוֹ מֵהֶן. הָרַמְבַּ"ם: {יא} <B> בִּתְרוּמַת.</B> וְזָכַר בְּכָאן זֶה הַדִּין לְפִי שֶׁקָּדַם לוֹ הַדָּבָר בַּשִּׁעוּרִין הַקְּטַנִּים מִן הַתְּרוּמָה שֶׁאֵינָן רְאוּיִן לָחוּשׁ בָּהֶן מִפְּנֵי מִעוּטָן כוּ', גַּם כֵּן זָכַר שִׁעוּר תְּרוּמַת מַעֲשֵׂר שֶׁל דְּמַאי שֶׁאֵין אָדָם חַיָּב מִפְּנֵי מִעוּטָהּ. הָרַמְבַּ"ם: {יב} <B> וְיוֹלִיכֶנָּה לַכֹּהֵן.</B> כְּלוֹמַר שֶׁאֵינוֹ רַשַּׁאי לְהַנִּיחָהּ בִּמְקוֹם הֶפְקֵר, אֲבָל לֹא שֶׁמְּחֻיָּב הוּא לַהֲבִיאָהּ אֶל הַכֹּהֵן, דְּאֵין יִשְׂרָאֵל חַיָּבִים לְהִטָּפֵל וְלַהֲבִיאָהּ מִן הַגֹּרֶן לָעִיר. תּוֹסְפוֹת יוֹם טוֹב: {יג} <B> מִשְּׁמֹנָה לַשְּׁמִינִית.</B> וְהוּא הַקּוֹרְטוֹב. וְדַרְכָּן לְהַפְלִיג בְּמִדָּה זוֹ מִפְּנֵי קַטְנוּתָהּ, שֶׁכֵּן אָמְרוּ אֲפִלּוּ חָסֵר קוֹרְטוֹב. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק יא - משנה ט

{יד} <B> שֶׁשָּׁם פָּרָה.</B> דְּשָׁמִין פָּרָה וַחֲמוֹר וְכָל דָּבָר שֶׁעוֹשֶׂה וְאוֹכֵל כְּדִתְנַן בְּמִשְׁנָה ה' פֶּרֶק ה' דְּבָבָא מְצִיעָא: 

~ פרק יא - משנה י

{טו} <B> בִּרְשׁוּת כֹּהֵן.</B> מְפָרֵשׁ בַּיְרוּשַׁלְמִי דְּלֹא קָאֵי בִרְשׁוּת כֹּהֵן אֶלָּא אַחוֹלִין, הָא קָמַיְתָא אֲפִלּוּ שֶׁלֹּא בִרְשׁוּת כֹּהֵן. הָרַ"שׁ. וְכֵן פָּסַק הָרַמְבַּ"ם:
