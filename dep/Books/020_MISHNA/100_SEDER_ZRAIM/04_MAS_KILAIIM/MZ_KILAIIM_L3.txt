~ פרק א - משנה א

{א} <b>הַחִטִּים.</b> כְּדֵי לְהוֹרוֹת שֶׁשְּׁנוּיִם זוּגוֹת זוּגוֹת לְכָךְ הִפְסִיק בְּזוּג קַמָּא וְתָנֵי עֲלַהּ אֵינָן כִּלְאַיִם זֶה בָזֶה: {ב} <B> כִּלְאַיִם.</B> כְּתִיב שָׂדְךָ לֹא תִזְרַע כִּלְאָיִם, הָיִיתִי אוֹמֵר אֲפִלּוּ שְׁנֵי מִינֵי חִטִּים אוֹ שְׂעוֹרִים, וְכֵן לְגַבֵּי בְּהֵמָה אֲפִלּוּ שׁוֹר שָׁחוֹר עַל גַּבֵּי לָבָן, וְכֵן לְגַבֵּי בְּגָדִים, פֵּרוּשׁ בִּבְגָדִים צֶמֶר וּפִשְׁתִּים יַחְדָּיו, מַה זֶּה לֹא זֶה מִמִּין זֶה, אַף כִּלְאַיִם שֶׁאָסַרְתִּי לְךָ בְּכָל מָקוֹם לֹא זֶה מִמִּין זֶה וְלֹא זֶה מִמִּין זֶה. יְרוּשַׁלְמִי: {ג} דְּאִי לָאו הָכִי לֹא הָיוּ אוֹסְרִין אֶלָּא בְּאוֹתוֹ מָקוֹם בִּלְבָד כִּבְסוֹף פֶּרֶק ה': {ד} <b>וְהַשְּׁעוּעִית.</b> מִפְּנֵי שֶׁמְּשַׁעְשְׁעוֹת אֶת הַלֵּב וּמְהַלֶּכֶת אֶת בְּנֵי מֵעַיִם, יְרוּשַׁלְמִי. כְּלוֹמַר שֶׁמְּטַמְטֶמֶת אֶת הַלֵּב. וּמְשַׁעְשַׁעַת מִלְּשׁוֹן וְעֵינָיו הָשַׁע (יְשַׁעְיָה ו). וְטָח אֶת הַבָּיִת, תַּרְגּוּם וִישׁוּעַ: 

~ פרק א - משנה ב

{ה} הֵם דּוֹמִים קְצָת לִבְצָלִים וּלְשׁוּמִים בִּהְיוֹתָם יְרֻקִּים עֲדַיִן שֶׁלֹּא נִתְיַבְּשׁוּ עֲדַיִן: {ו} רְצוֹנוֹ לוֹמַר כָּךְ, דְּלַעַת מָרָה דְּאוֹתוֹ הַמִּין שֶׁהוּא כָּךְ וְטוֹמְנִין בְּרֶמֶץ, בָּהּ עָסְקִינַן, וְהִיא הִיא רְמוּצָה. וְעַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ד

{ז} <B> הַפָּרִישִׁים.</B> פָּרִישִׁין בִּלְשׁוֹן אַשְׁכְּנַז קוויט"ן, עֻזְרָדִין בִּלְשׁוֹן אַשְׁכְּנַז שפייארלינ"ג: 

~ פרק א - משנה ה

{ח} הוֹאִיל וְטַעַם הַפְּרִי זֶה רָחוֹק מִטַּעַם פְּרִי זֶה בְּיוֹתֵר, הָרַ"מ. מַה שֶּׁאֵין כֵּן לֶפֶת וְנָפוּץ דְּמִשְׁנָה ג' שֶׁאֵין רִחוּקָן רִחוּק גָּדוֹל הֵקֵלּוּ מִפְּנֵי דָבָר אַחֵר שֶׁהֵם דּוֹמִין בּוֹ דִּמְיוֹן גָּדוֹל עַד שֶׁיֵּרָאוּ כִּשְׁנֵי גְוָנִין מִמִּין אֶחָד, שֶׁאֵין הוֹלְכִין בְּכִלְאַיִם אֶלָּא אַחֲרֵי מַרְאִית הָעַיִן, אֶלָּא אִם כֵּן טַעַם פְּרִי זֶה רָחוֹק מִטַּעַם פְּרִי זֶה בְּיוֹתֵר. תּוֹסְפוֹת יוֹם טוֹב בְּשֵׁם הָרַ"מ: 

~ פרק א - משנה ז

{ט} <B> אִילָן בְּאִילָן.</B> אַף עַל גַּב דִּכְבָר תָּנֵי לַהּ בְּמִשְׁנָה ד' חָזַר כָּאן לְפֵרוּשֵׁי דְּהָא דְאִיתָא כִּלְאַיִם בְּאִילָן הַיְנוּ דַּוְקָא בְּהַרְכָּבָה, דְּאִלּוּ לִזְרֹעַ שְׁנֵי מִינֵי זֶרַע אִילָנוֹת מֻתָּר: {י} <B> רַבִּי יְהוּדָה מַתִּיר יָרָק בְּאִילָן.</B> הוּא הַדִּין אִיפְּכָא, דְּמַאי שְׁנָא. וְטַעֲמוֹ דְרַבִּי יְהוּדָה שֶׁסּוֹבֵר דְּדַוְקָא כְּשֶׁעַל יְדֵי הַהַרְכָּבָה הֵן מִתְאַחִין וְנַעֲשֶׂה עַל יְדֵיהֶן פְּרִי אוֹ יָרָק מְשֻׁנֶּה אָסְרָה הַתּוֹרָה מִשּׁוּם כִּלְאַיִם, אֲבָל יָרָק וְאִילָן שֶׁאֵינָם מִתְאַחִין לְהִשְׁתַּנּוֹת אַף עַל פִּי שֶׁיּוֹנְקִים זֶה מִזֶּה לֹא אָסְרָה הַתּוֹרָה. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ח

{יא} כִּי מַה שֶּׁחוֹצֵב וּמְחַלֵּק הָאָרֶץ בְּיֹשֶׁר לַחֲלָקִים שָׁוִים נִקְרָא חֲצוּב: {יב} <b>מְשַׁמַּרְתּוֹ. </b>הַחַלָּמִית מַרְטִיב הַזֶּרַע וְשׁוֹמְרוֹ עַד שֶׁיְּהֵא הַזֶּרַע נִשְׁרָשׁ בָּאָרֶץ: 

~ פרק א - משנה ט

{יג} <B> תַּחַת הַגֶּפֶן.</B> שֶׁהוּא רַךְ וּבְוַדַּאי יַעַבְרוּ שָׁרְשֵׁי הַלֶּפֶת וּצְנוֹנוֹת בּוֹ, אֲפִלּוּ הָכִי מֻתָּר: {יד} דִּרְגִילוּת הוּא שֶׁמִּתּוֹסְפִים מֵחֲמַת לַחְלוּחִית הַקַּרְקַע אֲפִלּוּ כְּשֶׁמֻּנָּחִים בַּחַלּוֹן: {טו} <b>הֲרֵי זֶה כִּלְאָיִם.</b> לְחִיּוּב מַלְקוֹת, אֲבָל לְעִנְיַן אִסּוּר אֲפִלּוּ בְּלֹא זְרוּעָה כְּאַחַת יֵשׁ אִסּוּר וְאָסוּר לְקַיְּמָן כִּדְמוּכָח לְהַלָּן: {טז} <B> רַבִּי יְהוּדָה אוֹמֵר כוּ'.</B> כְּהַאי גַוְנָא פְּלִיגֵי בְּפֶרֶק בַּתְרָא לְעִנְיַן כִּלְאֵי בְגָדִים, דִּלְרַבִּי יְהוּדָה שָׁלֹשׁ תְּכִיפוֹת הֲוָה חִבּוּר וּלְרַבָּנָן אֲפִלּוּ שְׁתֵּי תְכִיפוֹת:

~ פרק ב - משנה א

{א} <B> יְמַעֵט.</B> אוֹ מוֹסִיף עַל סְאָה אוֹ פּוֹחֵת מֵרֹבַע, וְאַף דְּכָל הָאִסּוּרִין שֶׁרִבָּה עֲלֵיהֶם מֵזִיד אָסוּר, הָכָא הָאִסּוּר אֵינוֹ אֶלָּא מַרְאִית הָעַיִן: {ב} דִּדְבַר תּוֹרָה חַד בִּתְרֵי בָּטֵל אֶלָּא דַחֲכָמִים אָסְרוּ מִפְּנֵי מַרְאִית הָעַיִן וְהֵם אָמְרוּ דְּפָחוֹת מֵרֹבַע לִסְאָה מִתְבַּטֵּל: 

~ פרק ב - משנה ג

{ג} <b>אֶזְרַע.</b> וְאַף עַל פִּי שֶׁהַצְּמָחִים נִכָּרִים הֵם שֶׁבְּוַדַּאי יַהֲפֹךְ כֻּלָּם, אֲפִלּוּ הָכִי יַהֲפֹךְ מִתְּחִלָּה. תּוֹסְפוֹת יוֹם טוֹב: {ד} <B> כְּתַלְמֵי.</B> תֶּלֶם הוּא בֶּקַע שֶׁעוֹשָׂה הַמַּחֲרֵשָׁה בְּעָבְרָהּ בָּאָרֶץ, וְהוּא מִלְּשׁוֹן עַל תַּלְמֵי שָׂדָי (הוֹשֵׁעַ י): {ה} <B> יְשַׁיֵּר.</B> יֵשׁ לְפָרֵשׁ בְּכָל הַבֵּית סְאָה רֹבַע בְּמָקוֹם אֶחָד, וְיֵשׁ לְפָרֵשׁ דִּכְשֶׁתִּצְטָרֵף מַה שֶּׁאֵינוֹ חָרוּשׁ לֹא יַעֲלֶה לְרֹבַע. הָרַ"שׁ: 

~ פרק ב - משנה ד

{ו} לְדֻגְמָא נָקֵיט תְּבוּאָה וְהוּא הַדִּין קִטְנִיּוֹת וְזֵרְעוֹנֵי גִנָּה שֶׁנִקְרָאִים זְרִיעָה. וְהוּא מֻשְׁאָל לְיָרָק: {ז} <B> הוֹפֵךְ.</B> פֵּרוּשׁ לְאַחַר שֶׁהִתְלִיעָה: {ח} <B> אֲשָׁרֵשׁ.</B> נִגְזַר מִמִּלַּת שֹׁרֶשׁ, וְעִנְיָנוֹ לִכְרֹת הַשֹּׁרֶשׁ וְלַעֲקֹר אוֹתוֹ, כְּמוֹ וְשֵׁרֶשְׁךָ מֵאֶרֶץ חַיִּים [תְּהִלִּים נב]: 

~ פרק ב - משנה ה

{ט} לְפִי שֶׁהוּא נִסְפָּח אֶל קְצִיר הַשָּׁנָה שֶׁעָבְרָה, מִלְּשׁוֹן וְנִסְפְּחוּ עַל בֵּית יַעֲקֹב (יְשַׁעְיָה יד). הָרַמְבַּ"ן: {י} פֵּרוּשׁ אַחֵר. דְּאֵין מְחַיְּבִין אוֹתוֹ לַעֲקֹר דְּלֵיכָּא אִסּוּרָא שֶׁהֲרֵי הוּא לֹא זְרָעָן אֶלָּא מֵאֲלֵיהֶן בָּאוּ, וּמִשּׁוּם מְקַיֵּם כִּלְאַיִם לֵיכָּא מִשּׁוּם דְּקָשֶׁה לְהוּ עֲשָׂבִים, וְדַוְקָא הֵיכָא דְנִיחָא לֵיהּ חִיְּבוּהוּ חֲכָמִים שֶׁלֹּא יַחְשְׁדוּהוּ שֶׁזְּרָעָם. רִ"י בַּתּוֹסָפוֹת: 

~ פרק ב - משנה ז

{יא} וּרְצוֹנוֹ לוֹמַר מִכָּל אֵיזֶה תַּבְנִית שֶׁתִּהְיֶה בֵּין מְשֻׁלָּשׁ בֵּין מְרֻבָּע אוֹ שְׁאָר תַּבְנִית. הָרַ"מ: {יב} <B> נִכְנָס.</B> דְּמִן הַתּוֹרָה לֹא נֶאֱסַר אֶלָּא כְּשֶׁזּוֹרֵעַ בְּמַפֹּלֶת יָד, וְרַבָּנָן אָסְרוּ כָּל מָקוֹם שֶׁיֵּשׁ מַרְאִית הָעַיִן, וּלְפִיכָךְ הִתִּירוּ עַל כַּמָּה דְרָכִים. רַ"מ: {יג} וּבִלְבָד שֶׁיִּסְמְכֵהוּ לִשְׂדֵה חֲבֵרוֹ. הָרַ"מ: {יד} וְנִמְצָא כְּזוֹרֵעַ לְהַשְׁחָתָה. הָרַ"מ: 

~ פרק ב - משנה ח
