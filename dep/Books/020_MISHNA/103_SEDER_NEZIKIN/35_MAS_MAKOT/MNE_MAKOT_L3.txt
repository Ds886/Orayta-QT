~ פרק א - משנה א

{א} דְּתַנָּא קָאֵי אַדִּתְנַן בְּסִיּוּמָא דְסַנְהֶדְרִין דְּזוֹמְמֵי בַת כֹּהֵן אֵין עוֹשִׂין בָּהֶם דִּין הֲזַמָּה שֶׁזָּמְמוּ לַעֲשׂוֹת לָהּ. וְקָאָמַר הָכָא יֵשׁ עוֹד עֵדִים זוֹמְמִים אֲחֵרִים שֶׁאֵין עוֹשִׂין בָּהֶם דִּין הֲזַמָּה כָל עִקָּר אֶלָּא מַלְקוּת אַרְבָּעִים. כֵּיצַד כוּ'. גְּמָרָא: {ב} כְּלוֹמַר קֹדֶם שֶׁהוֹלִידוֹ אָבִיו, שֶׁהוּא שְׁעַת הַהֵרָיוֹן. רַשִׁ"י: {ג} <B> גְּרוּשָׁה.</B> הָא דְלָא נָקַט מַמְזֵר דְּזֶה הֲוָה שַׁיָּךְ בֵּין בְּכֹהֲנִים בֵּין בְּיִשְׂרְאֵלִים, יֵשׁ לְיַשֵּׁב מִשּׁוּם דְּקָאֵי אַזּוֹמְמֵי בַת כֹּהֵן וּלְכָךְ נָקַט מִידֵי דְּשַׁיָּךְ בִּכְהֻנָּה. תּוֹסָפוֹת: {ד} <B> בֶּן חֲלוּצָה.</B> וְהוּא חָלָל מִדְּרַבָּנָן. הָרַ"ן: {ה} <B> לִגְלוֹת.</B> וְאִם תֹּאמַר הָאֵיךְ הֵם יְכוֹלִים לְחַיְּבוֹ גָלוּת בְּעֵדוּתָן, וְהָא יָכוֹל לוֹמַר מֵזִיד הָיִיתִי כוּ'. וְיֵשׁ לוֹמַר דְּמַיְרֵי כְשֶׁרָאוּ בוֹ רַגְלַיִם לַדָּבָר, שֶׁנִּשְׁמַט הַבַּרְזֶל מִקַּתּוֹ, וְאִיכָּא לְמֵימַר דְּלֹא נִתְכַּוֵּן. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} <B> בִּכְתֻבָּתָהּ.</B> וְאִם תֹּאמַר וְלִחַיְּבוּ בִשְׁאֵר כְּסוּת וְעוֹנָה, דְּהָא מֻפְסָד לָאִשָּׁה. וְיֵשׁ לוֹמַר דְּמַיְרֵי שֶׁמַּעֲשֵׂה יָדֶיהָ שֶׁהֵן בְּיָדוֹ שָׁוִין כִּשְׁאָר כוּ'. עוֹד יֵשׁ לוֹמַר שֶׁאוֹמֶרֶת גֵּרַשְׁתַּנִי וְאִם כֵּן מָחֲלָה אוֹתָן לְבַעְלָהּ. תּוֹסָפוֹת. וּבְפֵרוּשׁ שְׁאֵר וְעוֹנָה, עַיֵּן מַה שֶּׁכָּתַבְנוּ פֶּרֶק ד' דִּכְתֻבּוֹת מִשְׁנָה ד': {ז} וְרַשִׁ"י הִקְשָׁה, דְּלֵיכָּא לְמֵימַר שֶׁאוֹתָן הַדָּמִים יִתְּנוּ הַזּוֹמְמִים, שֶׁהֲרֵי לֹא זְכוּת סְפֵקָהּ בָּאוּ לְהַפְסִידוֹ, אֶלָּא זְכוּת סְפֵקוֹ, שֶׁהוּא מְצַפֶּה שֶׁאִם תָּמוּת בְּחַיָּיו יִירָשֶׁנָּה, וְיָפֶה כֹחַ זְכוּתוֹ מִכֹּחַ זְכוּתָהּ, שֶׁהוּא אוֹכֵל פֵּרוֹת הַקַּרְקַע הַמְיֻחֶדֶת לִכְתֻבָּתָהּ תָּמִיד, וְלֹא הִיא. וְעוֹד, שֶׁהוּא מֻחְזָק וְהִיא מְחֻסֶּרֶת גּוֹבַיְנָא כוּ'. וְאִם בָּאוּ לִמְכֹּר זְכוּתָם, שֶׁלּוֹ נִמְכָּר יוֹתֵר מִשֶּׁלָּהּ, וְזֶה יִהְיוּ הַזּוֹמְמִים מְחֻיָּבִים לְשַׁלֵּם. וְיֵשׁ לוֹמַר דְּהָכִי קָאָמַר, שָׁמִין זְכוּת סְפֵקָהּ כַּמָּה כוּ' וְאוֹתוֹ לֹא יְשַׁלֵּם שֶׁהֲרֵי גַם עַתָּה יִתֵּן זֶה בְרָצוֹן, אֶלָּא הַשְּׁאָר שֶׁהָיוּ מַפְסִידִים אוֹתוֹ יִתְּנוּ. אִי נַמִּי אִין הָכִי נַמִּי דְּשָׁמִין זְכוּת סְפֵקוֹ קָאָמַר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ח} כְּלוֹמַר בְּעֵדִים זוֹמְמִין, דִּבְהוּ קָיְמִינַן. וְעַיֵּן פֶּרֶק ח' דְּבָבָא קַמָּא מִשְׁנָה ג' בְּדִבְרֵי הָרַ"ב: {ט} דִּבְפֵרוּשׁ רִבְּתָה תּוֹרָה עֵדִים זוֹמְמִין לְתַשְׁלוּמִין. מִכְּדֵי כְּתִיב וַעֲשִׂיתֶם לוֹ כַּאֲשֶׁר זָמַם, יָד בְּיָד לָמָּה לִּי, דָּבָר הַנִּתָּן מִיָּד לְיָד, וּמַאי נִיהוּ, מָמוֹן. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ג

{י} הָכִי קָאָמַר, דְּהוֹאִיל וּמַשְׁכַּחַת לַהּ שֶׁחִיְּבָהוּ הַכָּתוּב לְהַלְקוֹת מִשּׁוּם לֹא תַעֲנֶה (וְאַף עַל גַּב שֶׁאֵין בּוֹ מַעֲשֶׂה) כְּשֶׁאִי אֶפְשָׁר לְקַיֵּם כוּ', כֵּיוָן דְּגַלֵּי לָן קְרָא דִּוְהִצְדִּיקוּ דְּשַׁיָּךְ בֵּיהּ מַלְקוֹת, הִלְכָּךְ לְעוֹלָם נְחַיְּבֵהוּ מִשּׁוּם לֹא תַעֲנֶה אֲפִלּוּ כְשֶׁמִּתְקַיֵּם וַעֲשִׂיתֶם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} דְּהָא מַקְּשִׁינַן שְׁלֹשָׁה לִשְׁנַיִם, כְּדִתְנַן בְּמִשְׁנָה ז'. וְעוֹד יֵשׁ לוֹמַר דְּמִשּׁוּם מַלְקוֹת אִצְטְרִיךְ לְאַסְבּוֹרֵי בִשְׁלֹשָׁה, שֶׁיַּגִּיעַ לְכָל אֶחָד וְאֶחָד שְׁלֹשׁ עֶשְׂרֵה וְאִי אֶפְשָׁר לַחֲצוֹת לִשְׁנַיִם: {יב} שֶׁלֹּא קִבֵּל כְּלוּם, אֶלָּא שֶׁמַּלְקִים לָעֵדִים כַּאֲשֶׁר זָמְמוּ לְהַלְקוֹתוֹ, וְאַף עַל פִּי שֶׁיִּהְיוּ נִלְקִים בֵּין שְׁלָשְׁתָּן הַמַּלְקוֹת הָאַחַת לֹא נִתְקַיֵּם כַּאֲשֶׁר זָמַם, שֶׁאֵין כָּאן מַלְקוֹת אַחַת כְּלָל. מַה שֶּׁאֵין כֵּן בְּמָמוֹן שֶׁמִּצְטָרֵף וְיֵשׁ כָּאן קַבָּלַת מָמוֹן כַּאֲשֶׁר זָמַם לְהוֹצִיא מִמֶּנּוּ: 

~ פרק א - משנה ד

{יג} מִדְּלֹא נֶאֱמַר עֵדוּת שֶׁקֶר. הָרַמְבַּ"ם: {יד} <B> אֵין כוּ'.</B> וְהִיא עֵדוּת מֻכְחֶשֶׁת, וְנִדְחוּ דִבְרֵיהֶם שֶׁל אֵלּוּ וְאֵלּוּ, וּכְאִלּוּ אֵין שָׁם עֵדוּת כָּל עִקָּר. הָרַמְבַּ"ם: {טו} <B> אַתֶּם הֱיִיתֶם.</B> וְאֵין אָנוּ יוֹדְעִים אִם זֶה הָרַג אֶת זֶה בְּיוֹם זֶה בְּמָקוֹם פְּלוֹנִי כְּמוֹ שֶׁאַתֶּם אוֹמְרִים אוֹ לֹא הֲרָגוֹ, הוֹאִיל וְהָעֵדִים שֶׁהִזִּמּוּם לֹא הִשְׁגִּיחוּ עַל עַצְמָהּ שֶׁל עֵדוּת כְּלָל אִם אֱמֶת הָיָה אוֹ שֶׁקֶר, הֲרֵי אֵלּוּ זוֹמְמִין. אֲבָל אִם אָמְרוּ בְּיוֹם זֶה וּבְמָקוֹם זֶה הָיִינוּ עִמָּכֶם וְעִם אֵלּוּ כָּל הַיּוֹם וְלֹא הָיוּ דְבָרִים מֵעוֹלָם, הוֹאִיל וְהַהַכְחָשָׁה בָעֵדוּת עַצְמָהּ, אֵינָם זוֹמְמִים כוּ'. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב לְדַעַת הַטּוּר: {טז} <B> בְּמָקוֹם פְּלוֹנִי.</B> רָחוֹק מֵאוֹתוֹ מָקוֹם שֶׁאַתֶּם אוֹמְרִים יוֹתֵר מִמַּהֲלַךְ יוֹם. הָרַ"ן: {יז} <B> וְנֶהֱרָגִין כוּ'.</B> רְצוֹנוֹ לוֹמַר שֶׁיֵּהָרְגוּ הָעֵדִים עַל פִּי אֵלּוּ שֶׁהִזִּימוּ, וְאַף עַל פִּי שֶׁהֵם תְּרֵי וּתְרֵי, לְפִי שֶׁהָעֵדוּת עַל עַצְמָם שֶׁל עֵדִים וְלֹא בָאוּ לְהָעִיד עַל עַצְמָהּ שֶׁל עֵדוּת לֹא לְקַיְּמָהּ וְלֹא לְבַטְּלָהּ, לְפִי שֶׁהֵם אוֹמְרִים אֵין אָנוּ יוֹדְעִים אִם זֶה הָרַג כְּמוֹ שֶׁאַתֶּם חוֹשְׁבִים אוֹ שֶׁלֹּא הֲרָגוֹ, אֵין עָלֵינוּ לָדַעַת עֵדוּת. אֲבָל מַה שֶּׁאָנוּ מְעִידִין שֶׁאַתֶּם הֱיִיתֶם בְּמָקוֹם פְּלוֹנִי בְּיוֹם פְּלוֹנִי. לְפִיכָךְ שׁוֹמְעִין דִּבְרֵיהֶם וְיֵהָרְגוּ הָעֵדִים. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יח} דְּאַטּוּ כֻלֵּי עָלְמָא גַּבַּיְהוּ הֲווּ קַיְמֵי. גְּמָרָא. וּפָרֵיךְ אֲפִלּוּ כַת רִאשׁוֹנָה נַמִּי לֹא כוּ'. קָשְׁיָא: {יט} וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם. וְיֵשׁ לִתְמֹהַּ, מַאי אִסְטָסִית הִיא זוֹ, וְהָא לָאו מִשּׁוּם הַמַּזִּימִין קָאָתִינַן עֲלַיְהוּ, אֶלָּא מַשְׁמַע אִפְּכָא, דְּהָא מִשּׁוּם דְּתוּ לֹא מְקַבְּלִים סָהֲדוּתָא וְכוּ' הוּא, וּמַחֲזִיקִין לַמֻּזַמִּין וּלְכָל הַמְּעִידִים כְּמוֹתָם שֶׁהֵם שַׁקְרָנִים, וְשֶׁהַמַּזִּימִים הֵעִידוּ בֶאֱמֶת. וְנִרְאֶה דְיֵשׁ לְפָרֵשׁ דְּרַבִּי יְהוּדָה לְדִבְרֵיהֶם דְּרַבָּנָן קָאָמַר לְהוּ, לְדִבְרֵיכֶם דְּסָבְרִיתוּ דִמְקַבְּלִים סָהֲדוּתָא דְכַת שְׁנִיָּה כוּ', אֲפִלּוּ הָכִי לֹא הֲוָה לְכוּ לָדוּן לַהֲרִיגָה לְכָל הַמֻּזַמִּים, אֶלָּא הֲוָה לְכוּ לוֹמַר אִסְטָסִית הִיא זוֹ. אֲבָל לְדִידִי בְּלָאו הָכִי אֵינָהּ נֶהֱרֶגֶת אֶלָּא כַת הָרִאשׁוֹנָה בִלְבַד מִשּׁוּם דְּאֵין מְקַבְּלִין כוּ'. וְהֶאֱרִיךְ, וּמַסִּיק דִּסְבָרָא זוֹ הִיא בַגְּמָרָא לְהַסָּלְקָא דַעְתָּא. וּפֵרוּשׁ הָרִאשׁוֹן הוּא לְהַמַּסְקָנָא, אֶלָּא דִלְהַסָּלְקָא דַעְתָּא לֹא קָשְׁיָא קוּשְׁיַת הַגְּמָרָא שֶׁהֵבֵאתִי. לְהָכִי פֵרֵשׁ הָרַמְבַּ"ם לְמַתְנִיתִין כְּהַסָּלְקָא דַעְתָּא. וְיָצָא לָנוּ דִּשְׁנֵי הַפֵּרוּשִׁים שֶׁכָּתַב הָרַב בְּמִלַּת אִסְטָסִית וְחִלּוּף הַגִּרְסָאוֹת עוֹלִים לִשְׁנֵי הַפֵּרוּשִׁים, וְאֵינוֹ נִתְלֶה כְלָל בְּפֵרוּשׁ הָעִנְיָן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{לב} <B> וְאֶחָד כוּ'.</B> תֵּימַהּ, אַמַּאי הֻצְרַךְ לוֹמַר וְאֶחָד מַתְרֶה שֶׁאֵין מִן הָעֵדִים כוּ'. וְנִרְאֶה לְמוֹרִי שֶׁיִּחְיֶה, דְּנָקַט לַאֲפוֹקֵי מִדִּבְרֵי רַבִּי יוֹסֵי דְבַסֵּיפָא כוּ'. תּוֹסָפוֹת:  {לג} וּלְשׁוֹן מִקְצָתָן לֹא דָיֵיק הָכִי. וּלְשׁוֹן הַתּוֹסָפוֹת, שֶׁמִּקְצָתָן רוֹאִין כוּ' מִצְטָרְפִין אַף אוֹתָם שֶׁאֵין רוֹאִין: {לד} <B> וְאִם לָאו.</B> מַיְרֵי כְּשֶׁאֵין רוֹאִין הַמַּתְרֶה וְאֵין הַמַּתְרֶה רוֹאֶה אוֹתָם, אֶלָּא שָׁמְעוּ הַמַּתְרֶה שֶׁמַּתְרֶה כֵן. וּבְכָךְ סַגִּי, דְּהָא קָיְמִינַן לְרַבָּנָן דִּפְלִיגֵי אַדְּרַבִּי יוֹסֵי דִבְסָמוּךְ וְלֹא בָעֵי פִּי שְׁנֵי עֵדָיו מַתְרִין בּוֹ כְּמוֹ רַבִּי יוֹסֵי. תּוֹסָפוֹת. וּכְלוֹמַר דְּאִי הַמַּתְרֶה רוֹאֶה אוֹתָן אוֹ הֵן רוֹאִין אוֹתוֹ מִצְטָרְפִין. וְכֵן כָּתַב הָרַמְבַּ"ם בְּחִבּוּרוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לה} <B> פְּטוּרָה.</B> אִם הוּזַמָּה אַחַר שֶׁנֶּהֱרַג הָרוֹצֵחַ. דְּכֵיוָן שֶׁנֶּהֱרַג הַהוֹרֵג עַל פִּיהֶן שׁוּב אֵין נֶהֱרָגִין, דְּקַיְמָא לָן לֹא הָרְגוּ נֶהֱרָגִין, הָרְגוּ אֵין נֶהֱרָגִין. דִּכְתִיב כַּאֲשֶׁר זָמַם לַעֲשׂוֹת, וְלֹא כַאֲשֶׁר עָשָׂה. וּלְעִנְיַן מָמוֹן, אֲפִלּוּ כַאֲשֶׁר עָשָׂה נַמִּי מְחֻיָּבִים לְשַׁלּוּמֵי (עַיֵּן בָּבָא בַּתְרָא פ"ג מ"ד). וְדַעַת הָרַמְבַּ"ם דַּאֲפִלּוּ לְעִנְיַן מַלְקוֹת נַמִּי אֲפִלּוּ לָקָה זֶה שֶׁהֵעִידוּ עָלָיו לוֹקִין אֵלּוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לו} <B> שֶׁנֶּאֱמַר כוּ'.</B> כְּלוֹמַר שֶׁיְּהֵא הַדָּבָר כֻּלּוֹ שֶׁהוּא הָעֵדוּת וְהַהַתְרָאָה קָם עַל פִּי שְׁנַיִם הָעֵדִים. נִמּוּקֵי יוֹסֵף: {לז} <B> שֶׁלֹּא כוּ'.</B> שֶׁמָּא יַחֲלִיף הַמְתֻרְגְּמָן בִּלְשׁוֹן הָעֵדוּת. וְעוֹד כְּשֶׁיִּשְׁמְעוּ הֵם מִפִּי הָעֵדִים יִתָּכֵן לָהֶם לַחְקֹר אוֹתָם יוֹתֵר. נִמּוּקֵי יוֹסֵף. וְנִרְאֶה בְעֵינַי דְּאַף עַל גַּב דְּרַבִּי שִׁמְעוֹן הוּא דְדָרֵישׁ טַעֲמָא דִקְרָא, בְּכָאן אָנוּ צְרִיכִין לְכָךְ, דְּאִי לָאו הָכִי מְנָא לֵיהּ לְרַבִּי יוֹסֵי לְהוֹצִיא שְׁתֵּי דְרָשׁוֹת מִן הַכָּתוּב, אֶלָּא דְּעִקָּרוֹ לְהַתְרָאָה אֲתָא, וְהַדָּבָר אַחֵר מִסְּבָרָא הוּא אֶלָּא דְּאַסְמְכֵיהּ אַקְּרָא:

{א} <B> וְנָפְלָה.</B> וְלֹא תָנֵי נִפְסַק הַחֶבֶל. וְנִרְאֶה לִי דְּנָפְלָה פֵּרוּשׁ שֶׁנִּשְׁמַט מִיָּדוֹ. וּלְכָךְ לֹא רָצָה לוֹמַר וְנִפְסַק, דְּאִם כֵּן לֹא אָתְיָא כְרַבִּי דְּאָמַר לְקַמָּן נִפְסַק אֵינוֹ גוֹלֶה. תּוֹסָפוֹת: {ב} <B> יוֹרֵד.</B> לֹא זוֹ אַף זוֹ קָתָנֵי. לֹא זוֹ מְעַגֵּל דְּוַדַּאי יֵשׁ לְחַיְּבוֹ גָלוּת, דִּלְמָא לֹא נִזְהַר, אֶלָּא אַף מְשַׁלְשֵׁל חָבִית דְּשַׁפִּיר מִזְהַר שֶׁלֹּא יְשַׁבֵּר חָבִיתוֹ, אִם נָפַל וְהָרַג חַיָּב. וְעוֹד דְּלֹא זוֹ אֶלָּא אַף יוֹרֵד בְּסֻלָּם דִּטְפֵי מִזְהַר כְּדֵי שֶׁלֹּא יִפֹּל, אִם הָרַג חַיָּב. תּוֹסָפוֹת: {ג} וְהָרַמְבַּ"ם הִטְעִים הַטַּעַם יָפֶה, שֶׁהֲרֵי דֶרֶךְ נְפִילָה מָצוּי בְּרֹב הָעִתִּים לְהַזִּיק, וְדָבָר קָרוֹב הוּא לִהְיוֹת, שֶׁהֲרֵי טֶבַע הַכָּבֵד לֵירֵד לְמַטָּה בִמְהֵרָה, וְהוֹאִיל וְלֹא זֵרַז עַצְמוֹ וְתִקֵּן מַעֲשָׂיו יָפֶה בִשְׁעַת יְרִידָה יִגְלֶה. אֲבָל דֶּרֶךְ עֲלִיָּה אֵין זֶה הַדָּבָר הַקָּרוֹב לִהְיוֹת בְּרֹב הָעִתִּים, אֶלָּא כְמוֹ פֶלֶא הוּא וְהוּא כְמוֹ אָנוּס: {ד} כְּגוֹן הָרוֹצֶה לְהָרִים יָדוֹ בְכֹחַ וְגַרְזֶן בְּיָדוֹ וּמַשְׁפִּיל תְּחִלָּה אֶת גּוּפוֹ וְשׁוֹחֶה לְפָנָיו כְּדֵי לְהָרִים בְּכֹחַ, וּבְהַשְׁפָּלָתוֹ הָרַג. דְּלֹא תֵימָא הוֹאִיל וְצֹרֶךְ עֲלִיָּה הִיא פָּטוּר מִגָּלוּת דְּהָא כְתִיב וַיַּפֵּל. רַשִׁ"י. וּכְשֶׁהָרַג בַּעֲלִיָּתוֹ לַאֲחוֹרָיו, פָּטוּר. וְהַיְנוּ דִתְנַן שֶׁלֹּא בְדֶרֶךְ יְרִידָתוֹ אֵינוֹ גוֹלֶה: {ה} עַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב שֶׁהֶאֱרִיךְ בְּטַעַם פְּלוּגְתָּתָן בַּקְּרָא עַל פִּי הַדִּקְדּוּק: {ו} <B>מִן הָעֵץ כוּ'.</B> יָצָא קֵיסָם וְנִתַּז לְמֵרָחוֹק וְהָרַג. רַשִׁ"י: {ז} כְּלוֹמַר וְרַבִּי סָבַר דְּהַאי לָאו כֹּחַ כֹּחוֹ הוּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ב - משנה ב

{ח} כְּלוֹמַר וְזָרַק בָּהּ בַּיּוֹם. אֲבָל בְּאַשְׁפָּה שֶׁאֵינָהּ עֲשׂוּיָה לְהִפָּנוֹת בָּהּ אַף בַּלַּיְלָה וְזָרַק בָּהּ בַּלַּיְלָה וּמִקְּרֵי וִיתֵיב כוּ', אָנוּס הוּא, דְּלֵית לֵיהּ לְאַסּוֹקֵי אַדַּעְתֵּיהּ דְּמִקְּרֵי וִיתֵיב כֵּיוָן שֶׁאֵינָהּ עֲשׂוּיָה לִפָּנוֹת כְּלָל. תּוֹסָפוֹת: {ט} <B> הַלָּז.</B> נִמְצָא בַכָּתוּב מִי הָאִישׁ הַלָּזֶה (בְּרֵאשִׁית כד). וּבַיְרוּשַׁלְמִי (וְכֵן בַּבַּבְלִי. רַבֵּנוּ מֹשֶׁה זַכּוּתָא) גָּרַס הַלָּה: {י} <B> וְלַמַּזִּיק.</B> אִצְטְרִיךְ לְמִתְנֵי וְלַמַּזִּיק, דְּלֹא תֵימָא שֶׁאִם יֵשׁ לַמַּזִּיק גַּם כֵּן רְשׁוּת לִכָּנֵס דְּפָטוּר מִשּׁוּם דְּעַל הַנִּזָּק לְהַרְחִיק וְלִשְׁמֹר אֶת עַצְמוֹ מִן הַנֶּזֶק כְּשֶׁיּוֹדֵעַ שֶׁיֵּשׁ לְהַלָּה גַּם כֵּן רְשׁוּת לִכָּנֵס: {יא} <B> אֶת בְּנוֹ.</B> לְהַטּוֹתוֹ לְדֶרֶךְ אַחֶרֶת. רַשִׁ"י: {יב} <B> וּשְׁלִיחַ בֵּית דִּין.</B> הַמַּלְקֶה אַרְבָּעִים לַחַיָּב מַלְקוֹת. וּלְשׁוֹן הָרַאֲבַ"ד, שְׁלִיחַ בֵּית דִּין שֶׁהוּא מַלְקֶה יוֹתֵר מִמַּה שֶּׁאֲמָדוּהוּ. וְהָרַמְבַּ"ם כָּתַב שֶׁהִכָּה אֶת הַנִּמְנָע לָבֹא לַדִּין וֶהֱמִיתוֹ בִשְׁגָגָה: 

~ פרק ב - משנה ג

{יג} דְּמוּסָר נַמִּי כְתִיב יַסֵּר בִּנְךָ וִינִיחֶךָ. וְאֻמָּנוּת הָא תְנַן בְּסוֹף קִדּוּשִׁין לְעוֹלָם יְלַמֵּד אֶת בְּנוֹ כוּ': {יד} שֶׁנֶּאֱמַר וְהָיוּ לָכֶם הֶעָרִים לְמִקְלָט, לָכֶם לְמִקְלָט וְלֹא לַגֵּרִים. גְּמָרָא. וְהָא דְנֶהֱרָג, מִשּׁוּם דְּקַיְמָא לָן דְּבֶן נֹחַ אֵין צָרִיךְ הַתְרָאָה, אֶלָּא אַזְהָרָתוֹ זוֹ הִיא מִיתָתוֹ. וּמְסַיֵּם הָרַמְבַּ"ם, אֲבָל יִשְׂרָאֵל שֶׁהֲרָגוֹ גּוֹלֶה. עַד כָּאן. וּמַתְנִיתִין הָכִי קָאָמַר, הַכֹּל כוּ', כְּלוֹמַר אִית בְּהוּ תַרְתֵּי, שֶׁבֵּין הוּא הַהוֹרֵג אוֹ הַנֶּהֱרָג יֵשׁ כָּאן גָּלוּת, חוּץ מֵעַל יְדֵי גֵר תּוֹשָׁב שֶׁאַף עַל פִּי שֶׁאִם יִשְׂרָאֵל הֲרָגוֹ גּוֹלֶה, מִכָּל מָקוֹם אִם הוּא הָרַג לְיִשְׂרָאֵל אֵינוֹ גוֹלֶה אֶלָּא נֶהֱרָג. כֵּן נִרְאֶה לִי. וְכֵן מַשְׁמַע בַּגְּמָרָא שֶׁהַהוּא עַל יְדֵי, לָאו כְּגַוְנָא דְאִינָךְ הוּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} <B> אֶלָּא כוּ'.</B> וְעַל יָדוֹ מִיהָא גּוֹלֶה. דִּכְתִיב (בְּמִדְבַּר לה) לִבְנֵי יִשְׂרָאֵל וְלַגֵּר וְלַתּוֹשָׁב בְּתוֹכָם תִּהְיֶינָה שֵׁשׁ הֶעָרִים. גְּמָרָא: {טז} דְּמַשְׁמַע כָּאן לֹא רָאָה אֲבָל רוֹאֶה בְּמָקוֹם אַחֵר, פְּרָט לַסּוּמָא שֶׁאֵינוֹ רוֹאֶה בְּשׁוּם מָקוֹם. רַשִׁ"י: {יז} לָאו דִּבְלֹא רְאוֹת רִבּוּיָא הוּא. אֶלָּא הָכִי מְפָרְשִׁינַן בַּגְּמָרָא, בְּלֹא רְאוֹת לְמַעֵט, בִּבְלִי דַעַת לְמַעֵט, וְאֵין מִעוּט אַחַר מִעוּט אֶלָּא לְרַבּוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יח} <B> כְּמוּעָד.</B> כְּמַתְרֶה עָלָיו וְעוֹבֵר עַל הַתְרָאָה, דְּוַדַּאי לְדַעַת הֲרָגוֹ. רַשִׁ"י. וּבַגְּמָרָא, וְהָא לֹא אַתְרוּ בֵיהּ כוּ', וּמְשָׁנֵי רַבִּי יוֹסֵי בַר יְהוּדָה הִיא דִּסְבִירָא לֵיהּ הַתְרָאָה לֹא נִתְּנָה אֶלָּא לְהַבְחִין בֵּין שׁוֹגֵג לְמֵזִיד: {יט} <B> יָכוֹל כוּ'.</B> כֵּיצַד, אָמַר רַבִּי שִׁמְעוֹן יֵשׁ שׂוֹנֵא כוּ'. נִפְסַק גּוֹלֶה, נִשְׁמַט אֵינוֹ גוֹלֶה. פֵּרֵשׁ רַשִׁ"י נִפְסַק הַחֶבֶל דּוּמְיָא דְנִשְׁמַט הַבַּרְזֶל מִקַּתּוֹ הוּא, שֶׁנִּשְׁאַר הָאֶגֶד בְּיָדוֹ. וְנִשְׁמַט, דּוּמְיָא דְמִן הָעֵץ הַמִּתְבַּקֵּעַ, שֶׁלֹּא נִשְׁאַר בְּיָדוֹ כְלוּם:

{כ} דִּכְתִיב וַעֲלֵיהֶם תִּתְּנוּ אַרְבָּעִים וּשְׁתַּיִם עִיר, וּכְתִיב כָּל הֶעָרִים אֲשֶׁר תִּתְּנוּ לַלְוִיִּם אַרְבָּעִים וּשְׁמֹנֶה עִיר, הִקִּישָׁן הַכָּתוּב זוֹ לְזוֹ לִקְלֹט. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כא} גְּמָרָא. לְתָרוֹצֵי דְּלֹא תְנַן אֶלָּא שֵׁשׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} <B> לֹא הָיוּ כוּ'.</B> וּמִפְּנֵי מָה הִבְדִּילָם מֹשֶׁה, אָמַר מִצְוָה שֶׁבָּאָה לְיָדִי אֲקַיְּמֶנָּה. גְּמָרָא: 

~ פרק ב - משנה ה

{כג} בְּכָל מָקוֹם שֶׁהָיוּ שְׁנֵי דְרָכִים מְפֻצָּלִים, אֶחָד פּוֹנֶה לְעִיר מִקְלָט, הָיָה עֵץ תָּקוּעַ בְּאוֹתוֹ דֶרֶךְ וְכָתוּב בּוֹ מִקְלָט. גְּמָרָא. מַאי קְרָא, תָּכִין לְךָ הַדֶּרֶךְ, עֲשֵׂה הֲכָנָה לַדֶּרֶךְ: {כד} בַּגְּמָרָא. וְהַיְנוּ דְכָתַב הָרַמְבַּ"ם, כְּשֶׁמְּשִׁיבִין אוֹתוֹ מוֹסְרִים לוֹ כוּ' כְּדִתְנַן לְקַמָּן. וְלֹא מְפָרֵשׁ לַהּ דִּכְשֶׁמַּקְדִּימִין כוּ' כְּדִלְקַמָּן שֶׁמּוֹסְרִים לוֹ, מִשּׁוּם דְּאִי בִתְחִלַּת הַנִּיסָה הָאֵיךְ יוּכְלוּ לְהַחְלִיט וְלוֹמַר בִּשְׁגָגָה, דִּלְמָא לֹא הִיא: {כה} וְכֵן כָּתַב הָרַמְבַּ"ם. וּלְדִבְרֵיהֶם לֹא גָרְסִינַן אַף. וְכֵן הוּא בַבְּרַיְתָא. וְתוּ הָתָם, אָמְרוּ לוֹ הַרְבֵּה שְׁלִיחוּת עוֹשָׂה, פֵּרֵשׁ רַשִׁ"י דְּבָרִים שֶׁאֵינָם נַעֲשִׂים עַל יְדֵי אָדָם נַעֲשִׂים וּמִתְקַבְּלִים עַל יְדֵי הַרְבֵּה שְׁלוּחִים:
