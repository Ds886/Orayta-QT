~ פרק א - משנה א
{א} וְיֵשׁ לְכָל אֶחָד לְגַלְגֵּל עַל חֲבֵרוֹ שֶׁכָּל מַה שֶּׁנָּטַל כַּדִּין נָטַל. הָרַמְבַּ"ם. וְנִרְאֶה לִי דְּטַעֲמוֹ שֶׁעַל מַה שֶּׁבְּיָדוֹ מַמָּשׁ לֹא שַׁיְכָא תַּקָּנָה זוֹ, דְּאִם לֹא כֵן בְּכוֹפֵר הַכֹּל נַמִּי הֲוָה לְהוּ לְתַקְנָהּ: {ב} בַּגְּמָרָא, בְּמִקָּח וּמִמְכָּר. פֵּרֵשׁ רַשִׁ"י דְּאִיכָּא לְמֵימַר שְׁנֵיהֶם קְנָאוּהָ וְלִשְׁנֵיהֶם נִתְרַצָּה הַמּוֹכֵר. אֲבָל זֶה אוֹמֵר אֲנִי אֲרַגְתִּיהָ וְזֶה אוֹמֵר אֲנִי כוּ' לֹא יַחֲלֹקוּ, דְּוַדַּאי חַד מִנַּיְהוּ רַמַּאי הוּא, וּתְהֵא מֻנַּחַת עַד שֶׁיָּבֹא אֵלִיָּהוּ. וְדַעַת הַתּוֹסָפוֹת דְּגַם בִּכְהַאי גַוְנָא נַמִּי יַחֲלֹקוּ, וְלֹא אָמְרוּ יְהֵא מֻנָּח אֶלָּא בִּשְׁנַיִם שֶׁהִפְקִידוּ כֵּיוָן שֶׁיָּדוּעַ לַנִּפְקָד שֶׁהוּא וַדַּאי שֶׁל אֶחָד מֵהֶן: {ג} דְּאִי מֵחַד אֵינוֹ נֶאֱמָן לוֹמַר שֶׁמָּכַר לְאַחֵר. וְיֵשׁ שִׁטּוֹת רַבּוֹת בָּזֶה: {ד} שֶׁאֵינוֹ נוֹגֵעַ בָּעֵדוּת כְּלָל. דְּאַף עַל גַּב דְּקִבֵּל דָּמִים אֵינוֹ עוֹמֵד אֶלָּא בְּמִי שֶׁפָּרַע. וְאֵינוֹ דּוֹמֶה לְעֵד הַצָּרִיךְ שְׁבוּעָה דְּאֵינוֹ עֵד, דְּהָתָם רַחֲמָנָא אָמַר עַל פִּי שְׁנַיִם יָקוּם דָּבָר, דְּמַשְׁמַע הַנֶּאֱמָנִים בְּדִבּוּרָם בִּלְבַד וְלֹא שֶׁצָּרִיךְ שְׁבוּעָה לְהַאֲמִין דְּבָרָיו. וּמִי שֶׁפָּרַע לֵיכָּא אֶלָּא דַּאֲנַן לָיְטִינַן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} וְכֵן פּוֹטֵר מִשְּׁבוּעָה לְאוֹתוֹ הַמְסַיְּעוֹ. הַמַּגִּיד מִשְׁנֶה: {ו} מֵאַחַר שֶׁזֶּה תָּפוּס וְעוֹמֵד וְזֶה תָּפוּס וְעוֹמֵד. גְּמָרָא. דְּהֵיכָא דְּלֵיכָּא חֲזָקָה לְשׁוּם אֶחָד וּתְפוּסִים בּוֹ יֵשׁ לְהַחְזִיק שְׁנֵיהֶם בְּשָׁוֶה: {ז} מַסְּקִינַן דְּאַף עַל גַּב דְּחָשׁוּד אַמָּמוֹנָא, לֹא חָשׁוּד אַשְּׁבוּעָתָא. וְהַיְנוּ טַעֲמָא מִשּׁוּם דִּשְׁבוּעָה חֲמוּרָה, שֶׁהָעוֹלָם נִזְדַּעְזַע עַל לֹא תִשָּׂא. תּוֹסָפוֹת: {ח} שֶׁאִם יוֹדֵעַ שֶׁחֲבֵרוֹ קְנָאָהּ לֵיהּ לֹא הָוֵי תָּפִיס, וּמִדְּתָפַס בָּהּ סָבַר שֶׁלּוֹ הִיא וְנִתְרַצָּה הַמּוֹכֵר, דְּהָא נָקַט זוּזֵי מִתַּרְוַיְהוּ, אֵימָא לֹא כוּ', דְּמִשּׁוּם שְׁבוּעָה לֹא יִפְרֹשׁ שֶׁהֲרֵי סָבוּר הוּא לוֹמַר אֱמֶת, קָא מַשְׁמַע לָן דַּאֲפִלּוּ הָכִי יִשָּׁבַע. תּוֹסָפוֹת: {ט} חֲבֵרִי יֵלֵךְ וְיִמְצָא אַחֶרֶת. רַשִׁ"י: {י} <B> וְזֶה כוּ'.</B> אֲפִלּוּ לְבֶן נַנָּס דְּמִשְׁנָה ה' פֶּרֶק ז' דִּשְׁבוּעוֹת דְּאָמַר כֵּיצַד אֵלּוּ וְאֵלּוּ בָּאִין לִידֵי שְׁבוּעַת שָׁוְא, הָתָם וַדַּאי אִיכָּא שְׁבוּעַת שָׁוְא, (דְּחַד וַדַּאי מִשְׁתָּבַע לְשִׁקְרָא), הָכָא אִיכָּא לְמֵימַר דְּלֵיכָּא שְׁבוּעַת שָׁוְא, דְּתַרְוַיְהוּ בַּהֲדֵי הֲדָדֵי אַגְּבוּהָ. גְּמָרָא. (וּלְעִנְיָן מִקָּח וּמִמְכָּר נַמִּי שֶׁמָּא לִשְׁנֵיהֶם נִתְרַצָּה. רַשִׁ"י): {יא} לְרַבָּנָן דְּבֶן נַנָּס אִצְטְרִיךְ, דְּלֹא חָיְשִׁינַן לִשְׁבוּעַת שָׁוְא. תּוֹסָפוֹת: {יב} וּכְלוֹמַר דִּלְשׁוֹן הַשְּׁבוּעָה אֵינָהּ אֶלָּא שֶׁאֵין לוֹ בָּהּ פָּחוֹת מֵחֶצְיָהּ וּכְלִישְׁנָא דְּמַתְנִיתִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} <B> וְיַחֲלֹקוּ.</B> הַיְנוּ לִדְּמֵי אִי הֲוָה פְּסֵדָא בַּחֲלֻקַּת הַטַּלִּית עַצְמָהּ. גְּמָרָא: {יד} <B> זֶה כוּ'.</B> הָא תוּ לָמָּה לִי, סַלְקָא דַּעְתָּךְ אֲמֵינָא הַאי דְּקָאָמַר חֶצְיָהּ שֶׁלִּי לֶהֱוֵי כְּמֵשִׁיב אֲבֵדָה וְלִפָּטֵר, קָא מַשְׁמַע לָן דְּהַאי אַעֲרוּמֵי קָא מַעֲרִים, סָבַר אִי אֲמֵינָא כֻּלָּהּ שֶׁלִּי בָּעִינָא אִשְׁתְּבוֹעֵי, אֵימָא הָכִי דְּאֶהֱוֵי כְּמֵשִׁיב אֲבֵדָה וְאִפָּטֵר. גְּמָרָא: {טו} <B> מִשְּׁלֹשָׁה.</B> וְאִם תֹּאמַר עַל הַחֵצִי שֶׁמּוֹדֶה חֲבֵרוֹ לָמָּה צָרִיךְ לִשָּׁבַע. וְיֵשׁ לוֹמַר שֶׁאִלּוּ לֹא הָיָה נִשְׁבַּע אֶלָּא עַל הָרְבִיעַ בִּלְבַד, שֶׁמָּא הָיָה מִתְכַּוֵּן לְאוֹתוֹ חֵלֶק שֶׁמּוֹדֶה לוֹ חֲבֵרוֹ כוּ'. נִמּוּקֵי יוֹסֵף:

{טז} דִּלְגוּפֵיהּ לֹא אִצְטְרִיךְ, אֶלָּא לְאַשְׁמוֹעִינַן דְּרוֹכֵב קָנֵי אֲפִלּוּ שֶׁלֹּא מַנְהִיג כוּ'. רַשִׁ"י: {יז} תֵּימָהּ אַמַּאי כֵּיוָן שֶׁהַקִּנְיָן דִּרְכִיבָה לְחוּד הָוֵי קִנְיָן לָמָּה יִגָּרַע כְּנֶגֶד אַחֵר, כֵּיוָן דְּקִנְיָן הוּא לִקְנֵי. וּבַגְּמָרָא לֹא אָמְרוּ כוּ' אֶלָּא לִשְׁמוּאֵל דִּסְבִירָא לֵיהּ רוֹכֵב לְחוּדָא לֹא קָנֵי, מוֹקֵי לְסֵיפָא דְּקָנוּ שְׁנֵיהֶם כְּשֶׁרוֹכֵב וּמַנְהִיג בְּרַגְלָיו (וּמַתְנִיתִין אַשְׁמְעִינַן דִּשְׁנַיִם רוֹכְבִים קָנוּ שְׁנֵיהֶם, וְלֹא אָמְרִינַן קַמָּא עִקָּר, אִי נַמִּי רוֹכֵב וּמַנְהִיג כְּאֶחָד קָנוּ שְׁנֵיהֶם. הָרֹא"שׁ). אֲבָל אִי הֲוָה סָבוּר דְּרוֹכֵב לְחוּדָא קָנֵי לֹא אִצְטְרִיךְ לְאוֹקֵי מַתְנִיתִין שֶׁמַּנְהִיג בְּרַגְלָיו גַּם כֵּן. תּוֹסְפוֹת יוֹם טוֹב: {יח} וְתֵימָהּ דַּאֲפִלּוּ הָכִי מַאי רְבוּתָא הִיא. וּבַגְּמָרָא אִי בְּמִקָּח וּמִמְכָּר צְרִיכָא לְמֵימַר. אֶלָּא לָאו בִּמְצִיאָה. שְׁמַע מִנַּהּ דְּהַמַּגְבִּיהַּ מְצִיאָה לַחֲבֵרוֹ אַף עַל פִּי שֶׁלֹּא עֲשָׂאוֹ שָׁלִיחַ לְכָךְ. קָנָה חֲבֵרוֹ. דְּאִי סַלְקָא דַּעְתָּךְ לֹא קָנָה חֲבֵרוֹ, תֵּעָשֶׂה זוֹ כְּמִי שֶׁמֻּנַּחַת עַל גַּבֵּי קַרְקַע וְלֹא יִקְנֶה לֹא זֶה וְלֹא זֶה, וְכָל הָרוֹצֶה יַחְטְפֶנָּה מִיָּדָם. רַשִׁ"י:

{יט} <B> רוֹכֵב.</B> אוֹרְחָא דְמִלְּתָא נָקַט, דְּאִי מְהַלֵּךְ וְרָאָה יִטְּלֶנּוּ בְּעַצְמוֹ: {כ} <B> נְטָלָהּ כוּ'.</B> מַסְּקִינַן בַּגְּמָרָא דַּאֲפִלּוּ כִּי לֹא בָּעֵי לְמִזְכֵּי לְנַפְשֵׁיהּ מָצֵי זָכֵי לַחֲבְרֵיהּ בְּמִגּוֹ דְּאִי בָּעֵי זָכֵי לְנַפְשֵׁיהּ כוּ'. וְאִלּוּ מַתְנִיתִין דִּתְנַן וְאָמַר אֲנִי זָכִיתִי בָהּ זָכָה מִשּׁוּם דְּאָמַר לֵיהּ תְּחִלָּה, פֵּרֵשׁ רַשִׁ"י הֲרֵי אֲנִי זָכִיתִי בָהּ דְּקָאָמַר, תְּחִלָּה קָאָמַר לֵיהּ. מִתְּחִלָּה הִגְבַּהְתִּיהָ לְצָרְכִּי וְלֹא לְצָרְכְּךָ, וּלְעוֹלָם הַמַּגְבִּיהַּ מְצִיאָה לַחֲבֵרוֹ קָנָה חֲבֵרוֹ. עַד כָּאן. וְרִיהֲטָא דִּשְׁמַעְתְּתָא דְּאִי מִתְּחִלָּה הָיְתָה דַעְתּוֹ לְצֹרֶךְ חֲבֵרוֹ תּוּ לֹא מָצֵי מִהֲדַר. אֲבָל הַטּוּר לֹא סְבִירָא לֵיהּ. וְהַסְּמַ"ע הִסְכִּים לִדְבָרָיו. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כא} <B> תְּחִלָּה.</B> מִשּׁוּם רֵישָׁא דְּמַיְרֵי דַּוְקָא שֶׁאָמַר תְּחִלָּה, הִלְכָּךְ דִּיֵּק בַּסֵּיפָא לְמִתְנֵי תְּחִלָּה, כְּלוֹמַר וְאִם מִשֶּׁנְּתָנָהּ לוֹ טָעַן אוֹתָהּ טַעֲנָה דְּרֵישָׁא. אֲבָל מִשּׁוּם סֵיפָא לֹא אִצְטְרִיךְ לְמֵידַק שֶׁיִּטְעֹן תְּחִלָּה, דַּאֲפִלּוּ אִם לֹא טָעַן תְּחִלָּה מִסְּתָמָא תְּחִלָּה קָאָמַר. דְּמִי מָצֵי לְמֵימַר אֲנִי זוֹכֶה בָּהּ עַכְשָׁיו, וְהַלֹּא אֵינוֹ בְּיָדוֹ. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ה} <B> וְהִשְׁאִילָהּ.</B> לַעֲשׂוֹת בָּהּ יְמֵי שְׂכִירוּתוֹ. רַשִׁ"י: {ו} <B> יְשַׁלֵּם לַשּׂוֹכֵר.</B> וּבַגְּמָרָא, וְנֵימָא מַשְׂכִּיר לַשּׂוֹכֵר, דַּל אַנְתְּ וְדַל שְׁבוּעֲתָךְ, וַאֲנָא מִשְׁתָּעִינָא דִינָא בַּהֲדֵי שׁוֹאֵל. וּמְשָׁנֵי, מִי סָבַרְתָּ שׂוֹכֵר בִּשְׁבוּעָה הוּא דְּקָנֵי, מִשְּׁעַת מִיתָה הוּא דְּקָנֵי, וּשְׁבוּעָה לְהָפִיס דַּעְתּוֹ שֶׁל בַּעַל הַבַּיִת. וּפֵרֵשׁ רַשִׁ"י דְּמַה שֶּׁהוּא נִשְׁבַּע לַמַּשְׂכִּיר, לְהָפִיס דַּעְתּוֹ שֶׁלֹּא יֹאמַר פָּשַׁעְתָּ בָּהּ. וּמַה שֶּׁכָּתַב הָרַ"ב בִּשְׁבוּעָה כוּ', הָעְתַּק מִנֻּסְחָא מֻטְעֵית בְּרַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} <B> כֵּיצַד כוּ'.</B> טַעֲמָא דְּרַבִּי יוֹסֵי מִשּׁוּם דְּחָשׁוּב שׂוֹכֵר כְּאִלּוּ הִשְׁאִילוֹ בִּשְׁלִיחוּת הַמַּשְׂכִּיר, הִלְכָּךְ דִּין הַמַּשְׂכִּיר עִם הַשּׁוֹאֵל. הָרֹא"שׁ:

{יח} וְנִמּוּקִי יוֹסֵף פֵּרֵשׁ דְּהָכָא אֲתָא לְאַשְׁמוֹעִינַן דְּאֵין מְקַבְּלִין כְּלָל, וַאֲפִלּוּ יָהֵיב לֵיהּ אַגְרָא, הוֹאִיל דְּכֻלָּהּ מִלְוָה:  {יט} וּפַרְכִינַן בַּגְּמָרָא דְּהָא אֵין שְׁלִיחוּת לְנָכְרִי, מַה אַתֶּם בְּנֵי בְרִית אַף שֶׁלּוּחֲכֶם בְּנֵי בְרִית, וּלְכָל הַתּוֹרָה כֻּלָּהּ יָלְפִינַן לָהּ. וּמַסְּקִינַן כְּגוֹן שֶׁנָּטַל הַנָּכְרִי בַּיָּד וְקִבְּלָן מִן הָרִאשׁוֹן. וְקָא מַשְׁמַע לָן דְּלָא תֵימָא נָכְרִי גּוּפֵיהּ כִּי עֲבֵיד, אַדַּעְתָּא דְיִשְׂרָאֵל קָא גְמִיר וְיָהֵיב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} <B> אֲבָל כוּ'.</B> כְּגוֹן נָכְרִי שֶׁלָּוָה מִיִּשְׂרָאֵל בְּרִבִּית וּבִקֵּשׁ לְהַחְזִירוֹ לוֹ. מְצָאוֹ יִשְׂרָאֵל אַחֵר וְאָמַר לוֹ תְּנֵם לִי וַאֲנִי אֶעֱלֶה לְךָ כְּדֶרֶךְ שֶׁאַתָּה מַעֲלֶה לוֹ, מֻתָּר. וְאִם הֶעֱמִידוֹ אֵצֶל יִשְׂרָאֵל, אָסוּר. גְּמָרָא. וּלְרַשִׁ"י וּלְרַבֵּנוּ חֲנַנְאֵל טַעֲמָא דְּאָסוּר מִשּׁוּם דְּיֵשׁ שְׁלִיחוּת לְנָכְרִי לְחֻמְרָא מִדְּרַבָּנָן. וְרַבֵּנוּ תַם מוֹקֵי הָכָא גַּם כֵּן בְּנָטַל וְנָתַן בַּיָּד:

{כא} <B> אֵין כוּ'.</B> אֵין נוֹתְנִין מָעוֹת לְיִשְׂרָאֵל חֲבֵרוֹ עַל מְנָת לִתֵּן לוֹ בָּהֶם פֵּרוֹת לְסַךְ כָּךְ לִזְמַן פְּלוֹנִי. דְּכֵיוָן שֶׁלֹּא יָצָא הַשַּׁעַר בְּכַמָּה נִמְכָּרִים, שֶׁמָּא יִתְיַקְּרוּ כְּשֶׁיֵּצֵא שַׁעַר שֶׁלָּהֶם. נִמְצָא שֶׁהָיָה מִשְׂתַּכֵּר זֶה בְהַמְתָּנַת מְעוֹתָיו, שֶׁיִּתֵּן לוֹ פֵּרוֹת כְּפִי מַה שֶּׁפָּסְקוּ בְּזוֹל. נִמּוּקֵי יוֹסֵף: {כב} <B> וְאַף עַל פִּי כוּ'.</B> בַּגְּמָרָא, דְּאָמַר לֵיהּ שְׁקֹל לְטִיבוּתִיךְ וְשַׁדְיָא אַחִיזְרֵי, מַאי אַהֲנֵית לִי, אִי הֲוָה לִי זוּזֵי בְּיָדִי הָוֵי מְזַבְּנִינָא בְהִינֵי וְשִׁילֵי. וְהַתּוֹסָפוֹת כָּתְבוּ אַף עַל גַּב דְּלֹא מָשַׁךְ אֵין זֶה רִבִּית, כֵּיוָן שֶׁאִם הָיָה בָּא מוֹכֵר לַחֲזֹר הֲוָה קָאֵי בְּמִי שֶׁפָּרַע, וְלָכֵן חֲשִׁיבֵי כְּנִתְיַקְּרוּ בִּרְשׁוּת לוֹקֵחַ. עַד כָּאן. נִרְאֶה שֶׁדַּעְתָּם שֶׁאֵין לְהוֹצִיא מִיְדֵי אִסּוּר רִבִּית אֶלָּא אִם כֵּן שֶׁנּוּכַל לוֹמַר שֶׁהָיוּ קְנוּיִים לְזֶה שֶׁנָּתַן הַמָּעוֹת וּבִרְשׁוּתֵיהּ אַיְקוּר. וְזֶה כְּדַעַת רַשִׁ"י שֶׁכָּתַב בְּרֵישׁ פִּרְקִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} רַשִׁ"י. וּכְלוֹמַר הוֹאִיל וּלְעִנְיָן מִי שֶׁפָּרַע מִיהָא קָנָה:

{כד} <B> וּפוֹסֵק וְכוּ'.</B> אַרֵישָׁא קָאֵי, יָצָא הַשַּׁעַר פּוֹסְקִין. וְאַף עַל פִּי שֶׁיֵּשׁ נֻסְחָא דְּלֹא גָּרְסִינַן לְאוֹתָהּ בָּבָא, כְּמוֹ שֶׁכָּתְבוּ תּוֹסָפוֹת, מִכָּל מָקוֹם מִדִּיּוּקָא שָׁמְעִינַן לַהּ, וַעֲלַהּ קָאֵי וּפוֹסֵק עִמּוֹ: {כה} וְאַף עַל פִּי שֶׁאֵין לוֹ, וְגַם מִתְּחִלָּה כְּשֶׁנָּתַן לוֹ הַמָּעוֹת לֹא הִתְנָה, לֹא הֲוָה כְּרֵשָׁא גַּבֵּי יַיִן אֵין לוֹ, דְּהָתָם זָקְפוּ עָלָיו בְּהַלְוָאָה, אֲבָל הָכָא בְּתוֹרַת מִקָּח נוֹתֵן לוֹ. וְכֵן כָּתְבוּ הַתּוֹסָפוֹת: {כו} <B> בְּיֹקֶר כוּ'.</B> שֶׁאִם הָיָה הַסְּאָה בִּשְׁנֵי סְלָעִים, וְהוּזְלוּ, שֶׁסְּאָה עוֹמֶדֶת בְּסֶלַע, הָיָה נוֹטֵל רַק סְאָה, אַף עַל פִּי שֶׁבְּשָׁעָה שֶׁנָּתַן הָיָה נוֹתֵן לוֹ מַה שֶּׁשָּׁוֶה שְׁנֵי סְלָעִים. וְאִם כְּשֶׁנָּתַן הָיָה סְאָה בְּסֶלַע וְאַחַר כָּךְ עָמַד סְאָה בִּשְׁנַיִם, לֹא הָיָה נוֹטֵל אֶלָּא חֲצִי סְאָה, שֶׁבְּשָׁעָה שֶׁנָּתַן לוֹ לֹא הָיָה נוֹתֵן אֶלָּא מַה שֶּׁשָּׁוֶה סֶלַע אֶחָד: {כז} בָּבָא הָרִאשׁוֹנָה נָקַט אֲבָל בְּזוֹל וְהוּקְרוּ הוּא רְבוּתָא טְפֵי דְּהַשְׁתָּא מְקַבֵּל חִטִּין מֻעָטִין מִמַּה שֶּׁנָּתַן. וְכֵן הוּא בְּנִמּוּקֵי יוֹסֵף:

{ו} <B> מִן הַתּוֹרָה.</B> מִשּׁוּם דְּאַיְרֵי מֵעִקָּרָא בְּפוֹעֲלִים שֶׁאוֹכְלִים עַל יְדֵי שֶׁפָּסְקוּ לָהֶם מְזוֹנוֹת אוֹ עַל יְדֵי מִנְהַג הַמְּדִינָה, לְהָכִי נָקַט וְאֵלּוּ אוֹכְלִין מִן הַתּוֹרָה בְּלֹא פְּסַק וּבְלֹא מִנְהָג. תּוֹסָפוֹת: {ז} כְּלוֹמַר שֶׁנִּגְמַר מְלֶאכֶת בִּשּׁוּלָן, שֶׁאֵינָן צְרִיכִין עוֹד לַקַּרְקַע: {ח} וּבְפוֹעֵל הַכָּתוּב מְדַבֵּר, נֶאֱמַר כָּאן בִּיאָה וְנֶאֱמָר לְהַלָּן [דְּבָרִים כד] לֹא תָבֹא עָלָיו הַשָּׁמֶשׁ. גְּמָרָא: {ט} כְּלוֹמַר וְאֵין שְׁעַת גְּמַר מְלַאכְתּוֹ לְמַעֲשֵׂר פוֹטַרְתָּן: {י} דְּהָא שׁוֹר לָאו דַּוְקָא, דְּכָל מִלֵּי אִיתְנְהוּ בַּחֲסִימָה כִּדְיָלְפִינַן בְּסוֹף פֶּרֶק ה' דְּבָבָא קַמָּא. גְּמָרָא: {יא}  אוֹכֵל בִּמְחֻבָּר. אַף נֶחְסָם אוֹכֵל בִּמְחֻבָּר. גְּמָרָא: {יב} דְּעִקַּר פְּעֻלָּה מִשּׁוּם תִּקּוּן גְּדוֹלִים הֵם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} <B> שֶׁאֵין אוֹכְלִין.</B> אֲפִלּוּ מֵהִלְכוֹת מְדִינָה. גְּמָרָא:

{יד} <B> בִּכְתֵפוֹ.</B> דְּאֵינוֹ מֵזִיז יָדָיו וְרַגְלָיו. רַשִׁ"י: {טו} <B> אוֹכֵל.</B> מַאי טַעֲמָא, כִּי תָבֹא בְּכֶרֶם רֵעֲךָ, בְּכָל מַה דַּעֲבֵיד. גְּמָרָא. וְנִרְאֶה לִי דְּמַפֵּיק לֵיהּ מִדְּלֹא כְתִיב כִּי תָבֹא לִבְצֹר בְּכֶרֶם רֵעֶךָ:

{טז} <B> בִּתְאֵנִים כוּ'.</B> שֶׁנֶּאֱמַר בְּכֶרֶם וְאָכַלְתָּ עֲנָבִים, עֲנָבִים וְלֹא תְאֵנִים. סִפְרִי (וִירוּשַׁלְמִי). וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} <B> לֹא כוּ'.</B> וַאֲפִלּוּ שְׂכָרוֹ לַעֲשׂוֹת בִּשְׁנֵיהֶן. טוּר: {יח} <B> לֹא אָמְרוּ.</B> שֶׁיֹּאכַל אֶלָּא בְּשָׁעָה שֶׁהוּא עוֹסֵק בַּמְּלָאכָה, וְלֹא שֶׁיֵּשֵׁב לוֹ וְיֹאכַל לוֹמַר הֲרֵי מָנַעְתִּי אֶת עַצְמִי עַד כָּאן וְלֹא נָטַלְתִּי לֶאֱכֹל וְעַכְשָׁיו אֵשֵׁב לִי וְאוֹכַל: {יט} דְבַגְּמָרָא מַקְּשׁוּ בָּהּ. כְּשֶׁהִיא פוֹרֶקֶת מֵהֵיכָא אָכְלָה (כָּל מַשָּׂאָהּ פּוֹרְקִין בְּבַת אַחַת וְהוֹלְכִין), אֵימָא עַד שֶׁתְּהֵא פוֹרֶקֶת. וּפֵרְשׁוּ הַתּוֹסָפוֹת דְּסַלְקָא דַּעְתָּךְ אֲמֵינָא שֶׁאֵינָהּ אוֹכֶלֶת אֶלָּא כְּשֶׁהִיא רוֹאָה בְּפָנֶיהָ כוּ', קָא מַשְׁמַע לָן. וְלֹא קָאֵי אַמִּפְּנֵי הֲשָׁבַת אֲבֵדָה, דְּמִן הַדִּין אוֹכֶלֶת. אֶלָּא מִלְּתָא בְּאַפֵּי נַפְשַׁהּ הִיא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ז - משנה ה

{כ} <B> מַתִּירִין.</B> וְהָא דִּכְתִיב כְּנַפְשְׁךָ שָׂבְעֶךָ מְפָרֵשׁ בַּגְּמָרָא דְּדָרְשִׁינַן לֵיהּ מַה נַּפְשְׁךָ אִם חָסַמְתָּ פָּטוּר, אַף פּוֹעֵל אִם חָסַמְתָּ פָּטוּר כוּ'. וּבַסִּפְרִי, וַחֲכָמִים אוֹמְרִים שָׂבְעֶךָ, מְלַמֵּד שֶׁאוֹכֵל יָתֵר עַל שְׂכָרוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כא} <B> קוֹצֵץ כוּ'.</B> וּמִדַּעְתָּן. טוּר. וּמִשּׁוּם הָכִי לְכֻלֵּי עָלְמָא גְּדוֹלִים מַמָּשׁ. וְלֹא פְּלִיגֵי אָמוֹרָאֵי כְּמוֹ בְּמִשְׁנָה ה' דְּפֶרֶק קַמָּא: {כב} <B> עַבְדּוֹ כוּ'.</B> כְּנַעֲנִים. הַמַּגִּיד: {כג} <B> שֶׁאֵין כוּ'.</B> שֶׁהַפּוֹעֵל מִשֶּׁל שָׁמַיִם הוּא אוֹכֵל (וְלֵית לְהוּ לְמָרַיְהוּ זְכִיָּה בַּאֲכִילָתוֹ עַד שֶׁיִּתְּנֵהוּ לְתוֹךְ פִּיו. רַשִׁ"י). גְּמָרָא:

{כד} <B> לֹא כוּ'.</B> דְּלֹא אָמְרִינַן שֶׁהָיוּ סְבוּרִים שֶׁפָּדָה אוֹתָם כְּדֵי שֶׁיֹּאכְלוּ. אֶלָּא מִדְּלֹא אִתְנוּ, אַדַּעְתָּא דְּהָכִי נַחְתֵי כְּאִלּוּ קָצַץ כֵּן עִמָּהֶם. נִמּוּקֵי יוֹסֵף: {כה} <B> אִם לֹא כוּ'.</B> לֹא שֶׁשְּׂכָרָן סְתָם לַעֲשׂוֹת לוֹ מְלָאכָה, דְּבִכְהַאי גַוְנָא אַמַּאי פּוֹדֶה וּמַאֲכִילָן, אֶלָּא כְּגוֹן שֶׁשְּׂכָרָן לִבְצֹר כֶּרֶם וְלֹא הוֹדִיעָן שֶׁהוּא נֶטַע רְבָעִי כוּ' וְהָוֵי כְמִקַּח טָעוּת. הַמַּגִּיד: {כו} <B> נִתְפָּרְסוּ כוּ'.</B> רֵישָׁא בִּמְחֻבָּר וְסֵיפָא בְּתָלוּשׁ: {כז} <B> אִם לֹא כוּ'.</B> כְּגוֹן שֶׁאָמַר לָהֶם תְּחִלָּה לִדְרֹס בְּעִגּוּלִין, סְבוּרִין שֶׁזּוֹ הִיא תְּחִלַּת דְּרִיסָתָן וַעֲדַיִן לֹא נִגְמְרָה מְלַאכְתָּן כוּ'. הַמַּגִּיד מִשְׁנֶה:

{כח} <B> אַרְבָּעָה שׁוֹמְרִין.</B> וְדִינֵיהֶן שְׁלֹשָׁה. גְּמָרָא. דְּשׂוֹכֵר כְּנוֹשֵׂא שָׂכָר: {כט} <B> נִשְׁבָּע.</B> פְּשִׁיעָה לֹא חָשֵׁיב, דִּבְהָא כֻּלְּהוּ חַיָּבִים. תּוֹסָפוֹת: {ל} <B> הַכֹּל.</B> מַה שֶּׁהֻזְכַּר בַּפָּרָשָׁה. אֲבָל מֵתָה מֵחֲמַת מְלָאכָה לֹא. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לא} וְאַף עַל גַּב דְּהַשְׁתָּא מִחַיַּב בִּכְפֵלָא מִשֶּׁטָּעַן טַעֲנַת גַּנָּב וּבָאוּ עֵדִים שֶׁהוּא גְּנָבוֹ כְּדִתְנַן פֶּרֶק ט' דְּבָבָא קַמָּא, וּפָרָשָׁה שְׁנִיָּה דִּמְחֻיָּב בִּגְנֵבָה לֹא אָתֵי לִידֵי כְּפֵלָא, שֶׁכְּשֶׁטָּעַן שֶׁנִּגְנַב מְחֻיָּב בְּקֶרֶן, אֲפִלּוּ הָכִי קַרְנָא בְּלֹא כְּפֵלָא עֲדִיפָא, דְּהָא שׁוֹאֵל חָמוּר וְאֵינוֹ מְשַׁלֵּם אֶלָּא קֶרֶן. גְּמָרָא: {לב} לַאֲפוֹקֵי שׁוֹאֵל דְּמַשְׁכַּחַת לַהּ דְּכָל הֲנָאָה שֶׁלּוֹ, בְּקַיְמָא בָּאֲגַם, דְּלֹא בָּעֵי מְזוֹנֵי, וּבְנָטַר מָתָא דְּלֹא בָּעֵי נְטִירוּתָא, אִי נַמִּי בִּשְׁאִילַת כֵּלִים. גְּמָרָא:

{א} <B>לִקְצֹר כוּ'.</B> דְּלִשְׁנֵיהֶם יֵשׁ קְפֵידָא. דְּאִם הַמְּקַבֵּל רוֹצֶה לַעֲקֹר בְּמָקוֹם שֶׁנָּהֲגוּ לִקְצֹר, יָכוֹל בַּעַל הַקַּרְקַע לוֹמַר בָּעִינָא דְּתִזְדַּבֵּל אַרְעָאִי מֵהַקַּשׁ שֶׁיִּשָּׁאֵר בּוֹ אַחַר הַקָּצִיר, וְאִם בַּעַל קַרְקַע רוֹצֶה לַעֲקֹר יֹאמַר הַמְּקַבֵּל לֹא מָצִינָא דְּאֶטְרַח, שֶׁהַקְּצִירָה נוֹחָה. וּכְשֶׁהַמִּנְהָג לַעֲקֹר וְהַמְּקַבֵּל רוֹצֶה לִקְצֹר, יֹאמַר בַּעַל הַקַּרְקַע בָּעִינָא דְּתִנַּקַּר אַרְעָאִי. וְאִם הַבַּעַל הַקַּרְקַע רוֹצֶה לְקָצְרוֹ, יֹאמַר הַמְּקַבֵּל בָּעִינָא לְקוֹשֵׁשׁ הַקַּשׁ לְבַהֲמוֹתַי. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> יַחֲרֹשׁ.</B> פְּשִׁיטָא. לֹא צְרִיכָא בְּאַתְרָא דְּלֹא מְנַכְּשֵׁי וְאָזִיל אִיהוּ וְנַכֵּישׁ, מַהוּ דְּתֵימָא מָצֵי אָמַר לֵיהּ הַאי דְּנַכִּישְׁנָא אַדַּעְתָּא דְּלֹא כָּרִיבְנָא לַהּ, קָא מַשְׁמַע לָן דְּאִיבָּעֵי לֵיהּ לְפֵרוּשֵׁי לֵיהּ. גְּמָרָא: {ג} <B> הַכֹּל כוּ'.</B> לַאֲתוֹיֵי הָא דְּתָנוּ רַבָּנָן מָקוֹם שֶׁנָּהֲגוּ לְהַשְׂכִּיר אִילָנוֹת עַל גַּבֵּי קַרְקַע, שֶׁיִּטֹּל הָאָרִיס חֶלְקוֹ בְּפֵרוֹת הָאִילָן, אַף עַל פִּי שֶׁאֵינוֹ טוֹרֵחַ בָּהֶן שֶׁאֵינָן צְרִיכִין לָעֲבוֹדָה, מַשְׂכִּירִין, כְּלוֹמַר מִסְּתָמָא הֵם מֻשְׂכָּרִין. וְקָא מַשְׁמַע לָן דְּאַף עַל גַּב דְּכֻלֵּי עָלְמָא נוֹתְנִים קַרְקַע לָאָרִיס עַל מְנָת שֶׁיִּקְּחוּ הֵן שְׁלִישׁ וְאָרִיס שְׁנֵי חֲלָקִים, וְאָזִיל אִיהוּ וְיָהֵיב בְּרִבְעָא, מַהוּ דְּתֵימָא דְּאָמַר לֵיהּ הַאי דְּבָצְרָא לָךְ אַדַּעְתָּא דְּלֹא יָהִיבְנָא לָךְ בָּאִילָנוֹת, קָא מַשְׁמַע לָן דְּאִיבָּעֵי לֵיהּ לְפֵרוּשֵׁי לֵיהּ. גְּמָרָא: {ד} <B> כְּשֵׁם כוּ'.</B> לֵיתָא אֶלָּא בְּקַבְּלָנוּת. דְּאִי בַּחֲכִירוּת, לֹא שַׁיְכָא חֲלֻקָּה. רַשִׁ"י: {ה} <B> בַּתֶּבֶן וּבַקַּשׁ.</B> אוֹתוֹ שֶׁנִּקְצָץ עִם הַשִּׁבֹּלֶת קָרוּי תֶּבֶן. וְהַנִּשְׁאָר בָּאָרֶץ קָרוּי קַשׁ [וְעַיֵּן סוֹף פי"ז דְּכֵלִים]. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ט - משנה ב

{ו} כְּלוֹמַר שֶׁנִּמְשָׁךְ בָּהּ מִן הַנָּהָר הַגָּדוֹל. דְּאִלּוּ יָבַשׁ הַנָּהָר הַגָּדוֹל הָוֵי מַכַּת מְדִינָה דִמְנַכִּין מֵחֲכוֹרוֹ כְּדִתְנַן בְּמִשְׁנָה ו'. גְּמָרָא: {ז} רַשִׁ"י. וּמִשּׁוּם דְּמַתְנִיתִין חֲכוֹרוֹ תְּנַן, נַקְטֵי לְהוּ נַמִּי בְּקִבְּלָהּ בַּחֲכִירוּת. אֲבָל בְּהֶדְיָא אִיתָא בַּגְּמָרָא דְּמַתְנִיתִין בֵּין בַּחֲכִירוּת בֵּין בְּקַבְּלָנוּת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} רַשִׁ"י. וְלִישְׁנָא דִּגְמָרָא הָכִי הִיא, מִדְּקָאָמַר זֶה, מִכְּלָל דְּקָאֵי בְּגַוֵּיהּ עָסְקִינַן, בֵּית הַשְּׁלָחִין לָמָּה לֵיהּ לְמֵימַר, דְּקָאָמַר לֵיהּ בֵּית הַשְּׁלָחִין כִּדְקַיְמָא הַשְׁתָּא, מַשְׁמַע דְּלִישְׁנָא יְתֵרָא דְּבֵית הַשְּׁלָחִין דָּיֵק. וּבָרִי"ף וּבָרַמְבַּ"ם דְּמִזֶּה דָּיֵק: 

~ פרק ט - משנה ג

{ט} אֲבָל בַּחֲכִירוּת לֹא שַׁיָּךְ. דְּמַאי שָׁמִין אִיכָּא. חֲכִירוּתוֹ יִתֵּן. רַשִׁ"י: {י} <B> שֶׁכָּךְ כוּ'.</B> דְּאִם לֹא כֵן פָּטוּר. וּכְלוֹמַר שֶׁכָּךְ רְגִילִים לִכְתֹּב, וּלְפִיכָךְ אַף עַל פִּי שֶׁלֹּא כָּתַב כְּמוֹ שֶׁכָּתַב דָּמֵי. דְּאִי דַּוְקָא שֶׁכָּתַב פְּשִׁיטָא שֶׁיֵּשׁ לוֹ לְקַיֵּם כְּמוֹ שֶׁהִתְנָה, דְּהָא לֹא גָּזִים. אֶלָּא אֲשַׁלֵּם בְּמֵיטְבָא הִתְנָה. תּוֹסָפוֹת: 

~ פרק ט - משנה ד

{יא} <B> וְלֹא כוּ'.</B> וּפֵרוּשׁ בְּמָקוֹם שֶׁאֵין שָׁם מִנְהָג יָדוּעַ שֶׁלֹּא לְנַכֵּשׁ. אֲבָל אִם נָהֲגוּ שֶׁלֹּא לְנַכֵּשׁ הַכֹּל הוֹלֵךְ אַחַר הַמִּנְהָג: {יב} <B> נוֹתֵן כוּ'.</B> וְהָא לֵיתָא אֶלָּא בַּחֲכִירוּתָא, דְּאִי בְּקַבְּלָנוּת פְּשִׁיטָא, מִי מָצֵי אָמַר לֵיהּ מַאי אִכְפַּת לְךָ. רַשִׁ"י: {יג} <B> וּמַעֲלָה כוּ'.</B> כְּלוֹמַר אֲפִלּוּ אָמַר לוֹ בָּאַחֲרוֹנָה אֲנִי חוֹרֵשׁ אוֹתָהּ, מָצֵי אָמַר לֵיהּ שֶׁמַּעֲלָה לְפָנַי עֲשָׂבִין עַל יְדֵי הַזֶּרַע שֶׁנּוֹפֵל מֵהֶן לָאָרֶץ וְיִצְמְחוּ לְשָׁנָה הַבָּאָה. וּלְזוֹ לֹא תוֹעִיל חֲרִישָׁה. גְּמָרָא: 

~ פרק ט - משנה ה

{יד} דְּאִי בַּחֲכִירוּתָא מַה לּוֹ לְבַעַל הַשָּׂדֶה אִם יֶחְדַּל לוֹ, סוֹף סוֹף חֲכוֹרוֹ הוּא נוֹתֵן לוֹ. רַשִׁ"י: {טו} <B> חַיָּב כוּ'.</B> מְפֹרָשׁ בַּבְּרַיְתָא שֶׁכָּךְ כּוֹתֵב לוֹ, אֲנָא אֵיקוּם וְאֶבוֹר וְאֶזְרַע וְכוּ' וְאוֹקִים כַּרְיָא קַדֳּמָךְ. וְנִרְאֶה לִי דִלְהָכִי לֹא תָּנֵי לֵיהּ תַּנָּא דִּידָן, לְאַשְׁמְעִינַן דַּאֲפִלּוּ לֹא כָּתַב, וּכְמוֹ שֶׁכָּתַבְתִּי לְעֵיל (אוֹת י): {טז} וּנְפִילָה פֵּרֵשׁ רַשִׁ"י זֶרַע שֶׁמְּפַזְּרִין בְּמַפֹּלֶת יָד: 

~ פרק ט - משנה ו

{יז} וְאִי בְּקַבְּלָנוּת מַאי מְנַכֶּה אִיכָּא, מַה שֶּׁיִּמְצְאוּ יַחֲלֹקוּ. רַשִׁ"י: {יח} <B> נִשְׁדְּפָה.</B> לְשׁוֹן שִׁדָּפוֹן: {יט} <B> מְנַכֶּה.</B> הַכֹּל לְפִי הַהֶפְסֵד שֶׁאֵרְעוֹ. הָרַמְבַּ"ם. וּרְצוֹנוֹ לוֹמַר דְּאָזְלִינַן בָּתַר שָׂדֶה שֶׁלּוֹ בֵּין לְהָקֵל בֵּין לְהַחְמִיר, אִם אֵין לִקּוּי הַשָּׂדוֹת כֻּלָּן שָׁווֹת. סְמַ"ע: 

~ פרק ט - משנה ז

{כ} <B> בַּעֲשָׂרָה כוּ'.</B> דְּאִי בְּקַבְּלָנוּת פְּשִׁיטָא. דְּכָל אֶחָד זוֹכֶה וּמִתְחַיֵּב בְּחֶלְקוֹ. רַשִׁ"י: 

~ פרק ט - משנה ח

{כא} דְּאִי בְּקַבְּלָנוּת הָא אָמְרִינַן תִּכְחוֹשׁ אַרְעָא וְלֹא תִּכְחוֹשׁ מָרָהּ. וַהֲרֵי הוּא נֶהֱנֶה בְּשִׁנּוּי שֶׁל זֶה. רַשִׁ"י. מִפְּנֵי שֶׁהַשֶּׁבַח יִהְיֶה מֵעַתָּה בְּכִיסוֹ וְהוּא טוֹב לוֹ מִשֶּׁיִּשָּׁאֵר אוֹתוֹ שֶׁבַח עַצְמוֹ בַּקַּרְקַע. נִמּוּקֵי יוֹסֵף. וְהַמַּגִּיד בְּשֵׁם הָרַמְבַּ"ן כָּתַב דְּאַדְּרַבָּה בְּקַבְּלָנוּת לֹא יְשַׁנֶּה, שֶׁיָּכוֹל לוֹמַר בְּזוֹ אֲנִי רוֹצֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} הָכִי נַמִּי פָּסַק עִמּוֹ חִטִּין שֶׁמָּא הוּא זָרְעָהּ חִטִּין בְּשָׁנָה שֶׁעָבְרָה, וְאִם יִזְרָעֶנָּה זֶה שְׂעֹרִין, נִמְצָא זֶה מְקַלְקְלָהּ. רַשִׁ"י: {כג} <B> לֹא כוּ'.</B> פֵּרֵשׁ רַשִׁ"י שֶׁהַקִּטְנִית מַכְּחֶשֶׁת הַקַּרְקַע יוֹתֵר מִן הַתְּבוּאָה. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כד} וְהָא לֹא מִתּוֹקְמָא אֶלָּא בַּחֲכִירוּת. דְּאִי בְּקַבְּלָנוּת, מַאי דְּבָעֵי לִזְרַע שֶׁהֲרֵי הַבְּעָלִים חוֹלְקִים בּוֹ. רַשִׁ"י. כְּלוֹמַר וְלִכְחֹשׁ אַרְעָא וְלֹא לִכְחֹשׁ מָרָהּ. נִמּוּקֵי יוֹסֵף. הָרַמְבַּ"ן חוֹלֵק כְּדִלְעֵיל (אוֹת כא). וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כה} רַשִׁ"י. וְזֶה לְשׁוֹן הַטּוּר, קוֹצֵץ קוֹרַת שִׁקְמָה בֵּין אִם יָרַד לְתוֹכָהּ בְּסוֹף שֶׁבַע מִשֶּׁנִּקְצְצָה בֵּין אִם יָרַד לְתוֹכָהּ בְּשָׁנָה שְׁלִישִׁית אוֹ רְבִיעִית, קוֹצְצָהּ כְּשֶׁיַּשְׁלִימוּ לָהּ שֶׁבַע. שֶׁהֲרֵי יִגְדַּל לִכְשֶׁיֵּצֵא עַד שֶׁיַּחְזְרוּ לִכְמוֹ שֶׁהָיְתָה כְּשֶׁיָּרַד לְתוֹכָהּ: 

~ פרק ט - משנה י

{כו} <B> מִן הַמִּנְיָן.</B> כְּלוֹמַר חַיָּב לִתֵּן לוֹ הַשְּׁבַע מֵאוֹת. וְאַף עַל פִּי שֶׁאֵין שָׁנָה שְׁבִיעִית שָׁוָה לִשְׁאָר שָׁנִים לַפֵּרוֹת, מִכָּל מְקוֹם שָׁבוּעַ קָאָמַר, וְלֹא תָלָה הַשָּׂכָר לְפִי הַשָּׁנִים כְּמוֹ בְּסֵיפָא. נִמּוּקֵי יוֹסֵף:

{כז} לְשׁוֹן רַשִׁ"י הֶעְתִּיק, דְּקָאֵי אַבְּרַיְתָא דְּסַיְמָא בִּשְׂכִיר לַיְלָה שֶׁגּוֹבֶה כָּל הַיּוֹם שֶׁנֶּאֱמַר בְּיוֹמוֹ תִּתֵּן שְׂכָרוֹ. כְּלוֹמַר וְסֵיפָא לֹא תָבֹא וְגוֹ'. וּפָתַח רַשִׁ"י בְּלֹא תָבֹא דְּסִיֵּם: {כח} דְּשָׁאנֵי שֶׁל יוֹם מִשֶּׁל לַיְלָה. מִשּׁוּם דְּאֵין לַיְלָה הוֹלֵךְ אַחַר הַיּוֹם. וְהַיּוֹם הוֹלֵךְ אַחַר הַלַּיְלָה כְּשֶׁשְּׂכִירָתוֹ נִמְשָׁךְ בַּלַּיְלָה. וְלֹא דָּמֵי לִשְׂכִיר יוֹם דְּאֶתְמוֹל שֶׁאֵינוֹ גּוֹבֶה אֶלָּא כָּל הַלַּיְלָה. וְהַךְ סֵיפָא אֲתָאן לְרַבִּי שִׁמְעוֹן וּסְבִירָא לֵיהּ הָכִי נַמִּי בִּשְׂכִיר שָׁעוֹת דְּרֵישָׁא. וְרֵישָׁא אֲתָאן כְּרַבִּי יְהוּדָה וְלִצְדָדִין קָתָנֵי (וּכְמוֹ שֶׁפֵּרֵשׁ הָרַ"ב) וּסְבִירָא לֵיהּ נַמִּי הָכִי בַּסֵּיפָא. וְצָרִיךְ עִיּוּן עַל הָרַ"ב וְהָרַמְבַּ"ם דַּהֲוָה לְהוּ לְפָרֵשׁ כָּל זֶה, וּדְהַסֵּיפָא אֵינָהּ הֲלָכָה, וְעוֹד דְּלֹא תַקְּשֵׁי רֵישָׁא לַסֵּיפָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כט} <B> בְּיוֹמוֹ.</B> כְּלוֹמַר מִשּׁוּם כֻּלָּהּ קְרָא דִּבְיוֹמוֹ, דְּאִית בֵּיהּ עֲשֵׂה דְּתִתֵּן, וְלֹא תַעֲשֶׂה דְלֹא תָבֹא וְגוֹ': {ל} וְנִרְאֶה לִי שֶׁהוּא מִלְּשׁוֹן וּמָחָה עַל כָּתֵף יַם כִּנֶּרֶת דְּפָרָשַׁת מַסְּעֵי (בְּמִדְבַּר לד), שֶׁעִנְיָנוֹ הִגִּיעַ: {לא} גְּמָרָא. וְלָאו דַּוְקָא פוֹעֲלָיו, דַּאֲפִלּוּ אֵין לוֹ אֶלָּא פוֹעֵל זֶה, דִּינָא הָכִי, אֶלָּא טָרוּד בַּעֲסָקָיו קָאָמְרִינַן. נִמּוּקֵי יוֹסֵף: {לב} פֵּרוּשׁ אִי מוֹדֶה מִקְצָת הוּא דְּחַיָּב שְׁבוּעָה מִדְּאוֹרַיְתָא. וַאֲפִלּוּ הֵיכָא דְּכוֹפֵר הַכֹּל טְפֵי הֲוָה לֵיהּ לְחִיּוּבֵי לְבַעַל הַבַּיִת שֶׁעָלָיו לְשַׁלֵּם, שֶׁהֲרֵי אִם מוֹדֶה מִקְצָת הוּא נִשְׁבָּע. וּלְפִי שֶׁהָיְתָה רְאוּיָה לִהְיוֹת עַל בַּעַל הַבַּיִת, קָאָמַר דְּשָׁקְלוּהָ. תּוֹסָפוֹת: {לג} <B> עֵדִים כוּ'.</B> בַּגְּמָרָא. וְהָא קָא תָּבַע לֵיהּ קַמָּן. וּמַסִּיק דְּיֵשׁ עֵדִים שֶׁתְּבָעוֹ כָּל זְמַנּוֹ, וּפֵרֵשׁ הָרֹא"שׁ כְּגוֹן שְׂכִיר יוֹם בְּסוֹף הַלַּיְלָה וּשְׂכִיר לַיְלָה בְּסוֹף הַיּוֹם. וּפָרֵיךְ וּלְעוֹלָם כוּ' וְכִי שָׂכִיר מַשְׁהֶה שְׂכָרוֹ כָּל כָּךְ זְמַן אָרֹךְ. וּמַסִּיק אֵינוֹ נִשְׁבָּע וְנוֹטֵל אֶלָּא לְמָחָר בִּלְבַד כְּנֶגֶד אוֹתוֹ יוֹם שֶׁל תְּבִיעָה, אֲבָל מִכָּאן וָאֵילָךְ הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה:

{לד} וְאוֹמֵר רִ"י דְּמִדְּרַבָּנָן הִיא, כִּדְאָמְרִינַן בַּגְּמָרָא. תּוֹסָפוֹת וְנִמּוּקֵי יוֹסֵף. וְהַסְּמַ"ע כָּתַב דְּהָרַמְבַּ"ם וְהַטּוּר סְבִירָא לְהוּ דְּמִן הַתּוֹרָה עוֹבֵר אֲפִלּוּ בַּחוּץ. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לה} דְּתַנְיָא מִמַּשְׁמַע שֶׁנֶּאֱמָר בַּחוּץ תַּעֲמֹד וְגוֹ' אֵינִי יוֹדֵעַ שֶׁהָאִישׁ אֲשֶׁר אַתָּה נוֹשֶׁה בּוֹ יוֹצִיא. אֶלָּא מַה תַּלְמוּד לוֹמַר וְהָאִישׁ, לְרַבּוֹת שְׁלִיחַ בֵּית דִּין: {לו} וְתֵימַהּ וְאִם לֹא יִמָּצֵא בְּיָדוֹ הָאֵיךְ יִכְפּוּהוּ לִפְרֹעַ מִלְוָתוֹ. וְרַבֵּנוּ תַם מְפָרֵשׁ דְּלֹא מַיְרֵי בְּמִשְׁכְּנוֹ לְפֵרָעוֹן אֶלָּא שֶׁרוֹצֶה לִהְיוֹת בָּטוּחַ בִּמְעוֹתָיו (רְצוֹנוֹ לוֹמַר כְּמוֹ שֶׁכָּתַבְנוּ אוֹת ל"ט וְדוֹ"ק. בֵּית דָּוִד). אֲבָל כְּשֶׁרוֹצֶה לִפָּרַע מִמֶּנּוּ, שְׁלִיחַ בֵּית דִּין נִכְנָס לְבֵיתוֹ. הָרֹא"שׁ: {לז} <B> וּמַנִּיחַ אֶחָד.</B> כְּלוֹמַר אַחַר שֶׁנָּטַל שְׁנֵיהֶם לְמַשְׁכּוֹן מַנִּיחַ אֶת שֶׁהוּא צָרִיךְ לוֹ. וַהֲדַר מְפָרֵשׁ אֵיזֶהוּ מַנִּיחַ. וְנָקַט לֵיהּ בִּלְשׁוֹן חֲזָרָה. וְהָרַ"ב הֶעְתִּיק וּמַחֲזִיר וְנִיחָא טְפֵי: {לח} <B> הַמַּחֲרֵשָׁה.</B> הִקְשׁוּ הַמְּפָרְשִׁים וְהַלֹּא כְּלִי שֶׁעוֹשִׂין בּוֹ אֹכֶל נֶפֶשׁ הוּא וְאֵין מְמַשְׁכְּנִין אוֹתָהּ. וְתֵרְצוּ, בְּמִשְׁכְּנוֹ בַּלַּיְלָה שֶׁאֵינוֹ זְמַן מְלֶאכֶת הַחֲרִישָׁה, וְלֹא אָסְרָה תּוֹרָה אֶלָּא שֶׁלֹּא לְמַשְׁכְּנוֹ בְּשָׁעָה שֶׁהוּא רָאוּי לַעֲשׂוֹת מְלָאכָה. וְנִמּוּקֵי יוֹסֵף תֵּרֵץ דְּמַיְרֵי בְּאִם עָבַר וְחָבַל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לט} <B> אֵינוֹ כוּ'.</B> אֶלָּא יִמְכְּרֶנּוּ וְיִגְבֶּה חוֹבוֹ. רַשִׁ"י. אֲבָל בְּחַיָּיו לְעוֹלָם הֵם בְּתוֹרַת חֲזָרָה כוּ', וּמְמַשְׁכְּנִין מֵעִקָּרָא שֶׁלֹּא תְהֵא שְׁבִיעִית מְשַׁמְּטְתוֹ וְלֹא יֵעָשֶׂה מִטַּלְטְלִין אֵצֶל בָּנָיו. גְּמָרָא. וּבְפַעַם אַחַת דַּי. וְחוֹזְרִין וּמְמַשְׁכְּנִין כְּדֵי שֶׁלֹּא יוּכַל לִכְפֹּר, וְעוֹד, כְּדֵי שֶׁיִּתְבַּיֵּשׁ וִימַהֵר לִפְרֹעַ. תּוֹסָפוֹת: {מ} דְּהָא וְאִם אִישׁ עָנִי הוּא לֹא תִשְׁכַּב בַּעֲבוֹטוֹ הוּא דִּכְתִיב: {מא} <B> וְלֹא כוּ'.</B> אֲפִלּוּ לְאָדָם אַחֵר הוּא אָסוּר לְמַשְׁכֵּן מִבֵּיתוֹ. אֶלָּא לַעֲבֹר עָלֶיהָ בִּשְׁנֵי לָאוִין. וְדַעַת הָרַמְבַּ"ם דִּבְאַלְמָנָה אֲפִלּוּ בַּחוּץ וּבְבֵית דִּין אָסוּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מב} <B> הַחוֹבֵל כוּ'.</B> דַּעַת הָרַמְבַּ"ם שֶׁאֲפִלּוּ בִּשְׁעַת הַלְוָאָה אָסוּר (וְכֵן דַּעְתּוֹ בְּאַלְמָנָה), וְאֵינוֹ מְחֻוָּר. נִמּוּקֵי יוֹסֵף: {מג} <B> רֵחַיִם.</B> פֵּרוּשׁ שֶׁל יָד שֶׁהוּא מִטַּלְטְלִין וְשַׁיָּךְ בֵּיהּ מַשְׁכּוֹן. אֲבָל בָּתֵּי רֵחַיִם שֶׁל מַיִם, הֲרֵי הֵם כְּקַרְקַע, וְלֹא שַׁיָּךְ בֵּיהּ מַשְׁכּוֹן אֶלָּא גּוֹבַיְנָא כִּשְׁאָר קַרְקָעוֹת דְּלֹא שַׁיָּךְ בְּהוּ מַשְׁכּוֹן אֶלָּא בְּמִטַּלְטְלִים, שֶׁאֵין מְמַשְׁכְּנִין לִקַּח קַרְקַע לְמַשְׁכּוֹן. טוּר: {מד} <B> מִשּׁוּם שְׁנֵי וְכוּ'.</B> פֵּרוּשׁ אֲפִלּוּ בְּהַתְרָאָה אַחַת אִם חָבַל בְּפַעַם אַחַת רֵחַיִם וָרֶכֶב לוֹקֶה שְׁתַּיִם. דְּאִי בִּשְׁתֵּי הַתְרָאוֹת אֲפִלּוּ רֵחַיִם וְרֵחַיִם חַיָּב [שְׁתַּיִם]. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מה} <B> שֶׁנֶּאֱמַר כוּ'.</B> כְּלוֹמַר וְכֵיוָן דִּקְרָא פֵּרֵשׁ תַּרְוַיְהוּ וְכָל אֶחָד שֵׁם בִּפְנֵי עַצְמוֹ הוּא, הַתְרָאוֹתֵיהֶן חֲלוּקוֹת וְלֹא הָוֵי לָאו שֶׁבִּכְלָלוּת: {מו} <B> וְלֹא כוּ' אָמְרוּ.</B> כְּלוֹמַר מַה שֶּׁאָמְרוּ חֲכָמִים דְּחַיָּב מִשּׁוּם שְׁנֵי כֵּלִים אַף עַל פִּי שֶׁהֵם לִמְלָאכָה אַחַת, לֹא כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} <B> וּבֶעָפָר.</B> שֶׁכֵּן הָיָה דֶּרֶךְ בִּנְיָנֵיהֶם לְחַבֵּר הָאֲבָנִים בְּעָפָר כְּעִנְיָן שֶׁנֶּאֱמַר [וַיִּקְרָא יד] וְעָפָר אַחֵר יִקַּח וְטָח אֶת הַבַּיִת: {ב} <B> לְהִשְׁתַּבֵּר.</B> אִם יֵשׁ אֲבָנִים שְׁבוּרוֹת. שֶׁאַבְנֵיהֶם שֶׁל לְבֵנִים הָיוּ. רַשִׁ"י: {ג} וְאוֹתָם שֶׁפִּנּוּ אָזְלוּ לְעָלְמָא, דְּלֹא מָצִינוּ לְשַׁיְלִינְהוּ. גְּמָרָא. וַאֲפִלּוּ הֵן בִּרְשׁוּת אֶחָד מֵהֶם לֹא הֲוָה אִידָךָ הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה, דְּלֹא חַשְׁבִינַן לֵיהּ מֻחְזָק, דְּשֻׁתָּפִין לֹא קַפְּדִי אַהֲדָדֵי לְמִקְנֵי רְשׁוּתָא דִּידֵיהּ, הִלְכָּךְ בִּרְשׁוּתָא דְּהַאי נַמִּי יַתְבֵי, דְּהָא לֹא קָפֵיד עֲלֵיהּ מִלְּאוֹשְׁלֵיהּ דּוּכְתָּא. בַּגְּמָרָא: {ד} לְדַעַת הַסּוֹבְרִים דְּהֵיכָא דְּלֹא הֲוָה לֵיהּ לְמֵידָע אֵין אוֹמְרִים מְחֻיָּב שְׁבוּעָה וְאֵינוֹ יָכוֹל לִשָּׁבַע מְשַׁלֵּם, פֵּרְשׁוּ הַתּוֹסָפוֹת דְּהָכָא רְגִילוּת הוּא שֶׁכָּל אֶחָד מַכִּיר מִקְצָת אֲבָנָיו. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} <B> וְעוֹלוֹת כוּ'.</B> לְפִי חֶשְׁבּוֹן שְׁלֵמִים, דְּהַלָּה נוֹטֵל שְׁלֵמִים כְּנֶגְדָּן, דְּמִדְּהָנֵי יָדַע טְפֵי לֹא יָדַע, תּוּ לֵית לֵיהּ. וּמַהֲנֵי לֵיהּ לְהָא שֶׁנּוֹטֵל אֵלּוּ שֶׁמַּכִּירָן, כְּגוֹן שֶׁהֵם גְּדוֹלוֹת וְטוֹבוֹת יוֹתֵר מִשְּׁאָר הַלְּבֵנִים. גְּמָרָא: 

~ פרק י - משנה ב

{ו} מוֹקֵי לָהּ הָכִי דִּבְאַחִין שֶׁחָלְקוּ אֵין עַל הַתַּחְתּוֹן לִבְנוֹת תִּקְרָה. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} <B> נִפְחֲתָה כוּ'.</B> תִּקְרַת הַבַּיִת שֶׁבְּנֵי עֲלִיָּה דּוֹרְסִים עָלֶיהָ. שֶׁבָּאתָה הַתַּקָּלָה מֵחֲמַת תִּקְרַת הַבַּיִת, וְזֶה שִׁעְבֵּד לוֹ הַבַּיִת כְּשֶׁאָמַר לֵיהּ שֶׁעַל גַּבֵּי בַּיִת זֶה. שֶׁכָּל זְמַן שֶׁנִּתְקַלְקְלָה עֲלִיָּתוֹ מֵחֲמַת בַּיִת, אוֹ יְתַקֵּן אוֹ יָדוּר עִמּוֹ בַּבַּיִת. אֲבָל אִם נָפְלוּ כָּתְלֵי הָעֲלִיָּה אֵינוֹ חַיָּב לְתַקֵּן, וְלֹא יָדוּר עִמּוֹ בַּבַּיִת, שֶׁאֵין זֶה הַקִּלְקוּל מֵחֲמַת הַבַּיִת. מַגִּיד: {ח} דְּאִי לָאו דְּשִׁעְבְּדוֹ לָמָּה לֵיהּ לְמֵימַר שֶׁעַל גַּבֵּי בַּיִת זֶה, וְהָא קָא חָזֵי לֵיהּ לַעֲלִיָּה, דַּעֲלִיָּה זוֹ קָאָמַר לֵיהּ. רַשִׁ"י: 

~ פרק י - משנה ג

{ט} <B> אָמַר כוּ'.</B> וְאִם תֹּאמַר לָמָּה הוּא צָרִיךְ לִבְנוֹת הַבַּיִת, יִכְפּוּהוּ לְבַעַל הַבַּיִת לִבְנוֹת כוּ'. תֵּרֵץ בַּיְרוּשַׁלְמִי, בְּשֶׁאֵינוֹ, כְּלוֹמַר כְּשֶׁהָלַךְ לִמְדִינַת הַיָּם, דְּבִכְהַאי גַוְנָא אִי אֶפְשָׁר לִכְפּוֹתוֹ, שֶׁאֵין בֵּית דִּין יוֹרְדִין לִנְכָסָיו שֶׁלֹּא בְפָנָיו. וְדַעַת הָרַשְׁבָּ"א דְּאֵין כּוֹפִין אֲפִלּוּ בְּיֶשְׁנוֹ, לְפִי שֶׁלֹּא נִשְׁתַּעְבְּדוּ נִכְסֵי בַּעַל הַבַּיִת לִכְשֶׁנָּכוֹף אוֹתוֹ לִבְנוֹתוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} וְלֹא דָּמֵי לְמַה שֶּׁכָּתַבְתִּי (אוֹת ו), דְּוַדַּאי כְּשֶׁנָּפְלוּ לְגַמְרֵי מֻטָּל עַל הַתַּחְתּוֹן, דְּאֵין בַּיִת בְּלֹא תִקְרָה. אֲבָל הָתָם מִסְּתָמָא אַדַּעְתָּא דְּהָכִי נִכְנְסוּ תְּחִלָּה, שֶׁכָּל אֶחָד יְתַקֵּן הַצָּרִיךְ לוֹ, וַהֲרֵי בַּעַל הַבַּיִת אֵינוֹ צָרִיךְ לְהַתִּקְרָה כֵּיוָן שֶׁיֵּשׁ גַּג עַל גַּבָּיו. סְמַ"ע. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} וְקָשֶׁה, דְּהָא נֶהֱנֶה הוּא שֶׁאֵינוֹ צָרִיךְ לַעֲלוֹת. וְנִרְאֶה דְגָרַס וּמְקָרֶה אֶת הָעֲלִיָּה וְיוֹשֵׁב בְּתוֹכָהּ, פֵּרוּשׁ בַּעֲלִיָּה, וְלֹא יָנִיחַ לְבַעַל הַבַּיִת לִכְנוֹס בְּתוֹכוֹ עַד שֶׁיִּתֵּן לוֹ יְצִיאוֹתָיו. תּוֹסָפוֹת: {יב} וְאִם תֹּאמַר אַמַּאי אֵינוֹ חָסֵר, וְהַלֹּא מַשְׁחִיר לוֹ כְּתָלָיו בְּעָשָׁן כְּשֶׁדָּר שָׁם. לְפִיכָךְ כָּתַב הָרַשְׁבָּ"א זַ"ל דְּמֵהָא שָׁמְעִינַן דְּכָל שֶׁזֶּה אֵינוֹ נֶהֱנֶה אַף עַל גַּב דְּאִידָךְ חָסֵר, פָּטוּר, שֶׁאֵינוֹ אֶלָּא כְּמוֹנְעוֹ לִכָּנֵס לְבֵיתוֹ שֶׁהוּא פָטוּר. וְעוֹד רְאָיָה מִמִּשְׁנָה ג' פֶּרֶק ט' דְּדַוְקָא מִשּׁוּם דְּכָךְ כּוֹתֵב לוֹ. עַיֵּן שָׁם: 

~ פרק י - משנה ד

{יג} וְכֵן כָּתַב רַשִׁ"י, בְּשֻׁתָּפִין. וְכָתַב הָרֹא"שׁ דְּהָכָא עַל הַתַּחְתּוֹן לְתַקֵּן הַתִּקְרָה, לְפִי שֶׁהוּא צָרִיךְ לָהּ, כְּדֵי שֶׁלֹּא יִתְקַלְקֵל בֵּית הַבַּד שֶׁלּוֹ. מַה שֶּׁאֵין כֵּן בְּבַיִת שֶׁל שֻׁתָּפִין שֶׁאֵינוֹ צָרִיךְ לַתִּקְרָה, כִּי דַּי לוֹ בַּגַּג. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} <B> וְזוֹרֵעַ כוּ'.</B> וּמַשְׁמַע דִּלְגַמְרֵי יוֹרֵד וְזוֹרֵעַ לְמַטָּה כְּשִׁעוּר כָּל גִּנָּתוֹ (דְּאֵין זוֹרְעִין לַחֲצָאִין), וְאַף עַל פִּי שֶׁכָּל שְׁאָר בֵּית הַבַּד מְחֻפֶּה בְּקֵרוּי, אֶפְשָׁר לִזְרֹעַ זְרָעִים שֶׁאֵינָן צְרִיכִים לַגְּשָׁמִים כְּעֵין חַזֶּרֶת. בֵּית יוֹסֵף: {טו} וְעַל שֵׁם שֶׁהִיא כְּפוּפָה בְּעִגּוּל נִקְרֵאת כִּפִּין. לְשׁוֹן הֲלָכֹף כְּאַגְמוֹן [יְשַׁעְיָה נח]: {טז} וְאִם כֵּן מַיְרֵי דַּוְקָא שֶׁבְּנָאוֹ כָּרָאוּי, אֲבָל שֶׁלֹּא כָּרָאוּי חַיָּב כוּ'.וְכִי תְּנַן פָּטוּר אֲפִלּוּ אַנְּזָקִין דִּלְאַחַר נְפִילָה בְּשֶׁלֹּא הָיָה לוֹ שָׁהוּת לְסַלְּקָם, אִי נַמִּי בְּהִפְקִיר נְזָקָיו דְּהוֹאִיל דִּלְאַחַר נְפִילַת אֹנֶס הוּא פָטוּר, וְכֵן חַיָּב דְּסֵיפָא נַמִּי בְּכָל עִנְיָן הוּא. נִמּוּקֵי יוֹסֵף. וְכָתְבוּ הַתּוֹסָפוֹת דְּבִשְׁעַת נְפִילָה דּוֹמֶה לְאֵשׁ. וְהָרֹא"שׁ כָּתַב דְּגַם בִּשְׁעַת נְפִילָה דּוֹמֶה לַבּוֹר: 

~ פרק י - משנה ה

{יז} <B> אֵין כוּ'.</B> אַף עַל גַּב דְּמַפְקִיר נְזָקָיו לְאַחַר נְפִילַת אֹנֶס פָּטוּר כְּדִלְעֵיל, הָכָא אִשְׁתַּמּוֹטֵי קָא מִשְׁתַּמִּיט וְאֵינוֹ מַפְקִירוֹ, כְּדִלְקַמָּן. תּוֹסָפוֹת: {יח} <B> הֵילָךְ כוּ'.</B> מִכְּלָל דְּפַנִּינְהוּ. הָא לֹא פַנִּינְהוּ לֹא אָמְרִינַן קָנְתָה לוֹ חֲצֵרוֹ כִּדְאָמַר בְּמִשְׁנָה ד' פֶּרֶק קַמָּא, דְּלֹא מְכַוֵּן לְאַקְנוֹיֵי לֵיהּ אֶלָּא אִשְׁתַּמּוֹטֵי דְּקָא מִשְׁתַּמִּיט לֵיהּ. גְּמָרָא: {יט} <B> בְּתֶבֶן כוּ'.</B> אֲבָל מִידֵי דַּאֲכִילָה כְּגוֹן חִטִּין וּשְׂעֹרִין, שׁוֹמְעִין לוֹ. מָרְדְּכַי: {כ} <B> שְׂכָרְךָ כוּ'.</B> וְלֹא תְּנַן הָכָא וִיצִיאוֹתֶיךָ, דְּהָכָא מִיָּד שֶׁקִּבֵּל עָלָיו קָנְתָה לוֹ רְשׁוּתוֹ, כְּמוֹ שֶׁכָּתַב הַטּוּר סִימָן של"ו. וְכָתַב הַסְּמַ"ע כוּ' וְלִי נִרְאֶה דְּהַיְנוּ טַעֲמָא מִשּׁוּם דְּאִיכָּא בַּל תָּלִין, וְאִי אֶפְשָׁר דְּלֹא גָּמַר וּמַקְנֵי, קָרוֹב הוּא לַעֲבֹר עָלָיו. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כא} <B> שׁוֹרִין.</B> יֵשׁ מְקוֹמוֹת שֶׁשּׁוֹרִין אוֹתוֹ יָמִים רַבִּים צָבוּר בְּמָקוֹם אֶחָד. רַשִׁ"י: {כב} <B> לְבֵנִים.</B> לְפִי שֶׁמַּשְׁהֶה אוֹתָן לְיַבְּשָׁן כוּ'. רַשִׁ"י: 

~ פרק י - משנה ו

{כג} וְכֵן פֵּרֵשׁ רַשִׁ"י. וְטַעְמָא, דְּעַד כָּאן לֹא פְּלִיגֵי אֶלָּא בְּנוֹפוֹ, אֲבָל בְּעִקָּרוֹ כֻּלֵּי עָלְמָא לֹא פְּלִיגֵי דְּעֶלְיוֹן הָוֵי, כִּדְאִיתָא בַּגְּמָרָא. וּמִכָּל מָקוֹם נִרְאֶה לִי שֶׁהַלָּשׁוֹן צָרִיךְ לִהְיוֹת הוּא מֻנָּח:
