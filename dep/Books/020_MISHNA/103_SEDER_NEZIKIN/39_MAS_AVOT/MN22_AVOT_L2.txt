<b>מֹשֶׁה קִבֵּל תּוֹרָה מִסִּינַי.</b> אוֹמֵר אֲנִי, לְפִי שֶׁמַּסֶּכֶת זוֹ אֵינָהּ מְיֻסֶּדֶת עַל פֵּרוּשׁ מִצְוָה מִמִּצְוֹת הַתּוֹרָה כִּשְׁאָר מַסֶּכְתּוֹת שֶׁבַּמִּשְׁנָה, אֶלָּא כֻּלָּהּ מוּסָרִים וּמִדּוֹת, וְחַכְמֵי אֻמּוֹת הָעוֹלָם גַּם כֵּן חִבְּרוּ סְפָרִים כְּמוֹ שֶׁבָּדוּ מִלִּבָּם בְּדַרְכֵי הַמּוּסָר כֵּיצַד יִתְנַהֵג הָאָדָם עִם חֲבֵרוֹ, לְפִיכָךְ הִתְחִיל הַתַּנָּא בְּמַסֶּכֶת זוֹ מֹשֶׁה קִבֵּל תּוֹרָה מִסִּינַי, לוֹמַר לְךָ שֶׁהַמִּדּוֹת וְהַמּוּסָרִים שֶׁבְּזוֹ הַמַּסֶּכְתָּא לֹא בָּדוּ אוֹתָם חַכְמֵי הַמִּשְׁנָה מִלִּבָּם, אֶלָּא אַף אֵלּוּ נֶאֶמְרוּ בְּסִינַי: <B>מִסִּינַי.</B> מִמִּי שֶׁנִּגְלָה בְּסִינַי {א}: <B>לִזְקֵנִים.</B> שֶׁהֶאֱרִיכוּ יָמִים אַחֲרֵי יְהוֹשֻׁעַ. וְאוֹתָם הַזְּקֵנִים לִזְקֵנִים אֲחֵרִים עַד שֶׁהִגִּיעוּ לִתְחִלָּתָם שֶׁל נְבִיאִים שֶׁהֵן עֵלִי הַכֹּהֵן וּשְׁמוּאֵל הָרָמָתִי: <B>לְאַנְשֵׁי כְנֶסֶת הַגְּדוֹלָה.</B> מֵאָה וְעֶשְׂרִים זְקֵנִים הָיוּ. זְרֻבָּבֶל שְׂרָיָה רְעֵלָיָה מָרְדְּכַי בִּלְשָׁן, שֶׁהָיוּ בִּימֵי עֶזְרָא כְשֶׁעָלוּ מִן הַגּוֹלָה בְּבַיִת שֵׁנִי. וּמִכְּלָלָם, חַגַּי זְכַרְיָה וּמַלְאָכִי וּנְחֶמְיָה בֶּן חֲכַלְיָה וְחַבְרֵיהֶם. וְנִקְרְאוּ כְּנֶסֶת הַגְּדוֹלָה לְפִי שֶׁהֶחֱזִירוּ הָעֲטָרָה לְיָשְׁנָהּ [יוֹמָא סט:] שֶׁמֹּשֶׁה אָמַר (דְּבָרִים י) הָאֵל הַגָּדוֹל הַגִּבּוֹר וְהַנּוֹרָא, בָּאוּ יִרְמְיָה וְדָנִיֵּאל וְלֹא אָמְרוּ גִּבּוֹר וְנוֹרָא, וְהֵם הֶחֱזִירוּם כְּבַתְּחִלָּה, לְפִי שֶׁאָמְרוּ הֵן הֵן גְּבוּרוֹתָיו הֵן הֵן נוֹרְאוֹתָיו, שֶׁאִלְמָלֵא כֵן הֵיאַךְ אֻמָּה כָּזוֹ יְכוֹלָה לְהִתְקַיֵּם בִּפְנֵי כַּמָּה אֻמּוֹת: <B>הֵם אָמְרוּ שְׁלֹשָׁה דְבָרִים.</B> הַרְבֵּה דְּבָרִים אָמְרוּ, אֶלָּא שְׁלֹשָׁה דְּבָרִים הַלָּלוּ אָמְרוּ [הֵן] שֶׁיֵּשׁ בָּהֶם קִיּוּם הַתּוֹרָה: <B>הֱווּ מְתוּנִים בַּדִּין.</B> שֶׁאִם בָּא דִּין לְפָנֶיךָ פַּעַם וּשְׁתַּיִם וְשָׁלֹשׁ, לֹא תֹאמַר דִּין זֶה כְּבָר בָּא לְפָנַי וְשָׁנִיתִי וְשִׁלַּשְׁתִּי בּוֹ, אֶלָּא הֱווּ מְתוּנִים, כְּלוֹמַר מַמְתִּינִים קֹדֶם שֶׁתִּפְסְקוּ הַדִּין: <B>וְהַעֲמִידוּ תַלְמִידִים הַרְבֵּה.</B> לַאֲפוֹקֵי מֵרַבָּן גַּמְלִיאֵל דְּאָמַר [בְּרָכוֹת כח.] כָּל תַּלְמִיד שֶׁאֵין תּוֹכוֹ כְּבָרוֹ אַל יִכָּנֵס לְבֵית הַמִּדְרָשׁ, קָא מַשְׁמַע לָן שֶׁמְּלַמְּדִין תּוֹרָה לְכָל אָדָם וְאֵין צָרִיךְ לִבְדֹּק אַחֲרָיו, וּבִלְבַד שֶׁלֹּא יִהְיֶה יָדוּעַ מֵעִנְיָנוֹ שֶׁמַּעֲשָׂיו מְקֻלְקָלִים וְסָאנֵי שׁוּמְעָנֵיהּ. אִי נַמִּי אַשְׁמוֹעִינַן [יְבָמוֹת סב:] שֶׁאִם הֶעֱמִיד תַּלְמִידִים בְּבַחֲרוּתוֹ יַעֲמִיד תַּלְמִידִים בְּזִקְנוּתוֹ, דִּכְתִיב (קֹהֶלֶת יא) בַּבֹּקֶר זְרַע זַרְעֶךָ וְלָעֶרֶב אַל תַּנַּח יָדֶךָ: <B>וַעֲשׂוּ סְיָג לַתּוֹרָה.</B> גָּדֵר שֶׁלֹּא יָבֹא לִגַּע בְּאִסּוּר תּוֹרָה, כְּגוֹן שְׁנִיּוֹת לָעֲרָיוֹת, וּשְׁבוּת לְשַׁבָּת. דִּכְתִיב (וַיִּקְרָא יח) וּשְׁמַרְתֶּם אֶת מִשְׁמַרְתִּי, עֲשׂוּ מִשְׁמֶרֶת לְמִשְׁמַרְתִּי:

<B>מִשְּׁיָרֵי.</B> מִשִּׁיּוּרֵי. שֶׁלְּאַחַר שֶׁמֵּתוּ כֻּלָּן נִשְׁתַּיְּרָה הַקַּבָּלָה בְּיָדוֹ. וְהוּא הָיָה כֹּהֵן גָּדוֹל אַחַר עֶזְרָא: <B>הוּא הָיָה אוֹמֵר.</B> כְּלוֹמַר כָּךְ הָיָה מְרַגְּלָא בְּפוּמֵיהּ תָּמִיד. וְכֵן כָּל רַבִּי פְּלוֹנִי אוֹמֵר, הוּא הָיָה אוֹמֵר, שֶׁבְּמַסֶּכְתָּא זוֹ, פֵּרוּשָׁן הָיָה רָגִיל לוֹמַר כֵּן תָּמִיד: <B>הָעוֹלָם עוֹמֵד.</B> לֹא נִבְרָא הָעוֹלָם אֶלָּא בִּשְׁבִיל שְׁלֹשָׁה דְּבָרִים הַלָּלוּ: <B>עַל הַתּוֹרָה.</B> שֶׁאִלְמָלֵי לֹא קִבְּלוּ {ה} יִשְׂרָאֵל אֶת הַתּוֹרָה לֹא נִבְרְאוּ שָׁמַיִם וָאָרֶץ, דִּכְתִיב (יִרְמְיָה לג) אִם לֹא בְרִיתִי יוֹמָם וָלָיְלָה חֻקּוֹת שָׁמַיִם וָאָרֶץ לֹא שָׂמְתִּי [שַׁבָּת פח.]: <B>וְעַל הָעֲבוֹדָה.</B> עֲבוֹדַת הַקָּרְבָּנוֹת. שֶׁכָּךְ שָׁנִינוּ בְּמַסֶּכֶת תַּעֲנִית [כז:], שֶׁאִלְמָלֵא מַעֲמָדוֹת לֹא נִתְקַיְּמוּ שָׁמַיִם וָאָרֶץ {ו}. וּמָצִינוּ שֶׁבִּשְׁבִיל הַקָּרְבָּנוֹת שֶׁהִקְרִיב נֹחַ נִשְׁבַּע [הַקָּדוֹשׁ בָּרוּךְ הוּא] שֶׁלֹּא יָבִיא מַבּוּל לָעוֹלָם. הֲרֵי שֶׁהָעוֹלָם עוֹמֵד עַל הַקָּרְבָּנוֹת: <B>וְעַל גְּמִילוּת חֲסָדִים.</B> דִּכְתִיב (תְּהִלִּים פט) עוֹלָם חֶסֶד יִבָּנֶה. וּגְמִילוּת חֲסָדִים, הוּא לְשַׂמֵּחַ חֲתָנִים, וּלְנַחֵם אֲבֵלִים, וּלְבַקֵּר חוֹלִים, וְלִקְבֹּר מֵתִים, וְכַיּוֹצֵא בָזֶה:

<B>פְּרָס.</B> לְשׁוֹן עֶרֶךְ. בְּעֶרְכֶּךָ, מְתַרְגְּמִינַן פּוּרְסָנֵיהּ. וְהוּא מַה שֶּׁאָדָם נוֹתֵן לְמִי שֶׁעָבַד אוֹתוֹ וְהוּא אֵינוֹ חַיָּב בַּדִּין לִתֵּן לוֹ כְּלוּם {ח}, כְּמוֹ מַה שֶּׁיִּתֵּן אָדָם לִבְנוֹ קָטָן אוֹ לְאִשְׁתּוֹ אוֹ לְעַבְדּוֹ מִפְּנֵי קוֹרַת רוּחַ שֶׁעָשׂוּ לוֹ. וַאֲפִלּוּ לְתִקְוַת פְּרָס כָּזֶה לֹא יַעֲבֹד אָדָם אֶת בּוֹרְאוֹ, אֶלָּא מֵאַהֲבָה בִּלְבַד: <B>וִיהִי מוֹרָא שָׁמַיִם עֲלֵיכֶם.</B> אַף עַל פִּי שֶׁאַתָּה עוֹבֵד מֵאַהֲבָה עֲבֹד גַּם כֵּן מִיִּרְאָה. שֶׁהָעוֹבֵד מֵאַהֲבָה זָרִיז בְּמִצְוֹת עֲשֵׂה, וְהָעוֹבֵד מִיִּרְאָה נִזְהָר מִמִּצְוֹת לֹא תַעֲשֶׂה, וְנִמְצֵאת דַּעֲבוֹדָתוֹ שְׁלֵמָה. וְכֵן אָמְרוּ חֲזַ"ל, עֲבֹד מֵאַהֲבָה עֲבֹד מִיִּרְאָה. עֲבֹד מֵאַהֲבָה, שֶׁאִם בָּאתָ לִשְׂנֹא דַּע שֶׁאַתָּה אוֹהֵב וְאֵין אוֹהֵב שׂוֹנֵא. עֲבֹד מִיִּרְאָה, שֶׁאִם בָּאתָ לִבְעֹט דַּע שֶׁאַתָּה יָרֵא וְאֵין יָרֵא בּוֹעֵט: 

~ פרק א - משנה ד
<B>יוֹסֵי בֶן יוֹעֶזֶר אִישׁ צְרֵדָה וְיוֹסֵי בֶן יוֹחָנָן אִישׁ יְרוּשָׁלַיִם.</B> כָּל הַתַּנָּאִים הַנִּזְכָּרִים בְּפֶרֶק זֶה זוּג זוּג פְּלוֹנִי וּפְלוֹנִי קִבְּלוּ מִפְּלוֹנִי וּפְלוֹנִי, הָרִאשׁוֹן מֵהֶם נָשִׂיא וְהַשֵּׁנִי אַב בֵּית דִּין: <B>יְהִי בֵיתְךָ בֵית וַעַד לַחֲכָמִים.</B> כְּשֶׁיִּרְצוּ הַחֲכָמִים לְהִתְקַבֵּץ וּלְהִוָּעֵד, יִהְיֶה בֵּיתְךָ מוּכָן לְדָבָר זֶה, שֶׁיִּהְיוּ רְגִילִים לוֹמַר נִתְקַבֵּץ בַּבַּיִת שֶׁל פְּלוֹנִי. שֶׁאִי אֶפְשָׁר שֶׁלֹּא תִּלְמֹד מֵהֶם אֵיזֶה דְּבַר חָכְמָה. מָשְׁלוּ מָשָׁל לְמַה הַדָּבָר דּוֹמֶה, לְנִכְנָס לַחֲנוּתוֹ שֶׁל בַּשָּׂם, אַף עַל פִּי שֶׁלֹּא לָקַח כְּלוּם, מִכָּל מָקוֹם רֵיחַ טוֹב קָלַט וְהוֹצִיא עִמּוֹ: <B>וֶהֱוֵי מִתְאַבֵּק בַּעֲפַר רַגְלֵיהֶם.</B> כְּלוֹמַר שֶׁתֵּלֵךְ אַחֲרֵיהֶם. שֶׁהַמְהַלֵּךְ מַעֲלֶה אָבָק בְּרַגְלָיו, וְהַהוֹלֵךְ אַחֲרָיו מִתְמַלֵּא מֵאָבָק שֶׁהוּא מַעֲלֶה בְּרַגְלָיו. פֵּרוּשׁ אַחֵר, שֶׁתֵּשֵׁב לְרַגְלֵיהֶם עַל הָאָרֶץ. שֶׁכָּךְ הָיוּ נוֹהֲגִים, שֶׁהָרַב יוֹשֵׁב עַל הַסַּפְסָל וְהַתַּלְמִידִים יוֹשְׁבִים לְרַגְלָיו עַל גַּבֵּי קַרְקַע: <B>וֶהֱוֵי שׁוֹתֶה בְצָמָא אֶת דִּבְרֵיהֶם.</B> כְּאָדָם צָמֵא שֶׁשּׁוֹתֶה לְתֵאָבוֹן, וְלֹא כְּאָדָם שָׂבֵעַ שֶׁקָּץ בִּמְזוֹנוֹ שֶׁמּוֹאֵס אֲפִלּוּ בִּדְבָרִים הָעֲרֵבִים וְטוֹבִים: 

~ פרק א - משנה ה
<B>יְהִי בֵיתְךָ פָתוּחַ לִרְוָחָה.</B> כְּבֵיתוֹ שֶׁל אַבְרָהָם אָבִינוּ עָלָיו הַשָּׁלוֹם שֶׁהָיָה פָּתוּחַ לְאַרְבַּע רוּחוֹת הָעוֹלָם, כְּדֵי שֶׁלֹּא יִצְטָרְכוּ הָאוֹרְחִים לְהַקִּיף לִמְצֹא הַפֶּתַח: <B>וְיִהְיוּ עֲנִיִּים בְּנֵי בֵיתְךָ.</B> וְלֹא יִקְנֶה עֲבָדִים לְשַׁמְּשׁוֹ. מוּטָב שֶׁיֵּהָנוּ יִשְׂרָאֵל מִנְּכָסָיו וְלֹא זֶרַע כְּנַעַן הָאָרוּר: <B>בְּאִשְׁתּוֹ אָמְרוּ.</B> מִדְּקָאָמַר עִם הָאִשָּׁה וְלֹא אָמַר עִם אִשָּׁה, לָמַדְנוּ שֶׁבְּאִשְׁתּוֹ אָמְרוּ. אִית דִּמְפָרְשֵׁי, בְּאִשְׁתּוֹ נִדָּה בִּלְבַד, שֶׁלֹּא יָבֹא לִידֵי הֶרְגֵּל עֲבֵרָה. וּמִדִּבְרֵי הַמִּשְׁנָה נִרְאֶה, דַּאֲפִלּוּ בְּאִשְׁתּוֹ טְהוֹרָה. וְכֵן אָמְרוּ חֲזַ"ל [חֲגִיגָה ה'] מַגִּיד לְאָדָם מַה שֵּׂחוֹ, אֲפִלּוּ שִׂיחָה קַלָּה שֶׁבֵּין אִישׁ לְאִשְׁתּוֹ מַגִּידִין לוֹ בִּשְׁעַת הַדִּין. אֶלָּא אִם כֵּן צָרִיךְ לְפַיְּסָהּ לִדְבַר מִצְוָה, כְּגוֹן רַב דְּשָׂח וְשָׂחַק וְעָשָׂה צְרָכָיו: <B>מִכָּאן אָמְרוּ חֲכָמִים.</B> רַבֵּנוּ הַקָּדוֹשׁ שֶׁסִּדֵּר הַמִּשְׁנָיוֹת כְּתָבָהּ, מִדִּבְרֵי חָכָם זֶה שֶׁאָמַר וְאַל תַּרְבֶּה שִׂיחָה עִם הָאִשָּׁה לָמְדוּ חֲכָמִים לוֹמַר, כָּל זְמַן שֶׁאָדָם מַרְבֶּה שִׂיחָה עִם הָאִשָּׁה גּוֹרֵם רָעָה לְעַצְמוֹ. מָצָאתִי כָּתוּב, כְּשֶׁאָדָם מְסַפֵּר לְאִשְׁתּוֹ קוֹרוֹתָיו כָּךְ וְכָךְ אֵרַע לִי עִם פְּלוֹנִי, הִיא מְלַמֶּדְתּוֹ לְחַרְחֵר רִיב, כְּגוֹן קֹרַח שֶׁסִּפֵּר לְאִשְׁתּוֹ מַה שֶּׁעָשָׂה מֹשֶׁה שֶׁהֵנִיף אֶת הַלְּוִיִּים תְּנוּפָה, וֶהֱבִיאַתּוּ בִּדְבָרֶיהָ לִידֵי מַחֲלֹקֶת. אִי נַמִּי, מִתּוֹךְ שֶׁהוּא מְסַפֵּר לָהּ שֶׁחֲבֵרָיו גִּנּוּהוּ וּבִיְּשׁוּהוּ אַף הִיא מְבַזָּה אוֹתוֹ בְּלִבָּהּ, וְזֶה גּוֹרֵם רָעָה לְעַצְמוֹ: <B>וּבוֹטֵל מִדִּבְרֵי תוֹרָה.</B> שֶׁנִּמְשַׁךְ אַחַר דְּבָרִים בְּטֵלִים וְאֵינוֹ עוֹסֵק בַּתּוֹרָה: 

~ פרק א - משנה ו
<B>עֲשֵׂה לְךָ רַב.</B> רַמְבַּ"ם פֵּרֵשׁ, אַף עַל פִּי שֶׁאֵינוֹ רָאוּי לִהְיוֹת לְךָ רַב, עֲשֵׂה אוֹתוֹ רַב עָלֶיךָ וְלֹא תִּלְמֹד בֵּינְךָ לְבֵין עַצְמְךָ. וַאֲנִי שָׁמַעְתִּי, עֲשֵׂה לְךָ רַב, שֶׁיְּקַבֵּל לוֹ רַב אֶחָד שֶׁיִּלְמֹד מִמֶּנּוּ תָּמִיד, וְלֹא יִלְמֹד הַיּוֹם מֵאֶחָד וּלְמָחָר מִן הָאַחֵר. וְאַף עַל גַּב דִּבְמַסֶּכֶת עֲבוֹדָה זָרָה [דַּף י"ט] אָמְרוּ הַלּוֹמֵד תּוֹרָה מֵרַב אֶחָד אֵינוֹ רוֹאֶה סִימַן בְּרָכָה, כְּבָר פֵּרְשׁוּ וְאָמְרוּ הָנֵי מִלֵּי בִּסְבָרָא, שֶׁטּוֹב לוֹ לִשְׁמֹעַ סְבָרַת הָרַבִּים, אֲבָל לְעִנְיַן גְּמָרָא, מֵחַד רַב מְעַלֵּי, כִּי הֵיכִי דְּלֹא לִפְגֹּם לִשָׁנֵיהּ: <B>וּקְנֵה לְךָ חָבֵר.</B> וַאֲפִלּוּ אַתָּה צָרִיךְ לִקְנוֹתוֹ בְּדָמִים יְקָרִים וּלְפַזֵּר עָלָיו מָמוֹן כְּדֵי שֶׁתִּקְנֶה אַהֲבָתוֹ {יד}. אֲבָל בְּרַב לֹא שַׁיָּךְ לוֹמַר קְנֵה לְךָ רַב, שֶׁהָרַב צָרִיךְ לְלַמֵּד בְּחִנָּם: <B>וֶהֱוֵי דָן אֶת כָּל הָאָדָם לְכַף זְכוּת.</B> כְּשֶׁהַדָּבָר בְּכַף מֹאזְנַיִם וְאֵין לוֹ הֶכְרֵעַ לְכָאן וּלְכָאן, כְּגוֹן אָדָם שֶׁאֵין אָנוּ יוֹדְעִים מִמַּעֲשָׂיו אִם צַדִּיק אִם רָשָׁע וְעָשָׂה מַעֲשֶׂה שֶׁאֶפְשָׁר לְדוּנוֹ לִזְכוּת וְאֶפְשָׁר לְדוּנוֹ לְחוֹבָה, מִדַּת חֲסִידוּת הִיא לְדוּנוֹ לְכַף זְכוּת. אֲבָל אָדָם שֶׁהֻחְזַק בְּרָשָׁע, מֻתָּר לְדוּנוֹ לְחוֹבָה, שֶׁלֹּא אָמְרוּ אֶלָּא הַחוֹשֵׁד בִּכְשֵׁרִים לוֹקֶה בְּגוּפוֹ [שַׁבָּת צז.], מִכְּלָל שֶׁהַחוֹשֵׁד בִּרְשָׁעִים אֵינוֹ לוֹקֶה: 

~ פרק א - משנה ז
<B>הַרְחֵק מִשָּׁכֵן רָע.</B> שֶׁלֹּא תִּלְמֹד מִמַּעֲשָׂיו. וְעוֹד, שֶׁלֹּא תִּלְקֶה עִמּוֹ בְּמַפַּלְתּוֹ, דְּאוֹי לָרָשָׁע וְאוֹי לִשְׁכֵנוֹ: <B>וְאַל תִּתְחַבֵּר לָרָשָׁע.</B> שֶׁכָּךְ אָמְרוּ חֲכָמִים, כָּל הַמִּתְדַּבֵּק לָרְשָׁעִים אַף עַל פִּי שֶׁאֵינוֹ עוֹשֶׂה כְּמַעֲשֵׂיהֶם נוֹטֵל שָׂכָר כַּיּוֹצֵא בָּהֶם. לְמַה הַדָּבָר דּוֹמֶה, לְנִכְנָס לְבֵיתוֹ שֶׁל בּוּרְסִי, אַף עַל פִּי שֶׁלֹּא לָקַח מִמֶּנּוּ כְּלוּם, מִכָּל מָקוֹם רֵיחַ רַע קָלַט וְהוֹצִיא עִמּוֹ: <B>וְאַל תִּתְיָאֵשׁ מִן הַפֻּרְעָנוּת.</B> שֶׁלֹּא תֹּאמַר רָשָׁע זֶה מַעֲשָׂיו מַצְלִיחִין אֵלֵךְ וְאֶדָּבֵק עִמּוֹ הוֹאִיל וְהַשָּׁעָה מְשַׂחֶקֶת לוֹ, לְכָךְ אָמַר וְאַל תִּתְיָאֵשׁ מִן הַפֻּרְעָנוּת, כְּלוֹמַר דַּע שֶׁמְּהֵרָה תָּבֹא עָלָיו פֻּרְעָנוּת כִּי פִּתְאֹם יָבֹא אֵידוֹ:

<B>אַל תַּעַשׂ עַצְמְךָ כְעוֹרְכֵי הַדַּיָּנִין.</B> כְּאוֹתָן אֲנָשִׁים שֶׁעוֹרְכִים וּמְסַדְּרִים טַעֲנוֹת בַּעֲלֵי הַדִּין {יז} לִפְנֵי הַדַּיָּנִים. שֶׁאָסוּר לָאָדָם לְגַלּוֹת דִּינוֹ לְאֶחָד מִבַּעֲלֵי הַדִּין וְלוֹמַר לוֹ עֲשֵׂה כָּךְ {יח} בִּשְׁבִיל שֶׁתִּזְכֶּה בְּדִינְךָ, וְאַף עַל פִּי שֶׁיּוֹדֵעַ שֶׁהַדִּין עִמּוֹ. פֵּרוּשׁ אַחֵר, כְּעוֹרְכֵי הַדַּיָּנִין, כִּגְדוֹלֵי הַדַּיָּנִין. וּבְתַלְמִיד הַיּוֹשֵׁב לִפְנֵי רַבּוֹ מְדַבֵּר, שֶׁלֹּא יַעֲשֶׂה עַצְמוֹ כִּגְדוֹלֵי הַדַּיָּנִים לְדַבֵּר לִפְנֵי רַבּוֹ כְּפוֹסֵק אֶת הַדִּין. עוֹרְכֵי, לְשׁוֹן עַרְכָּאוֹת שֶׁל נָכְרִים [גִּטִּין ט:] <B>י:, עַרְכָּאוֹת שֶׁל בֵּית דָּוִד.</B> וּמִפִּי אֲחֵרִים שָׁמַעְתִּי, אַל תַּעַשׂ עַצְמְךָ כִּגְדוֹלֵי הַדַּיָּנִים לָכוֹף בַּעֲלֵי דִּינִים שֶׁיָּבֹאוּ לַדִּין לְפָנֶיךָ: <B>יִהְיוּ בְעֵינֶיךָ כִרְשָׁעִים.</B> שֶׁלֹּא יַטֶּה לִבְּךָ לְאֶחָד מֵהֶם לוֹמַר אִישׁ פְּלוֹנִי חָשׁוּב הוּא וְלֹא יִטְעֹן טַעֲנַת שֶׁקֶר. שֶׁאִם אַתָּה אוֹמֵר כֵּן, אֵין אַתָּה רוֹאֶה לוֹ חוֹבָה: <B>כְּשֶׁקִּבְּלוּ עֲלֵיהֶם אֶת הַדִּין.</B> שֶׁלֹּא תַּחְשֹׁד הַחַיָּב לוֹמַר גַּזְלָן הָיָה זֶה, אֶלָּא תֹּאמַר שֶׁמָּא טוֹעֶה הָיָה וְלֹא נִתְכַּוֵּן לִגְזֹל {כ}. אִי נַמִּי, נִתְחַיֵּב אֶחָד מֵהֶם שְׁבוּעָה וְנִשְׁבַּע, לֹא תֹּאמַר לַשֶּׁקֶר נִשְׁבַּע: 

~ פרק א - משנה ט
<B>וֶהֱוֵי זָהִיר בִּדְבָרֶיךָ.</B> שֶׁלֹּא יֹאמַר הַדַּיָּן שֶׁמָּא כָּךְ וְכָךְ הָיָה מַעֲשֶׂה, אוֹ אִם הָיָה הַדָּבָר כָּךְ הָיָה פְּלוֹנִי זַכַּאי, וּמִתּוֹךְ דְּבָרִים הַלָּלוּ לוֹמֵד בַּעַל הַדִּין אוֹ הָעֵדִים לוֹמַר דָּבָר שֶׁלֹּא הָיָה: 

~ פרק א - משנה י
<B>שְׁמַעְיָה וְאַבְטַלְיוֹן.</B> גֵּרֵי צֶדֶק הָיוּ {כא}, וּמִבְּנֵי בָּנָיו שֶׁל סַנְחֵרִיב הֵם. וְשָׁמַעְתִּי שֶׁמִּפְּנֵי שֶׁהָיָה אַבְטַלְיוֹן אַב בֵּית דִּין נִקְרָא בְּשֵׁם זֶה, שֶׁפֵּרוּשׁוֹ אַב לַקְּטַנִּים. כִּי טַלְיָא בִּלְשׁוֹן אֲרַמִּי, קָטָן, כְּמוֹ [מְגִלָּה ה:] אָמַר רַב יוֹחָנָן כַּד הֲוֵינָא טַלְיָא, כְּשֶׁהָיִיתִי קָטָן. לֵיתוּ טַלְיָא וְטַלְיָתָא [יְבָמוֹת קיד.] יָבִיאוּ קָטָן וּקְטַנָּה. אַף כָּאן אַבְטַלְיוֹן אֲבִיהֶן שֶׁל יְתוֹמִים קְטַנִּים: <B>אֱהֹב אֶת הַמְּלָאכָה.</B> אֲפִלּוּ יֵשׁ לוֹ בְּמַה לְּהִתְפַּרְנֵס {כב}, חַיָּב לַעֲסֹק בִּמְלָאכָה. שֶׁהַבַּטָּלָה מְבִיאָה לִידֵי שִׁעֲמוּם [כְּתֻבּוֹת נט:]: <B>וּשְׂנָא אֶת הָרַבָּנוּת.</B> וְלֹא תֹּאמַר אָדָם גָּדוֹל אֲנִי וּגְנַאי לִי לַעֲסֹק בִּמְלָאכָה, דְּאָמַר לֵיהּ רַב לְרַב כַּהֲנָא [פְּסָחִים קיג.], פְּשֹׁט נְבֵלְתָא בַּשּׁוּק וְטֹל אַגְרָא, וְלֹא תֵימָא כַּהֲנָא אֲנָא גַּבְרָא רַבָּא אֲנָא וְסַנְיָא בִּי מִלְּתָא. פֵּרוּשׁ אַחֵר, וּשְׂנָא אֶת הָרַבָּנוּת, הִתְרַחֵק מִלִּנְהֹג שְׂרָרָה עַל צִבּוּר, שֶׁהָרַבָּנוּת מְקַבֶּרֶת אֶת בְּעָלֶיהָ: <B>וְאַל תִּתְוַדַּע לָרָשׁוּת.</B> כְּדֵי לִטֹּל רַבָּנוּת עַל יָדֶיהָ. אִי נַמִּי, אַל תִּתְוַדַּע לָרָשׁוּת, שֶׁלֹּא יַעֲבִירוּךָ עַל דַּעַת קוֹנְךָ, כְּמוֹ שֶׁאֵרַע לְדוֹאֵג הָאֲדוֹמִי. לָרָשׁוּת, הַשְּׂרָרָה קְרוּיָה רָשׁוּת, מִפְּנֵי שֶׁהָרְשׁוּת בְּיָדָהּ לַעֲשׂוֹת כִּרְצוֹנָהּ: 

~ פרק א - משנה יא
<B>הִזָּהֲרוּ בְדִבְרֵיכֶם.</B> שֶׁלֹּא תַּנִּיחוּ מָקוֹם לַמִּינִים לִטְעוֹת בְּדִבְרֵיכֶם: <B>שֶׁמָּא תָחוּבוּ חוֹבַת גָּלוּת.</B> כְּלוֹמַר, אַף עַל פִּי שֶׁבַּמָּקוֹם שֶׁאַתֶּם בּוֹ אֵין שָׁם מִינִים, יֵשׁ לָחוּשׁ שֶׁמָּא יִגְרֹם הַחֵטְא וְתִתְחַיְּבוּ גָּלוּת, וְתִגְלוּ לְמָקוֹם שֶׁיֵּשׁ שָׁם אֲנָשִׁים שֶׁמַּרְאִים פָּנִים בַּתּוֹרָה שֶׁלֹּא כַּהֲלָכָה, וְהֵם כִּנּוּי לַמַּיִם הָרָעִים {כד}, וְיָבִינוּ מִתּוֹךְ דִּבְרֵיכֶם דְּבָרִים שֶׁאֵינָם הֲגוּנִים, וְיִשְׁתּוּ הַתַּלְמִידִים הַבָּאִים אַחֲרֵיכֶם מֵאוֹתָם דְּבָרִים דִּבְרֵי מִינוּת וְיָמוּתוּ בַּעֲוֹנָם: <B>וְנִמְצָא שֵׁם שָׁמַיִם מִתְחַלֵּל.</B> שֶׁיִּשָּׁאֲרוּ אוֹתָן דֵּעוֹת בְּטֵלוֹת בָּעוֹלָם. כְּמוֹ שֶׁקָּרָה לְאַנְטִיגְנוֹס אִישׁ סוֹכוֹ עִם צָדוֹק וּבַיְתּוֹס תַּלְמִידָיו, שֶׁאָמַר לָהֶם אַל תִּהְיוּ כַּעֲבָדִים הַמְשַׁמְּשִׁים אֶת הָרַב עַל מְנָת לְקַבֵּל פְּרָס, וְאָמְרוּ הֵם, אֶפְשָׁר פּוֹעֵל עוֹשֶׂה מְלָאכָה כָּל הַיּוֹם וְטוֹרֵחַ וְלָעֶרֶב אֵינוֹ מְקַבֵּל פְּרָס. וְיָצְאוּ לְמִינוּת הֵם {כה} וְתַלְמִידֵיהֶם, וְנִקְרָאִים צְדוֹקִים וּבַיְתּוֹסִים עַד הַיּוֹם: 

~ פרק א - משנה יב
<B>הֱוֵי מִתַּלְמִידָיו שֶׁל אַהֲרֹן אוֹהֵב שָׁלוֹם וְרוֹדֵף שָׁלוֹם.</B> פֵּרְשׁוּ בְּאָבוֹת דְּרַבִּי נָתָן, כֵּיצַד הָיָה אַהֲרֹן אוֹהֵב שָׁלוֹם, כְּשֶׁהָיָה רוֹאֶה שְׁנֵי בְּנֵי אָדָם מִתְקוֹטְטִים הָיָה הוֹלֵךְ לְכָל אֶחָד מֵהֶם שֶׁלֹּא מִדַּעַת חֲבֵרוֹ, וְאוֹמֵר לוֹ רְאֵה חֲבֵרְךָ אֵיךְ הוּא מִתְחָרֵט וּמַכֶּה אֶת עַצְמוֹ עַל שֶׁחָטָא לְךָ, וְהוּא אָמַר לִי שֶׁאָבֹא אֵלֶיךָ שֶׁתִּמְחַל לוֹ, וּמִתּוֹךְ כָּךְ כְּשֶׁהָיוּ פּוֹגְעִים זֶה בָּזֶה הָיוּ מְנַשְּׁקִים זֶה אֶת זֶה. וְכֵיצַד הָיָה מְקָרֵב אֶת הַבְּרִיּוֹת לַתּוֹרָה, כְּשֶׁהָיָה יוֹדֵעַ בְּאָדָם שֶׁעָבַר עֲבֵרָה הָיָה מִתְחַבֵּר {כח} עִמּוֹ וּמַרְאֶה לוֹ פָּנִים צְהֻבּוֹת, וְהָיָה אוֹתוֹ אָדָם מִתְבַּיֵּשׁ וְאוֹמֵר אִלּוּ הָיָה יוֹדֵעַ צַדִּיק זֶה מַעֲשַׂי הָרָעִים כַּמָּה הָיָה מִתְרַחֵק מִמֶּנִּי, וּמִתּוֹךְ כָּךְ הָיָה חוֹזֵר לַמּוּטָב. הוּא שֶׁהַנָּבִיא מֵעִיד עָלָיו (מַלְאָכִי ב) בְּשָׁלוֹם וּבְמִישׁוֹר הָלַךְ אִתִּי וְרַבִּים הֵשִׁיב מֵעָוֹן: 

~ פרק א - משנה יג
<B>נָגֵד שְׁמָא אָבֵד שְׁמֵהּ.</B> מִי שֶׁנִּמְשַׁךְ שְׁמוֹ לְמֵרָחוֹק מִתּוֹךְ שְׂרָרָה וְרַבָּנוּת, מְהֵרָה יֵאָבֵד שְׁמוֹ, שֶׁהָרַבָּנוּת מְקַבֶּרֶת אֶת בְּעָלֶיהָ. נָגֵד לְשׁוֹן מְשִׁיכָה. תִּרְגּוּם מִשְׁכוּ, נְגִידוּ: <B>וּדְלֹא מוֹסִיף.</B> מִי שֶׁאֵינוֹ מוֹסִיף עַל לִמּוּדוֹ: <B>יָסֵף.</B> יָסוּף מִפִּיו מַה שֶּׁכְּבָר לָמַד וְיִשְׁכַּח תַּלְמוּדוֹ. וְיֵשׁ שֶׁגּוֹרְסִין יֵאָסֵף, כְּלוֹמַר יֵאָסֵף אֶל עַמָּיו וְיָמוּת בְּלֹא עִתּוֹ: <B>וּדְלֹא יָלֵיף.</B> קָשֶׁה מִדְּלֹא מוֹסִיף, הִלְכָּךְ קְטָלָא חַיָּב, כְּלוֹמַר רָאוּי הוּא שֶׁיַּהַרְגוּהוּ, כְּמוֹ שֶׁאָמְרוּ [פְּסָחִים מט] עַם הָאָרֶץ מֻתָּר לְקָרְעוֹ כְּדָג, וּמִגַּבּוֹ: <B>וּדְאִשְׁתַּמֵּשׁ בְּתָגָא.</B> הַמִּשְׁתַּמֵּשׁ בְּכִתְרָהּ שֶׁל תּוֹרָה {לא} כְּאָדָם הַמִּשְׁתַּמֵּשׁ בְּכֵלָיו, חָלַף וְעָבַר מִן הָעוֹלָם. תָּגָא בִּלְשׁוֹן יִשְׁמָעֵאל כֶּתֶר. וְיֵשׁ מְפָרְשִׁים, תג"א רָאשֵׁי תֵּבוֹת תַּ'לְמִיד גַּ'בְרָא אַ'חֲרִינָא. שֶׁאָסוּר לָאָדָם לְהִשְׁתַּמֵּשׁ בְּתַלְמִידִים שֶׁאֵינָן תַּלְמִידָיו. וַאֲנִי שָׁמַעְתִּי, וּדְיִשְׁתַּמֵּשׁ בְּתָגָא, הַמִּשְׁתַּמֵּשׁ בַּשֵּׁם הַמְפֹרָשׁ, חָלַף וְאָבַד, שֶׁאֵין לוֹ חֵלֶק לָעוֹלָם הַבָּא: 

~ פרק א - משנה יד
<B>אִם אֵין אֲנִי לִי.</B> אִם אֵין אֲנִי זוֹכֶה לְעַצְמִי, מִי יִזְכֶּה בִּשְׁבִילִי: <B>וּכְשֶׁאֲנִי לְעַצְמִי.</B> וַאֲפִלּוּ זָכִיתִי לְעַצְמִי, מַה הוּא הַזְּכוּת הַזֶּה וּבַמֶּה נֶחְשַׁב הוּא כְּנֶגֶד מַה שֶּׁאֲנִי חַיָּב לַעֲשׂוֹת: <B>וְאִם לֹא עַכְשָׁיו.</B> בָּעוֹלָם הַזֶּה: <B>אֵימָתָי.</B> כִּי אַחַר הַמָּוֶת אִי אֶפְשָׁר לִזְכּוֹת עוֹד. פֵּרוּשׁ אַחֵר, אִם לֹא עַכְשָׁיו בִּימֵי הַבַּחֲרוּת, אֵימָתַי, שֶׁמָּא בִּימֵי הַזִּקְנָה לֹא יַעֲלֶה לְיָדִי: 

~ פרק א - משנה טו
<B>עֲשֵׂה תּוֹרָתְךָ קֶבַע.</B> שֶׁיִּהְיֶה עִקַּר עִסְקְךָ בַּיּוֹם וּבַלַּיְלָה בַּתּוֹרָה, וּכְשֶׁתִּהְיֶה יָגֵעַ מִן הַלִּמּוּד תַּעֲשֶׂה מְלָאכָה. וְלֹא שֶׁיִּהְיֶה עִקַּר עִסְקְךָ בַּמְּלָאכָה וּכְשֶׁתִּפְנֶה מִן הַמְּלָאכָה תַּעֲסֹק בַּתּוֹרָה. וּמָצָאתִי כָּתוּב, עֲשֵׂה תּוֹרָתְךָ קֶבַע, שֶׁלֹּא תַּחְמִיר לְעַצְמְךָ וְתָקֵל לָאֲחֵרִים {לב}, אוֹ תַּחְמִיר לַאֲחֵרִים וְתָקֵל לְעַצְמְךָ, אֶלָּא תְּהֵא תּוֹרָתְךָ קֶבַע לְךָ כְּמוֹ לַאֲחֵרִים, וְכֵן הוּא אוֹמֵר בְּעֶזְרָא [ז] כִּי עֶזְרָא הֵכִין לְבָבוֹ לִדְרֹשׁ אֶת תּוֹרַת ה' וְלַעֲשׂוֹת וּלְלַמֵּד לִבְנֵי יִשְׂרָאֵל, כְּמוֹ שֶׁהֵכִין לִבּוֹ לַעֲשׂוֹת כָּךְ הָיָה מְלַמֵּד לִבְנֵי יִשְׂרָאֵל: <B>אֱמֹר מְעַט וַעֲשֵׂה הַרְבֵּה.</B> כְּמוֹ שֶׁמָּצִינוּ בְּאַבְרָהָם אָבִינוּ עָלָיו הַשָּׁלוֹם, שֶׁאָמַר תְּחִלָּה (בְּרֵאשִׁית יד) וְאֶקְּחָה פַת לֶחֶם, וּבַסּוֹף וַיִקַּח בֶּן בָּקָר רַךְ וָטוֹב: <B>וֶהֱוֵי מְקַבֵּל אֶת כָּל הָאָדָם.</B> כְּשֶׁאַתָּה מַכְנִיס אוֹרְחִים לְבֵיתְךָ לֹא תִּתֵּן לָהֶם וּפָנֶיךָ כְּבוּשׁוֹת בַּקַּרְקַע, שֶׁכָּל הַנּוֹתֵן וּפָנָיו כְּבוּשׁוֹת בַּקַּרְקַע אֲפִלּוּ נָתַן כָּל מַתָּנוֹת שֶׁבָּעוֹלָם מַעֲלִין עָלָיו כְּאִלּוּ לֹא נָתַן כְּלוּם. שָׁלֹשׁ אַזְהָרוֹת הִזְהִיר שַׁמַּאי, כְּנֶגֶד שָׁלֹשׁ מַעֲלוֹת שֶׁהִזְכִּיר יִרְמְיָה, חָכָם גִּבּוֹר וְעָשִׁיר. כְּנֶגֶד חָכָם אָמַר עֲשֵׂה תּוֹרָתְךָ קֶבַע, כְּנֶגֶד עָשִׁיר אָמַר אֱמֹר מְעַט וַעֲשֵׂה הַרְבֵּה, כְּנֶגֶד גִּבּוֹר אָמַר הֱוֵי מְקַבֵּל אֶת כָּל הָאָדָם בְּסֵבֶר פָּנִים יָפוֹת, שֶׁיִּכְבֹּשׁ יִצְרוֹ וְיִלָּחֵם כְּנֶגֶד לִבּוֹ הָרַע, וְשָׁנִינוּ (לְקַמָּן פ"ד מ"א) אֵיזֶהוּ גִּבּוֹר הַכּוֹבֵשׁ אֶת יִצְרוֹ: 

~ פרק א - משנה טז
<B>עֲשֵׂה לְךָ רַב.</B> לְעִנְיַן הוֹרָאָה מְדַבֵּר עַכְשָׁיו, אִם בָּא דִּין לְפָנֶיךָ וְנִסְתַּפַּקְתָּ בּוֹ, עֲשֵׂה לְךָ רַב, וְהִסְתַּלֵּק מִן הַסָּפֵק. וְלֹא תִּפְסֹק עָלָיו אַתָּה לְבַדְּךָ. כִּי הָא דְרָבָא כִּי הֲוֵי אָתֵי טְרֵפְתָא לְגַבֵּיהּ הֲוֵי מְכַנֵּיף לְכֻלְּהוּ טַבָּחֵי דְּמָתָא מַחְסַיָּא, אָמַר, כִּי הֵיכִי דְלִמְטְיַן שִׁיבָא מִכְּשׁוּרָא: <B>וְאַל תַּרְבֶּה לְעַשֵּׂר אֹמָדוֹת.</B> שֶׁהַמַּפְרִישׁ מַעַשְׂרוֹת מֵאֹמֶד אֵינוֹ נִצּוֹל מִן הַקַּלְקָלָה. אִם פּוֹחֵת מִמַּה שֶּׁרָאוּי לִתֵּן מַעַשְׂרוֹתָיו מְתֻקָּנִים וּפֵרוֹתָיו מְקֻלְקָלִים. וְאִם הֶעֱדִיף עַל שֶׁהָיָה רָאוּי לִתֵּן, פֵּרוֹתָיו מְתֻקָּנִים וּמַעַשְׂרוֹתָיו מְקֻלְקָלִים {לד}: 

~ פרק א - משנה יז
<B>וְלֹא מָצָאתִי לַגּוּף טוֹב אֶלָּא שְׁתִיקָה.</B> מִי שֶׁשּׁוֹמֵעַ חֶרְפָּתוֹ וְשׁוֹתֵק: <B>וְלֹא הַמִּדְרָשׁ הוּא הָעִקָּר אֶלָּא הַמַּעֲשֶׂה.</B> וְתֵדַע לְךָ שֶׁהַשְּׁתִיקָה טוֹבָה {לו}, שֶׁאֲפִלּוּ הַמִּדְרָשׁ וְהַדְּרָשׁ וְהַדִּבּוּר בַּתּוֹרָה שֶׁאֵין לְךָ מִדָּה טוֹבָה הֵימֶנָּה, אֵין עִקַּר קִבּוּל הַשָּׂכָר אֶלָּא בִּשְׁבִיל הַמַּעֲשֶׂה, וְהַדּוֹרֵשׁ וְאֵינוֹ מְקַיֵּם, נוֹחַ לוֹ אִם הָיָה שׁוֹתֵק וְלֹא הָיָה דּוֹרֵשׁ: <B>וְכָל הַמַּרְבֶּה דְבָרִים מֵבִיא חֵטְא.</B> שֶׁכֵּן מָצִינוּ בְּחַוָּה שֶׁהִרְבְּתָה דְּבָרִים, וְאָמְרָה, אָמַר אֱלֹהִים לֹא תֹאכְלוּ מִמֶּנּוּ וְלֹא תִגְעוּ בּוֹ, וְהוֹסִיפָה נְגִיעָה שֶׁלֹּא נֶאֶסְרָה בָּהּ, וּדְחָפָהּ נָחָשׁ עַד שֶׁנָּגְעָה בּוֹ, וְאָמַר לָהּ, כְּשֵׁם שֶׁאֵין מִיתָה בִּנְגִיעָה כָּךְ אֵין מִיתָה בָּאֲכִילָה, וּמִתּוֹךְ כָּךְ בָּאָת לִידֵי חֵטְא, שֶׁאָכְלָה מִן הַפְּרִי. הוּא שֶׁשְּׁלֹמֹה אוֹמֵר (מִשְׁלֵי ל) אַל תּוֹסֵף עַל דְּבָרָיו פֶּן יוֹכִיחַ בְּךָ וְנִכְזָבְתָּ: 

~ פרק א - משנה יח
<B>הָעוֹלָם עוֹמֵד.</B> יִשּׁוּבָן שֶׁל בְּנֵי אָדָם מִתְקַיֵּם. אֵין זֶה כְּמוֹ הָעוֹלָם עוֹמֵד דִּלְעֵיל {לח}: <B>עַל הַדִּין.</B> לְזַכּוֹת אֶת הַזַּכַּאי וּלְחַיֵּב אֶת הַחַיָּב: <B>וְעַל הָאֱמֶת.</B> שֶׁלֹּא יְשַׁקְּרוּ אִישׁ בַּעֲמִיתוֹ: <B>וְעַל הַשָּׁלוֹם.</B> בֵּין הַמַּלְכֻיּוֹת, וּבֵין אָדָם לַחֲבֵרוֹ:

<B>רַבִּי אוֹמֵר.</B> שֶׁיָּבֹר. שֶׁיִּבְרֹר: <B>כָּל שֶׁהִיא תִפְאֶרֶת לְעוֹשֶׂיהָ וְתִפְאֶרֶת לוֹ מִן הָאָדָם.</B> שֶׁיְּהֵא נוֹחַ לוֹ, וְיִהְיוּ נוֹחִין בְּנֵי אָדָם מִמֶּנּוּ. וְזֶה יִהְיֶה, כְּשֶׁיֵּלֵךְ בְּכָל הַמִּדּוֹת בַּדֶּרֶךְ הָאֶמְצָעִי וְלֹא יִטֶּה לְאֶחָד מִשְּׁנֵי הַקְּצָווֹת. שֶׁאִם הוּא כִּילַי בְּיוֹתֵר, יֵשׁ תִּפְאֶרֶת לוֹ, שֶׁאוֹסֵף מָמוֹן הַרְבֵּה, אֲבָל אֵין בְּנֵי אָדָם מְפָאֲרִין אוֹתוֹ בְּמִדָּה זוֹ. וְאִם הוּא מְפַזֵּר יוֹתֵר מִן הָרָאוּי, בְּנֵי אָדָם הַמְקַבְּלִים מִמֶּנּוּ מְפָאֲרִים אוֹתוֹ, אֲבָל אֵינוֹ תִפְאֶרֶת לְעוֹשֶׂיהָ וְאֵינוֹ נוֹחַ לוֹ, שֶׁהוּא בָּא לִידֵי עֲנִיּוּת בִּשְׁבִיל כָּךְ. אֲבָל מִדַּת הַנְּדִיבוּת שֶׁהִיא אֶמְצָעִית בֵּין הַכִּילוּת וְהַפִּזּוּר, הִיא תִּפְאֶרֶת לְעוֹשֶׂיהָ, שֶׁשּׁוֹמֵר אֶת מָמוֹנוֹ וְאֵינוֹ מְפַזֵּר יוֹתֵר מִן הָרָאוּי. וְתִפְאֶרֶת לוֹ מִן הָאָדָם, שֶׁבְּנֵי אָדָם מְפָאֲרִים אוֹתוֹ, שֶׁנּוֹתֵן כְּמוֹ שֶׁרָאוּי לוֹ לִתֵּן. וְכֵן הַדִּין בְּכָל שְׁאָר הַמִּדּוֹת: <B>שֶׁאֵין אַתָּה יוֹדֵעַ מַתַּן שְׂכָרָן שֶׁל מִצְוֹת.</B> לֹא נִתְפָּרֵשׁ בַּתּוֹרָה שְׂכַר הַמְקַיֵּם מִצְוֹת עֲשֵׂה {ג}, וְלֹא עֹנֶשׁ הַמְבַטֵּל מֵעֲשׂוֹתָן. דְּאִלּוּ עָנְשָׁן שֶׁל מִצְוֹת לֹא תַּעֲשֶׂה מְפֹרָשִׁין הֵן, סְקִילָה, שְׂרֵפָה, הֶרֶג, וְחֶנֶק, כָּרֵת, וּמִיתָה בִּידֵי שָׁמַיִם, וּמַלְקוֹת. הָעֹנֶשׁ הַקַּל לַעֲבֵרָה קַלָּה וְהֶחָמוּר לַחֲמוּרָה: <B>וֶהֱוֵי מְחַשֵּׁב הֶפְסֵד מִצְוָה.</B> מַה שֶּׁאַתָּה מַפְסִיד מִסְּחוֹרָתְךָ וּמָמוֹנְךָ מִפְּנֵי עֵסֶק הַמִּצְוָה, כְּנֶגֶד הַשָּׂכָר שֶׁיַּעֲלֶה לְךָ מִמֶּנָּהּ בָּעוֹלָם הַזֶּה אוֹ בָּעוֹלָם הַבָּא, שֶׁיִּהְיֶה יוֹתֵר מֵאוֹתוֹ הֶפְסֵד: <B>וּשְׂכַר עֲבֵרָה.</B> הֲנָאָה שֶׁאַתָּה נֶהֱנֶה בָּעֲבֵרָה, כְּנֶגֶד הֶפְסֵד שֶׁעָתִיד לָבֹא לְךָ מִמֶּנָּהּ:

<B>דֶּרֶךְ אֶרֶץ.</B> מְלָאכָה אוֹ סְחוֹרָה: <B>שֶׁיְּגִיעַת שְׁנֵיהֶם מְשַׁכַּחַת עָוֹן.</B> שֶׁהַתּוֹרָה מַתֶּשֶׁת כֹּחוֹ שֶׁל אָדָם, וְהַמְּלָאכָה מְפָרֶכֶת וּמְשַׁבֶּרֶת אֶת הַגּוּף, וּמִתּוֹךְ כָּךְ יֵצֶר הָרַע בָּטֵל מִמֶּנּוּ: <B>וְכָל תּוֹרָה שֶׁאֵין עִמָּהּ מְלָאכָה.</B> וְאִם תֹּאמַר יְהֵא עָמֵל בַּתּוֹרָה תָּמִיד וִיגִיעָתָהּ תְּהֵא מְשַׁכַּחַת עָוֹן, וּמַה צֹּרֶךְ לַמְּלָאכָה, לְכָךְ הֻזְקַק לוֹמַר וְכָל תּוֹרָה שֶׁאֵין עִמָּהּ מְלָאכָה סוֹפָהּ בְּטֵלָה, לְפִי שֶׁאִי אֶפְשָׁר לוֹ בְּלֹא מְזוֹנוֹת, וּמְלַסְטֵם אֶת הַבְּרִיּוֹת וּמְשַׁכֵּחַ תַּלְמוּדוֹ: <B>יִהְיוּ עוֹסְקִין עִמָּהֶם לְשֵׁם שָׁמַיִם.</B> וְלֹא כְּדֵי לִטֹּל עֲטָרָה לוֹמַר כָּךְ וְכָךְ עָשִׂיתִי בִּשְׁבִיל הַצִּבּוּר: <B>שֶׁזְּכוּת אֲבוֹתָם מְסַיַּעְתָּן.</B> שֶׁזְּכוּת אֲבוֹתָן שֶׁל צִבּוּר וְצִדְקָתָן הָעוֹמֶדֶת לָעַד הִיא הַמְסַיַּעַת אֶל הָעוֹסְקִין עִמָּהֶן לְהוֹצִיא לָאוֹר צִדְקָן, וְלֹא מִצַּד טוּב הִשְׁתַּדְּלוּתָן שֶׁל הָעוֹסְקִים: <B>וְאַתֶּם מַעֲלֶה אֲנִי עֲלֵיכֶם שָׂכָר.</B> אַף עַל פִּי שֶׁאֵין הַדָּבָר בָּא לִידֵי גְמַר טוֹב מִצַּד מַעֲשֵׂיכֶם אֶלָּא בִּשְׁבִיל זְכוּת אֲבוֹתָם שֶׁל צִבּוּר, מַעֲלֶה אֲנִי עֲלֵיכֶם שָׂכָר כְּאִלּוּ אַתֶּם עֲשִׂיתֶם הַיְשׁוּעָה הַזֹּאת בְּיִשְׂרָאֵל, הוֹאִיל וְאַתֶּם עוֹסְקִים לְשֵׁם שָׁמַיִם. פֵּרוּשׁ אַחֵר, כָּל הָעוֹסְקִים עִם הַצִּבּוּר לְהַכְרִיחַ וְלָכוֹף אֶת הַצִּבּוּר לִדְבַר מִצְוָה לִצְדָקָה אוֹ לְפִדְיוֹן שְׁבוּיִים, יִהְיוּ עוֹסְקִים עִמָּהֶם לְשֵׁם שָׁמַיִם {יא}, שֶׁזְּכוּת אָבוֹת שֶׁל צִבּוּר מְסַיַּעְתָּן לָתֵת מַה שֶּׁפָּסְקוּ עֲלֵיהֶם וַאֲפִלּוּ מָמוֹן הַרְבֵּה, וְהַצְּדָקָה שֶׁיַּעֲשׂוּ תַּעֲמֹד לָהֶם לָעַד. וְאַתֶּם הַמִּתְעַסְּקִים לָכוֹף אֶת הַצִּבּוּר לְמִצְוָה זוֹ, מַעֲלֶה אֲנִי עֲלֵיכֶם שָׂכָר כְּאִלּוּ עֲשִׂיתֶם זֹאת הַמִּצְוָה מִמָּמוֹנְכֶם מַמָּשׁ. וְרַמְבַּ"ם פֵּרֵשׁ, וְאַתֶּם מַעֲלֶה אֲנִי עֲלֵיכֶם שָׂכָר כְּאִלּוּ עֲשִׂיתֶם, אִם בְּשָׁעָה שֶׁאַתֶּם עוֹסְקִים עִם הַצִּבּוּר נִתְבַּטַּלְתֶּם מֵעֲשִׂיַּת אֵיזוֹ מִצְוָה בִּשְׁבִיל עִסְקֵי הַצִּבּוּר, מַעֲלֶה אֲנִי עֲלֵיכֶם שָׂכָר כְּאִלּוּ עֲשִׂיתֶם אוֹתָהּ מִצְוָה שֶׁנִּתְבַּטַּלְתֶּם מִמֶּנָּהּ:

<B>הֱווּ זְהִירִים בָּרָשׁוּת.</B> אַתֶּם הַמִּתְעַסְּקִים בְּצָרְכֵי צִבּוּר, אַף עַל פִּי שֶׁאַתֶּם צְרִיכִים לְהִתְוַדֵּעַ לָרָשׁוּת כְּדֵי לְפַקֵּחַ עַל עִסְקֵי רַבִּים {יג}, הֱווּ זְהִירִים וְכוּ':

<B>עֲשֵׂה רְצוֹנוֹ כִרְצוֹנְךָ.</B> פַּזֵּר מָמוֹנְךָ בְּחֶפְצֵי שָׁמַיִם, כִּרְצוֹנְךָ, כְּאִלּוּ פִּזַּרְתָּ אוֹתָם בַּחֲפָצֶיךָ. שֶׁאִם עָשִׂיתָ כֵן יַעֲשֶׂה רְצוֹנְךָ כִּרְצוֹנוֹ, כְּלוֹמַר יִתֵּן לְךָ טוֹבָה בְּעַיִן יָפָה: <B>כְּדֵי שֶׁיְּבַטֵּל רְצוֹן אֲחֵרִים.</B> יָפֵר עֲצַת כָּל הַקָּמִים עָלֶיךָ לְרָעָה. וַאֲנִי שָׁמַעְתִּי, שֶׁאֵין זֶה אֶלָּא דֶּרֶךְ כָּבוֹד כְּלַפֵּי מַעֲלָה, וַהֲרֵי הוּא כְּאִלּוּ כָּתוּב כְּדֵי שֶׁיְּבַטֵּל רְצוֹנוֹ מִפְּנֵי רְצוֹנְךָ. כְּאוֹתָהּ שֶׁאָמְרוּ בְּפֶרֶק בַּמֶּה אִשָּׁה [שַׁבָּת סג.] כָּל הָעוֹשֶׂה מִצְוָה כְּתִקְנָהּ אֲפִלּוּ גְּזַר דִּין שֶׁל שִׁבְעִים שָׁנָה מִתְבַּטֵּל מֵעָלָיו: <B>אַל תִּפְרֹשׁ מִן הַצִּבּוּר.</B> אֶלָּא הִשְׁתַּתֵּף בְּצָרָתָם. שֶׁכָּל הַפּוֹרֵשׁ מִן הַצִּבּוּר אֵינוֹ רוֹאֶה בְּנֶחָמַת הַצִּבּוּר {טו} [תַּעֲנִית יא.]: <B>וְאַל תַּאֲמִין בְּעַצְמְךָ עַד יוֹם מוֹתְךָ.</B> שֶׁהֲרֵי יוֹחָנָן כֹּהֵן גָּדוֹל שִׁמֵּשׁ בִּכְהֻנָּה גְּדוֹלָה שְׁמֹנִים שָׁנָה וּלְבַסּוֹף נַעֲשָׂה צְדוֹקִי [בְּרָכוֹת כט.]: <B>וְאַל תָּדִין אֶת חֲבֵרְךָ עַד שֶׁתַּגִּיעַ לִמְקוֹמוֹ.</B> אִם רָאִיתָ חֲבֵרְךָ שֶׁבָּא לִידֵי נִסָּיוֹן וְנִכְשַׁל, אַל תְּדִינֵהוּ לְחוֹבָה עַד שֶׁתַּגִּיעַ לִידֵי נִסָּיוֹן כְּמוֹתוֹ וְתִנָּצֵל: <B>וְאַל תֹּאמַר דָּבָר שֶׁאִי אֶפְשָׁר לִשְׁמֹעַ וְסוֹפוֹ לְהִשָּׁמֵעַ.</B> כְּלוֹמַר, לֹא יִהְיוּ דְּבָרֶיךָ מְסֻפָּקִים שֶׁאִי אֶפְשָׁר לְהָבִינָם בִּתְחִלָּה וּבְעִיּוּן רִאשׁוֹן, וְתִסְמֹךְ שֶׁאִם יַעֲמִיק הַשּׁוֹמֵעַ בָּהֶם לְבַסּוֹף יְבִינֵם. כִּי זֶה יָבִיא בְּנֵי אָדָם לִטְעוֹת בִּדְבָרֶיךָ, שֶׁמָּא יִטְעוּ וְיֵצְאוּ לְמִינוּת עַל יָדְךָ {טז}. פֵּרוּשׁ אַחֵר, לֹא תְּגַלֶּה סוֹדְךָ אֲפִלּוּ בֵּינְךָ לְבֵין עַצְמְךָ וְתֹאמַר שֶׁאֵין כָּאן מִי שֶׁיִּשְׁמָעֲךָ, לְפִי שֶׁסּוֹפוֹ לְהִשָּׁמַע, כִּי עוֹף הַשָּׁמַיִם יוֹלִיךְ אֶת הַקּוֹל {יז}. וְהַגִּרְסָא הִיא לְפִי פֵּרוּשׁ זֶה, שֶׁסּוֹפוֹ לְהִשָּׁמַע. אֲבָל רַשִׁ"י גּוֹרֵס, וְאַל תֹּאמַר דָּבָר שֶׁאֶפְשָׁר לִשְׁמֹעַ שֶׁסּוֹפוֹ לְהִשָּׁמַע. וּבְדִבְרֵי תּוֹרָה מְדַבֵּר, אַל תֹּאמַר עַל דְּבַר תּוֹרָה שֶׁאַתָּה יָכוֹל לִשְׁמֹעַ עַכְשָׁיו שֶׁתִּשְׁמַע אוֹתוֹ לְבַסּוֹף, אֶלָּא הַט אָזְנֶיךָ וּשְׁמַע מִיָּד: <B>לִכְשֶׁאִפָּנֶה.</B> מֵעֲסָקַי, אֶעֱסֹק בַּתּוֹרָה:

<B>בּוּר.</B> רֵיק מִכָּל דָּבָר, וַאֲפִלּוּ בְּטִיב מַשָּׂא וּמַתָּן אֵינוֹ יוֹדֵעַ. וְהוּא גָּרוּעַ מֵעַם הָאָרֶץ. תַּרְגּוּם וְהָאֲדָמָה לֹא תֵשָׁם [בְּרֵאשִׁית מז], וְאַרְעָא לֹא תְּבוּר: <B>וְלֹא עַם הָאָרֶץ חָסִיד.</B> אֲבָל יְרֵא חֵטְא אֶפְשָׁר שֶׁיִּהְיֶה, שֶׁהֲרֵי הוּא בָּקִי בְּטִיב מַשָּׂא וּמַתָּן: <B>וְלֹא הַבַּיְשָׁן לָמֵד.</B> שֶׁהַמִּתְבַּיֵּשׁ לִשְׁאֹל שֶׁמָּא יַלְעִיגוּ עָלָיו, יִשָּׁאֵר תָּמִיד בִּסְפֵקוֹתָיו: <B>וְלֹא הַקַּפְּדָן.</B> הָרַב שֶׁמַּקְפִּיד עַל הַתַּלְמִידִים כְּשֶׁשּׁוֹאֲלִים אוֹתוֹ, לֹא יְלַמֵּד כָּרָאוּי. אֶלָּא צָרִיךְ שֶׁיַּסְבִּיר פָּנִים בַּהֲלָכָה לְתַלְמִידָיו: <B>וְלֹא כָל הַמַּרְבֶּה בִסְחוֹרָה מַחְכִּים.</B> דִּכְתִיב (דְּבָרִים ל) וְלֹא מֵעֵבֶר לַיָּם הִיא, אֵין הַתּוֹרָה מְצוּיָה בְּאוֹתָם שֶׁהוֹלְכִים מֵעֵבֶר לַיָּם [עֵרוּבִין נה.]: <B>וּבְמָקוֹם שֶׁאֵין אֲנָשִׁים.</B> לֵישֵׁב בָּרֹאשׁ וּלְהוֹרוֹת הוֹרָאוֹת:

<B>עַל דַּאֲטֵפְתְּ אַטְפוּךְ.</B> רָשָׁע הָיִיתָ וּמְלַסְטֵם אֶת הַבְּרִיּוֹת וּמְצִיפָן בַּנָּהָר, וּבְאוֹתָהּ מִדָּה מָדְדוּ לְךָ: <B>וְסוֹף מְטִיפַיִךְ יְטוּפוּן.</B> שֶׁלֹּא הָיָה הַדָּבָר מָסוּר בְּיָדָן לְהָרְגְךָ, אֶלָּא לְבֵית דִּין, וְהַקָּדוֹשׁ בָּרוּךְ הוּא מְסָרְךָ בְּיָדָן, שֶׁמְּגַלְגְּלִים חוֹבָה עַל יְדֵי חַיָּב, וְעָתִיד הוּא אַחַר כָּךְ לִתְבֹּעַ מֵהֶן מִיתָתְךָ {כג}:

<B>מַרְבֶּה בָשָׂר מַרְבֶּה רִמָּה.</B> הַמַּרְבֶּה בַּאֲכִילָה וּשְׁתִיָּה עַד שֶׁנַּעֲשֶׂה שָׁמֵן וּבַעַל בָּשָׂר, מַרְבֶּה עָלָיו רִמָּה בַּקֶּבֶר, וְקָשָׁה רִמָּה לַמֵּת כְּמַחַט בִּבְשַׂר הַחַי {כד}. וְקָמַשְׁמַע לַן הַאי תַּנָּא, שֶׁכָּל רִבּוּיִין קָשִׁים לָאָדָם חוּץ מֵרִבּוּי תּוֹרָה וְחָכְמָה וּצְדָקָה: <B>מַרְבֶּה דְאָגָה.</B> שֶׁמָּא יִגְזְלוּהוּ מִבֵּית הַמֶּלֶךְ אוֹ שֶׁמָּא יָבֹאוּ עָלָיו לִסְטִים וְיַהַרְגוּהוּ. וְחָסִיד אֶחָד הָיָה מִתְפַּלֵּל, הַמָּקוֹם יַצִּילֵנִי מִפִּזּוּר הַנֶּפֶשׁ. וְשָׁאֲלוּ מִמֶּנּוּ מַהוּ פִּזּוּר הַנֶּפֶשׁ, אָמַר לָהֶם שֶׁיִּהְיוּ לוֹ נְכָסִים מְרֻבִּים מְפֻזָּרִים בִּמְקוֹמוֹת הַרְבֵּה וְצָרִיךְ לְפַזֵּר נַפְשׁוֹ לַחֲשֹׁב לְכָאן וּלְכָאן: <B>מַרְבֶּה נָשִׁים מַרְבֶּה כְשָׁפִים.</B> סִדְרָא נָקֵיט וְאָזֵיל. בִּתְחִלָּה אָדָם מְקַשֵּׁט עַצְמוֹ וְאוֹכֵל וְשׁוֹתֶה וּמַרְבֶּה בָּשָׂר, וְאַחַר כָּךְ מְבַקֵּשׁ לְהַרְבּוֹת נְכָסִים, וּלְאַחַר שֶׁהִרְבָּה נְכָסִים וְרוֹאֶה שֶׁיֵּשׁ בְּיָדוֹ סִפּוּק לְפַרְנֵס נָשִׁים רַבּוֹת הוּא מַרְבֶּה לוֹ נָשִׁים, וּלְאַחַר שֶׁהִרְבָּה נָשִׁים כָּל אַחַת צְרִיכָה שִׁפְחָה שֶׁתְּשַׁמְּשֶׁנָּהּ, הֲרֵי מַרְבֶּה שְׁפָחוֹת, וְכֵיוָן שֶׁיֵּשׁ לוֹ בְנֵי בַּיִת מְרֻבִּין צָרִיךְ לְשָׂדוֹת וּכְרָמִים לְסַפֵּק יַיִן וּמָזוֹן לִבְנֵי בֵּיתוֹ, וּמַרְבֶּה עֲבָדִים לַעֲבֹד הַשָּׂדוֹת וְהַכְּרָמִים. לְכָךְ הֵן שְׁנוּיִין בְּסֵדֶר זֶה: <B>מַרְבֶּה תוֹרָהּ מַרְבֶּה חַיִּים.</B> דִּכְתִיב (דְּבָרִים ל) כִּי הִיא חַיֶּיךָ וְאֹרֶךְ יָמֶיךָ: <B>מַרְבֶּה חָכְמָה.</B> שֶׁנּוֹתֵן טַעַם לִדְבָרָיו וְהַסְּבָרוֹת שֶׁלּוֹ מִתְיַשְּׁבוֹת עַל לֵב הַשּׁוֹמְעִין: <B>מַרְבֶּה יְשִׁיבָה.</B> הַרְבֵּה תַּלְמִידִים מִתְקַבְּצִים וּבָאִים לִשְׁמֹעַ סְבָרוֹתָיו. וְאִית דְּגַרְסֵי, מַרְבֶּה יְשִׁיבָה מַרְבֶּה חָכְמָה, מִי שֶׁמַּרְבֶּה בְּתַלְמִידִים, הַתַּלְמִידִים מְחַדְּדִים אוֹתוֹ וּמַרְבִּין וּמוֹסִיפִין עַל חָכְמָתוֹ: <B>מַרְבֶּה עֵצָה.</B> מִי שֶׁמַּרְבֶּה לִטֹּל עֵצָה: <B>מַרְבֶּה תְבוּנָה.</B> מֵבִין דָּבָר מִתּוֹךְ דָּבָר מֵעֵצוֹת שֶׁיּוֹעֲצָיו נוֹתְנִים לוֹ: <B>מַרְבֶּה צְדָקָה מַרְבֶּה שָׁלוֹם.</B> שֶׁנֶּאֱמַר (יְשַׁעְיָה לב) וְהָיָה מַעֲשֵׂה הַצְּדָקָה שָׁלוֹם: 

~ פרק ב - משנה ח
<B>אַל תַּחֲזִיק טוֹבָה לְעַצְמְךָ.</B> לוֹמַר הַרְבֵּה תּוֹרָה לָמַדְתִּי. וּלְפִי שֶׁהוּא לֹא הִנִּיחַ לֹא מִקְרָא וְלֹא מִשְׁנָה הֲלָכָה וְאַגָּדָה שֶׁלֹּא לָמַד [סֻכָּה כח., בָּבָא בַּתְרָא קלד.], הָיָה אוֹמֵר כֵּן: <B>בּוֹר סוּד.</B> גָּרְסִינַן, וְלֹא בּוֹר סִיד {ל}. כְּלוֹמַר, כְּבוֹר טוּחַ בְּסִיד: <B>שֶׁאֵינוֹ מְאַבֵּד טִפָּה.</B> כָּךְ הוּא אֵינוֹ שׁוֹכֵחַ דָּבָר אֶחָד מִתַּלְמוּדוֹ: <B>אַשְׁרֵי יוֹלַדְתּוֹ.</B> מְאֻשָּׁר בְּמִדּוֹת טוֹבוֹת עַד שֶׁכָּל הָעוֹלָם אוֹמְרִים עָלָיו אַשְׁרֵי מִי שֶׁיְּלָדַתּוּ. וְיֵשׁ אוֹמְרִים, עַל שֵׁם שֶׁהִיא גָּרְמָה לוֹ שֶׁיְּהֵא חָכָם, שֶׁהָיְתָה מְחַזֶּרֶת עַל כָּל בָּתֵּי מִדְרָשׁוֹת שֶׁבְּעִירָהּ וְאוֹמֶרֶת לָהֶם בְּבַקָּשָׁה מִכֶּם בַּקְּשׁוּ רַחֲמִים עַל הָעֻבָּר הַזֶּה שֶׁבְּמֵעַי שֶׁיִּהְיֶה חָכָם, וּמִיּוֹם שֶׁנּוֹלַד לֹא הוֹצִיאָה עֲרַשְׂתּוֹ מִבֵּית הַמִּדְרָשׁ כְּדֵי שֶׁלֹּא יִכָּנְסוּ בְּאָזְנָיו אֶלָּא דִּבְרֵי תּוֹרָה: <B>חָסִיד.</B> שֶׁעוֹשֶׂה לִפְנִים מִשּׁוּרַת הַדִּין: <B>יְרֵא חֵטְא.</B> מַחְמִיר עַל עַצְמוֹ וְאוֹסֵר עָלָיו דְּבָרִים הַמֻּתָּרִים, מִיִּרְאָתוֹ שֶׁמָּא יָבֹא לִידֵי חֵטְא. דְּאִם לֹא כֵן מַאי רְבוּתֵיהּ, אֲפִלּוּ עַם הָאָרֶץ אֶפְשָׁר לִהְיוֹת יְרֵא חֵטְא: <B>מַעְיָן הַמִּתְגַּבֵּר.</B> לִבּוֹ רָחָב וּמוֹסִיף פִּלְפּוּל וּסְבָרוֹת מִדַּעְתּוֹ: <B>אַבָּא שָׁאוּל אוֹמֵר מִשְּׁמוֹ.</B> שֶׁל רַבָּן יוֹחָנָן בֶּן זַכַּאי: <B>וְאֶלְעָזָר בֶּן עֲרָךְ בְּכַף שְׁנִיָּה.</B> מָצָאתִי כָּתוּב, דְּאַבָּא שָׁאוּל לֹא נֶחְלַק עַל תַּנָּא קַמָּא, וּשְׁנֵי הַדְּבָרִים אֲמָרָם רַבָּן יוֹחָנָן בֶּן זַכַּאי, וּשְׁנֵיהֶם אֱמֶת, דִּלְעִנְיַן הַבְּקִיאוּת וְהַזִּכָּרוֹן הָיָה רַבִּי אֱלִיעֶזֶר מַכְרִיעַ, וּלְעִנְיַן הַחֲרִיפוּת וְהַפִּלְפּוּל הָיָה רַבִּי אֶלְעָזָר בֶּן עֲרָךְ מַכְרִיעַ: 

~ פרק ב - משנה ט
<B>עַיִן טוֹבָה.</B> מִסְתַּפֵּק בְּמַה שֶּׁיֵּשׁ לוֹ וְאֵינוֹ מְבַקֵּשׁ דְּבָרִים יְתֵרִים וְאֵינוֹ מְקַנֵּא כְּשֶׁרוֹאֶה שֶׁיֵּשׁ לַחֲבֵרָיו יוֹתֵר מִמֶּנּוּ: <B>חָבֵר טוֹב.</B> שֶׁמּוֹכִיחוֹ כְּשֶׁרוֹאֶה אוֹתוֹ עוֹשֶׂה דָּבָר שֶׁאֵינוֹ הָגוּן: <B>שָׁכֵן טוֹב.</B> מָצוּי אֶצְלוֹ בֵּין בַּיּוֹם בֵּין בַּלַּיְלָה, וְאִלּוּ חָבֵר טוֹב אֵינוֹ מָצוּי אֶצְלוֹ בְּכָל עֵת: <B>הָרוֹאֶה אֶת הַנּוֹלָד.</B> צוֹפֶה וּמַבִּיט בְּמַה שֶּׁעָתִיד לִהְיוֹת, וּמִתּוֹךְ כָּךְ נִמְצָא מְחַשֵּׁב הֶפְסֵד מִצְוָה כְּנֶגֶד שְׂכָרָהּ וּשְׂכַר עֲבֵרָה כְּנֶגֶד הֶפְסֵדָהּ: <B>לֵב טוֹב.</B> לְפִי שֶׁהַלֵּב הוּא הַמֵּנִיעַ לְכָל שְׁאָר הַכֹּחוֹת וְהוּא הַמָּקוֹר שֶׁמִּמֶּנּוּ נוֹבְעִים כָּל הַפְּעֻלּוֹת. וְאַף עַל פִּי שֶׁיֵּשׁ לַפְּעֻלּוֹת אֵבָרִים מְיֻחָדִים, מִכָּל מָקוֹם הַכֹּחַ הַמְעוֹרֵר לְכָל הַתְּנוּעוֹת הוּא בַּלֵּב, לְפִיכָךְ אָמַר רַבִּי אֶלְעָזָר לֵב טוֹב. וּמִפְנֵי זֶה אָמַר רַבָּן יוֹחָנָן בֶּן זַכַּאי רוֹאֶה אֲנִי אֶת דִּבְרֵי אֶלְעָזָר בֶּן עֲרָךְ מִדִּבְרֵיכֶם: <B>אֵיזוֹ הִיא דֶּרֶךְ רָעָה שֶׁיִּתְרַחֵק מִמֶּנָּה הָאָדָם.</B> הֻצְרַךְ לִשְׁאֹל מֵהֶם זֶה, וְלֹא הֵבִין מִתּוֹךְ דִּבְרֵיהֶם הָרִאשׁוֹנִים שֶׁהַדֶּרֶךְ הָרָעָה הִיא הֵפֶךְ הַדֶּרֶךְ הַטּוֹבָה, לְפִי שֶׁלֹּא כָּל מַה שֶּׁהוּא טוֹב יִהְיֶה הֶפְכוֹ רַע, כִּי מִדַּת הַחֲסִידוּת וְהוּא הָעוֹשֶׂה לִפְנִים מִשּׁוּרַת הַדִּין הִיא טוֹבָה, וּמִי שֶׁאֵינוֹ חָסִיד וּמַעֲמִיד דְּבָרָיו עַל דִּין תּוֹרָה אֵינוֹ רַע, וְאֶפְשָׁר הָיָה לוֹמַר, שֶׁאִם הַהִסְתַּפְּקוּת וְהִיא עַיִן טוֹבָה הִיא דֶּרֶךְ טוֹבָה, בַּקָּשַׁת הַיִּתְרוֹנוֹת אֵינוֹ דֶּרֶךְ רָעָה, לְפִי שֶׁאֵינוֹ מַזִּיק לְשׁוּם אָדָם בָּזֶה, וְכֵן בִּשְׁאָר הַמִּדּוֹת. לְפִיכָךְ הֻצְרַךְ לִשְׁאֹל מֵהֶם אֵיזוֹ הִיא דֶּרֶךְ רָעָה שֶׁיִּתְרַחֵק מִמֶּנָּה הָאָדָם: <B>הַלּוֶֹה וְאֵינוֹ מְשַׁלֵּם.</B> הוּא הֵפֶךְ הָרוֹאֶה אֶת הַנּוֹלָד, שֶׁאִם לֹא יְשַׁלֵּם לֹא יִמְצָא מִי שֶׁיַּלְוֶנּוּ וְיִהְיֶה מֻטָּל בָּרָעָב. וְלֹא אָמַר סְתָם מִי שֶׁאֵינוֹ רוֹאֶה אֶת הַנּוֹלָד, לְפִי שֶׁאֶפְשָׁר לָאָדָם שֶׁאֵינוֹ רוֹאֶה אֶת הַנּוֹלָד שֶׁלֹּא יָבֹא לִידֵי תַּקָּלָה, שֶׁיַּצִּיל עַצְמוֹ בְּבֹא הַנּוֹלָד הַהוּא {לד}: <B>וְצַדִּיק חוֹנֵן וְנוֹתֵן.</B> הַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁהוּא צַדִּיקוֹ שֶׁל עוֹלָם {לו} חוֹנֵן וְנוֹתֵן לַמַּלְוֶה מַה שֶּׁלָּוָה זֶה מִמֶּנּוּ וְלֹא פָּרַע לוֹ, נִמְצָא הַלּוֶֹה נִשְׁאַר חַיָּב לַמָּקוֹם:

<B>הֵם אָמְרוּ שְׁלֹשָׁה דְּבָרִים.</B> בְּדֶרֶךְ אֶרֶץ וּבְמוּסָרִים וּבְמִדּוֹת. דְּאִלּוּ בְּעִנְיַן הָאָסוּר וְהַמֻּתָּר וְהַפְּטוּר וְהַחִיּוּב דְּבָרִים הַרְבֵּה אָמְרוּ. אִי נַמִּי, אֵלּוּ הַשְּׁלֹשָׁה דְּבָרִים הֲוֵי מַרְגְּלָא בְּפוּמַיְהוּ תָּמִיד: <B>יְהִי כְבוֹד חֲבֵרְךָ חָבִיב עָלֶיךָ כְּשֶׁלָּךְ.</B> וְאֵימָתַי יְהֵא זֶה, כְּשֶׁלֹּא תְּהִי נוֹחַ לִכְעֹס, שֶׁאִם אַתָּה נוֹחַ לִכְעֹס אִי אֶפְשָׁר שֶׁלֹּא תְּזַלְזֵל בִּכְבוֹד חֲבֵרְךָ. הֲרֵי דִּבּוּר אֶחָד. וְשׁוּב יוֹם אֶחָד לִפְנֵי מִיתָתְךָ, הֲרֵי שְׁנַיִם. וְהַשְּׁלִישִׁי, הֱוֵי מִתְחַמֵּם כְּנֶגֶד אוּרָן שֶׁל חֲכָמִים וְכוּ': <B>שׁוּב יוֹם אֶחָד לִפְנֵי מִיתָתֶךָ.</B> וּלְפִי שֶׁאֵין אָדָם יוֹדֵעַ אֵימָתַי יָמוּת, יַעֲשֶׂה תְּשׁוּבָה הַיּוֹם שֶׁמָּא יָמוּת לְמָחָר: <B>וֶהֱוֵי זָהִיר בְּגַחַלְתָּן.</B> שֶׁלֹּא תִּנְהַג בָּהֶן קַלּוּת רֹאשׁ {לח}: <B>שֶׁלֹּא תִכָּוֶה.</B> שֶׁלֹּא תֵּעָנֵשׁ עַל יָדָן: <B>נְשִׁיכַת שׁוּעָל.</B> רְפוּאָתוֹ קָשָׁה, לְפִי שֶׁשִּׁנָּיו דַּקּוֹת עֲקֻמּוֹת וַעֲקֻשּׁוֹת וְצָרִיךְ הָרוֹפֵא לַחְתֹּךְ הַבָּשָׂר בְּאִזְמֵל וּלְהַרְחִיב הַנְּשִׁיכָה: <B>עֲקִיצַת עַקְרָב.</B> קָשָׁה מִנְּשִׁיכַת נָחָשׁ: <B>וּלְחִישָׁתָן.</B> דִּבּוּרָן: <B>לְחִישַׁת שָׂרָף.</B> כְּהֶבֶל הַיּוֹצֵא מִפִּי הַשָּׂרָף, שֶׁשּׂוֹרֵף בְּשָׁעָה שֶׁהוּא לוֹחֵשׁ. פֵּרוּשׁ אַחֵר, הַשָּׂרָף הַזֶּה אֵינוֹ מְקַבֵּל לַחַשׁ כִּשְׁאָר נְחָשִׁים, כְּדִכְתִיב (תְּהִלִּים נח) אֲשֶׁר לֹא יִשְׁמַע לְקוֹל מְלַחֲשִׁים {מ}, כָּךְ תַּלְמִיד חָכָם אִם תַּקְנִיטֶנּוּ וְתָבֹא לְפַיְּסוֹ אֵינוֹ מְקַבֵּל פִּיּוּס {מא}:

~ פרק ג - משנה א
<B>עֲקַבְיָא.</B> מִטִּפָּה סְרוּחָה. טִפַּת שִׁכְבַת זֶרַע. וְאַף עַל פִּי שֶׁבִּשְׁעַת הָעִבּוּר אֵינָהּ סְרוּחָה, שֶׁאֵינָהּ מַסְרַחַת בְּרֶחֶם הָאִשָּׁה עַד לְאַחַר שְׁלֹשָׁה יָמִים וּכְשֶׁתַּסְרִיחַ אֵינָהּ רְאוּיָה לְהַזְרִיעַ, מִכָּל מָקוֹם קָרֵי לָהּ טִפָּה סְרוּחָה מִפְּנֵי שֶׁהִיא קְרוֹבָה לְהַסְרִיחַ מִיָּד כְּשֶׁהִיא חוּץ לִמְעֵי הָאִשָּׁה. וְהַמִּסְתַּכֵּל שֶׁבָּא מִטִּפָּה סְרוּחָה, נִצּוֹל מִן הַגַּאֲוָה. וְהַמִּסְתַּכֵּל שֶׁעָתִיד לֵילֵךְ לִמְקוֹם עָפָר רִמָּה וְתוֹלֵעָה, נִצּוֹל מִן הַתַּאֲוָה וְהַחֶמְדָּה אֶל הַמָּמוֹן. וְהַמִּסְתַּכֵּל שֶׁעָתִיד לִתֵּן דִּין וְחֶשְׁבּוֹן, פּוֹרֵשׁ מִן הַחֵטְא וְאֵינוֹ נִכְשָׁל בָּעֲבֵרָה: 

~ פרק ג - משנה ב
<B>בִּשְׁלוֹמָהּ שֶׁל מַלְכוּת.</B> וַאֲפִלּוּ שֶׁל אֻמּוֹת הָעוֹלָם: <B>חַיִּים בְּלָעוֹ.</B> דִּכְתִיב (חֲבַקּוּק א) וַתַּעֲשֶׂה אָדָם כִּדְגֵי הַיָּם, מַה דָּגִים שֶׁבַּיָּם כָּל הַגָּדוֹל מֵחֲבֵרוֹ בּוֹלֵעַ אֶת חֲבֵרוֹ, אַף בְּנֵי אָדָם אִלְמָלֵא מוֹרָאָהּ שֶׁל מַלְכוּת כָּל הַגָּדוֹל מֵחֲבֵרוֹ בּוֹלֵעַ אֶת חֲבֵרוֹ [עֲבוֹדָה זָרָה נד:]: <B>יִרְאֵי ה' אִישׁ אֶל רֵעֵהוּ.</B> הֲרֵי כָּאן שְׁנַיִם: <B>וְיִדֹּם.</B> לְשׁוֹן קוֹל דְּמָמָה דַּקָּה, כְּדֶרֶךְ הַשּׁוֹנֶה יְחִידִי שֶׁהוּא שׁוֹנֶה בְּלַחַשׁ: <B>כִּי נָטַל עָלָיו.</B> כְּאִלּוּ נְתִינַת כָּל הַתּוֹרָה כֻּלָּהּ הָיְתָה בַּעֲבוּרוֹ בִּלְבַד {יא}: 

~ פרק ג - משנה ג
<B>וְלֹא אָמְרוּ עָלָיו דִּבְרֵי תוֹרָה.</B> וּבְבִרְכַּת הַמָּזוֹן שֶׁמְּבָרְכִים עַל הַשֻּׁלְחָן, יוֹצְאִין יְדֵי חוֹבָתָן, וְחָשׁוּב כְּאִלּוּ אָמְרוּ עָלָיו דִּבְרֵי תּוֹרָה. כָּךְ שָׁמַעְתִּי {יג}: <B>מִזִּבְחֵי מֵתִים.</B> תִּקְרֹבֶת עֲבוֹדָה זָרָה, דִּכְתִיב (תְּהִלִּים קו) וַיִּצָּמְדוּ לְבַעַל פְּעוֹר וַיֹּאכְלוּ זִבְחֵי מֵתִים: <B>מָלְאוּ קִיא צֹאָה.</B> וַעֲבוֹדָה זָרָה קְרוּיָה צֹאָה, דִּכְתִיב (יְשַׁעְיָה ל) צֵא תֹּאמַר לוֹ: <B>בְּלִי מָקוֹם.</B> בִּשְׁבִיל שֶׁלֹּא הִזְכִּירוּ שְׁמוֹ שֶׁל מָקוֹם בָּרוּךְ הוּא עַל הַשֻּׁלְחָן: <B>שֶׁנֶּאֱמָר וַיְּדַבֵּר אֵלַי זֶה הַשֻּׁלְחָן אֲשֶׁר לִפְנֵי ה'.</B> מִיָּד כְּשֶׁדִּבֵּר בְּדִבְרֵי תּוֹרָה נִקְרָא שֻׁלְחָן אֲשֶׁר לִפְנֵי ה'. וְיֵשׁ אוֹמְרִים מֵרֹאשׁוֹ שֶׁל מִקְרָא הוּא יוֹצֵא, דִּכְתִיב וְהַמִּזְבֵּחַ עֵץ שָׁלֹשׁ אַמּוֹת, אַל תִּקְרֵי אַמּוֹת אֶלָּא אִמּוֹת, כְּמוֹ יֵשׁ אֵם לַמִּקְרָא, שָׁלֹשׁ כְּנֶגֶד תּוֹרָה נְבִיאִים וּכְתוּבִים, וְאָמְרֵי לַהּ מִקְרָא מִשְׁנָה וְתַלְמוּד. שֶׁצָּרִיךְ אָדָם לְדַבֵּר בָּהֶן עַל הַשֻּׁלְחָן, וְאָז קָרוּי שֻׁלְחָן אֲשֶׁר לִפְנֵי ה'. כָּךְ פֵּרֵשׁ רַשִׁ"י: 

~ פרק ג - משנה ד
<B>הַנֵּעוֹר בַּלַּיְלָה וְהַמְהַלֵּךְ בַּדֶּרֶךְ יְחִידִי.</B> וּמְחַשֵּׁב בְּלִבּוֹ דִּבְרֵי הֲבַאי: <B>הֲרֵי זֶה מִתְחַיֵּב בְּנַפְשׁוֹ.</B> לְפִי שֶׁלַּיְלָה הוּא זְמַן לַמַּזִּיקִים, וְהַמְהַלֵּךְ בַּדֶּרֶךְ יְחִידִי הוּא בְּסַכָּנָה מִפְּנֵי הַלִּסְטִים וְכַמָּה פְּגָעִים רָעִים, וְאִם הָיָה מְחַשֵּׁב בְּדִבְרֵי תּוֹרָה הָיְתָה מְשַׁמַּרְתּוֹ: 

~ פרק ג - משנה ה
<B>עֹל מַלְכוּת.</B> מַשָּׂא מֶלֶךְ וְשָׂרִים: <B>דֶּרֶךְ אֶרֶץ.</B> עֲמַל וְטֹרַח הַפַּרְנָסָה, לְפִי שֶׁמְּלַאכְתּוֹ מִתְבָּרֶכֶת: <B>הַפּוֹרֵק מִמֶּנּוּ עֹל תּוֹרָה.</B> הָאוֹמֵר קָשֶׁה עֻלָּהּ שֶׁל תּוֹרָה וְאֵינִי יָכוֹל לְסוֹבְלָהּ {יז}: 

~ פרק ג - משנה ו
<B>עֲשָׂרָה שֶׁהָיוּ יוֹשְׁבִים בַּדִּין.</B> גָּרְסִינַן {יח}: <B>בַּעֲדַת אֵל.</B> וְאֵין עֵדָה פְּחוּתָה מֵעֲשָׂרָה, שֶׁנֶּאֱמַר בַּמְרַגְּלִים (בְּמִדְבָּר יד) עַד מָתַי לָעֵדָה הָרָעָה הַזֹּאת, יָצְאוּ יְהוֹשֻׁעַ וְכָלֵב, הֲרֵי עֲשָׂרָה: <B>וּמִנַּיִן אֲפִלּוּ חֲמִשָּׁה שֶׁנֶּאֱמַר בְּקֶרֶב אֱלֹהִים יִשְׁפֹּט.</B> שְׁלֹשָׁה דַּיָּנִים וּשְׁנֵי בַּעֲלֵי דִּינִים: <B>וּמִנַּיִן אֲפִלּוּ שְׁלֹשָׁה שֶׁנֶּאֱמַר וַאֲגֻדָּתוֹ עַל אֶרֶץ יְסָדָהּ.</B> הָאֵשׁ וְהָאֲוִיר וְהַמַּיִם שֶׁהֵם שְׁלֹשָׁה, עַל אֶרֶץ יְסָדָהּ, עַל יְסוֹד הָאָרֶץ הֵם מַקִּיפִים. הֲרֵי לְךָ שֶׁשְּׁלֹשָׁה קְרוּיִין אֲגֻדָּה. אִי נַמִּי, מָצִינוּ שְׁלֹשָׁה קְרוּיִין אֲגֻדָּה, אֲגֻדַּת אֵזוֹב שֶׁהֵם שְׁלֹשָׁה קְלָחִים. וְיֵשׁ סְפָרִים שֶׁכָּתוּב בָּהֵן וּמִנַּיִן אֲפִלּוּ חֲמִשָּׁה שֶׁנֶּאֱמַר וַאֲגֻדָּתוֹ עַל אֶרֶץ יְסָדָהּ, שֶׁאָדָם אוֹגֵד בְּיָדוֹ אַחַת שֶׁיֵּשׁ בָּהּ חָמֵשׁ אֶצְבָּעוֹת, וּכְלַל אֶצְבָּעוֹת שֶׁבַּיָּד קְרוּיִין אֲגֻדָּה. וּבְרֹאשׁ הַמִּקְרָא הוּא אוֹמֵר, הַבּוֹנֶה בַּשָּׁמַיִם מַעֲלוֹתָיו, כְּלוֹמַר הַשְּׁכִינָה שֶׁהִיא בַּשָּׁמַיִם יוֹרֶדֶת לְמַטָּה לָאָרֶץ כְּשֶׁיֵּשׁ שָׁם אֲגֻדָּה עוֹסְקִים בַּתּוֹרָה {יט}. וּמִנַּיִן אֲפִלּוּ שְׁלֹשָׁה שֶׁנֶּאֱמַר בְּקֶרֶב אֱלֹהִים יִשְׁפֹּט, שֶׁהַדַּיָּנִים הֵם שְׁלֹשָׁה: <B>כִּי נָטַל עָלָיו.</B> לְשׁוֹן סְכָךְ. תַּרְגּוּם וְסַכּוֹתָ, וְתַטֵּל, כְּלוֹמַר שֶׁהַשְּׁכִינָה סוֹכֶכֶת עָלָיו: 

~ פרק ג - משנה ז
<B>תֵּן לוֹ מִשֶּׁלּוֹ.</B> לֹא תִּמָּנַע מִלְּהִתְעַסֵּק בְּחֶפְצֵי שָׁמַיִם בֵּין בְּגוּפְךָ בֵּין בְּמָמוֹנְךָ, [שֶׁאֵינְךָ נוֹתֵן מִשֶּׁלְּךָ, לֹא מִגּוּפְךָ וְלֹא מִמָּמוֹנְךָ], שֶׁאַתָּה וּמָמוֹנְךָ שֶׁלּוֹ: <B>מַה נָּאֶה אִילָן זֶה מַה נָּאֶה נִיר זֶה.</B> הוּא הַדִּין לְכָל שִׂיחָה בְּטֵלָה, אֶלָּא שֶׁדִּבֵּר בַּהֹוֶה שֶׁדֶּרֶךְ הוֹלְכֵי דְּרָכִים לְדַבֵּר בְּמַה שֶּׁרוֹאִים בְּעֵינֵיהֶם. וְיֵשׁ אוֹמְרִים, דְּאַשְׁמוֹעִינַן רְבוּתָא דְּאַף עַל גַּב דְּעַל יְדֵי כֵן הוּא מְבָרֵךְ בָּרוּךְ שֶׁכָּכָה לוֹ בְּעוֹלָמוֹ, אַף עַל פִּי כֵן מַעֲלִין עָלָיו כְּאִלּוּ מִתְחַיֵּב בְּנַפְשׁוֹ, מִפְּנֵי שֶׁהִפְסִיק מִמִּשְׁנָתוֹ: <B>נִיר.</B> תֶּלֶם הַמַּחֲרֵשָׁה, כְּמוֹ נִירוּ לָכֶם נִיר (יִרְמְיָה ד): 

~ פרק ג - משנה ח
<B>כָּל הַשּׁוֹכֵחַ דָּבָר אֶחָד מִמִּשְׁנָתוֹ.</B> בִּשְׁבִיל שֶׁלֹּא חָזַר עָלֶיהָ, מַעֲלִין עָלָיו כְּאִלּוּ מִתְחַיֵּב בְּנַפְשׁוֹ. שֶׁמִּתּוֹךְ שִׁכְחָתוֹ הוּא בָּא לְהַתִּיר אֶת הָאָסוּר וְנִמְצֵאת תַּקָּלָה בָּאָה עַל יָדוֹ, וְשִׁגְגָתוֹ עוֹלָה זָדוֹן. אִי נַמִּי כְּאִלּוּ מִתְחַיֵּב בְּנַפְשׁוֹ, לְפִי שֶׁאוֹתָהּ מִשְׁנָה הָיְתָה מְשַׁמַּרְתּוֹ וְעַכְשָׁיו שֶׁשְּׁכָחָהּ אֵינָהּ מְשַׁמַּרְתּוֹ: <B>תָּקְפָה עָלָיו מִשְׁנָתוֹ.</B> שֶׁהָיְתָה קָשָׁה עָלָיו וּמִתּוֹךְ הַקֹּשִׁי שֶׁבָּהּ שְׁכָחָהּ: 

~ פרק ג - משנה ט
<B>כָּל שֶׁיִּרְאַת חֶטְאוֹ קוֹדֶמֶת לְחָכְמָתוֹ.</B> אֲנִי שָׁמַעְתִּי, שֶׁמַּקְדִּים בְּמַחֲשַׁבְתּוֹ יִרְאַת חֶטְאוֹ לְחָכְמָתוֹ, שֶׁהוּא חוֹשֵׁב בְּלִבּוֹ אֶלְמַד בִּשְׁבִיל שֶׁאֶהְיֶה יְרֵא חֵטְא {כח}. וְזֶה עַל דֶּרֶךְ שֶׁאָמְרוּ תְּחִלַּת הַמַּחֲשָׁבָה [הִיא] סוֹף הַמַּעֲשֶׂה: <B>חָכְמָתוֹ מִתְקַיֶּמֶת.</B> שֶׁהַחָכְמָה מְבִיאָתוֹ לְמַה שֶּׁלִּבּוֹ חָפֵץ וְהוּא נֶהֱנֶה בָּהּ: <B>וְכָל שֶׁחָכְמָתוֹ קוֹדֶמֶת לְיִרְאַת חֶטְאוֹ.</B> שֶׁאֵינוֹ לוֹמֵד עַל מְנָת לַעֲשׂוֹת, הוֹאִיל וְאֵין לְבָבוֹ פּוֹנֶה לִהְיוֹת יְרֵא חֵטְא: <B>אֵין חָכְמָתוֹ מִתְקַיֶּמֶת.</B> שֶׁמִּתּוֹךְ שֶׁהִיא מוֹנַעַת אוֹתוֹ מִלָּלֶכֶת אַחֲרֵי שְׁרִירוּת לִבּוֹ הוּא קָץ בָּהּ וּמוֹאֲסָהּ וּמַנִּיחָהּ: <B>כָּל שֶׁמַּעֲשָׂיו מְרֻבִּים מֵחָכְמָתוֹ.</B> הָכָא מַיְרֵי שֶׁזָּרִיז בְּמִצְוֹת עֲשֵׂה {לא}, וּלְעֵיל בְּכָל שֶׁיִּרְאַת חֶטְאוֹ קוֹדֶמֶת לְחָכְמָתוֹ אַיְרֵי שֶׁנִּזְהָר בְּמִצְוֹת לֹא תַעֲשֶׂה: 

~ פרק ג - משנה י
<B>כָּל שֶׁרוּחַ הַבְּרִיּוֹת נוֹחָה הֵימֶנּוּ.</B> כָּל מִי שֶׁאָהוּב לְמַטָּה בְּיָדוּעַ שֶׁהוּא אָהוּב לְמַעְלָה {לב}: <B>שֵׁנָה שֶׁל שַׁחֲרִית.</B> שֶׁהוֹלֵךְ וְיָשֵׁן עַד שֶׁעוֹנַת קְרִיאַת שְׁמַע עוֹבֶרֶת: <B>וְיַיִן שֶׁל צָהֳרַיִם.</B> מוֹשֵׁךְ לִבּוֹ שֶׁל אָדָם, דִּכְתִיב [קֹהֶלֶת ב] לִמְשֹׁךְ בַּיַּיִן אֶת בְּשָׂרִי, וּמֵבִיא אוֹתוֹ לִידֵי שִׁכְרוּת {לד}: <B>וְשִׂיחַת הַיְלָדִים.</B> מְבַטֶּלֶת אֶת אֲבוֹתֵיהֶם מִלַּעֲסֹק בַּתּוֹרָה: <B>וִישִׁיבַת בָּתֵּי כְנֵסִיּוֹת שֶׁל עַמֵּי הָאָרֶץ.</B> שֶׁמִּתְכַּנְּסִים וּמְדַבְּרִים בִּדְבָרִים בְּטֵלִים: 

~ פרק ג - משנה יא
<B>הַמְחַלֵּל אֶת הַקָּדָשִׁים.</B> הַמֵּבִיא קָדְשֵׁי הַמִּזְבֵּחַ לִידֵי פִּגּוּל וְנוֹתָר וְטָמֵא, אוֹ הַמּוֹעֵל וְנֶהֱנֶה בֵּין בְּקָדְשֵׁי מִזְבֵּחַ בֵּין בְּקָדְשֵׁי בֶּדֶק הַבַּיִת: <B>וְהַמְבַזֶּה אֶת הַמּוֹעֲדוֹת.</B> יָמִים שֶׁל חֹל הַמּוֹעֵד, עוֹשֶׂה בָּהֶן מְלָאכָה אוֹ נוֹהֵג בָּהֶן מִנְהַג חֹל בַּאֲכִילָה וּשְׁתִיָּה: <B>וְהַמַּלְבִּין פְּנֵי חֲבֵרוֹ בָרַבִּים.</B> הַמִּתְבַּיֵּשׁ פָּנָיו מַאֲדִימוֹת תְּחִלָּה וְאַחַר כָּךְ מִתְלַבְּנוֹת, שֶׁהַנֶּפֶשׁ יֵשׁ לָהּ שְׁתֵּי תְּנוּעוֹת אַחַת לַחוּץ וְאַחַת לִפְנִים, וּכְשֶׁמְּבַיְּשִׁין אֶת הָאָדָם בִּתְחִלָּה הָרוּחַ מִתְנוֹעֵעַ לְצַד חוּץ כְּמִי שֶׁמִּתְמַלֵּא חֵמָה וּפָנָיו מַאֲדִימוֹת, וּכְשֶׁאֵינוֹ מוֹצֵא טַעֲנָה כֵּיצַד יָסִיר הַבֹּשֶׁת הַהוּא מֵעַל פָּנָיו, הוּא דּוֹאֵג בְּקִרְבּוֹ וְנִכְנַס הָרוּחַ לְצַד פָּנָיו מִפְּנֵי הַצַּעַר, וּפָנָיו מִתְכַּרְכְּמִים וּמִתְלַבְּנִים. וְזֶהוּ שֶׁאָמְרוּ [בָּבָא מְצִיעָא נח:] בְּעִנְיַן הַהַלְבָּנָה, דְּאַזִּיל סוּמְקָא וְאָתִי חִיוְרָא: <B>וְהַמֵּפֵר בְּרִיתוֹ.</B> שֶׁלֹּא מָל, אוֹ שֶׁמָּל וּמוֹשֵׁךְ עָרְלָתוֹ לְכַסּוֹת הַמִּילָה כְּדֵי שֶׁלֹּא יֵרָאֶה שֶׁהוּא מָהוּל: <B>וְהַמְגַלֶּה פָנִים בַּתּוֹרָה.</B> שֶׁמַּרְאֶה פָּנִים וּפֵרוּשִׁים בַּתּוֹרָה שֶׁלֹּא כַּהֲלָכָה. כְּגוֹן הַמְתַרְגֵּם וּמִזַּרְעֲךָ לֹא תִתֵּן לְהַעֲבִיר לַמּוֹלֵךְ, וּמִזַּרְעֲךָ לֹא תִתֵּן לְאַעֲבָרָא לְאַרְמִיּוּתָא. וְאֵין זֶה פְּשׁוּטוֹ שֶׁל מִקְרָא. וּבִכְלָל זֶה הַדּוֹרֵשׁ דְּרָשׁוֹת שֶׁל דֹּפִי {לו}. פֵּרוּשׁ אַחֵר, מְגַלֶּה פָּנִים, שֶׁמֵּעֵז פָּנָיו לַעֲבֹר עַל דִּבְרֵי תּוֹרָה בְּפַרְהֶסְיָא בְּיָד רָמָה וְאֵין לוֹ בּשֶׁת פָּנִים: <B>אַף עַל פִּי שֶׁיֵּשׁ בְּיָדוֹ תּוֹרָה וּמַעֲשִׂים טוֹבִים.</B> וְלֹא חָזַר בִּתְשׁוּבָה מֵאַחַת מֵעֲבֵרוֹת הַלָּלוּ שֶׁבְּיָדוֹ, אַף עַל פִּי שֶׁבָּאוּ עָלָיו יִסּוּרִים וּמֵת בַּיִּסּוּרִים {לז}, אֵין לוֹ חֵלֶק לָעוֹלָם הַבָּא. אֲבָל אִם חָזַר בִּתְשׁוּבָה קֹדֶם מוֹתוֹ, אֵין לְךָ דָּבָר שֶׁעוֹמֵד בִּפְנֵי הַתְּשׁוּבָה: 

~ פרק ג - משנה יב
<B>הֱוֵי קַל לְרֹאשׁ.</B> לִפְנֵי אָדָם גָּדוֹל זָקֵן וְיוֹשֵׁב בָּרֹאשׁ בַּיְּשִׁיבָה, הֱוֵי קַל לַעֲבֹד עֲבוֹדָתוֹ וּלְשַׁמֵּשׁ לְפָנָיו: <B>וְנוֹחַ לְתִשְׁחֹרֶת.</B> לְאָדָם בָּחוּר שֶׁשַּׂעֲרוֹתָיו שְׁחֹרוֹת אֵינְךָ צָרִיךְ לְהָקֵל עַצְמְךָ כָּל כָּךְ, אֶלָּא תַּעֲמֹד לְפָנָיו בְּנַחַת וּבְיִשּׁוּב: <B>וֶהֱוֵי מְקַבֵּל וְכוּ'.</B> נֶגֶד כָּל אָדָם בֵּין רֹאשׁ בֵּין תִּשְׁחֹרֶת תַּעֲמֹד בְּשִׂמְחָה. מְקַבֵּל, כְּמוֹ מַקְבִּיל. מִתַּרְגּוּם נֶגֶד, קֳבֵל. פֵּרוּשׁ אַחֵר, הֱוֵי קַל לְרֹאשׁ, בְּרֵאשִׁיתְךָ כְּשֶׁאַתָּה בָּחוּר הֱוֵי קַל לַעֲשׂוֹת רְצוֹן בּוֹרַאֲךָ, וּבְזִקְנוּתְךָ כְּשֶׁהֻשְׁחֲרוּ פָּנֶיךָ מִפְּנֵי הַזִּקְנָה תְּהֵא נוֹחַ לוֹ: 

~ פרק ג - משנה יג
<B>מָסֹרֶת סְיָג לַתּוֹרָה.</B> הַמָּסֹרֶת שֶׁמָּסְרוּ לָנוּ חֲכָמִים בַּחֲסֵרוֹת וִיתֵרוֹת שֶׁבַּתּוֹרָה, הֵם גָּדֵר וְחִזּוּק לַתּוֹרָה שֶׁבִּכְתָב, שֶׁעַל יְדֵיהֶם אָנוּ מְבִינִים כַּמָּה מִצְוֹת אֵיךְ יַעֲשׂוּ אוֹתָם. כְּמוֹ בַּסֻּכּוֹת בַּסֻּכֹּת בַּסֻּכֹּת, שְׁנַיִם חֲסֵרִים וְאֶחָד מָלֵא, שֶׁמִּמֶּנּוּ אָנוּ לוֹמְדִים לְהַכְשִׁיר סֻכָּה בְּשָׁלֹשׁ דְּפָנוֹת. וּכְגוֹן מוֹעֲדֵי ה' אֲשֶׁר תִּקְרְאוּ אֹתָם, שֶׁנִּכְתָּב חָסֵר בִּשְׁלֹשָׁה מְקוֹמוֹת, לְלַמֵּד, אַתֶּם אֲפִלּוּ שׁוֹגְגִים, אַתֶּם אֲפִלּוּ מְזִידִים, אַתֶּם אֲפִלּוּ מֻטְעִים {מ}: <B>מַעַשְׂרוֹת סְיָג לָעֹשֶׁר.</B> דִּכְתִיב עַשֵּׂר תְּעַשֵּׂר, עַשֵּׂר בִּשְׁבִיל שֶׁתִּתְעַשֵּׁר [תַּעֲנִית ט]: <B>נְדָרִים סְיָג לַפְּרִישׁוּת.</B> בִּזְמַן שֶׁאָדָם מַתְחִיל בִּפְרִישׁוּת וּמִתְיָרֵא שֶׁלֹּא יַעֲבֹר, מְקַבֵּל עָלָיו בִּלְשׁוֹן נֶדֶר שֶׁלֹּא יַעֲשֶׂה כָּךְ וְכָךְ, וְעַל יְדֵי כֵן הוּא כּוֹבֵשׁ אֶת יִצְרוֹ: <B>סְיָג לַחָכְמָה שְׁתִיקָה.</B> בְּמַה אֲנַן קָיְמִין, אִי בִּשְׁתִיקָה מִדִּבְרֵי תּוֹרָה, הֲרֵי כְּבָר כְּתִיב וְהָגִיתָ בּוֹ וְגוֹ', אִי בִּשְׁתִיקָה מֵרְכִילוּת וְלָשׁוֹן הָרָע וּקְלָלָה, דְּאוֹרַיְתָא נִינְהוּ. הָא אֵינוֹ מְדַבֵּר אֶלָּא בִּשְׁתִיקָה מִדִּבְרֵי הָרְשׁוּת שֶׁבֵּין אָדָם לַחֲבֵרוֹ, שֶׁיֵּשׁ לוֹ לָאָדָם לְמַעֵט הַדִּבּוּר בָּהֶם כָּל מַה שֶּׁאֶפְשָׁר. וַעֲלֵיהֶם אָמַר שְׁלֹמֹה (מִשְׁלֵי יז) גַּם אֱוִיל מַחֲרִישׁ חָכָם יֵחָשֵׁב {מג}: 

~ פרק ג - משנה יד
<B>חִבָּה יְתֵרָה נוֹדַעַת לוֹ.</B> רַמְבַּ"ם פֵּרֵשׁ, חִבָּה יְתֵרָה הֶרְאָה הַקָּדוֹשׁ בָּרוּךְ הוּא לָאָדָם שֶׁהוֹדִיעוֹ וְאָמַר לוֹ רְאֵה שֶׁבְּרָאתִיךָ בְּצֶלֶם, שֶׁהַמֵּטִיב לַחֲבֵרוֹ וּמוֹדִיעוֹ הַטּוֹבָה שֶׁעָשָׂה עִמּוֹ, מַרְאֶה חִבָּה יְתֵרָה יוֹתֵר מִשֶּׁאִלּוּ הֵטִיב עִמּוֹ וְאֵינוֹ חָשׁוּב בְּעֵינָיו לְהוֹדִיעוֹ הַטּוֹבָה שֶׁעָשָׂה עִמּוֹ. וְיֵשׁ לְפָרֵשׁ, חִבָּה יְתֵרָה נוֹדַעַת לוֹ, חִבָּה גְּלוּיָה וּמְפֻרְסֶמֶת, שֶׁלֹּא בִּלְבַד אַהֲבָה מְסֻתֶּרֶת הָיָה לוֹ לַמָּקוֹם בָּרוּךְ הוּא עִם הָאָדָם אֶלָּא אַף חִבָּה גְּלוּיָה וִידוּעָה לַכֹּל: <B>שֶׁנֶּאֱמַר כִּי לֶקַח טוֹב.</B> כָּל מַעֲשֵׂה בְרֵאשִׁית שֶׁנֶּאֱמַר בּוֹ וַיַּרְא אֱלֹהִים כִּי טוֹב, לֹא נִבְרָא אֶלָּא בִּשְׁבִיל הַתּוֹרָה שֶׁנִּקְרֵאת לֶקַח, כְּמַה דְּאַתְּ אָמַר (דְּבָרִים לב) יַעֲרֹף כַּמָּטָר לִקְחִי: 

~ פרק ג - משנה טו
<B>הַכֹּל צָפוּי.</B> כָּל מַה שֶּׁאָדָם עוֹשֶׂה בְּחַדְרֵי חֲדָרִים גָּלוּי לְפָנָיו: <B>וְהָרְשׁוּת נְתוּנָה.</B> בְּיָדוֹ שֶׁל אָדָם לַעֲשׂוֹת טוֹב וָרַע, כְּדִכְתִיב (שָׁם ל) רְאֵה נָתַתִּי לְפָנֶיךָ הַיּוֹם אֶת הַחַיִּים וְגוֹ': <B>וּבְטוֹב הָעוֹלָם נִדּוֹן.</B> בְּמִדַּת רַחֲמִים. וְאַף עַל פִּי כֵן אֵין הַכֹּל שָׁוִין בְּמִדָּה זוֹ, לְפִי שֶׁהַכֹּל לְפִי רֹב הַמַּעֲשֶׂה. הַמַּרְבֶּה בְּמַעֲשִׂים טוֹבִים נוֹתְנִים לוֹ רַחֲמִים מְרֻבִּים, וְהַמַּמְעִיט בְּמַעֲשִׂים טוֹבִים מְמַעֲטִים לוֹ בְּרַחֲמִים. פֵּרוּשׁ אַחֵר, וְהַכֹּל לְפִי רֹב הַמַּעֲשֶׂה, לְפִי רֹב מַעֲשָׂיו שֶׁל אָדָם הוּא נִדּוֹן, אִם רֹב זְכֻיּוֹת זַכַּאי, אִם רֹב עֲוֹנוֹת חַיָּב. וְרַמְבַּ"ם פֵּרֵשׁ, הַכֹּל צָפוּי, כָּל מַעֲשֵׂה בְנֵי אָדָם מַה שֶּׁעָשָׂה וּמַה שֶּׁעָתִיד לַעֲשׂוֹת הַכֹּל גָּלוּי לְפָנָיו, וְלֹא תֹּאמַר כֵּיוָן שֶׁהַקָּדוֹשׁ בָּרוּךְ הוּא יוֹדֵעַ מַה שֶּׁיַּעֲשֶׂה הָאָדָם אִם כֵּן הוּא מֻכְרָח בְּמַעֲשָׂיו שֶׁיִּהְיֶה צַדִּיק אוֹ רָשָׁע, כִּי הָרְשׁוּת נְתוּנָה בְּיָדוֹ לַעֲשׂוֹת טוֹב וָרַע וְאֵין שׁוּם דָּבָר שֶׁיַּכְרִיחֵהוּ כְּלָל {מו}, וְכֵיוָן שֶׁכֵּן הוּא, בְּטוֹב הָעוֹלָם נִדּוֹן, לְהִפָּרַע מִן הָרְשָׁעִים וְלִתֵּן שָׂכָר טוֹב לַצַּדִּיקִים. שֶׁהַחוֹטֵא חָטָא בִּרְצוֹנוֹ, רָאוּי שֶׁיֵּעָנֵשׁ. וְהַצַּדִּיק הָיָה צַדִּיק בִּרְצוֹנוֹ, וְרָאוּי שֶׁיְּקַבֵּל שָׂכָר {מז}: <B>וְהַכֹּל לְפִי רֹב הַמַּעֲשֶׂה.</B> לְפִי מַה שֶּׁאָדָם כּוֹפֵל וּמַתְמִיד בַּעֲשִׂיַּת הַטּוֹב כָּךְ שְׂכָרוֹ מְרֻבֶּה. שֶׁאֵינוֹ דּוֹמֶה הַמְחַלֵּק מֵאָה זְהוּבִים לִצְדָקָה בְּמֵאָה פְּעָמִים, לְנוֹתֵן אוֹתָם בְּפַעַם אַחַת. וְגִרְסַת רַמְבַּ"ם הִיא, וְהַכֹּל לְפִי רֹב הַמַּעֲשֶׂה, אֲבָל לֹא עַל פִּי הַמַּעֲשֶׂה {מח}: 

~ פרק ג - משנה טז
<B>הַכֹּל נָתוּן בְּעֵרָבוֹן.</B> רַגְלוֹהִי דְּבַר נַשׁ אִינוּן עֲרֵבִין בֵּיהּ [סֻכָּה נג.] לַאֲתַר דְּמִתְבָּעֵי תָּמָן מוֹבִילִין יָתֵיהּ: <B>וּמְצוּדָה פְרוּסָה.</B> יִסּוּרִין וּמִיתָה: <B>הַחֲנוּת פְּתוּחָה.</B> וּבְנֵי אָדָם נִכְנָסִים שָׁם וְלוֹקְחִים בַּהַקָּפָה: <B>וְהַחֶנְוָנִי מֵקִיף.</B> וּמַאֲמִין לְכָל הַבָּא לִטֹּל. כָּךְ בְּנֵי אָדָם חוֹטְאִים בְּכָל יוֹם וְהַקָּדוֹשׁ בָּרוּךְ הוּא מַמְתִּין לָהֶם עַד בֹּא עִתָּם: <B>וְהַפִּנְקָס פָּתוּחַ.</B> לִכְתֹּב בָּהּ הַהַקָּפוֹת כְּדֵי שֶׁלֹּא יִשָׁכַח: <B>וְהַיָּד כּוֹתֶבֶת.</B> שֶׁלֹּא תֹּאמַר אַף עַל פִּי שֶׁהַפִּנְקָס פָּתוּחַ פְּעָמִים שֶׁהַחֶנְוָנִי טָרוּד וְאֵינוֹ כּוֹתֵב הַכֹּל, לְכָךְ אָמַר וְהַיָּד כּוֹתֶבֶת: <B>וְכָל הָרוֹצֶה לִלְווֹת וְכוּ'.</B> הַיְנוּ וְהָרְשׁוּת נְתוּנָה דִּלְעֵיל, שֶׁאֵין שׁוּם אָדָם מֻכְרָח לִלְווֹת שֶׁלֹּא בִּרְצוֹנוֹ: <B>וְהַגַּבָּאִים.</B> יִסּוּרִים וּפְגָעִים רָעִים {מט}: <B>לְדַעְתּוֹ.</B> פְּעָמִים שֶׁזּוֹכֵר אֶת חוֹבוֹ וְאוֹמֵר יָפֶה דַּנְתַּנִי: <B>וְשֶׁלֹּא לְדַעְתּוֹ.</B> פְּעָמִים שֶׁשָּׁכַח וְקוֹרֵא תִּגָּר כְּנֶגֶד דִּינוֹ שֶׁל מָקוֹם בָּרוּךְ הוּא: <B>וְיֵשׁ לָהֶם עַל מַה שֶּׁיִּסְמֹכוּ.</B> עַל הַפִּנְקָס, וְעַל הַחֶנְוָנִי שֶׁהוּא נֶאֱמָן עַל פִּנְקָסוֹ. כָּךְ הַיִּסּוּרִים הַלָּלוּ הֵן נִסְמָכִים עַל מַעֲשָׂיו שֶׁל אָדָם הַנִּזְכָּרִים לִפְנֵי הַמָּקוֹם בָּרוּךְ הוּא, אַף עַל פִּי שֶׁהֵם נִשְׁכָּחִים מִן הָאָדָם: <B>וְהַדִּין דִּין אֱמֶת.</B> שֶׁאֵין הַקָּדוֹשׁ בָּרוּךְ הוּא בָּא בִּטְרוּנְיָא {נא} עִם בְּרִיּוֹתָיו [עֲבוֹדָה זָרָה ג.]: <B>וְהַכֹּל מְתֻקָּן לַסְּעוּדָה.</B> אֶחָד צַדִּיקִים וְאֶחָד רְשָׁעִים {נב} יֵשׁ לָהֶם חֵלֶק לָעוֹלָם הַבָּא לְאַחַר שֶׁגָּבוּ מֵהֶן אֶת חוֹבָן: 

~ פרק ג - משנה יז
<B>אִם אֵין תּוֹרָה וְכוּ'.</B> אֵין מַשָּׂאוֹ וּמַתָּנוֹ יָפֶה עִם הַבְּרִיּוֹת: <B>אִם אֵין דֶּרֶךְ אֶרֶץ וְכוּ'.</B> סוֹף שֶׁתּוֹרָתוֹ מִשְׁתַּכַּחַת מִמֶּנּוּ {נג}: <B>אִם אֵין דַּעַת וְכוּ'.</B> דַּעַת, הוּא שֶׁמּוֹצֵא טַעַם לַדָּבָר {נה}. וּבִינָה, הוּא שֶׁמֵּבִין דָּבָר מִתּוֹךְ דָּבָר אֲבָל אֵינוֹ מַרְאֶה לוֹ טַעַם. וְאִם אֵין דַּעַת לָתֵת טַעַם לַדָּבָר אֵין בִּינָה, אַחַר שֶׁלֹּא יָדַע טַעֲמוֹ שֶׁל דָּבָר כְּאִלּוּ לֹא יְדָעוֹ. וּמִכָּל מָקוֹם הַבִּינָה הִיא קוֹדֶמֶת, לְפִיכָךְ אִם אֵין בִּינָה אֵין דַּעַת: <B>אִם אֵין קֶמַח אֵין תּוֹרָה.</B> מִי שֶׁאֵין לוֹ מַה יֹּאכַל הֵיאַךְ יַעֲסֹק בַּתּוֹרָה: <B>אִם אֵין תּוֹרָה אֵין קֶמַח.</B> מַה יּוֹעִיל לוֹ הַקֶּמַח שֶׁבְּיָדוֹ, הוֹאִיל וְאֵין בּוֹ תּוֹרָה, נוֹחַ לוֹ שֶׁלֹּא הָיָה לוֹ קֶמַח וְהָיָה מֵת בָּרָעָב:

כְּאַבַּיֵי [שָׁם] דְּעָסַק בַּתּוֹרָה וּבִגְמִילוּת חֲסָדִים, מַסְפִּיקִין בְּיָדוֹ לְהַשְׁלִים מַחֲשַׁבְתּוֹ, וְיִזְכֶּה לִלְמֹד וּלְלַמֵּד לִשְׁמֹר וְלַעֲשׂוֹת. וְיֵשׁ מְפָרְשִׁים הַלּוֹמֵד עַל מְנָת לְלַמֵּד, בִּשְׁבִיל שֶׁיִּקָּרֵא רַב, וְגוֹרְסִין אֵין מַסְפִּיקִין בְּיָדוֹ לִלְמֹד וּלְלַמֵּד. וּבְרֹב הַסְּפָרִים מָצָאתִי כְּגִרְסָא רִאשׁוֹנָה, וְעִקָּר: <B>אַל תַּעֲשֵׂם עֲטָרָה לְהִתְגַּדֵּל בָּהֶם.</B> שֶׁלֹּא תֹּאמַר אֶלְמַד בִּשְׁבִיל שֶׁאֶקָּרֵא רַבִּי וְיוֹשִׁיבוּנִי בָּרֹאשׁ, אֶלָּא לְמַד מֵאַהֲבָה וְסוֹף הַכָּבוֹד לָבֹא: <B>וְלֹא קַרְדֹּם לַחְפֹּר בָּהֶם.</B> וְלֹא תִּלְמַד תּוֹרָה כְּדֵי לַעֲשׂוֹת מִמֶּנָּה מְלָאכָה שֶׁתִּתְפַּרְנֵס בָּהּ, כְּמוֹ קַרְדֹּם לַחְפֹּר בָּהּ, שֶׁהָעוֹשֶׂה כֵן מוֹעֵל בִּקְדֻשָּׁתָהּ שֶׁל תּוֹרָה וְחַיָּב מִיתָה בִּידֵי שָׁמַיִם כְּמִי שֶׁנֶּהֱנֶה מִן הַהֶקְדֵּשׁ. וּמְלַמְּדֵי תִּינוֹקוֹת נוֹטְלִין שְׂכָר שִׁמּוּר הַתִּינוֹקוֹת בִּלְבַד, שֶׁמְּשַׁמְּרִין אוֹתָן שֶׁלֹּא יִפְשְׁעוּ וְיַזִּיקוּ, וּשְׂכַר פִּסּוּק טְעָמִים, שֶׁאֵין הָרַב חַיָּב לִטְרֹחַ וּלְלַמֵּד לַתַּלְמִידִים פִּסּוּק הַטְּעָמִים. אֲבָל שְׂכַר לִמּוּד אָסוּר לִטֹּל, דִּכְתִיב (דְּבָרִים ד) וְאוֹתִי צִוָּה ה' בָּעֵת הַהִיא לְלַמֵּד אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים כַּאֲשֶׁר צִוַּנִי ה' אֱלֹהַי, מָה אֲנִי בְּחִנָּם, אַף אַתֶּם בְּחִנָּם {טז}. וְכֵן הַדַּיָּן אָסוּר לִקַּח שְׂכַר פְּסַק הַדִּין, אֶלָּא שְׂכַר בַּטָּלָה בִּלְבַד, דָּבָר שֶׁהוּא נִכָּר כַּמָּה מַפְסִיד מִבִּטּוּל מְלַאכְתוֹ כְּדֵי לִשְׁמֹעַ טַעֲנוֹת בַּעֲלֵי הַדִּין, וּבִלְבַד שֶׁיִּטֹּל מִשְּׁנֵיהֶם בְּשָׁוֶה, וְאִם נוֹטֵל יוֹתֵר מִזֶּה, דִּינָיו בְּטֵלִים. וּמַה שֶּׁהִתִּירָה תּוֹרָה לְתַלְמִיד חָכָם לֵהָנוֹת מִדִּבְרֵי תּוֹרָה, הוּא, שֶׁתִּמָּכֵר סְחוֹרָתוֹ בַּשּׁוּק {יז} קֹדֶם כָּל אָדָם, וְגַם שֶׁיִּפָּטֵר מִכָּל מִסִּים וְעֻלִּים וְאַרְנוֹנִיּוֹת {יח}, וַאֲפִלּוּ כֶּסֶף גּוּלְגַלְתָּא חַיָּבִין הַצִּבּוּר לִתֵּן עָלָיו, וַאֲפִלּוּ הוּא עָשִׁיר וְיֵשׁ לוֹ מָמוֹן הַרְבֵּה יָכוֹל הוּא לִשְׁאֹל בַּדִּין שֶׁיִּפְטְרוּהוּ. וְאִם תַּלְמִיד חָכָם חוֹלֶה הוּא וּמְדֻכָּא בְּיִסּוּרִין וּמַרְבִּין הָעָם לְהָבִיא מָנוֹת גְּדוֹלוֹת מִפְּנֵי כְּבוֹד תּוֹרָתוֹ, מִצְוָה עָלָיו שֶׁיִּטֹּל, וְאֵין זֶה בִּכְלָל נֵאוֹת מִדִּבְרֵי תּוֹרָה, הוֹאִיל וְאִי אֶפְשָׁר לוֹ לְהִתְפַּרְנֵס בְּדֶרֶךְ אַחֶרֶת. וְכֵן תַּלְמִיד חָכָם שֶׁמִּנּוּ אוֹתוֹ הַצִּבּוּר עֲלֵיהֶם פַּרְנָס אוֹ רֹאשׁ הַסֵּדֶר וּמִתְעַסֵּק בְּצָרְכֵי צִבּוּר, מֻתָּר לוֹ לִקַּח מֵהֶם פְּרָס וַאֲפִלּוּ שָׂכָר הַרְבֵּה יוֹתֵר מִכְּדֵי פַּרְנָסָתוֹ, כְּדֵי שֶׁיִּהְיוּ מְזוֹנוֹתָיו בְּרֶוַח וּמִתּוֹךְ כָּךְ יִהְיֶה גָּדוֹל וְיָרְאוּי וּמְאוּיָּם בְּעֵינֵיהֶם. כִּדְאַשְׁכְּחָן בְּכֹהֵן גָּדוֹל דִּכְתִיב בֵּיהּ הַגָּדוֹל מֵאֶחָיו, וְדָרְשׁוּ חֲכָמִים [יוֹמָא יח.] גַּדְּלֵהוּ מֵאֶחָיו, שֶׁיִּהְיוּ אֶחָיו הַכֹּהֲנִים מַעֲשִׁירִים אוֹתוֹ מִשֶּׁלָּהֶן. וַחֲכָמִים הָרִאשׁוֹנִים שֶׁהָיוּ נִמְנָעִים מִזֶּה, מִדַּת חֲסִידוּת הָיְתָה בָּהֶם, אֲבָל לֹא מִן הַדִּין: 

~ פרק ד - משנה ו
<B>כָּל הַמְכַבֵּד אֶת הַתּוֹרָה.</B> הַדּוֹרֵשׁ הַחֲסֵרוֹת וְהַיְתֵרוֹת שֶׁבַּתּוֹרָה וּמַרְאֶה טַעַם עַל כָּל קוֹץ וְקוֹץ לוֹמַר שֶׁאֵין בָּהּ דָּבָר לְבַטָּלָה, אֵין לְךָ כְּבוֹד תּוֹרָה גָּדוֹל מִזֶּה. אִי נַמִּי, הַמְכַבֵּד סֵפֶר תּוֹרָה וּמְכַבֵּד לוֹמְדֵי הַתּוֹרָה וְהָעוֹסְקִים בָּהּ, כָּל אֵלּוּ בִּכְלָל מְכַבֵּד אֶת הַתּוֹרָה הֵן: 

~ פרק ד - משנה ז
<B>הַחוֹשֵׁךְ עַצְמוֹ מִן הַדִּין.</B> בְּמָקוֹם שֶׁיֵּשׁ גָּדוֹל מִמֶּנּוּ. אִי נַמִּי, אוֹמֵר לְבַעֲלֵי הַדִּין שֶׁיִּתְפַּשְּׁרוּ: <B>פּוֹרֵק מִמֶּנּוּ אֵיבָה.</B> שֶׁהַיּוֹצֵא מִבֵּית דִּין חַיָּב, שׂוֹנֵא לַדַּיָּן, שֶׁאוֹמֵר בְּלִבּוֹ לֹא הִפֵּךְ בִּזְכוּתִי: <B>וְגָזֵל.</B> שֶׁמָּא יְחַיֵּב הַזַּכַּאי וְנִמְצָא שֶׁגָּזֵל בָּא עַל יָדוֹ: <B>וּשְׁבוּעַת שָׁוְא.</B> שֶׁמָּא יְחַיֵּב שְׁבוּעָה לְמִי שֶׁאֵינוֹ חַיָּב בָּהּ וְנִמְצָא מֵבִיא לִידֵי שְׁבוּעַת שָׁוְא. אִי נַמִּי, קָרֵי תַּנָּא לִשְׁבוּעַת שֶׁקֶר שְׁבוּעַת שָׁוְא, כְּגוֹן הָאוֹמֵר לֹא לָוִיתִי וְלָוָה, דְּבִשְׁעַת שְׁבוּעָה יָצְאָה מִפִּיו לַשָּׁוְא: <B>וְהַגַּס לִבּוֹ בַהוֹרָאָה.</B> בְּלֹא עִיּוּן וּבְלֹא הַמְתָּנָה: 

~ פרק ד - משנה ח
<B>אַל תְּהִי דָּן יְחִידִי.</B> אַף עַל פִּי שֶׁיָּחִיד מֻמְחֶה יָכוֹל לָדוּן, מִדַּרְכֵי הַחֲסִידוּת הוּא שֶׁאֲפִלּוּ מֻמְחֶה לֹא יְהִי דָּן יְחִידִי. וְדַוְקָא כְּשֶׁלֹּא קִבְּלוּהוּ בַּעֲלֵי הַדִּין עֲלֵיהֶם לְדִין, אֲבָל קִבְּלוּהוּ בַּעֲלֵי הַדִּין עֲלֵיהֶם, דָּן יְחִידִי וַאֲפִלּוּ מִדֶּרֶךְ חֲסִידוּת {כא}: <B>שֶׁאֵין דָּן יְחִידִי אֶלָּא אֶחָד.</B> הַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁהוּא יָחִיד בְּעוֹלָמוֹ: <B>וְאַל תֹּאמַר.</B> לַחֲבֵרֶיךָ הַחוֹלְקִים עָלֶיךָ: <B>קַבְּלוּ דַּעְתִּי.</B> שֶׁאֲנִי מֻמְחֶה וּבְלֹא הַצֵּרוּף שֶׁלָּכֶם הָיִיתִי יָכוֹל לָדוּן: <B>שֶׁהֵן רַשָּׁאִין וְלֹא אַתָּה.</B> הוֹאִיל וְצֵרַפְתָּ אוֹתָם עִמְּךָ רָאוּי שֶׁיִּפָּסֵק הַדִּין עַל פִּי הָרַבִּים: 

~ פרק ד - משנה ט
<B>הַמְקַיֵּם אֶת הַתּוֹרָה מֵעֹנִי.</B> שֶׁהוּא דָּחוּק לִמְזוֹנוֹת וּמִתְבַּטֵּל מִמְּלַאכְתּוֹ כְּדֵי לַעֲסֹק {כב} בַּתּוֹרָה: <B>הַמְבַטֵּל אֶת הַתּוֹרָה מֵעֹשֶׁר.</B> מֵחֲמַת רֹב הַמָּמוֹן צָרִיךְ לִתֵּן לִבּוֹ פַּעַם לְכָאן וּפַעַם לְכָאן וְאֵין לוֹ פְּנַאי לַעֲסֹק בַּתּוֹרָה: 

~ פרק ד - משנה י
<B>הֱוֵי מְמַעֵט בְּעֵסֶק.</B> הַמְעֵט בְּעֵסֶק סְחוֹרָתְךָ וּמְלַאכְתְּךָ, וְיִהְיֶה עִקַּר עִסְקְךָ בַּתּוֹרָה: <B>וֶהֱוֵי שְׁפַל רוּחַ בִּפְנֵי כָל אָדָם.</B> לִלְמֹד אֲפִלּוּ מִמִּי שֶׁהוּא קָטָן מִמְּךָ בְּחָכְמָה {כה}: <B>יֵשׁ לְךָ בְטֵלִים הַרְבֵּה כְנֶגְדָּךְ.</B> שֶׁיְּסַיְּעוּ לְבַטֶּלְךָ. פֵּרוּשׁ אַחֵר, יֵשׁ בָּעוֹלָם כַּמָּה בְּטֵלִים שֶׁהֵם הָרְשָׁעִים וְחַיּוֹת רָעוֹת שֶׁהֵם בְּטֵלִים, וְעַל יָדָם הַקָּדוֹשׁ בָּרוּךְ הוּא יָבִיא עָלֶיךָ פֻּרְעָנוּת: <B>וְאִם עָסַקְתָּ בַתּוֹרָה.</B> הוּא עַצְמוֹ יְשַׁלֵּם שְׂכָרְךָ וְלֹא עַל יְדֵי שָׁלִיחַ. וּבְזֶה יְתֵרָה מִדָּה טוֹבָה עַל מִדַּת פֻּרְעָנוּת: 

~ פרק ד - משנה יא
<B>פְּרַקְלִיט.</B> מַלְאַךְ מֵלִיץ טוֹב: <B>קַטֵּגוֹר.</B> מֵלִיץ רַע. וּלְשׁוֹן קַטֵּגוֹר, קוֹרֵא תִּגָּר. וְסַנֵּגוֹר שֶׁהוּא הֵפֶךְ קַטֵּגוֹר, פֵּרוּשׁוֹ שׂוֹנֵא תִּגָּר: 

~ פרק ד - משנה יב
<B>יְהִי כְבוֹד תַּלְמִידְךָ חָבִיב עָלֶיךָ כְּשֶׁלְּךָ.</B> שֶׁכֵּן מָצִינוּ בְּמֹשֶׁה שֶׁאָמַר לִיהוֹשֻׁעַ תַּלְמִידוֹ בְּחַר לָנוּ אֲנָשִׁים (שְׁמוֹת יז), עֲשָׂאוֹ שָׁוֶה לוֹ: <B>וּכְבוֹד חֲבֵרְךָ כְּמוֹרָא רַבְּךָ.</B> דְּאָמַר לֵיהּ אַהֲרֹן לְמֹשֶׁה בִּי אֲדֹנִי (בְּמִדְבָּר יב), וְאָחִיו גָּדוֹל מִמֶּנּוּ בְּשָׁנִים הָיָה {לג} וְקָרֵי לֵיהּ אֲדֹנִי: <B>וּמוֹרָא רַבְּךָ כְּמוֹרָא שָׁמָיִם.</B> דִּכְתִיב (שָׁם י"א) אֲדֹנִי מֹשֶׁה כְּלָאֵם, כָּלֵּם מִן הָעוֹלָם, כֵּיוָן שֶׁמָּרְדוּ בְּךָ כְּאִלּוּ מָרְדוּ בְּהַקָּדוֹשׁ בָּרוּךְ הוּא וְחַיָּבִים כְּלָיָה: 

~ פרק ד - משנה יג
<B>הֱוֵי זָהִיר בַּתַּלְמוּד.</B> שֶׁיְּהֵא תַּלְמוּדְךָ בְּדִקְדּוּק וְכַהֲלָכָה: <B>שֶׁשִּׁגְגַת תַּלְמוּד עוֹלָה זָדוֹן.</B> אִם תִּטְעֶה בְּהוֹרָאָה מִתּוֹךְ שֶׁלֹּא דִּקְדַּקְתָּ בְּתַלְמוּדְךָ וְתָבֹא לְהַתִּיר אֶת הָאָסוּר, הַקָּדוֹשׁ בָּרוּךְ הוּא מַעֲלֶה עַל יָדְךָ כְּאִלּוּ עָשִׂיתָ מֵזִיד: <B>שְׁלֹשָׁה כְתָרִים הֵם.</B> הַכְּתוּבִים בַּתּוֹרָה, שֶׁחִיְּבָה הַתּוֹרָה לִנְהֹג בָּהֶם כָּבוֹד: <B>כֶּתֶר תּוֹרָה.</B> כְּתִיב בֵּיהּ (וַיִּקְרָא יט) וְהָדַרְתָּ פְּנֵי זָקֵן, זֶה שֶׁקָּנָה חָכְמָה: <B>כֶּתֶר כְּהֻנָּה.</B> כְּתִיב בֵּיהּ (שָׁם כ"א) וְקִדַּשְׁתּוֹ קָדוֹשׁ יִהְיֶה לָךְ: <B>כֶּתֶר מַלְכוּת.</B> כְּתִיב בֵּיהּ (דְּבָרִים י"ז) שׂוֹם תָּשִׂים עָלֶיךָ מֶלֶךְ, שֶׁתְּהֵא אֵימָתוֹ עָלֶיךָ: <B>וְכֶתֶר שֵׁם טוֹב.</B> מִי שֶׁיֵּשׁ בְּיָדוֹ מַעֲשִׂים טוֹבִים וּשְׁמוּעָתוֹ טוֹבָה מֵחֲמַת מַעֲשָׂיו, לֹא מָצִינוּ לוֹ כֶּתֶר בַּתּוֹרָה שֶׁיִּהְיוּ חַיָּבִים לִנְהֹג בּוֹ כָּבוֹד, וְהוּא עוֹלֶה עַל גַּבֵּי כֻּלָּן, שֶׁכָּל הַשְּׁלֹשָׁה כְּתָרִים צְרִיכִים לוֹ. שֶׁאִם תַּלְמִיד חָכָם הוּא וְסַנִּי שׁוּמְעָנֵיהּ, שָׁרֵי לְבִזּוּיֵהּ. וְאִם כֹּהֵן גָּדוֹל הוּא, אָמְרִינַן בְּיוֹמָא (דַּף ע"א) יֵיתוֹן בְּנֵי עַמְמַיָא לִשְׁלָם דְּעָבְדִין עוּבְדֵי אַהֲרֹן, וְלֹא יֵיתוֹן בְּנֵי אַהֲרֹן לִשְׁלָם דְּלֹא עָבְדִין עוּבְדֵי אַהֲרֹן. וְאִם מֶלֶךְ הוּא, כְּתִיב (שְׁמוֹת כב) וְנָשִׂיא בְּעַמְּךָ לֹא תָאֹר, בְּעוֹשֶׂה מַעֲשֶׂה עַמְּךָ: 

~ פרק ד - משנה יד
<B>הֱוֵי גּוֹלֶה לִמְקוֹם תּוֹרָה.</B> אִם אֵין תַּלְמִידֵי חֲכָמִים בִּמְקוֹמְךָ: <B>וְאַל תֹּאמַר שֶׁהִיא תָּבֹא אַחֲרֶיךָ.</B> תַּלְמִידֵי חֲכָמִים יָבֹאוּ לְכָאן: <B>שֶׁחֲבֵרֶיךָ יְקַיְּמוּהָ בְיָדֶךָ.</B> וְאַל תִּסְמֹךְ שֶׁחֲבֵרֶיךָ כְּשֶׁיָּבֹאוּ מִבֵּית הָרַב יְקַיְּמוּ הַתּוֹרָה בְּיָדְךָ, שֶׁתִּלְמֹד מֵהֶם מַה שֶּׁלָּמְדוּ הֵם מִן הָרַב. אֶלָּא הֱוֵי גּוֹלֶה אַתָּה בְעַצְמְךָ לִמְקוֹם הָרַב, שֶׁאֵינוֹ דּוֹמֶה שׁוֹמֵעַ מִפִּי תַּלְמִיד לְשׁוֹמֵעַ מִפִּי הָרַב. פֵּרוּשׁ אַחֵר, שֶׁחֲבֵרֶיךָ יְקַיְּמוּהָ בְּיָדֶךָ, מִפְּנֵי מָה אֲנִי אוֹמֵר לְךָ הֱוֵי גּוֹלֶה לִמְקוֹם תּוֹרָה, שֶׁחֲבֵרֶיךָ יְקַיְּמוּהָ בְּיָדֶךָ, שֶׁאֲפִלּוּ אַתָּה חָרִיף וּמְפֻלְפָּל בְּיוֹתֵר לֹא תִּתְקַיֵּם הַתּוֹרָה בְּךָ אֶלָּא עַל יְדֵי חֲבֵרֶיךָ שֶׁתִּשָּׂא וְתִתֵּן עִמָּהֶם, וְהַיְנוּ דִּמְסַיֵּם וְאֶל בִּינָתְךָ אַל תִּשָּׁעֵן: 

~ פרק ד - משנה טו
<B>אֵין בְּיָדֵינוּ.</B> כְּמוֹ לֹא הֲוָה בְּיָדֵיהּ [יְבָמוֹת קה.]. כְּלוֹמַר אֵין הַדָּבָר הַזֶּה יָדוּעַ לָנוּ מַדּוּעַ דֶּרֶךְ רְשָׁעִים צָלֵחָה וּמִפְּנֵי מָה הַצַּדִּיקִים מְדֻכָּאִים בְּיִסּוּרִין {לה}. פֵּרוּשׁ אַחֵר, אֵין לָנוּ בִּזְמַן הַגָּלוּת לֹא מִשַּׁלְוָה וְהַשְׁקֵט שֶׁרָגִיל הַקָּדוֹשׁ בָּרוּךְ הוּא לָתֵת לָרְשָׁעִים כְּדֵי לְטָרְדָן מִן הָעוֹלָם הַבָּא, וְאַף לֹא מִן הַיִּסּוּרִים הַמְיֻחָדִים לַצַּדִּיקִים {לו} שֶׁהֵן יִסּוּרִים שֶׁל אַהֲבָה שֶׁאֵין בָּהֵן בִּטּוּל תּוֹרָה. כְּלוֹמַר יָצָאנוּ מִכְּלָל רְשָׁעִים, שֶׁאֵין לָנוּ הַשַּׁלְוָה שֶׁיֵּשׁ לָרְשָׁעִים, וְלִכְלַל צַדִּיקִים לֹא הִגַּעְנוּ, שֶׁהַיִּסּוּרִים שֶׁלָּנוּ אֵינָן שֶׁל אַהֲבָה כְּיִסּוּרֵי הַצַּדִּיקִים: <B>הֱוֵי מַקְדִּים שָׁלוֹם לְכָל אָדָם.</B> וַאֲפִלּוּ לְנָכְרִי בַּשּׁוּק: <B>וֶהֱוֵי זָנָב לָאֲרָיוֹת.</B> לִגְדוֹלִים מִמְּךָ {לז}: <B>וְלֹא רֹאשׁ לַשּׁוּעָלִים.</B> לַקְּטַנִּים מִמְּךָ: 

~ פרק ד - משנה טז
<B>בַּפְּרוֹזְדוֹר.</B> בֵּית שַׁעַר. תַּרְגּוּם אוּלָם, פְּרוֹזְדוֹרָא: <B>טְרַקְלִין.</B> מְקוֹם מוֹשַׁב הַמֶּלֶךְ. כָּךְ אַתָּה הַתְקֵן עַצְמְךָ בָּעוֹלָם הַזֶּה כְּדֵי שֶׁתִּזְכֶּה לָעוֹלָם הַבָּא: 

~ פרק ד - משנה יז
<B>יָפָה שָׁעָה אַחַת בִּתְשׁוּבָה.</B> לְצֹרֶךְ תְּשׁוּבָה וּמַעֲשִׂים טוֹבִים: <B>מִכָּל חַיֵּי הָעוֹלָם הַבָּא.</B> לְפִי שֶׁבְּאוֹתוֹ זְמַן אֵין הַתְּשׁוּבָה וּמַעֲשִׂים טוֹבִים מְהַנִּים לָאָדָם, שֶׁאֵין הָעוֹלָם הַבָּא אֶלָּא לְקַבֵּל שָׂכָר עַל מַה שֶּׁקִּיֵּם בָּעוֹלָם הַזֶּה: 

~ פרק ד - משנה יח
<B>אַל תְּרַצֶּה אֶת חֲבֵרְךָ בִשְׁעַת כַּעֲסוֹ.</B> דִּכְתִיב (שְׁמוֹת לג) פָּנַי יֵלֵכוּ וַהֲנִחוֹתִי לָךְ, אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא לְמֹשֶׁה, הַמְתֵּן עַד שֶׁיַּעַבְרוּ פָּנִים שֶׁל זַעַם [בְּרָכוֹת ז.]: <B>וְאַל תְּנַחֲמֶנּוּ בְשָׁעָה שֶׁמֵּתוֹ מֻטָּל לְפָנָיו.</B> שֶׁבִּשְׁעַת הַחֻרְבָּן כִּבְיָכוֹל הָיָה הַקָּדוֹשׁ בָּרוּךְ הוּא מִתְאַבֵּל, בִּקְּשׁוּ מַלְאֲכֵי הַשָּׁרֵת לְנַחֲמוֹ, וְרוּחַ הַקֹּדֶשׁ מְשִׁיבָן (יְשַׁעְיָה כב) אַל תָּאִיצוּ לְנַחֲמֵנִי: <B>וְאַל תִּשְׁאַל לוֹ בִשְׁעַת נִדְרוֹ.</B> לִמְצֹא פְּתָחִים לְהַתִּיר לוֹ אֶת נִדְרוֹ. מִפְּנֵי שֶׁאָז עַל כָּל פֶּתַח וּפֶתַח שֶׁתִּמְצָא לוֹ יֹאמַר אַדַּעְתָּא דְהָכִי נָדַרְתִּי, וְשׁוּב לֹא תִּמְצָא לוֹ פֶּתַח. וּמָצִינוּ כְּשֶׁנִּשְׁבַּע הַקָּדוֹשׁ בָּרוּךְ הוּא לְמֹשֶׁה שֶׁלֹּא יִכָּנֵס לָאָרֶץ לֹא הִתְחַנֵּן מִיָּד אֶלָּא הִמְתִּין וְאַחַר כָּךְ הִתְחִיל לְהִתְחַנֵּן: <B>בִּשְׁעַת קַלְקָלָתוֹ.</B> כְּשֶׁנִּתְקַלְקֵל בְּחֵטְא. לְפִי שֶׁהוּא מִתְבַּיֵּשׁ מִכָּל אָדָם. וְכֵן אַתָּה מוֹצֵא כְּשֶׁחָטָא אָדָם הָרִאשׁוֹן לֹא נִגְלָה עֲלֵיהֶם הַקָּדוֹשׁ בָּרוּךְ הוּא עַד שֶׁעָשׂוּ חֲגוֹרוֹת, דִּכְתִיב (בְּרֵאשִׁית ג) וַיַּעֲשׂוּ לָהֶם חֲגוֹרוֹת, וְאַחַר כָּךְ וַיִּשְׁמְעוּ אֶת קוֹל ה' אֱלֹהִים: 

~ פרק ד - משנה יט
<B>בִּנְפֹל אוֹיִבְךָ אַל תִּשְׂמָח.</B> פָּסוּק הוּא בְּמִשְׁלֵי, אֶלָּא שֶׁשְּׁמוּאֵל הַקָּטָן הָיָה רָגִיל לְהוֹכִיחַ בְּנֵי אָדָם בְּמִדָּה זוֹ: <B>פֶּן יִרְאֶה ה' וְרַע בְּעֵינָיו.</B> שֶׁעֲשִׂיתוֹ בִּלְבָבְךָ כְּאִלּוּ הוּא שְׁלוּחֲךָ לְמַלְאוֹת תַּאֲוָתְךָ: <B>וְהֵשִׁיב מֵעָלָיו אַפּוֹ.</B> מִדְּלֹא כְּתִיב וְשָׁב אֶלָּא וְהֵשִׁיב, מַשְׁמַע יַסִּירֵהוּ מֵעַל אוֹיְבֶיךָ וִישִׁיבֵהוּ עָלֶיךָ:

וְנַעֲשֵׂית הַצְּבָת מִיָּד:

<B>גֹּלֶם.</B> לְשׁוֹן גָּלְמֵי כֵּלִים, שֶׁלֹּא נִגְמְרָה מְלַאכְתָּן {כט}. כָּךְ אָדָם שֶׁאֵינוֹ נִגְמָר בְּדַעְתּוֹ לֹא בְּמִדּוֹת וְלֹא בְּחָכְמָה קָרוּי גֹּלֶם: <B>אֵינוֹ מְדַבֵּר לִפְנֵי מִי שֶׁגָּדוֹל מִמֶּנּוּ בְחָכְמָה.</B> שֶׁכֵּן מָצִינוּ בְּאֶלְעָזָר וְאִיתָמָר שֶׁלֹּא רָצוּ לְדַבֵּר בִּפְנֵי אֲבִיהֶן כְּשֶׁקָּצַף עֲלֵיהֶם מֹשֶׁה, וְהֵשִׁיב אַהֲרֹן {לא}: <B>וְאֵינוֹ נִכְנָס לְתוֹךְ דִּבְרֵי חֲבֵרוֹ.</B> שֶׁלֹּא [יְעַרְבְּבֶנּוּ]. דִּכְתִיב (בְּמִדְבָּר יב) שִׁמְעוּ נָא דְּבָרָי, הַמְתִּינוּ לִי עַד שֶׁאֲדַבֵּר. קַל וָחֹמֶר לְהֶדְיוֹט: <B>וְאֵינוֹ נִבְהָל לְהָשִׁיב.</B> כְּדֵי שֶׁתְּהֵא תְּשׁוּבָתוֹ כַּהֲלָכָה {לג}. וְכֵן בֶּאֱלִיהוּא הוּא אוֹמֵר (אִיּוֹב לו) כַּתַּר לִי זְעֵיר וַאֲחַוֶּךָ: <B>שׁוֹאֵל כָּעִנְיָן וּמֵשִׁיב כַּהֲלָכָה.</B> לְדָבָר אֶחָד הוּא נִמְנָה כָּאן. וְהָכִי פֵּרוּשׁוֹ, הַתַּלְמִיד שׁוֹאֵל כָּעִנְיָן כְּלוֹמַר בְּאוֹתוֹ עִנְיָן שֶׁהֵם עֲסוּקִים בּוֹ, וְאָז הָרַב מֵשִׁיב כַּהֲלָכָה. אֲבָל אִם יִשְׁאַל הַתַּלְמִיד שֶׁלֹּא כְּעִנְיָן, הוּא מֵבִיא לָרַב שֶׁיָּשִׁיב שֶׁלֹּא כַּהֲלָכָה. עַל דֶּרֶךְ שֶׁאָמַר רַבִּי חִיָּא לְרַב (שַׁבָּת דַּף ג) כִּי קָאֵי רַבִּי בְּהַאי מַסֶּכְתָּא לֹא תְשַׁיְלֵיהּ בְּמַסֶּכְתָּא אַחֲרִיתִי. וְכֵן אַתָּה מוֹצֵא בָּאֲנָשִׁים אֲשֶׁר הָיוּ טְמֵאִים לְנֶפֶשׁ אָדָם שֶׁרָאוּ אֶת מֹשֶׁה עוֹסֵק בְּהִלְכוֹת הַפֶּסַח וְשָׁאֲלוּ לוֹ בְּאוֹתוֹ עִנְיָן: <B>וְאוֹמֵר עַל רִאשׁוֹן רִאשׁוֹן.</B> וְכֵן מָצִינוּ בְּהַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁאָמַר לוֹ מֹשֶׁה (שְׁמוֹת ג) מִי אָנֹכִי כִּי אֵלֵךְ אֶל פַּרְעֹה, הִיא הָרִאשׁוֹנָה. וְכִי אוֹצִיא אֶת בְּנֵי יִשְׂרָאֵל, הֲרֵי שְׁנִיָּה. וְהֵשִׁיבוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא עַל הָרִאשׁוֹנָה כִּי אֶהְיֶה עִמָּךְ, וְעַל הַשְּׁנִיָּה, בְּהוֹצִיאֲךָ אֶת הָעָם מִמִּצְרַיִם תַּעַבְדוּן אֶת הָאֱלֹהִים: <B>וְעַל מַה שֶּׁלֹּא שָׁמַע אוֹמֵר לֹא שָׁמָעְתִּי.</B> אִם פּוֹסֵק דִּין מִן הַסְּבָרָא מִדַּעְתּוֹ לֹא יֹאמַר כָּךְ שָׁמַעְתִּי מֵרַבּוֹתַי {לד}. וּמָצִינוּ בְּאַנְשֵׁי חָרָן (בְּרֵאשִׁית כט) כְּשֶׁשָּׁאַל מֵהֶם יַעֲקֹב הֲשָׁלוֹם לוֹ, אָמְרוּ לוֹ שָׁלוֹם, וְהִנֵּה רָחֵל בִּתּוֹ בָּאָה עִם הַצֹּאן, כְּלוֹמַר, זֶה אָנוּ יוֹדְעִים, וְאִם תִּשְׁאַל יוֹתֵר, הִנֵּה רָחֵל בִּתּוֹ בָּאָה עִם הַצֹּאן וְהִיא תַּגִּיד לְךָ, כִּי אֲנַחְנוּ לֹא יָדַעְנוּ יוֹתֵר מִזֶּה: <B>וּמוֹדֶה עַל הָאֱמֶת.</B> וְאַף עַל פִּי שֶׁיָּכוֹל לְהַעֲמִיד אֶת דְּבָרָיו עַל יְדֵי טְעָנוֹת שֶׁבְּיָדוֹ. וְכֵן מָצִינוּ בְּמֹשֶׁה (וַיִּקְרָא י), כְּשֶׁסִּלְּקוֹ אַהֲרֹן וְאָמַר לוֹ, אִם שָׁמַעְתָּ בְּקָדְשֵׁי שָׁעָה אֵין לְךָ לְהָקֵל בְּקָדְשֵׁי דּוֹרוֹת, וַיִּשְׁמַע מֹשֶׁה וַיִּיטַב בְּעֵינָיו, הוֹדָה וְלֹא בּוֹשׁ לוֹמַר לֹא שָׁמַעְתִּי, אֶלָּא שָׁמַעְתִּי וְשָׁכַחְתִּי: <B>וְחִלּוּפֵיהֶן בַּגֹּלֶם.</B> חִלּוּף הַדְּבָרִים הַלָּלוּ שֶׁבֶּחָכָם הֵם בַּגֹּלֶם: 

~ פרק ה - משנה ח
<B>רָעָב שֶׁל בַּצֹּרֶת.</B> הַגְּשָׁמִים מֻעָטִים וּמִתּוֹךְ כָּךְ הַשַּׁעַר מִתְיַקֵּר: <B>רָעָב שֶׁל מְהוּמָה.</B> מִפְּנֵי הַגְּיָסוֹת אֵין יְכוֹלִין לֶאֱסֹף הַתְּבוּאָה: <B>רָעָב שֶׁל כְּלָיָה.</B> הַשָּׁמַיִם כַּבַּרְזֶל וְהָאָרֶץ כַּנְּחוּשָׁה: <B>שֶׁלֹּא נִמְסְרוּ לְבֵית דִּין.</B> שֶׁלֹּא עָשׂוּ בָּהֶן דִּין תּוֹרָה {לט}: <B>וְעַל פֵּרוֹת שְׁבִיעִית.</B> שֶׁעוֹשִׂים בָּהֶן סְחוֹרָה וְאֵין נוֹהֲגִין בָּהֶן קְדֻשַּׁת שְׁבִיעִית: <B>עִנּוּי הַדִּין.</B> שֶׁיּוֹדְעִים לְהֵיכָן הַדִּין נוֹטֶה וּמְעַכְּבִין וְאֵין פּוֹסְקִין אוֹתוֹ: <B>עִוּוּת הַדִּין.</B> לְזַכּוֹת אֶת הַחַיָּב וּלְחַיֵּב אֶת הַזַּכַּאי: <B>וְעַל הַמּוֹרִים בַּתּוֹרָה שֶׁלֹּא כַהֲלָכָה.</B> לֶאֱסֹר אֶת הַמֻּתָּר וּלְהַתִּיר אֶת הָאָסוּר: 

~ פרק ה - משנה ט
<B>שְׁבוּעַת שָׁוְא.</B> לְבַטָּלָה, שֶׁלֹּא לְצֹרֶךְ: <B>חִלּוּל הַשֵּׁם.</B> הָעוֹבֵר עֲבֵרָה בְּפַרְהֶסְיָא בְּיָד רָמָה. אִי נַמִּי, שֶׁבְּנֵי אָדָם רוֹאִים וְלוֹמְדִים מִמַּעֲשָׂיו: <B>וְעַל הַשְׁמָטַת הָאָרֶץ.</B> שֶׁחוֹרְשִׁים וְזוֹרְעִים בַּשְּׁבִיעִית: <B>מִפְּנֵי מַעֲשַׂר עָנִי שֶׁבַּשְּׁלִישִׁית.</B> שֶׁבִּמְקוֹם מַעֲשֵׂר שֵׁנִי שֶׁמַּפְרִישִׁים בִּשְׁאָר שָׁנִים שֶׁל שְׁמִיטָה, בַּשְּׁלִישִׁית וּבַשִּׁשִּׁית מַפְרִישִׁין מַעֲשַׂר עָנִי: <B>גֶּזֶל מַתְּנוֹת עֲנִיִּים.</B> לֶקֶט שִׁכְחָה וּפֵאָה פֶּרֶט וְעוֹלְלוֹת: 

~ פרק ה - משנה י
<B>שֶׁלִּי שֶׁלִּי וְשֶׁלְּךָ שֶׁלָּךְ.</B> אֵינִי רוֹצֶה לְהַנּוֹתְךָ, וְהַלְוַאי שֶׁלֹּא תְּהַנֶּה אוֹתִי: <B>וְיֵשׁ אוֹמְרִים זוֹ מִדַּת סְדוֹם.</B> קָרוֹב הַדָּבָר לָבֹא לִידֵי מִדַּת סְדוֹם, שֶׁמִּתּוֹךְ שֶׁהוּא רָגִיל בְּכָךְ, אֲפִלּוּ בְּדָבָר שֶׁחֲבֵרוֹ נֶהֱנֶה וְהוּא אֵינוֹ חָסֵר לֹא יִרְצֶה לְהַנּוֹת אֶת חֲבֵרוֹ, וְזוֹ הָיְתָה מִדַּת סְדוֹם, שֶׁהָיוּ מִתְכַּוְּנִים לְכַלּוֹת הָרֶגֶל מִבֵּינֵיהֶם, וְאַף עַל פִּי שֶׁהָיְתָה הָאָרֶץ רַחֲבַת יָדַיִם לִפְנֵיהֶם וְלֹא הָיוּ חֲסֵרִים כְּלוּם: <B>שֶׁלִּי שֶׁלְּךָ וְשֶׁלְּךָ שֶׁלִּי עַם הָאָרֶץ.</B> שֶׁנֶּהֱנֶה וּמְהַנֶּה בְּשָׁוֶה, וְזֶהוּ יִשּׁוּבָהּ שֶׁל הָאָרֶץ {מו}, אֲבָל אֵינוֹ יוֹדֵעַ קְרָא דִּכְתִיב (מִשְׁלֵי טו) וְשׂוֹנֵא מַתָּנוֹת יִחְיֶה. וְזֶהוּ לְשׁוֹן עַם הָאָרֶץ הָאָמוּר בְּכָל מָקוֹם, שֶׁרוֹצֶה בְּתִקּוּנָהּ שֶׁל הָאָרֶץ אֲבָל אֵין בּוֹ חָכְמָה לְהַבְדִּיל בַּתִּקּוּנִין הָרְאוּיִין: <B>שֶׁלִּי שֶׁלְּךָ וְשֶׁלְּךָ שֶׁלָּךְ.</B> מְהַנֶּה אֶת הַבְּרִיּוֹת מִנְּכָסָיו, וְהוּא אֵינוֹ נֶהֱנֶה מֵאֲחֵרִים: <B>חָסִיד.</B> שֶׁעוֹשֶׂה לִפְנִים מִשּׁוּרַת הַדִּין:

הָכִי גָּרְסִינַן נוֹחַ לִכְעֹס וְנוֹחַ לִרְצוֹת יָצָא שְׂכָרוֹ בְהֶפְסֵדוֹ, קָשֶׁה לִכְעֹס וְקָשֶׁה לִרְצוֹת יָצָא הֶפְסֵדוֹ בִשְׂכָרוֹ. אָדָם שֶׁכּוֹעֵס מַהֵר עַל כָּל דָּבָר, אַף עַל פִּי שֶׁהוּא חוֹזֵר וּמִתְרַצֶּה מְהֵרָה, הֶפְסֵדוֹ מְרֻבֶּה מִשְּׂכָרוֹ, שֶׁרֹב מַעֲשָׂיו מְקֻלְקָלִים מֵאַחַר שֶׁהוּא נוֹחַ לִכְעֹס עַל כָּל דָּבָר וְדָבָר. אֲבָל הַקָּשֶׁה לִכְעֹס, אַף עַל פִּי שֶׁיֵּשׁ לוֹ מִדָּה רָעָה שֶׁהוּא קָשֶׁה לִרְצוֹת, יָצָא הֶפְסֵדוֹ הַמֻּעָט שֶׁהוּא קָשֶׁה לִרְצוֹת בִּשְׂכָרוֹ הַמְרֻבֶּה שֶׁהוּא קָשֶׁה לִכְעֹס, וְרֹב מַעֲשָׂיו מְתֻקָּנִים. וְאִית דְּגַרְסֵי אִפְּכָא. וְגִרְסָא זוֹ נִרְאֵית לִי עִקָּר:

<B>מַהֵר לִשְׁמֹעַ וּמַהֵר לְאַבֵּד יָצָא שְׂכָרוֹ בְהֶפְסֵדוֹ.</B> דְּמֵאַחַר שֶׁשּׁוֹכֵחַ מַה שֶּׁלּוֹמֵד מַה הֲנָאָה יֵשׁ בְּמַה שֶּׁהוּא מַהֵר לִשְׁמֹעַ, נִמְצָא הֶפְסֵדוֹ גָּדוֹל מִשְּׂכָרוֹ: <B>קָשֶׁה לִשְׁמֹעַ וְקָשֶׁה לְאַבֵּד יָצָא הֶפְסֵדוֹ בִשְׁכָרוֹ.</B> שֶׁמִּדָּה טוֹבָה שֶׁיֵּשׁ בּוֹ יְתֵרָה עַל הַמִּדָּה הַנִּפְסֶדֶת, הוֹאִיל וּמַה שֶּׁשָּׁמַע אַחַר הַקֹּשִׁי הוּא זוֹכֵר וְאֵינוֹ שׁוֹכֵחַ. וְנָפְקָא מִנַּהּ, שֶׁאִם יֵשׁ לְפָנֵינוּ שְׁנֵי תַּלְמִידִים וְאֵין לָנוּ לְסַפֵּק מָזוֹן אֶלָּא לְאֶחָד מֵהֶם, נַקְדִּים הַקָּשֶׁה לְאַבֵּד עַל הַמַּהֵר לִשְׁמֹעַ: <B>זֶה חֵלֶק רָע.</B> לֹא הֲוָה שַׁיָּךְ לְמִתְנֵי הָכָא חָסִיד אוֹ רָשָׁע, שֶׁאֵין זֶה דָּבָר הַתָּלוּי בִּבְחִירָתוֹ {נ} שֶׁל אָדָם, אֶלָּא חִסָּרוֹן שֶׁהָיָה בּוֹ מֵעִקַּר בְּרִיָּתוֹ: 

~ פרק ה - משנה יג
<B>אַרְבַּע מִדּוֹת בְּנוֹתְנֵי צְדָקָה.</B> כְּלוֹמַר בִּנְתִינַת הַצְּדָקָה, וְלֹא בְּנוֹתְנֵי צְדָקָה מַמָּשׁ, דְּהָא אִיכָּא בְּהוּ מִי שֶׁאֵינוֹ נוֹתֵן. וְכֵן בְּסָמוּךְ בְּהוֹלְכֵי לְבֵית הַמִּדְרָשׁ, הַיְנוּ בַּהֲלִיכַת בֵּית הַמִּדְרָשׁ: <B>עֵינוֹ רָעָה בְּשֶׁל אֲחֵרִים.</B> דְּיוֹדֵעַ שֶׁהַצְּדָקָה מְעַשֶּׁרֶת וְאֵינוֹ רוֹצֶה שֶׁאֲחֵרִים יִתְעַשְּׁרוּ {נא}. פֵּרוּשׁ אַחֵר, יֵשׁ מִי שֶׁחָס עַל מָמוֹן קְרוֹבָיו יוֹתֵר מֵעַל מָמוֹנוֹ, וְאַף עַל פִּי שֶׁהוּא נוֹתֵן אֵינוֹ רוֹצֶה שֶׁקְּרוֹבָיו יִתְּנוּ, שֶׁלֹּא יְאַבְּדוּ אֶת מָמוֹנָם, וְהַיְנוּ עֵינוֹ רָעָה בְּשֶׁל אֲחֵרִים דּוּמְיָא דְּעֵינוֹ רָעָה בְּשֶׁלּוֹ דִּבְסֵיפָא: 

~ פרק ה - משנה יד
<B>הוֹלֵךְ וְאֵינוֹ עוֹשֶׂה.</B> הוֹלֵךְ לְבֵית הַמִּדְרָשׁ לִשְׁמֹעַ וְאֵינוֹ שׁוֹנֶה וְלוֹמֵד וְלֹא מֵבִין {נב}: <B>עוֹשֶׂה וְאֵינוֹ הוֹלֵךְ.</B> לוֹמֵד וְשׁוֹנֶה בְּבֵית הַמִּדְרָשׁ שֶׁבְּבֵיתוֹ: 

~ פרק ה - משנה טו
<B>בְּיוֹשְׁבִים לִפְנֵי חֲכָמִים.</B> לְעֵיל בְּאַרְבַּע מִדּוֹת בַּתַּלְמִידִים אַיְרֵי בְּעִנְיַן הַזִּכָּרוֹן וְהַשִּׁכְחָה, וְהַשְׁתָּא מַיְרֵי בְּעִנְיַן הַסְּבָרָא הַיְשָׁרָה וּבְרֵרַת הַדָּבָר הַצּוֹדֵק מֵהַבִּלְתִּי צוֹדֵק: <B>סְפוֹג.</B> הוּא סוֹפֵג אֶת הַמַּיִם, בֵּין עֲכוּרִים בֵּין צְלוּלִים, כָּךְ יֵשׁ מִי שֶׁלִּבּוֹ רָחָב וּמְקַבֵּל כָּל מַה שֶּׁשּׁוֹמֵעַ, וְאֵין בּוֹ כֹּחַ לִבְרֹר הָאֱמֶת מִן הַשֶּׁקֶר: <B>מַשְׁפֵּךְ.</B> כְּלִי שֶׁנּוֹתְנִים עַל פִּי חָבִית אוֹ עַל פִּי הַנּוֹד כְּשֶׁרוֹצִים לְמַלְאוֹתוֹ יַיִן אוֹ שֶׁמֶן: <B>שֶׁמַּכְנִיס בְּזוֹ וּמוֹצִיא בְזוֹ.</B> כָּךְ יֵשׁ מִי שֶׁמְּקַבֵּל כָּל מַה שֶּׁלּוֹמֵד וּכְבוֹלְעוֹ כָּךְ פּוֹלְטוֹ: <B>מְשַׁמֶּרֶת.</B> מוֹצִיא כָּל מַה שֶּׁשּׁוֹמֵעַ בְּבֵית הַמִּדְרָשׁ, וְקוֹלֵט דָּבָר שֶׁל בַּטָּלָה: <B>נָפָה.</B> לְאַחַר שֶׁמּוֹצִיאִים הַסֻּבִּין וְהַמֻּרְסָן מִן הַקֶּמַח הַנִּטְחָן וְנִשְׁאַר הַקֶּמַח הַדַּק עִם הַסֹּלֶת הַגַּס וְהוּא הֶחָשׁוּב, מַעֲבִירִין אוֹתוֹ בְּנָפָה דַּקָּה מְאֹד וְיוֹרֵד מִמֶּנָּה כָּל אוֹתוֹ הַקֶּמַח הַדַּק שֶׁהוּא כְּעֵין עַפְרוּרִית לְבָנָה וְנִשְׁאַר הַסֹּלֶת הַגַּס הֶחָשׁוּב. וְכֵן הָיוּ עוֹשִׂין לַמְּנָחוֹת. כָּךְ יֵשׁ מִי שֶׁיֵּשׁ בּוֹ כֹּחַ לִבְרֹר וּלְלַבֵּן שְׁמוּעוֹתָיו, וְקוֹלֵט הָאֱמֶת מִן הַשֶּׁקֶר וְהַבָּטֵל: 

~ פרק ה - משנה טז
<B>כָּל אַהֲבָה שֶׁהִיא תְלוּיָה בְדָבָר בָּטֵל.</B> שֶׁאֵינוֹ מִתְקַיֵּם, כְּשֶׁיִּתְבַּטֵּל הַדָּבָר שֶׁהָיָה סִבָּה לְאוֹתָהּ אַהֲבָה, גַּם הָאַהֲבָה בְּטֵלָה. וְכָל אַהֲבָה שֶׁאֵינָהּ תְּלוּיָה בְּדָבָר בָּטֵל אֶלָּא בְּדָבָר קַיָּם, כְּגוֹן אַהֲבַת הַצַּדִּיקִים וְהַחֲכָמִים, אֵינָהּ בְּטֵלָה לְעוֹלָם. כְּשֵׁם שֶׁהַדָּבָר שֶׁהוּא סִבָּה לְאוֹתָהּ אַהֲבָה אֵינָהּ בָּטֵל, כָּךְ אֵין הָאַהֲבָה בְּטֵלָה: <B>אַהֲבַת אַמְנוֹן וְתָמָר.</B> מִפְּנֵי יָפְיָהּ: <B>אַהֲבַת דָּוִד וְיוֹנָתָן.</B> לְהַשְׁלִים רְצוֹן קוֹנָם. דְּאָמַר לוֹ יְהוֹנָתָן לְדָוִד אַתָּה תִּהְיֶה לְמֶלֶךְ עַל יִשְׂרָאֵל וַאֲנִי אֶהֱיֶה לְךָ לְמִשְׁנֶה: 

~ פרק ה - משנה יז
<B>כָּל מַחֲלֹקֶת שֶׁהִיא לְשֵׁם שָׁמַיִם סוֹפָהּ לְהִתְקַיֵּם.</B> כְּלוֹמַר שֶׁאַנְשֵׁי הַמַּחֲלֹקֶת הַהִיא מִתְקַיְּמִים וְאֵינָם אוֹבְדִין, כְּמַחֲלֹקֶת הִלֵּל וְשַׁמַּאי שֶׁלֹּא אָבְדוּ לֹא תַּלְמִידֵי בֵּית שַׁמַּאי וְלֹא תַּלְמִידֵי בֵּית הִלֵּל. אֲבָל קֹרַח וַעֲדָתוֹ אָבְדוּ. וַאֲנִי שָׁמַעְתִּי, פֵּרוּשׁ סוֹפָהּ, תַּכְלִיתָהּ הַמְבֻקָּשׁ מֵעִנְיָנָהּ. וְהַמַּחֲלֹקֶת שֶׁהִיא לְשֵׁם שָׁמַיִם, הַתַּכְלִית וְהַסּוֹף הַמְבֻקָּשׁ מֵאוֹתָהּ מַחֲלֹקֶת לְהַשִּׂיג הָאֱמֶת, וְזֶה מִתְקַיֵּם, כְּמוֹ שֶׁאָמְרוּ מִתּוֹךְ הַוִּכּוּחַ יִתְבָּרֵר הָאֱמֶת, וּכְמוֹ שֶׁנִּתְבָּאֵר בְּמַחֲלֹקֶת הִלֵּל וְשַׁמַּאי שֶׁהֲלָכָה כְּבֵית הִלֵּל. וּמַחֲלֹקֶת שֶׁאֵינָהּ לְשֵׁם שָׁמַיִם, תַּכְלִית הַנִּרְצֶה בָּהּ הִיא בַּקָּשַׁת הַשְּׂרָרָה וְאַהֲבַת הַנִּצּוּחַ, וְזֶה הַסּוֹף אֵינוֹ מִתְקַיֵּם, כְּמוֹ שֶׁמָּצִינוּ בְּמַחֲלֹקֶת קֹרַח וַעֲדָתוֹ שֶׁתַּכְלִית וְסוֹף כַּוָּנָתָם הָיְתָה בַּקָּשַׁת הַכָּבוֹד וְהַשְּׂרָרָה וְהָיוּ לְהֶפֶךְ: 

~ פרק ה - משנה יח
<B>אֵין חֵטְא בָּא עַל יָדוֹ.</B> כְּדֵי שֶׁלֹּא יְהֵא הוּא בַּגֵּיהִנֹּם וְתַלְמִידָיו בְּגַן עֵדֶן {נט}: <B>אֵין מַסְפִּיקִין בְּיָדוֹ לַעֲשׂוֹת תְּשׁוּבָה.</B> שֶׁלֹּא יְהֵא הוּא בְּגַן עֵדֶן וְתַלְמִידָיו בַּגֵּיהִנֹּם: <B>וְזִכָּה אֶת הָרַבִּים.</B> שֶׁלִּמֵּד תּוֹרָה לְכָל יִשְׂרָאֵל: <B>צִדְקַת ה' עָשָׂה וּמִשְׁפָּטָיו עִם יִשְׂרָאֵל.</B> וּמִשְׁפָּטָיו שֶׁעִם יִשְׂרָאֵל כְּאִלּוּ הוּא עֲשָׂאָן: <B>אֲשֶׁר חָטָא וַאֲשֶׁר הֶחֱטִיא אֶת יִשְׂרָאֵל.</B> מִדְּלֹא קָאָמַר עַל חַטֹּאת יָרָבְעָם וְיִשְׂרָאֵל, שְׁמַע מִנַּהּ שֶׁהַכֹּל תָּלוּי בְּיָרָבְעָם:

<B>מִתַּלְמִידָיו שֶׁל אַבְרָהָם אָבִינוּ.</B> לָמֵד מִמֶּנּוּ וְהוֹלֵךְ בִּדְרָכָיו: <B>עַיִן טוֹבָה.</B> מִסְתַּפֵּק בְּמַה שֶּׁיֵּשׁ לוֹ וְאֵינוֹ חוֹמֵד מָמוֹן אֲחֵרִים. שֶׁכֵּן מָצִינוּ בְּאַבְרָהָם שֶׁאָמַר לְמֶלֶךְ סְדוֹם (בְּרֵאשִׁית יד) אִם מִחוּט וְעַד שְׂרוֹךְ נַעַל וְאִם אֶקַּח מִכָּל אֲשֶׁר לְךָ: <B>וְרוּחַ נְמוּכָה.</B> עֲנָוָה יְתֵרָה. וְכֵן מָצִינוּ אַבְרָהָם אוֹמֵר (שָׁם יח) וְאָנֹכִי עָפָר וָאֵפֶר: <B>וְנֶפֶשׁ שְׁפָלָה.</B> זְהִירוּת וְהַפְּרִישָׁה מִן הַתַּאֲווֹת. וּמָצִינוּ זֶה בְּאַבְרָהָם, דִּכְתִיב (שָׁם יב) הִנֵּה נָא יָדַעְתִּי כִּי אִשָּׁה יְפַת מַרְאֶה אָתְּ, שֶׁעַד עַכְשָׁיו לֹא הִכִּיר בָּהּ מֵרֹב צְנִיעוּת. וּבְבִלְעָם אַשְׁכְּחַן עַיִן רָעָה, שֶׁהָיָה יוֹדֵעַ שֶׁהָיָה רַע בְּעֵינֵי הַמָּקוֹם שֶׁיֵּלֵךְ אֵצֶל בָּלָק וְהָיָה הוֹלֵךְ כְּדֵי לִטֹּל שָׂכָר, דִּכְתִיב (בְּמִדְבָּר כב) אִם יִתֶּן לִי בָלָק מְלֹא בֵיתוֹ כֶּסֶף וְזָהָב: <B>וְרוּחַ גְּבוֹהָה.</B> דְּאָמַר (שָׁם כד) נְאוּם שׁוֹמֵעַ אִמְרֵי אֵל וְיוֹדֵעַ דַּעַת עֶלְיוֹן: <B>וְנֶפֶשׁ רְחָבָה.</B> שֶׁאִם לֹא הָיָה רַב הַתַּאֲוָה לֹא הָיָה יוֹעֵץ לְהַפְקִיר בְּנוֹת מוֹאָב לִזְנוּת {סא}. וְאָמְרוּ חֲכָמִים [סַנְהֶדְרִין קה.] בִּלְעָם בּוֹעֵל אֲתוֹנוֹ הָיָה: <B>לְהַנְחִיל אֹהֲבַי יֵשׁ.</B> אַבְרָהָם אִקְּרֵי אוֹהֵב, דִּכְתִיב (יְשַׁעְיָה מא) זֶרַע אַבְרָהָם אוֹהֲבִי. יֵשׁ, בָּעוֹלָם הַבָּא, וְאוֹצְרוֹתֵיהֶם אֲמַלֵּא, בָּעוֹלָם הַזֶּה: <B>אַנְשֵׁי דָמִים.</B> בִּלְעָם אִקְּרֵי אִישׁ דָּמִים, שֶׁהִפִּיל בַּעֲצָתוֹ עֶשְׂרִים וְאַרְבָּעָה אֲלָפִים מִיִּשְׂרָאֵל {סד}:

<B>עַז כַּנָּמֵר.</B> הַנָּמֵר הַזֶּה נוֹלַד מִן חֲזִיר הַיַּעַר וְהַלְּבִיאָה, כִּי בְּעֵת יַחֵם הָאֲרָיוֹת הַלְּבִיאָה מַכְנֶסֶת רֹאשָׁהּ בְּסִבְכֵי הַיַּעַר וְנוֹהֶמֶת וְתוֹבַעַת אֶת הַזָּכָר, וְהַחֲזִיר שׁוֹמֵעַ קוֹלָהּ וְרוֹבְעָהּ, וְנָמֵר יוֹצֵא מִבֵּין שְׁנֵיהֶן. וּלְפִי שֶׁהוּא מַמְזֵר הוּא עַז פָּנִים, אַף עַל פִּי שֶׁאֵין בּוֹ גְּבוּרָה כָּל כָּךְ. אַף אַתָּה הֱוֵי עַז וְלֹא תִּתְבַּיֵּשׁ לִשְׁאֹל מֵרַבְּךָ מַה שֶּׁלֹּא הֵבַנְתָּ, כְּאוֹתָהּ שֶׁשָּׁנִינוּ לֹא הַבַּיְשָׁן לָמֵד: <B>וְקַל כַּנֶּשֶׁר.</B> לַחֲזֹר אַחַר תַּלְמוּדְךָ, וְלֹא תִיגַע. כְּדִכְתִיב (יְשַׁעְיָה מ) יַעֲלוּ אֵבֶר כַּנְּשָׁרִים יָרוּצוּ וְלֹא יִיגָעוּ: <B>וְרָץ כַּצְּבִי.</B> לִרְדֹּף אַחַר הַמִּצְוֹת: <B>וְגִבּוֹר כָּאֲרִי.</B> לִכְבֹּשׁ אֶת יִצְרְךָ מִן הָעֲבֵרוֹת: <B>עַז פָּנִים.</B> לְפִי שֶׁהָעַזּוּת נִכָּר בַּפָּנִים כְּדִכְתִיב (מִשְׁלֵי כא) הֵעֵז אִישׁ רָשָׁע בְּפָנָיו {סו}, לְפִיכָךְ נִקְרָא עַז פָּנִים: <B>שֶׁתִּבְנֶה עִירְךָ בִּמְהֵרָה בְיָמֵינוּ.</B> כְּלוֹמַר, כְּשֵׁם שֶׁחוֹנַנְתָּנוּ זֹאת הַמִּדָּה, שֶׁסִּימָן לְזֶרַע אַבְרָהָם בַּיְשָׁנִים רַחֲמָנִים וְגוֹמְלֵי חֲסָדִים [יְבָמוֹת עט.], כֵּן יְהִי רָצוֹן שֶׁתִּבְנֶה עִירְךָ וְכוּ':

<B>בֶּן חָמֵשׁ שָׁנִים לַמִּקְרָא.</B> מֵעָרְלָה יָלְפִינַן, דִּכְתִיב (וַיִּקְרָא יט) שָׁלֹשׁ שָׁנִים יִהְיֶה לָכֶם עֲרֵלִים וְגוֹ' וּבַשָּׁנָה הָרְבִיעִית יִהְיֶה כָּל פִּרְיוֹ קֹדֶשׁ הִלּוּלִים, שָׁחוּחַ וְכָפוּף. וְיֵשׁ אוֹמְרִים, לְשׁוֹן שׁוּחָה עֲמֻקָּה: 

~ פרק ה - משנה כב
<B>הֲפֹךְ בָּהּ וַהֲפֹךְ בָּהּ.</B> בַּתּוֹרָה {עד}: <B>דְּכֹלָּא בָהּ.</B> שֶׁהַכֹּל תִּמְצָא בָּהּ: <B>וְסִיב וּבְלֵה בָהּ.</B> גַּם עַד זִקְנָה וְשֵׂיבָה לֹא תַּעַזְבֶנָּה: <B>וּמִנַּהּ לֹא תָּזוּעַ.</B> שֶׁלֹּא תֹּאמַר לָמַדְתִּי חָכְמַת יִשְׂרָאֵל אֵלֵךְ וְאֶלְמַד חָכְמַת יְוָנִית, שֶׁאֵין מֻתָּר לִלְמֹד חָכְמַת יְוָנִית אֶלָּא בְּמָקוֹם שֶׁאָסוּר לְהַרְהֵר בְּדִבְרֵי תּוֹרָה, כְּגוֹן בְּבֵית הַמֶּרְחָץ אוֹ בְּבֵית הַכִּסֵּא. שֶׁשָּׁאֲלוּ אֶת רַבִּי יְהוֹשֻׁעַ מַהוּ לְלַמֵּד אָדָם אֶת בְּנוֹ חָכְמַת יְוָנִית, אָמַר לָהֶם יְלַמְּדֶנּוּ בְּשָׁעָה שֶׁאֵינוֹ לֹא יוֹם וְלֹא לַיְלָה, דְּהָא כְּתִיב וְהָגִיתָ בּוֹ יוֹמָם וָלַיְלָה: 

~ פרק ה - משנה כג
<B>לְפוּם צַעֲרָא אַגְרָא.</B> כְּפִי רֹב הַצַּעַר שֶׁאַתָּה סוֹבֵל בְּלִמּוּד הַתּוֹרָה וַעֲשִׂיַּת הַמִּצְוָה, יִהְיֶה שְׂכָרְךָ מְרֻבֶּה {עח}:
