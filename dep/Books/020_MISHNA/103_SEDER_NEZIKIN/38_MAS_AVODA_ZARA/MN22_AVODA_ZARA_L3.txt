~ פרק א - משנה א

{א} <B> שֶׁל גּוֹיִם.</B> עוֹבְדֵי עֲבוֹדָה זָרָה. וְעַיֵּן בְּפֵרוּשׁ הָרַ"ב מִשְׁנָה ב': {ב} <B> שְׁלֹשָׁה יָמִים.</B> דִּלְהָכִין צָרְכֵי הַקְרָבָה (עַיֵּן פֶּרֶק ה' דְּחֻלִּין מִשְׁנָה ג') צָרִיךְ שְׁלֹשָׁה יָמִים. הִלְכָּךְ אָסוּר לָשֵׂאת וְלָתֵת כוּ', מִשּׁוּם דְּשֵׁם עֲבוֹדָה זָרָה שָׁגוּר בְּפִיו וְאָזֵיל וּמוֹדֶה. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} וְעָבַר יִשְׂרָאֵל מִשּׁוּם לֹא יִשָּׁמַע עַל פִּיךָ. רַשִׁ"י. כְּלוֹמַר לֹא יִשָּׁמַע בִּגְרָם שֶׁלְּךָ. הָרַ"ן: {ד} <B> וְלִשְׁאֹל.</B> שֶׁהַנָּכְרִי יִשְׂמַח כְּשֶׁיִּשְׂרָאֵל צָרִיךְ לוֹ. וּבַגְּמָרָא עֲבִיד צְרִיכוּתָא: {ה} <B> לְהַלְוֹתָן.</B> פֵּרֵשׁ רַבֵּנוּ תָם דַּוְקָא בְחִנָּם, אֲבָל בְּרִבִּית שָׁרֵי. מִשּׁוּם דְּמִצְטַעֵר טוּבָא, כִּדְאָמְרִינַן בְּקִדּוּשִׁין דַּף כ', לְזַבֵּין אֱינִישׁ בְּרַתֵּיהּ וְלֹא לוֹזִיף בְּרִבִּית: 

~ פרק א - משנה ב

{ו} <B> וַחֲכָמִים אוֹמְרִים כוּ'.</B> חֲכָמִים הַיְנוּ תַנָּא קַמָּא. הֵן בְּלֹא אֵידֵיהֶן אִיכָּא בֵינַיְהוּ. תַּנָּא קַמָּא סָבַר הֵן בְּלֹא אֵידֵיהֶן, וְרַבָּנָן בָּתְרָאֵי סָבְרֵי הֵן וְאֵידֵיהֶן. וְאִי בָעֵית אֵימָא כוּ'. גְּמָרָא: {ז} לְשׁוֹן הָרַמְבַּ"ם בְּמִשְׁנָה ד', אֲבָל אֲנַחְנוּ אֲנוּסִים, וְנִתְקַיֵּם בָּנוּ מַה שֶּׁנֶּאֱמַר (דְּבָרִים ד) וַעֲבַדְתֶּם שָׁם אֱלֹהִים עֵץ וָאֶבֶן. עַד כָּאן: {ח} <B> לְאַחַר אֵידֵיהֶן מֻתָּר.</B> וְהַאִידְנָא מֻתָּר לָשֵׂאת וְלָתֵת עִם הָעַמִּים אֲשֶׁר אֲנַחְנוּ יוֹשְׁבִים בֵּינֵיהֶם, דְּכָל הָנֵי דְאִתַּסְּרוּ בְזוֹ הַמַּסֶּכֶת מַיְרֵי בְעוֹבְדֵי אֱלִילִים וַעֲבוֹדָה זָרָה מַמָּשׁ. וְאֵלּוּ הַגּוֹיִים לֹא עוֹבְדֵי עֲבוֹדָה זָרָה מַמָּשׁ הֵם, כְּלוֹמַר שֶׁאֵינָם יוֹדְעִים בְּטִיב עֲבוֹדָה זָרָה כָּל כָּךְ, וְכִדְאָמְרִינַן בְּפֶרֶק קַמָּא דְחֻלִּין דַּף י"ג ב', גּוֹיִים שֶׁבְּחוּץ לָאָרֶץ לָאו עוֹבְדֵי עֲבוֹדָה זָרָה הֵם, אֶלָּא מִנְהַג אֲבוֹתֵיהֶם בִּידֵיהֶם. בֵּית יוֹסֵף סוֹף סִימָן קמ"ח: 

~ פרק א - משנה ג

{ט} גְּמָרָא. וְנִרְאֶה דְּמִשּׁוּם שֶׁאֵלּוּ הַשְּׁמֹנָה יָמִים שֶׁאַחַר הַתְּקוּפָה עָשָׂה אוֹתָן יָמִים טוֹבִים קֹדֶם שֶׁעָשָׂה לְאוֹתָן שֶׁקֹּדֶם הַתְּקוּפָה, שֶׁלֹּא עֲשָׂאָן אֶלָּא לְשָׁנָה הַבָּאָה, לְפִיכָךְ הִקְדִימָן לִשְׁנוֹת, קָלֶנְדָּא קֹדֶם סְטַרְנוּרָא. וּבַיְרוּשַׁלְמִי כֵּיוָן דַּחֲזָא אַיְּמָמָא אָמַר קָלֶנְדָּא. קְלָן דִיא"וֹ. וְנִרְאֶה לִי כִּי קְלָן הוּא לְשׁוֹן שְׂרָרָה וּגְדֻלָּה. כְּמוֹ שֶׁכָּתַב הָרַ"ב פֶּרֶק ד' דְּסֻכָּה מִשְׁנָה ה', קָלֶנְיָא חָפְשִׁי מִמַּס, וְאֵין לְךָ חָפְשִׁי כְּמֶלֶךְ מַלְכֵי הַמְּלָכִים הַקָּדוֹשׁ בָּרוּךְ הוּא, וְהוּא דִיא"וֹ, שֶׁבִּלְשׁוֹן יָוָן קוֹרִין אֱלוֹהַּ דִיא"וֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} בִּלְשׁוֹן יָוָן הוּא קָרַט, לְשׁוֹן תְּפִיסָה. תּוֹסָפוֹת: {יא} בַּגְּמָרָא, גְּנִיסֵי מַלְכוּת, פֵּרוּשׁ תַּכְסִיסֵי מַלְכוּת. עָרוּךְ: {יב} <B> וּבְלוֹרִיתוֹ.</B> פֵּרוּשׁ צִיצִית שֶׁל שֵׂעָר. עָרוּךְ: 

~ פרק א - משנה ד

{יג} <B> לְשָׁם.</B> לְאוֹתָהּ הָעִיר בְּיוֹם עֲבוֹדָה זָרָה שֶׁלָּהֶם לְסַפֵּר עִם אֶחָד מִבְּנֵי הָעִיר. רַשִׁ"י: {יד} <B> אֲסוּרוֹת.</B> מִשּׁוּם דְּנֶהֱנֶה מִשֶּׁל עֲבוֹדָה זָרָה אָסוּר, כְּדִכְתִיב [דְּבָרִים יג] וְלֹא יִדְבַּק בְּיָדְךָ מְאוּמָה מִן הַחֵרֶם, כָּל שֶׁכֵּן לִמְהַנֶּה לַעֲבוֹדָה זָרָה דְּאָסוּר. הָרַ"ן: {טו} וְאִי מִשּׁוּם דְּנוֹשֵׂא וְנוֹתֵן בְּיוֹם אֵידָם, הָנֵי מִלֵּי לְזַבּוֹנֵי לְהוּ, דִּלְמָא מְזַבֵּן לֵיהּ בְּהֵמָה וְאָזֵיל וּמַקְרִיב לַהּ לַעֲבוֹדָה זָרָה. אִי נַמִּי דֶּרֶךְ לוֹקֵחַ לִהְיוֹת שָׂמֵחַ וְאָזֵיל וּמוֹדֶה. אֲבָל לְמִזְבַּן מִנֵּיהּ דָּבָר הַמִּתְקַיֵּם שָׁרֵי, דִּסְתָם מוֹכֵר עָצֵב הוּא. רַשִׁ"י. וְאַף עַל פִּי שֶׁהָרַ"ב פֵּרֵשׁ בְּרֵישׁ פִּרְקִין לִמְכֹּר וְלִקְנוֹת, לֹא קַשְׁיָא כִּדְפֵרֵשׁ הָרֹא"שׁ דִּלְרַשִׁ"י יֵשׁ בִּמְכִירָה צַד אִסּוּר וְצַד הֶתֵּר וְכֵן בִּלְקִיחָה. כְּלוֹמַר צַד אִסּוּר דִּמְכִירָה דָּבָר הַמִּתְקַיֵּם, וְצַד הֶתֵּר שֶׁאֵינוֹ מִתְקַיֵּם. וְדִלְקִיחָה הוּא בְהֵפֶךְ: 

~ פרק א - משנה ה

{טז} <B>הָאֲסוּרִים לִמְכֹּר.</B> לְעוֹלָם. הָרַמְבַּ"ם. וְטַעֲמָא, דְּכֵיוָן דִּלְהַקְרָבָה צָרִיךְ לָהֶם קָא עָבַר עַל לִפְנֵי עִוֵּר לֹא תִתֵּן מִכְשׁוֹל, דְּבֶן נֹחַ מֻזְהָר עַל עֲבוֹדָה זָרָה. נִרְאֶה לִי. וְהַבֵּית יוֹסֵף כָּתַב דְּאֶפְשָׁר שֶׁכָּתַב לְעוֹלָם, שֶׁאֲפִלּוּ שֶׁלֹּא בְיוֹם חַגָּם אָסוּר. אִי נַמִּי לוֹמַר שֶׁלְּעִנְיָן זֶה אֵין חִלּוּק [בֵּין בִּימֵיהֶם לְהָאִידְנָא כְדִלְעֵיל בְּמִשְׁנָה ב']. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} קָשׁוּר. רַשִׁ"י: {יח} רַשִׁ"י וְתוֹסָפוֹת. עוֹד פֵּרְשׁוּ, שֶׁהוּא מִין עֵשֶׂב שֶׁבּוֹ תִחֵם יְהוֹשֻׁעַ אֶת הָאָרֶץ: 

~ פרק א - משנה ז

{יא} <B> הַיַּיִן כוּ'.</B> כְּלוֹמַר, אַף עַל פִּי שֶׁלֹּא יָדַעְנוּ בוֹ שֶׁנִּתְנַסֵּךְ. וּבַגְּמָרָא מַיְתֵי לֵיהּ מִקְּרָא דְּיַיִן שֶׁנִּתְנַסַּךְ אָסוּר בַּהֲנָאָה. וְגָזְרוּ חֲכָמִים עַל סְתָם יֵינָם שֶׁיִּהְיֶה גַם כֵּן אָסוּר בַּהֲנָאָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} <B> שֶׁהָיָה כוּ'.</B> וּפְשִׁיטָא דְמִשּׁוּם דְּאַחְמִיץ לֹא פָּקַע מִנֵּיהּ אִסּוּרָא. וְהַיְנוּ דְכָתַב הָרַ"ב לַאֲפוֹקֵי כוּ': {יג} וְכֵן לְשׁוֹן רַשִׁ"י. וְאִם כֵּן אַף בִּשְׁתִיָּה הָיָה לָנוּ לְהַתִּירוֹ. אֶלָּא נִרְאֶה לִי שֶׁרַשִׁ"י לֹא רָצָה לִכְתֹּב הֶתֵּר בִּשְׁתִיָּה מִטַּעַם שֶׁכָּתְבוּ הַתּוֹסָפוֹת כִּי אֵין אָנוּ בְקִיאִים בְּטִיב חֹמֶץ אִם הוּא חֹמֶץ אוֹ יַיִן: {יד} וּבַעֲבוֹדָה זוֹ מַקְרִיבִין אַף הָעוֹר, וּלְפִיכָךְ כָּל עוֹר שֶׁיֵּשׁ בָּהּ סִימָן כָּזֶה אָסוּר. וְזֶה לְשׁוֹן הָרַמְבַּ"ם, בְּהֵמָה שֶׁהִקְרִיבָהּ כֻּלָּהּ לַעֲבוֹדָה זָרָה אֲסוּרָה בַהֲנָאָה, אֲפִלּוּ פִּרְשָׁהּ וְעַצְמוֹתֶיהָ וְקַרְנֶיהָ וּטְלָפֶיהָ וְעוֹרָהּ, הַכֹּל אָסוּר בַּהֲנָאָה: {טו} <B> הַנִּכְנָס כוּ'.</B> שֶׁהַנָּכְרִי רוֹצֶה לְהַכְנִיסוֹ, מֻתָּר בַּהֲנָאָה אִם נִזְהָר יִשְׂרָאֵל לִטְּלוֹ עִם כְּנִיסָתוֹ קֹדֶם שֶׁיַּקְרִיבוּ אוֹתוֹ. רַשִׁ"י: {טז} <B> מִפְּנֵי כוּ'.</B> מַהֲדַר וְתָנֵי מִפְּנֵי כוּ', לוֹמַר שֶׁאָסוּר וּמְטַמֵּא. וְרַבֵּנוּ תָם פֵּרֵשׁ דְּהָכָא בַחֲתִיכָה שֶׁאֵינוֹ זִבּוּחַ לִפְנֵי עֲבוֹדָה זָרָה, אֶלָּא שֶׁמְּבִיאִים דּוֹרוֹן, וּלְהָכִי תָנֵי כְזִבְחֵי. וּבַיְרוּשַׁלְמִי גָרַס זִבְחֵי:

{יז} הַיְנוּ שְׁלֹשָׁה יָמִים. וּמִשּׁוּם הָכִי מְעָרָן שְׁלֹשָׁה יָמִים, דִּכְבוֹלְעוֹ כָּךְ פּוֹלְטוֹ (מָרְדְּכָי). וּבְפָחוֹת מִכֵּן מְשַׁכְשֵׁךְ. טוּר:

{יח} <B> חֲמוּרָה.</B> שֶׁהָעוֹלָה אֲסוּרָה בַהֲנָאָה וּמוֹעֲלִין בָּהּ. רַשִׁ"י: {יט} <B> יָפָה.</B> שֶׁאֵינוֹ קָץ וּמוֹאֵס בָּהּ. שׂוֹרְפָהּ, גּוֹמְעָהּ. וְלֹא הוֹדוּ לָאוֹמְרִים כֵּן לְהַתִּירָהּ לְכַתְּחִלָּה, אֲבָל אָמְרוּ לֹא נֶהֱנִין וְלֹא מוֹעֲלִין כוּ'. וְהָכִי מַשְׁמַע, לֹא נֶהֱנִים מִדְּרַבָּנָן, וְאִם נֶהֱנָה אֵין בָּהּ מְעִילָה (כְּלוֹמַר מִדְּאוֹרַיְתָא). אַלְמָא מִדְּלֵית בָּהּ מְעִילָה שְׁמַע מִנַּהּ הַאי חָלָב הַכָּנוּס בְּקֵבַת הָעוֹלָה, כְּפִרְשָׁא בְעָלְמָא הוּא. רַשִׁ"י. וּבְעוֹלָה גָזְרוּ, מִשּׁוּם דִּבְעוֹלָה יֵשׁ מְעִילָה מִן הַתּוֹרָה, מַה שֶּׁאֵין כֵּן בִּנְבֵלָה דְלֵית בָּהּ אֶלָּא אִסּוּר אֲכִילָה, הִלְכָּךְ לֹא שַׁיָּךְ לֶאֱסֹר בָּהּ, דְּלָאו מִידֵי דְמֵיכַל הוּא: {כ} <B> שֶׁמַּעֲמִידִין כוּ'.</B> לֹא הָיָה מְכַוֵּן כִּי אִם לִדְחוֹתוֹ. דְּהָא אֲפִלּוּ לְרַבִּי מֵאִיר דְּחָיֵשׁ לְמִעוּטָא, לֹא אָסַר אֶלָּא גְבִינוֹת בֵּית אֻנְיָקִי, אֲבָל שְׁאָר מְקוֹמוֹת חָשֵׁיב מִעוּטָא דְמִעוּטָא. תּוֹסָפוֹת: {כא} <B> הִשִּׂיאוֹ.</B> לְשׁוֹן מַשִּׂיא עֵצָה. וְאִית דְּגָרְסֵי הִשִּׁיאוֹ. וּנְפָרֵשׁ הִטְעָהוּ לְשׁוֹן הִשִּׁיאַנִי [בְּרֵאשִׁית ג]: {כב} גְּמָרָא, וּמְסַיֵּם בְּמִדְרַשׁ שִׁיר הַשִּׁירִים [רַבָּה]. וְרַבִּי יִשְׁמָעֵאל הָיָה קָטָן כוּ', מְגַלֶּה לָהֶם סִתְרֵי תוֹרָה: {כג} וְעוֹר לָאו פִּרְשָׁא הוּא וְאָסוּר. רַשִׁ"י. וּשְׁחִיטָה, לִרְבוּתָא נָקַט: {כד} <B> דֹּדֶיךָ.</B> לְשׁוֹן זָכָר. דֹּדַיִךְ, לְשׁוֹן נְקֵבָה. וְהָכִי קָאָמְרָה כְנֶסֶת יִשְׂרָאֵל, יִשָּׁקֵנִי הַקָּדוֹשׁ בָּרוּךְ הוּא מִנְּשִׁיקוֹת פִּיהוּ כִּי כָךְ אָמַר לִי טוֹבִים דֹּדַיִךְ מִיַּיִן. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כה} <B> דֹּדַיִךְ.</B> מִשּׁוּם דִּבְהָכִי מִתְיַשֵּׁב רֵישֵׁיהּ דִּקְרָא לַנִּסְתָּר, יִשָּׁקֵנִי, וְכֵן פִּיהוּ. וְסֵיפָא דֹּדֶיךָ לְנוֹכֵחַ. אֶלָּא דְהָכִי קָאָמַר יִשָּׁקֵנִי כוּ' כִּי כָךְ אָמַר לִי כִּי טוֹבִים כוּ'. שָׁמַעְתִּי מִפִּי רַבִּי מֵאִיר בַּאךְ זִכְרוֹנוֹ לִבְרָכָה: {כו} וְהַיְנוּ דְשַׁיְלֵיהּ לְהַאי קְרָא, כְּלוֹמַר, כְּנֶסֶת יִשְׂרָאֵל אוֹמְרִים טוֹבִים תַּקָּנוֹת כוּ' וְאַף עַל פִּי שֶׁאֵין אָנוּ יוֹדְעִים הַטְּעָמִים, כִּי טַעְמָם וְנִמּוּקָם עִמָּהֶם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כז} <B> חֲבֵרוֹ כוּ'.</B> שֶׁהוּא לְשׁוֹן זָכָר, וּכְנֶסֶת יִשְׂרָאֵל הִיא דְקָאָמְרָה לִפְנֵי שְׁכִינָה {כח} <B> חָלָב כוּ'.</B> וְאַף עַל גַּב דְּטָהוֹר חִוַּר, וְטָמֵא יָרוֹק, חָיְשִׁינַן שֶׁמָּא נִתְעָרֵב. וַאֲפִלּוּ לְהַעֲמִיד בּוֹ גְבִינָה אָסוּר, דְּשֶׁמָּא הַטָּמֵא שֶׁאֵינוֹ מַעֲמִיד יִשָּׁאֵר בַּגֻּמּוֹת עִם הַנַּסְיוּבֵי. גְּמָרָא. אֲבָל לְעֵיל בִּגְבִינוֹת הַנָּכְרִים לֵיכָּא לְמֵיסַר מֵהַאי טַעֲמָא, דְּכֵיוָן שֶׁהַנָּכְרִי עוֹשֶׂה לְצָרְכּוֹ לֵיכָּא לְמֵיחַשׁ דִּלְמָא עֵרֵב בּוֹ דָבָר טָמֵא. תּוֹסָפוֹת: {כט} <B> וּכְבָשִׁין.</B> לִכְבֹּשׁ דָּגִים וְיָרָק, וְרֹאשׁ וְרַגְלַיִם שֶׁל בְּהֵמָה. רַשִׁ"י: {ל} <B> שֶׁדַּרְכָּן כוּ'.</B> אֲבָל בְּיָדוּעַ, אָסוּר בַּהֲנָאָה. וּפֵרֵשׁ הַטּוּר, שֶׁדַּרְכָּן, הַיְנוּ שֶׁלִּפְעָמִים הָיוּ נוֹתְנִין בָּהֶן יַיִן. וְשֶׁיָּדוּעַ, הַיְנוּ שֶׁכֻּלָּן נוֹתְנִין לְתוֹכָן יַיִן אַף עַל פִּי שֶׁאֵינוֹ יוֹדֵעַ שֶׁנָּתְנוּ לְתוֹךְ אֵלּוּ. עַד כָּאן. וּמַאי שְׁנָא מִמֻּרְיָס (דְּמִשְׁנָה ד) דְּשָׁרֵי בַהֲנָאָה (וְרֻבָּא דְעָלְמָא שָׁדוּ בֵיהּ חַמְרָא). הָתָם לַעֲבוֹרֵי זֻהֲמָא (וְהָוֵי כְמַאן דְּאָזֵיל לְאִבּוּד), הָכָא לְמַתּוֹקֵי טַעֲמָא. גְּמָרָא: {לא} <B> וָחֹמֶץ.</B> אַדִּלְעֵיל קָאֵי, שֶׁדַּרְכָּן לָתֵת לְתוֹכָן יַיִן וָחֹמֶץ. וְאֵין לְפָרֵשׁ דְּחֹמֶץ הַיְנוּ שֶׁלֹּא הָיָה מִתְּחִלָּתוֹ יַיִן וּכְשֶׁקָּנָה חֹמֶץ מִיִּשְׂרָאֵל, דְּהָא מְבֹאָר לְעֵיל דְּאַף בִּשְׁתִיָּה שָׁרֵי מִן הַדִּין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לב} דְּאִי לָאו הָכִי הָא אָמַר מַר נוֹתֵן טַעַם לִפְגָם מֻתָּר. גְּמָרָא: {לג} <B> הֲרֵי כוּ'.</B> לִמְעוּטֵי בְיָדוּעַ דִּכְבָשָׁן בְּיַיִן. גְּמָרָא:

{טו} וּבַיְרוּשַׁלְמִי מְפָרֵשׁ, כְּשֶׁקָּדַם הַכֹּתֶל לָאֱלִיל, וּלְפִיכָךְ כְּשֶׁסָּמַךְ יִשְׂרָאֵל בְּהֶתֵּר סָמַךְ. הָרַ"ן: {טז} דְּלֹא בָעֵי צְנִיעוּתָא, דְּאִלּוּ לִגְדוֹלִים צְנִיעוּתָא בָעֵי. גְּמָרָא. וּמַה שֶּׁכָּתַב וְעוֹשֶׂה כוּ', אוֹ אוֹ קָאָמַר. וְכֵן הוּא בַגְּמָרָא: {יז} וְאַף עַל גַּב דְּבִדְרַבָּנָן קַיְמָא לָן דְּיֵשׁ בְּרֵרָה, יֵשׁ לוֹמַר דְּמִשּׁוּם חֻמְרַת עֲבוֹדָה זָרָה אָמְרוּ דְאֵין בְּרֵרָה: {יח} דְּאִתַּקַּשׁ נַמִּי לְמֵת, שֶׁנֶּאֱמַר וַיַּשְׁלֵךְ אֶת עֲפָרוֹ עַל קֶבֶר בְּנֵי הָעָם [מְלָכִים ב כג]. גְּמָרָא: {יט} <B> שַׁקֵּץ.</B> לְשׁוֹן שֶׁרֶץ. רַשִׁ"י: {כ} <B> כַּנִּדָּה.</B> וְאִתְּקִישׁ לְשֶׁרֶץ לִמְשַׁמְּשֶׁיהָ. וְתַנָּא קַמָּא סָבַר דְּאִתְּקִישׁ לְנִדָּה שֶׁאֵינָהּ מְטַמְּאָה לְאֵבָרִים אִם נֶחְתַּךְ אֵבֶר מִמֶּנָּה, כְּנִדָּה שֶׁאֵינָהּ מְטַמְּאָה לְאֵבָרִים. גְּמָרָא: {כא} <B> שֶׁנֶּאֱמַר כוּ'.</B> בַּעֲבוֹדָה זָרָה מִשְׁתַּעֵי. רַשִׁ"י:

{כב} רַשִׁ"י. וּמִשּׁוּם הָכִי אָסוּר מִיָּד, דְּאִי לָאו הָכִי הָוֵי מְשַׁמְּשֵׁי עֲבוֹדָה זָרָה דְּלֹא נֶאֱסָר עַד שֶׁיֵּעָבֵד. הָרַ"ן: {כג} וּלְשׁוֹן אֲרַמִּית בְּבֵית אֲרָזִים, בְּבֵיתָא דְמִטַּלְּלָא בְכִיּוּרָא דְאַרְזַיָּא. וַאֲחֵרִים כֵּן: {כד} וְאֶת הַבַּיִת עַצְמוֹ לֹא נִתְכַּוֵּן לַעֲבֹד, וְהוֹצִיאָהּ יִשְׂרָאֵל, הֲרֵי בַיִת זֶה מֻתָּר. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כה} וּמְסַיֵּם רַשִׁ"י, וְעוֹבְדִין אֶת הַבִּימוֹס עַצְמוֹ כַעֲבוֹדָה זָרָה, הֲרֵי זוֹ אֲסוּרָה וַאֲפִלּוּ לֹא הֶעֱמִיד עָלֶיהָ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כו} וְכֵן פֵּרֵשׁ רַשִׁ"י כָאן וּלְעֵיל בְּבַיִת. וּמְסַיֵּם וַאֲפִלּוּ שֶׁלֹּא בִטְּלוֹ הָעוֹבֵד כּוֹכָבִים אוֹ שֶׁהִיא שֶׁל יִשְׂרָאֵל מוּמָר כוּ', הָכָא בָטִיל, שֶׁאֵין נֶאֱסָר אֶלָּא חִדּוּשׁ. וְהַאי חִדּוּשׁ אָסוּר בַּהֲנָאָה לְעוֹלָם. וּבַגְּמָרָא מוֹקֵי לַהּ כְּגוֹן שֶׁלֹּא נֶעֱבַד הַבַּיִת מֵעוֹלָם, דְּאִי לָאו הָכִי בָּעֵי נָכְרִי לְבַטְּלָהּ. וְאִם הִשְׁתַּחֲוָה לָהּ יִשְׂרָאֵל, אֲסוּרָה עוֹלָמִית. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כז} וְאִי לֹא גִדְּעוֹ וּפִסְּלוֹ, לֹא. דִּכְרַבָּנָן דִּסְבִירָא לְהוּ בְמִשְׁנָה ה' דְּאִילָן שֶׁנְּטָעוֹ וּלְבַסּוֹף עֲבָדוֹ מֻתָּר. וּמִיהוּ מַסְּקִינַן בַּגְּמָרָא דִּבְתוֹסֶפֶת מוֹדוּ רַבָּנָן. וְדִתְנַן גִּדְּעוֹ וּפִסְּלוֹ, לְאַשְׁמְעִינַן דְּאַף עַל גַּב דְּעָשָׂה מַעֲשֶׂה בְגוּף הָאִילָן, אֲפִלּוּ הָכִי אֵין הָעִקָּר נֶאֱסָר: {כח} <B> וּבִטְּלָהּ.</B> וְאַף עַל גַּב דְּכִי לֹא בִטְּלָהּ נַמִּי לֹא הָוֵי מְשַׁמְּשֶׁיהָ, דְּהָא אַלִּיבָּא דְרַבָּנָן קָא מוֹקְמֵי לַהּ, דִּנְטָעוֹ וּלְבַסּוֹף עֲבָדוֹ מֻתָּר כוּ', מִכָּל מָקוֹם כָּל זְמַן שֶׁלֹּא בִטְּלָהּ אִסּוּרָא דְרַבָּנָן אִיכָּא, שֶׁבְּשָׁעָה שֶׁאֱלִיל נִרְאֵית תַּחְתֶּיהָ נִרְאֶה כְאִלּוּ נְטוּעָה מִתְּחִלָּה לְכָךְ. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כט} <B> בְּצִלָּהּ.</B> מִשּׁוּם דְּקָא מִתְהַנֵי מֵעֲבוֹדָה זָרָה. וְדַוְקָא לֵישֵׁב, דְּחָיְשִׁינַן שֶׁמָּא יִשְׁהֶה בִישִׁיבָתוֹ וְאָתֵי לְכַוּוֹנֵי. דְּאִי לָאו הָכִי קַיְמָא לָן אֶפְשָׁר וְלֹא מְכַוֵּין שָׁרֵי. אֲבָל הַעֲבָרָה בְעָלְמָא שָׁרֵי. וְתַחְתֶּיהָ דְאָסוּר מִשּׁוּם טֻמְאַת עֲבוֹדָה זָרָה. הָרַ"ן: {ל} וְכָל זְמַן שֶׁלֹּא עָבְרָה מִדַּת אֹרֶךְ הַצֵּל אֶת מִדַּת גֹּבַהּ קוֹמַת הָאִילָן הַצֵּל עָב וְחָשׁוּךְ. מִכָּאן וָאֵילָךְ הַצֵּל דַּק וְקָלוּשׁ. רַשִׁ"י: {לא} <B> טָמֵא.</B> מַאי טַעֲמָא, אִי אֶפְשָׁר דְּלֵיכָּא תִּקְרֹבֶת עֲבוֹדָה זָרָה (וְנִמְצָא הָאִילָן מַאֲהִיל עַל הַתִּקְרֹבֶת וְעָלָיו). מַנִּי, רַבִּי יְהוּדָה בֶן בְּתֵירָא הוּא, דְּאָמַר מִנַּיִן לְתִקְרֹבֶת עֲבוֹדָה זָרָה שֶׁמְּטַמֵּא בְאֹהֶל, שֶׁנֶּאֱמַר וַיִּצָּמְדוּ וְגוֹ' וַיֹּאכְלוּ זִבְחֵי מֵתִים [תְּהִלִּים קו], מַה מֵּת מְטַמֵּא בְאֹהֶל אַף תִּקְרֹבֶת עֲבוֹדָה זָרָה וְכוּ'. גְּמָרָא. וְדַוְקָא תִקְרֹבֶת, אֲבָל בָּאֱלִיל עַצְמָהּ לֵיכָּא טֻמְאַת אֹהֶל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לב} <B> וְעָבַר.</B> בִּדְאִיכָּא דַרְכָּא אַחֲרִינָא (הַקָּצָר כָּזֶה. תּוֹסָפוֹת) מִשּׁוּם הָכִי דִיעֲבַד אִין, לְכַתְּחִלָּה לֹא. אֲבָל בִּדְלֵיכָּא דַרְכָּא אַחֲרִינָא עוֹבֵר אַף לְכַתְּחִלָּה. גְּמָרָא: {לג} וּקְרָא דְמַיְתִינַן, אַסְמַכְתָּא הִיא. רַשִׁ"י: {לד} <B> אֲבָל כוּ'.</B> וְלֹא קָתָנֵי וְאִם זָרַע אָסוּר כְּמוֹ דְאִם עָבַר טָמֵא, דְּלֹא אָסְרוּ הַהֲנָאָה בְדִיעֲבַד. הָרֹא"שׁ: {לה} וְלֹא קַיְמָא לָן כְּמַתְנִיתִין דַּאֲבָל לֹא בִימוֹת הַחַמָּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יג} <B> יְאַבֵּד כוּ'.</B> וְאִם תֹּאמַר, יְשִׁיבוּם, יְאַבֵּד הַשּׁוֹטִים, לֹא קַשְׁיָא, שֶׁהֲרֵי מָצִינוּ שֶׁאוֹבְדִים בִּמְלֹאת סִפֻּקָם וּסְאָתָם, כְּעִנְיַן יוֹשְׁבֵי אֶרֶץ כְּנַעַן כוּ', כְּמוֹ כֵן אֵלּוּ הַשּׁוֹטִים כְּשֶׁתִּתְמַלֵּא סְאָתָם יֹאבֵדוּ. וְקֹדֶם לָכֵן לֹא. שֶׁאִם יֹאבְדוּ קֹדֶם לָכֵן הֲרֵי זֶהוּ בִכְלַל יְאַבֵּד עוֹלָמוֹ, לְפִי שֶׁהָעוֹלָם נִבְרָא שֶׁתְּהֵא הָרְשׁוּת נְתוּנָה בִידֵי אָדָם לְהֵימִין אוֹ לְהַשְׂמִאיל. וּלְפִיכָךְ אֵינָם נֶאֱבָדִים קֹדֶם שֶׁנִּתְמַלֵּא סְאָתָם. נִרְאֶה לִי. וּבְסֵפֶר דֶּרֶךְ חַיִּים עַל מִשְׁנָה [אָבוֹת פֶּרֶק ג'] הַכֹּל צָפוּי כוּ', פֵּרֵשׁ שֶׁאִם יְאַבֵּד הָעוֹבְדִים לֹא הָיָה הָאָדָם בַּעַל בְּחִירָה, שֶׁהֲרֵי יָרֵא מִן הָאִבּוּד לְגַמְרֵי:

{יד} <B> אַף עַל פִּי כוּ'.</B> דְּלֹא תֵימָא דְלָא שָׁרִינַן לִקַּח גַּת בְּעוּטָה אַף עַל פִּי שֶׁסְּתָם יֵינָם אָסוּר כְּדִלְקַמָּן אֶלָּא הֵיכָא דְיָדְעִינַן דְּלֹא נָגַע, אֲבָל כִּי נָגַע אוֹסֵר, קָא מַשְׁמַע לָן דְּהָכָא שָׁרֵי, מִשּׁוּם דְּאַכַּתִּי לָאו יַיִן הוּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} שֶׁהַגַּת עֲשׂוּיָה כְמִדְרוֹן, וּמִשָּׁעָה שֶׁהוּא נִמְשָׁךְ מִצַּד הָעֶלְיוֹן לְצַד הַתַּחְתּוֹן קָרוּי יַיִן. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טז} <B> לַבּוֹר.</B> בּוֹר הָיוּ עוֹשִׂין לִפְנֵי הַגַּת. וְהַכְּלִי נָתוּן שָׁם לְקַבֵּל הַיַּיִן. וְיֵשׁ שֶׁטָּחִין הַבּוֹר בְּסִיד וְהַיַּיִן מִשְׁתַּמֵּר בְּתוֹכוֹ וְאֵינוֹ מְאַבֵּד טִפָּה, וְהוּא בּוֹר סוּד. רַשִׁ"י: {יז} <B> וְהַשְּׁאָר מֻתָּר.</B> וַאֲפִלּוּ יִגַּע בּוֹ נָכְרִי. דּוּמְיָא דְרֵישָׁא דִנְתִינָה לַתַּפּוּחַ מַיְרֵי בִנְגִיעָה, אַף מֻתָּר דְּסֵיפָא מַיְרֵי בִנְגִיעָה. תּוֹסָפוֹת: {יח} <B> מֻתָּר.</B> אַף בִּשְׁתִיָּה. דּוּמְיָא דְרֵישָׁא דְלוֹקְחִין מַשְׁמַע דְּמֻתָּר אַף בִּשְׁתִיָּה. תּוֹסָפוֹת:

{יט} וְאֵין לְפָרֵשׁ בִּתְחִלַּת דְּרִיכָה וּמִטַּעַם שֶׁנִּטְמְאוּ הַכֹּל בְּגַת טְמֵאָה, דְּאִם כֵּן יִשְׂרָאֵל דְּסֵיפָא אַמַּאי לֹא דוֹרְכִין עִמּוֹ כֵּיוָן שֶׁכְּבָר נִטְמְאוּ בַגַּת. תּוֹסָפוֹת: {כ} וּמִשּׁוּם מְסַיֵּעַ יְדֵי עוֹבְרֵי עֲבֵרָה לֵיכָּא, שֶׁהָעוֹבֵד כּוֹכָבִים לֹא נִצְטַוָּה עַל כָּךְ. רַשִׁ"י: {כא} תֵּימַהּ, תֵּיפּוּק לֵיהּ אִי מִשּׁוּם טֻמְאַת הַגַּת לְחוּד, אִי מִשּׁוּם מַגָּע לְחוּד, וּשְׁתֵּיהֶן לָמָּה, וְהָרַ"ב הִרְכִּיב שְׁנֵי פֵרוּשִׁים יַחַד, שֶׁזֶּה שֶׁאָמַר לְפִי שֶׁנּוֹתְנָם בְּגַת טְמֵאָה, הוּא פֵרוּשׁ רַשִׁ"י לְפִי שֶׁעַל יְדֵי מַשְׁקִין הַיּוֹצְאִין בִּשְׁעַת דְּרִיכָה בַגַּת הֻכְשְׁרוּ וְנִטְמְאוּ בַגַּת הַטְּמֵאָה, וְקֹדֶם דְּרִיכָה אַף בְּמַגָּע לֹא נִטְמְאוּ, כֵּיוָן שֶׁלֹּא הֻכְשְׁרוּ וַדַּאי, דְּהָא דְהַבּוֹצֵר לַגַּת הֻכְשָׁר [בְּפֶרֶק קַמָּא דְשַׁבָּת] הַיְנוּ הֵיכָא דְהֻכְשָׁר וַדַּאי אֲבָל לֹא בִסְתָמָא. וּמַה שֶּׁכָּתַב הָרַ"ב וְהָעוֹבֵד כּוֹכָבִים מְטַמֵּא בְמַגָּעוֹ הוּא פֵרוּשׁ הָרַמְבַּ"ם דִּסְבִירָא לֵיהּ דִּסְתָמָא הֻכְשְׁרוּ בַּבְּצִירָה. וְאֵין שְׁנֵי הַפֵּרוּשִׁים עוֹלִים כְּאֶחָד: {כב} מִשּׁוּם מַגַּע עוֹבֵד כּוֹכָבִים לַעֲשׂוֹתוֹ יֵין נֶסֶךְ אִתְּמַר בַּגְּמָרָא: {כג} <B>לֹא דוֹרְכִין.</B> שֶׁדְּרִיכָתָן בַּעֲבֵרָה בְּגַת טְמֵאָה, וּדְרִיכָתָן הִיא עִקַּר טֻמְאָתָן, כְּשֶׁנַּעֲשִׂין מַשְׁקֶה הֵן טְמֵאִים עַל יְדֵי הַגַּת. רַשִׁ"י. וְאַשְׁמְעִינַן דַּאֲפִלּוּ בְסוֹף דְּרִיכָה שֶׁכְּבָר נִטְמְאוּ אָסוּר לְסַיֵּעַ יְדֵי עוֹבְרֵי עֲבֵרָה, וּמִשּׁוּם דִּדְרִיכָה חַד מַעֲשֶׂה הוּא. תּוֹסָפוֹת. וּלְהָרַמְבַּ"ם, בְּצִירָה בִכְלַל מְלֶאכֶת הַדְּרִיכָה הִיא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כד} <B> וְלֹא כוּ'.</B> שֶׁכָּל מַעֲשֶׂה עֲרִיכָה זוֹ בַעֲבֵרָה. רַשִׁ"י:

{כה} <B> אָסוּר.</B> שֶׁמָּא נָגַע בּוֹ. לְשׁוֹן הָרַמְבַּ"ם, כָּל יַיִן שֶׁיִּגַּע בּוֹ הָעוֹבֵד כּוֹכָבִים הֲרֵי זֶה אָסוּר, שֶׁמָּא נִסֵּךְ אוֹתוֹ, שֶׁמַּחֲשֶׁבֶת הָעוֹבֵד כּוֹכָבִים לָעֲבוֹדָה זָרָה. הָא לָמַדְתָּ שֶׁיַּיִן שֶׁל יִשְׂרָאֵל שֶׁנָּגַע בּוֹ הָעוֹבֵד כּוֹכָבִים דִּינוֹ כִסְתָם יֵינוֹ [שֶׁהוּא אָסוּר בַּהֲנָיָה]: {כו} שֶׁעָשָׂה לוֹ יִשְׂרָאֵל יַיִן זֶה אַפּוֹתֵיקֵי לִגְבּוֹת הֵימֶנּוּ. רַשִׁ"י: {כז} <B> מֻתָּר.</B> דְּנִתְפָּס כְּגַנָּב אֲפִלּוּ עַל הַכְּנִיסָה, וּמִשּׁוּם הָכִי לֹא חָיְשִׁינַן שֶׁמָּא נָגַע, דְּמִרְתַּת. הָרַ"ן: {כח} וּמִסְּתָמָא נַסְּכֵיהּ בַּעֲלִיָּתוֹ, וְאָסוּר בַּהֲנָאָה. רַשִׁ"י:

{א} <B> הַשּׂוֹכֵר כוּ'.</B> עוֹבֵד כּוֹכָבִים שֶׁשָּׂכַר אֶת יִשְׂרָאֵל. רַשִׁ"י: {ב} וּמִשּׁוּם חֻמְרָא דְיֵין נֶסֶךְ קָנְסוּ. אַף עַל גַּב דְּבִשְׁבִיעִית לֹא קָנְסוּ פוֹעֵל דְּלֹא נְפִישׁ אַגְרֵיהּ: {ג} וּבְחָבִיּוֹת שֶׁאֵינָן שֶׁל יֵין נֶסֶךְ אֶלָּא כְגוֹן שֶׁל שֵׁכָר, וְאַף עַל פִּי שֶׁאוֹמֵר לוֹ הַעֲבֵר לִי חָבִית שֶׁל יֵין נֶסֶךְ [פֵּרוּשׁ, אַף עַל פִּי שֶׁבְּשָׁעָה שֶׁשְּׂכָרוֹ הוֹדִיעוֹ שֶׁבִּכְלַל הַמְּלָאכָה אִיכָּא חָבִית שֶׁל יֵין נֶסֶךְ] שְׂכָרוֹ מֻתָּר. וּפֵרֵשׁ רַשִׁ"י, שָׁדֵי פְרוּטָה לְנַהֲרָא. אִי נַמִּי לֹא שָׁקֵיל מִיַּד עוֹבֵד כּוֹכָבִים, וְאִינָךְ מִשְׁתְּרֵי דְּהָא לֹא שַׁיָּךְ שְׂכַר הַהוּא חָבִית בַּהֲדֵי אִינָךְ, דְּכָל חֲדָא קִנְיָן אַגְרָא בְאַפֵּי נַפְשֵׁיהּ: {ד} דְּכָל כַּמָּה דְלָא אַעַבְרִינְהוּ לְכֻלְּהוּ לֹא יָהֵיב לֵיהּ מִידֵי, הִלְכָּךְ אַגְרָא שַׁיָּךְ בֵּיהּ. רַשִׁ"י. דְּכֻלֵּהּ אַגְרָא אַכָּל חֲדָא וַחֲדָא יָהֵיב. הָרַמְבַּ"ן. וַאֲפִלּוּ לֹא הוֹדִיעוֹ בִשְׁעַת הַשְּׂכִירוּת אֶלָּא נִמְצָא אַחַר כָּךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} דְּכֵיוָן דְּאִי לֹא מוֹתִיב לֵיהּ לֹא מָצֵי אָמַר לֵיהּ נַכֵּי לֵיהּ אַגְרָא דְלָגִינָא, הֲרֵי זֶה כְאִלּוּ לֹא שְׂכָרָהּ לְכָךְ. גְּמָרָא: 

~ פרק ה - משנה ב

{ו} <B> מֻתָּרוֹת.</B> בַּיְרוּשַׁלְמִי מְפָרֵשׁ דְּמַיְרֵי בַעֲנָבִים הַמְחֻבָּרִים לָאֶשְׁכּוֹל. אֲבָל אִם נִטַּל הָעֹקֶץ הֲווּ כִמְבֻקָּעוֹת מִפְּנֵי נֶקֶב הַנִּשְׁאָר בִּזְנָבוֹ. תּוֹסָפוֹת: {ז} <B> נָפַל עַל גַּבֵּי וְכוּ' אָסוּר.</B> לֹא גָרְסִינַן בִּסְפָרִים אֲחֵרִים. דְּאִי גָרְסִינַן הֲוָה לָן לְמֵימַר חֲסוּרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, אֲבָל בִּתְאֵנִים יְבֵשִׁים נוֹתֵן טַעַם לִפְגָם הוּא: {ח} <B> מֻתָּר.</B> בַּגְּמָרָא יָלְפִינַן לֵיהּ מִקְּרָא: {ט} פֵּרוּשׁ בְּמַשְׁקֶה הַנַּעֲשֶׂה מִדְּבַשׁ. [דְּאִלּוּ הַדְּבַשׁ עַצְמוֹ מְשֻׁבָּח בַּבָּשָׂר]: {י} תֵּימַהּ לִי, דְּמָה עִנְיָן זֶה לְמַשְׁבִּיחַ וּלְבַסּוֹף פּוֹגֵם, דְּלֹא אָמְרָן אֶלָּא עַל הַדָּבָר שֶׁבִּשְׁעַת תַּעֲרָבְתּוֹ מַשְׁבִּיחַ וּלְבַסּוֹף פּוֹגֵם, וּלְכָךְ יֵשׁ לֶאֱסֹר הַמַּאֲכָל הַהוּא מִפְּנֵי שֶׁבִּתְחִלַּת תַּעֲרָבְתּוֹ הָיָה מְשֻׁבָּח כוּ', אֲבָל בִּכְלִי שֶׁאֵינוֹ בֶן יוֹמוֹ כְּבָר נִפְגַּם הַטַּעַם וְנִמְצָא שֶׁזֶּה הַתַּעֲרֹבֶת פּוֹגֵם מִתְּחִלָּה וְעַד סוֹף הוּא כוּ'. וּבַגְּמָרָא דְמִשּׁוּם גְּזֵרָה אַטּוּ בֶן יוֹמוֹ אָסוּר, אֲבָל מִשּׁוּם עַצְמוֹ נוֹתֵן טַעַם לִפְגָם מַמָּשׁ הוּא: 

~ פרק ה - משנה ג

{יא} <B> מַעֲבִיר כוּ'.</B> מַיְרֵי בְחָבִית סְתוּמָה בִּפְקָק שֶׁל עֵץ וְכַיּוֹצֵא בוֹ. אַף עַל גַּב דְּהָעוֹבֵד כּוֹכָבִים יָכוֹל לְסַלְּקוֹ וְלִגַּע בְּלֹא זִיּוּף כְּלָל, כֵּיוָן דִּבְחֶזְקַת מִשְׁתַּמֵּר מֻתָּר. אֲבָל בִּפְתוּחִים לְגַמְרֵי לֹא, דְּכֵיוָן שֶׁהֵם עַל כְּתֵפוֹ וְיָכוֹל לִגַּע בָּהֶן בְּהֶדְיָא, חָיְשִׁינַן דִּלְמָא נָגַע וְלֹא מִרְתַּת כְּלָל, דְּמִשְׁתַּמֵּט לְמֵימַר לְהַחֲזִיק בֶּחָבִית נִתְכַּוַּנְתִּי שֶׁלֹּא יִפֹּל. וְאַף עַל גַּב דִּסְתָמָא קָתָנֵי, וּמַשְׁמַע דְּאַיְרֵי בִּפְתוּחוֹת דּוּמְיָא דְאַחֲרִינֵי, סְתָמָא דְמַעֲבִיר חָבִית מִמָּקוֹם לְמָקוֹם הַיְנוּ סָתוּם קְצָת שֶׁלֹּא יֵצֵא הַיַּיִן בְּטִלְטוּל הֶחָבִית, וַאֲפִלּוּ הָכִי פְּקָק שֶׁל עֵץ פְּתוּחוֹת מִקְּרוּ. הָרֹא"שׁ: {יב} אֲבָל הַתּוֹסָפוֹת דִּקְדְּקוּ דְּדַוְקָא בְּבָא דֶרֶךְ עֲקַלָּתוֹן, שֶׁיָּכוֹל לָבֹא עָלָיו פִּתְאוֹם שֶׁלֹּא יִרְאֶנּוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} וְזֶה לְשׁוֹן הָרֹא"שׁ, וְהוֹדִיעוֹ שֶׁהוּא מַפְלִיג, פֵּרוּשׁ, שֶׁאָמַר לוֹ שֶׁדַּעְתּוֹ לְהַפְלִיג בִּסְתָם, אוֹ שֶׁאָמַר לוֹ הַמָּקוֹם, וְיֵשׁ בְּאוֹתָהּ הַפְלָגָה כְּדֵי שֶׁיִּשְׁתֹּם וְיִסְתֹּם וְיִגֹּב. אֲבָל אִם אֵין בְּאוֹתָהּ הַפְלָגָה כְּדֵי שֶׁיִּשְׁתֹּם כוּ', אֲפִלּוּ שָׁהָה זְמַן מְרֻבֶּה מֻתָּר, דִּבְכָל עִנְיָן מִרְתַּת עוֹבֵד כּוֹכָבִים דִּלְמָא אָתֵי הַשְׁתָּא וְחָזֵי לֵיהּ: {יד} וְאִי אֶפְשָׁר לְנָטְלָהּ שֶׁלֹּא תִשָּׁבֵר כֻּלָּהּ. רַשִׁ"י: {טו} וְרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל סְבִירָא לֵיהּ נְהִי דִלְמַעְלָה לֹא יְדִיעַ, לְפִי שֶׁלְּמַעְלָה מֵרְחוֹ הָעוֹבֵד כּוֹכָבִים וְהִשְׁוָה סִתּוּמוֹ, לְמַטָּה מִיהָא יְדִיעַ, דִּלְמַטָּה לֹא יוּכַל הָעוֹבֵד כּוֹכָבִים לְמָרְחוֹ, וּכְשֶׁיִּפְתַּח יִשְׂרָאֵל אֶת הַמְּגוּפָה יִרְאֶה מִתַּחְתֶּיהָ אוֹתָהּ סְתִימָה. וְרַבָּנָן, כֵּיוָן דִּלְמַעְלָה לֹא יְדִיעַ, לֹא מַסִּיק אַדַּעְתֵּהּ דַּאָפֵיךְ וְחָזֵי לֵיהּ. אִי נַמִּי, זִמְנִין דְּחָלֵים [מִתְמַלֵּא הַנֶּקֶב מִלְּמַטָּה כִלְמַעְלָה. רַשִׁ"י]:

{טז} <B> יוֹצֵא וְכוּ'.</B> כָּל הַיּוֹם כֻּלּוֹ. הָרַמְבַּ"ם: 

~ פרק ה - משנה ה

{יז} <B> וּלְגִינָה כוּ'.</B> וְהֵם פְּתוּחוֹת דּוּמְיָא דְשֶׁעַל הַשֻּׁלְחָן. הָרֹא"שׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ה - משנה ו

{יח} <B> סְתוּמוֹת כוּ'.</B> כֵּיוָן שֶׁהֵם סְתוּמוֹת הֲרֵי זוֹ רְאָיָה שֶׁלֹּא פְתָחוּם, שֶׁאִלּוּ פְתָחוּם לֹא חָזְרוּ לְסָתְמָם, שֶׁאֵין אֵימַת אָדָם עֲלֵיהֶם. וּמִיהוּ דַוְקָא סְתוּמוֹת בִּמְגוּפָה שֶׁל טִיט. אֲבָל בִּפְקָק שֶׁל עֵץ כִּפְתוּחוֹת דָּמֵי, וְכִי מַהֲדַר לִפְקָק, לָאו מִשּׁוּם אֵימָתוֹ אֶלָּא אוֹרְחָא דְמִלְּתָא. הָרַ"ן: {יט} <B> מֻתָּרוֹת.</B> וּמִיהוּ דַוְקָא בִּפְתוּחוֹת מִתְּחִלָּתָן, אֲבָל סְתוּמוֹת שֶׁנִּפְתְּחוּ חָיְשִׁינַן לְהוּ. הָרַ"ן: 

~ פרק ה - משנה ז

{כ} <B> אָסוּר.</B> פֵּרוּשׁ, שֶׁקִּבְּלוּהָ בַשָּׂכָר. אֲבָל בְּהַכְנָסָה לְבַד לֹא נֶאֱסָר, שֶׁאֵין חֲצֵרוֹ שֶׁל אָדָם קוֹנָה לוֹ בְעַל כָּרְחוֹ. בֵּית יוֹסֵף: {כא} וּמִכִּי מָדַד מִסְּתָמָא מוֹשֵׁךְ. וְעַיֵּן בְּסָמוּךְ: {כב} וּבַיְרוּשַׁלְמִי גָרְסֵי עַקֶּבֶת בְּקוֹ"ף, וּמְפָרֵשׁ כְּמָה דְתֵימָא עֲקֻבָּה מִדָּם. פֵּרוּשׁ לִכְלוּךְ יַיִן. הָרַ"ן: {כג} דְּמִתַּסְרָא מִשּׁוּם נִצּוֹק שֶׁחִבֵּר יַיִן שֶׁבַּמַּשְׁפֵּךְ וְיַיִן שֶׁבְּקַרְקָעִית שֶׁל צְלוֹחִית. גְּמָרָא. וְכָתַב הָרַ"ן, וְהַאי אָסוּר, בַּהֲנָאָה קָאָמַר, וּמִשּׁוּם הָכִי דַּוְקָא כִי אִית בֵּיהּ עַכֶּבֶת יַיִן, הָא לָאו הָכִי לֹא עֲדִיף מִנּוֹדוֹת הָעוֹבְדֵי כוֹכָבִים דְּפֶרֶק ב' מִשְׁנָה ד', וּלְפִי זֶה אֲנַן דְּקַיְמָא לָן בְּתַעֲרֹבוֹת כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל [דְּמִשְׁנָה י'] לֹא קַיְמָא לָן כִּי הַךְ מַתְנִיתִין. וּלְהָכִי מְפָרֵשׁ הָרַ"ב לְעֵיל דַּוְקָא בְכֵלָיו שֶׁל יִשְׂרָאֵל, דְּאִי בְכֵלָיו שֶׁל עוֹבֵד כּוֹכָבִים הָיָה אָסוּר אַף בְּפָסַק קֹדֶם שֶׁמָּדַד, לְתַנָּא דְמַתְנִיתִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כד} וְכֵיוָן שֶׁהַכְּלִי בְיָדוֹ וְלֹא עַל גַּבֵּי הַקַּרְקַע, מִסְּתָמָא מְנַדְנְדוֹ וְנֶאֱסָר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כה} <B> יֵין נֶסֶךְ.</B> כָּל הֵיכָא דְתָנֵי יֵין נֶסֶךְ, אַף סְתָם יֵינָם בְּמַשְׁמַע, אֶלָּא אִם כֵּן פֵּרֵשׁ. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כו} שֶׁמִּפְּנֵי שֶׁאֵין דֶּרֶךְ לְעָרוֹת בִּכְלִי כָזֶה, אֵין אָנוּ רוֹאִין כְּנָפוּל וּמְעֹרָב אֶלָּא מַה שֶּׁיָּרַד כְּבָר. מַה שֶּׁאֵין כֵּן בִּשְׁאָר כֵּלִים שֶׁדֶּרֶךְ הוּא לְעָרוֹת מֵהֶן, כֵּיוָן שֶׁסּוֹף הַיַּיִן לֵירֵד עַל יָדוֹ אִי אֶפְשָׁר לוֹמַר בּוֹ רִאשׁוֹן רִאשׁוֹן בָּטֵל, שֶׁכָּל הָעוֹמֵד לִפֹּל וּלְהִתְעָרֵב חָשְׁבִינַן לֵיהּ כִּמְעֹרָב כְּבָר. הָרַ"ן. וּתְנָאִים אֵלּוּ שֶׁכָּתַב הָרַ"ב, הוּא הַדִּין לְשֶׁאֵינוֹ מִינוֹ שֶׁהוּא בְנוֹתֵן טַעַם שֶׁהַדִּין כֵּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כז} לֹא בָא לְמַעֵט כָּל שֶׁנִּתְבָּאֵר עַד עַתָּה, אֶלָּא לְהוֹרוֹת הֲלָכָה בְכָל הָאִסּוּרִין הוּא דַאֲתָא: {כח} דְּזֶה הַכְּלָל לַאֲתוֹיֵי טֶבֶל. גְּמָרָא: {כט} וְהוּא הַדִּין מַיִם בְּמַיִם. תּוֹסָפוֹת: {ל} <B> בְּנוֹתֵן טָעַם.</B> יֵשׁ מְפָרְשִׁים דְּהַיְנוּ בְשִׁשִּׁים כְּמוֹ בִשְׁאָר אִסּוּרִין. וְהָרַאֲבַ"ד כָּתַב כֵּיוָן שֶׁנִּתְעָרֵב בְּמַיִם עַד שֶׁיֵּשׁ בּוֹ יוֹתֵר מִכְּדֵי מְזִיגָה פּוֹגְמוֹ וּמֻתָּר אֲפִלּוּ בִשְׁתִיָּה, הִלְכָּךְ אִם יֵשׁ בַּמַּיִם שִׁשָּׁה חֲלָקִים מִן הַיַּיִן מֻתָּר אֲפִלּוּ בִשְׁתִיָּה. וְכֵן פָּסַק הַטּוּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לא} וּשְׁבִיעִית דְּאוֹסֵר בְּמַשֶּׁהוּ, אֵינוֹ אוֹסֵר בַּאֲכִילָה, אֶלָּא שֶׁחַיָּב לְאָכְלוֹ בִקְדֻשַּׁת שְׁבִיעִית. וְחָמֵץ יָצָא מִן הַכְּלָל, לְפִי שֶׁהוּא לִזְמַן קָצוּב וּבְהֵעָבֵר הַזְּמַן סָר אִסּוּרוֹ. הָרַמְבַּ"ם:

{לב} <B> בְּכָל שֶׁהֵן.</B> הַאי בְּכָל שֶׁהֵן אֵין פֵּרוּשׁוֹ שֶׁאֲפִלּוּ בְמַשֶּׁהוּ מֵאִסּוּרִין הַלָּלוּ יֶאֱסֹר תַּעֲרָבְתּוֹ בַהֲנָאָה, דְּהָא דַוְקָא דָבָר שֶׁבְּמִנְיָן קָא חָשֵׁיב, הִלְכָּךְ וַדַּאי הַאי כָּל שֶׁהֵן הָכִי קָאָמַר, שֶׁכָּל אֶחָד מֵאִסּוּרִים הַלָּלוּ אִם הוּא דָבָר שֶׁבְּמִנְיָן אוֹסֵר תַּעֲרָבְתּוֹ אֲפִלּוּ רִבָּה עָלָיו אֶלֶף כַּיּוֹצֵא בוֹ. הָרַ"ן: {לג} וְהָכָא כְּשֶׁאִסּוּר וְהֶתֵּר כָּל אֶחָד בְּעֵין. וְלֹא דָמֵי לְתַעֲרֹבֶת דִּלְעֵיל וּמַתְנִיתִין דִּלְקַמָּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לד} וְהוּא הַדִּין לִשְׁאָר הָאוֹסְרִין בְּכָל שֶׁהֵן, שֶׁיִּמָּכְרוּ חוּץ מִדְּמֵי אִסּוּר. הָרַמְבַּ"ם. וְהָרַאֲבַ"ד: {לה} דְּהַשְׁתָּא הָוֵי דָבָר הַנִּמְכָּר בְּמִנְיָן, כַּמָּה גִזּוֹת בְּסֶלַע, וְעוֹשִׂין מֵהֶן שַׂקִּין. רַשִׁ"י: {לו} וְאִם כֵּן מִשְׁנָה ה' פֶּרֶק ז' דְּחֻלִּין דְּאוֹסֶרֶת גִּיד הַנָּשֶׁה מִשּׁוּם דִּבְרִיָּה הִיא, וַחֲתִיכַת נְבֵלָה דִרְאוּיָה לְהִתְכַּבֵּד הִיא, פְּלִיגָא אַמַּתְנִיתִין, וְכֵן מַתְנִיתִין דְּרֵישׁ פֶּרֶק ח' דִּזְבָחִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{לז} <B> שֶׁנָּפַל.</B> כְּשֶׁנָּפַל בְּחָבִית גְּדוֹלָה וְנִשְׁתַּבְּרָה וְנָפַל כֻּלּוֹ כְאֶחָד לְגוֹ הֶתֵּרָא, וְלֹא אָמְרִינַן קַמָּא קַמָּא בָטִיל לֵיהּ בְּרֹב. רַשִׁ"י: {לח} וְקָא מַשְׁמַע לָן דַּאֲפִלּוּ נִתְעָרֵב הַיַּיִן מַמָּשׁ. אֲבָל כִּי נִתְעָרֵב חָבִית בְּחָבִיּוֹת אֲפִלּוּ יֵין נֶסֶךְ גָּמוּר נַמִּי שָׁרֵי בְּהוֹלָכַת הֲנָאָה לְיָם הַמֶּלַח. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{לט} <B> יִקְלֹף.</B> וְאַחַר כָּךְ יְנַגְּבוֹ. הָרַ"ן וְהָרֹא"שׁ: {מ} <B> אֲסוּרָה.</B> וְאֵינָהּ נִתֶּרֶת בְּנִגּוּב אֶלָּא בְעֵרוּי. דְּנִגּוּב אֵינוֹ מַפְלִיט בִּכְלִי חֶרֶס. וְכֵיוָן דְּצָרִיךְ עֵרוּי, אֵין צָרִיךְ קְלִיפָה, דְּעֵרוּי מַפְלִיט אַף בְּלֹא קְלִיפָה כָּל זְמַן שֶׁלֹּא דָרַךְ בָּהּ, דְּלֹא גָרַע מִקַּנְקַנִּים שֶׁמַּכְנִיסִין לְקִיּוּם. הָרֹא"שׁ: {מא} כְּלוֹמַר, שֶׁאַף עַל פִּי שֶׁהַזֶּפֶת נִקְלַף, לֹא הוּסַר בָּזֶה טַעַם הַיַּיִן, לְפִי שֶׁהַחֶרֶס עַצְמוֹ מִלְּבַד הַזֶּפֶת בָּלַע כְּשֶׁנָּתַן הַיַּיִן בַּזֶּפֶת לְזַפְּתָהּ כוּ'. וּלְשׁוֹן הָרַ"ן, מִשּׁוּם דְּחֶרֶס בָּלַע טוּבָא:

{מב} וַאֲפִלּוּ חֲדָשִׁים, דְּהָא יְשָׁנִים וְלִבְּנָן כַּחֲדָשִׁים דָּמוּ, וַאֲפִלּוּ הָכִי בָעוּ טְבִילָה: {מג} דְּבָתַר סוֹפוֹ אָזְלִינַן, שֶׁחִפָּהוּ מַתָּכוֹת. גְּמָרָא: {מד} הוֹאִיל וְכִי נִשְׁבְּרוּ יֵשׁ לָהֶן תַּקָּנָה [עַל יְדֵי הַתָּכָה], כִּכְלֵי מַתָּכוֹת דָּמֵי. גְּמָרָא: {מה} יָלְפִינַן לֵיהּ בַּגְּמָרָא מִקְּרָא. וְהַתֵּימַהּ עַל הָרַ"ב שֶׁלֹּא הִצְרִיךְ לְהָסִיר הַחֲלוּדָה כְּמוֹ שֶׁהִצְרִיךְ בְּהַגְעָלָה. וּכְמוֹ שֶׁכָּתְבוּ הַפּוֹסְקִים. וּבְלִבּוּן שֶׁלֹּא הִצְרִיךְ כֵּן, הוּא כְדַעַת הַטּוּר שֶׁהָאֵשׁ יַעֲבִיר הַכֹּל: {מו} שֶׁזֶּה שֶׁנֶּאֱמַר כָּל אֲשֶׁר יָבֹא בָאֵשׁ תַּעֲבִירוּ בָאֵשׁ, פֵּרוּשׁ כְּדֶרֶךְ תַּשְׁמִישׁוֹ כֵּן הַגְעָלָתוֹ. רַשִׁ"י: {מז} וְיֵשׁ לְדַקְדֵּק מָה רָאָה בְמַתְנִיתִין שֶׁשָּׁנָה בְּשַׁפּוּד וְאַסְכְּלָה הֶכְשֵׁרוֹ לְרוֹתֵחַ, וּבְסַכִּין הֶכְשֵׁרוֹ שֶׁל צוֹנֵן. וְיֵשׁ לוֹמַר שֶׁלֹּא הָיָה צָרִיךְ כְּלָל לִשְׁנוֹתוֹ, לְפִי שֶׁהוּא כְבָר בִּכְלַל מַה שֶּׁשָּׁנָה לְהַגְעִיל יַגְעִיל, לְלַבֵּן כוּ'. אֲבָל שֶׁהֻצְרַךְ לִשְׁנוֹתוֹ בְשַׁפּוּד וְאַסְכְּלָה, שֶׁמִּפְּנֵי שֶׁשָּׁנִינוּ בָהֶן לְעִנְיַן קָדָשִׁים דִּבְהַגְעָלָה סַגִּי לְהוּ כִּדְאִיתָא בִזְבָחִים [פּי"א מי"א] מִשּׁוּם דְּכִי בָלַע דְּהֶתֵּרָא בָלַע, קָא מַשְׁמַע לָן דְּהָכָא מְלַבְּנָן כוּ'. וּבְסַכִּין הוֹדִיעָנוּ דְּאַגַּב דּוֹחֲקָא דְסַכִּינָא לֹא סַגִּי לֵיהּ בַּהֲדָחָה אֲפִלּוּ לְצוֹנֵן וְצָרִיךְ שִׁיפָה כוּ'. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מח} דְּנוֹתֵן טַעַם לִפְגָם מֻתָּר. וְהַתּוֹרָה לֹא אָסְרָה אֶלָּא קְדֵרָה בַת יוֹמָא. גְּמָרָא:
