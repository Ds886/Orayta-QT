~ פרק א - משנה א
<b>הַשֻּׁתָּפִין שֶׁרָצוּ לַעֲשׂוֹת מְחִצָּה.</b> בְּחָצֵר שֶׁאֵין בָּהּ דִּין חֲלֻקָּה מַיְרֵי, שֶׁאֵין בָּהּ כְּדֵי אַרְבַּע אַמּוֹת לְכָל אֶחָד מִן הַשֻּׁתָּפִין. לְהָכִי קָתָנֵי שֶׁרָצוּ {ב}, דְּמַשְׁמַע דַּוְקָא שְׁנֵיהֶם רוֹצִים, אֲבָל אֵין אֶחָד מֵהֶם יָכוֹל לָכוֹף אֶת חֲבֵרוֹ. דְּאִלּוּ בְּחָצֵר שֶׁיֵּשׁ בָּהּ דִּין חֲלֻקָּה, כָּל אֶחָד יָכוֹל לָכוֹף אֶת חֲבֵרוֹ לַחֲלֹק. וְהַשְׁתָּא אַשְׁמוֹעִינַן דְּכֵיוָן שֶׁרָצוּ לַחֲלֹק וְהֶחֱזִיק {ג} כָּל אֶחָד {ד} מֵהֶם בָּרוּחַ הַמַּגִּיעַ לוֹ כְּפִי מַה שֶּׁנִּתְפַּשְּׁרוּ, בּוֹנִין אֶת הַכֹּתֶל בָּאֶמְצַע. שֶׁזֶּה נוֹתֵן מֵחֶלְקוֹ חֲצִי מָקוֹם עֳבִי הַכֹּתֶל {ה} וְכֵן זֶה: <B>גָּוִיל.</B> אֲבָנִים שֶׁאֵינָם מְשֻׁפּוֹת וּמְתֻקָּנוֹת: <B>גָּזִית.</B> אֲבָנִים מְשֻׁפּוֹת וּמְגֹרָרוֹת בִּמְגֵרָה: <B>כְּפִיסִין.</B> אָרִיחַ. שֶׁהוּא חֲצִי לְבֵנָה, וְהוּא טֶפַח וּמֶחֱצָה. וְהַלְּבֵנָה שְׁלֹשָׁה טְפָחִים: <B>הַכֹּל כְּמִנְהַג הַמְּדִינָה.</B> הַכֹּל לַאֲתוֹיֵי אַתְרָא דִּנְהִיגֵי לַעֲשׂוֹת מְחִצָּה בְּלוּלְבֵי גְּפָנִים וְעַנְפֵי עֵצִים, שֶׁעוֹשִׂים כְּפִי הַמִּנְהָג {ז}. וּבִלְבַד שֶׁתְּהֵא הַמְּחִצָּה עָבָה שֶׁלֹּא יוּכַל אֶחָד מֵהֶם לְהַבִּיט בְּשֶׁל חֲבֵרוֹ, דְּהֶזֵּק רְאִיָּה שְׁמֵיהּ הֶזֵּק. וְגֹבַהּ הַכֹּתֶל וְהַמְּחִצָּה {ח} אֵין פָּחוּת מֵאַרְבַּע אַמּוֹת: <B>זֶה נוֹתֵן שְׁלֹשָׁה טְפָחִים.</B> שֶׁכֹּתֶל הַגָּוִיל צָרִיךְ טֶפַח יוֹתֵר מִכֹּתֶל הַגָּזִית, בִּשְׁבִיל רָאשֵׁי הָאֲבָנִים שֶׁבּוֹלְטוֹת לַחוּץ שֶׁאֵינָן חֲלָקוֹת: <B>וּבִכְפִיסִים זֶה נוֹתֵן טְפָחַיִם.</B> שֶׁעֳבִי כֹּתֶל הַכְּפִיסִים טֶפַח יוֹתֵר מֵעֳבִי כֹּתֶל הַלְּבֵנִים, לְפִי שֶׁמֵּשִׂים אָרִיחַ מִכָּאן וְאָרִיחַ מִכָּאן, שֶׁהֵם שְׁלֹשָׁה טְפָחִים, וְטֶפַח בָּאֶמְצַע שֶׁנּוֹתֵן שָׁם טִיט לְחַבְּרָן. וְכֹתֶל הֶעָשׂוּי מִלְּבֵנִים נוֹתֵן לְבֵנָה שְׁלֵמָה לָעֳבִי הַכֹּתֶל, שֶׁהִיא שְׁלֹשָׁה טְפָחִים, וְאֵין צָרִיךְ טִיט בָּאֶמְצַע: <B>הַמָּקוֹם וְהָאֲבָנִים שֶׁל שְׁנֵיהֶם.</B> הָא קָמַשְׁמַע לָן דְּאַף עַל גַּב דְּנָפְלוּ הָאֲבָנִים לִרְשׁוּתָא דְּחַד מִנַּיְהוּ, אִי נַמִּי דְּקָדֵים חַד וּפָנִינְהוּ לִרְשׁוּתֵיהּ, מַהוּ דְּתֵימָא לֶהֱוֵי אִידָךְ הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה, קָא מַשְׁמַע לָן {י}: 

~ פרק א - משנה ב
<B>וְכֵן בְּגִנָּה.</B> הָכִי קָאָמַר, וְכֵן בְּגִנָּה סְתָם כִּמְקוֹם שֶׁנָּהֲגוּ לִגְדֹּר הוּא {יא}, וּמְחַיְּבִים אוֹתוֹ שֶׁלָּקַח מָקוֹם שָׁם {יב} לִגְדֹּר: <B>אֲבָל בְּבִקְעָה.</B> סְתָם, הֲרֵי הוּא כִּמְקוֹם שֶׁנָּהֲגוּ שֶׁלֹּא לִגְדֹּר, וְאֵין מְחַיְּבִים אוֹתוֹ. אֶלָּא אִם רָצָה חֲבֵרוֹ לִגְדֹּר, כּוֹנֵס בְּתוֹךְ שֶׁלּוֹ: <B>וְעוֹשֶׂה לוֹ חֲזִית.</B> סִימָן {טו}, לְהֶכֵּר שֶׁהַכֹּתֶל שֶׁלּוֹ. וְהַסִּימָן מְפָרֵשׁ בַּגְּמָרָא, שֶׁטָּח רֹאשׁ הַכֹּתֶל אַמָּה בְּסִיד לְצַד חֲבֵרוֹ. וְלֹא לַצַּד שֶׁלּוֹ, שֶׁמָּא יָטוּחַ גַּם חֲבֵרוֹ מִן הַצַּד שֶׁלּוֹ, וְיֹאמַר שֶׁהַכֹּתֶל שֶׁל שְׁנֵיהֶם הוּא. אֲבָל כְּשֶׁטָּח לְצַד חֲבֵרוֹ וְאֵינוֹ טָח לְתוֹךְ שֶׁלּוֹ, הָוֵי סִימָן שֶׁהַכֹּתֶל שֶׁלּוֹ. וְאִם יְקַלְּפֶנּוּ חֲבֵרוֹ, קִלּוּפָהּ מֵידָע יְדִיעַ: <B>וְעוֹשִׂין חֲזִית מִכָּאן וּמִכָּאן.</B> לְהוֹדִיעַ שֶׁשְּׁנֵיהֶם עָשׂוּ אוֹתוֹ: 

~ פרק א - משנה ג
<B>אֵין מְחַיְּבִין אוֹתוֹ.</B> דְּהָא לֹא אַהֲנֵי לֵיהּ מִידֵי, שֶׁעֲדַיִן קַרְקַע שֶׁלּוֹ פְּתוּחָה. אֲבָל אִם גָּדַר אֶת הָרְבִיעִית, דְּהַשְׁתָּא וַדַּאי אַהֲנֵי לֵיהּ, מְחַיְּבִין אוֹתוֹ לְשַׁלֵּם {יט} מַחֲצִית דְּמֵי קָנִים בְּזוֹל. אֲבָל לֹא חֲצִי יְצִיאוֹתָיו, דְּאָמַר לֵיהּ לְדִידִי סַגִּי לִי בִּמְחִצָּה שֶׁל קָנִים וְאִי אֶפְשִׁי בִּיצִיאָה שֶׁל גֶּדֶר אֲבָנִים: <B>מְגַלְגְּלִין עָלָיו אֶת הַכֹּל.</B> מַחֲצִית יְצִיאוֹתָיו כְּפִי מַה שֶּׁגָּדַר. וְהַיְנוּ דְּאִיכָּא בֵּין תַּנָּא קַמָּא לְרַבִּי יוֹסֵי. וַהֲלָכָה כְּרַבִּי יוֹסֵי. וְאֵין צָרִיךְ לוֹמַר אִם הַנִּקָּף הוּא שֶּׁגָּדַר רוּחַ רְבִיעִית, דְּגַלֵּי דַּעְתֵּיהּ דְּנִיחָא לֵיהּ בְּמַה שֶּׁגָּדַר חֲבֵרוֹ, שֶׁמְּגַלְגְּלִין עָלָיו אֶת הַכֹּל וּמְשַׁלֵּם לַחֲבֵרוֹ חֲצִי יְצִיאוֹתָיו: 

~ פרק א - משנה ד
<B>כֹּתֶל חָצֵר.</B> שֶׁל שֻׁתָּפִין: <B>עַד אַרְבַּע אַמּוֹת.</B> דִּבְהָכִי סַגִּי לֵיהּ לְהֶזֵּק רְאִיָּה: <B>בְּחֶזְקַת שֶׁנָּתַן.</B> כְּשֶׁזֶּה תּוֹבְעוֹ חֲצִי הַיְצִיאָה וְזֶה אוֹמֵר כְּבָר נָתַתִּי חֶלְקִי, נֶאֱמָן, עַד שֶׁיָּבִיא הַתּוֹבֵעַ עֵדִים שֶׁתְּבָעוֹ וְלֹא נָתַן {כא}. דְּמִשְׁפָּט זֶה גָּלוּי לַכֹּל שֶׁחוֹבָה עָלָיו לְסַיְּעוֹ, וְלֹא הָיָה זֶה בּוֹנֵהוּ מִשֶּׁלּוֹ אֶלָּא הָיָה מְבִיאוֹ לְבֵית דִּין: <B>סָמַךְ לוֹ כֹתֶל אַחֵר.</B> לְאַחַר שֶׁבָּנָה הָאֶחָד לְמַעְלָה מֵאַרְבַּע אַמּוֹת הַרְבֵּה וְלֹא רָצָה לְסַיְּעוֹ בְּהַגְבָּהָתוֹ, סָמַךְ הַשֵּׁנִי כֹּתֶל אַחֵר כְּנֶגֶד כֹּתֶל זֶה כְּדֵי לְסַכֵּךְ וְלָתֵת תִּקְרָה מִכֹּתֶל לְכֹתֶל, אַף עַל פִּי שֶׁלֹּא נָתַן עָלָיו עֲדַיִן הַתִּקְרָה: <B>מְגַלְגְּלִין עָלָיו אֶת הַכֹּל.</B> דְּגַלֵּי דַּעְתֵּיהּ דְּנִיחָא לֵיהּ בַּהַגְבָּהָה דְּהַאִיךְ {כג}: <B>בְּחֶזְקַת שֶׁלֹּא נָתַן.</B> אִם בָּא הָרִאשׁוֹן וּתְבָעוֹ לַדִּין לְאַחַר שֶׁסָּמַךְ אֶת כָּתְלוֹ לְכֹתֶל רִאשׁוֹן, וְאוֹמֵר לוֹ תֵּן חֶלְקְךָ בְּמַה שֶּׁהִגְבַּהְתִּי אוֹתוֹ, וְזֶה אוֹמֵר נָתַתִּי חֶלְקִי, אֵינוֹ נֶאֱמָן אֶלָּא בְּעֵדִים. שֶׁאֵין מִשְׁפָּט זֶה גָּלוּי, וְעַד שֶׁחִיְּבוּהוּ בֵּית דִּין אֵינוֹ עָשׂוּי לִתֵּן: 

~ פרק א - משנה ה
<B>כּוֹפִין אוֹתוֹ.</B> אֶת בֶּן הֶחָצֵר שֶׁאֵינוֹ רוֹצֶה לְסַיֵּעַ לִבְנֵי הֶחָצֵר: <B>לִבְנוֹת.</B> לֶחָצֵר בֵּית שַׁעַר לִהְיוֹת שׁוֹמֵר הַפֶּתַח יוֹשֵׁב שָׁם בַּצֵּל וּמַרְחִיק אֶת בְּנֵי רְשׁוּת הָרַבִּים מִלְּהָצִיץ בֶּחָצֵר: <B>וְדֶלֶת.</B> לְשַׁעַר הֶחָצֵר: <B>לֹא כָל הַחֲצֵרוֹת רְאוּיוֹת לְבֵית שָׁעַר.</B> חָצֵר שֶׁאֵינָהּ סְמוּכָה לִרְשׁוּת הָרַבִּים אֵינָהּ רְאוּיָה לְבֵית שַׁעַר. וְאֵין הֲלָכָה כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל. דַּאֲפִלּוּ שֶׁאֵינָהּ סְמוּכָה לִרְשׁוּת הָרַבִּים זִמְנִין דְּדַחְקֵי רַבִּים וְעָיְלוּ וְאָתוּ: <B>לֹא כָל הָעֲיָרוֹת רְאוּיוֹת לְחוֹמָה.</B> עִיר שֶׁאֵינָהּ סְמוּכָה לִגְבוּל הָאוֹיְבִים אֵינָהּ צְרִיכָה לְחוֹמָה. וְאֵין הֲלָכָה כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל. דַּאֲפִלּוּ אֵינָהּ סְמוּכָה לִגְבוּל הָאוֹיְבִים צְרִיכָה לְחוֹמָה, דְּזִמְנִין דְּמִקְרוּ וְאָתוּ גְּיָסוֹת: <B>וִיהֵא כְאַנְשֵׁי הָעִיר.</B> לָשֵׂאת עִמָּהֶם בָּעֹל {כה}: <B>שְׁנֵים עָשָׂר חֹדֶשׁ.</B> וְהָאִידְנָא דְּנַיְדֵי, נָהוּג עָלְמָא שְׁלֹשִׁים יוֹם:

<B>אֵין חוֹלְקִין אֶת הֶחָצֵר.</B> אֵין אֶחָד מִן הַשֻּׁתָּפִין יָכוֹל לָכוֹף אֶת חֲבֵרוֹ לַחְלֹק הֶחָצֵר עַד שֶׁיְּהֵא בּוֹ אַרְבַּע אַמּוֹת לְכָל אֶחָד מִן הַשֻּׁתָּפִין חוּץ מִן הַפְּתָחִים, שֶׁכָּל בַּיִת שֶׁבֶּחָצֵר צָרִיךְ שֶׁיִּהְיֶה לִפְנֵי פִּתְחוֹ אַרְבַּע אַמּוֹת לְפָרֵק מַשָּׂאוֹ מֵעַל חֲמוֹרוֹ, וְחוּץ מֵאוֹתָן אַרְבַּע אַמּוֹת צָרִיךְ שֶׁיִּהְיֶה בֶּחָצֵר אַרְבַּע אַמּוֹת אֲחֵרוֹת לְכָל אֶחָד מִן הַשֻּׁתָּפִין לִשְׁאָר תַּשְׁמִישִׁים, וְאָז יִהְיֶה בֶּחָצֵר דִּין חֲלֻקָּה {כו}: <B>רַבִּי יְהוּדָה אוֹמֵר תִּשְׁעָה חֲצָאֵי קַבִּין.</B> מָר כִּי אַתְרֵיהּ וּמָר כִּי אַתְרֵיהּ. וְלֹא פְּלִיגֵי. בְּאַתְרֵיהּ דְּרַבִּי יְהוּדָה הָיְתָה עוֹשָׂה תְּבוּאָה שָׂדֶה בַּת תִּשְׁעָה חֲצָאֵי קַבִּין כְּשָׂדֶה בַּת תִּשְׁעָה קַבִּין בְּאַתְרַיְהוּ דְּרַבָּנָן: <B>בֵּית רֹבַע.</B> מָקוֹם לִזְרֹעַ בּוֹ רֹבַע הַקַּב. וּפְסַק הַהֲלָכָה, אֵין דִּין חֲלֻקָּה בְּקַרְקָעוֹת הָעוֹמְדוֹת לַחֲרִישָׁה וְלִזְרִיעָה אֶלָּא אִם כֵּן יִשָּׁאֵר לְכָל אֶחָד מִן הַשֻּׁתָּפִים קַרְקַע כְּדֵי עֲבוֹדַת יוֹם אֶחָד {כח} בַּחֲרִישָׁה [אוֹ בִּזְרִיעָה] וְכַיּוֹצֵא בָהֶן {כט}: <B>מוֹרָן וּטְרַקְלִין.</B> מִינֵי פְּלָטִין הֵן: <B>כְּדֵי לָזֶה וּכְדֵי לָזֶה.</B> כְּדִמְפָרֵשׁ וְאָזִיל שֶׁלְּאַחַר שֶׁיַּחֲלֹקוּ וְיִטֹּל כָּל אֶחָד חֶלְקוֹ יִהְיֶה בְּחֵלֶק כָּל אֶחָד מֵהֶם מָקוֹם רָאוּי שֶׁיִּקָּרֵא טְרַקְלִין אוֹ מוֹרָן, שֶׁיִּשָּׁאֵר בִּקְצָתוֹ הַשֵּׁם שֶׁהָיָה בְּכֻלּוֹ: <B>בִּזְמַן שֶׁאֵין שְׁנֵיהֶם רוֹצִין.</B> אֵין יָכוֹל הָאֶחָד לָכוֹף אֶת חֲבֵרוֹ לַחְלֹק עַל כָּרְחוֹ. אֲבָל יָכוֹל לְכוֹפוֹ לְעִנְיַן גּוֹד אוֹ אֱגוֹד, כְּלוֹמַר, קְנֵה חֶלְקִי אוֹ מְכֹר לִי חֶלְקְךָ {ל} בְּמַה שֶּׁהוּא שָׁוֶה: <B>וְכִתְבֵי הַקֹּדֶשׁ.</B> תּוֹרָה נְבִיאִים וּכְתוּבִים: <B>לֹא יַחֲלֹקוּ.</B> וְדַוְקָא כְּשֶׁהֵם בְּכֶרֶךְ אֶחָד. אֲבָל בִּכְרָכִים הַרְבֵּה כָּל סֵפֶר בִּפְנֵי עַצְמוֹ, אִם רָצוּ חוֹלְקִים {לג}:

וְטָעַן זֶה שֶׁנִּכְנְסוּ בְּבֵיתוֹ לְקוּחִים הֵם בְּיָדִי וַהֲרֵי אֲנִי מֻחְזָק בָּהֶם, אֵין זוֹ חֲזָקָה, שֶׁכֵּן דַּרְכָּם לָלֶכֶת מִבַּיִת לְבַיִת. אֲבָל אִם הֶחֱזִיק הָעֶבֶד שָׁלֹשׁ שָׁנִים הַוְיָא חֲזָקָה וְאֵין צָרִיךְ לִשְׁטַר מְכִירָה: <B>שְׂדֵה הַבַּעַל.</B> הַמִּסְתַּפֶּקֶת מִמֵּי גְּשָׁמִים, אֵינָהּ עוֹשָׂה פֵּרוֹת אֶלָּא פַּעַם אַחַת בְּשָׁנָה: <B>חֶזְקָתָהּ שָׁלֹשׁ שָׁנִים.</B> וְאֵינָהּ צְרִיכָה מִיּוֹם לְיוֹם: <B>רַבִּי יִשְׁמָעֵאל אוֹמֵר שְׁלֹשָׁה חֳדָשִׁים וְכוּ'.</B> שֶׁיֵּשׁ תְּבוּאָה גְּדֵלָה לִשְׁלֹשָׁה חֳדָשִׁים, כְּגוֹן שְׂעֹרִים וְשִׁבֹּלֶת שׁוּעָל וַעֲדָשִׁים, וְנִמְצָא אוֹכֵל שָׁלֹשׁ תְּבוּאוֹת בִּשְׁמֹנָה עָשָׂר חֹדֶשׁ: <B>רַבִּי עֲקִיבָא אוֹמֵר.</B> יֵשׁ דָּבָר שֶׁגָּדֵל לִשְׁלֹשִׁים יוֹם, כְּגוֹן שַׁחַת וִירָקוֹת. לְפִיכָךְ אִם אָכְלָה אַרְבָּעָה עָשָׂר חֹדֶשׁ הַוְיָא חֲזָקָה. וְרַבִּי יִשְׁמָעֵאל סָבַר, דְּבַאֲכִילַת שַׁחַת וִירָקוֹת לֹא הַוְיָא חֲזָקָה, אֶלָּא בַּאֲכִילַת תְּבוּאָה וּפֵרוֹת גְּמוּרִים: <B>בַּמֶּה דְבָרִים אֲמוּרִים.</B> שֶׁצָּרִיךְ שְׁמֹנָה עָשָׂר חֹדֶשׁ לִשְׂדֵה הַבַּעַל: <B>בִּשְׂדֵה הַלָּבָן.</B> שֶׁכָּל פֵּרוֹתֶיהָ נִלְקָטִין בְּפֶרֶק אֶחָד, לְפִיכָךְ צָרִיךְ שָׁלֹשׁ שָׁנִים: <B>אֲבָל בִּשְׂדֵה הָאִילָן.</B> שֶׁפֵּרוֹתֶיהָ נִלְקָטִים לִפְרָקִים, עֲנָבִים בְּפֶרֶק אֶחָד, וְזֵיתִים בְּפֶרֶק אֶחָד, וּתְאֵנִים בְּפֶרֶק אֶחָד: <B>כָּנַס אֶת תְּבוּאָתוֹ.</B> יַיִן שֶׁל גְּפָנִים {ז}: <B>מָסַק אֶת זֵיתָיו וְכָנַס אֶת קֵיצוֹ.</B> לִקֵּט תְּאֵנִים וְיִבְּשָׁן וְהִכְנִיסָן לְבֵיתוֹ, הֲרֵי זוֹ חֲזָקָה כְּאִלּוּ הֵן שָׁלֹשׁ שָׁנִים. וְאֵין הֲלָכָה לֹא כְּרַבִּי עֲקִיבָא וְלֹא כְּרַבִּי יִשְׁמָעֵאל {ח}: 

~ פרק ג - משנה ב
<B>שָׁלֹשׁ אֲרָצוֹת לַחֲזָקָה.</B> שָׁלֹשׁ אֲרָצוֹת בְּאֶרֶץ יִשְׂרָאֵל חֲלוּקוֹת זוֹ מִזּוֹ לְעִנְיַן חֲזָקָה, שֶׁאִם הֶחֱזִיק בְּאֶחָד מֵאֵלּוּ הָאֲרָצוֹת וְהָיָה בַּעַל הַקַּרְקַע בְּאַחֶרֶת, אֵין חֶזְקָתוֹ חֲזָקָה, לְפִי שֶׁאֵין שְׁיָרוֹת מְצוּיוֹת מִזּוֹ לָזוֹ. וְאַף עַל פִּי שֶׁאֵינָן שְׁעַת חֵרוּם וּמִלְחָמָה, כִּשְׁעַת חֵרוּם דָּמִי {ט}, וְאִם הָיָה בַּעַל הַקַּרְקַע מוֹחֶה לֹא הָיָה מִי שֶׁיּוֹדִיעַ לְזֶה הַמַּחֲזִיק, לְפִיכָךְ הָיָה רָאוּי לְזֶה הַמַּחֲזִיק לִזָּהֵר בִּשְׁטָרוֹ, וְכֵיוָן שֶׁלֹּא נִזְהַר הִפְסִיד: <B>עַד שֶׁיְּהֵא עִמּוֹ בַמְּדִינָה.</B> שֶׁיִּהְיוּ שְׁנֵיהֶם בִּיהוּדָה אוֹ שְׁנֵיהֶם בְּעֵבֶר הַיַּרְדֵּן, וְאַף עַל פִּי שֶׁזֶּה בְּעִיר אַחַת וְזֶה בְּעִיר אַחֶרֶת, דְּכֵיוָן דִּשְׁיָרוֹת מְצוּיוֹת הָיָה לוֹ לִמְחוֹת, וְכֵיוָן דְּלֹא מִחָה הִפְסִיד: <B>אָמַר רַבִּי יְהוּדָה וְכוּ'.</B> קָסָבַר רַבִּי יְהוּדָה טַעְמָא דַּחֲזָקָה לַאו מִשּׁוּם דְּעַד תְּלַת שְׁנִין מִזְדְּהַר אֱינָשׁ בִּשְׁטָרֵיהּ, טְפֵי לֹא מִזְדְּהַר, אֶלָּא טַעְמָא דַּחֲזָקָה לְדִידֵיהּ, מִשּׁוּם דְּלֹא שָׁבִיק אֱינִישׁ דְּאָכְלֵי אַרְעָא דִּידֵיהּ אֲפִלּוּ שָׁעָה אַחַת וְשָׁתֵיק, וְהַאי דְּהִרְחִיקוּ תְּלַת שְׁנִין כְּדֵי שֶׁאִם הָיָה בְּאַסְפַּמְיָא וְכוּ', אֲבָל אִם הָיָה עִמּוֹ בָּעִיר, לְאַלְתַּר הָוֵי חֲזָקָה. וּבְשָׁלֹשׁ אֲרָצוֹת הַלָּלוּ שֶׁאֵין שְׁיָרוֹת מְצוּיוֹת, בִּתְלַת שְׁנִין מִיהָא הָוֵי חֲזָקָה. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: 

~ פרק ג - משנה ג
<B>כָּל חֲזָקָה שֶׁאֵין עִמָּהּ טַעֲנָה.</B> שֶׁיִּטְעֹן לָמָּה מַחֲזִיק בְּשֶׁל חֲבֵרוֹ, אֵינָהּ חֲזָקָה: <B>וְהַבָּא מֵחֲמַת יְרֻשָּׁה.</B> שֶׁהֶחֱזִיק בָּהּ שָׁלֹשׁ שָׁנִים מִכֹּחַ יְרֻשַּׁת אָבִיו, שֶׁהָיְתָה שֶׁל אָבִיו בְּיוֹם מוֹתוֹ: <B>אֵין צָרִיךְ טַעֲנָה.</B> לְבָרֵר אֵיךְ בָּאָה לְיַד אָבִיו {יד}. וּמִיהוּ רְאָיָה בָּעֵי שֶׁרָאוּ אָבִיו דָּר בּוֹ יוֹם אֶחָד: <B>הָאֻמָּנִים.</B> שֶׁמְּתַקְּנִים כֵּלִים. אֵין לָהֶם חֲזָקָה. אִם הֵם מֻחְזָקִים בְּכֵלִים שֶׁל אֲחֵרִים {טו} אֵין יְכוֹלִים לִטְעֹן לְקוּחִים הֵם בְּיָדִי, וַאֲפִלּוּ הֵם כֵּלִים שֶׁאֵין עֲשׂוּיִין לְהַשְׁאִיל וּלְהַשְׂכִּיר. וְהָנֵי מִלֵּי כְּשֶׁהַכְּלִי מָצוּי בְּפָנֵינוּ בְּיַד הָאֻמָּן {טז}. אֲבָל אִם לֹא נִמְצָא הַכְּלִי בְּפָנֵינוּ בְּיַד הָאֻמָּן אֶלָּא שֶׁבָּא אֶחָד מִן הַשּׁוּק וְטָעַן לָאֻמָּן כְּלִי יֵשׁ לִי אֶצְלְךָ שֶׁנָּתַתִּי לְךָ לְתַקֵּן, הַחֲזִירֵהוּ לִי, וְטָעַן הָאֻמָּן אֱמֶת שֶׁהוּא אֶצְלִי אֲבָל אַתָּה מָכַרְתָּ לִי, נֶאֱמָן הָאֻמָּן בִּשְׁבוּעָה, בְּמִגּוֹ דְּאִי בָּעֵי אָמַר לֹא הָיָה דְּבָרִים מֵעוֹלָם אוֹ הֶחֱזַרְתִּי לְךָ {יז}. וְכֵן טָעַן הָאֻמָּן כָּךְ פָּסַקְתָּ לִי בִּשְׂכַר תִּקּוּנוֹ, וְהַלָּה אוֹמֵר לֹא פָּסַקְתִּי אֶלָּא פָּחוֹת. אִם הַכְּלִי מָצוּי בְּפָנֵינוּ בְּיַד הָאֻמָּן בַּעַל הַכְּלִי נֶאֱמָן. וְאִם אֵין הַכְּלִי מָצוּי בְּפָנֵינוּ בְּיַד הָאֻמָּן הֲרֵי הָאֻמָּן נֶאֱמָן בִּשְׁבוּעָה וַאֲפִלּוּ מָסַר לוֹ בְּעֵדִים, בְּמִגּוֹ דְּאִי בָּעֵי אָמַר הֶחֱזַרְתִּיו לְךָ. וְהַשֻּׁתָּפִים. שֶׁיֵּשׁ לָהֶן קַרְקַע בְּשֻׁתָּפוּת וְאָכַל אֶחָד מֵהֶן שָׁלֹשׁ שָׁנִים כָּל הַפֵּרוֹת, אֵינָהּ חֲזָקָה {יט}. וְהָנֵי מִלֵּי, דְּאֵין בַּקַּרְקַע דִּין חֲלֻקָּה. אֲבָל אִם יֵשׁ בַּקַּרְקַע דִּין חֲלֻקָּה וַאֲכָלָהּ אֶחָד מֵהֶן שָׁלֹשׁ שָׁנִים, הָוֵי חֲזָקָה {כ}: <B>וְהָאֲרִיסִים.</B> שֶׁיּוֹרְדִים לַקַּרְקַע לְמֶחֱצָה לִשְׁלִישׁ וְלִרְבִיעַ, וְאָכַל כָּל הַפֵּרוֹת שָׁלֹשׁ שָׁנִים, אֵינָהּ חֲזָקָה. וְדַוְקָא בַּאֲרִיסֵי בָּתֵּי אָבוֹת {כא}, שֶׁהוּא כְּמוֹ אַפּוֹטְרוֹפּוֹס עַל הַבֵּן, אֲבָל הָאָרִיס שֶׁהוֹרִידוֹ בַּעַל הַקַּרְקַע עַצְמוֹ וְאָכַל כָּל הַפֵּרוֹת שָׁלֹשׁ שָׁנִים, יֵשׁ לוֹ חֲזָקָה: <B>וְאֵין לָאִישׁ חֲזָקָה בְּנִכְסֵי אִשְׁתּוֹ.</B> וַאֲפִלּוּ כָּתַב לָהּ בְּעוֹדָהּ אֲרוּסָה דִּין וּדְבָרִים אֵין לִי בִּנְכָסַיִךְ וְלֹא בְּפֵרוֹתֵיהֶן, דְּהַשְׁתָּא אֵינוֹ אוֹכֵל פֵּרוֹת מִן הַדִּין, וְאַחַר כָּךְ הֵבִיא רְאָיָה שֶׁאָכַל פֵּרוֹת שָׁלֹשׁ שָׁנִים, לֹא הַוְיָא חֲזָקָה, דְּאוֹרְחָא דְּאִתְּתָא לְהָנִיחַ לְבַעְלָהּ שֶׁיֹּאכַל פֵּרוֹת נְכָסֶיהָ בֵּין כַּדִּין בֵּין שֶׁלֹּא כַּדִּין: <B>וְאֵין לָאִשָּׁה חֲזָקָה בְּנִכְסֵי בַעְלָהּ.</B> וַאֲפִלּוּ יִחֵד לָהּ קַרְקַע לִמְזוֹנוֹתֶיהָ וְאָכְלָה פֵּרוֹת מִקַּרְקַע אַחֶרֶת שֶׁל בַּעְלָהּ שָׁלֹשׁ שָׁנִים, אֲפִלּוּ הָכִי לֹא הַוְיָא חֲזָקָה, שֶׁדֶּרֶךְ הָאִישׁ לְהָנִיחַ לְאִשְׁתּוֹ שֶׁתֹּאכַל בִּנְכָסָיו אֲפִלּוּ בְּדָבָר שֶׁאֵין לָהּ בּוֹ זְכוּת: <B>וְלֹא לָאָב בְּנִכְסֵי הַבֵּן וְלֹא לַבֵּן וְכוּ'.</B> לְפִי שֶׁהֵם כְּאַפּוֹטְרוֹפְּסִים זֶה עַל זֶה: <B>בַּמֶּה דְבָרִים אֲמוּרִים.</B> דְּאֵינָהּ חֲזָקָה {כה}: <B>בְּמַחֲזִיק.</B> בַּחֲזָקָה שֶׁיֵּשׁ עָלֶיהָ עִרְעוּר, שֶׁחֲבֵרוֹ טוֹעֵן גְּזוּלָה הִיא בְּיָדְךָ: <B>אֲבָל הַנּוֹתֵן מַתָּנָה.</B> בְּפָנֵינוּ {כז}, וְאָמַר לִמְקַבֵּל הַמַּתָּנָה לֵךְ חֲזֵק וּקְנֵה, כָּל הָנָךְ דְּאָמְרִינַן לְעֵיל בְּמַתְנִיתִין דְּאֵין לָהֶם חֲזָקָה, מִשֶּׁהֶחֱזִיקוּ כִּשְׁאָר מְקַבְּלֵי מַתָּנָה, קָנוּ, וְאֵין הַנּוֹתֵן יָכוֹל לַחֲזֹר בּוֹ. וְהָאִשָּׁה שֶׁנָּתְנָה אוֹ שֶׁמָּכְרָה לְבַעְלָהּ נִכְסֵי מְלוֹג שֶׁלָּהּ, מִשֶּׁהֶחֱזִיק בָּהֶן הַבַּעַל קָנָה, וְאֵינָהּ יְכוֹלָה לוֹמַר נַחַת רוּחַ עָשִׂיתִי לְבַעֲלִי. דְּדַוְקָא בְּנִכְסֵי צֹאן בַּרְזֶל אוֹ בְּקַרְקַע שֶׁיִּחֵד לָהּ בַּעְלָהּ בִּכְתֻבָּתָהּ אָמְרִינַן שֶׁאֵין מִכְרָהּ מֶכֶר וְאֵין מַתְּנָתָהּ מַתָּנָה מִפְּנֵי שֶׁיְּכוֹלָה לוֹמַר נַחַת רוּחַ עָשִׂיתִי לְבַעֲלִי, מִפְּנֵי שֶׁיֵּשׁ לְבַעְלָהּ שַׁיָּכוּת בָּהֶן. אֲבָל בְּנִכְסֵי מְלוּג שֶׁאֵין לְבַעְלָהּ בְּעִקָּרָן שׁוּם שַׁיָּכוּת, אֵינָהּ יְכוֹלָה לוֹמַר נַחַת רוּחַ עָשִׂיתִי לְבַעֲלִי. וְכֵן הָאִישׁ שֶׁמָּכַר לְאִשְׁתּוֹ מִנְּכָסָיו, אִם הַמָּעוֹת שֶׁקָּנְתָה בָּהֶן הָאִשָּׁה אוֹתָן נְכָסִים לֹא הָיוּ טְמוּנִים אֶצְלָהּ, הַמֶּכֶר קַיָּם, וְיִהְיוּ אוֹתָן הַנְּכָסִים לָאִשָּׁה וְהַבַּעַל אוֹכֵל פֵּרוֹת. וְאִם הָיוּ אוֹתָן מָעוֹת טְמוּנִים אֶצְלָהּ, הַמֶּכֶר בָּטֵל, דְּאָמַר, לִגְּלוּיֵי זוּזֵי שֶׁהָיוּ טְמוּנִים אֶצְלָהּ אָמַרְתִּי שֶׁאֲנִי מוֹכֵר לָהּ: <B>וְהָאַחִין שֶׁחָלְקוּ.</B> וְהֶחֱזִיק כָּל אֶחָד מֵהֶן {כח} בְּחֶלְקוֹ, אֵין יְכוֹלִים לַחֲזֹר: <B>וְהַמַּחֲזִיק בְּנִכְסֵי הַגֵּר.</B> שֶׁמֵּת וְאֵין לוֹ יוֹרְשִׁים, וְכָל הַקּוֹדֵם לְהַחֲזִיק בִּנְכָסָיו זָכָה: <B>נָעַל.</B> שֶׁעָשָׂה דֶּלֶת: <B>וְגָדַר.</B> אוֹ שֶׁעָשָׂה חוֹמָה: <B>וּפָרַץ.</B> אוֹ שֶׁפָּרַץ בָּהּ פִּרְצָה {כט}: 

~ פרק ג - משנה ד
<B>מְשַׁלְּשִׁין בֵּינֵיהֶם.</B> כָּל כַּת יִתֵּן הַשְּׁלִישׁ. שֶׁהֲרֵי הֵן שָׁלֹשׁ כִּתּוֹת לְשָׁלֹשׁ שָׁנִים: <B>שְׁלֹשָׁה אַחִים.</B> לְכָל שָׁנָה אָח אֶחָד וְאַחֵר עִם הָאָח, שֶׁאוֹתוֹ אַחֵר מֵעִיד עִם כֻּלָּן: <B>הֲרֵי כָּאן שָׁלֹשׁ עֵדֻיּוֹת.</B> דְּמַאי דְּמַסְהֵיד הַאי לֹא מַסְהֵיד הַאי, וּלְכָךְ עֵדוּתָן כְּשֵׁרָה: <B>וְהֵן עֵדוּת אַחַת.</B> לְעִנְיַן הֲזָמָה, שֶׁאִם הוּזְמוּ מְשַׁלְּשִׁין בֵּינֵיהֶן, וְאֵין נַעֲשִׂים זוֹמְמִין עַד שֶׁיּוּזְמוּ כֻּלָּן {לג}: 

~ פרק ג - משנה ה
<B>שֶׁיֵּשׁ לָהֶם חֲזָקָה.</B> אִם הֶחֱזִיק בְּקַרְקַע חֲבֵרוֹ לְעִנְיָן זֶה שָׁלֹשׁ שָׁנִים הַוְיָא חֲזָקָה: <B>הָיָה מַעֲמִיד בְּהֶמְתּוֹ בֶּחָצֵר.</B> הַאי תַּנָּא בַּחֲצַר הַשֻּׁתָּפִים קָא מַיְרֵי, דְּלֹא קָפְדֵי אַהֲדָדֵי {לה} אַהַעֲמָדַת בְּהֵמָה וְכַיּוֹצֵא בָּהּ, וּמִשּׁוּם הָכִי לֹא הַוְיָא חֲזָקָה אֲפִלּוּ כִּי הֶחֱזִיק לְהַאי מִלְּתָא שָׁלֹשׁ שָׁנִים: <B>אֲבָל אִם עָשָׂה מְחִצָּה וְכוּ'.</B> דִּבְכִי הַאי גַּוְנָא וַדַּאי קָפְדֵי, אִי שָׁתִיק לָהּ שָׁלֹשׁ שָׁנִים וְלֹא מִחָה, וַדַּאי הַוְיָא חֲזָקָה. וְרַבֵּינוּ מֹשֶׁה בַּר מַיְמוֹנִי פֵּרֵשׁ, דִּבְשֻׁתָּף דְּקָפֵיד אַעֲשִׂיַּת מְחִצָּה, אִם עָשָׂה שֻׁתָּפוֹ מְחִצָּה וְלֹא מִחָה לְאַלְתַּר הַוְיָא חֲזָקָה. אֲבָל אֱינָשׁ דְּעָלְמָא אַף עַל פִּי שֶׁעָשָׂה מְחִצָּה וְלֹא מִחָה לֹא הַוְיָא חֲזָקָה {לז} עַד לְאַחַר שָׁלֹשׁ שָׁנִים {לח}: 

~ פרק ג - משנה ו
<B>הַמַּרְזֵב.</B> צִנּוֹר קָטָן שֶׁנּוֹתְנִים אוֹתוֹ בְּסוֹף צִנּוֹר הַגָּדוֹל הַהוֹלֵךְ עַל פְּנֵי כָּל הַגַּג: <B>מַרְזֵב.</B> מָקוֹם שֶׁהַטִּפָּה שֶׁל גְּשָׁמִים זָבָה מִמֶּנּוּ. מָר, טִפָּה. כְּמוֹ הֵן גּוֹיִם כְּמַר מִדְּלִי (יְשַׁעְיָה מ): <B>אֵין לוֹ חֲזָקָה וְיֵשׁ לִמְקוֹמוֹ חֲזָקָה.</B> אִם הָיָה שָׁם מַרְזֵב וּבָא בַּעַל הֶחָצֵר לְעָקְרוֹ לְגַמְרֵי שֶׁלֹּא יִשָּׁפְכוּ מֵי הַגַּג לְתוֹךְ חֲצֵרוֹ, אֵינוֹ יָכוֹל, שֶׁכְּבָר יֶשׁ לוֹ חֲזָקָה לָזֶה שֶׁיִּהְיוּ מֵימֵי גַּגּוֹ שׁוֹפְכִים דֶּרֶךְ אוֹתוֹ מַרְזֵב. אֲבָל אִם בָּא לְהָפְכוֹ, שֶׁהָיָה מֻנָּח לְרוּחַ דְּרוֹמִי וּבָא לָתֵת אוֹתוֹ לְרוּחַ צְפוֹנִי, אֵין בַּעַל הַמַּרְזֵב יָכוֹל לְעַכֵּב עָלָיו, דְּלַאו מִידֵי אַפְסְדֵיהּ, וְאֵין לוֹ חֲזָקָה שֶׁיִּהְיֶה תָּמִיד עוֹמֵד בְּרוּחַ זוֹ: <B>מַזְחִילָה.</B> צִנּוֹר גָּדוֹל הַמַּחֲזִיק כָּל אֹרֶךְ הַגַּג. וְהָוֵי מִלְּתָא דִּקְבִיעוּתָא, לְכָךְ יֶשׁ לוֹ חֲזָקָה: <B>סֻלָּם הַמִּצְרִי.</B> קָטָן {מב} וְאֵינוֹ קָבוּעַ, וְלֹא קָפֵיד עֲלֵיהּ אִם נוֹתֵן אוֹתוֹ לְתוֹךְ חֲצַר חֲבֵרוֹ כְּדֵי לַעֲלוֹת בּוֹ לְגַגּוֹ וּלְשׁוֹבָכוֹ, לְכָךְ אֵין לוֹ חֲזָקָה: <B>חַלּוֹן הַמִּצְרִית.</B> חַלּוֹן קָטָן שֶׁאֵין רֹאשׁוֹ שֶׁל אָדָם נִכְנָס בּוֹ: <B>אֵין לוֹ חֲזָקָה.</B> וְאִי בָּעֵי בַּעַל הֶחָצֵר לִבְנוֹת כְּנֶגֶד הַחַלּוֹן וּלְסָתְמוֹ, לֹא מָצֵי בַּעַל הַחַלּוֹן לוֹמַר כְּבָר הֶחֱזַקְתִּי בּוֹ וְאֵינְךָ יָכוֹל לִסְתֹּם, דְּאָמַר לֵיהּ הִיאַךְ, לֹא נִתְפַּיַּסְתִּי אֶלָּא לְפִי שֶׁלֹּא הָיָה לִי בּוֹ הֶזֵּק. וַאֲפִלּוּ הָכִי אִי הֲוָה בָּעֵי מֵעִקָּרָא כְּשֶׁפְּתָחוֹ הָוֵי מָצֵי מְעַכֵּב עָלָיו שֶׁלֹּא יִפְתְּחֶנּוּ {מג}, וַאֲפִלּוּ הָיָה גָּבוֹהַּ יוֹתֵר מֵאַרְבַּע אַמּוֹת {מד}, דְּמָצֵי אָמַר לֵיהּ שֶׁמָּא תָּשִׂים סַפְסָל סָמוּךְ לַחַלּוֹן וְתַבִּיט בִּי, וְאַחַר שֶׁהִנִּיחוֹ נַמִּי וּפָתַח, אֵין לוֹ חֲזָקָה: <B>וְלַצּוֹרִית יֶשׁ לוֹ חֲזָקָה.</B> חַלּוֹן שֶׁרֹאשׁוֹ שֶׁל אָדָם נִכְנָס בּוֹ, וַאֲפִלּוּ הוּא לְמַעְלָה מֵאַרְבַּע אַמּוֹת. אוֹ חַלּוֹן הֶעָשׂוּי לְאוֹרָה {מה}, וַאֲפִלּוּ הוּא קָטָן מְאֹד. אוֹ חַלּוֹן שֶׁהוּא לְמַטָּה מֵאַרְבַּע אַמּוֹת. כָּל אֶחָד מִשְּׁלֹשָׁה חַלּוֹנוֹת הַלָּלוּ אִם הִנִּיחוֹ וּפָתַח, יֶשׁ לוֹ חֲזָקָה וְשׁוּב אֵין יָכוֹל לִבְנוֹת כְּנֶגְדּוֹ וּלְסָתְמוֹ, דְּיֵשׁ חֲזָקָה לִנְזָקִין. חוּץ מֵעָשָׁן וְאָבָק וּבֵית הַכִּסֵּא כְּלוֹמַר רֵיחַ רַע, שֶׁאֵלּוּ הַשְּׁלֹשָׁה אֵין לָהֶם חֲזָקָה אֲפִלּוּ הֶחֱזִיק בָּהֶם כַּמָּה שָׁנִים. אֲבָל הֶזֵּק רְאִיָּה יֶשׁ לָהּ חֲזָקָה {מו}. וְלֹא בָּעֵי חֲזָקָה לִנְזָקִין שָׁלֹשׁ שָׁנִים, אֶלָּא מִכִּי קָאֵי נִזְקָא כְּדֵי שֶׁיּוּכַל לוֹמַר לַנִּזָּק כְּבָר יָדַעְתָּ בַּנֵּזֶק וְלֹא מָחִיתָ, הַוְיָא חֲזָקָה: <B>מַלְבֵּן.</B> בִּנְיָן מִבַּחוּץ כְּעֵין מַשְׁקוֹף מִלְּמַעְלָה <B>מָכַר אֶת הַבָּתִּים.</B> הַפְּתוּחִים לֶחָצֵר: <B>וּבוֹרוֹת שִׁיחִים וּמְעָרוֹת.</B> שֶׁבְּתוֹךְ הַבָּתִּים. וְאַף עַל גַּב דְּהַמּוֹכֵר בַּיִת לֹא מָכַר בּוֹר שִׁיחַ וּמְעָרָה, לְגַבֵּי חָצֵר מִיהָא בְּטֵלִים הֵן: <B>בֵּין כָּךְ וּבֵין כָּךְ.</B> אֲפִלּוּ אָמַר כָּל מַה שֶּׁבְּתוֹכָהּ, לֹא מָכַר אֶת הַמֶּרְחָץ וְכוּ'. דְּאֵינָן בִּכְלָל חָצֵר: <B>רַבִּי אֱלִיעֶזֶר אוֹמֵר וְכוּ'.</B> וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר:

<B>מָכַר אֶת הַיָּם.</B> הָאֶבֶן שֶׁנּוֹתְנִים בְּתוֹכָהּ הַזֵּיתִים בִּשְׁעַת טְחִינָתָן: <B>הַמַּמָּל.</B> הָרֶכֶב הָעֶלְיוֹנָה שֶׁבָּהּ מְפָרְכִים הַזֵּיתִים: <B>הַבְּתוּלוֹת.</B> כְּלוּנְסוֹת שֶׁל אֶרֶז שֶׁמַּעֲמִידִין בָּהֶם קוֹרוֹת בֵּית הַבַּד {יד}: <B>עֲכִירִין.</B> נְסָרִים כְּבֵדִים שֶׁנּוֹתְנִים עַל הָאַמְתָּחוֹת שֶׁשָּׂמִים בָּהֶם הַזֵּיתִים הַכְּתוּשִׁים כְּדֵי לְכָבְשָׁן: <B>הַגַּלְגַּל.</B> שֶׁמְּגַלְגֵּל הָאֶבֶן וּמַכְבִּיד עַל הַזֵּיתִים הַכְּתוּשִׁים לְהוֹצִיא שַׁמְנָן: 

~ פרק ד - משנה ו
<B>הַנְּסָרִים.</B> לוּחוֹת שֶׁמְּשִׂימִין עֲלֵיהֶם הַבְּגָדִים: <B>הַסַּפְסָלִים.</B> שֶׁיּוֹשְׁבִים עֲלֵיהֶם. וְיֵשׁ גּוֹרְסִים סְפָלִים, שֶׁנּוֹתְנִים בָּהֶם מַיִם לִרְחֹץ: <B>וִילָאוֹת.</B> מָסָךְ כְּנֶגֶד הַפֶּתַח. פֵּרוּשׁ אַחֵר, אֲלוּנְטִית שֶׁמִּסְתַּפְּגִין בָּהּ. וְיֵשׁ אוֹמְרִים סִנָּר שֶׁמְּכַסִּין בּוֹ הָעֶרְוָה כְּשֶׁיּוֹשְׁבִין בַּמֶּרְחָץ: <B>מְגֻרוֹת שֶׁל מַיִם.</B> בְּרֵכוֹת הַשּׁוֹפְכוֹת מַיִם לַמֶּרְחָץ: <B>אוֹצָרוֹת שֶׁל עֵצִים.</B> שֶׁמְּחַמְּמִין בּוֹ הַמֶּרְחָץ: 

~ פרק ד - משנה ז
<B>מָכַר בָּתִּים.</B> וְכָל שֶׁכֵּן חֲצֵרוֹת שֶׁהֵן עִקַּר הָעִיר: <B>בֵּית הַשְּׁלָחִין.</B> גִּנּוֹת וּפַרְדֵּסִין הַשַּׁיָּכִין לָעִיר {יח}: <B>אֲבָל לֹא הַמִּטַּלְטְלִין.</B> מָאנֵי תַּשְׁמִישְׁתָּא, כְּגוֹן מַפְתֵּחַ וָקֶלֶת וְכַיּוֹצֵא בָּהֶן. וְכָל שֶׁכֵּן שֶׁלֹּא מָכַר חִטֵּי וּשְׂעָרֵי: <B>הֲרֵי כֻלָּן מְכוּרִים.</B> אֲפִלּוּ עֲבָדִים וּבְהֵמוֹת דְּמִטַּלְטְלֵי דְּנַיְדֵי נִינְהוּ. וְכָל שֶׁכֵּן חִטֵּי וּשְׂעָרֵי דַּהֲווּ מִטַּלְטְלֵי דְּלֹא נַיְדֵי: <B>הַסַּנְטֵר.</B> הָעֶבֶד הַמְמֻנֶּה לִשְׁמֹר הָעִיר {כ}. וְאֵין הֲלָכָה כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל: 

~ פרק ד - משנה ח
<B>הָאֲבָנִים שֶׁהֵם לְצָרְכָּהּ.</B> לַעֲשׂוֹת גָּדֵר {כא}: <B>וְאֶת הַקָּנִים.</B> שֶׁקּוֹשְׁרִים בָּהֶם הַגְּפָנִים: <B>וְאֶת מְחִצַּת הַקָּנִים.</B> קָנִים הַרְבֵּה גְּדֵלִים בְּקֶלַח אֶחָד, כְּשֶׁהִיא פְּחוּתָה מִבֵּית רֹבַע אָז הִיא בְּטֵלָה לְגַבֵּי הַשָּׂדֶה: <B>שׁוֹמֵרָה.</B> סֻכַּת שׁוֹמְרִים הַמְחֻבֶּרֶת לַקַּרְקַע בְּטִיט {כג}: <B>וְאֶת הֶחָרוּב.</B> בְּבַחֲרוּתוֹ שֶׁעֲדַיִן לֹא הֻרְכָּב. דִּכְשֶׁמַּזְקִין מִתְגַּבֵּר וּמַרְכִּיבִין אוֹתוֹ וְיֵשׁ לוֹ שֵׁם בִּפְנֵי עַצְמוֹ וְלֹא בָּטֵיל לְגַבֵּי שָׂדֶה: <B>וְאֶת בְּתוּלַת הַשִּׁקְמָה.</B> בְּבַחֲרוּתָהּ קֹדֶם שֶׁחָתְכוּ עֲנָפֶיהָ. דִּכְשֶׁמַּזְקִין קוֹצְצִים הָעֲנָפִים וּגְדֵלִין הַרְבֵּה בִּמְקוֹמָן וְקָרוּי סַדַּן שִׁקְמָה: 

~ פרק ד - משנה ט
<B>לֹא מָכַר אֶת מְחִצַּת הַקָּנִים וְכוּ'.</B> שֶׁכָּל אֵלֶּה חֲשׁוּבִים כְּשָׂדֶה בִּפְנֵי עַצְמָן {כה}: <B>לֹא מָכַר אֶת הַבּוֹר.</B> אַף עַל פִּי שֶׁמָּכַר לוֹ הַשָּׂדֶה אֵין אֵלּוּ מְכוּרִין: <B>וְצָרִיךְ הַמּוֹכֵר לִקַּח לוֹ דֶרֶךְ מִן הַלּוֹקֵחַ.</B> לָבֹא לַבּוֹר וְלַגַּת. דְּמוֹכֵר בְּעַיִן יָפָה מוֹכֵר כִּדְפָרִישְׁנָא לְעֵיל: <B>נָתַן אֶת כֻּלָּן.</B> וַאֲפִלּוּ לְמַאן דְּאָמַר מוֹכֵר בְּעַיִן רָעָה מוֹכֵר וְצָרִיךְ הַלּוֹקֵחַ לִקַּח לוֹ דֶּרֶךְ הֵיכָא דְּמָכַר לוֹ אֶחָד מִכָּל אֵלּוּ וְשִׁיֵּר הַשָּׂדֶה לְעַצְמוֹ, דַּוְקָא בְּמוֹכֵר אָמְרִינַן הָכִי, מִשּׁוּם דַּהֲוָה לֵיהּ לַלּוֹקֵחַ לְאַתְנוּיֵי וּלְפֵרוּשֵׁי, וְכֵיוָן דְּלֹא אַתְנֵי אִיהוּ הוּא דְּאַפְסִיד, אֲבָל גַּבֵּי נוֹתֵן שֶׁמְּקַבֵּל הַמַּתָּנָה מִתְבַּיֵּשׁ לוֹמַר לַנּוֹתֵן פָּרֵשׁ לִי מַה שֶּׁאַתָּה נוֹתֵן לִי, לֹא אָמְרִינַן מִדְּלֹא פֵּרֵשׁ אִיהוּ הוּא דְּאַפְסִיד, אֶלָּא אָמְרִינַן בְּעַיִן יָפָה נָתַן וְאֵין צָרִיךְ מְקַבֵּל הַמַּתָּנָה לִקַּח לוֹ דֶּרֶךְ. וְכֵן נַמִּי כָּל הָנֵי דְּאָמְרִינַן לְעֵיל לֹא מָכַר לְפִי שֶׁאֵינָן בִּכְלָל שָׂדֶה, וַאֲפִלּוּ הָנֵי דְּאֵינָן בִּכְלָל כָּל מַה שֶּׁבְּתוֹכָהּ כְּגוֹן חָרוּב וְסַדַּן הַשִּׁקְמָה וּבוֹר וָגַת, אִם נָתַן הַשָּׂדֶה מַתָּנָה, נָתַן אֶת כֻּלָּן, מִשּׁוּם דִּבְעַיִן יָפָה יוֹתֵר מִדַּאי הוּא נוֹתֵן: <B>זָכוּ בַשָּׂדֶה זָכוּ בְכֻלָּן.</B> אֲפִלּוּ בּוֹר וָגַת וְשׁוֹבָךְ שֶׁבַּשָּׂדֶה. מִפְּנֵי שֶׁכָּל אֶחָד זוֹכֶה בְּשֶׁלּוֹ וּמִסְתַּלֵּק מֵחֲבֵרוֹ מִכֹּל וָכֹל: <B>הֶחֱזִיק בְּכֻלָּם.</B> וְאַף עַל גַּב דְּחָרוּב וְשִׁקְמָה {לא} לֹא בְּטִילֵי לְגַבֵּי שָׂדֶה, הֲווּ כִּשְׁתֵּי שָׂדוֹת וְאֵין מֵצַר בֵּינֵיהֶן, דְּאִם הֶחֱזִיק בְּאַחַת מֵהֶן קָנָה חֲבֶרְתָּהּ: <B>הִקְדִּישׁ אֶת כֻּלָּם.</B> דְּמַקְדִּישׁ בְּעַיִן יָפָה מַקְדִּישׁ: <B>לֹא הִקְדִּישׁ אֶלָּא אֶת הֶחָרוּב.</B> מִכָּל הָנָךְ דְּאָמְרִינַן בְּמַתְנִיתִין גַּבֵּי מֶכֶר לֹא מָכַר, גַּבֵּי הֶקְדֵּשׁ נַמִּי לֹא הִקְדִּישׁ, דְּדַעַת מַקְדִּישׁ כְּדַעַת מוֹכֵר, לְבַד מֵחָרוּב הַמֻּרְכָּב וְסַדַּן הַשִּׁקְמָה, דְּאַף עַל גַּב דִּבְמֶכֶר לֹא מָכַר, בְּהֶקְדֵּשׁ קַדְּשֵׁי. וְטַעְמָא, הוֹאִיל וּמִשְּׂדֵה הֶקְדֵּשׁ קָא יָנְקֵי {לב}:

~ פרק ה - משנה א
<B>הַמּוֹכֵר אֶת הַסְּפִינָה.</B> סְתָם: <B>מָכַר אֶת הַתֹּרֶן.</B> עֵץ גָּבוֹהַּ שֶׁתּוֹלִין עָלָיו אֶת הַנֵּס. וְהַנֵּס הוּא כְּמִין וִילוֹן שֶׁתּוֹלִין בַּתֹּרֶן לְנֶגֶד הָרוּחַ: <B>עוֹגִין.</B> בַּרְזֶל שֶׁקּוֹשְׁרִין אוֹתוֹ בַּחֶבֶל וְזוֹרְקִין אוֹתוֹ בְּעֹמֶק הַמַּיִם לְעַכֵּב וּלְעַגֵּן הַסְּפִינָה: <B>מַנְהִיגִים.</B> אֵלּוּ הַמְּשׁוֹטוֹת שֶׁבָּהֶם מוֹלִיכִין הַסְּפִינָה לַמָּקוֹם שֶׁרוֹצִים: <B>מַרְצוּפִין.</B> אַמְתָּחוֹת שֶׁנּוֹתְנִים בָּהֶם הַסְּחוֹרָה: <B>אַנְתִּיקִי.</B> הַפְּרַקְמַטְיָא שֶׁבְּתוֹךְ הַסְּפִינָה. וְכָל הָנֵי דְּחָשֵׁיב הָכָא דִּבְמֶכֶר לֹא מָכַר, אִם נָתַן וְהִקְדִּישׁ אֶת הַסְּפִינָה לֹא הֲווּ הָנָךְ בִּכְלָל הַמַּתָּנָה וְהַהֶקְדֵּשׁ. וְלֹא דָּמֵי לְבוֹר וְדוּת וְגַת דְּפִרְקִין דִּלְעֵיל דְּכֻלְּהוּ מְקַרְקְעֵי נִינְהוּ וּבָטְלֵי לְגַבֵּי שָׂדֶה {ב} לְעִנְיַן הֶקְדֵּשׁ וּמַתָּנָה, אֲבָל הָנָךְ לֹא בָּטְלֵי כְּלָל: <B>פְּרָדוֹת.</B> בְּהֵמוֹת הַמּוֹשְׁכוֹת אֶת הַקָּרוֹן. וְהוּא שֶׁאֵין אֲדוּקִים בּוֹ {ד} בִּשְׁעַת הַמְּכִירָה. וְאִית דִּמְפָרְשֵׁי פְּרָדוֹת, עֵצִים הַנִּפְרָדִים מִן הָעֲגָלָה שֶׁמּוֹשְׁכִים אֶת הָעֲגָלָה בָּהֶם. וְהוּא שֶׁלֹּא יִהְיוּ קְשׁוּרִים בָּהּ בִּשְׁעַת הַמְּכִירָה: <B>צֶמֶד.</B> הוּא הָעֹל שֶׁצּוֹמֵד וּמְחַבֵּר אֶת הַשְּׁוָרִים יַחַד: <B>אֵין הַדָּמִים רְאָיָה.</B> וְהָא דְּאָמַר דִּשְׁתוּת קָנָה וּמַחֲזִיר אוֹנָאָה יוֹתֵר מִשְּׁתוּת בָּטֵל מִקָּח, הָנֵי מִלֵּי בִּכְדֵי שֶׁהַדַּעַת טוֹעָה, דְּסָבוּר הַלּוֹקֵחַ שֶׁשָּׁוֶה כָּךְ, וְהָוֵי מִקַּח טָעוּת, אֲבָל בִּכְדֵי שֶׁאֵין הַדַּעַת טוֹעָה, כְּגוֹן דְּקָנָה צֶמֶד שֶׁשָּׁוֶה זוּז בְּמָאתַיִם {ח}, אָמְרִינַן בְּמַתָּנָה נְתָנָם לוֹ וְאֵין כָּאן בִּטּוּל מִקָּח. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: 

~ פרק ה - משנה ב
<B>לֹא מָכַר אֶת כֵּלָיו.</B> בְּכֵלִים הָעֲשׂוּיִים לִרְכִיבָה כְּגוֹן אֻכָּף וּמִרְדַּעַת, כֻּלֵּי עָלְמָא לֹא פְּלִיגֵי דְּקָנָה אֲפִלּוּ אֵינָם עָלָיו בִּשְׁעַת מְכִירָה {ט}. כִּי פְּלִיגֵי בְּכֵלִים שֶׁל מַשְּׂאוֹי, כְּגוֹן שַׂק וְדִיסְקְיָא, תַּנָּא קַמָּא סָבַר לֹא מָכַר הַכֵּלִים שֶׁל מַשְּׂאוֹי שֶׁעָלָיו, וְנַחוּם הַמָּדִי סָבַר מָכַר {י} הַכֵּלִים שֶׁל מַשְּׂאוֹי שֶׁעָלָיו. וַהֲלָכָה כַּתַּנָּא קַמָּא, וְלֹא קָנָה הַכֵּלִים שֶׁל מַשְּׂאוֹי וַאֲפִלּוּ הָיוּ עָלָיו בִּשְׁעַת מְכִירָה: <B>חֲמוֹרְךָ זֶה.</B> מַשְׁמַע כְּמוֹת שֶׁהוּא עִם כֵּלָיו: <B>חֲמוֹרְךָ הוּא.</B> הָוֵי כְּשׁוֹאֵל לוֹ חֲמוֹרְךָ הוּא, מְכֹר אוֹתוֹ לִי. וְהָוֵי כְּמוֹכֵר חֲמוֹר סְתָם, וְאֵין כֵּלָיו מְכוּרִים {יב} וַאֲפִלּוּ הֵן עָלָיו בִּשְׁעַת מְכִירָה. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: 

~ פרק ה - משנה ג
<B>הַמּוֹכֵר אֶת הַחֲמוֹר מָכַר אֶת הַסְּיָח.</B> בְּנוֹ שֶׁל חֲמוֹר {יג}. וּבַגְּמָרָא מוֹקֵי לָהּ כְּגוֹן שֶׁאָמַר לוֹ חֲמוֹר מֵנִיקָה אֲנִי מוֹכֵר לְךָ. דַּחֲלֵב הָאָתוֹן לְמַאי מַהֲנֵי, אֶלָּא וַדַּאי הִיא וּבְנָהּ קָאָמַר לֵיהּ. וְסֵיפָא דְּאָמַר פָּרָה מֵנִיקָה אֲנִי מוֹכֵר לְךָ לֹא קָנָה אֶת בְּנָהּ, דְּפָרָה לַחֲלָבָהּ עוֹמֶדֶת, וּפָרָה וַחֲלָבָהּ קָאָמַר לֵיהּ: <B>מָכַר אַשְׁפָּה.</B> מָקוֹם גָּבוֹהַּ שְׁלֹשָׁה טְפָחִים אוֹ יוֹתֵר, אוֹ עָמֹק שְׁלֹשָׁה טְפָחִים אוֹ יוֹתֵר, שֶׁרָגִיל לְהָנִיחַ שָׁם זֶבֶל בְּהֵמוֹתָיו: <B>מָכַר בּוֹר מָכַר מֵימָיו.</B> מַתְנִיתִין יְחִידָאָה הִיא, וּפְלִיגֵי רַבָּנָן עֲלָהּ וְאָמְרֵי מָכַר בּוֹר לֹא מָכַר מֵימָיו. וַהֲלָכָה כַּחֲכָמִים: <B>הַלּוֹקֵחַ פֵּרוֹת שׁוֹבָךְ.</B> מַה שֶּׁיָּלְדוּ הַיּוֹנִים כָּל הַשָּׁנָה {טו}: <B>מַפְרִיחַ.</B> הַלּוֹקֵחַ: <B>בְּרֵכָה רִאשׁוֹנָה.</B> וְיַנִּיחֶנָּה לַמּוֹכֵר. וְכָל שְׁנֵי וְלָדוֹת קְרוּיִין בְּרֵכָה {טז}. וּמַנִּיחָן עִם אִמָּם כְּדֵי לִהְיוֹת לְצַוְתָּא לְאִמָּם שֶׁלֹּא יִבְרְחוּ: <B>פֵּרוֹת כַּוֶּרֶת.</B> דְּבוֹרִים הַנּוֹלָדִים מִן הַכַּוֶּרֶת זֹאת הַשָּׁנָה. נוֹטֵל הַלּוֹקֵחַ שְׁלֹשָׁה נְחִילִים, שְׁלֹשָׁה חֲבוּרוֹת מִוְּלָדוֹת הַדְּבוֹרִים הָרִאשׁוֹנִים הַיּוֹצְאִים מִן הַכַּוֶּרֶת, וּמַכְנִיסָן לְתוֹךְ הַכַּוֶּרֶת שֶׁלּוֹ. שֶׁהָרִאשׁוֹנִים הֵם חֲשׁוּבִים, וְאַחֲרוֹן אַחֲרוֹן גָּרוּעַ. וּמִשָּׁם וָאֵילַךְ מְסָרֵס, כְּלוֹמַר נוֹטֵל אַחַת וּמַנִּיחַ אַחַת {יז} לַמּוֹכֵר, כְּדֵי שֶׁיִּגְדְּלוּ וְיִתְחַבְּרוּ עִם הָאָבוֹת כְּדֵי שֶׁתִּתְיַשֵּׁב הַכַּוֶּרֶת {יח}. וְכֵן לְעוֹלָם {יט}: <B>חַלּוֹת דְּבַשׁ.</B> הַדְּבַשׁ עִם הַשַּׁעֲוָה יַחַד מְעֹרָבִים עָשׂוּי חַלּוֹת חַלּוֹת: <B>מַנִּיחַ שְׁתֵּי חַלּוֹת.</B> שֶׁמִּתְפַּרְנְסִים הַדְּבוֹרִים מֵהֶם כָּל יְמוֹת הַגְּשָׁמִים {כב}: <B>גְּרוֹפִיּוֹת.</B> עֲנָפִים {כג}. וְחוֹזֵר וְצוֹמֵחַ: 

~ פרק ה - משנה ד
<B>הַקּוֹנֶה שְׁנֵי אִילָנוֹת.</B> סְתָמָא, לֹא קָנָה קַרְקַע כְּלָל {כד} סְבִיבוֹתָיו: <B>הִגְדִּילוּ.</B> הִרְחִיבוּ הָעֲנָפִים: <B>לֹא יְשַׁפֶּה.</B> לֹא יִכְרֹת {כה} אוֹתָם בַּעַל הַקַּרְקַע אַף עַל פִּי שֶׁהַצֵּל שֶׁלָּהֶם מַזִּיק לַקַּרְקַע שֶׁלּוֹ, דְּכֵיוָן דְּאֵין לוֹ קַרְקַע לְבַעַל הָאִילָנוֹת הֲרֵי בַּעַל הַקַּרְקַע שִׁעְבֵּד לוֹ קַרְקָעוֹ לְכָל צָרְכָּן כָּל זְמַן שֶׁהֵם קַיָּמִים: <B>גֶּזַע.</B> כָּל שֶׁהוּא לְמַעְלָה מִן הָאָרֶץ שֶׁרוֹאֶה פְּנֵי חַמָּה: <B>שָׁרָשִׁין.</B> כָּל שֶׁהֵן לְמַטָּה מִן הָאָרֶץ: <B>שֶׁלּוֹ.</B> שֶׁל בַּעַל הָאִילָן. וְלֹא שֶׁיַּנִּיחֶנּוּ שָׁם וְיִגְדַּל, דְּחָיְשִׁינַן שֶׁמָּא תַּגְבִּיהַּ הַקַּרְקַע עַד שֶׁיִּתְכַּסֶּה הָאִילָן הַיּוֹצֵא מִן הַגֶּזַע מִקְצָתוֹ בַּקַּרְקַע וְיֵרָאוּ כִּשְׁלֹשָׁה אִילָנוֹת {כו}, וְיֹאמַר לוֹ הַלּוֹקֵחַ שְׁלֹשָׁה אִילָנוֹת מָכַרְתָּ לִי וְיֵשׁ לִי קַרְקַע, אֶלָּא יָקֹץ אוֹתוֹ וְיִשְׂרְפֶנּוּ: <B>וְאִם מֵתוּ.</B> שֶׁיָּבַשׁ הָאִילָן: <B>אֵין לוֹ קַרְקַע.</B> שֶׁיּוּכַל לִטַּע אַחֵר בִּמְקוֹמוֹ: <B>קָנָה קַרְקַע.</B> דַּחֲשִׁיבֵי שְׂדֵה הָאִילָן, וְקָנָה בַּקַּרְקַע שֶׁיֵּשׁ בֵּין אִילָן לְאִילָן, וְתַחְתֵּיהֶן וְחוּצָה לָהֶן כִּמְלֹא אוֹרֶה {כז} וְסַלּוֹ, כְּדֵי שֶׁיָּכוֹל לַעֲמֹד מְלַקֵּט הַפֵּרוֹת עִם הַסַּל שֶׁלּוֹ. וְהָנֵי מִלֵּי, כְּשֶׁיֵּשׁ בֵּין אִילָן לְאִילָן לֹא פָּחוֹת מֵאַרְבַּע אַמּוֹת {כח} וְלֹא יוֹתֵר עַל שֵׁשׁ עֶשְׂרֵה אַמָּה {כט}, אָז קָנָה בַּקַּרְקַע שֶׁבֵּין אִילָן לְאִילָן וְתַחְתֵּיהֶן וְחוּצָה לָהֶן כִּדְאָמְרָן. אֲבָל אִם יֵשׁ בֵּין אִילָן לְאִילָן פָּחוֹת מֵאַרְבַּע אַמּוֹת אוֹ יוֹתֵר מִשֵּׁשׁ עֶשְׂרֵה אַמָּה, לֹא קָנָה קַרְקַע: 

~ פרק ה - משנה ה
<B>לֹא מָכַר אֶת הָרַגְלָיִם.</B> בַּתּוֹסֶפְתָּא תַּנְיָא, בַּמֶּה דְּבָרִים אֲמוּרִים, בִּמְקוֹם שֶׁלֹּא נָהֲגוּ, אֲבָל בִּמְקוֹם שֶׁנָּהֲגוּ הַכֹּל כְּמִנְהַג הַמְּדִינָה {לב}: <B>מָכַר אֶת הַקָּנֶה.</B> אֶת הָרֵאָה. וְנִקְרֵאת עַל שֵׁם הַקָּנֶה שֶׁלָּהּ {לג}: 

~ פרק ה - משנה ו
<B>אַרְבַּע מִדּוֹת.</B> דִּינִים חֲלוּקִים זֶה מִזֶּה: <B>מָכַר לוֹ חִטִּים יָפוֹת.</B> הִתְנָה עִמּוֹ שֶׁיִּתֵּן לוֹ חִטִּין יָפוֹת וְנִמְצְאוּ רָעוֹת, הָוֵי כְּאוֹנָאָה {לה}, וּלְכָךְ יָכוֹל לַחֲזֹר בּוֹ, מִי שֶׁנִּתְאַנָּה בִּלְבַד, דְּהוּא לוֹקֵחַ, אֲבָל מוֹכֵר אֵינוֹ יָכוֹל לַחֲזֹר בּוֹ וַאֲפִלּוּ הוּקְרוּ הַחִטִּין הַרְבֵּה: <B>אֵין אֶחָד מֵהֶן יָכוֹל לַחֲזֹר בּוֹ.</B> וַאֲפִלּוּ הוּקְרוּ אוֹ הוּזְלוּ. וְלֹא מָצֵי לוֹקֵחַ לְמֵימַר אֲנִי לְיָפוֹת נִתְכַּוַּנְתִּי וְהַאי דַּאֲמֵינָא רָעוֹת מִשּׁוּם דְּרַע רַע יֹאמַר הַקּוֹנֶה. וְכֵן אִפְּכָא לֹא מָצֵי מוֹכֵר לְמֵימַר אֲנִי לְרָעוֹת נִתְכַּוַּנְתִּי וְהַאי דַּאֲמֵינָא יָפוֹת דְּדֶרֶךְ הַמּוֹכֵר לוֹמַר לָרַע טוֹב: <B>שְׁחַמְתִּית.</B> אֲדֻמָּה. וְכָל חוּם (בְּרֵאשִׁית ל) מְתַרְגְּמִינַן שָׁחוּם: <B>וְנִמְצֵאת לְבָנָה.</B> וְאִיכָּא דְּנִיחָא לֵיהּ בְּהַאי וְאִיכָּא דְּנִיחָא לֵיהּ בְּהַאי. וְכָל כְּהַאי גַּוְנָא הָוֵי מִקַּח טָעוּת לִשְׁנֵיהֶם וּשְׁנֵיהֶם יְכוֹלִים לַחֲזֹר. דְּאִלּוּ בְּיָפוֹת וְנִמְצְאוּ רָעוֹת כֻּלֵּי עָלְמָא נִיחָא לְהוּ בְּיָפוֹת: <B>יַיִן וְנִמְצָא חֹמֶץ.</B> אִיכָּא דְּנִיחָא לֵיהּ בְּחַמְרָא וְאִיכָּא דְּנִיחָא לֵיהּ בְּחַלָּא {לח}: 

~ פרק ה - משנה ז
<B>מָשַׁךְ.</B> מֵרְשׁוּת הָרַבִּים בְּסִמְטָא אוֹ בְּחָצֵר {לט} שֶׁל שְׁנֵיהֶם, קָנָה. אֲבָל מְשִׁיכָה בִּרְשׁוּת הָרַבִּים לֹא מַהֲנְיָא: <B>אַף עַל פִּי שֶׁלֹּא מָדַד.</B> וּכְגוֹן שֶׁפָּסַק לוֹ מִתְּחִלָּה הַדָּמִים בְּכָךְ וְכָךְ יִתֵּן לוֹ הַמִּדָּה. אֲבָל לֹא פָּסַק לוֹ הַדָּמִים, אֲפִלּוּ מָדַד וּמָשַׁךְ לֹא קָנָה, דְּלֹא סַמְכָא דַּעְתַּיְהוּ, שֶׁהַמּוֹכֵר יָכוֹל לְהַעֲלוֹת הַדָּמִים כְּמוֹ שֶׁרוֹצֶה, וְהַקּוֹנֶה אוֹמֵר אֵינִי קוֹנֶה אֶלָּא בְּזוֹל: <B>מָדַד.</B> הַמּוֹכֵר לַלּוֹקֵחַ בִּרְשׁוּת הָרַבִּים, אֲפִלּוּ בְּכֵלָיו שֶׁל לוֹקֵחַ, לֹא קָנָה, שֶׁאֵין כֵּלָיו שֶׁל אָדָם קוֹנִים לוֹ בִּרְשׁוּת הָרַבִּים {מ}. וְאִם לוֹקֵחַ עַצְמוֹ מָדַד, אֲפִלּוּ בִּרְשׁוּת הָרַבִּים קָנָה בְּהַגְבָּהָה: <B>אִם הָיָה פִקֵּחַ.</B> הַלּוֹקֵחַ: <B>שׂוֹכֵר אֶת מְקוֹמוֹ.</B> אִם הוּא בִּרְשׁוּת בְּעָלִים {מא}. וּמְקוֹמוֹ קוֹנֶה לוֹ. וּבְמַשָּׂאוֹת גְּדוֹלוֹת אַיְרֵי, דְּאֵין דַּרְכָּן לְהַגְבִּיהַּ, לְכָךְ קָנִי בִמְשִׁיכָה: <B>עַד שֶׁיְּטַלְטְלֶנּוּ.</B> הַיְנוּ הַגְבָּהָה {מב}. וְאוֹרְחָא דְּמִלְּתָא נָקַט, דְּדֶרֶךְ מַגְבִּיהַּ לְטַלְטֵל מִמָּקוֹם לְמָקוֹם {מג}: <B>וְאִם הָיָה מְחֻבָּר לַקַּרְקַע.</B> בַּגְּמָרָא מוֹקֵי לַהּ כְּגוֹן שֶׁאָמַר לוֹ מוֹכֵר לַלּוֹקֵחַ לֵךְ וְיַפֵּה לִי קַרְקַע כָּל שֶׁהוּא וּקְנֵי כָּל מַה שֶּׁעָלֶיהָ, דְּכֵיוָן שֶׁשְּׂכָרוֹ לַעֲשׂוֹת מְלָאכָה בַּקַּרְקַע שֶׁלּוֹ וְעָשָׂה בּוֹ מְלָאכָה כָּל שֶׁהוּא, קָנָה שְׂכִירוּתוֹ, וְקָנָה נַמִּי אוֹתוֹ דָּבָר שֶׁרוֹצֶה לְהַקְנוֹת לוֹ עִם הַשְּׂכִירוּת {מד}, לְפִיכָךְ אִם תָּלַשׁ כָּל שֶׁהוּא קָנָה:

<B>עַד שֶׁלֹּא נִתְמַלְּאָה הַמִּדָּה לַמּוֹכֵר.</B> בְּמִדָּה שֶׁאֵינָהּ שֶׁל שְׁנֵיהֶם מַיְרֵי {מז}, וְהַמַּשְׁאִיל הִשְׁאִילָהּ לִשְׁנֵיהֶם, וַהֲרֵי הִיא שְׁאוּלָה לַמּוֹכֵר עַד שֶׁיְּמַלְּאֶנָּה, וּמִשָּׁם וָאֵילַךְ הִיא שְׁאוּלָה לַלּוֹקֵחַ. לְפִיכָךְ עַד שֶׁלֹּא נִתְמַלְּאָה הַמִּדָּה, שֶׁהַכְּלִי שָׁאוּל לַמּוֹכֵר, לֹא קָנָה לוֹקֵחַ, וּמִשֶּׁנִּתְמַלְּאָה הַמִּדָּה, שֶׁהַכְּלִי שָׁאוּל לַלּוֹקֵחַ, קָנָה לוֹקֵחַ, שֶׁקָּנוּ לוֹ כֵּלָיו. וּבְסִמְטָא מַיְרֵי אוֹ בִּרְשׁוּת הַלּוֹקֵחַ {מח}: <B>הָיָה סַרְסוּר בֵּינֵיהֶם וְנִשְׁבְּרָה הֶחָבִית.</B> וְהָיְתָה הֶחָבִית שֶׁלּוֹ: <B>נִשְׁבְּרָה לַסַּרְסוּר.</B> וְלֹא אָמְרִינַן שְׁלוּחוֹ שֶׁל לוֹקֵחַ הוּא {מט} וְנִשְׁבְּרָה לַלּוֹקֵחַ: <B>וְחַיָּב לְהַטִּיף לוֹ שָׁלֹשׁ טִפִּין.</B> הַמּוֹכֵר לַלּוֹקֵחַ, מִן הַנִּדְבָּק בְּדָפְנֵי הַמִּדָּה לְאַחַר שֶׁעֵרָה הַיַּיִן וְהַשֶּׁמֶן: <B>הִרְכִּינָהּ וּמִצָּה.</B> הִטָּה אֶת הַמִּדָּה עַל צִדָּהּ לְאַחַר שֶׁעֵרָה הַשֶּׁמֶן וְהַיַּיִן וְהִטִּיף שָׁלֹשׁ טִפִּין, וְנִתְאַסֵּף מִצַּת הַמִּדָּה לְשׁוּלֵי הַכְּלִי אֶל מָקוֹם אֶחָד: <B>הֲרֵי הוּא שֶׁל מוֹכֵר.</B> שֶׁהֲרֵי נִתְיָאֵשׁ הַלּוֹקֵחַ מִמֶּנּוּ: <B>וְהַחֶנְוָנִי אֵינוֹ חַיָּב לְהַטִּיף שָׁלֹשׁ טִפִּין.</B> לְפִי שֶׁהוּא טָרוּד, שֶׁמּוֹכֵר כָּל שָׁעָה: <B>רַבִּי יְהוּדָה אוֹמֵר.</B> לֹא אָמְרוּ שֶׁחֶנְוָנִי פָּטוּר מִלְּהַטִּיף שָׁלֹשׁ טִפִּין, אֶלָּא עֶרֶב שַׁבָּת עִם חֲשֵׁכָה. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

<B>הַשּׁוֹלֵחַ בְּנוֹ.</B> קָטָן אֵצֶל הַחֶנְוָנִי וּבְיָדוֹ פֻּנְדְּיוֹן שֶׁהֵם שְׁנֵי אִסָּרִין, לְהָבִיא לוֹ בְּאִסָּר שֶׁמֶן וְאִסָּר אֶחָד יִתֵּן לוֹ, וְכֵן עָשָׂה, וְשָׁבַר הַתִּינוֹק אֶת הַצְּלוֹחִית וְנִשְׁפַּךְ הַשֶּׁמֶן וְאִבֵּד אֶת הָאִסָּר שֶׁנָּתַן לוֹ הַחֶנְוָנִי: <B>חֶנְוָנִי חַיָּב.</B> בַּשֶּׁמֶן וּבַצְּלוֹחִית וּבָאִסָּר. דְּלֹא שָׁלַח הָאָב אֶת בְּנוֹ הַקָּטָן אֶל הַחֶנְוָנִי אֶלָּא לְאוֹדוּעֵי לֵיהּ שֶׁהוּא צָרִיךְ לַשֶּׁמֶן, לֹא שֶׁיִּשְׁלַח הַשֶּׁמֶן עַל יָדוֹ. וּבַגְּמָרָא פָּרֵיךְ, אַמַּאי חַיָּב הַחֶנְוָנִי עַל הַצְּלוֹחִית, אֲבֵדָה מִדַּעַת הִיא {נא}. שֶׁהֲרֵי הוּא עַצְמוֹ שְׁלָחָהּ בְּיַד בְּנוֹ. וּמְשָׁנֵי, כְּגוֹן שֶׁנְּטָלָהּ חֶנְוָנִי לָמֹד בָּהּ שֶׁמֶן לַאֲחֵרִים, הִלְכָּךְ נַעֲשֶׂה גַּזְלָן עֲלֵיהּ וְקָמָה לֵיהּ בִּרְשׁוּתֵיהּ עַד דִּמְהַדֵּר לָהּ לְמָרָהּ, וְאַף עַל גַּב דְּאַהַדְרָהּ לַתִּינוֹק לֹא מִפְטַר מִנַּהּ: <B>רַבִּי יְהוּדָה פּוֹטֵר שֶׁעַל מְנָת כֵּן שְׁלָחוֹ.</B> כִּי הֵיכִי דְּלִשְׁדַר לֵיהּ חֶנְוָנִי בְּיַד בְּנוֹ: <B>שֶׁהוּא פָטוּר.</B> מִדְּמֵי צְלוֹחִית. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

הוֹסִיף רֹבַע, הִגִּיעוֹ. וְאִם אָמַר בֵּית כּוֹר סְתָם, סְתָמָא נַמִּי כְּהֵן חָסֵר הֵן יָתֵר דָּמִי {ט}: <B>יוֹתֵר מִכָּאן.</B> יוֹתֵר מֵרֹבַע לְכָל סְאָה אִם הוֹתִיר, יַעֲשֶׂה חֶשְׁבּוֹן, יַחְשֹׁב מֻעָט דְּלֹא חֲזִי לֵיהּ לְמִידִי: <B>שֶׁאִם שִׁיֵּר.</B> בְּבֵית כּוֹר שָׂדֶה בֵּית תִּשְׁעָה קַבִּין, דְּהַשְׁתָּא חֲזִי לֵיהּ הַאי קַרְקַע, כִּדְאָמְרִינַן דַּף י"א, דְּבֵית תִּשְׁעָה קַבִּין חָשׁוּב לִהְיוֹת נִקְרָא שָׂדֶה, אָז אֵין מַחֲזִיר לוֹ מָעוֹת, אֶלָּא הַמּוֹכֵר נוֹטֵל אֶת הַקַּרְקַע שֶׁלּוֹ. וְשָׁמְעִינַן מִמַּתְנִיתִין דְּשִׁבְעַת קַבִּין וּמֶחֱצָה לְכוֹר, שֶׁהוּא רֹבַע הַקַּב <B>מִדָּה בַחֶבֶל וְכוּ'.</B> הָאוֹמֵר לַחֲבֵרוֹ שְׁתֵּי לְשׁוֹנוֹת הַלָּלוּ הַסּוֹתְרִים זֶה אֶת זֶה, מִדָּה בַּחֶבֶל דְּמַשְׁמַע בְּצִמְצוּם, הֵן חָסֵר הֵן יָתֵר דְּמַשְׁמַע דְּלֹא קָפֵיד עַל מְעַט פָּחוֹת אוֹ יוֹתֵר. בִּטֵּל לָשׁוֹן אַחֲרוֹן שֶׁהוּא הֵן חָסֵר הֵן יָתֵר, אֶת לָשׁוֹן רִאשׁוֹן שֶׁהוּא מִדָּה בַּחֶבֶל. וַאֲפִלּוּ פִּחֵת רֹבַע לִסְאָה אוֹ הוֹתִיר רֹבַע לִסְאָה, הִגִּיעוֹ: <B>דִּבְרֵי בֶן נַנָּס.</B> דִּסְבִירָא לֵיהּ תְּפוֹס לָשׁוֹן אַחֲרוֹן. וּפְלִיגֵי רַבָּנָן עֲלֵיהּ, וְאָמְרֵי הַלֵּךְ אַחַר פָּחוּת שֶׁבַּלְּשׁוֹנוֹת, בֵּין רִאשׁוֹן בֵּין אַחֲרוֹן. דִּמְסָפְקָא לְהוּ אִי תְּפוֹס לָשׁוֹן רִאשׁוֹן אוֹ לָשׁוֹן אַחֲרוֹן, וְנוֹטֵל הַלּוֹקֵחַ כַּפָּחוּת שֶׁבַּשִּׁעוּרִין {יב}. וַהֲלָכָה כַּחֲכָמִים: <B>בְּסִימָנָיו וּבִמְצָרָיו.</B> אָמַר לוֹ בֵּית כּוֹר עָפָר אֲנִי מוֹכֵר לְךָ בְּתוֹךְ הַסִּימָנִים וְהַמְּצָרִים הָאֵלֶּה: <B>הָכִי גָּרְסִינַן פָּחוֹת מִשְּׁתוּת הִגִּיעוֹ עַד שְׁתוּת יְנַכֶּה.</B> וְהָכִי פֵּרוּשָׁא, אִם פִּחֵת מִן הַבֵּית כּוֹר שְׁתוּת וְכָל שֶׁכֵּן פָּחוֹת מִשְּׁתוּת, הִגִּיעוֹ {יג}. אֲבָל יוֹתֵר עַל שְׁתוּת וּלְמַטָּה עַד שְׁתוּת וְלֹא שְׁתוּת בִּכְלָל, יְנַכֶּה {יד}:

~ פרק ח - משנה ב
<B>כָּל הַקּוֹדֵם לַנַּחֲלָה יוֹצְאֵי יְרֵכוֹ קוֹדְמִין.</B> וְאִם אֵין יוֹצְאֵי יְרֵכוֹ קַיָּמִין, חָזְרָה נַחֲלָתָן אֵצֶל הָאָב. כֵּיצַד, רְאוּבֵן שֶׁמֵּת, חֲנוֹךְ וּפַלּוּא חֶצְרוֹן וְכַרְמִי [בָּנָיו] יוֹרְשִׁין אוֹתוֹ. וְאִם מֵת אֶחָד מִן הַבָּנִים קֹדֶם רְאוּבֵן, וְהַהוּא בֵּן הִנִּיחַ בֵּן אוֹ בַּת [אוֹ בֶּן בֵּן אוֹ בַּת בֵּן] אוֹ בֶּן בַּת אוֹ בַּת בַּת] עַד מֵאָה דּוֹרוֹת, יִירַשׁ בִּמְקוֹם אָבִיו בְּנִכְסֵי אָבִיו רְאוּבֵן, וְיִטֹּל כְּאֶחָד מִבְּנֵי רְאוּבֵן. וְאִם אֵין לִרְאוּבֵן בָּנִים וְלֹא בְּנֵי בָנִים וְאַף לֹא בְּנוֹת בָּנִים, אָז יוֹרְשׁוֹת בְּנוֹת רְאוּבֵן אֶת רְאוּבֵן. וְאִם אֵין לִרְאוּבֵן בָּנוֹת וְלֹא בְּנֵי בָּנוֹת וְלֹא בְּנוֹת בָּנוֹת עַד סוֹף כָּל הַדּוֹרוֹת, אָז יִירַשׁ יַעֲקֹב אֶת רְאוּבֵן בְּנוֹ. וְאִם אֵין יַעֲקֹב קַיָּם, יִירְשׁוּ אֶת רְאוּבֵן אֶחָיו בְּנֵי אָבִיו, שִׁמְעוֹן לֵוִי וִיהוּדָה [יוֹסֵף וּבִנְיָמִין] וְכוּ'. וְאִם אֵינָן קַיָּמִין, יִירְשׁוּ אֶת רְאוּבֵן בְּנֵי שִׁמְעוֹן וְלֵוִי וְכוּ'. וְאִם אֵין לָהֶם בָּנִים וְלֹא בְּנֵי בָנִים, יִירְשׁוּ בְּנוֹתֵיהֶן אוֹ בְּנֵי בְנוֹתֵיהֶן אוֹ בְּנוֹת בְּנוֹתֵיהֶן. דִּבְכָל מָקוֹם הַזָּכָר וְיוֹרְשָׁיו קוֹדְמִין לַנְּקֵבָה {טו}. וְאִם מֵתוּ אֶחָיו שֶׁל רְאוּבֵן בְּלֹא זֶרַע אוֹ שֶׁלֹּא הָיָה לוֹ אָח מֵעוֹלָם, תִּירְשֶׁנּוּ אֲחוֹתוֹ דִּינָה בַּת יַעֲקֹב אוֹ בָּנֶיהָ אוֹ בְּנֵי בָנֶיהָ [אוֹ בְּנוֹתֶיהָ וּבְנוֹת בְּנוֹתֶיהָ] עַד סוֹף הָעוֹלָם. וְאִם אֵין לִרְאוּבֵן אָחוֹת וְלֹא זֶרַע אָחוֹת, תַּחֲזֹר יְרֻשָּׁתוֹ לַאֲבִי אָבִיו דְּהַיְנוּ לְיִצְחָק. וְאִם אֵין יִצְחָק קַיָּם, תַּחֲזֹר יְרֻשַּׁת רְאוּבֵן לְעֵשָׂו בֶּן יִצְחָק, שֶׁהוּא אֲחִי אָבִיו שֶׁל רְאוּבֵן. וְאִם אֵין עֵשָׂו קַיָּם, תַּחֲזֹר יְרֻשַּׁת רְאוּבֵן לֶאֱלִיפַז בֶּן עֵשָׂו אוֹ לְבָנָיו וְלִבְנֵי בָנָיו אוֹ לִבְנוֹתָיו וְלִבְנוֹת בְּנוֹתָיו עַד סוֹף כָּל הַדּוֹרוֹת. וְאִם אֵין אַחִים לְאָבִיו שֶׁל רְאוּבֵן וְלֹא מִבְּנֵי בְנֵיהֶן, תַּחֲזֹר יְרֻשַּׁת רְאוּבֵן לַאֲחוֹת אָבִיו אוֹ לְבָנֶיהָ וְלִבְנֵי בָנֶיהָ אוֹ לִבְנוֹתֶיהָ עַל הַסֵּדֶר שֶׁאָמַרְנוּ. וְאִם אֵין לְאָבִיו שֶׁל רְאוּבֵן לֹא אַחִים וְלֹא בְּנֵי אַחִים וְלֹא אָחוֹת וְלֹא בְּנֵי אָחוֹת, תַּחְזֹר יְרֻשַּׁת רְאוּבֵן לְאַבְרָהָם אֲבִי אֲבִי אָבִיו. וְכֵן לְעוֹלָם עַד אָדָם הָרִאשׁוֹן: <B>הָאָב קוֹדֵם.</B> לְעוֹלָם לַאֲחֵי הָאָב {טז} וְלִבְנֵיהֶם, [וַאֲחֵי הָאָב קוֹדְמִים לַאֲחוֹת הָאָב], וַאֲחֵי הָאָב וַאֲחוֹת הָאָב קוֹדְמִים [לַאֲבִי הָאָב. וַאֲבִי הָאָב קוֹדֵם לַאֲחֵי אֲבִי הָאָב. וַאֲחֵי אֲבִי הָאָב וְאַף אֲחוֹתוֹ קוֹדְמִין] לַאֲבִי אֲבִי הָאָב. וְכֵן לְעוֹלָם: 

~ פרק ח - משנה ג
<B>נָטְלוּ שְׁלֹשָׁה חֲלָקִים בַּנַּחֲלָה.</B> בְּנַחֲלַת אֶרֶץ יִשְׂרָאֵל. וְסָבַר הַאי תַּנָּא לְיוֹצְאֵי מִצְרַיִם נִתְחַלְּקָה הָאָרֶץ {יח}, דִּכְתִיב (בַּמִדְבָּר כו) לִשְׁמוֹת מַטּוֹת אֲבוֹתָם יִנְחָלוּ. וּצְלָפְחָד וְחֵפֶר אָבִיו שְׁנֵיהֶם נָטְלוּ חֵלֶק בָּאָרֶץ, שֶׁמִּיּוֹצְאֵי מִצְרַיִם הָיוּ שְׁנֵיהֶם. וְנָטְלוּ בְּנוֹת צְלָפְחָד חֵלֶק אֲבִיהֶם שֶׁהָיָה מַגִּיעַ לוֹ בָּאָרֶץ, וְחֵלֶק הַמַּגִּיעַ לוֹ מִירֻשַּׁת חֵפֶר אָבִיו, וְחֵלֶק בְּכוֹרָתוֹ שֶׁהָיָה בְּכוֹר וְנוֹטֵל פִּי שְׁנַיִם. וְאַף עַל פִּי שֶׁעֲדַיִן לֹא יָרְשׁוּ אֶת הָאָרֶץ וְאֵין הַבְּכוֹר נוֹטֵל פִּי שְׁנַיִם בָּרָאוּי לָבֹא לְאַחַר מִיתָה, אֶרֶץ יִשְׂרָאֵל מֻחְזֶקֶת הָיְתָה {יט}: 

~ פרק ח - משנה ד
<B>אֶחָד הַבֵּן וְאֶחָד הַבַּת בַּנַּחֲלָה.</B> הָכִי קָאָמַר, אֶחָד הַבֵּן וְאֶחָד הַבַּת שָׁוִין בִּירֻשַּׁת נִכְסֵי הָאֵם כְּבִירֻשַּׁת נִכְסֵי הָאָב, וְאֵין הֶפְרֵשׁ בֵּין יְרֻשַּׁת נִכְסֵי הָאֵם לִירֻשַּׁת נִכְסֵי הָאָב, אֶלָּא שֶׁהַבְּכוֹר נוֹטֵל פִּי שְׁנַיִם בְּנִכְסֵי הָאָב וְאֵינוֹ נוֹטֵל פִּי שְׁנַיִם בְּנִכְסֵי הָאֵם: 

~ פרק ח - משנה ה
<B>לֹא אָמַר כְּלוּם.</B> וְאֵין יָכוֹל לְסַלְּקוֹ מִן הַיְרֻשָּׁה אֶלָּא עַל יְדֵי שֶׁיִּתֵּן בְּמַתָּנָה נְכָסָיו לִשְׁאָר בָּנָיו: <B>עַל פִּיו.</B> מִשּׁוּם דִּמְצַוֶּה מֵחֲמַת מִיתָה, וּשְׁכִיב מְרַע דְּבָרָיו כִּכְתוּבִים וְכִמְסוּרִים וְאֵינוֹ צָרִיךְ קִנְיָן. מִשּׁוּם הָכִי תָּנֵי עַל פִּיו: <B>וְהִשְׁוָה לָהֶן אֶת הַבְּכוֹר.</B> בִּלְשׁוֹן מַתָּנָה: <B>דְּבָרָיו קַיָּמִין.</B> וְאֵין כָּאן מַתְנֶה עַל מַה שֶּׁכָּתוּב בַּתּוֹרָה {כד}, שֶׁיֵּשׁ כֹּחַ בָּאָדָם לִתֵּן מָמוֹנוֹ בְּמַתָּנָה לְכָל מִי שֶׁיִּרְצֶה: <B>וְאִם אָמַר מִשּׁוּם יְרֻשָּׁה.</B> וְאִם רִבָּה לְאֶחָד וּמִעֵט לְאֶחָד בְּתוֹרַת יְרֻשָּׁה, שֶׁאָמַר פְּלוֹנִי בְּנִי יִירַשׁ שָׂדֶה בֵּית כּוֹר, וּפְלוֹנִי [בְּנִי] יִירַשׁ שָׂדֶה בֵּית לֶתֶךְ, וְעַל בְּנוֹ בְּכוֹר אָמַר שֶׁיִּירַשׁ כְּשֶׁל חֲבֵרוֹ, לֹא אָמַר כְּלוּם, מִפְּנֵי שֶׁמַּתְנֶה עַל מַה שֶּׁכָּתוּב בַּתּוֹרָה: <B>כָּתַב בֵּין בַּתְּחִלָּה וְכוּ'.</B> תִּנָּתֵן שָׂדֶה פְּלוֹנִית לִפְלוֹנִי וְיִירָשֶׁנָּה, זֶהוּ בַּתְּחִלָּה. יִירָשֶׁנָּה וְתִנָּתֵן לוֹ, זוֹ הִיא בַּסּוֹף. יִירַשׁ שָׂדֶה פְּלוֹנִית וְתִנָּתֵן לוֹ וְיִירָשֶׁנָּה, זוֹ הִיא מַתָּנָה בָּאֶמְצַע: <B>אִם אָמַר עַל מִי שֶׁרָאוּי לְיָרְשׁוֹ.</B> כְּגוֹן עַל בֵּן בֵּין הַבָּנִים, וְעַל בַּת בֵּין הַבָּנוֹת, פְּלוֹנִי יִירָשֵׁנִי: <B>דְּבָרָיו קַיָּמִין.</B> דִּכְתִיב (דְּבָרִים כא) וְהָיָה בְּיוֹם הַנְחִילוֹ אֶת בָּנָיו {כו}, הַתּוֹרָה נָתְנָה רְשׁוּת לָאָב לְהַנְחִיל לַבָּנִים לְמִי שֶׁיִּרְצֶה. וּמוֹדֶה רַבִּי יוֹחָנָן דְּעַל אָח בִּמְקוֹם בַּת וְעַל בַּת בִּמְקוֹם בֵּן, לֹא אָמַר כְּלוּם, שֶׁאֵין הַבַּת רְאוּיָה לִירַשׁ בִּמְקוֹם בֵּן, וְלֹא הָאָח בִּמְקוֹם בַּת. וְכֵן מוֹדֶה רַבִּי יוֹחָנָן, שֶׁאִם הִשְׁוָה אֶת הַבְּכוֹר לַבָּנִים לֹא אָמַר כְּלוּם, דִּכְתִיב (שָׁם) לֹא יוּכַל לְבַכֵּר וְכוּ'. וַהֲלָכָה כְּרַבִּי יוֹחָנָן בֶּן בְּרוֹקָה: <B>אֵין רוּחַ חֲכָמִים נוֹחָה הֵימֶנּוּ.</B> אֵין לַחֲכָמִים נַחַת רוּחַ בְּמַעֲשָׂיו, וַאֲפִלּוּ אֵין בָּנָיו נוֹהֲגִים כַּשּׁוּרָה, דִּלְמָא נָפֵיק מִנַּיְהוּ זַרְעָא מַעַלְיָא. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן בֶּן גַּמְלִיאֵל:

~ פרק ט - משנה א
<B>מִי שֶׁמֵּת.</B> שֶׁהַנְּכָסִים מְרֻבִּים. כְּדֵי שֶׁיִּזּוֹנוּ הַבָּנִים וְהַבָּנוֹת עַד שֶׁיִּבְגְּרוּ הַבָּנוֹת: <B>שֶׁהַנְּכָסִים מֻעָטִין.</B> שֶׁאֵין בָּהֶם כְּשִׁעוּר זֶה: <B>בִּשְׁבִיל שֶׁאֲנִי זָכָר הִפְסַדְתִּי.</B> בִּתְמִיהָה. אֶלָּא לֹא אַפְסִיד, וְנִזּוֹן כֻּלָּנוּ יַחַד. וְאֵין הֲלָכָה כְּאַדְמוֹן. וְעָשׂוּ אַלְמָנָה אֵצֶל הַבַּת בִּנְכָסִים מֻעָטִים, כְּבַת אֵצֶל הָאַחִין, מַה בַּת אֵצֶל הָאַחִין הַבַּת נִזּוֹנֵת וְהַבָּנִים יִשְׁאֲלוּ עַל הַפְּתָחִים, אַף אַלְמָנָה אֵצֶל הַבַּת הָאַלְמָנָה נִזּוֹנֵת וְהַבַּת תִּשְׁאַל עַל הַפְּתָחִים: 

~ פרק ט - משנה ב
<B>וְטֻמְטוּם.</B> סָפֵק זָכָר סָפֵק נְקֵבָה: <B>הַזְּכָרִים דּוֹחִין אוֹתוֹ אֵצֶל הַנְּקֵבוֹת.</B> דְּאָמְרֵי לֵיהּ אַיְתֵי רְאָיָה דְּזָכָר אַתְּ וּשְׁקוֹל: <B>הַנְּקֵבוֹת דּוֹחוֹת אוֹתוֹ אֵצֶל הַזְּכָרִים.</B> דְּאָמְרֵי לֵיהּ אַיְתֵי רְאָיָה דִּנְקֵבָה אַתְּ וְתִזּוֹן עִמָּנוּ: <B>יָלְדָה זָכָר נוֹטֵל מָנֶה.</B> וְאַף עַל גַּב דְּקַיְמָא לָן הַמְזַכֶּה לָעֻבָּר לֹא קָנָה, הָכָא בִּשְׁכִיב מְרַע שֶׁזָּכָה לִבְנוֹ קָנָה, שֶׁדַּעְתּוֹ שֶׁל אָדָם קְרוֹבָה אֵצֶל בְּנוֹ {ב}: <B>יָלְדָה זָכָר וּנְקֵבָה.</B> תְּאוֹמִים יַחַד {ד}: <B>יָלְדָה טֻמְטוּם אֵינוֹ נוֹטֵל.</B> הַאי מַתְנִיתִין אִדַּחְיָא לַהּ, וְהִלְכָתָא שֶׁטֻּמְטוּם נוֹטֵל כַּפָּחוּת שֶׁבִּשְׁנֵיהֶם {ה}: <B>יוֹרֵשׁ אֶת הַכֹּל.</B> דְּלָא תֵימָא בִּרְיָה בִּפְנֵי עַצְמָהּ הוּא וְלָאו בַּר יְרֻשָּׁה הוּא, קָמַשְׁמַע לָן: 

~ פרק ט - משנה ג
<B>הִשְׁבִּיחוּ גְּדוֹלִים אֶת הַנְּכָסִים.</B> בְּעוֹדָן בִּתְפוּסַת הַבַּיִת: <B>הִשְׁבִּיחוּ לָאֶמְצַע.</B> וְהָנֵי מִלֵּי כְּשֶׁשָּׁבְחוּ הַנְּכָסִים מֵחֲמַת נְכָסִים עַצְמָן, שֶׁלֹּא הוֹצִיאוּ מִשֶּׁלָּהֶן כְּלוּם אֶלָּא מִנִּכְסֵי אֲבִיהֶן שָׂכְרוּ פּוֹעֲלִים וְהִשְׁבִּיחוּ הַנְּכָסִים מֵחֲמַת עַצְמָן. אֲבָל אִם חָפְרוּ וְנָטְעוּ וְהוֹצִיאוּ מִשֶּׁלָּהֶן, מַה שֶּׁהִשְׁבִּיחוּ הִשְׁבִּיחוּ לְעַצְמָן: <B>וְכֵן הָאִשָּׁה שֶׁהִשְׁבִּיחָה.</B> בַּגְּמָרָא מוֹקְמִינַן לַהּ בְּאִשָּׁה יוֹרֶשֶׁת. כְּגוֹן שֶׁנָּשָׂא רְאוּבֵן אֶת בַּת שִׁמְעוֹן אָחִיו, וּמֵת בְּלֹא בָנִים, וּבְנוֹת שִׁמְעוֹן אָחִיו יוֹרְשׁוֹת אוֹתוֹ, נִמְצֵאת אִשְׁתּוֹ שֶׁהִיא בַּת שִׁמְעוֹן אָחִיו יוֹרֶשֶׁת אוֹתוֹ עִם שְׁאָר אַחְיוֹתֶיהָ, וְסָלְקָא דַּעְתָּךְ אֲמֵינָא דִּבְהַהִיא הֲנָאָה דְּנָפֵיק עֲלַהּ קָלָא שֶׁשּׁוֹלֶטֶת וּמַשְׁבַּחַת אֲפִלּוּ כִּי אָמְרָה רְאוּ מַה הִנִּיחַ לִי בַּעְלִי וְכוּ' אֲפִלּוּ הָכִי מָחֲלָה וְרוֹצָה שֶׁיִּהְיֶה הַשֶּׁבַח לָאֶמְצַע, קָמַשְׁמַע לָן דְּהִשְׁבִּיחָה לְעַצְמָהּ {ח}: 

~ פרק ט - משנה ד
<B>שֶׁנָּפַל אֶחָד מֵהֶן לָאֻמָּנוּת.</B> לַעֲבוֹדַת הַמֶּלֶךְ. שֶׁכֵּן מִנְהַג הַמֶּלֶךְ לְהַעֲמִיד מִכָּל בָּתֵּי הָעִיר אָדָם אֶחָד לְמוֹכֵס חֹדֶשׁ אֶחָד אוֹ שְׁנֵי חֳדָשִׁים: <B>נָפַל לָאֶמְצַע.</B> כָּל הָרֶוַח. מִפְּנֵי שֶׁמֵּחֲמַת אֲבִיהֶם בָּאָת לוֹ. אֲבָל אִם מֵחֲמַת עַצְמוֹ, מִפְּנֵי חֲרִיפוּתוֹ וַחֲשִׁיבוּתוֹ וְחָכְמָתוֹ נִלְקַח לַעֲבוֹדַת הַמֶּלֶךְ {יא}, מַה שֶּׁהִרְוִיחַ הִרְוִיחַ לְעַצְמוֹ: <B>חָלָה וְנִתְרַפָּא.</B> אִם חָלָה בְּאֹנֶס, כְּלוֹמַר שֶׁלֹּא פָּשַׁע הוּא בְּעַצְמוֹ, מִתְרַפֵּא מִן הָאֶמְצַע. אֲבָל אִם חָלָה מִצִּנִּים פַּחִים וְכַיּוֹצֵא בָהֶם {יב}, דִּכְתִיב בְּהוּ (משלי כב) שׁוֹמֵר נַפְשׁוֹ יִרְחַק מֵהֶם, בְּהָא אָמְרִינַן בְּמַתְנִיתִין דְּאֵינוֹ מִתְרַפֵּא אֶלָּא מִשֶּׁל עַצְמוֹ: <B>שׁוּשְׁבִינוּת.</B> דֶּרֶךְ שׁוּשְׁבִינוּת נוֹטֵל {יד} סְעוּדָה וְדוֹרוֹן לַחֻפָּה וְאוֹכֵל עִמּוֹ, וְזֶה חוֹזֵר וְעוֹשֶׂה כֵן לָזֶה גַּם כֵּן כְּשֶׁיִּשָּׂא אִשָּׁה. וְאִם שָׁלַח הָאָב שׁוּשְׁבִינוּת בִּסְתָם עַל יַד אֶחָד מִבָּנָיו וְחָזְרָה הַשּׁוּשְׁבִינוּת אַחַר כָּךְ לְאַחַר שֶׁמֵּת הָאָב, חָזְרָה לָאֶמְצַע {טו}, מִפְּנֵי שֶׁהִיא כְּמִלְוָה, שֶׁהֲרֵי נִגְבֵּית בְּבֵית דִּין, שֶׁיָּכוֹל הַמּוֹלִיךְ שׁוּשְׁבִינוּת לַחֲבֵרוֹ לַחֲזֹר וְלִתְבֹּעַ מִמֶּנּוּ בְּבֵית דִּין שֶׁיִּתֵּן לוֹ שׁוּשְׁבִינוּת כְּדֶרֶךְ שֶׁנָּתַן לוֹ. וְהוּא שֶׁיִּהְיוּ נִשּׂוּאֵי הַשֵּׁנִי כְּנִשּׂוּאֵי הָרִאשׁוֹן, אִם בְּתוּלָה בְּתוּלָה, וְאִם אַלְמָנָה אַלְמָנָה, וְאִם פַרְהֶסְיָא פַרְהֶסְיָא, וְאִם בְּצִנְעָא בְּצִנְעָא. שֶׁיָּכוֹל הַשֵּׁנִי לוֹמַר לָרִאשׁוֹן, אֵינִי עוֹשֶׂה עִמְּךָ אֶלָּא כְּדֶרֶךְ שֶׁעָשִׂיתָ עִמִּי: <B>אֲבָל הַשּׁוֹלֵחַ לַחֲבֵרוֹ כַּדֵּי יַיִן וְכַדֵּי שֶׁמֶן.</B> בְּלֹא חֻפָּה. אוֹ אִי נַמִּי בְּחֻפָּה וְאֵינוֹ הוֹלֵךְ לֶאֱכֹל עִמּוֹ, אֵין זוֹ שׁוּשְׁבִינוּת אֶלָּא מַתָּנָה, וְאֵינָהּ נִגְבֵּית בְּבֵית דִּין: 

~ פרק ט - משנה ה
<B>הַשּׁוֹלֵחַ סִבְלוֹנוֹת.</B> מִנְהַג חֲתָנִים לְמָחֳרַת הַקִּדּוּשִׁין שׁוֹלְחִין לְבֵית הָאֲרוּסָה תַּכְשִׁיטִים וּמִינֵי מִגְדָּנוֹת וְכַדֵּי יַיִן וְכַדֵּי שֶׁמֶן, וּפְעָמִים הוֹלֵךְ הֶחָתָן וְאוֹכֵל שָׁם: <B>אֵינָן נִגְבִּין.</B> אִם מֵת הוּא אוֹ מֵתָה הִיא אוֹ שֶׁבָּא לְגָרְשָׁהּ. דְּמֵחֲמַת חִבַּת שִׂמְחַת אֲכִילָה מָחַל. וְדַוְקָא שֶׁאָכַל בְּדִינָר, אֲבָל אָכַל פָּחוֹת מִדִּינָר לֹא מָחַל וְגוֹבֶה הַסִּבְלוֹנוֹת: <B>שָׁלַח סִבְלוֹנוֹת מְרֻבִּים.</B> וַאֲפִלּוּ מֻעָטִים, אִם פֵּרֵשׁ עַל מְנָת שֶׁיָּבֹאוּ עִמָּהּ לְבֵית בַּעְלָהּ הֲרֵי אֵלּוּ נִגְבִּין. וְאוֹרְחָא דְּמִלְּתָא נָקַט, שֶׁדֶּרֶךְ הַסִּבְלוֹנוֹת שֶׁשּׁוֹלְחִין כְּדֵי שֶׁיָּבֹאוּ לְבֵית בַּעְלָהּ לִהְיוֹת מְרֻבִּין, וְאוֹתָן שֶׁשּׁוֹלְחִין כְּדֵי שֶׁתִּתְקַשֵּׁט בָּהֶן בְּבֵית אָבִיהָ דַּרְכָּן לִהְיוֹת מֻעָטִין:

<B>וְשִׁיֵּר קַרְקַע כָּל שֶׁהוּא.</B> וְהוּא הַדִּין אִם שִׁיֵּר לְעַצְמוֹ מִטַּלְטְלִין, מַתְּנָתוֹ מַתָּנָה, וְאִם עָמַד אֵינוֹ חוֹזֵר. וְהוּא שֶׁקָּנוּ מִיָּדוֹ עַל אוֹתוֹ מַתָּנָה, דְּמַתְּנַת שְׁכִיב מְרַע בְּמִקְצָת בָּעֵי קִנְיָן בֵּין עָמַד בֵּין לֹא עָמַד. וְאִם לֹא קָנוּ מִיָּדוֹ לֹא קָנָה מְקַבֵּל הַמַּתָּנָה אֲפִלּוּ מֵת הַשְּׁכִיב מְרַע. וְדַוְקָא כְּשֶׁבָּא לְהַקְנוֹת הַמַּתָּנָה מֵחַיִּים {יח}. אֲבָל אִם אָמַר יִתְּנוּ לִפְלוֹנִי כָּךְ וְכָךְ אַחַר מוֹתוֹ אֵינוֹ צָרִיךְ קִנְיָן, וַאֲפִלּוּ שִׁיֵּר לְעַצְמוֹ. וּשְׁכִיב מְרַע שֶׁבָּא לִתֵּן מַתָּנָה מֵחַיִּים, אִם לֹא שִׁיֵּר לְעַצְמוֹ כְּלוּם וְעָמַד, חוֹזֵר וַאֲפִלּוּ קָנוּ מִיָּדוֹ. וּמְצַוֶּה מֵחֲמַת מִיתָה דִּשְׁמַעְנָא לֵיהּ דְּקָאָמַר וַי לֵיהּ דְּמָיֵת, לְעוֹלָם חוֹזֵר, בֵּין שִׁיֵּר בֵּין לֹא שִׁיֵּר, וַאֲפִלּוּ קָנוּ מִיָּדוֹ. וְאִם מֵת, קָנָה {כ} מְקַבֵּל הַמַּתָּנָה וַאֲפִלּוּ בְּלֹא קִנְיָן: <B>לֹא כָתַב בָּהּ שְׁכִיב מְרַע.</B> כַּד קָצִיר וְרָמֵי בְּעַרְסֵיהּ. וְאַף לֹא לְשׁוֹן מַתְּנַת בָּרִיא, כַּד הָלַךְ עַל רַגְלוֹהִי בְּשׁוּקָא. וְלֹא שִׁיֵּר כְּלוּם: <B>הוּא אוֹמֵר שְׁכִיב מְרַע הָיִיתִי.</B> וְחוֹזְרַנִי בִּי. וְהַלָּה אוֹמֵר בָּרִיא הָיִיתָ וְלֹא תַחֲזֹר: <B>צָרִיךְ לְהָבִיא רְאָיָה שֶׁהָיָה שְׁכִיב מְרַע.</B> דְּבָתַר הַשְׁתָּא אַזְלִינַן, דְּהָאִידְנָא בָּרִיא הוּא, וְעָלָיו לְהָבִיא רְאָיָה שֶׁהָיָה שְׁכִיב מְרַע בִּשְׁעַת הַמַּתָּנָה: <B>וַחֲכָמִים אוֹמְרִים.</B> לֹא אַזְלִינַן בָּתַר הַשְׁתָּא אֶלָּא אָמְרִינַן הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה, וּמְקַבֵּל הַמַּתָּנָה שֶׁבָּא לְהוֹצִיא מִיַּד הַנּוֹתֵן שֶׁהוּא מֻחְזָק, עָלָיו לְהָבִיא רְאָיָה וְעֵדִים שֶׁבָּרִיא הָיָה בְּאוֹתָהּ שָׁעָה. וַהֲלָכָה כַּחֲכָמִים: 

~ פרק ט - משנה ז
<B>רַבִּי אֱלִיעֶזֶר אוֹמֵר אֶחָד בָּרִיא וְאֶחָד מְסֻכָּן.</B> רַבִּי אֱלִיעֶזֶר לֵית לֵיהּ דִּבְרֵי שְׁכִיב מְרַע כִּכְתוּבִין וְכִמְסוּרִין דָּמוּ, וַאֲפִלּוּ מְצַוֶּה מֵחֲמַת מִיתָה {כב} סְבִירָא לֵיהּ דְּאֵין מַתְּנָתוֹ מַתָּנָה אֶלָּא בְּקִנְיָן, כְּמַתְּנַת בָּרִיא. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: <B>כְּבִינָתִי.</B> תַּרְגּוּם רְדִידִים כְּבִנְתָּא: <B>תְּקַבְּרֵם אִמָּם.</B> כְּלוֹמַר אֵין מְבִיאִין רְאָיָה מֵהֶם, שֶׁרְשָׁעִים הָיוּ {כד}, וּקְנָסָא קָנְסוּ רַבָּנָן לַבָּנִים לְקַיֵּם לַבַּת מַתְּנַת אִמָּן, אַף עַל פִּי שֶׁלֹּא הָיָה מִן הַדִּין: <B>אֲבָל לֹא בְחֹל.</B> אִם חִלֵּק הַשְּׁכִיב מְרַע נְכָסָיו בְּחֹל: <B>קַל וָחֹמֶר בְּחֹל.</B> וְכֵן הֲלָכָה דְּבֵין בְּחֹל בֵּין בְּשַׁבָּת דִּבְרֵי שְׁכִיב מְרַע כִּכְתִיבֵי וְכִמְסִירֵי דָּמוּ {כז}, וְאֵין צָרִיךְ קִנְיָן. וְאִם שָׁאַל שֶׁיִּקְנוּ מִיָּדוֹ, קוֹנִים מִמֶּנּוּ בֵּין בְּחֹל בֵּין בְּשַׁבָּת {כח}, כְּדֵי שֶׁלֹּא תִּטָּרֵף דַּעְתּוֹ עָלָיו: 

~ פרק ט - משנה ח
<B>יוֹרְשֵׁי הָאָב אוֹמְרִים הַבֵּן מֵת רִאשׁוֹן.</B> וְאֵין בַּעַל חוֹב יָכוֹל לְהִפָּרַע מִנְּכָסִים אֵלּוּ, שֶׁהֲרֵי לֹא זָכָה בָּהֶן הַבֵּן מֵעוֹלָם: <B>וּבַעֲלֵי הַחוֹב אוֹמְרִים הָאָב מֵת רִאשׁוֹן.</B> וְנָפְלוּ הַנְּכָסִים שָׁעָה אַחַת לִפְנֵי הַבֵּן, וְהֵם מְשֻׁעְבָּדִים לִכְתֻבַּת אִשָּׁה וּבַעֲלֵי חוֹב: <B>בֵּית שַׁמַּאי אוֹמְרִים יַחֲלֹקוּ.</B> דְּקָסָבְרֵי בֵּית שַׁמַּאי שְׁטָר הָעוֹמֵד לִגָּבוֹת כְּגָבוּי דָּמֵי, וְיוֹרְשֵׁי הָאָב וּבַעֲלֵי חוֹב תַּרְוַיְהוּ מֻחְזָקִים הֲווּ, לְפִיכָךְ יַחֲלֹקוּ: <B>וּבֵית הִלֵּל אוֹמְרִים.</B> הַנְּכָסִים בְּחֶזְקַת יוֹרְשִׁים וּכְמֻחְזָקִים דָּמוּ {לג} וּבַעַל חוֹב בָּא לְהוֹצִיא מִיָּדָם וְעָלָיו לְהָבִיא רְאָיָה שֶׁהָאָב מֵת תְּחִלָּה: 

~ פרק ט - משנה ט
<B>הָאִשָּׁה מֵתָה תְּחִלָּה.</B> וְאֵין לְיוֹרְשֵׁי הָאִשָּׁה כְּלוּם, שֶׁהֲרֵי הַבַּעַל מֵת אַחֲרוֹן וְיָרַשׁ אֶת אִשְׁתּוֹ: <B>נְכָסִים בְּחֶזְקָתָן.</B> נִכְסֵי צֹאן בַּרְזֶל בְּחֶזְקָתָן. וְלֹא פֵּרְשׁוּ בֵּית הִלֵּל אִי בְּחֶזְקַת יוֹרְשֵׁי הָאִשָּׁה שֶׁהָיוּ שֶׁלָּהּ {לו}, אִי בְּחֶזְקַת יוֹרְשֵׁי הַבַּעַל שֶׁאַחֲרָיוּתָן עָלָיו, הִלְכָּךְ נִכְסֵי צֹאן בַּרְזֶל יַחֲלֹקוּ: <B>וּכְתֻבַּת אִשָּׁה.</B> שֶׁהֵם מָנֶה מָאתַיִם וְתוֹסֶפֶת, נִשְׁאָרִים בְּחֶזְקַת יוֹרְשֵׁי הַבַּעַל {לז}: <B>וּנְכָסִים הַנִּכְנָסִים וְיוֹצְאִים עִמָּהּ.</B> הֵם נִכְסֵי מְלוֹג. שֶׁבְּשָׁעָה שֶׁהִיא נִכְנֶסֶת נִכְנָסִים עִמָּהּ וּכְשֶׁהִיא יוֹצֵאת יוֹצְאִים עִמָּהּ, אִם פָּחֲתוּ פָּחֲתוּ לָהּ וְאִם הוֹתִירוּ הוֹתִירוּ לָהּ. אֵלּוּ הֵם בְּחֶזְקַת יוֹרְשֵׁי הָאִשָּׁה. נִמְצְאוּ יוֹרְשֵׁי הָאִשָּׁה נוֹטְלִים כָּל נִכְסֵי מְלוֹג וַחֲצִי נִכְסֵי צֹאן בַּרְזֶל:

<B>שְׁטָרֵי אֲרִיסוּת.</B> הַיּוֹרֵד לְקַרְקַע לְעָבְדָהּ וּלְשָׁמְרָהּ לְמֶחֱצָה לִשְׁלִישׁ וְלִרְבִיעַ: <B>קַבְּלָנוּת.</B> בְּכָךְ וְכָךְ כּוֹרִין לְשָׁנָה, בֵּין עָשְׂתָה בֵּין לֹא עָשְׂתָה: <B>שְׁטָרֵי בֵרוּרִין.</B> זֶה בּוֹרֵר לוֹ אֶחָד שֶׁיָּדוּן לוֹ וְזֶה בּוֹרֵר לוֹ אֶחָד, וְכוֹתְבִין בִּכְתָב, פְּלוֹנִי בֵּרֵר דַּיָּן פְּלוֹנִי וְטַעֲנוֹתָיו כָּךְ וְכָךְ, כְּדֵי שֶׁלֹּא יִהְיוּ חוֹזְרִים וְטוֹעֲנִים: <B>כּוֹתְבִים שְׁנַיִם.</B> לְכָל אֶחָד מִבַּעֲלֵי דִּינִים שְׁטָר בִּפְנֵי עַצְמוֹ שֶׁיִּהְיוּ טַעֲנוֹתָיו סְדוּרִים בּוֹ. וְאֵין הֲלָכָה כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל, אֶלָּא בִּשְׁטָר אֶחָד כּוֹתְבִין טְעָנוֹת שֶׁל שְׁנֵי בַּעֲלֵי דִּינִים, וְשֶׁפְּלוֹנִי בֵּרֵר פְּלוֹנִי דַּיָּן שֶׁיָּדוּן לוֹ וּפְלוֹנִי בֵּרֵר פְּלוֹנִי דַּיָּן אַחֵר שֶׁיָּדוּן לוֹ: 

~ פרק י - משנה ה
<B>וְהִשְׁלִישׁ אֶת שְׁטָרוֹ.</B> הַמַּלְוֶה וְהַלֹּוֶה מָסְרוּ הַשְּׁטָר בְּיַד שָׁלִישׁ. שֶׁטֹּרַח הָיָה לָהֶם לִכְתֹּב שׁוֹבֵר וְסָמְכוּ עַל הַשָּׁלִישׁ: <B>רַבִּי יוֹסֵי אוֹמֵר יִתֵּן.</B> דִּסְבִירָא לֵיהּ אַסְמַכְתָּא קַנְיָא. הַמַּבְטִיחַ לַחֲבֵרוֹ דָּבָר עַל מְנָת שֶׁיַּעֲשֶׂה לוֹ דָּבָר לֶעָתִיד וְסוֹמֵךְ בְּלִבּוֹ בִּשְׁעַת הַתְּנַאי שֶׁיּוּכַל לְקַיֵּם וּכְשֶׁהִגִּיעַ הַזְּמַן אֵינוֹ יָכוֹל לְקַיֵּם, זוֹ קְרוּיָה אַסְמַכְתָּא, וּלְדִבְרֵי רַבִּי יוֹסֵי קַנְיָא. וּלְעִנְיַן פְּסַק הֲלָכָה, אַסְמַכְתָּא לֹא קַנְיָא, אֶלָּא אִם כֵּן קָנוּ מִיָּדוֹ בְּבֵית דִּין חָשׁוּב, וְהִתְפִּיס זְכֻיּוֹתָיו בְּאוֹתוֹ בֵּית דִּין שֶׁמָּסַר שְׁטָרוֹתָיו וּרְאָיוֹתָיו, וְאָמַר אִי לֹא אֲתִינָא מִכָּאן וְעַד שְׁלֹשִׁים יוֹם לִבָּטְלָן זְכֻיּוֹתַי. וְרַבּוֹתַי מְפָרְשִׁים שֶׁכָּל בֵּית דִּין שֶׁבְּקִיאִים וְיוֹדְעִים בְּדִינֵי אַסְמַכְתָּא נִקְרָאִים בֵּית דִּין חָשׁוּב לְעִנְיָן זֶה. אֲבָל הָרַמְבַּ"ם אוֹמֵר שֶׁאֵין בֵּית דִּין חָשׁוּב אֶלָּא בֵּית דִּין הַסָּמוּךְ בְּאֶרֶץ יִשְׂרָאֵל: 

~ פרק י - משנה ו
<B>מִי שֶׁנִּמְחַק שְׁטַר חוֹבוֹ.</B> וְיֵשׁ עֵדִים שֶׁרָאוּ כְּשֶׁנִּמְחַק מֵעַצְמוֹ אוֹ נִטַּשְׁטֵשׁ עַל יְדֵי מַיִם: <B>מַעֲמִיד עָלָיו עֵדִים.</B> שֶׁיּוֹדְעִים מֵאֵימָתַי נִכְתַּב וּמֶה הָיָה כָּתוּב בּוֹ, וְהֵן עוֹשִׂין לוֹ קִיּוּם בִּפְנֵי בֵּית דִּין וְכוֹתְבִין לוֹ כָּל מַה שֶּׁהָיָה כָּתוּב בְּאוֹתוֹ שְׁטָר: <B>יַחֲלִיף.</B> יִקְרַע אוֹתוֹ שְׁטָר וְיִכְתְּבוּ לוֹ אַחֵר כְּפִי הַחֶשְׁבּוֹן הַנִּשְׁאָר: <B>נִמְצָא זֶה.</B> הַלֹּוֶה צָרִיךְ לִשְׁמֹר שׁוֹבְרוֹ מִן הָעַכְבָּרִים {ל}, שֶׁאִם יֹאבַד, יִגְבֶּה הַמַּלְוֶה אֶת כָּל חוֹבוֹ: <B>אָמַר רַבִּי יוֹסֵי וְכֵן יָפֶה לוֹ.</B> לַמַּלְוֶה, שֶׁיְּהֵא הַלֹּוֶה צָרִיךְ לִשְׁמֹר שׁוֹבְרוֹ וִימַהֵר לִפְרֹעַ לוֹ, וְלֹא יֵרַע כֹּחוֹ שֶׁל מַלְוֶה לִכְתֹּב שְׁטָר אַחֵר, שֶׁהָיָה זְמַן חוֹבוֹ מֻקְדָּם וְיָשׁוּב עַתָּה מְאֻחָר וְלֹא יוּכַל לִטְרֹף לָקוֹחוֹת אֶלָּא מִזְּמַן הַשְּׁטָר הַשֵּׁנִי. וַהֲלָכָה כְּרַבִּי יוֹסֵי שֶׁכּוֹתְבִין שׁוֹבֵר: 

~ פרק י - משנה ז
<B>קַח לְךָ עֲבָדִים.</B> שֶׁיְּחַמְּמוּ לְךָ הַמֶּרְחָץ. דִּכְמוֹ שֶׁהִנִּיחָהּ אָבִינוּ כֵּן תִּהְיֶה לְעוֹלָם. וְאַף עַל גַּב דְּאִית לָן בְּפֶרֶק קַמָּא דִּבְמִידֵי דְּלֵית בָּהּ דִּין חֲלֻקָּה יָכוֹל לְמֵימַר גּוֹד אוֹ אֱגוֹד, כְּלוֹמַר מְכֹר לִי חֶלְקְךָ אוֹ אֲנִי אֶמְכֹּר לְךָ חֶלְקִי, שָׁאנֵי הָכָא, שֶׁאֵין הֶעָנִי יָכוֹל לוֹמַר אֱגוֹד, שֶׁאֵין לוֹ בַּמֶּה לִקְנוֹת {לא}: <B>אֵינָן יְכוֹלִין לְהוֹצִיא שְׁטָר חוֹב זֶה עַל זֶה.</B> דְּכָל אֶחָד מֵהֶם יָכוֹל לִטְעֹן זֶה הַשְּׁטָר שֶׁבְּיָדְךָ אֲנִי הֶחֱזַרְתִּי לְךָ כְּשֶׁפָּרַעְתָּ לִי הַמָּעוֹת שֶׁהִלְוִיתִיךָ {לג}: <B>וְלֹא אַחֵר יָכוֹל לְהוֹצִיא שְׁטָר חוֹב עֲלֵיהֶם.</B> דְּכָל אֶחָד מָצֵי מַדְּחֵי לֵיהּ לְגַבֵּי חַבְרֵיהּ: <B>יְשַׁלֵּשׁוּ.</B> יִכְתֹּב שֵׁם אֲבִי אָבִיו: <B>וְאִם הָיוּ מְשֻׁלָּשִׁים.</B> שֶׁשְּׁמוֹתָן וּשְׁמוֹת אֲבוֹתֵיהֶן וַאֲבוֹת אֲבוֹתֵיהֶן שָׁוִין: <B>יִכְתְּבוּ סִימָן.</B> פְּלוֹנִי, שֶׁהוּא גִּיחוֹר {לה} אוֹ אָרֹךְ אוֹ גּוּץ. וְאִם הֵן שָׁוִין בַּסִּימָנִין, יִכְתְּבוּ כֹּהֵן. אִם אֶחָד כֹּהֵן וְאֶחָד יִשְׂרָאֵל: <B>נִמְצָא.</B> לְלֹוֶה אֶחָד שָׁם שְׁנֵי שְׁטָרוֹת מִשְּׁתֵּי הַלְוָאוֹת שֶׁלָּוָה מִמֶּנּוּ: <B>הַקָּטָן אֵינוֹ פָרוּעַ.</B> דִּשְׁטָר אֶחָד בֵּין שְׁטָרוֹתָיו קָאָמַר, וְלֹא שְׁנַיִם: <B>לֹא יִפָּרַע מִן הֶעָרֵב.</B> תְּחִלָּה, עַד שֶׁיִּתְבַּע אֶת הַלֹּוֶה לַדִּין וִיחַיְּבוּ אוֹתוֹ בֵּית דִּין, וְאִם אֵין לוֹ מַה לְּשַׁלֵּם אָז יִפָּרַע מִן הֶעָרֵב: <B>רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר אִם יֵשׁ נְכָסִים לַלֹּוֶה.</B> לֹא יִפָּרַע מִן הֶעָרֵב. לַאו מִכְּלָל דְּתַנָּא קַמָּא סָבַר אֲפִלּוּ אִם יֵשׁ נְכָסִים לַלֹּוֶה יִפָּרַע מִן הֶעָרֵב, אֶלָּא מַתְנִיתִין חֲסוּרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, הַמַּלְוֶה אֶת חֲבֵרוֹ עַל יְדֵי עָרֵב לֹא יִפָּרַע מִן הֶעָרֵב, וְאִם אָמַר עַל מְנָת שֶׁאֶפָּרַע מִמִּי שֶׁאֶרְצֶה, יִפָּרַע מִן הֶעָרֵב. בַּמֶּה דְּבָרִים אֲמוּרִים, כְּשֶׁאֵין נְכָסִים לַלֹּוֶה {מ}, אֲבָל יֵשׁ נְכָסִים לַלֹּוֶה לֹא יִפָּרַע מִן הֶעָרֵב. וְקַבְּלָן, אַף עַל פִּי שֶׁיֵּשׁ נְכָסִים לַלֹּוֶה יִפָּרַע מִן הַקַּבְּלָן. רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר, אֶחָד עָרֵב וְאֶחָד קַבְּלָן, אִם יֵשׁ נְכָסִים לַלֹּוֶה לֹא יִפָּרַע מֵהֶן. וְאֵין הֲלָכָה כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל. עָרֵב, הוּא שֶׁאוֹמֵר תֵּן לוֹ וַאֲנִי עָרֵב. קַבְּלָן, הוּא שֶׁאוֹמֵר תֵּן לוֹ וַאֲנִי נוֹתֵן לְךָ {מא}: <B>הֶעָרֵב לָאִשָּׁה בִּכְתֻבָּתָהּ.</B> וְאֵין לַבַּעַל נְכָסִים, וְצָרִיךְ הֶעָרֵב לִפְרֹעַ כְּתֻבָּתָהּ, לֹא יִפְרַע הַכְּתֻבָּה עַד שֶׁיַּדִּירֶנָּה הַבַּעַל תְּחִלָּה הֲנָאָה עַל דַּעַת רַבִּים, נֶדֶר שֶׁאֵין לוֹ הַתָּרָה, שֶׁלֹּא יוּכַל לְהַחֲזִיר אוֹתָהּ, דְּחַיְשִׁינַן שֶׁמָּא דַּעְתּוֹ לְהַחֲזִירָהּ וְלֶאֱכֹל כְּתֻבָּתָהּ לְאַחַר שֶׁתִּגְבֶּנָּה מִן הֶעָרֵב. וּלְעִנְיַן דִּינָא, עָרֵב דִּכְתֻבָּה לֹא מִשְׁתַּעְבֵּד וְאֵינוֹ חַיָּב לִפְרֹעַ, וַאֲפִלּוּ אֵין לוֹ נְכָסִים לַבַּעַל {מג}. מַאי טַעְמָא, דְּמִצְוָה עָבַד וְלָאו מִידֵי חִסְּרָהּ {מד}. וְאִי עָרֵב כְּתֻבָּתָא דִּבְרֵיהּ הוּא, מִשְׁתַּעְבֵּד, דְּאַבָּא לְגַבֵּיהּ בְּרֵיהּ שַׁעְבּוֹדֵי שַׁעְבֵּד נַפְשֵׁיהּ. וְקַבְּלָן דִּכְתֻבָּה מִשְׁתַּעְבֵּד {מה}, וִיכוֹלָה הָאִשָּׁה שֶׁתִּתְבַּע מִמֶּנּוּ תְּחִלָּה, וַאֲפִלּוּ יֵשׁ לוֹ נְכָסִים לַבַּעַל. וְהוּא שֶׁיַּדִּירֶנָּה הַבַּעַל הֲנָאָה תְּחִלָּה עַל דַּעַת רַבִּים: 

~ פרק י - משנה ח
<B>הַמַּלְוֶה אֶת חֲבֵרוֹ בִּשְׁטָר.</B> אַף עַל פִּי שֶׁאֵין כָּתוּב בּוֹ אַחֲרָיוּת נְכָסִים גּוֹבֶה מִנְּכָסִים מְשֻׁעְבָּדִים, דְּקַיְמָא לָן אַחֲרָיוּת טָעוּת סוֹפֵר הוּא אִם לֹא כְּתָבוֹ בִּשְׁטָר, וַהֲרֵי הוּא כְּאִלּוּ נִכְתַּב: <B>הוֹצִיא עָלָיו כְּתַב יָדוֹ שֶׁהוּא חַיָּב לוֹ.</B> וְאֵין שָׁם עֵדוּת אַחֵר: <B>גּוֹבֶה מִנְּכָסִים בְּנֵי חוֹרִין.</B> וְלֹא מִמְּשֻׁעְבָּדִים. דְּכֵיוָן דְּאֵין בּוֹ עֵדִים לֵית לֵיהּ קָלָא וְלֹא יָדְעֵי בֵּיהּ לָקוֹחוֹת כִּי הֵיכִי דְּלִזְדַּהֲרוּ: <B>אַחַר חִתּוּם שְׁטָרוֹת.</B> אַחַר שֶׁחָתְמוּ הָעֵדִים בִּשְׁטָר, כָּתַב אֲנִי פְּלוֹנִי בַּר פְּלוֹנִי עָרֵב: <B>גּוֹבֶה.</B> מִן הֶעָרֵב, מִבְּנֵי חוֹרִין בִּלְבַד. דְּכֵיוָן דְּלֹא חֲתִימֵי סַהֲדֵי עַל הָעַרְבוּת הַוְיָא לָהּ מִלְוָה עַל פֶּה: <B>אָמַר לוֹ בֶּן נַנָּס וְכוּ'.</B> דְּקָסָבַר כָּל עָרֵב שֶׁאַחַר מַתַּן מָעוֹת לֹא הָוֵי עָרֵב, שֶׁלֹּא עַל אֱמוּנַת הֶעָרֵב וְהַבְטָחָתוֹ הִלְוָהוּ. וְאַף עַל פִּי שֶׁקִּלֵּס רַבִּי יִשְׁמָעֵאל אֶת בֶּן נַנָּס, הֲלָכָה כְּרַבִּי יִשְׁמָעֵאל. וְעָרֵב דִּלְאַחַר מַתַּן מָעוֹת בָּעֵי קִנְיָן, וְאִי לֹא לֹא מִשְׁתַּעְבֵּד {מז}. וְשֶׁלִּפְנֵי מַתַּן מָעוֹת לֹא בָּעֵי קִנְיָן:
