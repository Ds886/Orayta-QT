~ פרק א - משנה א

{א} כְּרַבִּי עֲקִיבָא מִשְׁנָה ה' פֶּרֶק ג'. גְּמָרָא: {ב} שֶׁאֵין הַכָּתוּב מְדַבֵּר אֶלָּא בְטֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו, כִּדְמוֹכְחֵי לַהּ בַּגְּמָרָא: {ג} וּדְקָרֵי לְהוּ יְדִיעוֹת, פֵּרֵשׁ הָרַ"ב בְּרֵישׁ פֶּרֶק ב': {ד} כְּרַבִּי יִשְׁמָעֵאל דְּסוֹף פֶּרֶק ב'. גְּמָרָא: {ה} עַיֵּן מַה שֶּׁכָּתַבְתִּי בְרֵישׁ פֶּרֶק קַמָּא דְשַׁבָּת. וְהָא דְהָתָם תָּנָא שְׁתַּיִם שֶׁהֵן אַרְבַּע בִּפְנִים כוּ' וּשְׁתַּיִם כוּ' בַּחוּץ וְאִלּוּ הָכָא לֹא תָנֵי אֶלָּא שְׁתַּיִם שֶׁהֵן אַרְבַּע, מְתָרְצִים בַּגְּמָרָא, הָתָם דְּעִקַּר שַׁבָּת תָּנָא חִיּוּבֵי וּפְטוֹרֵי, הָכָא חִיּוּבֵי תָּנָא וּפְטוֹרֵי לֹא תָּנֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ב

{ו} <B> בַּתְּחִלָּה כוּ'.</B> בַּגְּמָרָא יָלְפִינַן לְהוּ וּמַסִּיק דִּידִיעַת בֵּית רַבּוֹ (שֶׁלָּמַד בְּבֵית רַבּוֹ שֶׁהַנּוֹגֵעַ בְּטֻמְאָה טָמֵא) שְׁמָהּ יְדִיעָה. פְּרָט לְנִשְׁבָּה בֵּין הַנָּכְרִים: {ז} <B> בִּפְנִים.</B> שֶׁהֲרֵי הוּא אוֹמֵר בְּשָׂעִיר זֶה וְכִפֶּר עַל הַקֹּדֶשׁ, מִטֻּמְאָה שֶׁבַּקֹּדֶשׁ. יָכוֹל עַל כָּל טֻמְאָה שֶׁבַּקֹּדֶשׁ יְהֵא שָׂעִיר זֶה מְכַפֵּר, תַּלְמוּד לוֹמַר וּמִפִּשְׁעֵיהֶם לְכָל חַטֹּאתָם, חֲטָאִים דּוּמְיָא דִפְשָׁעִים, מַה פְּשָׁעִים שֶׁאֵינָן בְּנֵי קָרְבָּן, אַף חֲטָאִים שֶׁאֵינָן בְּנֵי קָרְבָּן. וּמִנַּיִן לְיֵשׁ בָּהּ יְדִיעָה בַתְּחִלָּה, פֵּרֵשׁ הָרַ"ב מִשְׁנָה דִלְקַמָּן: {ח} שְׁלֹשָׁה שְׂעִירִים נַעֲשִׂין לַצִּבּוּר בְּיוֹם הַכִּפּוּרִים, אֶחָד מַתַּן דָּמוֹ לִפְנִים, וַחֲבֵרוֹ מִשְׁתַּלֵּחַ לַעֲזָאזֵל, וְאֶחָד קָרֵב בַּמִּזְבֵּחַ הַחִיצוֹן בְּמוּסָפִין. וְהוּא כָתוּב בְּפָרָשַׁת פִּינְחָס עִם שְׁאָר כָּל מוּסְפֵי הַמּוֹעֲדוֹת. רַשִׁ"י: {ט} <B> וְיוֹם הַכִּפּוּרִים.</B> הוּא עַצְמוֹ מוֹעִיל עִם הַקָּרְבָּנוֹת. שֶׁנֶּאֱמַר [וַיִּקְרָא כג] כִּי יוֹם כִּפֻּרִים הוּא. רַשִׁ"י: {י} שֶׁאַף עַל פִּי שֶׁאֵינוֹ יוֹדֵעַ שֶׁחָטָא, יֵשׁ לוֹ לִדְאֹג. שֶׁכָּל הַשּׁוֹגְגִים צְרִיכִין כַּפָּרָה לִכְשֶׁיֵּדְעוּ, אַלְמָא קֹדֶם יְדִיעָה עֲנוּשִׁים הֵן. רַשִׁ"י: {יא} וַחֲדָא מִתַּרְתֵּי נָקַט. וְהוּא הַדִּין שֶׁנִּכְנַס לַמִּקְדָּשׁ: 

~ פרק א - משנה ג

{יב} לְמַסְקָנָא דִגְמָרָא שֶׁכָּתַבְתִּי בְרֵישׁ מַתְנִיתִין דִּלְעֵיל, לֹא מַשְׁכַּחַת אֶלָּא בְתִינוֹק שֶׁנִּשְׁבָּה בֵּין הַנָּכְרִים: {יג} וְאִם תֹּאמַר כֵּיוָן שֶׁהֻקְּשׁוּ זֶה לָזֶה יְכַפֵּר כָּל אֶחָד אַדִּידֵיהּ וְאַדְּחַבְרֵיהּ. וְנָפְקָא מִנַּהּ לְטֻמְאָה שֶׁאֵרַע בֵּין זֶה לָזֶה, וְאִי נַמִּי אִי לֹא עֲבַד חִיצוֹן. מְשָׁנֵי בַגְּמָרָא אָמַר קְרָא אַחַת, כַּפָּרָה אַחַת מְכַפֵּר, וְאֵינוֹ מְכַפֵּר שְׁתֵּי כַפָּרוֹת. אָמַר קְרָא אַחַת בַּשָּׁנָה, כַּפָּרָה זוֹ לֹא תְהֵא אֶלָּא אַחַת בַּשָּׁנָה: 

~ פרק א - משנה ד

{יד} גְּמָרָא, וּנְכַפֵּר נַמִּי אֲפִלּוּ בִשְׁאָר עֲבֵרוֹת (כָּל חֵטְא שֶׁלֹּא נוֹדַע). תָּנָא דְבֵי רַבִּי יִשְׁמָעֵאל הוֹאִיל וְזֶה בָא בִזְמַן קָבוּעַ (שְׂעִיר יוֹם הַכִּפּוּרִים דְּחִיצוֹן) וְזֶה בָא בִזְמַן קָבוּעַ, מַה זֶּה אֵינוֹ מְכַפֵּר אֶלָּא עַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו אַף כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
{טו} בַּגְּמָרָא, עֲצֶרֶת [וְיוֹם הַכִּפּוּרִים] דְּלֹא כְתִיב בְּהוּ וּשְׂעִיר, מְנָלָן. אֶלָּא אָמַר רַבִּי יוֹנָה אָמַר קְרָא אֵלֶּה תַּעֲשׂוּ לַה' בְּמוֹעֲדֵיכֶם, הֻקְּשׁוּ כָל הַמּוֹעֲדוֹת כֻּלָּן זֶה לָזֶה. וְהָא רֹאשׁ חֹדֶשׁ לָאו מוֹעֵד הוּא, אִבְּרָא כוּ' רֹאשׁ חֹדֶשׁ נַמִּי אִקְּרֵי מוֹעֵד, דִּכְתִיב [אֵיכָה א] קָרָא עָלַי מוֹעֵד כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טז} <B> הָרְגָלִים.</B> דְּאִית לֵיהּ נַמִּי הֶקֵּשָׁא דִבְמוֹעֲדֵיכֶם דְּהֻקְּשׁוּ כֻלָּן לִדְרֹאשׁ חֹדֶשׁ, מַה דְּרֹאשׁ חֹדֶשׁ בְּמִלְּתָא דְקֹדֶשׁ מְכַפֵּר, אַף דִּרְגָלִים כוּ'. וְכִי תֵימָא נִיכַפְּרוּ אַדְּרֹאשׁ חֹדֶשׁ, וְאֹתָהּ נָתַן לָכֶם לָשֵׂאת אֶת עֲוֹן הָעֵדָה וְגוֹ', אֹתָהּ נוֹשֵׂא עָוֹן, וְאֵין אַחֵר נוֹשֵׂא עָוֹן כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} <B> עַל הַטָּהוֹר כוּ'.</B> דְּאָמַר קְרָא וְאֹתָהּ נָתַן לָכֶם לָשֵׂאת אֶת עֲוֹן הָעֵדָה, וְהַאי קְרָא בְּשָׂעִיר דְּרֹאשׁ חֹדֶשׁ כְּתִיב, וְיָלֵיף עָוֹן עָוֹן מִצִּיץ, מַה לְּהַלָּן טֻמְאַת בָּשָׂר, שֶׁאֵין הַצִּיץ מְרַצֶּה עַל טֻמְאַת הַגּוּף אֶלָּא עַל טֻמְאַת חֵלֶב דָּם וּבָשָׂר שֶׁקָּרַב בְּטֻמְאָה, אַף כָּאן טֻמְאַת בָּשָׂר כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יח} <B> שָׁוָה.</B> דְּיָלֵיף הֶקֵּשָׁא דְמוֹעֲדֵיכֶם אַף לַחִיצוֹן דְּיוֹם הַכִּפּוּרִים. דַּעֲדִיפָא לֵיהּ הַךְ הֶקֵּשָׁא מֵהֶקֵּשָׁא דְּלַפְּנִימִי. תּוֹסָפוֹת:
{יט} שֶׁהוּא אֵינוֹ מְכַפֵּר כַּפָּרָתָן, כַּפָּרָה אַחַת מְכַפֵּר וְאֵינוֹ מְכַפֵּר שְׁתֵּי כַפָּרוֹת. וְהֵן אֵינָן מְכַפְּרִין כַּפָּרָתוֹ, אַחַת בַּשָּׁנָה, כַּפָּרָה זוֹ לֹא תְהֵא אֶלָּא אַחַת בַּשָּׁנָה. גְּמָרָא:
{כ} מִשּׁוּם דִּלְרַבִּי שִׁמְעוֹן אַלִּימָא קֻשְׁיָא, דַּחֲלוּקִים הֵם שְׂעִירֵי רָאשֵׁי חֳדָשִׁים גַּם מִשְּׂעִירֵי הָרְגָלִים, לְהָכִי פָּרֵיךְ לֵיהּ. וְהוּא הַדִּין נַמִּי דַּהֲוָה מָצֵי לְמִפְרַךְ לְתַנָּא קַמָּא מִשְּׂעִירֵי יוֹם הַכִּפּוּרִים עִם שְׁאָר הַשְּׂעִירִים. וּמִסְתַּבֵּר נַמִּי דְּמַה שֶּׁכָּתַב הָרַ"ב וְכֵן רַשִׁ"י, רַבִּי מֵאִיר הוּא דְּקָאָמַר לֵיהּ, לָאו דַּוְקָא כוּ', אֶלָּא חֲכָמִים הֵשִׁיבוּ לוֹ לְרַבִּי שִׁמְעוֹן אַלִּיבָּא דְרַבִּי מֵאִיר: 

~ פרק א - משנה ה

{כא} <B> מוֹסִיף כוּ'.</B> דְּלֹא דָרֵישׁ אֹתָהּ נוֹשֵׂא עָוֹן וְאֵין אַחֵר כוּ'. וּסְבִירָא לֵיהּ נַמִּי דְּאִתַּקּוּשׁ רְגָלִים לַשָּׂעִיר הַחִיצוֹן כוּ'. תּוֹסָפוֹת: {כב} <B> מוֹסִיף כוּ'.</B> וְדִרְגָלִים לֹא מְכַפְּרֵי אַדְּיוֹם הַכִּפּוּרִים, דִּסְבִירָא לֵיהּ דְּרָשָׁא דְאַחַת בַּשָּׁנָה. וּדְיוֹם הַכִּפּוּרִים מוֹסִיף, דְּאָמַר קְרָא וְכִפֶּר אַהֲרֹן עַל קַרְנֹתָיו אַחַת בַּשָּׁנָה, קַרְנוֹתָיו דְּמִזְבֵּחַ הַפְּנִימִי הוּא דְאֵינוֹ מְכַפֵּר שְׁתֵּי כַפָּרוֹת, הָא דְחִיצוֹן אֲפִלּוּ שְׁתֵּי כַפָּרוֹת. גְּמָרָא:

{כג} <B> וְעַל זְדוֹן כוּ'.</B> אַף עַל גַּב דְּאָמְרִינַן אַחַת כַּפָּרָה אַחַת כוּ'. כְּבָר כָּתְבוּ הַתּוֹסָפוֹת דְּאֵין זֶה שְׁתֵּי כַפָּרוֹת אֶלָּא מַה שֶּׁקָּרְבָּן אַחֵר מְכַפֵּר. אִי נַמִּי, כַּפָּרָה אַחַת לְשׁוֹגֵג וְאַחַת לְמֵזִיד: {כד} וְכֵן הוּא אוֹמֵר [מְלָכִים ב ג] מֶלֶךְ מוֹאָב פָּשַׁע בִּי. וְאוֹמֵר [שָׁם ח] אָז תִּפְשַׁע לִבְנָה בָּעֵת הַהִיא. גְּמָרָא:

{א} רַשִׁ"י. מַשְׁמַע שֶׁפֵּרֵשׁ כֻּלַּהּ מַתְנִיתִין בִּידִיעָה גְמוּרָה. שֶׁפֵּרֵשׁ וְיָדַע שֶׁנִּטְמָא כוּ'. וְהַמְדַקְדֵּק יָכוֹל לְפָרֵשׁ כֻּלַּהּ מַתְנִיתִין בִּידִיעַת בֵּית רַבּוֹ. תּוֹסָפוֹת. וְעַיֵּן פֶּרֶק קַמָּא מִשְׁנָה ג'. וְעוֹד דַּע, שֶׁעִם זֶה שֶׁצָּרִיךְ שֶׁתְּהֵא לוֹ יְדִיעָה שֶׁנִּטְמָא צָרִיךְ גַּם כֵּן שֶׁתְּהֵא לוֹ יְדִיעָה שֶׁזֶּה מִקְדָּשׁ וְשֶׁזֶּה קֹדֶשׁ:
{ב} וְהַיְנוּ טַעֲמָא נַמִּי דְּלֹא תָנֵי אַרְבַּע שֶׁהֵם שְׁמֹנֶה, וְהַיְנוּ עוֹד אַרְבַּע יְדִיעוֹת בַּסּוֹף, דְּלֹא חָשֵׁיב הָנֵי יְדִיעוֹת דְּאִיתְנְהוּ בְכָל הַתּוֹרָה כֻּלָּהּ, שֶׁאֵין אָדָם מֵבִיא חַטָּאת אֶלָּא אִם כֵּן נוֹדַע לוֹ שֶׁחָטָא, כִּדְאִיתָא בַגְּמָרָא: {ג} בְּהֶדְיָא. אֶלָּא מֵרִבּוּיָא דִּוְנֶעְלַם דָּרְשִׁינַן לַהּ בְּסוֹף פִּרְקִין:

{ה} <B>לָעֲזָרָה.</B> דְּבָעֲזָרָה נֶאֱמַר כָּרֵת לְטָמֵא הַנִּכְנָס, כִּדְאִיתָא בְּמִשְׁנָה ח' פֶּרֶק קַמָּא דְכֵלִים:
{ו} שֶׁעָלָיו נֶאֱמַר [דְּבָרִים לג] וַיְהִי בִישֻׁרוּן מֶלֶךְ. הָרַמְבַּ"ם: {ז} כְּלוֹמַר וְכֵיוָן שֶׁאָחִיו כֹּהֵן גָּדוֹל הָיָה אִם כֵּן הָיָה שָׁם אוּרִים וְתֻמִּים. וְהִקְשׁוּ הַתּוֹסָפוֹת אַף עַל גַּב דְּאַהֲרֹן הָיָה בְבִנְיַן הַמִּשְׁכָּן וְכִי נִשְׁאֲלוּ בְאוּרִים וְתֻמִּים לַעֲשׂוֹת מִשְׁכָּן, הֲלֹא קֹדֶם שֶׁנִּתְכַּהֵן אַהֲרֹן עַל יְדֵי הַמִּלּוּאִים נַעֲשָׂה הַמִּשְׁכָּן: {ח} רַשִׁ"י וְהָרַמְבַּ"ם. וּבְסַנְהֶדְרִין פֵּרֵשׁ הָרַ"ב מִשּׁוּם דְּמֹשֶׁה בִּמְקוֹם סַנְהֶדְרִין הֲוָה קָאֵי. וְטַעֲמָא דְמִסְתַּבֵּר הוּא, לְפִי שֶׁלֹּא נִמְצָא מִנּוּי שִׁבְעִים לַסַּנְהֶדְרִין כִּי אִם בְּפָרָשַׁת מִתְאוֹנְנִים, וְהָיָה זֶה אַחַר שֶׁנַּעֲשָׂה הַמִּשְׁכָּן כוּ'. וּלְפֵרוּשָׁם דְּהָכָא, צָרִיךְ לוֹמַר דְּשִׁבְעִים זְקֵנִים שֶׁנֶּאֶמְרוּ בְסוֹף פָּרָשַׁת מִשְׁפָּטִים הֵם הָיוּ מְיֻחָדִים שֶׁבָּעֵדָה לְכָל דָּבָר שֶׁבִּקְדֻשָּׁה, כְּעֵין סַנְהֶדְרִין: {ט} כְּלוֹמַר שֶׁאֵין טַעַם בַּדָּבָר לָמָּה זוֹ נֶאֱכֶלֶת וְלָמָּה זוֹ נִשְׂרֶפֶת, אֶלָּא עַל פִּי נָבִיא בְרוּחַ הַקֹּדֶשׁ עָשׂוּ כֵן. דְּלֵיכָּא לְמֵימַר דְּחִיצוֹנָה כִי אָתְיָא קִדְּשָׁה הַמָּקוֹם וְהַפְּנִימִית הִיא לֹא נִפְסֶלֶת בְּיוֹצֵא וְשֶׁלְּכָךְ נֶאֱכֶלֶת, דְּהָא בְּלֹא הָא לֹא סַגִּי: {י} בַּגְּמָרָא, מַאי גְּדוֹלוֹת, אִילֵימָא כוּ', אֶלָּא גְדוֹלוֹת שֶׁבַּתּוֹדָה, וּמַאי נִיהוּ חָמֵץ. דְּחַלּוֹת הֶחָמֵץ הָיוּ גְדוֹלוֹת מִשֶּׁל מַצָּה (כְּדִתְנַן רֵישׁ פּ"ז דִּמְנָחוֹת):
{יא} פֵּרֵשׁ רַשִׁ"י, הוּא מִזְמוֹר שֶׁל חֲנֻכַּת הַבַּיִת, לְכָךְ אוֹמֵר אוֹתוֹ. וּבַגְּמָרָא מְסַיֵּם וְשִׁיר שֶׁל פְּגָעִים עַד כִּי אַתָּה ה' מַחְסִי עֶלְיוֹן שַׂמְתָּ מְעוֹנֶךָ, וְחוֹזֵר וְאוֹמֵר מִזְמוֹר לְדָוִד בְּבָרְחוֹ [מִפְּנֵי אַבְשָׁלוֹם בְּנוֹ] ה' מָה רַבּוּ צָרָי כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} <B> וּבֵית דִּין כוּ'.</B> לְמֵימְרָא דְּבֵית דִּין קַמֵּי תוֹדָה אָזְלֵי, וְהַכְּתִיב (נְחֶמְיָה יב) וַיֵּלֵךְ אַחֲרֵיהֶם הוֹשַׁעְיָה וַחֲצִי שָׂרֵי יְהוּדָה. הָכִי קָאָמַר, בֵּית דִּין מְהַלְּכִין וּשְׁתֵּי תוֹדוֹת מְהַלְּכוֹת וּבֵית דִּין אַחֲרֵיהֶם. גְּמָרָא. כְּלוֹמַר דְּתָנֵי בֵית דִּין כוּ' וַהֲדַר מְפָרֵשׁ כֵּיצַד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} <B> הַפְּנִימִית.</B> פְּלִיגֵי תַנָּאֵי בַגְּמָרָא, חַד אָמַר זוֹ כְנֶגֶד זוֹ הָיוּ מְהַלְּכוֹת, כְּלוֹמַר שְׁנֵי כֹהֲנִים נוֹשְׂאִים אוֹתָם זֶה אֵצֶל זֶה, וְהַוְיָא הַפְּנִימִית הַךְ דְּמִקָּרְבָא אֵצֶל חוֹמָה. וְחַד אָמַר זוֹ אַחַר זוֹ, וְהָוֵי הַפְּנִימִית הַךְ דְּמִקָּרְבָא לְבֵית דִּין: {יד} <B> אֵין חַיָּבִין.</B> הָכִי קָאָמַר, הַנִּכְנָס לְמָקוֹם פְּלוֹנִי וְהַנִּכְנָס לְמָקוֹם פְּלוֹנִי וְהַנִּכְנָס כוּ' כֻּלָּם אֵין חַיָּבִין עַל הַמָּקוֹם שֶׁנִּכְנָס כָּל אֶחָד וְאֶחָד:

{א} <B> שְׁבוּעַת כוּ'.</B> שְׁבוּעָה לֹא כְתִיב בְּהֶדְיָא בַקְּרָא, אֶלָּא יָלְפִינַן מִקּוֹל, כְּמוֹ שֶׁכָּתוּב בְּרֵישׁ מַתְנִיתִין דְּסוֹף פִּרְקִין: {ב} כְּלוֹמַר, אַף עַל גַּב דְּאֵינוֹ בַר מַלְקוֹת, שֶׁנִּתָּק לַעֲשֵׂה דְהִשָּׁבוֹן: {ג} מִכְּלָל דְּמֶלֶךְ מִדְּאוֹרַיְתָא אֵינוֹ כָשֵׁר לְהָעִיד, וְהָכִי אִיתָא בְהֶדְיָא בַגְּמָרָא דִּמְמַעֲטִינַן מְשַׂחֵק בְּקֻבְיָא וְכָל שֶׁכֵּן מֶלֶךְ. וּכְלוֹמַר שֶׁהוּא מִדְּאוֹרַיְתָא. וּפֵרֵשׁ רַשִׁ"י, דִּכְתִיב (דְּבָרִים יז) שׂוֹם תָּשִׂים עָלֶיךָ מֶלֶךְ, שֶׁתְּהֵא אֵימָתוֹ עָלֶיךָ, וְהַבָּא לְהָעִיד צָרִיךְ לַעֲמֹד לִפְנֵי הַדַּיָּן וְדֶרֶךְ בִּזָּיוֹן הוּא לוֹ. וּלְפִי זֶה אַף בְּמַלְכֵי בֵית דָּוִד אָיְרִינַן, דְּאֵינָן רְאוּיִין לְהָעִיד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} <B> מִפִּי עַצְמוֹ.</B> דְּיָלֵיף לַהּ מִגְּזֵרָה שָׁוָה דְתֶחֱטָא תֶּחֱטָא מִפִּקָּדוֹן. וְסָבַר דּוּן מִנַּהּ. וּמִנַּהּ כְּלוֹמַר, דְּדָיְנִינַן מִנַּהּ לְחַיְּבוֹ אַף מִפִּי עַצְמוֹ, וּמִנַּהּ כִּי הָתָם דְּמִחַיְּבֵי בֵּין בְּבֵית דִּין בֵּין שֶׁלֹּא בְבֵית דִּין. דְּנִשְׁבַּע עַל שָׁקֶר כְּתִיב, מַשְׁמַע בֵּין בְּבֵית דִּין בֵּין שֶׁלֹּא בְבֵית דִּין. רַשִׁ"י: {ה} וְכֵן אִם עָנָה אָמֵן אַחַר הַשְּׁבוּעָה. רַשִׁ"י: {ו}  אַף עַל גַּב דְּלֹא עָנָה אָמֵן. שֶׁכֵּן הוּא אוֹמֵר וְשָׁמְעָה קוֹל אָלָה אִם לוֹא יַגִּיד וְנָשָׂא עֲוֹנוֹ: {ז} שֶׁנֶּאֱמַר וְשָׁמְעָה קוֹל אָלָה: {ח} <B> בֵּין כוּ'.</B> דִּסְבִירָא לְהוּ דּוּן מִנַּהּ וְאוֹקֵי בְאַתְרַהּ. כְּלוֹמַר, לְאַחַר שֶׁהֵבֵאתָ מִפִּי עַצְמוֹ לִשְׁבוּעַת הָעֵדוּת מִפִּקָּדוֹן, אוֹקֵי בְאַתְרַהּ בְּדִין הַכָּתוּב בִּשְׁבוּעַת הָעֵדוּת, דְּמִפִּי אֲחֵרִים שֶׁמִּשְׁפָּטוֹ דַוְקָא בְבֵית דִּין, הָכִי נַמִּי מִפִּי עַצְמוֹ שֶׁהֵבֵאתָ אֵינוֹ אֶלָּא בְבֵית דִּין: 

~ פרק ד - משנה ב

{ט} וּמִמֵּילָא שֶׁיּוֹדֵעַ לוֹ עֵדוּת. דְּאִי לָאו הָכִי אֵינָהּ אֲסוּרָה. וְהַיְנוּ זְדוֹן הָעֵדוּת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} רַשִׁ"י. וְתָמְהוּ הַתּוֹסָפוֹת, דְּתֵיפּוּק לֵיהּ דִּבְקוּשְׁטָא קָא מִשְׁתַּבְעֵי. וְתֵרֵץ הָרַ"ן, דְּרַשִׁ"י הָכִי קָאָמַר, שֶׁבְּאוֹתָהּ שָׁעָה שֶׁנִּשְׁבְּעוּ אֵין אָנוּ יוֹדְעִין, יוֹדְעִין הָיוּ אֶלָּא שֶׁלֹּא הָיוּ מִתְכַּוְּנִים לָשׂוּם לֵב, אֲבָל לֹא הָיוּ צְרִיכִין יְדִיעָה אַחֶרֶת, וְנִמְצָא מַה שֶּׁנִּשְׁבְּעוּ אֵין אָנוּ יוֹדְעִים לֹא נִשְׁבְּעוּ עַל הָאֱמֶת, שֶׁאִלּוּ הָיוּ שָׂמִים לֵב בְּאוֹתָהּ שָׁעָה יוֹדְעִים הָיוּ, שֶׁהֲרֵי מִיָּד כְּשֶׁנִּתְיַשְּׁבוּ נִזְכְּרוּ בְּלֹא שׁוּם חִדּוּשׁ יְדִיעָה, וַאֲפִלּוּ הָכִי כֵּיוָן שֶׁהָיוּ סְבוּרִים לִשָּׁבַע עַל הָאֱמֶת, פְּטוּרִים, מִשּׁוּם דַּאֲנוּסִים הָיוּ כוּ'. אֲבָל וַדַּאי דְּהוּא הַדִּין וְכָל שֶׁכֵּן כְּשֶׁסְּבוּרִים שֶׁמֻּתָּרִים לְגַמְרֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ד - משנה ג

{יא} <B> שֶׁאָמְרוּ.</B> בְּלֹא שְׁבוּעָה, אֵין אָנוּ כוּ', וְאָמַר לָהֶם מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם שֶׁיְּהֵא אֱמֶת מַה שֶּׁאֲמַרְתֶּם, וְאָמְרוּ אָמֵן, חַיָּבִים. אֲבָל אִם שָׁתְקוּ אַחַר הַשְּׁבוּעָה, אֵין כָּאן קַבָּלַת שְׁבוּעָה. רַשִׁ"י. וְנִרְאֶה מִלְּשׁוֹנוֹ דְּדַוְקָא שָׁתְקוּ, אֲבָל כָּפְרוּ וְאָמְרוּ אֵין אָנוּ יוֹדְעִים כוּ' חַיָּבִים. וּמַיְרֵי בְבֵית דִּין. וּמַתְנִיתִין אֲפִלּוּ כְרַבָּנָן. וְאָמֵן לָאו דַּוְקָא, וְלֹא נָקְטוּ אֶלָּא כְדֵי לְקַצֵּר. וְכֵן כָּתְבוּ הַתּוֹסָפוֹת, וְעַיֵּן בְּסָמוּךְ (אוֹת יג): {יב} <B> חוּץ לְבֵית דִּין.</B> אָתְיָא נַמִּי כְרַבָּנָן, דִּלְרַבָּנָן נַמִּי לֹא בָעִינַן שְׁבוּעָה בְבֵית דִּין, אֶלָּא הַכְּפִירָה. דְּאַהַגָּדָה דַוְקָא קָפֵיד קְרָא דְּלֶהֱוֵי בְמָקוֹם הָרָאוּי לְהַגָּדָה. תּוֹסָפוֹת:
{יג} וְרֵישָׁא דְתָנֵי הֲרֵי אֵלּוּ חַיָּבִין, וְלֹא אָמְרִינַן דַּהֲוֵי כְפִירַת דְּבָרִים בְּעָלְמָא כֵּיוָן דִּכְבָר אָמְרוּ אֵין אָנוּ יוֹדְעִים כוּ' וְשׁוּב אֵין יְכוֹלִין לַחֲזֹר וּלְהוֹדוֹת, נִרְאֶה לִי מִשּׁוּם דְּלֹא קַפְּדָה הַתּוֹרָה אַחוֹזֵר וּמַגִּיד אֶלָּא בָתַר שְׁבוּעָה. דְּאִם לֹא יַגִּיד, בָּתַר וְשָׁמְעָה קוֹל אָלָה כְּתִיב. וְהַטַּעַם מִפְּנֵי שֶׁדֶּרֶךְ הָעֵדִים לְהִשָּׁמֵט שֶׁלֹּא לְהָעִיד, וּכְשֶׁאוֹמְרִים בָּתַר שְׁבוּעָה יוֹדְעִים אָנוּ, אֵין זֶה חוֹזֵר כוּ'. עוֹד נִרְאֶה לִי, דְּאֵין אָנוּ יוֹדְעִים לֹא מִקְּרֵי מַגִּיד כְּלָל, דְּהָא לֹא הִגִּידוּ כְלוּם, וְהָכָא לָאו מִשּׁוּם חוֹזֵר וּמַגִּיד אַיְתִּינַן עֲלַהּ, אֶלָּא מִשּׁוּם דְּכָפְרוּ בָתַר שְׁבוּעָה שׁוּב לֹא מָצוּ מְשַׁוּוּ נַפְשַׁיְהוּ רְשִׁיעֵי לוֹמַר יוֹדְעִים אָנוּ. וּבְהָכִי מִתָּרְצָא אַמַּאי נָקַט רֵישָׁא בְּמֻשְׁבָּע מִפִּי עַצְמוֹ, דִּלְאַשְׁמוֹעִינַן הַךְ רְבוּתָא נָקַט בְּמִפִּי עַצְמוֹ. הָרַ"ן. כְּלוֹמַר דְּאִלּוּ הֲוָה נָקַט בְּמֻשְׁבָּע מִפִּי אֲחֵרִים, לֹא הָיָה יָכוֹל לְהַתְחִיל אָמְרוּ אֵין אָנוּ יוֹדְעִים כוּ', לְפִי שֶׁשּׁוּב אַחַר כָּךְ כְּשֶׁמַּשְׁבִּיעָן אֵין רְגִילוּת לְהָשִׁיב אֶלָּא אָמֵן, וְאִם כֵּן לֹא הֲוָה יָדְעִינַן רְבוּתָא זוֹ:

{יד} גְּמָרָא. דְּאִי אֶפְשָׁר לְצַמְצֵם. וּפֵרוּשׁ תּוֹךְ כְּדֵי דִבּוּר, עַיֵּן בְּרֵישׁ פֶּרֶק ד' דְּנָזִיר. וְלֹא שֶׁיִּהְיֶה כָל הַכְּפִירָה תּוֹךְ כְּדֵי דִבּוּר אֶלָּא שֶׁתַּתְחִיל תּוֹךְ כְּדֵי דִּבּוּר. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} דְּאַף עַל גַּב דְּקָם לִשְׁבוּעָה (וְרֹב בְּנֵי אָדָם אֵין נִשְׁבָּעִין לַשֶּׁקֶר וְהָיָה מְשַׁלֵּם), דָּבָר הַגּוֹרֵם לְמָמוֹן לָאו כְּמָמוֹן דָּמֵי. גְּמָרָא: {טז} <B> הַכּוֹפֵר כוּ'.</B> גְּמָרָא הַשְׁתָּא בָּזֶה אַחַר זֶה דְתַרְוַיְהוּ קָא כָפְרֵי אָמַרְתְּ הָרִאשׁוֹן חַיָּב וְשֵׁנִי פָטוּר, כָּפַר אֶחָד וְהוֹדָה אֶחָד מִבַּעְיָא. לֹא צְרִיכָא כְּגוֹן שֶׁכָּפְרוּ שְׁנֵיהֶם וְחָזַר אֶחָד מֵהֶן וְהוֹדָה בְתוֹךְ כְּדֵי דִבּוּר, וְהָא קָמַשְׁמַע לָן דְּמַהוּ דְתֵימָא דְהָא דְרֵישָׁא תּוֹךְ כְּדֵי דִבּוּר מַהֲנֵי, הָנֵי מִלֵּי כְפִירָה וּכְפִירָה, אֲבָל כְּפִירָה וְהוֹדָאָה אֵימָא לֹא, קָא מַשְׁמַע לָן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} שֶׁהָיוּ יְכוֹלִים לוֹמַר יָדַעְנוּ כִי הָיָה שָׁם כַּת אַחֶרֶת בִּשְׁעַת תְּבִיעָתָן. הָרַמְבַּ"ם: {יח} נָשְׂאוּ שְׁתֵּי אֲחָיוֹת, וְאֵינָן כְּשֵׁרִים בְּעֵדוּת אַחַת. וְקָשֶׁה גּוֹסְסוֹת דְּקָאָמַר דְּמַשְׁמַע דְּתַרְוַיְהוּ גוֹסְסוֹת לְמַאי, דִּבְאַחַת סַגִּי. וְכָתְבוּ הַתּוֹסָפוֹת דְּמַיְרֵי כְּגוֹן שֶׁבִּשְׁעַת רְאִיַּת הָעֵדוּת הָיוּ רְחוֹקִים. דְּבָעִינַן תְּחִלָּתָן וְסוֹפָן בְּכַשְׁרוּת: {יט} וְלִמֶּדְךָ שֶׁהַגּוֹסֵס הָוֵי כְחַי לְכָל דְּבָרָיו וְאַף עַל פִּי שֶׁרֻבָּן לְמִיתָה. רַשִׁ"י וְהָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} <B> בִּשְׁתֵּיהֶן.</B> שֶׁאַחַר שֶׁמֵּתוּ הַנָּשִׁים אָז כָּפְרָה הַשְּׁנִיָּה הָעֵדוּת גַּם כֵּן. הָרַמְבַּ"ם:

{ח} דְּאִי לָאו הָכִי שֶׁלֹּא יָדְעוּ הָעֵדִים שֶׁהוֹצִיא כְּלָל, אֵין בַּעַל הַבַּיִת נֶאֱמָן בִּשְׁבוּעָה. דְּשֶׁמָּא לֹא הוֹצִיא אֶלָּא אֲבָנִים אוֹ צְרוֹרוֹת כוּ': {ט} <B> נִשְׁבָּע וְנוֹטֵל.</B> דְּקָנְסוּ חֲכָמִים לַגַּזְלָנִים, כְּדֵי שֶׁלֹּא יִרְבֶּה הֶזֵּק בְּנֵי אָדָם. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ז - משנה ג

{י} <B> הַנֶּחְבָּל.</B> קְנָס לַחוֹבֵל, כְּדֵי שֶׁלֹּא יִרְבֶּה הֶזֵּק בְּנֵי אָדָם וְהִתְגָּרוּתָם. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} דִּנְשִׁיכַת שִׁנַּיִם לָאו חִכּוּךְ כֹּתֶל הוּא. רַשִׁ"י:
{יב} וְלֹא קַשְׁיָא דְּאִם כֵּן יִתְבָּעֵהוּ לְאוֹתוֹ אַחֵר. דְּהָכִי קָאָמַר, שֶׁמָּא אַחֵר עָשָׂה עַל פִּיו דְּנֶחְבָּל, שֶׁאָמַר לוֹ לַחֲבֹל בּוֹ כְּדֵי שֶׁיִּתְבַּע לָזֶה, מִידֵי דַהֲוָה דְחָיְשִׁינַן שֶׁמָּא חָבַל בְּעַצְמוֹ. וְהָכִי אִיתָא בַגְּמָרָא: 

~ פרק ז - משנה ד

{יג} <B>הָעֵדוּת.</B> הִקְשָׁה הָרַשְׁבָּ"א זַ"ל, מַאן מוֹכָח דְּעָבַר עֲלֵיהּ בְּמֵזִיד, דִּלְמָא אִשְׁתְּלֵי וְלֹא מֵזִיד הוּא. וּלְדִידִי לֹא קַשְׁיָא, דְּכָל שֶׁהָיָה עֵדוּת קָרוֹב כָּל כָּךְ דְּאִי אֶפְשָׁר לִתְלוֹת בּוֹ שִׁכְחָה, לֹא אָמְרִינַן דִּלְמָא אִשְׁתְּלֵי. וּבְמוֹדֶה שֶׁהֵזִיד נַמִּי מַשְׁכַּחַת לַהּ, וְלֹא אָמְרִינַן בָּזֶה אֵין אָדָם מֵשִׂים עַצְמוֹ רָשָׁע, דְּהָא הוֹדָאַת מָמוֹן הוּא, וְהוֹדָאַת בַּעַל דִּין כְּמֵאָה עֵדִים דָּמֵי. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} לְהוֹצִיא שֶׁקֶר מִפִּיו. רַשִׁ"י: {טו} <B> שֶׁכְּנֶגְדּוֹ כוּ'.</B> וְלֹא אָמְרִינַן מִתּוֹךְ שֶׁאֵינוֹ יָכוֹל לִשָּׁבַע מְשַׁלֵּם, כְּמוֹ שֶׁכָּתַב הָרַ"ב פֶּרֶק ח' דְּבָבָא מְצִיעָא מִשְׁנָה ב', כְּדֵי שֶׁלֹּא יַפְקִיעוּ מִמֶּנּוּ מָמוֹן, שֶׁמִּי שֶׁהָיָה רוֹצֶה הָיָה מַלְוֶה לוֹ פְרוּטָה וְיִתְבַּע כָּל אֲשֶׁר לוֹ, וִיהֵא מוֹדֶה מִקְצָת כַּמָּה שֶׁחַיָּב, וְיִתְחַיֵּב שְׁבוּעָה וּמִתּוֹךְ שֶׁאֵינוֹ יָכוֹל לִשָּׁבַע יִקַּח כָּל אֲשֶׁר לוֹ בְּלֹא שְׁבוּעָה, לְכָךְ תִּקְּנוּ שֶׁיִּשָּׁבַע כְּנֶגְדּוֹ. מַה שֶּׁאֵין כֵּן הָתָם שֶׁזֶּה שֶׁאֵינוֹ יָכוֹל לִשָּׁבַע לֹא מֵחֲמַת שֶׁהוּא חָשׁוּד אֶלָּא עַתָּה בִתְבִיעָה זוֹ אֵינוֹ יָכוֹל לִשָּׁבַע מֵחֲמַת הַמְאֹרָע, לֹא הִצְרִיכוּ לִשָּׁבַע שֶׁכְּנֶגְדּוֹ. תּוֹסָפוֹת: {טז} וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם. וּמְסַיֵּם, וְאֵין לוֹ עַל אוֹתוֹ שֶׁתְּבָעוֹ זוּלַת חֵרֶם סְתָם. וַהֲלָכָה כְּרַבִּי מֵאִיר וְלֹא כְרַבִּי יוֹסֵי דְּאָמַר יַחֲלֹקוּ. עַד כָּאן. הָכִי מַסְּקִינַן בַּגְּמָרָא, דְּרַבִּי יוֹסֵי הוּא דְּאָמַר יַחֲלֹקוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ז - משנה ה

{יז} <B> לֹא כוּ'.</B> אֶלָּא בִזְמַן שֶׁיֵּשׁ רַגְלַיִם לַדָּבָר, שֶׁבַּעַל הַבַּיִת מוֹדֶה שֶׁאָמַר לוֹ תֵּן כוּ'. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יח} <B> לִבְנִי כוּ'.</B> וְהָא דְקָתָנֵי פוֹעֲלִים וּבָנִים, רְבוּתָא אַשְׁמְעִינַן, דְּלֹא תֵימָא הָא דְחֶנְוָנִי מְחֻיָּב שְׁבוּעָה הַיְנוּ כְשֶׁהִמְחָה פוֹעֲלִים אֶצְלוֹ שֶׁהוּא צָרִיךְ לְשַׁלֵּם נַמִּי לַפּוֹעֲלִים, אֲבָל הֵיכָא דְאֵינוֹ חַיָּב לְשַׁלֵּם אֶלָּא לַחֶנְוָנִי בִלְבַד, סָלְקָא דַעְתָּךְ שֶׁיִּטֹּל חֶנְוָנִי בְּלֹא שְׁבוּעָה, קָא מַשְׁמַע לָן. הָרַ"ן: {יט} <B> בְּסֶלַע כוּ'.</B> כְּלוֹמַר, בְּסֶלַע שֶׁאֶתֵּן לְךָ תֵּן לָהֶם מָעוֹת שֶׁיּוּכַל הַפּוֹעֵל לְהוֹצִיא אוֹתָם בִּפְרוֹטְרוֹט: {כ} <B> וְהֵן כוּ'.</B> וְלֹא הָוֵי כְאוֹמֵר הִלְוִיתַנִי וְאֵינִי יוֹדֵעַ אִם פְּרַעְתִּיךָ דְּחַיָּב לְשַׁלֵּם, דְּהָכָא כֵּיוָן דְּמַפְסִיד בַּעַל הַבַּיִת, סְבִירָא לֵיהּ לְתַנָּא קַמָּא דְּתִקְּנוּ רַבָּנָן שְׁבוּעָה שֶׁיִּשָּׁבְעוּ פוֹעֲלִים. הָרַ"ן: {כא} <B> אֵלּוּ וָאֵלּוּ.</B> בָּאִים לְבֵית דִּין לִשָּׁבַע שְׁבוּעַת שָׁוְא מִבֵּינֵיהֶם, דְּלֵית בְּהוּ דְּאֵינוֹ מְשַׁקֵּר. רַשִׁ"י: {כב} שֶׁלֹּא כוּ'. הָרַ"ן הֶעְתִּיק, אֶלָּא הוּא נוֹטֵל בִּשְׁבוּעָה, לְפִי שֶׁאֵין חִיּוּבוֹ בָרוּר. וְהֵן נוֹטְלִין בְּלֹא שְׁבוּעָה, דַּהֲוֵי לֵיהּ כְּמָנֶה הִלְוִיתַנִי וְאֵינִי יוֹדֵעַ אִם פְּרַעְתִּיךָ: 

~ פרק ז - משנה ו

{כג} <B> וְנָתַן לוֹ.</B> פֵּרוּשׁ שֶׁהִגְבִּיהָן לוֹקֵחַ וְהִנִּיחָן בִּרְשׁוּת הָרַבִּים. הַמַּגִּיד. אֲבָל לְהָרַ"ן דִּלְקַמָּן אֵין נִרְאֶה שֶׁהִגְבִּיהָן הַלּוֹקֵחַ. וְכֵן נִרְאֶה דַּעַת הָרַ"ב:
{כד} וְאֵינוֹ תוֹבֵעַ מִמֶּנּוּ אֶלָּא דְמֵיהֶן. הָרַ"ן: {כה} וְזֶה לְשׁוֹן הָרַ"ן, מִשּׁוּם דְּטַעַנְתֵּיהּ דְּחֶנְוָנִי מְעַלְּיָא בְהוּ טְפֵי, לְפִי שֶׁלִּדְבָרָיו אֵין לְבַעַל הַבַּיִת שׁוּם דְּרָרָא בַּפֵּרוֹת הַלָּלוּ, וּבַעַל הַבַּיִת מוֹדֶה לַחֶנְוָנִי שֶׁהֵן שֶׁלּוֹ אֶלָּא שֶׁנָּתַן לוֹ הַדִּינָר בִּשְׁבִילָן. הָרַ"ן:
{כו} <B> אָמַר לוֹ כוּ'.</B> פֵּרוּשׁ חֶנְוָנִי טוֹעֵן שֶׁמַּעֲשֶׂה שֶׁהָיָה כָּךְ הָיָה, שֶׁבָּא אֵלָיו בַּעַל הַבַּיִת וְאֵין בְּיָדוֹ דִינָר וְאוֹמֵר לוֹ תֵן לִי בְדִינָר פֵּרוֹת וְעַכְשָׁיו אָבִיא לְךָ אֶת הַדִּינָר וְנָתַן לוֹ אֶת הַפֵּרוֹת וְהוֹלִיכָם לְתוֹךְ בֵיתוֹ וְאַחַר כָּךְ חָזַר אֵלָיו וּפָרַע לוֹ אֶת הַדִּינָר, וּבַעַל הַבַּיִת טוֹעֵן לֹא כִי, אֶלָּא קֹדֶם שֶׁנָּתַתִּי הַדִּינָר לֹא לָקַחְתִּי פֵרוֹת מֵעוֹלָם, וְלֹא נָתַתִּי לְךָ הַדִּינָר אֶלָּא בִשְׁבִיל אֵלּוּ הַפֵּרוֹת שֶׁהֵן בִּרְשׁוּת הָרַבִּים. הָרַ"ן: {כז} <B> לַשֻּׁלְחָנִי.</B> הָכָא נַמִּי שֶׁהַמָּעוֹת צְבוּרִין וּמֻנָּחִין בִּרְשׁוּת הָרַבִּים: {כח} מִשּׁוּם דְּפֵירֵי עֲבִידֵי דְּמַרְקְבֵי, לֹא מַשְׁהֵי לְהוּ. גְּמָרָא. וְהַיְנוּ טַעֲמָא דְּרַבִּי יְהוּדָה לֹא אָמַר בַּחֶנְוָנִי בְּהַהוּא לִישְׁנָא דְּאָמַר בַּשֻּׁלְחָנִי, לְפִי שֶׁדֶּרֶךְ הַחֶנְוָנִי לְהַקִּיף אֶלָּא דְהָכָא מוֹדֶה הַחֶנְוָנִי שֶׁלֹּא מָכַר בְּהַקָּפָה. הָרַ"ן וְתוֹסָפוֹת. וּבַסֵּיפָא לֹא אָמַר כֹּל שֶׁהָאִסָּר כוּ', מִשּׁוּם דְּלִפְעָמִים בַּעַל הַבַּיִת נוֹתֵן אִסָּר בִּפְרוּטוֹת, וְלֹא שַׁיָּךְ לְמֵימַר כֹּל שֶׁהָאִסָּר כוּ'. תּוֹסָפוֹת: 

~ פרק ז - משנה ז

{כט} <B> כְּשֵׁם.</B> מִשּׁוּם דִּתְנִינְהוּ בְּפֶרֶק הַכּוֹתֵב מִשְׁנָה ט', קָתָנֵי כְּשֵׁם, וְקָאֵי אַנִּשְׁבָּעִין וְנוֹטְלִין דְּרֵישָׁא. הָרַמְבַּ"ן:
{ל} וְלֹא דָמֵי לְמִשְׁנָה ד' דְּחָזְרָה שְׁבוּעָה לִמְקוֹמָהּ דִּמְפָרְשִׁינַן שֶׁהַנִּתְבָּע יְשַׁלֵּם, דְּהָכָא הַיְנוּ טַעֲמָא דְּאָמְרִינַן כֵּיוָן דְּתוֹבֵעַ אֵינוֹ יָכוֹל לִשָּׁבַע שֶׁיַּפְסִיד, מִשּׁוּם שֶׁאֵין כָּאן פְּשִׁיעָה לַנִּתְבָּע כְּלָל. אֲבָל בְּמַתְנִיתִין דִּלְעֵיל כֵּיוָן שֶׁהַנִּתְבָּע אֵינוֹ יָכוֹל לִשָּׁבַע מִפְּנֵי פְשִׁיעָתוֹ, וּמִדִּינָא מְשַׁלֵּם, וְאֵינוֹ בְדִין שֶׁיְּהֵא חוֹטֵא נִשְׂכָּר, כְּשֶׁאֵין הַתּוֹבֵעַ יָכוֹל לִשָּׁבַע שֶׁיִּפָּטֵר. וְאַף עַל פִּי שֶׁהַתּוֹבֵעַ גַּם כֵּן מִפְּנֵי חֲשָׁדוֹ אֵינוֹ יָכוֹל לִשָּׁבַע, אֵינָהּ דּוֹמָה שְׁבוּעַת הַתַּקָּנָה בְעָלְמָא לִשְׁבוּעַת הַתּוֹרָה שֶׁמְּחַיַּבְתּוֹ לְשַׁלֵּם. הָרַ"ן:
{לא} בַּגְּמָרָא, מִשּׁוּם דְּאִיכָּא מַאן דְּפָלֵיג אַהַאי דִּינָא, הִלְכָּךְ אָמְרִינַן מַאן דַּעֲבַד וְדָן כוּ' מַאי דַּעֲבַד עֲבַד: {לב} <B>וְשֶׁלֹּא מָצִינוּ כוּ'.</B> שׁוֹבֵר עַל שְׁטָר זֶה. רַשִׁ"י: {לג} <B> שֶׁאָמַר כוּ'.</B> וְתַנָּא קַמָּא סָבַר, דַּאֲפִלּוּ הָכִי אֵין נוֹטְלִין אֶלָּא בִשְׁבוּעָה. דְּשֶׁמָּא לְהַשְׁבִּיעַ אֶת בָּנָיו אָמַר כֵּן כְּדֵי לְאַחְשְׁבִינְהוּ. תּוֹסָפוֹת:

{לד} <B> הַשֻּׁתָּפִין כוּ'.</B> וְהוּא הַדִּין לַשְּׁאָר, שֶׁאִם חָלְקוּ אֵינוֹ יָכוֹל לְהַשְׁבִּיעָן. הָרִי"ף: {לה} <B> נִתְגַּלְגְּלָה.</B> נִתְחַיֵּב לוֹ הֲוָה לֵיהּ לְמִתְנֵי, אֶלָּא דְתָנָא לָשׁוֹן נוֹפֵל עַל לָשׁוֹן מְגַלְגְּלִין עָלָיו כוּ':
{לו} כְּדִכְתִיב שָׁמוֹט כָּל בַּעַל מַשֵּׁה יָדוֹ. רַשִׁ"י. וּבַגְּמָרָא, מְנָא הָנֵי מִלֵּי, דְּאָמַר קְרָא וְזֶה דְּבַר הַשְּׁמִטָּה, וַאֲפִלּוּ דִבּוּר מְשַׁמֶּטֶת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

~ פרק ח - משנה א

{א} <B> מְשַׁלֵּם כוּ'.</B> וְיֵשׁ עָלָיו שְׁבוּעָה שֶׁאֵינוֹ בִרְשׁוּתוֹ:
{ב} וְהַאי דַּהֲדַרַהּ וְתַנְיַהּ הָכָא, מִשּׁוּם דְּבָעֵי לְמִתְנֵי חִיּוּבֵי וּפְטוֹרֵי דְּקָרְבַּן שְׁבוּעָה דִּידְהוּ, תָּנָא בָרֵישָׁא חִיּוּבֵי וּפְטוֹרֵי דְּמָמוֹן דִּידְהוּ, לְפִי שֶׁקָּרְבַּן הַשְּׁבוּעָה בְּחִיּוּבֵי וּפְטוֹרֵי תָּלוּי בְּכָךְ כוּ', כְּדִמְפָרֵשׁ וְאָזֵיל אָמַר לְשׁוֹמֵר חִנָּם כוּ':

~ פרק ח - משנה ב

{ג} פֵּרוּשׁ, מַה שֶּׁחַיָּב עַל פִּקָּדוֹן. אֲבָל מִשּׁוּם בִּטּוּי, קַיְמָא לָן כְּמַאן דְּאָמַר בַּגְּמָרָא דְּחַיָּב:

{ד} וְכֵן לְשׁוֹן רַשִׁ"י. דְּאִלּוּ אֵין כָּאן עֵדִים עַל הַגְּנֵבָה, מַאן מְחַיְּבוֹ אַטְּבִיחָה וּמְכִירָה אַף עַל פִּי שֶׁיֵּשׁ עֵדִים שֶׁטָּבַח וּמָכַר, אִי לָאו דְּאִיהוּ אוֹדֵי שֶׁגָּנַב אֵימָא דִּידֵיהּ טָבַח דִּידֵיהּ מָכַר. וְעַיֵּן פֶּרֶק ז' דְּבָבָא קַמָּא מִשְׁנָה ב' וּבְפֵרוּשׁ הָרַ"ב שָׁם מִשְׁנָה ד': 

~ פרק ח - משנה ה

{ה} <B> מֵת.</B> וְשֶׁלֹּא מֵחֲמַת מְלָאכָה: {ו} <B> מַשְׁבִּיעֲךָ אָנִי.</B> אַף עַל גַּב דְּלֹא פָטַר נַפְשֵׁיהּ בְּמִידֵי, מַשְׁבִּיעוֹ לְאַמֵּת דְּבָרָיו. וְאֵין לְפָרֵשׁ שֶׁמַּשְׁבִּיעוֹ שֶׁאֵינוֹ בִרְשׁוּתוֹ כוּ', דְּזֶה אֵינוֹ צָרִיךְ לִשָּׁבַע שֶׁכִּדְבָרָיו כֵּן הוּא אֶלָּא שֶׁאֵינָהּ בִּרְשׁוּתוֹ, וְאִם כֵּן מַאי קָא מַשְׁמַע לָן, הָא לֹא נִשְׁבָּע לַשֶּׁקֶר: 

~ פרק ח - משנה ו

{ז} <B>זֶה הַכְּלָל.</B> הָרַ"ן הֶעְתִּיקוֹ. וְאֶפְשָׁר שֶׁהוּא לֹא גָרַס זֶה הַכְּלָל בַּתְרָא: {ח} <B> לְהָקֵל כוּ' חַיָּב.</B> הָא נִשְׁבָּע שֶׁלֹּא לְהָקֵל, אֶלָּא שֶׁמְּשַׁנֶּה בֵּין מֵחוֹבָה לְחוֹבָה בֵּין מִפְּטוֹר לִפְטוֹר, נַעֲשֶׂה כְנִשְׁבָּע לְהַחְמִיר עַל עַצְמוֹ וּפָטוּר:
