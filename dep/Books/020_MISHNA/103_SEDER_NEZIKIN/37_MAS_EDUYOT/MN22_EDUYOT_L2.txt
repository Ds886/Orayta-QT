~ פרק א - משנה א
<b>שַׁמַּאי אוֹמֵר כָּל הַנָּשִׁים דַּיָּן שְׁעָתָן.</b> כָּל הַנָּשִׁים הָרוֹאוֹת דָּם, דַּיָּן שְׁעָתָן לְטַמֵּא טָהֳרוֹת שֶׁנָּגְעוּ בָּהֶן מִשְּׁעַת רְאִיָּתָן וְאֵילַךְ. וְלֹא אָמְרִינַן מִקַּמֵּי הָכִי נַמִּי הֲוָה דָּם וְכָתְלֵי בֵּית הָרֶחֶם הֶעֱמִידוּהוּ וְהָיְתָה טְמֵאָה מִקֹּדֶם שֶׁכָּל הַנָּשִׁים מְטַמְּאוֹת בַּבַּיִת הַחִיצוֹן וְאַף עַל פִּי שֶׁלֹּא יָצָא הַדָּם לַחוּץ. וְטַעֲמָא דְשַׁמַּאי דְלָא חָיֵשׁ דִּלְמָא מִקַּמֵּי הָכִי הֲוָה דָם, שֶׁאִם אַתָּה אוֹמֵר כֵּן, לִבּוֹ שֶׁל אָדָם נוֹקְפוֹ תָּמִיד בִּשְׁעַת בִּיאָה וּפוֹרֵשׁ מֵאִשְׁתּוֹ וְנִמְצֵאתָ מְבַטֵּל בְּנוֹת יִשְׂרָאֵל מִפְּרִיָּה וּרְבִיָּה: <B>מִפְּקִידָה לִפְקִידָה.</B> בָּדְקָה הַיּוֹם וּמָצְאָה טְהוֹרָה וּבָדְקָה לְסוֹף שָׁבוּעַ וּמָצְאָה טְמֵאָה, חוֹשְׁשִׁים לְמַגָּעָהּ מִבְּדִיקָה רִאשׁוֹנָה וְאֵילַךְ, שֶׁמָּא עִם סִלּוּק יָדֶיהָ רָאֲתָה וְכָתְלֵי בֵּית הָרֶחֶם הֶעֱמִידוּהוּ. וּלְבִטּוּל פְּרִיָּה וּרְבִיָּה לֹא חָיְשִׁינַן, דְּדַוְקָא לַטָּהֳרוֹת הוּא דִּמְטַמְּאִינַן לְהוּ מִפְּקִידָה לִפְקִידָה, וְלֹא לְבַעְלָהּ. וְשַׁמַּאי אוֹמֵר, אִי מְטַמֵּאת לָהּ לַטָּהֳרוֹת, לִבּוֹ נוֹקְפוֹ וּפוֹרֵשׁ נַמִּי מִתַּשְׁמִישׁ: <B>לֹא כְדִבְרֵי זֶה וְלֹא כְדִבְרֵי זֶה.</B> לֹא כְּדִבְרֵי שַׁמַּאי דְמֵקֵל טְפֵי וְאֵינוֹ עוֹשֶׂה סְיָג לִדְבָרָיו, וְלֹא כְּדִבְרֵי הִלֵּל שֶׁהִפְרִיז עַל מִדּוֹתָיו וּמַחְמִיר יוֹתֵר מִדַּאי, דְּוַדַּאי כֻלֵּי הַאי יָמִים רַבִּים לֹא מוֹקְמֵי כָּתְלֵי בֵּית הָרֶחֶם דָּם: <B>מֵעֵת לְעֵת מְמַעֶטֶת עַל יַד מִפְּקִידָה לִפְקִידָה.</B> שְׁנֵי זְמַנִּים הֻזְכְּרוּ בָּאִשָּׁה לְטַמֵּא טָהֳרוֹת לְמַפְרֵעַ, וְהַלֵּךְ אַחַר הַקַּל שֶׁבִּשְׁנֵיהֶן. אִם מִפְּקִידָה לִפְקִידָה יוֹתֵר עַל מֵעֵת לְעֵת, הַלֵּךְ אַחַר מֵעֵת לְעֵת וְלֹא תְטַמֵּא אֶלָּא טָהֳרוֹת שֶׁנָּגְעָה מֵאֶתְמוֹל בַּשָּׁעָה הַזֹּאת. וְאִם מֵעֵת לְעֵת יוֹתֵר עַל מִפְּקִידָה לִפְקִידָה, כְּגוֹן שֶׁבָּדְקָה עַצְמָהּ שַׁחֲרִית וּמָצְאָה טְהוֹרָה וְעַרְבִית מָצְאָה טְמֵאָה, לֹא מְטַמְּאִינַן אֶלָּא הָנָךְ טָהֳרוֹת דְּמִבְּדִיקַת שַׁחֲרִית וְאֵילַךְ. וַהֲלָכָה כַּחֲכָמִים: <B>עַל יַד.</B> אַחַר. כְּמוֹ וְעַל יָדוֹ הֶחֱזִיק, בְּסֵפֶר עֶזְרָא ב' [נֶחֶמְיָה ג], שֶׁהוּא כְּמוֹ וְאַחֲרָיו הֶחֱזִיק [שָׁם טז]: <B>וֶסֶת.</B> קָבוּעַ, שֶׁנִּקְבַּע לָהּ זְמַן הָאֹרַח שָׁלֹשׁ פְּעָמִים, וּבָדְקָה בִּשְׁעַת וֶסְתָּהּ וּמָצְאָה טְמֵאָה, דַּיָּהּ שְׁעָתָהּ, וְלֹא חָיְשִׁינַן דִּלְמָא מִקַּמֵּי הָכִי הֲוָה, דְּוַדַּאי אֹרַח בִּזְמַנּוֹ בָּא: <B>וְהַמְשַׁמֶּשֶׁת בְּעִדִּים הֲרֵי זוֹ כִפְקִידָה.</B> תַּרְתֵּי קָתָנֵי, וְהָכִי קָאָמַר, וּמְשַׁמֶּשֶׁת בָּעִדִּים, כְּלוֹמַר מִצְוָה עַל כָּל אִשָּׁה לְשַׁמֵּשׁ בִּשְׁנֵי עִדִּים שֶׁבּוֹדֶקֶת בָּהֶן, אֶחָד לִפְנֵי תַּשְׁמִישׁ וְאֶחָד לְאַחַר תַּשְׁמִישׁ: <B>הֲרֵי זוֹ כִפְקִידָה.</B> הָעֵד שֶׁלְאַחַר תַּשְׁמִישׁ הֲרֵי הוּא כִּפְקִידָה: <B>מְמַעֶטֶת עַל יַד מֵעֵת לְעֵת וְעַל יַד מִפְּקִידָה לִפְקִידָה.</B> דְּמַהוּ דְתֵימָא שֶׁמָּא תִּרְאֶה טִפַּת דָּם כְּחַרְדָּל וּתְחַפֶּנָּהּ שִׁכְבַת זֶרַע וְלֹא לֶהֱוֵי כִּפְקִידָה, קָא מַשְׁמַע לָן מַתְנִיתִין דְּעֵד שֶׁלְאַחַר תַּשְׁמִישׁ הֲוָה כִפְקִידָה. אֲבָל הָעֵד שֶׁלִּפְנֵי תַּשְׁמִישׁ אֵינוֹ כִּפְקִידָה, שֶׁמִּתּוֹךְ שֶׁהִיא מְהוּמָה לְבֵיתָהּ, כְּלוֹמַר בְּהוּלָה לְשַׁמֵּשׁ, אֵינָהּ מַכְנֶסֶת אוֹתוֹ לַחוֹרִין וְלַסְּדָקִין:

<B>מִקַּב לְחַלָּה.</B> עִסָּה שֶׁיֵּשׁ בָּהּ קַב חַיֶּבֶת בְּחַלָּה: <B>קַב וּמֶחֱצָה חַיָּבִים בְּחַלָּה.</B> שֶׁהֵן שִׁבְעָה לֻגִּין וּבֵיצָה וְחֹמֶשׁ בֵּיצָה, לְמִדָּה מִדְבָּרִית. וְהוּא עֹמֶר לַגֻּלְגֹּלֶת עֲשִׂירִית הָאֵיפָה שֶׁחַיָּב בְּחַלָּה, כְּדִכְתִיב (בְּמִדְבַּר טו) רֵאשִׁית עֲרִסֹתֵיכֶם, כְּדֵי עִסַּת מִדְבָּר. וְהוֹסִיפוּ שְׁתוּת בַּיְרוּשַׁלְמִיּוֹת, וְנִמְצְאוּ שֵׁשׁ מִדְבָּרִיּוֹת נִכְנָסִים בְּחָמֵשׁ יְרוּשַׁלְמִיּוֹת, וְלֹג הַנִּשְׁאָר וּבֵיצָה וְחֹמֶשׁ בֵּיצָה עוֹלִים לְלֹג יְרוּשַׁלְמִי. שֶׁהַלֹּג מִדְבָּרִי שִׁשָּׁה בֵּיצִים, תִּתְּנֵם בְּחָמֵשׁ בֵּיצִים גְּדוֹלוֹת, נִמְצָא הַלֹּג חָסֵר בֵּיצָה גְּדוֹלָה, תֵּן בֵּיצָה וְחֹמֶשׁ בֵּיצָה תַּחַת הַבֵּיצָה גְּדוֹלָה, שֶׁהַחֹמֶשׁ בֵּיצָה שֶׁהוּא שְׁתוּת מִלְּבַר נוֹסָף עַל הַבֵּיצָה, נִמְצְאוּ שֵׁשׁ בֵּיצִים גְּדוֹלִים שֶׁהֵן לֹג גָּדוֹל, הֲרֵי הֵן שִׁשָּׁה לֻגִּים גְּדוֹלִים שֶׁהֵן קַב וּמֶחֱצָה: <B>מִשֶּׁהִגְדִּילוּ הַמִּדּוֹת.</B> זוֹ הִיא מִדָּה צִפּוֹרִית, שֶׁהוֹסִיפוּ שְׁתוּת עַל שֶׁל יְרוּשַׁלְמִית, נִמְצְאוּ הַשִּׁשָּׁה לֻגִּין חֲמִשָּׁה, שֶׁהֵן חֲמֵשֶׁת רִבְעֵי קַב, שֶׁהַקַּב אַרְבָּעָה לֻגִּין: <B>חֲמִשָּׁה וָעוֹד חַיָּבִים.</B> דְּקָסָבַר רַבִּי יוֹסֵי, מִדָּה מִדְבָּרִית הָיוּ בֵּיצִים גְּדוֹלוֹת מִשֶּׁלָּנוּ אֶחָד מֵעֶשְׂרִים בְּבֵיצָה לְכָל בֵּיצָה. וַהֲלָכָה כַּחֲכָמִים, דְּקַב וּמֶחֱצָה יְרוּשַׁלְמִי דְּהַיְנוּ שִׁשָּׁה לֻגִּין יְרוּשַׁלְמִיּוֹת שֶׁהֵן שִׁבְעָה לֻגִּין וּבֵיצָה וְחֹמֶשׁ בֵּיצָה מִדְבָּרִיּוֹת, הֵן שִׁעוּר חַלָּה. וְהֵן אַרְבָּעִים וְשָׁלֹשׁ בֵּיצִים וְחֹמֶשׁ בֵּיצָה. וְרַמְבַּ"ם אִזֵּן וְחָקַר וְתִקֵּן וּבָדַק וּמָצָא שֶׁמִּשְׁקַל חֲמֵשׁ מֵאוֹת וְעֶשְׂרִים דְּרַהֲ"ם מִקֶּמַח חִטִּים הוּא שִׁעוּר הַקֶּמַח הַחַיָּב בְּחַלָּה. וּמִשְׁקַל הָדְּרַהֲ"ם יָדוּעַ בְּמִצְרַיִם הַיּוֹם וּבְכָל אֶרֶץ יִשְׂרָאֵל, שֶׁהוּא מִשְׁקַל שִׁשִּׁים וְאֶחָד גַּרְגִּירֵי שְׂעֹרוֹת בְּקֵרוּב: 

~ פרק א - משנה ג
<B>הִין.</B> שְׁנֵים עָשָׂר לֻגִּין: <B>פּוֹסְלִין אֶת הַמִּקְוֶה.</B> אִם נָפְלוּ בְּתוֹכוֹ קֹדֶם שֶׁנִּשְׁלַם שִׁעוּרוֹ. אֲבָל לְאַחַר שֶׁנִּשְׁלַם, אֲפִלּוּ הִשְׁלִיךְ לְתוֹכוֹ כָּל מַיִם שְׁאוּבִים שֶׁבָּעוֹלָם שׁוּב אֵינָן פּוֹסְלִים: <B>שֶׁחַיָּב אָדָם לוֹמַר בִּלְשׁוֹן רַבּוֹ.</B> כְּלוֹמַר, הִין אֵינוֹ לְשׁוֹן מִשְׁנָה אֶלָּא לְשׁוֹן תּוֹרָה, אֶלָּא כָּךְ שָׁמַע מֵרַבּוֹתָיו שְׁמַעְיָה וְאַבְטַלְיוֹן. וְרַמְבַּ"ם קִבֵּל מֵאָבִיו זַ"ל שֶׁמִּפְּנֵי שֶׁהָיוּ שְׁמַעְיָה וְאַבְטַלְיוֹן גֵּרֵי צֶדֶק לֹא הָיוּ יְכוֹלִין לְהוֹצִיא מִפִּיהֶן מִלַּת הִין, וְהָיוּ אוֹמְרִים אִין בִּמְקוֹם הִין, כְּדֶרֶךְ בְּנֵי אָדָם עַד הַיּוֹם שֶׁאֵינָם יְכוֹלִים לַחְתֹּךְ בְּאוֹתִיּוֹת [אחה"ע], וְהָיָה הִלֵּל גַּם הוּא אוֹמֵר אִין, כְּמוֹ שֶׁהָיוּ רַבּוֹתָיו שְׁמַעְיָה וְאַבְטַלְיוֹן גֵּרֵי צֶדֶק אוֹמְרִים: <B>גַּרְדִּיִּים.</B> אוֹרְגִים: <B>מִשַּׁעַר הָאַשְׁפּוֹת.</B> הִזְכִּיר הַתַּנָּא שֵׁם אֻמָּנוּתָן וְשֵׁם שְׁכוּנָתָן, לוֹמַר לָךְ שֶׁלֹּא יִמְנַע אָדָם עַצְמוֹ מִבֵּית הַמִּדְרָשׁ. שֶׁאֵין לְךָ אֻמָּנוּת פְּחוּתָה מִן הַגַּרְדִּי {ה}, שֶׁאֵין מַעֲמִידִים הֵימֶנּוּ לֹא מֶלֶךְ וְלֹא כֹּהֵן גָּדוֹל, וְאֵין שַׁעַר בִּירוּשָׁלַיִם פְּחוּתָה מִשַּׁעַר הָאַשְׁפּוֹת, וְהִכְרִיעוּ בְּעֵדוּתָן לְכָל חַכְמֵי יִשְׂרָאֵל: 

~ פרק א - משנה ד
<B>שֶׁלֹּא יְהֵא אָדָם עוֹמֵד עַל דְּבָרָיו.</B> שֶׁלֹּא יְהֵא סַרְבָן לַעֲמֹד קַיָּם בִּסְבָרָתוֹ: <B>אֲבוֹת הָעוֹלָם.</B> הִלֵּל וְשַׁמַּאי: 

~ פרק א - משנה ה
<B>וְיִסְמֹךְ עָלָיו.</B> כְּלוֹמַר שֶׁעָשָׂה כְּדִבְרֵי הַיָּחִיד וְהִנִּיחַ דַּעַת הָרַבִּים: <B>אֵין בֵּית דִּין.</B> אַחֵר שֶׁיַּעֲמֹד אַחֲרָיו יָכוֹל לְבַטֵּל דִּבְרֵי בֵּית דִּין הָרִאשׁוֹן שֶׁעָשָׂה כְּדִבְרֵי הַיָּחִיד, עַד שֶׁיִּהְיֶה גָּדוֹל מִמֶּנּוּ בְּחָכְמָה וּבְמִנְיָן. בְּחָכְמָה, הַיְנוּ שֶׁרֹאשׁ הַיְשִׁיבָה שֶׁל בֵּית דִּין הָאַחֲרוֹן יִהְיֶה גָּדוֹל בְּחָכְמָה מֵרֹאשׁ הַיְשִׁיבָה שֶׁל רִאשׁוֹן {ט}. וּבְמִנְיָן, שֶׁיִּהְיוּ מִנְיַן הַתַּלְמִידִים שֶׁבַּיְשִׁיבָה שֶׁל אַחֲרוֹן מְרֻבִּים מִמִּנְיַן הַתַּלְמִידִים שֶׁבַּיְשִׁיבָה שֶׁל רִאשׁוֹן {י}: 

~ פרק א - משנה ו
<B>לָמָּה מַזְכִּירִין דִּבְרֵי הַיָּחִיד בֵּין הַמְרֻבִּין.</B> כְּדֵי לְבַטְּלָן. יָחִיד שֶׁלֹּא עָשָׂה שׁוּם בֵּית דִּין כִּדְבָרָיו וְהֵן דְּחוּיִים וּבְטֵלִין {יג}, לָמָּה מַזְכִּירִין אוֹתָן כְּלָל. וּמְשָׁנֵי, שֶׁאִם יֹאמַר אָדָם כָּךְ אֲנִי מְקֻבָּל וְיִתְמַהּ כְּשֶׁרוֹאֶה שֶׁאֵין עוֹשִׂין כְּקַבָּלָתוֹ, יֹאמְרוּ לוֹ כְּדִבְרֵי אִישׁ פְּלוֹנִי שָׁמַעְתָּ וְנִדְחוּ דְּבָרָיו: 

~ פרק א - משנה ז
<B>רֹבַע עֲצָמוֹת.</B> רֹבַע קַב שֶׁל עַצְמוֹת הַמֵּת מְטַמְּאִים בְּאֹהֶל. וּפָחוֹת [מֵרֹבַע] אֵין מְטַמְּאִין אֶלָּא בְּמַגָּע וּבְמַשָּׂא אֲבָל לֹא בְּאֹהֶל. וְסָבְרֵי בֵּית שַׁמַּאי דְרֹבַע קַב מְטַמְּאִים וַאֲפִלּוּ הֵן מִמֵּתִים הַרְבֵּה: <B>וּבֵית הִלֵּל אוֹמְרִים רֹבַע עֲצָמוֹת מִן הַגְּוִיָּה.</B> כְּלוֹמַר, מִגּוּף אֶחָד שֶׁל מֵת אֶחָד, וְלֹא מִמֵּתִים הַרְבֵּה. וַאֲפִלּוּ שֶׁל מֵת אֶחָד אֵין מְטַמְּאִין עַד שֶׁיִּהְיֶה בְּרֹבַע הַקַּב רֹב בִּנְיָן {טו} דְּהַיְנוּ רֹב גֹּדֶל הַגּוּף, אוֹ רֹב מִנְיַן עַצְמוֹתָיו שֶׁל אָדָם, וְיִהְיוּ מֵאָה עֶשְׂרִים וְחָמֵשׁ עֲצָמוֹת, שֶׁמִּנְיַן עַצְמוֹתָיו שֶׁל אָדָם מָאתַיִם אַרְבָּעִים וּשְׁמֹנֶה: <B>אֲפִלּוּ מֵעֶצֶם אֶחָד.</B> אִם עֶצֶם אֶחָד שֶׁל מֵת מְמַלֵּא רֹבַע קַב, מְטַמֵּא בְּאֹהֶל. וַהֲלָכָה כְּבֵית הִלֵּל: 

~ פרק א - משנה ח
<B>כַּרְשִׁינֵי תְרוּמָה.</B> בַּעֲרָבִי קוֹרִין לָהּ כַּרְסְנָ"א. וְהֵם מַאֲכָל לַגְּמַלִּים וְאֵין בְּנֵי אָדָם אוֹכְלִין מֵהֶם אֶלָּא מִדֹּחַק בִּשְׁנוֹת רְעָבוֹן. וּמַפְרִישִׁין מֵהֶם תְּרוּמָה, הוֹאִיל וְנֶאֱכָלִים לָאָדָם לִפְרָקִים עַל יְדֵי הַדְּחָק. וְאֵינָהּ קְדוֹשָׁה כִּשְׁאָר תְּרוּמוֹת: <B>שׁוֹרִין.</B> אוֹתָן בְּמַיִם: <B>וְשָׁפִין.</B> עַל (בְּשָׂרָן) [הַבָּשָׂר]:  בטהרה
<B>בְּטָהֳרָה.</B> בִּנְטִילַת יָדַיִם, כְּדִין כָּל שְׁאָר אֳכָלִים שֶׁל תְּרוּמָה. שֶׁסְּתָם יָדַיִם שְׁנִיּוֹת הֵן וּפוֹסְלוֹת אֶת הַתְּרוּמָה: <B>וּמַאֲכִילִין.</B> לַבְּהֵמָה: <B>בְּטֻמְאָה.</B> וְאֵינוֹ חוֹשֵׁשׁ אִם מְטַמֵּא אוֹתָן בַּיָּדַיִם בְּשָׁעָה שֶׁמַּאֲכִילָן לַבְּהֵמָה. אֲבָל כָּל זְמַן שֶׁאֵינוֹ מַאֲכִילָן לַבְּהֵמָה, אָסוּר לְטַמְּאָן בַּיָּדַיִם: <B>שׁוֹרִין בְּטָהֳרָה.</B> שֶׁשְּׁרִיָּתָן בַּמַּיִם מַכְשִׁירָתָן לְקַבֵּל טֻמְאָה, וְאִם שׁוֹרֶה אוֹתָן בְּטֻמְאָה נִמְצָא הֶכְשֵׁרָן וְטֻמְאָתָן בָּאִים כְּאֶחָד. וְזֶה בִּלְבַד אוֹסְרִים בֵּית הִלֵּל, מִשּׁוּם הֶכֵּר כְּדֵי שֶׁיֵּדְעוּ שֶׁהֵן תְּרוּמָה: <B>יֵאָכְלוּ צָרִיד.</B> לְשׁוֹן יֹבֶשׁ. כְּמוֹ צָרִיד שֶׁל מְנָחוֹת, שֶׁהוּא מָקוֹם שֶׁל מְנָחוֹת שֶׁלֹּא הִגִּיעַ שָׁם שֶׁמֶן. אַף כָּאן יֵאָכְלוּ יְבֵשִׁים שֶׁלֹּא יְהֵא עֲלֵיהֶן מַשְׁקֶה בִּשְׁעַת אֲכִילָה, כְּדֵי שֶׁלֹּא יִהְיֶה נִכָּר שֶׁהֻכְשְׁרוּ לְקַבֵּל טֻמְאָה: <B>כָּל מַעֲשֵׂיהֶם בְּטֻמְאָה.</B> וַאֲפִלּוּ הַשְּׁרִיָּה. וַהֲלָכָה כְּבֵית הִלֵּל: 

~ פרק א - משנה ט
<B>הַפּוֹרֵט סֶלַע מִמְּעוֹת מַעֲשֵׂר שֵׁנִי.</B> מִי שֶׁיֵּשׁ לוֹ מְעוֹת נְחֹשֶׁת שֶׁל מַעֲשֵׂר שֵׁנִי וּבָא לְפָרְטָן בְּסֶלַע כֶּסֶף לְהַעֲלוֹת לִירוּשָׁלַיִם מִפְּנֵי מַשְּׂאוֹי הַדֶּרֶךְ: <B>בֵּית שַׁמַּאי אוֹמְרִים בְּכָל הַסֶּלַע מָעוֹת.</B> אִם בָּא לְפָרְטָן, יָכוֹל הוּא לִפְרֹט כֻּלָּן, וְנוֹתֵן מָעוֹת בִּשְׁבִיל כָּל הַסֶּלַע: <B>וּבֵית הִלֵּל אוֹמְרִים.</B> לֹא יִפְרֹט אֶלָּא חֶצְיָן. שֶׁכְּשֶׁיָּבֹא לִירוּשָׁלַיִם יִהְיֶה צָרִיךְ לַפְּרוּטוֹת מִיָּד לִקְנוֹת צָרְכֵי סְעֻדָּה, וְאִם יָרוּצוּ הַכֹּל אֵצֶל שֻׁלְחָנִי לִפְרֹט יוֹקִירוּ הַפְּרוּטוֹת, וְנִמְצָא מַעֲשֵׂר שֵׁנִי נִפְסָד. לְפִיכָךְ יִשָּׂא פְּרוּטוֹת עִמָּהֶן לְהוֹצִיא בְּמִקְצָת, וְלִכְשֶׁיִּכְלוּ יִפְרֹט בַּכֶּסֶף שֶׁבְּיָדוֹ מְעַט מְעַט. שֶׁקֶל הוּא חֲצִי סֶלַע: <B>אֵין מְחַלְּלִין כֶּסֶף וּפֵרוֹת עַל כֶּסֶף.</B> מִי שֶׁיֵּשׁ לוֹ חֲצִי דִּינָר כֶּסֶף שֶׁל מַעֲשֵׂר, וּפֵרוֹת שֶׁל מַעֲשֵׂר שָׁוִים חֲצִי דִּינָר, לֹא יְצָרְפֵם יַחַד לְחַלְּלָם עַל דִּינָר: <B>וַחֲכָמִים מַתִּירִים.</B> בִּכְהַאי גַוְנָא עַל יְדֵי צֵרוּף פֵּרוֹת, כֵּיוָן שֶׁאֵין לוֹ אֶלָּא חֲצִי דִּינַר כֶּסֶף. אֲבָל לְחַלֵּל דִּינַר כֶּסֶף, וּפֵרוֹת שָׁוִין דִּינָר, עַל חֲצִי סֶלַע שֶׁהוּא שְׁנֵי דִּינָרִין, מוֹדִים חֲכָמִים שֶׁאֵין מְחַלְּלִים. וַהֲלָכָה כַּחֲכָמִים: 

~ פרק א - משנה י
<B>הַפּוֹרֵט סֶלַע שֶׁל מַעֲשֵׂר שֵׁנִי בִּירוּשָׁלַיִם.</B> שֶׁהָיָה מַחֲלִיף סֶלַע שֶׁבְּיָדוֹ וְנוֹטֵל פְּרוּטוֹת לְהוֹצִיאָם לְצָרְכֵי סְעֻדַּת מַעֲשֵׂר: <B>בֵּית שַׁמַּאי אוֹמְרִים.</B> אִם בָּא לְהַחֲלִיף כָּל הַסְּלָעִים שֶׁבְּיָדוֹ בְּמָעוֹת יַחֲלִיף: <B>וּבֵית הִלֵּל אוֹמְרִים, לֹא יַחֲלִיף אֶלָּא חֶצְיָן.</B> שֶׁמָּא לֹא יִשְׁהֶה בָּעִיר עַד שֶׁיּוֹצִיא אֶת כֻּלָּן, וְיַפְקִידֵם בָּעִיר עַד רֶגֶל אַחֵר, וְהַפְּרוּטוֹת מִתְעַפְּשׁוֹת. וְאִם יַחֲזֹר וְיַחֲלִיפֵם בִּסְלָעִים נִמְצָא שֻׁלְחָנִי מִשְׂתַּכֵּר שְׁתֵּי פְּעָמִים וּמַעֲשֵׂר שֵׁנִי נִפְסָד: <B>הַדָּנִים לִפְנֵי חֲכָמִים.</B> שִׁמְעוֹן בֶּן עַזַּאי וְשִׁמְעוֹן בֶּן זוֹמָא וְחָנָן הַמִּצְרִי: <B>בִּשְׁלֹשָׁה דִינָרִים כֶּסֶף וּבְדִינָר מָעוֹת.</B> הַסֶּלַע הוּא אַרְבָּעָה דִּינָרִים, וּכְשֶׁבָּא לְהַחֲלִיף הַסֶּלַע לֹא יִקַּח אֶלָּא בְּדִינָר אֶחָד פְּרוּטוֹת, וּשְׁלֹשָׁה דִּינָרִים יִהְיוּ כֶּסֶף: <B>וּבִרְבִיעִית כֶּסֶף בִּרְבִיעִית מָעוֹת.</B> בְּדִינָר רְבִיעִי שֶׁל כֶּסֶף לֹא יִקַּח אֶלָּא בִּרְבִיעִיתוֹ מָעוֹת שֶׁל נְחֹשֶׁת וּשְׁלֹשָׁה חֲלָקִים כֶּסֶף, שֶׁנִּמְצָא לוֹקֵחַ מָעוֹת אֶחָד מִשִּׁשָּׁה עָשָׂר בְּסֶלַע בִּלְבַד: <B>אַרְבָּעָה אַסְפְּרֵי כָסֶף.</B> הַדִּינָר חֲמִשָּׁה אַסְפְּרֵי, וּמַטְבֵּעַ הוּא בְּאֶרֶץ יָוָן שֶׁעַד הַיּוֹם קוֹרִין לוֹ אַסְפְּר"וֹ. נִמְצָא הַסֶּלַע עֶשְׂרִים אַסְפְּר"וֹ. כְּשֶׁהוּא מְחַלֵּל בְּדִינָר יְחַלְּלֶנּוּ עַל אַרְבָּעָה אַסְפְּר"וֹ כֶּסֶף וְאַסְפְּר"וֹ אֶחָד נְחֹשֶׁת. נִמְצָא לוֹקֵחַ נְחֹשֶׁת אֶחָד מֵעֶשְׂרִים בְּסֶלַע בִּלְבַד: <B>יַנִּיחֶנָּה בַחֲנוּת וְיֹאכַל כְּנֶגְדָּהּ.</B> לֹא יְחַלֵּל כְּלָל עַל פְּרוּטוֹת, שֶׁמָּא יִשְׁכַּח וְיַעֲשֶׂה אוֹתָן חֻלִּין. אֶלָּא יַנִּיחַ הַסֶּלַע אֵצֶל הַחֶנְוָנִי וְיֹאכַל כְּנֶגְדָּהּ עַד שֶׁתִּכְלֶה. וַהֲלָכָה כְּדִבְרֵי בֵּית הִלֵּל בִּלְבַד: 

~ פרק א - משנה יא
<B>שֶׁנִּטְּלוּ חִפּוּיָיו.</B> רַבּוֹתַי פֵּרְשׁוּ מִלְּשׁוֹן חָפֵי פּוֹתַחַת [שַׁבָּת דַּף פא], שֶׁהֵם הַשִּׁנַּיִם הַבּוֹלְטִים מִן הַמַּפְתֵּחוֹת שֶׁנּוֹהֲגִים לַעֲשׂוֹת בְּאֶרֶץ יִשְׁמָעֵאל, אַף כָּאן רְגִילִים לַעֲשׂוֹת בַּכִּסֵּא שֶׁל כַּלָּה כְּמִין שִׁנַּיִם בּוֹלְטִים כְּדֵי שֶׁיִּסְמְכוּ עֲלֵיהֶן. וְרַמְבַּ"ם פֵּרֵשׁ חִפּוּיָיו, פִּתּוּחִין וְצִיּוּרִין שֶׁעוֹשִׂים מֵעֵצִים אוֹ מֵאֲבָנִים וּמַדְבִּיקִין אוֹתָן בְּכִסֵּא הַכַּלָּה: <B>בֵּית שַׁמַּאי מְטַמְּאִים.</B> דְּאַכַּתִּי חֲזִי לִישִׁיבָה: <B>וּבֵית הִלֵּל מְטַהֲרִין.</B> דְּלֹא חֲזִי לְכַלָּה, וְהָוֵי כְּנִשְׁבָּר: <B>אַף מַלְבֵּן הַכִּסֵּא טָמֵא.</B> כְּלוֹמַר אַף מַלְבֵּן הַכִּסֵּא בִּפְנֵי עַצְמוֹ בְּלֹא הַכִּסֵּא וּבְלֹא חִפּוּיָיו טָמֵא. כָּל שֶׁכֵּן שֶׁהַכִּסֵּא בְּלֹא חִפּוּיָיו טָמֵא. מַלְבֵּן, צוּרַת לְבֵנָה מְרֻבַּעַת עוֹשִׂין עַל הַכִּסֵּא וְיוֹשְׁבִין עָלָיו: <B>כִּסֵּא שֶׁקְּבָעוֹ בַעֲרֵבָה.</B> הֵבִיא כִּסֵּא מִמָּקוֹם אַחֵר וּקְבָעוֹ בָּעֲרֵבָה דֶּרֶךְ יְשִׁיבָתוֹ. וַעֲרֵבָה אֵינָהּ טְמֵאָה מִדְרָס, דַּחֲזְיָא לְלִישָׁה וְלֹא לֵישֵׁב עָלֶיהָ: <B>בֵּית שַׁמַּאי מְטַמְּאִין.</B> דְּלֹא בָּטֵל כִּסֵּא לְגַבֵּי עֲרֵבָה: <B>וּבֵית הִלֵּל מְטַהֲרִין.</B> מִמִּדְרָס, דְּבָטֵל כִּסֵּא לְגַבֵּי עֲרֵבָה. אֲבָל כִּסֵּא הֶעָשׂוּי בְּגוּף הָעֲרֵבָה עַצְמָהּ, מוֹדוּ בֵּית שַׁמַּאי לְבֵית הִלֵּל דְּטָהוֹר: <B>שַׁמַּאי אוֹמֵר.</B> אַף כִּסֵּא הֶעָשׂוּי בָּעֲרֵבָה עַצְמָהּ טְמֵא מִדְרָס:

<B>אֶלָּא בְּבָאָה מִן הַקָּצִיר.</B> כַּמַּעֲשֶׂה שֶׁהָיָה שֶׁהָלְכוּ בְּנֵי אָדָם לִקְצֹר חִטִּין וּנְשָׁכוֹ נָחָשׁ לְאֶחָד מֵהֶן וּמֵת, וּבָאָה וְהוֹדִיעָה בְּבֵית דִּין וְשָׁלְחוּ וּמָצְאוּ כִּדְבָרֶיהָ. וְלֹא הִתִּירוּ אֶלָּא דֻּגְמָתוֹ, שֶׁיְּהֵא הַדָּבָר קָרוֹב, אֲבָל מִמְּדִינַת הַיָּם אֵינָהּ נֶאֱמֶנֶת: <B>אֶלָּא בַהֹוֶה.</B> מַעֲשֶׂה שֶׁהָיָה כָּךְ הָיָה, וְהוּא הַדִּין לִשְׁאָר מְקוֹמוֹת: <B>שֶׁאֵין הָאַחִין נִכְנָסִין.</B> לְנַחֲלַת בַּעְלָהּ, דְּרַחֲמָנָא אָמַר עַל פִּי שְׁנֵי עֵדִים. וְגַבֵּי נִשּׂוּאִים דִּידָהּ הוּא דַּאֲקִילוּ מִשּׁוּם עִגּוּנָהּ: <B>מִסֵּפֶר כְּתֻבָּתָהּ.</B> מִנֻּסַּח שֶׁהִתְקִינוּ לִכְתֹּב בִּשְׁטַר כְּתֻבָּה: <B>כְּשֶׁתִּנָּשְׂאִי לְאַחֵר.</B> וַהֲרֵי הִיא נִשֵּׂאת וְאִם כֵּן תִּטֹּל כְּתֻבָּתָהּ: 

~ פרק א - משנה יג
<B>מִי שֶׁחֶצְיוֹ עֶבֶד וְחֶצְיוֹ בֶּן חוֹרִין.</B> כְּגוֹן עֶבֶד שֶׁל שְׁנֵי שֻׁתָּפִים וְשִׁחְרְרוֹ אֶחָד מֵהֶן. אִי נַמִּי, שֶׁקִּבֵּל רַבּוֹ מִמֶּנּוּ חֲצִי דָּמָיו וְשִׁחְרֵר חֶצְיוֹ בְּאוֹתָן הַדָּמִים: <B>תִּקַּנְתֶּם אֶת רַבּוֹ.</B> שֶׁאֵינוֹ חָסֵר כְּלוּם: <B>לִשָּׂא שִׁפְחָה אֵינוֹ יָכוֹל.</B> מִפְּנֵי צַד חֵרוּת שֶׁבּוֹ: <B>בַּת חוֹרִין אֵינוֹ יָכוֹל.</B> מִפְּנֵי צַד עַבְדוּת שֶׁבּוֹ: <B>כּוֹפִין אֶת רַבּוֹ וְעוֹשֵׂהוּ בֶן חוֹרִין.</B> וְהוּא הַדִּין אִם הָיָה עֶבֶד שֶׁל מֵאָה שֻׁתָּפִין וְאֶחָד מֵהֶם שִׁחְרְרוֹ, שֶׁכּוֹפִין אֶת כֻּלָּם לְשַׁחְרְרוֹ: 

~ פרק א - משנה יד
<B>כְּלִי חֶרֶס מַצִּיל עַל הַכֹּל.</B> כְּלִי חֶרֶס הַמֻּקָּף צָמִיד פָּתִיל מַצִּיל עַל כָּל מַה שֶּׁבְּתוֹכוֹ כְּשֶׁהוּא בְּאֹהֶל הַמֵּת, וְלֹא יִטַּמֵּא כָּל מַה שֶּׁבְּתוֹכוֹ, דִּכְתִיב (בְּמִדְבַּר יט) וְכֹל כְּלִי פָתוּחַ אֲשֶׁר אֵין צָמִיד פָּתִיל עָלָיו טָמֵא הוּא, הָא יֵשׁ עָלָיו צָמִיד פָּתִיל טָהוֹר הוּא וּמַה שֶּׁבְּתוֹכוֹ, לֹא שְׁנָא כֵּלִים וְלֹא שְׁנָא אֳכָלִין וְיִשְׁתַּמְּשׁוּ בָּהֶן הֵן שֶׁמַּחֲזִיקִין כְּלֵיהֶם בְּחֶזְקַת טְהוֹרִים, וְאֵין לָחֹשׁ שֶׁמָּא יִשְׁאַל הֶחָבֵר מֵהֶן וְיִשְׁתַּמֵּשׁ בָּהֶן, שֶׁהֲרֵי הֵן אֶצְלוֹ בְּחֶזְקַת טְמֵאִים וְאֵין לָהֶם ~ פרק ב - משנה א
<B>רַבִּי חֲנִינָא סְגַן הַכֹּהֲנִים הֵעִיד.</B> מִלִּשְׂרֹף אֶת הַבָּשָׂר שֶׁנִּטְמָא בִּוְלַד הַטֻּמְאָה. בְּוָלָד שֶׁל וָלָד קָאָמַר. הַבָּשָׂר שֶׁהוּא שְׁלִישִׁי, שֶׁנִּטְמָא בְּוָלָד שֶׁל וָלָד הוֹסִיף רַבִּי עֲקִיבָא עַל דִּבְרֵי רַבִּי חֲנִינָא סְגַן הַכֹּהֲנִים, שֶׁהֲרֵי רַבִּי חֲנִינָא לֹא הִתִּיר אֶלָּא לְהַחֲזִיר הַשְּׁלִישִׁי שֵׁנִי, וְרַבִּי עֲקִיבָא הִתִּיר לְהַחֲזִיר הַשְּׁלִישִׁי רִאשׁוֹן, שֶׁהַשֶּׁמֶן שֶׁנִּפְסַל בִּטְבוּל יוֹם שֶׁהוּא שְׁלִישִׁי, כְּשֶׁמַּדְלִיקִין אוֹתוֹ בְּנֵר שֶׁנִּטְמָא בִּטְמֵא מֵת שֶׁהַנֵּר עַצְמוֹ נַעֲשָׂה אָב כְּמוֹ שֶׁאָמַרְנוּ, נִמְצָא שֶׁחוֹזֵר <B>לֹא רָאִיתִי עוֹר יוֹצֵא לְבֵית הַשְּׂרֵפָה.</B> לְאַחַר שֶׁהֻפְשַׁט אִם נִמְצֵאת טְרֵפָה. אַף עַל פִּי שֶׁפְּסוּל זֶה הָיָה בּוֹ קֹדֶם הֶפְשֵׁט, הוֹאִיל וְלֹא נִכַּר אֶלָּא לְאַחַר הֶפְשֵׁט: <B>שֶׁהַמַּפְשִׁיט אֶת הַבְּכוֹר וְנִמְצָא טְרֵפָה.</B> הָא קָא מַשְׁמַע לָן רַבִּי עֲקִיבָא, דַּאֲפִלּוּ בְּכוֹר בַּעַל מוּם הַנִּשְׁחָט בַּמְּדִינָה עַל מוּמוֹ וְלֹא הִתִּירוֹ הַכָּתוּב אֶלָּא בַּאֲכִילָה כְּדִכְתִיב (דְּבָרִים טו) בִּשְׁעָרֶיךָ תֹאכֲלֶנּוּ, אֲבָל אִם מֵת, עוֹרוֹ אָסוּר וְטָעוּן קְבוּרָה, וְאַשְׁמְעִינַן רַבִּי עֲקִיבָא דְהֵיכָא דְּלֹא הֻכַּר טְרֵפוּתוֹ עַד לְאַחַר הֶפְשֵׁט, הִתִּירָתוֹ שְׁחִיטָתוֹ וְהֶפְשֵׁטוֹ לְעוֹרוֹ כְּאִלּוּ נִזְרַק דָּמוֹ בַּמִּקְדָּשׁ: <B>יֵאוֹתוּ הַכֹּהֲנִים בְּעוֹרוֹ.</B> וְאֵינוֹ נִשְׂרָף: <B>אֵין לֹא רָאִיתִי רְאָיָה.</B> שֶׁמָּא לֹא אֵרַע בְּיָמָיו שֶׁיִּמָּצֵא טְרֵפָה לְאַחַר הֶפְשֵׁט, וְאִם אֵרַע וּשְׂרָפוּהוּ הוּא לֹא רָאָה: <B>אֶלָּא יֵצֵא לְבֵית הַשְּׂרֵפָה.</B> הוֹאִיל וְקֹדֶם הֶפְשֵׁט בָּא. וַהֲלָכָה כְּרַבִּי עֲקִיבָא בִּבְכוֹר בַּעַל מוּם כְּשֶׁהִתִּירוֹ מֻמְחֶה. אֲבָל לֹא הִתִּירוֹ מֻמְחֶה לֹא. וַהֲלָכָה כַּחֲכָמִים בִּבְכוֹר תָּמִים, דְּבָשָׂר בִּקְבוּרָה וְעוֹר בִּשְׂרֵפָה:

<B>וְכוֹתֵב בִּכְתַב יָדוֹ.</B> שְׁטַר חוֹב עַל הַלֹּוֶה: <B>וַאֲחֵרִים חוֹתְמִים.</B> עֵדִים כְּשֵׁרִים הָיוּ חוֹתְמִים עַל הַשְּׁטָר: <B>וְהִתִּירוּ.</B> וְאַף עַל פִּי שֶׁכּוֹתֵב הַשְּׁטָר הוּא הַמַּלְוֶה, וְנוֹגֵעַ בְּעֵדוּתוֹ: <B>שֶׁהָאִשָּׁה כּוֹתֶבֶת אֶת גִּטָּהּ.</B> וְעֵדִים כְּשֵׁרִים חֲתוּמִים בּוֹ: <B>אֶת שׁוֹבְרוֹ.</B> שְׁטַר מְחִילָה שֶׁמָּחֲלָה לוֹ אִשְׁתּוֹ עַל כְּתֻבָּתָהּ: <B>שֶׁאֵין קִיּוּם הַגֵּט אֶלָּא בְחוֹתְמָיו.</B> הָעֵדִים הַחֲתוּמִים {ג} עַל הַגֵּט הֵם עִקַּר סִבַּת קִיּוּם הַגֵּט, הִלְכָּךְ כְּשֶׁהָעֵדִים כְּשֵׁרִים חֲתוּמִים בּוֹ כָּשֵׁר וְאַף עַל פִּי שֶׁהוּא כְּתַב יַד הָאִשָּׁה: <B>וְעַל מַחַט.</B> שֶׁיָּדוּעַ בָּהּ שֶׁנִּטְמֵאת בְּמֵת, וְנִמְצֵאת בִּבְשַׂר הַקֹּדֶשׁ כְּשֶׁחָתְכוּ אוֹתוֹ בָּעֲזָרָה, וְסָפֵק אִם נָגַע בָּהּ הַסַּכִּין וְהָאָדָם אִם לָאו, הַסַּכִּין וְהָאָדָם טְהוֹרִין, דִּסְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים הִיא, שֶׁהָעֲזָרָה דִּין רְשׁוּת הָרַבִּים {ד} יֵשׁ לָהּ לְעִנְיַן טֻמְאָה, וּסְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים סְפֵקוֹ טָהוֹר: <B>וְהַבָּשָׂר טָמֵא.</B> דְּוַדַּאי נָגְעָה בּוֹ טֻמְאָה. וּמַתְנִיתִין אַיְרֵי בְּבֶהֱמַת קָדָשִׁים שֶׁהֶעֱבִירָהּ בַּנָּהָר סָמוּךְ לִשְׁחִיטָתָהּ וַעֲדַיִן מַשְׁקֶה טוֹפֵחַ עָלֶיהָ {ה}, שֶׁהֻכְשַׁר הַבָּשָׂר לְקַבֵּל טֻמְאָה בְּאוֹתָן הַמַּיִם. דְּאִי לָאו הָכִי לֹא הָיָה הַבָּשָׂר טָמֵא, וְאַף עַל פִּי שֶׁרוֹחֲצִים אוֹתוֹ בְּבֵית הַמִּטְבְּחַיִם בָּעֲזָרָה, דְּכָל מַשְׁקֵה בֵית מִטְבְּחַיָּא דָכָן, וְאֵין מַכְשִׁירִין אֶת הַבָּשָׂר לְקַבֵּל טֻמְאָה {ו}. וְאִי קַשְׁיָא הֵיאַךְ הַבָּשָׂר טָמֵא וְהַיָּדַיִם טְהוֹרוֹת, וַהֲלֹא אֳכָלִין טְמֵאִין מְטַמְּאִין אֶת הַיָּדַיִם מִדְּרַבָּנָן. הָא לֹא קַשְׁיָא מִידֵי, דְּקַיְמָא לָן אֵין טֻמְאַת יָדַיִם בַּמִּקְדָּשׁ, שֶׁכְּשֶׁגָּזְרוּ עַל טֻמְאַת יָדַיִם לֹא גָּזְרוּ בָּהּ בַּמִּקְדָּשׁ:

<B>בַּכֶּרֶם.</B> שֶׁהָיוּ יוֹשְׁבִין שׁוּרוֹת שׁוּרוֹת שֶׁל חֲכָמִים כְּכֶרֶם {ח} זוֹ הַנְּטוּעָה שׁוּרוֹת שׁוּרוֹת שֶׁל גְּפָנִים: <B>עַל בֵּיצָה טְרוּפָה.</B> בִּקְעָרָה, שֶׁנִּתְעָרְבָה הַחֶלְבּוֹן וְהַחֶלְמוֹן יַחַד וּנְתוּנָה עַל גַּבֵּי הַיָּרָק, וְנָגַע טְבוּל יוֹם בַּבֵּיצָה, אַף עַל פִּי שֶׁהִיא חֻלִּין, דְּאֵין שַׁיָּךְ תְּרוּמָה בְּבֵיצָה, וּטְבוּל יוֹם אֵינוֹ פּוֹסֵל בְּחֻלִּין, אֲפִלּוּ הָכִי הָוֵי חִבּוּר וְנִפְסַל הַיָּרָק כְּאִלּוּ נָגַע בּוֹ: <B>אִם הָיָה כְמִין כּוֹבַע.</B> שֶׁנִּתְנַפְּחָה הַבֵּיצָה וְנַעֲשֵׂית כְּמִין כּוֹבַע עַל הַיָּרָק וַחֲלוּלָה תַּחְתֶּיהָ: <B>שִׁבֹּלֶת שֶׁבַּקָּצִיר.</B> שִׁיֵּר בִּקְצִירָתוֹ שִׁבֹּלֶת אַחַת שֶׁלֹּא קָצַר, וְרֹאשׁ אוֹתָהּ שִׁבֹּלֶת מַגִּיעַ לַקָּמָה: <B>אִם נִקְצְרָה.</B> אִם אוֹתָהּ שִׁבֹּלֶת נִקְצֶרֶת עִם הַקָּמָה, הֲרֵי הִיא שֶׁל בַּעַל הַבַּיִת, שֶׁהַקָּמָה מַצִּילָתָהּ, דְּלֹא קָרִינַן בָּהּ לֹא תָשׁוּב לְקַחְתּוֹ: <B>וְאִם לָאו.</B> הֲרֵי זוֹ שִׁכְחָה וְהַוְיָא לָעֲנִיִּים: <B>עָרִיס.</B> חָמֵשׁ גְּפָנִים הַנְּטוּעוֹת וּמוּדְלוֹת וְשׁוֹכְבוֹת עַל גַּבֵּי כְּלוֹנְסוֹת אוֹ עַל גַּבֵּי גָּדֵר קְרוּיוֹת עָרִיס, לְשׁוֹן עַל עֶרֶשׂ יְצוּעָי: <B>אִם יֶשׁ בָּהּ כִּמְלֹא בוֹצֵר וְסַלּוֹ.</B> אִם הַגִּנָּה גְּדוֹלָה כְּשִׁעוּר שֶׁיַּעֲמֹד בָּהּ בּוֹצֵר הָעֲנָבִים עִם סַלּוֹ שֶׁמַּכְנִיס הָעֲנָבִים לְתוֹכוֹ כְּשֶׁהוּא מְלַקֵּט הָעֲנָבִים מִכָּל צִדְדֵי הֶעָרִיס: <B>תִּזָּרֵעַ.</B> הַגִּנָּה. וְהוּא שֶׁיַּרְחִיק הַזֶּרַע מִן הַגְּפָנִים {ט} כְּדֵי עֲבוֹדַת הַכֶּרֶם שֶׁהֵם שִׁשָּׁה טְפָחִים לְכָל רוּחַ, וְזוֹרֵעַ הַשְּׁאָר: <B>וְאִם לָאו.</B> שֶׁאֵין הַגִּנָּה גְּדוֹלָה כָּל כָּךְ: <B>לֹא תִזָּרֵעַ.</B> וְאַף עַל פִּי שֶׁמַּרְחִיק כְּדֵי עֲבוֹדַת הַכֶּרֶם. מִפְּנֵי שֶׁהַזֶּרַע נִרְאֶה עִם הַכֶּרֶם כְּאִלּוּ הֵם כִּלְאַיִם:

<B>וּפֵרְשָׁן רַבִּי יְהוֹשֻׁעַ.</B> אֵימָתַי חַיָּב וְאֵימָתַי פָּטוּר {יא}: <B>אִם לַעֲשׂוֹת לָהּ פֶּה חַיָּב.</B> מִשּׁוּם בּוֹנֶה {יג}: <B>וְאִם לְהוֹצִיא מִמֶּנָּה לֵחָה פָּטוּר.</B> דִּמְלָאכָה שֶׁאֵינָהּ צְרִיכָה לְגוּפָהּ הִיא, שֶׁהַפֶּתַח הִיא הַמְּלָאכָה, וְזוֹ אֵינָהּ צְרִיכָה לִהְיוֹת לָהּ פֶּתַח מֵעַכְשָׁיו {יד}, וְאֵין כָּאן אֶלָּא אִסּוּרָא דְרַבָּנָן וּמִשּׁוּם צַעֲרָא לֹא גָזוּר, וְהָוֵי פָּטוּר וּמֻתָּר: <B>אִם מִתְעַסֵּק שֶׁלֹּא יִשְּׁכֶנּוּ פָּטוּר.</B> דְּאֵינוֹ צָרִיךְ לְגוּפוֹ שֶׁל דָּבָר הַנִּצּוֹד, וְאִם הָיָה יוֹדֵעַ שֶׁיַּעֲמֹד וְלֹא יִשְּׁכֶנּוּ לֹא הָיָה צָד. וּבְהַאי נַמִּי לֹא גָּזְרוּ רַבָּנָן, וְהָוֵי פָּטוּר וּמֻתָּר: <B>לְפָסִין אִירוֹנִיּוֹת.</B> כְּלֵי חֶרֶס סְתוּמִים עֲשׂוּיִין כְּמִין כַּדּוּר חָלוּל מִבִּפְנִים, וּלְאַחַר שֶׁמְּלַבְּנִים אוֹתוֹ בַּכִּבְשָׁן חוֹתְכִים אוֹתוֹ בְּאֶמְצָעוֹ וְהוּא נַעֲשֶׂה שְׁנֵי כֵּלִים: <B>טְהוֹרוֹת בְּאֹהֶל הַמֵּת.</B> כָּל זְמַן שֶׁלֹּא חֲתָכוּהוּ. דִּכְלִי חֶרֶס אֵין מְקַבֵּל טֻמְאָה מִגַּבּוֹ, אֶלָּא מֵאֲוִירוֹ, כְּדִכְתִיב (בְּמִדְבַּר יט) וְכֹל כְּלִי פָתוּחַ, דֶּרֶךְ פִּתְחוֹ מִטַּמֵּא וְאֵינוֹ מִטַּמֵּא מִגַּבּוֹ, וְהַאי אֵין לוֹ אֲוִיר: <B>וּטְמֵאוֹת בְּמַשָּׂא הַזָּב.</B> בְּטֻמְאוֹת הֶסֵּט. שֶׁאִם נִתְטַלְטְלוּ אוֹ נִתְנַעְנְעוּ מִמַּשָּׂא הַזָּב, טְמֵאִים, אַף עַל פִּי שֶׁאֵין לָהֶם אֲוִיר: <B>מִפְּנֵי שֶׁלֹּא נִגְמְרָה מְלַאכְתָּן.</B> דַּחֲתִיכָה שֶׁחוֹתְכִין אוֹתָן בְּאֶמְצָעָן זוֹ הִיא גְּמַר מְלַאכְתָּן. וְאֵין הֲלָכָה כְּרַבִּי אֶלְעָזָר בַּר צָדוֹק. אֶלָּא לִבּוּנָן בַּכִּבְשָׁן הוּא גְּמַר מְלַאכְתָּן, הִלְכָּךְ מִטַּמְּאוֹת בְּמַשָּׂא הַזָּב אֲפִלּוּ קֹדֶם שֶׁחִלְּקָן:

<B>אֹכֶל פָּרוּד.</B> כְּגוֹן צִבּוּר שֶׁל אֱגוֹזִים בָּטְנִים וּשְׁקֵדִים {א}, שֶׁכְּשֶׁהֵן נִפְרָדִים זֶה מִזֶּה אֵין בָּהֶם כַּבֵּיצָה שֶׁהוּא שִׁעוּר טֻמְאַת אֳכָלִים לְטַמֵּא אֲחֵרִים: <B>אֵינוֹ מִצְטָרֵף.</B> וְאִם הֵן טְמֵאִים וְנָגְעוּ בָּהֶן טָהֳרוֹת כְּשֶׁהֵן צְבוּרִין אֵינָן מִצְטָרְפִין זֶה עִם זֶה לְטַמֵּא הַטָּהֳרוֹת, כֵּיוָן שֶׁאֵין בְּכָל אֶחָד מֵהֶן בִּפְנֵי עַצְמוֹ כְּשִׁעוּר טֻמְאַת אֳכָלִין: <B>אֲסִימוֹן.</B> עָשׂוּי כְּמִדַּת הַמַּטְבֵּעַ הַיּוֹצֵא, אֶלָּא שֶׁעֲדַיִן לֹא טָבְעוּ עָלָיו צוּרָה. וְאֵין פּוֹדִין בּוֹ מַעֲשֵׂר שֵׁנִי, דִּכְתִיב (דְּבָרִים יד) וְצַרְתָּ הַכֶּסֶף בְּיָדְךָ, דָּבָר שֶׁיֵּשׁ עָלָיו צוּרָה {ב}: <B>מַטְבִּילִין יָדַיִם לַחַטָּאת.</B> מִי שֶׁנָּגַע בָּאֳכָלִין וּמַשְׁקִין טְמֵאִים וְכַיּוֹצֵא בָּהֶן מִדְּבָרִים שֶׁמְּטַמְּאִין אֶת הַיָּדַיִם וְאֵין מְטַמְּאִין כָּל הַגּוּף, הֲרֵי זֶה מַטְבִּיל יָדָיו בִּלְבַד, וְכָשֵׁר לִגַּע בְּמֵי חַטָּאת וּלְהַזּוֹת מֵאֵפֶר הַפָּרָה {ג} עַל הַטָּמֵא: <B>אִם נִטְמְאוּ יָדָיו נִטְמָא גוּפוֹ.</B> וְצָרִיךְ טְבִילָה לְכָל הַגּוּף, דְּמַעֲלָה עָשׂוּ בְּמֵי חַטָּאת שֶׁאִם נִטְמְאוּ יָדָיו נִטְמָא גּוּפוֹ, מַה שֶּׁאֵין כֵּן בִּתְרוּמָה וְקָדָשִׁים: 

~ פרק ג - משנה ג
<B>מְעֵי אֲבַטִּיחַ.</B> הַזֵּרְעוֹנִים וְהַמַּשְׁקֶה שֶׁבְּתוֹךְ הָאֲבַטִּיחַ: <B>וּקְנִיבַת יָרָק.</B> עָלִין מְעֻפָּשִׁים שֶׁמְּסִירִים מִן הַיָּרָק: <B>מָנֶה וּמָנֶה וּפְרָס.</B> כָּל אַחַת מָנֶה וָחֵצִי. וּבְצִיר מֵהָכִי לֹא חָשֵׁיב גֵּז, שֶׁזּוֹ פָּחוּת שֶׁבַּגִּזּוֹת: <B>כָּל שֶׁהֵן.</B> לָאו דַּוְקָא. דִּבְפָחוֹת מִשִּׁשִּׁים סְלָעִים לֹא מְחַיְּבֵי רַבָּנָן בְּרֵאשִׁית הַגֵּז. אֶלָּא מִשּׁוּם דְּרַבִּי דוֹסָא יָהֵיב שִׁעוּרָא רַבָּה, קָרֵי תַּנָּא לְשִׁעוּרָא זוּטָא דְרַבָּנָן כָּל שֶׁהֵן. וְהַסֶּלַע מִשְׁקָלוֹ עֶשְׂרִים וְאַרְבָּעָה מָעִים, וּמִשְׁקַל כָּל מָעָה שֵׁשׁ עֶשְׂרֵה גַּרְעִינֵי שְׂעֹרָה: 

~ פרק ג - משנה ד
<B>חוֹצָלוֹת.</B> מַחְצְלָאוֹת עֲשׂוּיוֹת מִגֶּמִי וְכַיּוֹצֵא בּוֹ, וְיֵשׁ לָהֶם שָׂפָה סָבִיב {ד}: <B>טְמֵאוֹת טְמֵא מֵת.</B> שֶׁהֲרֵי יֵשׁ לָהֶן בֵּית קִבּוּל. אֲבָל לֹא טֻמְאַת מִשְׁכָּב וּמוֹשָׁב, לְפִי שֶׁאֵין רְאוּיוֹת לִשְׁכִיבָה וְלֹא לִישִׁיבָה: <B>וַחֲכָמִים אוֹמְרִים מִדְרָס.</B> דַּחֲזוּ נַמִּי לִשְׁכִיבָה {ה}: <B>הַקְּלִיעוֹת.</B> חוּטִין מַעֲשֵׂה עֲבוֹת: <B>טְהוֹרוֹת.</B> שֶׁאֵינָן חֲשׁוּבוֹת אָרִיג וְאֵין זֶה בֶּגֶד: <B>חוּץ מִשֶּׁל גַּלְגִּילוֹן.</B> הַקְּלִיעוֹת שֶׁל אַבְנֵט שֶׁחוֹגְרִים בּוֹ, שֶׁבֶּגֶד הוּא חָשׁוּב: <B>כֻּלָּם טְמֵאוֹת.</B> דְּכֻלָּם חֲשׁוּבִים בֶּגֶד, חוּץ מִן הַקְּלִיעָה שֶׁעוֹשִׂים מוֹכְרֵי הַצֶּמֶר לִקְשֹׁר בָּהּ אֲגֻדּוֹת שֶׁל צֶמֶר: 

~ פרק ג - משנה ה
<B>קֶלַע.</B> שֶׁזּוֹרְקִים בּוֹ הָאֲבָנִים. כְּמָה דְאַתְּ אָמַר (שׁוֹפְטִים כ) קֹלֵעַ בָּאֶבֶן. (שְׁמוּאֵל א כה) יְקַלְּעֶנָּה בְּתוֹךְ כַּף הַקָּלַע: <B>בֵּית קִבּוּל שֶׁלָּהּ.</B> הַמָּקוֹם שֶׁמְּשִׂימִין בּוֹ הָאֶבֶן שֶׁרוֹצִים לְהַשְׁלִיךְ: <B>בֵּית אֶצְבַּע שֶׁלָּהּ.</B> כְּמִין נֶקֶב קָטָן שֶׁמַּכְנִיסִין בּוֹ הָאֶצְבַּע: <B>בֵּית הַפָּקִיעַ.</B> הַבֵּית יָד שֶׁאוֹחֵז בּוֹ כְּשֶׁמְּסַבֵּב הַקֶּלַע בָּאֲוִיר וְרוֹצֶה לְהַפְקִיעַ {ח} וּלְהַשְׁלִיךְ הָאֶבֶן: 

~ פרק ג - משנה ו
<B>הַשְּׁבוּיָה.</B> אֵשֶׁת כֹּהֵן שֶׁנִּשְׁבֵּית: <B>אוֹכֶלֶת בַּתְּרוּמָה.</B> וְלֹא חָיְשִׁינַן שֶׁמָּא הַנָּכְרִי בָּא עָלֶיהָ וַעֲשָׂאָהּ זוֹנָה וּפְסָלָהּ מִן הַתְּרוּמָה: <B>שֶׁהַפֶּה שֶׁאָסַר הוּא הַפֶּה שֶׁהִתִּיר.</B> כְּשֵׁם שֶׁהֶאֱמִינוּהָ כְּשֶׁאָמְרָה נִשְׁבֵּיתִי, כָּךְ יֵשׁ לְהַאֲמִינָהּ כְּשֶׁאוֹמֶרֶת וּטְהוֹרָה אֲנִי: 

~ פרק ג - משנה ז
<B>הַטָּמֵא עוֹמֵד.</B> מְצֹרָע עוֹמֵד תַּחַת הָאֹהֶל אוֹ תַּחַת הָאִילָן: <B>וְטָהוֹר עוֹבֵר.</B> דֶּרֶךְ שָׁם. סָפֵק נָגַע סָפֵק לֹא נָגַע, אוֹ סָפֵק הֶאֱהִיל הָאִילָן וְהָאֹהֶל עַל שְׁנֵיהֶם וְנִטְמָא הַטָּהוֹר, סָפֵק לֹא הֶאֱהִיל וְלֹא נִטְמָא, הֲרֵי סָפֵק אֶחָד. אוֹ שֶׁהָיָה הַטָּהוֹר עוֹמֵד וְהַמְצֹרָע עוֹבֵר, זֶה סָפֵק שֵׁנִי: <B>טֻמְאָה בִּרְשׁוּת הַיָּחִיד וְטָהֳרָה בִּרְשׁוּת הָרַבִּים.</B> כְּגוֹן חֲנוּת הַפְּתוּחָה לִרְשׁוּת הָרַבִּים, סָפֵק נִכְנַס סָפֵק לֹא נִכְנַס: <B>אוֹ שֶׁטָּהֳרָה בִּרְשׁוּת הַיָּחִיד וְהַטֻּמְאָה בִּרְשׁוּת הָרַבִּים.</B> הֲרֵי שְׁנֵי סְפֵקוֹת אֲחֵרוֹת: <B>רַבִּי יְהוֹשֻׁעַ מְטַמֵּא.</B> דְּסָבַר הוֹאִיל וּרְשׁוּת הַיָּחִיד מְשֻׁתֶּפֶת בִּסְפֵק טֻמְאָה זוֹ, הָוֵי סְפֵק טֻמְאָה בִּרְשׁוּת הַיָּחִיד וּסְפֵקוֹ טָמֵא: <B>וַחֲכָמִים מְטַהֲרִין.</B> דְּסָבְרֵי הוֹאִיל וּסְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים מְשֻׁתֶּפֶת בָּהּ, הָוֵי סְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים וּסְפֵקוֹ טָהוֹר. וְאֵין הֲלָכָה כְּרַבִּי יְהוֹשֻׁעַ: 

~ פרק ג - משנה ח
<B>מַסְמֵר הַשֻּׁלְחָנִי.</B> שֶׁהַשֻּׁלְחָנִי תּוֹלֶה בּוֹ כַּף הַמֹּאזְנַיִם. וְרַבּוֹתֵינוּ פֵּרְשׁוּ, מַסְמֵר קְבוּעָה בָּעַמּוּד שֶׁלִּפְנֵי הַשֻּׁלְחָנִי לְהַעֲמִיד בּוֹ תְּרִיסֵי הַחֲנוּת. וּמְטַהֲרִין רַבָּנָן מִשּׁוּם דְּתַשְׁמִישׁוֹ עִם הַקַּרְקַע: <B>וְאָרוֹן שֶׁל גָּרוֹסוֹת.</B> הָעוֹשִׂים גְּרִיסִים שֶׁל פּוֹל בָּרֵחַיִם שֶׁלָּהֶם, יֵשׁ לָהֶן כְּמִין אָרוֹן שֶׁל עֵץ: <B>אֶבֶן הַשָּׁעוֹת.</B> אֶבֶן שֶׁיֵּשׁ בָּהּ קַוִּים וּרְשׁוּמִים עֲלֵיהֶם שְׁמוֹת הַשָּׁעוֹת וְתָחוּב בָּהּ מַסְמֵר וּבוֹ מְכַוְּנִים אֶת הַשָּׁעוֹת. וְקוֹרִין לוֹ בְּלַעַ"ז אירליי"ו, וּבַעֲרָבִי אלבלאט"ה:
<B>רַבִּי צָדוֹק מְטַמֵּא.</B> אוֹתוֹ אָרוֹן, דְּתוֹרַת כְּלִי עָלָיו: <B>וַחֲכָמִים מְטַהֲרִין.</B> דִּכְלִי עֵץ הֶעָשׂוּי לְנַחַת הוּא וְאֵינוֹ עָשׂוּי לִטַּלְטֵל. וְרַבִּי צָדוֹק סָבַר, פְּעָמִים מְתַקְּנִים אוֹתוֹ לְטַלְטְלוֹ. וְאֵין הֲלָכָה כְּרַבִּי צָדוֹק: 

~ פרק ג - משנה ט
<B>טֶנִי.</B> כְּלִי שֶׁל מַתֶּכֶת שֶׁבַּעֲלֵי בָּתִּים מְשִׂימִים בּוֹ גְּרוּטָאוֹת וְשִׁבְרֵי כְּלֵי מַתָּכוֹת. וְרַבָּן גַּמְלִיאֵל מְטַמֵּא הַכִּסּוּי, דְּסָבַר הוּא נַמִּי חָשׁוּב כְּלִי, מִשּׁוּם דְּכִסּוּי נַמִּי יֵשׁ לוֹ בֵּית קִבּוּל. וַחֲכָמִים מְטַהֲרִין, דְּסָבְרֵי כֵּיוָן דְּלֹא נַעֲשָׂה אֶלָּא לְכַסּוֹת אֵינוֹ חָשׁוּב כְּלִי: <B>וּתְלוֹי הַמַּגְרֵדוֹת.</B> מַסְמְרִים שֶׁתּוֹלִים בָּהֶם הַמַּגְרֵדוֹת בְּבָתֵּי מֶרְחֲצָאוֹת, וְהַנִּכְנָסִים שָׁם כָּל אֶחָד לוֹקֵחַ מַגְרֶדֶת וּמְגָרֵד בָּהּ רַגְלָיו וְגוּפוֹ: <B>וְגָלְמֵי כְלֵי מַתָּכוֹת.</B> שֶׁצְּרִיכִים לָשׁוּף לְגָרֵד לְהַקִּישׁ בְּקֻרְנָס, אוֹ שֶׁמְּחֻסַּר אֹגֶן אוֹ אֹזֶן, אֵינָן טְמֵאִים לְדִבְרֵי חֲכָמִים עַד שֶׁתִּגָּמֵר מְלַאכְתָּן: <B>טַבְלָא שֶׁנֶּחְלְקָה לִשְׁנַיִם.</B> טַבְלָא שֶׁל חֶרֶס שֶׁיֵּשׁ בָּהּ לִבְזְבְּזִין, וְנַעֲשֵׂית שְׁתֵּי חֲתִיכוֹת שָׁווֹת שֶׁאֵין אַחַת מֵהֶן גְּדוֹלָה מֵחֲבֶרְתָּהּ, בְּהָא פְלִיגֵי רַבָּן גַּמְלִיאֵל וַחֲכָמִים. וַהֲלָכָה כַּחֲכָמִים:

<B>בֵּיצָה שֶׁנּוֹלְדָה בְיוֹם טוֹב.</B> בְּיוֹם טוֹב שֶׁל אַחַר שַׁבָּת עָסְקִינַן. וְטַעֲמַיְהוּ דְבֵית הִלֵּל דְּאָמְרֵי לֹא תֵּאָכֵל, מִשּׁוּם דְּכָל בֵּיצָה דְמִתְיַלְּדָא הָאִידְנָא מֵאֶתְמוֹל גָּמְרָה לַהּ, וְנִמְצֵאת שַׁבָּת מְכִינָה לְיוֹם טוֹב, וְהַתּוֹרָה אָמְרָה (שְׁמוֹת טז) וְהָיָה בַּיּוֹם הַשִּׁשִּׁי וְהֵכִינוּ, וּסְתָם שִׁשִּׁי חֹל הוּא, חֹל מֵכִין לְשַׁבָּת וְחֹל מֵכִין לְיוֹם טוֹב, דְּיוֹם טוֹב נַמִּי אִקְרֵי שַׁבָּת, וְאֵין יוֹם טוֹב מֵכִין לְשַׁבָּת וְלֹא שַׁבָּת מְכִינָה לְיוֹם טוֹב. וַהֲכָנָה דְבֵיצָה אַף עַל גַּב דְּבִידֵי שָׁמַיִם הִיא מִקַּרְיָא הֲכָנָה: <B>שְׂאֹר בְּכַזַּיִת.</B> לְעִנְיַן אֲכִילָה כֻלֵּי עָלְמָא לֹא פְלִיגֵי דְּזֶה וָזֶה בְּכַזַּיִת, מִדְּפָתַח הַכָּתוּב בִּשְׂאֹר, שְׂאֹר לֹא יִמָּצֵא בְּבָתֵּיכֶם (שָׁם יב), וְסִיֵּם בְּחָמֵץ, כִּי כָּל אֹכֵל מַחְמֶצֶת, לוֹמַר לָךְ זֶהוּ שְׂאֹר זֶהוּ חָמֵץ, כָּזֶה כֵּן זֶה. כִּי פְלִיגֵי לְעִנְיַן בִּעוּר, בֵּית שַׁמַּאי סָבְרֵי מִדְּכָתַב רַחֲמָנָא לְתַרְוַיְהוּ שְׂאֹר וְחָמֵץ, שְׁמַע מִנַּהּ שִׁעוּרוֹ שֶׁל זֶה לֹא כְּשִׁעוּרוֹ שֶׁל זֶה, וְלֹא יָלְפִינַן בִּעוּר מֵאֲכִילָה. וּבֵית הִלֵּל סָבְרֵי יָלְפִינַן בִּעוּר מֵאֲכִילָה:

<B>הַכֹּל מוֹדִים שֶׁהִיא מֻתֶּרֶת.</B> וְהוּא שֶׁיּוֹדֵעַ בָּעֻבָּר שֶׁכָּלוּ לוֹ חֳדָשָׁיו {ב}: <B>וְאֶפְרוֹחַ שֶׁיָּצָא.</B> הַכֹּל מוֹדִים שֶׁהוּא אָסוּר, מִשּׁוּם שֶׁרֶץ הָעוֹף {ג}: <B>הַשּׁוֹחֵט חַיָּה וָעוֹף.</B> הַבָּא לִשְׁחֹט חַיָּה וָעוֹף בְּיוֹם טוֹב, וְנִמְלַךְ בְּבֵית דִּין כֵּיצַד יַעֲשֶׂה: <B>בֵּית שַׁמַּאי אוֹמְרִים.</B> בֵּית דִּין מוֹרִים לוֹ שֶׁיִּשְׁחֹט לְכַתְּחִלָּה וַיַּחְפֹּר בְּדֶקֶר נָעוּץ שֶׁיֵּשׁ לוֹ מִבְּעוֹד יוֹם, כְּלוֹמַר שֶׁיַּעֲקֹר אוֹתוֹ מִמְּקוֹם נְעִיצָתוֹ וְיַעֲלֶה עָפָר וִיכַסֶּה בּוֹ. וּמַיְרֵי כְּשֶׁהוּא נָעוּץ בְּעָפָר תִּחוֹחַ הָרָאוּי לְכַסּוֹת, שֶׁאֵינוֹ מְחֻסַּר כְּתִישָׁה: <B>דֶּקֶר.</B> יָתֵד שֶׁנּוֹעָצִין בָּאָרֶץ. לְשׁוֹן וַיִּדְקֹר אֶת שְׁנֵיהֶם [בְּמִדְבָּר כה]: <B>שֶׁאֵפֶר כִּירָה מוּכָן הוּא.</B> לָאו אַמִּלְּתַיְהוּ דְּבֵית שַׁמַּאי וּבֵית הִלֵּל קָאֵי, אֶלָּא מִלְּתָא בְּאַנְפֵּי נַפְשָׁהּ הִיא. וְהָכִי קָאָמַר, וְאֵפֶר כִּירָה מוּכָן הוּא, וְלֹא שָׁנוּ אֶלָּא שֶׁהֻסַּק מֵעֶרֶב יוֹם טוֹב, אֲבָל הֻסַּק בְּיוֹם טוֹב, אָסוּר, דְּלֵיכָּא לְמֵימַר מֵאֶתְמוֹל דַּעְתֵּיהּ עֲלוֹיֵהּ. וְאִם רָאוּי לִצְלוֹת בּוֹ בֵּיצָה, שֶׁעֲדַיִן הוּא רֶמֶץ חַם, מֻתָּר לְכַסּוֹת בּוֹ, אַיְדֵי דַּחֲזִי לְהַפּוֹכֵי בּוֹ לִצְלוֹת בּוֹ בֵּיצָה, שָׁקֵיל לֵיהּ נַמִּי וּמְכַסֶּה בּוֹ:

<B>הֶפְקֵר לָעֲנִיִּים הֶפְקֵר.</B> מִי שֶׁהִפְקִיר לָעֲנִיִּים וְלֹא לָעֲשִׁירִים, דִּין הֶפְקֵר יֵשׁ לוֹ, וְאֵינוֹ חַיָּב בְּלֶקֶט שִׁכְחָה וּפֵאָה וְלֹא בְּמַעַשְׂרוֹת, דִּכְתִיב בְּלֶקֶט וּפֵאָה (וַיִּקְרָא יט) לֶעָנִי וְלַגֵּר תַּעֲזֹב אוֹתָם, מַה תַּלְמוּד לוֹמַר תַּעֲזֹב אֹתָם, לִמֵּד עַל עֲזִיבָה אַחֶרֶת, דְּהַיְנוּ הֶפְקֵר שֶׁהִיא כְּזוֹ, מַה זּוֹ לָעֲנִיִּים וְלֹא עֲשִׁירִים, אַף הֶפְקֵר לָעֲנִיִּים: <B>עַד שֶׁיַּפְקִיר אַף לָעֲשִׁירִים כַּשְּׁמִטָּה.</B> דִּכְתִיב (שְׁמוֹת כג) וְהַשְּׁבִיעִת תִּשְׁמְטֶנָּה וּנְטַשְׁתָּהּ, מַה תַּלְמוּד לוֹמַר וּנְטַשְׁתָּהּ, לִמֵּד עַל נְטִישָׁה אַחֶרֶת, דְּהַיְנוּ הֶפְקֵר, שֶׁהִיא כַּשְּׁבִיעִית, מַה שְּׁבִיעִית לָעֲנִיִּים וְלָעֲשִׁירִים אַף הֶפְקֵר לָעֲנִיִּים וְלָעֲשִׁירִים: <B>כָּל עָמְרֵי הַשָּׂדֶה שֶׁל קַב קַב.</B> אִם הָיוּ כָּל עָמְרֵי הַשָּׂדֶה כָּל אֶחָד מֵהֶן קַב וְאֶחָד שֶׁל אַרְבַּעַת קַבִּין, וּשְׁכָחוֹ, הֲרֵי זֶה שִׁכְחָה, יָתֵר עַל כֵּן אֵינוֹ שִׁכְחָה. וְכֵן אִם הָיוּ כָּל הָעֳמָרִים שְׁנֵי קַבִּין וְאֶחָד מִשְּׁמֹנָה קַבִּין, וּשְׁכָחוֹ, הֲרֵי זֶה שִׁכְחָה, יָתֵר עַל כֵּן אֵינָהּ שִׁכְחָה: 

~ פרק ד - משנה ד
<B>לַגַּפָּה.</B> גֶּדֶר אֲבָנִים סְדוּרוֹת זוֹ עַל זוֹ בְּלֹא טִיט: <B>וְלַכֵּלִים.</B> כְּלֵי הַמַּחֲרֵשָׁה: <B>בֵּית שַׁמַּאי אוֹמְרִים אֵינוֹ שִׁכְחָה.</B> פְּלוּגְתַּיְהוּ דְבֵית שַׁמַּאי וּבֵית הִלֵּל בְּעֹמֶר שֶׁהֶחֱזִיק בּוֹ לְהוֹלִיכוֹ לָעִיר וּנְתָנוֹ בְּצַד הַגַּפָּה אוֹ בְּצַד הַגָּדִישׁ וּשְׁכָחוֹ שָׁם, שֶׁבֵּית שַׁמַּאי אוֹמְרִים אֵינוֹ שִׁכְחָה, שֶׁהֲרֵי זָכָה בּוֹ, וּבֵית הִלֵּל אוֹמְרִים שִׁכְחָה. פֵּרוּשׁ אַחֵר, בֵּית שַׁמַּאי אוֹמְרִים אֵינוֹ שִׁכְחָה, אֲפִלּוּ בְּעֹמֶר שֶׁלֹּא הֶחֱזִיק בּוֹ כְּלָל לֹא הָוֵי שִׁכְחָה, דְּהוֹאִיל וְהִנִּיחוֹ אֵצֶל דָּבָר הַמְסֻיָּם עָתִיד לְזָכְרוֹ. וּבֵית הִלֵּל אוֹמְרִים שִׁכְחָה, כָּל זְמַן שֶׁלֹּא הֶחֱזִיק בּוֹ. וּמוֹדִים בֵּית הִלֵּל שֶׁאִם הֶחֱזִיק בּוֹ וְאַחַר כָּךְ שְׁכָחוֹ שֶׁאֵינוֹ שִׁכְחָה: 

~ פרק ד - משנה ה
<B>כֶּרֶם רְבָעִי.</B> שֶׁהוּא צָרִיךְ פִּדְיוֹן אִם בָּא לְאָכְלוֹ חוּץ לִירוּשָׁלַיִם. וְהוּא הַדִּין לְכָל עֵץ מַאֲכָל: <B>אֵין לוֹ חֹמֶשׁ.</B> דְּלֹא כָּתְבָה תּוֹרָה בּוֹ שֶׁהַבְּעָלִים מוֹסִיפִין אֶת הַחֹמֶשׁ כַּדֶּרֶךְ שֶׁכָּתְבָה בְּמַעֲשֵׂר שֵׁנִי: <B>וְאֵין לוֹ בִעוּר.</B> אֵינוֹ חַיָּב לְבַעֲרוֹ מִן הַבַּיִת בְּעֶרֶב פֶּסַח שֶׁל רְבִיעִית וְשֶׁל שְׁבִיעִית כְּשֶׁמְּבַעֵר הַמַּעַשְׂרוֹת כְּדִכְתִיב (דְּבָרִים כו) בִּעַרְתִּי הַקֹּדֶשׁ מִן הַבַּיִת: <B>יֶשׁ לוֹ חֹמֶשׁ וְיֶשׁ לוֹ בִעוּר.</B> בֵּית הִלֵּל יָלְפֵי קֹדֶשׁ קֹדֶשׁ מִמַּעֲשֵׂר, מַה מַּעֲשֵׂר יֵשׁ לוֹ חֹמֶשׁ וְיֵשׁ לוֹ בִּעוּר, אַף כֶּרֶם רְבָעִי יֵשׁ לוֹ חֹמֶשׁ וְיֵשׁ לוֹ בִּעוּר. וּבֵית שַׁמַּאי לֹא יָלְפֵי קֹדֶשׁ קֹדֶשׁ מִמַּעֲשֵׂר: <B>יֶשׁ לוֹ פֶרֶט וְיֶשׁ לוֹ עוֹלְלוֹת.</B> דִּכְחֻלִּין חָשְׁבֵי לֵיהּ: <B>וְהָעֲנִיִּים פּוֹדִין לְעַצְמָן.</B> הַפֶּרֶט וְהָעוֹלְלוֹת שֶׁלָּקְטוּ, וְאוֹכְלִין אוֹתָן בִּמְקוֹמָן, וּמַעֲלִין הַדָּמִים לִירוּשָׁלַיִם: <B>וּבֵית הִלֵּל אוֹמְרִים כֻּלּוֹ לַגָּת.</B> מִשּׁוּם דְּיָלְפֵי מִמַּעֲשֵׂר, וּסְבִירָא לְהוּ מַעֲשֵׂר שֵׁנִי מָמוֹן גָּבוֹהַּ הוּא, הִלְכָּךְ אֵין לָעֲנִיִּים חֵלֶק בּוֹ, וְדוֹרְכִים הָעוֹלְלוֹת עִם שְׁאָר הַיַּיִן, וְהַבְּעָלִים מַעֲלִין הַכֹּל לִירוּשָׁלָיִם:

<B>מְגֻלְגָּלִים.</B> כְּבוּשִׁים בְּמֶלַח כְּדֵי לְמַתְּקָן: <B>אֵינוֹ צָרִיךְ לְנַקֵּב.</B> הֶחָבִית, וְאַף עַל פִּי שֶׁהַמֹּהַל {ו} הַיּוֹצֵא מֵהֶן צָף עַל גַּבָּן, אֵינוֹ מַכְשִׁירָן לְקַבֵּל טֻמְאָה, דְּלֹא נִיחָא לֵיהּ בְּאוֹתוֹ מֹהַל הַיּוֹצֵא, וַאֲנָן בָּעִינַן מַשְׁקֶה דְנִיחָא לֵיהּ, דִּכְתִיב (וַיִּקְרָא יא) וְכִי יֻתַּן מַיִם עַל זֶרַע, כְּתִיב יִתֵּן, וְקָרִינַן יֻתַּן, מַה יִּתֵּן דְּנִיחָא לֵיהּ אַף יֻתַּן דְּנִיחָא לֵיהּ: <B>צָרִיךְ לְנַקֵּב.</B> לַעֲשׂוֹת מַעֲשֶׂה לְגַלּוֹת דַּעְתֵּיהּ דְּלֹא נִיחָא לֵיהּ בְּאוֹתוֹ מֹהַל שֶׁיִּהְיֶה צָף עַל גַּבֵּי הַזֵּיתִים וְרוֹצֶה הוּא שֶׁיֵּצֵא דֶּרֶךְ נֶקֶב שֶׁעוֹשֶׂה בֶּחָבִית: <B>וּסְתָמוּהָ שְׁמָרִים שֶׁהִיא טְהוֹרָה.</B> שֶׁכֵּיוָן שֶׁנִּקְּבָהּ, גָּלֵי דַעְתֵּיהּ דְּלֹא נִיחָא לֵיהּ, וְשׁוּב אֵין אוֹתוֹ הַמֹּהַל מַכְשִׁירָן לְקַבֵּל טֻמְאָה: <B>אַף עַל פִּי שֶׁהוּא מְנַטֵּף טָהוֹר.</B> אַף עַל פִּי שֶׁהַשֶּׁמֶן נוֹטֵף מִבְּשָׂרוֹ לְאַחַר שֶׁטָּבַל, טָהוֹר: <B>כְּדֵי סִיכַת אֵבֶר קָטָן.</B> אִם לֹא נִשְׁאַר מִן הַשֶּׁמֶן עַל בְּשָׂרוֹ אֶלָּא כְּדֵי סִיכַת אֵבֶר קָטָן, טָהוֹר. וְיוֹתֵר מִשִּׁעוּר זֶה, טָמֵא, בִּשְׁבִיל הַשֶּׁמֶן שֶׁנִּטְמָא בְּהִטָּמְאוֹ וְנִשְׁאַר עַל בְּשָׂרוֹ וְטִמְּאָהוּ {י}. וְהַשֶּׁמֶן שֶׁעַל בְּשָׂרוֹ לֹא טָהַר בַּמִּקְוֶה, שֶׁאֵין לְךָ מַשְׁקֶה מְקַבֵּל טָהֳרָה בַּמִּקְוֶה אֶלָּא הַמַּיִם בִּלְבַד עַל יְדֵי הַשָּׁקָה: <B>בֵּית שַׁמַּאי אוֹמְרִים כְּדֵי סִיכַת אֵבֶר קָטָן טָהוֹר.</B> טְפֵי מֵהָכִי, טָמֵא: <B>מַשְׁקֶה טוֹפֵחַ וּמַטְפִּיחַ.</B> שֶׁיֵּשׁ לַחְלוּחִית בַּכַּף עַד שֶׁכְּשֶׁיִּגַּע בְּכַף אַחֶרֶת יִדְבַּק בָּהּ לַחְלוּחִית גַּם הִיא. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

שַׁבָּתוֹת לְבֵית שַׁמַּאי, יוֹצִיא וְיִתֵּן כְּתֻבָּה, וַאֲפִלּוּ הָיָה גַּמָּל שֶׁעוֹנָתוֹ לִשְׁלֹשִׁים יוֹם, אוֹ סַפָּן שֶׁעוֹנָתוֹ לְשִׁשָּׁה חֳדָשִׁים: <B>הַמַּפֶּלֶת לְאוֹר שְׁמֹנִים וְאֶחָד.</B> שֶׁיָּלְדָה נְקֵבָה, וְלֵיל שְׁמֹנִים וְאֶחָד, שֶׁהָיְתָה רְאוּיָה לְמָחָר לְהָבִיא כַּפָּרָתָהּ, הִפִּילָה: <B>בֵּית שַׁמַּאי פּוֹטְרִים מִן הַקָּרְבָּן.</B> מִלֵּידָה שְׁנִיָּה, אַף עַל גַּב דִּלְאַחַר מְלֹאת הוּא, הוֹאִיל וְלַיְלָה הִיא וְלֹא יָצְאָה לְשָׁעָה רְאוּיָה לְקָרְבָּן, דְּלַיְלָה מְחֻסַּר זְמַן קָרְבָּן הוּא דִּכְתִיב (וַיִּקְרָא ז) בְּיוֹם צַוֹּתוֹ, הִלְכָּךְ לְעִנְיַן קָרְבָּן כְּיוֹם מְלֹאת דָּמֵי: <B>וּבֵית הִלֵּל מְחַיְּבִין.</B> הוֹאִיל וּלְאַחַר מְלֹאת הוּא דְּהִפִּילָה: <B>סָדִין.</B> שֶׁל פִּשְׁתִּים: <B>בֵּית שַׁמַּאי פּוֹטְרִין.</B> לְפִי שֶׁדֶּרֶךְ סָדִין שֶׁלּוֹבְשִׁין אוֹתוֹ בַּלַּיְלָה, וְצִיצִית שֶׁיֵּשׁ בּוֹ פְּתִיל תְּכֵלֶת שֶׁהוּא צֶמֶר צָבוּעַ בְּדַם חִלָּזוֹן הָוֵי כִּלְאַיִם בְּסָדִין שֶׁל פִּשְׁתִּים. וּבַיּוֹם, שֶׁהוּא מִצְוָתוֹ, אָתֵי עֲשֵׂה דְצִיצִית וְדָחֵי לֹא תַעֲשֶׂה דְּלֹא תִלְבַּשׁ שַׁעַטְנֵז {יד}, אֲבָל בַּלַּיְלָה, שֶׁאֵין מִצְוַת צִיצִית נוֹהֶגֶת בַּלַּיְלָה, דִּכְתִיב וּרְאִיתֶם אֹתוֹ, פְּרָט לִכְסוּת לַיְלָה, אִי לָבֵישׁ צִיצִית בַּסָּדִין, מִחַיַּב מִשּׁוּם לֹא תִלְבַּשׁ שַׁעַטְנֵז. וְסָבְרֵי בֵּית שַׁמַּאי גָּזְרִינַן סָדִין בְּצִיצִית אֲפִלּוּ בַּיּוֹם דְּלֹא מִחַיַּב, גְּזֵרָה מִשּׁוּם כְּסוּת לַיְלָה דְּמִחַיַּב מִשּׁוּם כִּלְאַיִם {טו}. וּבֵית הִלֵּל סָבְרֵי לֹא גָּזְרִינַן: <B>כַּלְכָּלַת הַשַּׁבָּת.</B> כַּלְכָּלָה מְלֵאָה פֵּרוֹת שֶׁיִּחֲדָהּ לְשַׁבָּת: <B>בֵּית הִלֵּל מְחַיְּבִין.</B> בְּמַעֲשֵׂר מִיָּד, אֲפִלּוּ קֹדֶם הַשַּׁבָּת, דְּכֵיוָן שֶׁיִּחֲדָהּ לְשַׁבָּת הֻקְבְּעָה מִיָּד. וּבֵית שַׁמַּאי סָבְרֵי דְּאֵין הַשַּׁבָּת קוֹבַעַת אֶלָּא לְאַחַר שֶׁתִּכָּנֵס:

<B>וְאַחַר כָּךְ בָּא לָאָרֶץ.</B> שֶׁאֵין נְזִירוּת נוֹהֶגֶת אֶלָּא בָּאָרֶץ, מִשּׁוּם טֻמְאַת אֶרֶץ הָעַמִּים. וּמִי שֶׁנָּדַר נְזִירוּת בְּחוּץ לָאָרֶץ מְחַיְּבִין אוֹתוֹ לַעֲלוֹת לְאֶרֶץ יִשְׂרָאֵל וְלִהְיוֹת נוֹהֵג שָׁם נְזִירוּתוֹ: <B>נָזִיר שְׁלֹשִׁים יוֹם.</B> קָנְסִינַן לֵיהּ שֶׁיְּקַיֵּם סְתָם נְזִירוּת בְּאֶרֶץ יִשְׂרָאֵל, שֶׁהוּא שְׁלֹשִׁים יוֹם: <B>נָזִיר בַּתְּחִלָּה.</B> צָרִיךְ לִנְהֹג בְּאֶרֶץ יִשְׂרָאֵל כְּמִנְיַן הַיָּמִים שֶׁנָּדַר בַּנְּזִירוּת. וְהַיָּמִים שֶׁנָּהַג נְזִירוּת בְּחוּץ לָאָרֶץ כְּאִלּוּ לֹא נָהַג בָּהֶם נְזִירוּת כְּלָל: <B>אֵלּוּ מְעִידִין שֶׁנָּדַר שְׁתַּיִם.</B> שְׁתֵּי נְזִירוּת: <B>וְאֵלּוּ מְעִידִין שֶׁנָּדַר חָמֵשׁ.</B> בְּאוֹתָהּ שָׁעָה שֶׁאַתֶּם אוֹמְרִים שֶׁנָּדַר שְׁתַּיִם אָנוּ מְעִידִים שֶׁנָּדַר חָמֵשׁ נְזִירֻיּוֹת. וְהוּא אוֹמֵר שֶׁלֹּא נָדַר כְּלָל: <B>נֶחְלְקָה הָעֵדוּת.</B> הוֹאִיל וּמַכְחִישׁוֹת זוֹ אֶת זוֹ נִתְבַּטְּלוּ דִּבְרֵיהֶם וְאֵין כָּאן עֵדוּת כְּלָל: <B>וּבֵית הִלֵּל אוֹמְרִים יֵשׁ בִּכְלָל חָמֵשׁ שְׁתַּיִם.</B> וְיִהְיֶה נָזִיר שְׁתַּיִם:

<B>אָדָם שֶׁהוּא נָתוּן תַּחַת הַסֶּדֶק.</B> אַכְסַדְרָה שֶׁנִּסְדְּקָה תִּקְרָתָהּ וְנֶחְלְקָה לִשְׁנַיִם, וְכֵלִים מִצַּד זֶה וְטֻמְאָה מִצַּד זֶה, הַכֵּלִים טְהוֹרִים, שֶׁאֲוִיר הַסֶּדֶק מַפְסִיק וְגוֹרֵם שֶׁאֵין הַטֻּמְאָה עוֹבֶרֶת לַצַּד הַשֵּׁנִי. וְאִם אָדָם נָתוּן שָׁם בָּאָרֶץ בְּרִצְפַּת הָאַכְסַדְרָה כְּנֶגֶד הַסֶּדֶק, בֵּית שַׁמַּאי אוֹמְרִים אֵינוֹ מֵבִיא אֶת הַטֻּמְאָה. שֶׁאֵין מֵבִיא אֶת הַטֻּמְאָה אֶלָּא דָּבָר שֶׁיֵּשׁ בּוֹ חָלָל טֶפַח: <B>וּבֵית הִלֵּל אוֹמְרִים אָדָם חָלוּל הוּא.</B> וְאַף עַל פִּי שֶׁבְּנֵי מֵעָיו בְּתוֹכוֹ, חָלָל שֶׁבְּתוֹךְ הַגּוּף חָשֵׁיב חָלָל טֶפַח:

<B>רַבִּי יְהוּדָה אוֹמֵר.</B> דַּם נְבֵלוֹת בֵּית שַׁמַּאי מְטַהֲרִין. לְגַמְרֵי, דְּלָא חָשֵׁיב כִּנְבֵלָה: <B>בֵּיצַת נְבֵלָה.</B> כְּגוֹן שֶׁנִּתְנַבְּלָה בַּשְּׁחִיטָה. וְנִמְצְאוּ בָּהּ בֵּיצִים לְאַחַר שְׁחִיטָה: <B>אִם כַּיּוֹצֵא בָהּ נִמְכֶּרֶת בַּשּׁוּק.</B> שֶׁקְּלִפָּתָהּ קָשָׁה וְנִגְמֶרֶת כִּשְׁאָר בֵּיצִים הַנִּמְכָּרוֹת בַּשּׁוּק: <B>וְאִם לָאו.</B> שֶׁאֵינָהּ נִגְמֶרֶת: <B>אֲסוּרָה.</B> דְּכִבְנֵי מֵעֶיהָ הִיא חֲשׁוּבָה: <B>דַּם נָכְרִית בֵּית שַׁמַּאי מְטַהֲרִים.</B> דִּכְתִיב בְּפָרָשַׁת זָב (וַיִּקְרָא טו) דַּבְּרוּ אֶל בְּנֵי יִשְׂרָאֵל, בְּנֵי יִשְׂרָאֵל מְטַמְּאִים בַּזִּיבָה וְאֵין הַנָּכְרִים מְטַמְּאִים בַּזִּיבָה, וְרַבָּנָן גָּזוּר עֲלַיְהוּ שֶׁיִּהְיוּ כְּזָבִים לְכָל דִּבְרֵיהֶן. בֵּית שַׁמַּאי סָבְרֵי, כִּי גָּזוּר רַבָּנָן בְּרֻקָּהּ וּמֵימֵי רַגְלֶיהָ שֶׁהֵן מְצוּיִין תָּדִיר, אֲבָל דַּם הַזִּיבָה שֶׁאֵין מָצוּי כָּל כָּךְ לֹא גָּזְרוּ בָּהּ רַבָּנָן, עֲבַדוּ רַבָּנָן הֶיכֵּרָא וְלֹא טִמְּאוּ דַּם זִיבָה שֶׁל נָכְרִית כְּדֵי שֶׁיֵּדְעוּ דְּטֻמְאַת הַנָּכְרִי דְּרַבָּנָן הִיא, כִּי הֵיכִי דְלָא נִשְׂרְפוּ עֲלַהּ תְּרוּמָה וְקָדָשִׁים: <B>וְדַם טָהֳרָה שֶׁל מְצֹרַעַת.</B> דַּם טֹהַר שֶׁרָאֲתָה הַמְצֹרַעַת שֶׁיָּלְדָה, לְאַחַר שִׁבְעָה לְזָכָר וּשְׁבוּעַיִם לִנְקֵבָה: <B>כְּרֻקָּהּ וּכְמֵימֵי רַגְלֶיהָ.</B> דַּם הַנָּכְרִית מְטַמֵּא כְּרֻקָּהּ וּכְמֵימֵי רַגְלֶיהָ, דְּהַיְנוּ לַח וְלֹא יָבֵשׁ. וְזוֹ הַהֶיכֵּרָא בִּלְבַד עָשׂוּ לָהּ, שֶׁאִם הָיְתָה טֻמְאָה דְּאוֹרַיְתָא הָיָה הַדָּם מְטַמֵּא לַח וְיָבֵשׁ, עַכְשָׁיו שֶׁאֵין מְטַמֵּא אֶלָּא לַח מַכִּירִים דְּטֻמְאָה דְּרַבָּנָן הִיא: <B>בְּטוֹבָה.</B> שֶׁהָאוֹכֵל מַחֲזִיק טוֹבָה לְבַעַל הַפֵּרוֹת שֶׁהֶאֱכִילָהוּ מִפֵּרוֹתָיו: <B>וְשֶׁלֹּא בְטוֹבָה.</B> שֶׁאֵינוֹ מַחֲזִיק לוֹ טוֹבָה: <B>אֵין אוֹכְלִים בְּטוֹבָה {ג}.</B> שֶׁאָסוּר לְהַחֲזִיק טוֹבָה לַבְּעָלִים, דְּרַחְמָנָא אַפְקְרִינְהוּ: <B>הַחֵמֶת.</B> כְּמִין נֹאד שֶׁל עוֹר שֶׁנִּקָּב וּצְרָרוֹ: <B>צְרוּרָה עוֹמֶדֶת.</B> אֵינָהּ מְקַבֶּלֶת טֻמְאָה אֶלָּא אִם כֵּן הִיא צְרוּרָה, כְּלוֹמַר שֶׁהָיְתָה קְשׁוּרָה וְנִקְמַט הָעוֹר וְנִתְקַשָּׁה עַל הַמָּקוֹם הַצָּרוּר עַד שֶׁהוּא עוֹמֵד מֵאֵלָיו וּמַחֲזִיק מַה שֶּׁמְּשִׂימִין בְּתוֹכוֹ: <B>אַף עַל פִּי שֶׁאֵינָהּ צְרוּרָה.</B> שֶׁלֹּא נִקְמַט הָעוֹר מֵאֵלָיו וְלֹא נִתְקַשָּׁה עַל מְקוֹם הַקֶּשֶׁר, הוֹאִיל וּקְשָׁרוֹ וּמַחֲזִיק, הֲרֵי זֶה מְקַבֵּל טֻמְאָה:

<B>הָעוֹף עוֹלֶה עִם הַגְּבִינָה.</B> שֶׁאֵין אִסּוּרוֹ אֶלָּא מִדִּבְרֵי סוֹפְרִים: <B>לֹא עוֹלֶה.</B> גְּזֵרָה שֶׁמָּא יַעֲלֶה גְּבִינָה עִם בְּשַׂר בְּהֵמָה בְּאִלְפָּס רוֹתֵחַ, דְּהוּא אָסוּר מִן הַתּוֹרָה, דְּהַיְנוּ מְבַשֵּׁל: <B>וּבֵית הִלֵּל אוֹמְרִים אֵין תּוֹרְמִין.</B> מִי שֶׁיֵּשׁ לוֹ שֶׁמֶן שֶׁחַיָּב לְהַפְרִישׁ מִמֶּנּוּ תְּרוּמָה וְזֵיתִים שֶׁחַיָּב לְהַפְרִישׁ מֵהֶן תְּרוּמָה, אֵינוֹ יָכוֹל לִתְרֹם מִן הַזֵּיתִים כְּפִי הַשִּׁעוּר שֶׁיֵּשׁ לוֹ לִתְרֹם מִן הַזֵּיתִים וּמִן הַשֶּׁמֶן וְלִפְטֹר הַשֶּׁמֶן בִּתְרוּמַת הַזֵּיתִים, דִּכְתִיב (בְּמִדְבַּר יח) כַּדָּגָן מִן הַגֹּרֶן, מִן הַגָּמוּר עַל הַגָּמוּר וְלֹא מִמַּה שֶּׁאֵינוֹ גָּמוּר עַל הַגָּמוּר: <B>הַזּוֹרֵעַ אַרְבַּע אַמּוֹת בַּכֶּרֶם.</B> שֶׁהוּא הַשִּׁעוּר שֶׁצָּרִיךְ אָדָם לְהַרְחִיק הַזֶּרַע מִן הַכֶּרֶם, וְהַזּוֹרֵעַ בְּתוֹךְ שִׁעוּר זֶה קִדֵּשׁ שׁוּרָה אַחַת שֶׁל כֶּרֶם, כְּדִכְתִיב (דְּבָרִים כב) פֶּן תִּקְדַּשׁ הַמְלֵאָה וְגוֹ' וּתְבוּאַת הַכָּרֶם. וּסְבִירָא לְהוּ לְבֵית שַׁמַּאי דְּשׁוּרָה אַחַת קְרוּיָה כֶּרֶם. וּבֵית הִלֵּל סָבְרֵי דְּאֵין נִקְרָא כֶּרֶם פָּחוֹת מִשְּׁתֵּי שׁוּרוֹת, וְכִי קָאָמַר רַחְמָנָא דִּתְבוּאַת הַכֶּרֶם קָדְּשָׁה, שְׁתֵּי שׁוּרוֹת שֶׁל כֶּרֶם קָאָמַר: <B>הַמְּעִיסָה.</B> קֶמַח שֶׁנְּתָנוּהוּ עַל גַּבֵּי מַיִם רוֹתְחִים וּמִתְבַּשֵּׁל וּמִתְעַבֶּה שָׁם. בֵּית שַׁמַּאי פּוֹטְרִים מֵחַלָּה. וּבֵית הִלֵּל מְחַיְּבִים. וּפְסַק הֲלָכָה בָּזֶה כָּתַבְנוּ בְּרֵישׁ חַלָּה: <B>חַרְדָּלִית.</B> כְּמוֹ הַר דָּלִית. זֶרֶם מַיִם הַבָּא מִדָּלְיוֹ שֶׁל הַר, כְּלוֹמַר מִגֹּבַהּ הָהָר. וַאֲפִלּוּ אֵין מִתְּחִלָּתָן וְעַד סוֹפָן אֶלָּא אַרְבָּעִים סְאָה מַטְבִּילִין בָּהֶן לְבֵית שַׁמַּאי. וּבֵית הִלֵּל אוֹמְרִים אֵין מַטְבִּילִין עַד שֶׁיְּהוּ אַרְבָּעִים סְאָה בְּמָקוֹם אֶחָד, שֶׁאֵין חַרְדָּלִית מְטַהֶרֶת אֶלָּא בְאַשְׁבֹּרֶן: <B>כְּפוֹרֵשׁ מִן הַקָּבֶר.</B> וְצָרִיךְ הַזָּאָה שְׁלִישִׁי וּשְׁבִיעִי. וְלֹא נֶחְלְקוּ בֵּית שַׁמַּאי וּבֵית הִלֵּל אֶלָּא בְּעָרֵל נָכְרִי שֶׁמָּל בְּיוֹם אַרְבָּעָה עָשָׂר, דְּבֵית הִלֵּל סָבְרֵי גְּזֵרָה שֶׁמָּא יִטַּמֵּא לְשָׁנָה הַבָּאָה וְיֹאמַר אֶשְׁתָּקַד לֹא טָהַרְתִּי מִכָּל טֻמְאָה עַד אַרְבָּעָה עָשָׂר וְטָבַלְתִּי וְאָכַלְתִּי, הַשְׁתָּא נַמִּי אֶטְבֹּל וְאֹכַל, וְלֹא יֵדַע דְּאֶשְׁתָּקַד נָכְרִי הָיָה וְלֹא מְקַבֵּל טֻמְאָה, עַכְשָׁיו יִשְׂרָאֵל הוּא וּמְקַבֵּל טֻמְאָה. וּבֵית שַׁמַּאי סָבְרֵי לֹא גָּזְרִינַן. אֲבָל עָרֵל יִשְׂרָאֵל, דִּבְרֵי הַכֹּל טוֹבֵל וְאוֹכֵל פִּסְחוֹ לָעֶרֶב: 

~ פרק ה - משנה ג
<B>קֹהֶלֶת אֵינוֹ מְטַמֵּא אֶת הַיָּדַיִם.</B> מִפְּנֵי שֶׁחָכְמָתוֹ שֶׁל שְׁלֹמֹה הִיא וְלֹא בְּרוּחַ הַקֹּדֶשׁ נֶאֶמְרָה: <B>מְטַמֵּא אֶת הַיָּדָיִם.</B> דְּסָבְרֵי אַף קֹהֶלֶת בְּרוּחַ הַקֹּדֶשׁ נֶאֶמְרָה, הִלְכָּךְ מְטַמֵּא אֶת הַיָּדַיִם כִּשְׁאָר כִּתְבֵי הַקֹּדֶשׁ: <B>שֶׁעָשׂוּ מִצְוָתָן.</B> לְאַחַר שֶׁהִזּוּ אוֹתָם עַל הַטָּמֵא וְטִהַר בָּהֶם, אִם נָטְפוּ מִגּוּפוֹ עַל גַּבֵּי אָדָם אוֹ עַל גַּבֵּי כֵּלִים: <B>הַקֶּצַח.</B> זֶרַע שָׁחֹר שֶׁקּוֹרִין לוֹ נייל"ו בְּלַעַ"ז {ו}. וּרְגִילִים לָתֵת אוֹתוֹ בַּלֶּחֶם, שֶׁהָרָגִיל בּוֹ אֵינוֹ בָּא לִידֵי כְּאֵב לֵב: <B>בֵּית שַׁמַּאי מְטַהֲרִין.</B> דְּלֹא חָשֵׁיב אֹכֶל: <B>וּבֵית הִלֵּל מְטַמְּאִין.</B> כֵּיוָן דִּרְגִילִין לָתֵת אוֹתוֹ בָּאֳכָלִין חָשֵׁיב אֹכֶל: <B>וְכֵן לַמַּעַשְׂרוֹת.</B> כְּמוֹ שֶׁנֶּחְלְקוּ בַּטֻּמְאָה כֵּן נֶחְלְקוּ לַמַּעַשְׂרוֹת, שֶׁהַמְטַמְּאוֹ טֻמְאַת אֳכָלִים מְחַיְּבוֹ בַּמַּעַשְׂרוֹת: 

~ פרק ה - משנה ד
<B>דַּם יוֹלֶדֶת.</B> שֶׁשָּׁהֲתָה שָׁבוּעַ לְזָכָר וּשְׁבוּעַיִם לִנְקֵבָה וְלֹא טָבְלָה: <B>בֵּית שַׁמַּאי אוֹמְרִים כְּרֻקָּהּ וּכְמֵימֵי רַגְלֶיהָ.</B> שֶׁמְּטַמְּאִים לַחִים וְאֵין מְטַמְּאִים יְבֵשִׁים, אַף דָּמָהּ מְטַמֵּא לַח וְאֵינוֹ מְטַמֵּא יָבֵשׁ. וְאֵינוֹ חָשֵׁיב כְּדַם נִדָּה שֶׁמְּטַמֵּא לַח וְיָבֵשׁ: <B>וּבֵית הִלֵּל אוֹמְרִים מְטַמֵּא לַח וְיָבֵשׁ.</B> כָּל זְמַן שֶׁלֹּא טָבְלָה כְּדַם נִדָּה הוּא חָשֵׁיב, וְאַף עַל פִּי שֶׁהוּא בְּתוֹךְ יְמֵי טֹהַר: <B>וּמוֹדִים בְּיוֹלֶדֶת בְּזוֹב.</B> שֶׁצְּרִיכָה לִסְפֹּר שִׁבְעָה יָמִים נְקִיִּים כְּדִין כָּל שְׁאָר זָבוֹת, שֶׁאִם לֹא סָפְרָה וְלֹא טָבְלָה וְרָאֲתָה דָּם בְּתוֹךְ יְמֵי טֹהַר, שֶׁהוּא מְטַמֵּא לַח וְיָבֵשׁ, דְּדַם זִיבָה הוּא חָשֵׁיב כָּל זְמַן שֶׁלֹּא סָפְרָה וְלֹא טָבְלָה: 

~ פרק ה - משנה ה
<B>חוֹלְצוֹת וְלֹא מִתְיַבְּמוֹת.</B> דְּכֵיוָן דְּתַרְוַיְהוּ זְקוּקִין לְהַאי וּלְהַאי, קַמָּא דִמְיַבֵּם פָּגַע בַּאֲחוֹת זְקוּקָתוֹ דְהַוְיָא כְּאִשְׁתּוֹ: <B>מִשּׁוּם בֵּית שַׁמַּאי יְקַיְּמוּ וּבֵית הִלֵּל אוֹמְרִים יוֹצִיאוּ.</B> בַּגְּמָרָא בִּיבָמוֹת פֶּרֶק ד' אַחִין מַפְּכִינַן לַהּ, בֵּית שַׁמַּאי אוֹמְרִים יוֹצִיאוּ וּבֵית הִלֵּל אוֹמְרִים יְקַיְּמוּ. וְכֵן הֲלָכָה, שֶׁאִם קָדְמוּ וְכָנְסוּ יְקַיְּמוּ: 

~ פרק ה - משנה ו
<B>שְׂעַר פְּקֻדָּה.</B> לְשׁוֹן פִּקָּדוֹן, שֶׁהִפְקִידָה הַבַּהֶרֶת אֶת הַשֵׂעָר בְּעוֹר הַבָּשָׂר וְהָלְכָה לָהּ, כְּגוֹן שֶׁהָיְתָה בּוֹ בַּהֶרֶת וּבָהּ שֵׂעָר לָבָן, הָלְכָה הַבַּהֶרֶת וְהִנִּיחָה לְשֵׂעָר לָבָן בִּמְקוֹמוֹ, וְאַחַר כָּךְ חָזְרָה הַבַּהֶרֶת. עֲקַבְיָא בֶּן מַהֲלַלְאֵל מְטַמֵּא, כֵּיוָן שֶׁנֶּהֱפַךְ שֵׂעָר לָבָן בַּבַּהֶרֶת, אַף עַל פִּי שֶׁזּוֹ הַבַּהֶרֶת שֶׁהִיא עַכְשָׁיו אֵינָהּ אוֹתָהּ שֶׁהָפְכָה אוֹתוֹ לְשֵׂעָר לָבָן, טָמֵא: <B>וְדַם הַיָּרוֹק.</B> עֲקַבְיָא בֶּן מַהֲלַלְאֵל מְטַמֵּא, דְּסָבַר כְּקֶרֶן כַּרְכֹּם הָיָה, שֶׁהוּא מִן הַדָּמִים הַטְּמֵאִים, אֶלָּא שֶׁלָּקָה: <B>וַחֲכָמִים מְטַהֲרִין.</B> דִּכְתִיב (וַיִּקְרָא יג) וְהִיא הָפְכָה שֵׂעָר לָבָן, שֶׁהֲפָכַתּוּ הִיא וְלֹא שֶׁהֲפָכַתּוּ חֲבֶרְתָּהּ: <B>הָיָה מַתִּיר שְׂעַר בְּכוֹר בַּעַל מוּם שֶׁנָּשַׁר.</B> הַצֶּמֶר שֶׁנָּשַׁר מִן הַבְּכוֹר בַּעַל מוּם, הָיָה מַתִּירוֹ בַּהֲנָאָה לַכֹּהֵן: <B>וַחֲכָמִים אוֹסְרִים.</B> דְּאִי שָׁרֵית לֵיהּ צֶמֶר הַנּוֹשֵׁר מֵחַיִּים, אָתֵי לְאַשְׁהוֹיֵי לַבְּכוֹר כְּדֵי שֶׁיָּשִׁיר צִמְרוֹ כָּל שָׁעָה, וְאָתוּ בָּהּ לִידֵי תַּקָּלָה שֶׁיִּגְזֹז וְיַעֲבֹד בּוֹ, וּפְסוּלֵי הַמֻּקְדָּשִׁים אֲסִירֵי בְּגִזָּה וַעֲבוֹדָה, דִּכְתִיב (דְּבָרִים יב) מְטַמֵּא בְּאֹהֶל, דִּכְתִיב (שָׁם) וְעַל הַנֹּגֵעַ בַּעֶצֶם, וַהֲלָכָה לְמֹשֶׁה מִסִּינַי שֶׁהוּא מְטַמֵּא בְּכַשְּׂעֹרָה. וּכְשֶׁפֵּרַשׁ מֵאֵבָר מִן הַחַי רַבִּי נְחוּנְיָא מְטַמֵּא וְכוּ': <B>מָצִינוּ אֵבֶר מִן הַחַי כְּמֵת שָׁלֵם.</B> כִּדְיָלְפִינַן מִקְרָא בַּחֲלַל חֶרֶב אוֹ בְּמֵת: <B>אִם טִמֵּאתָ כַּזַּיִת בָּשָׂר הַפּוֹרֵשׁ מִן הַמֵּת.</B> כְּלוֹמַר דִּין הוּא שֶׁכַּזַּיִת בָּשָׂר הַפּוֹרֵשׁ מִן הַמֵּת יְהֵא טָמֵא, שֶׁכֵּן יֵשׁ בּוֹ חֻמְרָא אַחֶרֶת, שֶׁעֶצֶם כַּשְּׂעֹרָה הַפּוֹרֵשׁ מִמֶּנּוּ נַמִּי טָמֵא: <B>אֲבָל נְטַמֵּא כַּזַּיִת בָּשָׂר הַפּוֹרֵשׁ מֵאֵבָר מִן הַחַי.</B> בִּתְמִיָּה. שֶׁהֲרֵי אֵין בּוֹ זֹאת הַחֻמְרָא: <B>שֶׁכֵּן טִהַרְתָּ עֶצֶם כַּשְּׂעֹרָה הַפּוֹרֵשׁ מִמֶּנּוּ.</B> דְּהָא תְּנַן לְעֵיל עֶצֶם כַּשְּׂעֹרָה הַפּוֹרֵשׁ מֵאֵבָר מִן הַחַי רַבִּי אֱלִיעֶזֶר וְרַבִּי יְהוֹשֻׁעַ מְטַהֲרִין. וּמִדִּבְרֵי עַצְמוֹ עָשׂוּ פִּרְכָא לְמַה מָּצִינוּ שֶׁהֵבִיא, וְכֵן לְרַבִּי נְחוּנְיָא: <B>שֶׁהַבָּשָׂר נוֹהֵג בַּנְּבֵלוֹת וּבַשְּׁרָצִים וְכוּ'.</B> דִּכְתִיב (וַיִּקְרָא יא) וְהַנֹּגֵעַ בְּנִבְלָתָם, בְּנִבְלָתָם וְלֹא בָּעֲצָמוֹת וְלֹא בַּקַּרְנַיִם וְלֹא בַּטְּלָפַיִם, אַלְמָא אֵין הָעֲצָמוֹת מְטַמְּאִים מִשּׁוּם נְבֵלָה: <B>אֵבֶר שֶׁיֵּשׁ עָלָיו בָּשָׂר כָּרָאוּי.</B> אֵבֶר אֵינוֹ מְטַמֵּא אֶלָּא אִם כֵּן יֵשׁ בּוֹ בָּשָׂר וְגִידִים וַעֲצָמוֹת, דִּכְתִיב (בְּמִדְבַּר יט) אוֹ בְּעֶצֶם אָדָם, מָה אָדָם שֶׁיֵּשׁ בּוֹ בָּשָׂר וְגִידִים וַעֲצָמוֹת, אַף כָּל שֶׁיֵּשׁ בּוֹ בָּשָׂר וְגִידִים וַעֲצָמוֹת. וְאִם חָסֵר מִן הַבָּשָׂר שֶׁהָיָה עָלָיו וְנִשְׁאַר בּוֹ בָּשָׂר כָּרָאוּי שֶׁיַּעֲלֶה אֲרוּכָה וְיַבְרִיא אִם הָיָה מְחֻבָּר בְּאָדָם חַי, מְטַמֵּא מִשּׁוּם אֵבֶר. וְהַיְנוּ דְּקָאָמַר חָסֵר הַבָּשָׂר טָמֵא. אֲבָל אִם נֶחְסַר כָּל שֶׁהוּא מִן הָעֶצֶם שֶׁבָּאֵבָר, שׁוּב אֵינוֹ מְטַמֵּא מִשּׁוּם אֵבֶר. וְהַיְנוּ דְּקָאָמַר חָסֵר הָעֶצֶם טָהוֹר, כְּלוֹמַר טָהוֹר מִשּׁוּם אֵבֶר אֲבָל טָמֵא מִשּׁוּם בָּשָׂר {ח}, וְאִם כֵּן מָצִינוּ שֶׁמְּרֻבָּה טֻמְאַת בָּשָׂר מִטֻּמְאַת עֶצֶם: <B>וְהוּא כִּבְרִיָתוֹ.</B> בָּשָׂר וְגִידִים וַעֲצָמוֹת: <B>כַּזַּיִת בָּשָׂר מְטַמֵּא בְמַגָּע וּבְמַשָּׂא וּבְאֹהֶל.</B> אָמְרוּ לְפִי שֶׁתְּחִלַּת בְּרִיָּתוֹ שֶׁל אָדָם כַּזַּיִת, לְפִיכָךְ שִׁעוּר טֻמְאָתוֹ בְּכַזַּיִת: <B>וְרֹב עֲצָמוֹת.</B> רֹב מִנְיַן עַצְמוֹתָיו שֶׁל אָדָם, שֶׁמִּנְיַן עַצְמוֹתָיו שֶׁל אָדָם מָאתַיִם אַרְבָּעִים וּשְׁמֹנֶה, נִמְצָא הָרֹב מֵאָה וְעֶשְׂרִים וְחָמֵשׁ: <B>חָסֵר הַבָּשָׂר.</B> מִשִּׁעוּר שֶׁל כַּזַּיִת: <B>טָהוֹר.</B> גָּמוּר מִלְּטַמֵּא, לֹא בְּמַגָּע וְלֹא בְּמַשָּׂא וְלֹא בְּאֹהֶל: <B>חָסֵר רֹב עֲצָמוֹת מְטַמֵּא בְמַגָּע וּבְמַשָּׂא.</B> דְּעֶצֶם כַּשְּׂעֹרָה מְטַמֵּא בְּמַגָּע וּבְמַשָּׂא וְאֵינוֹ מְטַמֵּא בְּאֹהֶל, אַלְמָא מְרֻבָּה טֻמְאַת עֲצָמוֹת מִטֻּמְאַת בָּשָׂר, דְּאִלּוּ עֲצָמוֹת כְּשֶׁנֶּחְסְרוּ מִשִּׁעוּרָן עֲדַיִן נִשְׁאַר בָּהֶן טֻמְאָה, וּבָשָׂר כְּשֶׁנֶּחְסַר מִשִּׁעוּרוֹ טָהוֹר גָּמוּר: <B>רֹב בִּנְיָנוֹ.</B> כְּגוֹן שְׁתֵּי שׁוֹקַיִם וְיָרֵךְ אֶחָד {י}. וְכָל בִּנְיָנוֹ שֶׁל אָדָם הֵן שְׁתֵּי שׁוֹקַיִם וְהַיְרֵכַיִם וְהַצְּלָעוֹת וְהַשִּׁדְרָה: <B>אַף עַל פִּי שֶׁאֵין בָּהֶן רֹבַע.</B> דְּרֹבַע קַב עֲצָמוֹת שֶׁל מֵת מְטַמְּאִים בְּאֹהֶל אַף עַל פִּי שֶׁאֵין בָּהֶם רֹב מִנְיָן וְלֹא רֹב בִּנְיָן. וְכֵן רֹב מִנְיָן אוֹ רֹב בִּנְיָן מְטַמְּאִים אַף עַל פִּי שֶׁאֵין בָּהֶן רֹבַע. אֲבָל בָּשָׂר כְּשֶׁהוּא פָּחוּת מִכַּזַּיִת אֵין לְךָ בּוֹ דָּבָר שֶׁיָּבִיא לוֹ אֶת הַטֻּמְאָה: <B>אִם אֲמַרְתֶּם בַּמֵּת שֶׁיֶּשׁ בּוֹ רֹב וְרֹבַע וְרָקָב.</B> דִּין הוּא שֶׁכַּזַּיִת בָּשָׂר וְעֶצֶם כַּשְּׂעֹרָה הַפּוֹרְשִׁים מִן הַמֵּת שֶׁיִּהְיוּ טְמֵאִים, שֶׁכֵּן יֵשׁ בַּמֵּת חֻמְרוֹת שֶׁל רֹב וְרֹבַע וְרָקָב: <B>תֹּאמְרוּ בַחַי שֶׁאֵין בּוֹ.</B> חֻמְרוֹת הַלָּלוּ, שֶׁאֵינוֹ דִּין שֶׁיִּהְיוּ לֹא כַּזַּיִת בָּשָׂר וְלֹא עֶצֶם כַּשְּׂעֹרָה הַפּוֹרְשִׁים מֵאֵבָר מִן הַחַי טְמֵאִים אֶלָּא טְהוֹרִים. וַהֲלָכָה כְּרַבִּי יְהוֹשֻׁעַ: <B>וְרָקָב.</B> גּוּפוֹ שֶׁל מֵת כְּשֶׁכָּלָה לַחוּתוֹ וְנַעֲשָׂה כְּמִין עָפָר, זֶהוּ רָקָב, שֶׁמְּטַמֵּא כִּמְלֹא תַּרְוָד, וְשִׁעוּרוֹ מְלֹא חָפְנַיִם שֶׁל אָדָם בֵּינוֹנִי. וְאֵין מְלֹא תַּרְוַד רָקָב מְטַמֵּא אֶלָּא מִן הַמֵּת שֶׁנִּקְבַּר עָרֹם בְּאָרוֹן שֶׁל שַׁיִשׁ וּמְכֻסֶּה בְּכִסּוּי שֶׁל שַׁיִשׁ, עַד שֶׁנּוֹדָע בְּוַדַּאי שֶׁאֵין בּוֹ תַּעֲרֹבֶת רִקָּבוֹן שֶׁל בֶּגֶד אוֹ שֶׁל עֵץ אוֹ עָפָר אַחֵר. אֲבָל מֵת שֶׁנִּקְבַּר בִּכְסוּתוֹ אוֹ בְּאָרוֹן שֶׁל עֵץ אוֹ בֶּעָפָר, אֵין לוֹ רָקָב. וְכֵן מֵת שֶׁנִּקְבַּר חֲסַר אֵבֶר אֵין לוֹ רָקָב:

~ פרק ז - משנה א
<B>הֵעִיד רַבִּי יְהוֹשֻׁעַ.</B> פִּדְיוֹן פֶּטֶר חֲמוֹר שֶׁמֵּת. שֶׂה שֶׁהִפְרִישׁוֹ לְפִדְיוֹן פֶּטֶר חֲמוֹר וּמֵת: <B>כְּחָמֵשׁ סְלָעִים.</B> שֶׁל פִּדְיוֹן הַבֵּן שֶׁחַיָּב בְּאַחֲרָיוּתָן: <B>וַחֲכָמִים אוֹמְרִים.</B> אָמַר קְרָא (בְּמִדְבַּר יח) אַךְ פָּדֹה תִּפְדֶּה אֶת בְּכוֹר הָאָדָם וְאֶת בְּכוֹר הַבְּהֵמָה הַטְּמֵאָה תִּפְדֶּה, לִפְדִיָּה הֵקַשְׁתִּיו וְלֹא לְדָבָר אַחֵר: <B>כְּפִדְיוֹן מַעֲשֵׂר שֵׁנִי.</B> דְּאִם אָבַד אֵינוֹ חַיָּב בְּאַחֲרָיוּתוֹ, דְּהַהוּא כֶּסֶף חַיְּבֵיהּ רַחְמָנָא לֶאֱכֹל בִּירוּשָׁלַיִם, וְהָא אָזַל לֵיהּ. וְטַעֲמָא דְרַבִּי אֱלִיעֶזֶר, שֶׁכֵּן מָצִינוּ שֶׁהֵקִישׁ הַכָּתוּב פֶּטֶר חֲמוֹר לִבְכוֹר אָדָם, שֶׁנֶּאֱמַר (שְׁמוֹת לד) וּפֶטֶר חֲמוֹר תִּפְדֶּה בְשֶׂה וְגוֹ' וְכֹל בְּכוֹר בָּנֶיךָ תִּפְדֶּה:

<B>לֹא פָסְלוּ אֶת צִירָן.</B> דְּהֵקֵלּוּ בְּצִיר שֶׁל חֲגָבִים טְמֵאִים שֶׁאֵינוֹ אוֹסֵר תַּעֲרָבְתּוֹ, מִפְּנֵי שֶׁאֵין לָהֶם דָּם וְאֵינוֹ אֶלָּא לַחְלוּחִית בְּעָלְמָא. וְזוֹ הָיְתָה מִשְׁנָה רִאשׁוֹנָה. וְעֵדוּתוֹ שֶׁל רַבִּי צָדוֹק הוֹסִיף עַל מִשְׁנָה רִאשׁוֹנָה לוֹמַר שֶׁהוּא טָהוֹר, דְּלֹא מִבַּעְיָא דְּאֵינוֹ אוֹסֵר אֶת תַּעֲרָבְתּוֹ אֶלָּא אֲפִלּוּ הוּא עַצְמוֹ טָהוֹר: 

~ פרק ז - משנה ג
<B>עַל זוֹחֲלִין שֶׁרַבּוּ עַל הַנּוֹטְפִים.</B> זוֹחֲלִין הֵן כְּגוֹן נְהָרוֹת שֶׁהֵם נִגָּרִים עַל גַּבֵּי קַרְקַע. לְשׁוֹן זֹחֲלֵי עָפָר (דְּבָרִים לב). וְדִינָן כְּמַעְיָן, שֶׁהֵן מְטַהֲרִים בְּזוֹחֲלִין וּבְכָל שֶׁהֵן, וְהֵן כְּשֵׁרִים לְקַדֵּשׁ בָּהֶן מֵי חַטָּאת וְלִטְבִילַת זָבִים. נוֹטְפִים, הֵן כְּגוֹן מֵי גְּשָׁמִים, וְדִינָן כְּמִקְוֶה לְטַהֵר בְּאַרְבָּעִים סְאָה וּבְאַשְׁבֹּרֶן, וּפְסוּלִים לְמֵי חַטָּאת, שֶׁאֵינָן מַיִם חַיִּים, וְכֵן לִטְבִילַת זָבִין. וּכְשֶׁנִּתְעָרְבוּ זוֹחֲלִין בְּנוֹטְפִין וְרַבּוּ הַזּוֹחֲלִין, דִּינָן כְּזוֹחֲלִין, וּכְשֵׁרִים לְקַדֵּשׁ בָּהֶן מֵי חַטָּאת וְלִטְבִילַת זָבִים, וּמְטַהֲרִים בְּזוֹחֲלִים וּבְכָל שֶׁהֵן: <B>בְּבִירַת הַפִּלְיָא.</B> שֵׁם מָקוֹם: 

~ פרק ז - משנה ד
<B>בַּעֲלֵה אֱגוֹז.</B> הַקְּלִפָּה הַחִיצוֹנָה שֶׁל אֱגוֹז, דְּהַיְנוּ הַקְּלִפָּה הַיְרֻקָּה כְּשֶׁהִיא לַחָה. אִם עֲשָׂאָהּ כְּמִין מַרְזֵב לִהְיוֹת הַמַּיִם הַזּוֹחֲלִים נִכְנָסִים בָּהּ וּמְקַלְּחִים מִמֶּנָּה לַחוּץ, דִּין זוֹחֲלִים יֵשׁ לַמַּיִם הַמְקַלְּחִים מִמֶּנָּה, וְהֵם כְּשֵׁרִים לְמֵי חַטָּאת וְלִטְבִילַת זָבִים וְכוּ'. וְלֹא אָמְרִינַן הוֹאִיל וְנִכְנְסוּ בְּתוֹךְ קְלִפָּה זוֹ שֶׁיֵּשׁ לָהּ בֵּית קִבּוּל וּמִשָּׁם מְקַלְּחִין לַחוּץ שׁוּב לֹא יִהְיֶה לָהֶם דִּין זוֹחֲלִים, לְפִי שֶׁאוֹתָהּ קְלִפָּה אֵינָהּ חֲשׁוּבָה כְּלִי: <B>לִפְנֵי לִשְׁכַּת הַגָּזִית.</B> סַנְהֶדְרִי גְּדוֹלָה שֶׁיּוֹשְׁבִים בְּלִשְׁכַּת הַגָּזִית: 

~ פרק ז - משנה ה
<B>קָלָל שֶׁל חַטָּאת.</B> כְּלִי חֶרֶס שֶׁאֵפֶר פָּרָה מֻנָּח בּוֹ. תַּרְגּוּם וְכַדָּהּ עַל שִׁכְמָהּ (בְּרֵאשִׁית כד) וְקוּלְתַהּ עַל כַּתְפַּהּ: <B>שֶׁנְּתָנוֹ עַל גַּבֵּי הַשֶּׁרֶץ.</B> וְלֹא נִטְמָא הַכְּלִי שֶׁאֵפֶר הַפָּרָה בְּתוֹכוֹ, דְּהָא אֵין כְּלִי חֶרֶס מִטַּמֵּא מִגַּבּוֹ, אַף עַל פִּי כֵן הָאֵפֶר טָמֵא, מִשּׁוּם דְּמֻנָּח בְּמָקוֹם טָמֵא, וְהַתּוֹרָה אָמְרָה (בְּמִדְבַּר יט) וְהִנִּיחַ מִחוּץ לַמַּחֲנֶה בְּמָקוֹם טָהוֹר: <B>שֶׁרַבִּי אֱלִיעֶזֶר מְטַהֵר.</B> דְּסָבַר הוֹאִיל וְהַכְּלִי שֶׁהָאֵפֶר בְּתוֹכוֹ טָהוֹר הוּא, שַׁפִּיר קָרֵינָא בֵּיהּ מָקוֹם טָהוֹר. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: <B>עַל מִי שֶׁנָּזַר שְׁתֵּי נְזִירוּת.</B> סְתָם. וְכָל סְתָם נְזִירוּת שְׁלֹשִׁים יוֹם: <B>שֶׁאִם גִּלַּח אֶת הָרִאשׁוֹנָה יוֹם שְׁלֹשִׁים.</B> דִּלְכַתְּחִלָּה לֹא מִבַּעְיָא לֵיהּ לְגַלֵּחַ אֶלָּא יוֹם שְׁלֹשִׁים וְאֶחָד, כְּדֵי שֶׁתִּהְיֶה נְזִירוּת שְׁלֹשִׁים יוֹם גְּמוּרִים. וְאִם גִּלַּח יוֹם שְׁלֹשִׁים, עָלְתָה לוֹ נְזִירוּת, דְּאָמְרִינַן מִקְצָת הַיּוֹם כְּכֻלּוֹ: <B>וְאִם גִּלַּח יוֹם שִׁשִּׁים חָסֵר אֶחָד יָצָא.</B> שֶׁיּוֹם שְׁלֹשִׁים שֶׁל נְזִירוּת רִאשׁוֹנָה עוֹלֶה לְכָאן וּלְכָאן, וְכֵיוָן שֶׁיּוֹם שְׁלֹשִׁים שֶׁל רִאשׁוֹנָה נִמְנֶה גַּם מִן הַנְּזִירוּת הַשְּׁנִיָּה, נִמְצְאוּ שְׁלֹשִׁים יוֹם שֶׁל נְזִירוּת שְׁנִיָּה כָּלִים בְּשִׁשִּׁים חָסֵר אֶחָד: 

~ פרק ז - משנה ו
<B>וְלַד שְׁלָמִים לֹא יִקְרַב שְׁלָמִים.</B> אֶלָּא כּוֹנְסוֹ לְכִפָּה וּמֵת. מִשּׁוּם גְּזֵרָה, דְּאִי אָמַרְתְּ וְלַד שְׁלָמִים יֵשׁ לוֹ תַּקָּנָה, אָתֵי לְאַשְׁהוֹיֵי לָאֵם עַד שֶׁתֵּלֵד וִיגַדֵּל עֲדָרִים מִן הַוְּלָדוֹת וְאָתֵי בְּהוּ לִידֵי גִּזָּה וַעֲבוֹדָה: <B>וְאָכַלְנוּ וְלָדָהּ שְׁלָמִים בֶּחָג.</B> בְּחַג הַשָּׁבוּעוֹת קָאָמַר. שֶׁאִם הָיָה מַמְתִּין וּמְצַפֶּה לְחַג הַסֻּכּוֹת, נִמְצָא עוֹבֵר בַּעֲשֵׂה, שֶׁנֶּאֱמַר (דְּבָרִים יב) וּבָאתָ שָׁמָּה וַהֲבֵאתֶם שָׁמָּה, דְּמַשְׁמַע בְּרֶגֶל רִאשׁוֹן שֶׁתָּבֹא שָׁמָּה הָבֵא כָּל נְדָרִים שֶׁעָלֶיךָ. מִיהוּ בְּלָאו דְּלֹא תְּאַחֵר אֵינוֹ עוֹבֵר עַד שֶׁיַּעַבְרוּ עָלָיו שָׁלֹשׁ רְגָלִים: 

~ פרק ז - משנה ז
<B>אֲרוּכוֹת שֶׁל נַחְתּוֹמִים.</B> פְּשׁוּטֵי כְּלֵי עֵץ, שֶׁהַנַּחְתּוֹמִים עוֹרְכִים שָׁם כִּכָּרוֹת שֶׁל לֶחֶם בְּשָׁעָה שֶׁעוֹשִׂין הַפַּת. אֲרוּכוֹת, כְּמוֹ עֲרוּכוֹת: <B>שֶׁהֵן טְמֵאוֹת מִדְּרַבָּנָן.</B> דִּנְהִי דְּמִן הַתּוֹרָה פְּשׁוּטֵי כְּלֵי עֵץ אֵין מְקַבְּלִין טֻמְאָה, רַבָּנָן גָּזוּר עֲלַיְהוּ, כִּדְמוּכָח בְּמַסֶּכֶת כֵּלִים פֶּרֶק ב': <B>שֶׁרַבִּי אֱלִיעֶזֶר מְטַהֵר.</B> דְּסָבַר לֹא חֲשִׁיבֵי כֵּלִים כְּלָל, וַאֲפִלּוּ כִּפְשׁוּטֵי כְּלֵי עֵץ לֹא הֲווּ. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: <B>שֶׁחִתְּכוֹ חֻלְיוֹת.</B> שֶׁחִתְּכוֹ לַתַּנּוּר בְּרָחְבּוֹ לְחֻלְיוֹת וְהִנִּיחַ חֻלְיָא עַל גַּבֵּי חֻלְיָא וְנָתַן חֹל בֵּין חֻלְיָא לְחֻלְיָא, וְטָח אוֹתוֹ בְּטִיט וְעָשָׂה לוֹ טְפֵלָה שֶׁל טִיט סָבִיב לְעַבּוֹתוֹ לִשְׁמֹר חֻמּוֹ כַּדֶּרֶךְ שֶׁעוֹשִׂים לַתַּנּוּרִים: <B>שֶׁרַבִּי אֱלִיעֶזֶר מְטַהֵר.</B> דְּכֵיוָן דְּיֵשׁ חֹל בֵּין הַסְּדָקִין, כְּשָׁבוּר דָּמֵי. וַחֲכָמִים מְטַמְּאִים, שֶׁהַטְּפֵלָה עוֹשָׂה אֶת כֻּלּוֹ אֶחָד וּמְחַבֶּרֶת אֶת הַסְּדָקִים, וְאַף עַל פִּי שֶׁיֵּשׁ חֹל בֵּין חֻלְיָא לְחֻלְיָא. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: <B>שֶׁמְּעַבְּרִין אֶת הַשָּׁנָה בְּכָל אֲדָר.</B> עַד עֶשְׂרִים וְתִשְׁעָה בַּאֲדָר יֵשׁ זְמַן לְבֵית דִּין לוֹמַר מְעֻבֶּרֶת הַשָּׁנָה, וְהַחֹדֶשׁ הַבָּא הוּא אֲדָר שֵׁנִי. אֲבָל בְּיוֹם שְׁלֹשִׁים לַאֲדָר אֵין יְכוֹלִים לְעַבֵּר אֶת הַשָּׁנָה, הוֹאִיל וְרָאוּי לְקָבְעוֹ נִיסָן, כַּמְפֹרָשׁ בִּפְסָחִים פֶּרֶק מָקוֹם שֶׁנָּהֲגוּ [דַּף נו]: <B>עַל תְּנָאי.</B> אִם יִרְצֶה הַנָּשִׂיא {ב} תִּהְיֶה מְעֻבֶּרֶת וְאִם לָאו לֹא תִּהְיֶה מְעֻבֶּרֶת: 

~ פרק ז - משנה ח
<B>מוּסַף הַיּוֹרָה שֶׁל שׁוֹלְקֵי זֵיתִים.</B> דֶּרֶךְ שׁוֹלְקֵי זֵיתִים וְצַבָּעִים, שֶׁיֵּשׁ לָהֶם יוֹרוֹת גְּדוֹלוֹת, וְעוֹשִׂים לָהֶם תּוֹסֶפֶת טִיט עַל שְׂפָתָם שֶׁיַּעֲלוּ בָּהּ הַמַּיִם בְּעֵת רְתִיחָתָם. וְשֶׁל שׁוֹלְקֵי זֵיתִים טָמֵא, לְפִי שֶׁאוֹתָהּ תּוֹסֶפֶת צְרִיכָה לַכְּלִי וּמִשְׁתַּמְּשִׁין בָּהּ, וְהַתּוֹרָה אָמְרָה גַּבֵּי תַּנּוּר וְכִירַיִם (וַיִּקְרָא יא) וּטְמֵאִים יִהְיוּ לָכֶם, וְדָרְשִׁינַן לָכֶם, כָּל שֶׁבְּצָרְכֵיכֶם, כְּלוֹמַר הַדָּבָר מִן הַכְּלִי שֶׁאַתֶּם צְרִיכִים לוֹ וּמִשְׁתַּמְּשִׁין בּוֹ הוּא שֶׁמְּקַבֵּל טֻמְאָה: <B>וְשֶׁל צַבָּעִים טָהוֹר.</B> שֶׁאֵין הַצַּבָּעִים מִשְׁתַּמְּשִׁים בְּאוֹתָהּ תּוֹסֶפֶת לְפִי שֶׁיְּרֵאִים שֶׁמָּא יִפָּסֵד צִבְעָם:

<B>הֵעִיד רַבִּי יְהוֹשֻׁעַ עַל דַּם נְבֵלוֹת שֶׁהוּא טָהוֹר.</B> מִלְּטַמֵּא כַּנְּבֵלָה בְּכַזַּיִת. אֲבָל מְטַמֵּא בִּרְבִיעִית. שֶׁאֵין לְךָ דָּבָר שֶׁיְּטַמֵּא דָּמוֹ כִבְשָׂרוֹ אֶלָּא הַשֶּׁרֶץ בִּלְבַד {א}. וּלְעֵיל [רֵישׁ פֶּרֶק ה'] דִּתְנַן דַּם נְבֵלוֹת בֵּית שַׁמַּאי מְטַהֲרִין, לְגַמְרֵי הָיוּ בֵּית שַׁמַּאי מְטַהֲרִים. וּבֵית הִלֵּל מְטַמְּאִין, אֲבָל לֹא כַּנְּבֵלָה שֶׁמְּטַמֵּא בְּכַזַּיִת, אֶלָּא בִּרְבִיעִית: <B>שֶׁנָּגַע טָמֵא בְמִקְצָתוֹ שֶׁטִּמֵּא אֶת כֻּלּוֹ.</B> מַעֲלָה עָשׂוּ בַּקֹּדֶשׁ וּבְאֵפֶר חַטָּאת {ב}, שֶׁאַף עַל פִּי שֶׁהֵם גּוּפִים חֲלוּקִים, אִם מֻנָּחִים בְּתוֹךְ כְּלִי אֶחָד וְנָגַע דָּבָר שֶׁהוּא מְטַמֵּא לַקֹּדֶשׁ בְּמִקְצָתָן, נִטְמְאוּ כֻּלָּן, שֶׁהַכְּלִי מְצָרְפָן לְהַחֲשִׁיבָן כְּאִלּוּ הֵן גּוּף אֶחָד. וְאַסְּמְכוּהּ אַקְּרָא דִכְתִיב (בְּמִדְבַּר ז) כַּף אַחַת עֲשָׂרָה זָהָב מְלֵאָה קְטֹרֶת, הַכָּתוּב עָשָׂה לְכָל מַה שֶּׁבְּכַף אַחַת: <B>הוֹסִיף רַבִּי עֲקִיבָא.</B> דְּאִלּוּ מֵעֵדוּתוֹ שֶׁל רַבִּי שִׁמְעוֹן בֶּן בְּתֵירָא לֹא שָׁמַעְנוּ אֶלָּא עַל הַטָּמֵא, וּבָא רַבִּי עֲקִיבָא וְהוֹסִיף אֲפִלּוּ טְבוּל יוֹם שֶׁאֵינוֹ מְטַמֵּא אֶלָּא פּוֹסֵל בִּלְבַד, אִם נָגַע בְּמִקְצָתָן פָּסַל אֶת כֻּלָּן. וְרַמְבַּ"ם פֵּרֵשׁ, דִּמֵעֵדוּתוֹ שֶׁל רַבִּי שִׁמְעוֹן לֹא שָׁמַעְנוּ שֶׁמְּצָרֵף אֶלָּא כְּלִי שֶׁיֵּשׁ לוֹ תּוֹךְ {ג}, אֲבָל כְּלִי שֶׁאֵין לוֹ תּוֹךְ לֹא שָׁמַעְנוּ שֶׁמְּצָרֵף, וּבָא רַבִּי עֲקִיבָא וְהוֹסִיף שֶׁאֲפִלּוּ סֹלֶת וּקְטֹרֶת וּלְבוֹנָה וְגֶחָלִים שֶׁאֵינָן בְּתוֹךְ כְּלִי, אֶלָּא צְבוּרִים וּמֻנָּחִים עַל לוּחַ אוֹ דַּף שֶׁאֵין לוֹ תּוֹךְ, הֲרֵי הֵן כְּאִלּוּ מֻנָּחִים בְּתוֹךְ כְּלִי, וְנַעֲשִׂים כְּאִלּוּ הֵן גּוּף אֶחָד: <B>וְהַגֶּחָלִים.</B> אוֹתָן שֶׁכֹּהֵן גָּדוֹל חוֹתֶה בַּמַּחְתָּה בְּיוֹם הַכִּפּוּרִים {ה}. אִם נָגַע טְבוּל יוֹם בְּמִקְצָתָן, נִפְסְלוּ כֻּלָּן. וּמַעֲלָה עָשׂוּ בָּהֶן, שֶׁאֵין הַגֶּחָלִים בְּנֵי קַבּוֹלֵי טֻמְאָה:

<B>קְטַנָּה בַת יִשְׂרָאֵל.</B> וְהִיא יְתוֹמָה, כִּדְפָרֵישְׁנָא בְּשִׁלְהֵי פִרְקִין דִּלְעֵיל. וְהוֹסִיף כָּאן כְּשֶׁנִּכְנְסָה לַחֻפָּה אַף עַל פִּי שֶׁלֹּא נִבְעֲלָה. דְּמֵעֵדוּת דִּלְעֵיל לֹא שָׁמַעְנוּ שֶׁאוֹכֶלֶת בַּתְּרוּמָה אֶלָּא כְּשֶׁנִּבְעֲלָה: <B>שֶׁהֻרְהֲנָה.</B> שֶׁנִּתְמַשְׁכְּנָה בִּידֵי נָכְרִים {ו}: <B>וְעֵדֶיהָ.</B> אוֹתָן שֶׁהֵעִידוּ שֶׁהֻרְהֲנָה: <B>הַאֲמִינוּ שֶׁלֹּא נִסְתְּרָה וְשֶׁלֹּא נִטְמָאָה.</B> וְדַוְקָא לְזֹאת שֶׁעֵדֶיהָ מְעִידִים עָלֶיהָ שֶׁלֹּא נִטְמְאָה, הוּא דְאָמְרֵי רַבָּנָן הַאֲמִינוּ לָהּ, וְשֶׁלֹּא כַּדִּין רִחֲקוּהָ בְּנֵי מִשְׁפַּחְתָּהּ. אֲבָל אִם אֵין לָהּ עֵדִים, כָּל אִשָּׁה שֶׁנֶּחְבְּשָׁה עַל יְדֵי מָמוֹן בִּזְמַן שֶׁיַּד הַנָּכְרִים תַּקִּיפָה {ח}, אֲסוּרָה לְבַעְלָהּ כֹּהֵן, לֹא שְׁנָא הֻרְהֲנָה מִדַּעְתָּהּ וְלֹא שְׁנָא נֶחְבְּשָׁה:

<B>אַלְמָנַת עִסָּה.</B> מִשְׁפָּחָה אַחַת שֶׁנִּתְעָרֵב בָּהּ סְפֵק חָלָל, כָּל אֶחָד מִבְּנֵי הַמִּשְׁפָּחָה סָפֵק אִם הוּא אוֹתוֹ סָפֵק שֶׁנִּתְעָרֵב בָּהֶן אִם לָאו. וְאִשָּׁה שֶׁנִּשֵּׂאת לְאֶחָד מִבְּנֵי אוֹתָהּ מִשְׁפָּחָה וּמֵת בַּעְלָהּ, הִיא קְרוּיָה אַלְמָנַת עִסָּה, שֶׁכְּמוֹ שֶׁהָעִסָּה נִלּוֹשָׁה וּמְעֹרֶבֶת, כָּךְ זֹאת הָאִשָּׁה מְעֹרֶבֶת מִסְּפֵקוֹת. סָפֵק אִם בַּעְלָהּ הוּא אוֹתוֹ סְפֵק חָלָל, וַאֲפִלּוּ אִם הוּא אוֹתוֹ סְפֵק חָלָל שֶׁמָּא אֵינוֹ חָלָל לְפִי הָאֱמֶת. וְרַבִּי יְהוֹשֻׁעַ מַכְשִׁיר אוֹתָהּ לַכְּהֻנָּה, שֶׁיֵּשׁ כָּאן סְפֵק סְפֵקָא וּלְקֻלָּא. וְרַבָּן גַּמְלִיאֵל סָבַר אַף עַל גַּב דִּבְעָלְמָא סְפֵק סְפֵקָא לְקֻלָּא, הָכָא שָׁאנֵי, דְּמַעֲלָה עָשׂוּ בְּיוֹחֲסִין {י}: <B>שֶׁהָעִסָּה כְשֵׁרָה לְטַמֵּא וּלְטַהֵר.</B> כְּלוֹמַר שֶׁהַמִּשְׁפָּחָה שֶׁנִּתְעָרֵב בָּהּ סְפֵק חָלָל הֲרֵי הִיא בְּחֶזְקַת כַּשְׁרוּת כִּשְׁאָר הַמִּשְׁפָּחוֹת הַמְיֻחָסוֹת, וּכְמוֹ שֶׁשְּׁאָר הַמִּשְׁפָּחוֹת אוֹמְרוֹת זוֹ טְמֵאָה וּמִתְרַחֲקִים מִמֶּנָּה וְזוֹ טְהוֹרָה וּמְקָרְבִים אוֹתָהּ, כָּךְ מִשְׁפָּחָה זוֹ שֶׁנִּטְמַע בָּהּ סְפֵק חָלָל, אֵין יְכוֹלִים לוֹמַר הוֹאִיל וְנִטְמָע בָּהּ סָפֵק אֵין צְרִיכִים יוֹתֵר לִבְדֹּק כְּשֶׁנּוֹשְׂאִים נָשִׁים אֵיזוֹ נִטְמְאָה וְאֵיזוֹ טְהוֹרָה, כְּדֵי לְרַחֵק הַטְּמֵאָה וּלְקָרֵב הַטְּהוֹרָה {יא}: <B>קִבַּלְנוּ עֵדוּתְכֶם.</B> נֶאֱמָנִים אַתֶּם בְּעֵינֵינוּ וּמַאֲמִינִים אָנוּ שֶׁכָּךְ שְׁמַעְתֶּם: <B>שֶׁלֹּא לְהוֹשִׁיב בֵּית דִּין עַל כָּךְ.</B> לְהַתִּיר אַלְמָנַת עִסָּה לְכַתְּחִלָּה: <B>הַכֹּהֲנִים שׁוֹמְעִים לָכֶם לְרַחֵק.</B> אִם תֹּאמְרוּ אֲסוּרָה: <B>אֲבָל לֹא לְקָרֵב.</B> אִם תֹּאמְרוּ מֻתֶּרֶת. וּפְסַק הֲלָכָה, שֶׁאַלְמָנַת עִסָּה אֲסוּרָה לְהִנָּשֵׂא לַכְּהֻנָּה לְכַתְּחִלָּה, וְאִם נִשֵּׂאת לֹא תֵּצֵא:

<B>אַיִל קַמְצָא.</B> מִין חָגָב הוּא {יב}. תַּרְגּוּם כַּחֲגָבִים כְּקַמְצִין: <B>דָּכָן.</B> טָהוֹר {יג} וּמֻתָּר בַּאֲכִילָה: <B>וְעַל מַשְׁקֵה בֵּית מִטְבְּחַיָּא.</B> הַדָּם וְהַמַּיִם שֶׁהֵם בְּבֵית הַמִּטְבְּחַיִם שֶׁבָּעֲזָרָה: <B>דְּאִינוּן דַּכְיָן.</B> אִיכָּא מַאן דְּאָמַר דַּכְיָן טְהוֹרִים לְגַמְרֵי. דְּטֻמְאַת מַשְׁקִין לָאו דְאוֹרַיְתָא {יד} אֶלָּא רַבָּנָן הוּא דְּגָזוּר בְּהוּ טֻמְאָה, וּבְהָנֵי לֹא גָזוּר. וְאִיכָּא לְמַאן דְּאָמַר דַּכְיָן טְהוֹרִים מִלְּטַמֵּא אֲחֵרִים, אֲבָל טֻמְאַת עַצְמָן יֵשׁ לָהֶן, דְּיֵשׁ לְמַשְׁקִין טֻמְאָה מִן הַתּוֹרָה לְהִטַּמֵּא {טו}, וְלֹא מָצוּ רַבָּנָן לְטַהֵר מַה שֶּׁטִּמְּאָה הַתּוֹרָה: <B>וּדְיִקְרַב בְּמִיתָא מְסָאַב.</B> הָכִי קָאָמַר, וְכָל דְּיִקְרַב וַדַּאי לְמִיתָא, מְסָאַב. אֲבָל סָפֵק, אֲפִלּוּ טֻמְאַת מֵת חֲמוּרָה, טָהוֹר. וְכָל שֶׁכֵּן טֻמְאַת שֶׁרֶץ קַלָּה. וּסְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים בָּא לְהַתִּיר. וְאַף עַל גַּב דִּסְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים טָהוֹר דְּאוֹרַיְתָא הוּא, דְּכָל אִסּוּר סְפֵק טֻמְאָה מִסּוֹטָה יָלְפִינַן, דְּאַפִּיק לַהּ קְרָא בִּלְשׁוֹן טֻמְאָה, וְנִסְתְּרָה וְהִיא נִטְמְאָה, מַגִּיד לְךָ הַכָּתוּב שֶׁעַל הַסָּפֵק אֲסוּרָה, וּמַה סּוֹטָה רְשׁוּת הַיָּחִיד, שֶׁהֲרֵי אֵין סְתִירָה בִּרְשׁוּת הָרַבִּים, אַף טֻמְאַת סָפֵק אֵינָהּ אֶלָּא בִּרְשׁוּת הַיָּחִיד {טז}, מִכָּל מָקוֹם קֹדֶם שֶׁבָּא יוֹסֵי בֶּן יוֹעֶזֶר הָיוּ אוֹמְרִים הֲלָכָה וְאֵין מוֹרִין כֵּן {יז}, וּבָא הוּא וְהֵעִיד שֶׁמּוֹרִין כֵּן לְכַתְּחִלָּה לְטַהֵר כָּל סְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים: <B>וְקָרוּ לֵיהּ יוֹסֵי שָׁרְיָא.</B> לְפִי שֶׁהִתִּיר שְׁלֹשָׁה דְּבָרִים שֶׁהָיוּ נוֹהֲגִין בָּהֶן אִסּוּר. שֶׁכָּל בֵּית דִּין שֶׁמַּתִּיר שְׁלֹשָׁה דְּבָרִים שֶׁאֵין הַהֶתֵּר שֶׁלָּהֶן פָּשׁוּט, קָרוּ לֵיהּ בֵּי דִינָא שָׁרְיָא:

<B>שֶׁמַּשִּׂיאִין הָאִשָּׁה עַל פִּי עֵד אֶחָד.</B> אִשָּׁה שֶׁהָלַךְ בַּעְלָהּ לִמְדִינַת הַיָּם וּבָא עֵד אֶחָד וְאָמַר שֶׁמֵּת, מַשִּׂיאִים אֶת אִשְׁתּוֹ עַל פִּיו: <B>בְּדִיר הָעֵצִים.</B> לִשְׁכָּה שֶׁשָּׁם הָיוּ אוֹצְרִים כָּל עֲצֵי הַמַּעֲרָכָה, וְהִיא הָיְתָה בְּמִקְצוֹעַ מִזְרָחִית צְפוֹנִית שֶׁל עֶזְרַת נָשִׁים. וּמָצְאוּ שָׁם עֲצָמוֹת שֶׁל מֵתִים, וְאָמְרוּ חֲכָמִים, מְלַקֵּט עֶצֶם עֶצֶם וְהַכֹּל טָהוֹר, וְלֹא חָיְשִׁינַן שֶׁמָּא נִטְמְאוּ בָּהֶם אָדָם וְכֵלִים, מִשּׁוּם דְּעֶזְרַת נָשִׁים דִּין רְשׁוּת הָרַבִּים יֵשׁ לָהּ {יט} וְהָוֵי סְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים דִּסְפֵקוֹ טָהוֹר. וּבְמַסֶּכֶת זְבָחִים פֶּרֶק בָּתְרָא [דַּף קיג] מַיְתֵי גְמָרָא שֶׁרָצוּ לִגְזֹר טֻמְאָה עַל כָּל יְרוּשָׁלַיִם בִּשְׁבִיל אוֹתָן עֲצָמוֹת שֶׁנִּמְצְאוּ בְּדִיר הָעֵצִים, אֶלָּא שֶׁאָמַר לָהֶם רַבִּי יְהוֹשֻׁעַ לֹא בּוּשָׁה וּכְלִמָּה הִיא לָנוּ שֶׁנִּגְזֹר טֻמְאָה עַל עִיר אֲבוֹתֵינוּ:

<B>שָׁמַעְתִּי שֶׁמַּקְרִיבִים אַף עַל פִּי שֶׁאֵין בַּיִת.</B> לְפִי שֶׁקְּדֻשָּׁה שֶׁקִּדֵּשׁ שְׁלֹמֹה אֶת הַבַּיִת, קִדְּשָׁה לִשְׁעָתָהּ וְקִדְּשָׁה לֶעָתִיד לָבֹא. וְכֵן קְדֻשַּׁת יְרוּשָׁלַיִם קִדְשָׁה לְעוֹלָם. אֲבָל קְדֻשַּׁת שְׁאָר אֶרֶץ יִשְׂרָאֵל לֹא קִדְּשָׁה בְּכִבּוּשׁ רִאשׁוֹן אֶלָּא לִשְׁעָתָהּ {כב}, עַד שֶׁחָזְרוּ עוֹלֵי בָּבֶל וְקִדְּשׁוּהָ קְדֻשָּׁה שְׁנִיָּה, וְאוֹתָהּ קְדֻשָּׁה קִדְּשָׁה לֶעָתִיד לָבֹא:

<B>הֲלָכָה לְמֹשֶׁה מִסִּינַי.</B> שֶׁהֶרְאָה לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא לְמֹשֶׁה בְּסִינַי דּוֹר דּוֹר וְדוֹרְשָׁיו, וְהֶרְאָהוּ שֶׁאֵין אֵלִיָּהוּ בָּא לְטַמֵּא וּלְטַהֵר, לְרַחֵק וּלְקָרֵב, לְבָרֵר סְפֵק מִשְׁפָּחוֹת הַמְטֻמָּעוֹת מִי נִטְמָע וּמִי לֹא נִטְמָע, אֶלָּא יַנִּיחֵם וְהֵם כְּשֵׁרִים לֶעָתִיד לָבֹא, דַּהֲלָכָה הִיא מִשְׁפָּחָה שֶׁנִּטְמְעָה נִטְמְעָה: <B>אֶלָּא לְרַחֵק הַמְקֹרָבִים בִּזְרוֹעַ.</B> שֶׁהַכֹּל מַחֲזִיקִים אוֹתָן בִּפְסוּלִים, אֶלָּא שֶׁנִּתְקָרְבוּ בִּזְרוֹעַ. אֲבָל מִשְׁפָּחָה שֶׁנִּטְמְעָה מֵחֲמַת שֶׁלֹּא נוֹדַע פְּסוּלָהּ, יַנִּיחֶנָּה בְּכַשְׁרוּתָהּ: <B>בֵּית צְרִיפָה.</B> כָּךְ שֵׁם הַמִּשְׁפָּחָה: <B>וְרִחֲקָהּ.</B> הִכְרִיז עֲלֵיהֶן שֶׁהֵן פְּסוּלִין: <B>בֶּן צִיּוֹן.</B> אֱינָשׁ אַלִּימָא הֲוָה וּבַעַל זְרוֹעַ: <B>וְעוֹד מִשְׁפָּחָה אַחֶרֶת הָיְתָה שָׁם.</B> שֶׁהָיְתָה פְּסוּלָה: <B>וְקֵרְבָהּ בֶּן צִיּוֹן בִּזְרוֹעַ.</B> וְהִכְרִיז עֲלֵיהֶן שֶׁהֵן כְּשֵׁרִים לְהִתְחַתֵּן בָּהֶם. וְחָס הַתַּנָּא עַל כְּבוֹד הַבְּרִיּוֹת וְלֹא הִזְכִּיר שֵׁם הַמִּשְׁפָּחָה הַפְּסוּלָה שֶׁקֵּרְבָהּ בֶּן צִיּוֹן בִּזְרוֹעַ כְּמוֹ שֶׁהִזְכִּיר שֵׁם הַמִּשְׁפָּחָה הַכְּשֵׁרָה, לְלַמֶּדְךָ כַּמָּה צָרִיךְ אָדָם לִזָּהֵר שֶׁלֹּא לְסַפֵּר בִּגְנוּת חֲבֵרוֹ, וְלִהְיוֹת כּוֹסֶה קָלוֹן. אִם כָּךְ בִּפְסוּלִים, כָּל שֶׁכֵּן בִּכְשֵׁרִים {כה}: <B>כְּגוֹן אֵלּוּ.</B> שֶׁכַּשְׁרוּתָן וּפַסְלוּתָן יְדוּעָה, אֶלָּא שֶׁנִּתְרַחֲקוּ בִּזְרוֹעַ וְנִתְקָרְבוּ בִּזְרוֹעַ: <B>לְקָרֵב אֲבָל לֹא לְרַחֵק.</B> הַמִּשְׁפָּחָה הַכְּשֵׁרָה שֶׁנִּתְרַחֲקָה בִּזְרוֹעַ, הוּא מְקָרֵב. אֲבָל אֵינוֹ מְרַחֵק אוֹתָהּ שֶׁנִּתְקָרְבָה בִּזְרוֹעַ: <B>שֶׁנֶּאֱמַר וְהֵשִׁיב לֵב אָבוֹת עַל בָּנִים.</B> שֶׁעָתִיד לוֹמַר בְּרוּחַ הַקֹּדֶשׁ זֶה מִבְּנֵי בָּנָיו שֶׁל זֶה. וּלְדִבְרֵי רַבִּי שִׁמְעוֹן, אָבוֹת אֵלּוּ הַחֲכָמִים, וּבָנִים אֵלּוּ הַתַּלְמִידִים {כז}, שֶׁיִּהְיֶה לֵב כֻּלָּן שָׁוֶה וְלֹא יִפֹּל בֵּינֵיהֶן מַחֲלֹקֶת:
