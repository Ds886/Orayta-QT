{א} <b> מִקַּבָּיִם</b>. בַּיְרוּשַׁלְמִי מְפָרֵשׁ טַעֲמָא, מִשּׁוּם דְּאֵיפָה שְׁמֹנָה עָשָׂר קַבִּין, וַעֲשִׂירִית מִלְּבַר הַיְנוּ שְׁנֵי קַבִּין, וּכְאוֹתוֹ שִׁעוּר צָרִיךְ לְהַפְרִישׁ חַלָּה לְהִלֵּל. וְשַׁמַּאי סָבַר דְּצָרִיךְ שְׁתֵּי חַלּוֹת לָעֹמֶר, שֶׁהָיוּ עוֹשִׂין מִמֶּנּוּ שְׁתֵּי אֲכִילוֹת, אַחַת בַּבֹּקֶר וְאַחַת בָּעֶרֶב, וּכְשִׁעוּר אֲכִילָה אַחַת חֲשִׁיב שַׁמַּאי עִסַּת מִדְבָּר. תּוֹסָפוֹת: 

~ פרק א - משנה ג

{ב} <B> רַבּוֹ.</B> וְהַטַּעַם לְדִבְרֵי רַבּוֹ, שֶׁרָצָה לְגַלּוֹת כִּי מִפְּנֵי שֶׁהַהִין הִיא הַמִּדָּה הַגְּדוֹלָה שֶׁל מִדּוֹת תּוֹרָה בַּלַּח, כְּדִכְתִיב [שְׁמוֹת ל] וְשֶׁמֶן זַיִת הִין, הִלְכָּךְ הִיא חֲשׁוּבָה לִפְסֹל אֶת הַמִּקְוֶה. וְאַף עַל פִּי שֶׁנֶּאֶמְרוּ בַּתּוֹרָה מִדּוֹת קְטַנּוֹת מִזֹּאת, כֵּיוָן דִּפְסוּל זֶה מִדְּרַבָּנָן הוּא, אָזְלִינַן בָּתַר הַמִּדָּה הַגְּדוֹלָה הַמְפֹרֶשֶׁת בַּתּוֹרָה. הָרַאֲבַ"ד: {ג} <B>בִּלְשׁוֹן רַבּוֹ.</B> שֶׁהֲרֵי יֵשׁ טַעַם בַּלָּשׁוֹן לָמָּה אָמַר כֵּן, כַּנִּזְכָּר לְעֵיל. כָּל שֶׁכֵּן שֶׁלִּפְעָמִים אֶפְשָׁר שֶׁכְּשֶׁיְּשַׁנֵּהוּ לְלָשׁוֹן אַחֵר שֶׁאֶפְשָׁר דְּלֹא לְמִשְׁמַע מִנֵּיהּ בְּאוֹתוֹ הַלָּשׁוֹן כְּפִי הַמְכֻוָּן בַּלָּשׁוֹן שֶׁאָמַר רַבּוֹ. הָרַאֲבַ"ד: {ד} <B> תִּשְׁעָה קַבִּין.</B> קָסָבַר, כֵּיוָן דְּהֵם רְאוּיִים לִטְבִילַת בַּעַל קֶרִי וּבָהֶם יֵשׁ לְהַשִּׂיג כְּדֵי לְהִשְׁתַּטֵּף כָּל גּוּפוֹ, לֹא פָּחוֹת מִכָּאן, הִלְכָּךְ הֵם חֲשׁוּבִים לִפְסֹל אֶת הַמִּקְוֶה בִּשְׁאוּבִין, דְּשִׁעוּר טְבִילָה פּוֹסֵל טְבִילָה, בְּצִיר מֵהָכִי לֹא. הָרַאֲבַ"ד: {ה} רְצוֹנוֹ לוֹמַר וְכָל דִּכְוָתֵיהּ הַשְּׁנוּיִים בַּבְּרַיְתָא קִדּוּשִׁין דַּף פ"ב, שֶׁאֵין מַעֲמִידִין מֵהֶם כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} <B> שְׁלֹשָׁה לֻגִּים.</B> טַעֲמַיְהוּ דְּרַבָּנָן, כֵּיוָן שֶׁהֵם נְתוּנִים לְשִׁעוּר נִסְכֵּי קָרְבְּנוֹת צִבּוּר חֲשׁוּבִים הֵם לִפְסֹל. וְאַף עַל פִּי שֶׁנֶּאֶמְרוּ בַּתּוֹרָה לְקָרְבָּנוֹת צִבּוּר מִדּוֹת גְּדוֹלוֹת מִזּוֹ כוּ', כֵּיוָן דְּאַשְׁכְּחַן דִּרְבִיעִית הַהִין דְּהוּא שְׁלֹשָׁה לֻגִּין חָשִׁיבָא לְנִסְכֵּי צִבּוּר, לְעִנְיָן זֶה נַמִּי חָשִׁיב כוּ'. הָרַאֲבַ"ד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ז} <B> לֹא עָמְדוּ כוּ'.</B> וְלֹא עָמְדוּ חֲכָמִים עַל דִּבְרֵיהֶם. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ה

{ח} <B> וְיִסְמֹךְ עָלָיו.</B> וּלְגִרְסָא זוֹ פֵּרוּשׁוֹ, שֶׁאִם יֵרָאֶה לְבֵית דִּין הָאַחֲרוֹן הֲלָכָה כְּדִבְרֵי הַיָּחִיד, יִסְמֹךְ עָלָיו, כְּלוֹמַר יִקְבַּע הֲלָכָה כְּמוֹתוֹ. כְּמוֹ שֶׁמָּצָאנוּ בַּאֲמוֹרָא מִן הָאַחֲרוֹנִים שֶׁהֵן קוֹבְעִין וַהֲלָכָה כַּיְחִידִים הָרִאשׁוֹנִים בְּכַמָּה מְקוֹמוֹת, וְאַף עַל פִּי שֶׁהַמְרֻבִּים חוֹלְקִים עֲלֵיהֶם, וְאִם לֹא שֶׁמָּצְאוּ דִּבְרֵי הַיָּחִיד הָרִאשׁוֹן לֹא הָיוּ אַחֲרוֹנִים יְכוֹלִין לִדְחוֹת דִּבְרֵי הָרִאשׁוֹנִים מִדַּעַת עַצְמָן לְפִי שֶׁאֵין בֵּית דִּין יָכוֹל לְבַטֵּל כוּ', אֶלָּא כֵּיוָן שֶׁמָּצְאוּ דִּבְרֵי הַיָּחִיד מִן הָרִאשׁוֹנִים שֶׁהָיָה חוֹלֵק עִמָּהֶם הָיָה לָהֶם בַּמֶּה לִתְלוֹת. וְעַיֵּן לְקַמָּן: {ט} כִּדְאַשְׁכְּחַן רַבָּה דַּהֲוָה גָּדוֹל מֵרַב יְהוּדָה. כֶּסֶף מִשְׁנֶה: {י} וְזֶה לְשׁוֹן הָרַמְבַּ"ם, וְהַאֵיךְ יִהְיוּ גְּדוֹלִים בְּמִנְיָן הוֹאִיל וְכָל בֵּית דִּין וּבֵית דִּין שֶׁל שִׁבְעִים וְאֶחָד הוּא, זֶה מִנְיַן חַכְמֵי הַדּוֹר שֶׁהִסְכִּימוּ וְקִבְּלוּ הַדָּבָר שֶׁאָמְרוּ בֵּית דִּין הַגָּדוֹל וְלֹא חָלְקוּ בּוֹ. וְהָרַאֲבַ"ד כָּתַב, מִנְיַן שָׁנִים קָאָמַר: {יא} <B> שֶׁאֵין כוּ'.</B> פֵּרוּשׁ בְּתַקָּנוֹת גְּזֵרוֹת וּמִנְהָגוֹת. אֲבָל אִם דָּרְשׁוּ בְּאַחַת מִן הַמִּדּוֹת כְּפִי מַה שֶּׁנִּרְאֶה בְּעֵינֵיהֶם שֶׁהַדִּין כָּךְ וְדָנוּ דִּין וְעָמַד אַחֲרֵיהֶם בֵּית דִּין אַחֵר וְנִרְאֶה לוֹ טַעַם אַחֵר לִסְתֹּר אוֹתוֹ, הֲרֵי זֶה סוֹתֵר וְדָן כְּפִי מַה שֶּׁנִּרְאֶה בְּעֵינָיו, שֶׁנֶּאֱמַר [דְּבָרִים יז] אֶל הַשֹּׁפֵט אֲשֶׁר יִהְיֶה בַּיָּמִים הָהֵם, אֵינְךָ חַיָּב לָלֶכֶת אֶלָּא אַחַר שׁוֹפֵט שֶׁבְּיָמֶיךָ. הָרַמְבַּ"ם. וְכָתַב הַכֶּסֶף מִשְׁנֶה דְּמִחֲתִימַת הַמִּשְׁנָה קִיְּמוּ וְקִבְּלוּ שֶׁלֹּא לַחֲלֹק עַל הַתַּנָּאִים. וְכֵן בַּחֲתִימַת הַגְּמָרָא קִבְּלוּ עֲלֵיהֶם שֶׁלֹּא לַחֲלֹק עַל הָאֲמוֹרָאִים: {יב} <B> הָיָה כוּ'.</B> לְהָכִי אַהְדְּרֵיהּ, דְּלֹא תֵימָא דְּאוֹ אוֹ קָתָנֵי, כִּדְאַשְׁכְּחַן כְּהַאי גַוְנָא בְּאָבוֹת פֶּרֶק ה' מִשְׁנָה ז', קָא מַשְׁמַע לָן דְּלֹא. וְהָא דְּקָתָנֵי עַד שֶׁיִּהְיֶה כוּ', נִרְאֶה לִי דְּאָתָא לְמֵימַר שֶׁיְּדַקְדֵּק הֵיטֵב הֵיטֵב עַד שֶׁיִּהְיֶה לוֹ בָּרוּר מְאֹד שֶׁבְּוַדַּאי הוּא גָּדוֹל כוּ', שֶׁכֵּן לֵב הָאָדָם אֵינוֹ רוֹאֶה חוֹבָה לְעַצְמוֹ וְכָל אִישׁ גָּדוֹל הוּא בְּעֵינָיו, וּלְפִיכָךְ הֻצְרַךְ לְהַזְהִיר מְאֹד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יג} וּתְמֵהַנִי כוּ' דְּהָא אֶפְשָׁר שֶׁיָּבֹא בֵּית דִּין בְּאַחֲרִית הַיָּמִים וְיִרְאֶה לַעֲשׂוֹת כִּדְבָרָיו. וְנִיחָא לְפִי מַה שֶּׁכָּתַב הָרַמְבַּ"ם, בַּמֶּה דְּבָרִים אֲמוּרִים בִּדְבָרִים שֶׁלֹּא אָסְרוּ אוֹתָן כְּדֵי לַעֲשׂוֹת סְיָג אֶלָּא בִּשְׁאָר דִּינֵי תּוֹרָה, אֲבָל דְּבָרִים שֶׁרָאוּ בֵּית דִּין לִגְזֹר וּלְאָסְרָן לַעֲשׂוֹת סְיָג, אִם פָּשַׁט אִסּוּרָן בְּכָל אֶרֶץ יִשְׂרָאֵל אֵין בֵּית דִּין הָאַחֵר יָכוֹל לְעָקְרָן וּלְהַתִּירָן אֲפִלּוּ הָיָה גָּדוֹל מִן הָרִאשׁוֹנִים. וְאַכַּתִּי לְשׁוֹן כְּדֵי לְבַטְּלָן שֶׁכָּתַב הָרַ"ב קָשֶׁה, מַאי כְּדֵי, אַדְּרַבָּא אִם לֹא הָיוּ נִזְכָּרִים מִכָּל שֶׁכֵּן שֶׁהָיוּ בְּטֵלִין וּמְבֻטָּלִין לֹא קַיָּמִין וְלֹא נִזְכָּרִים. וּלְכָךְ אֲנִי אוֹמֵר, שֶׁכִּשְׁגָגָה יָצְאָה מִלְּפָנָיו, וְעָרֵב לְשׁוֹן הָרַאֲבַ"ד בְּפֵרוּשׁוֹ שֶׁהוּא פֵּרֵשׁ בְּדֶרֶךְ אַחֵר, וְהוּא, שֶׁבֶּאֱמֶת נִזְכְּרוּ כְּדֵי לְבַטְּלָן לְדוֹרוֹת, שֶׁאִם יֹאמַר אֶחָד כָּךְ מְקֻבְּלַנִי יֹאמְרוּ כוּ', אֲבָל אִם לֹא הִזְכִּירוּ דִּבְרֵי הַיָּחִיד הָיָה סָבוּר הַמְקֻבָּל דְּדַעַת רַבִּים הִיא כוּ'. עַד כָּאן. וְרַבִּי יְהוּדָה פָּלֵיג אַטַּעַם שֶׁאִם יִרְאֶה כוּ':

{א} <B> שִׁשָּׁה כוּ'.</B> לָאו דְּאִיכָּא הָנֵי וְתוּ לֹא, אֶלָּא כְּשֶׁנִּכְנְסוּ לַכֶּרֶם בְּיַבְנֶה וְהֵעִידוּ כָּל אֶחָד עַל אוֹתוֹ שֶׁבְּיָדוֹ, הֵעִיד רַבִּי יְהוּדָה עַל אֵלּוּ. רַשִׁ"י. וְהָרַאֲבַ"ד כָּתַב דִּסְתָמָא דְּפֶרֶק ד' פָּלֵיג אַהַךְ פִּרְקָא, דִּיחִידָאֵי הוּא, וּמִכָּל מָקוֹם הַנְהוּ דְאַשְׁכְּחֵי בְהוּ סְתָמֵי, הִלְכָתָא נִינְהוּ: {ב} <B> בְּאִסּוּר.</B> מִשּׁוּם דְּבֵיצָה כְּגוּפָהּ דָּמֵי. וּלְפִיכָךְ כְּשֶׁנִּטְרְפָה הַתַּרְנְגוֹלֶת אַף הַבֵּיצָה אֲסוּרָה, וַאֲפִלּוּ שֶׁנִּגְמְרָה לְגַמְרֵי קֹדֶם שֶׁנִּטְרְפָה, דְּהָא אַכַּיּוֹצֵא בָּהּ נִמְכֶּרֶת בַּשּׁוּק קָאֵי. וְהָא דִתְנַן מִפְּנֵי שֶׁגָּדְלָה בְּאִסּוּר, הָא מְפָרְקִינַן לַהּ בַּגְּמָרָא, אֵימָא שֶׁנִּגְמְרָה, דִּטְרֵפָה אֵינָהּ יוֹלֶדֶת. הָרַ"ן: {ג} וְהָרַאֲבַ"ד גּוֹרֵס כְּגִרְסַת הַסֵּפֶר. וְזֶה לְשׁוֹנוֹ, נִרְאֶה לִי דְטַעֲמַיְהוּ דְבֵית הִלֵּל מִשּׁוּם דְּגָזְרֵי שְׁבִיעִית אַטּוּ שְׁאָר שְׁנֵי שָׁבוּעַ, כְּדֵי שֶׁלֹּא יְהֵא אָדָם רָגִיל לִכָּנֵס בִּשְׂדֵה חֲבֵרוֹ וּבְגִנָּתוֹ וּבְפַרְדֵּסוֹ שֶׁלֹּא מִדַּעְתּוֹ. אֲבָל מִן הַתּוֹרָה וַדַּאי מֻתָּר כוּ':

{ד} <B>תּוֹרְמִין כוּ' אֵין תּוֹרְמִין.</B> מַשְׁמַע דְּאִם תָּרְמוּ תְּרוּמָתָן תְּרוּמָה. וְאַיְרֵי בְּנָטַל רְשׁוּת מִכֹּהֵן. וּבְמִשְׁנָה ד' פֶּרֶק קַמָּא דִתְרוּמוֹת, דִּלְכַתְּחִלָּה לְכֻלֵּי עָלְמָא לֹא, וּלְבֵית הִלֵּל אַף בְּדִיעֲבַד, אַיְרֵי בְלֹא רְשׁוּת כֹּהֵן, וּמִשּׁוּם דְּאִיכָּא פְסֵידָא דְכֹהֵן (כְּמוֹ שֶׁכָּתַב הָרַ"ב שָׁם מִשְׁנָה ח). הָרַ"שׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ה} <B> מְטַמְּאִין.</B> דְּלֹא כִסְתָמָא דְמִשְׁנָה ד' פֶּרֶק י"ב דְּפָרָה. וְעַיֵּן שָׁם וּבְמִשְׁנָה ה': {ו} וְהָרַאֲבַ"ד מְפָרֵשׁ בְּדֶרֶךְ אַחֵר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ה - משנה ו

{ז} <B> שֶׁלֹּא כוּ'.</B> חֲדָא וְעוֹד קָאָמַר. כְּלוֹמַר, אֲפִלּוּ כְּשֶׁאֶחֱזֹר בִּי שֶׁלֹּא מִשּׁוּם הַשְּׂרָרָה וְלֹא אֶהֵא נַעֲשֶׂה רָשָׁע לִפְנֵי הַמָּקוֹם כִּי הוּא בּוֹחֵן לִבּוֹת, אֲפִלּוּ הָכִי לֹא אֶתְמַנֶּה לְאַב בֵּית דִּין, שֶׁלֹּא יִהְיוּ אוֹמְרִים כוּ'. נִרְאֶה לִי: {ח} לְשׁוֹנוֹ מְגֻמְגָּם וּמְהֻפָּךְ. וּלְשׁוֹן הָרַאֲבַ"ד, וַעֲקַבְיָא סָבַר לֹא גָזְרִינַן: {ט} <B> בְּכַרְכְּמִית.</B> כָּךְ שְׁמָהּ. אוֹ עַל שֵׁם מְקוֹמָהּ. רַשִׁ"י: {י} רְצוֹנוֹ לוֹמַר שֶׁלֹּא הִשְׁקוּהָ מֵי הַמָּרִים, אֶלָּא מַיִם אֲחֵרִים הָיוּ שֶׁהִשְׁקוּהָ. וּכְלוֹמַר שֶׁלֹּא כָּתְבוּ מְגִלַּת סוֹטָה וּמְחָקוּהָ, אֶלָּא מְגִלָּה בְּעָלְמָא כָּתְבוּ וּמָחֲקוּ וְהִשְׁקוּהָ. הָרַמְבַּ"ם. הָרַאֲבַ"ד וְהַתּוֹסָפוֹת: {יא} דְּגַם לַפֵּרוּשׁ הַשֵּׁנִי שֶׁאָמַר שֶׁעֲשָׂאוֹ דֻּגְמָא וְדִמְיוֹן כוּ', הָיָה גַּם כֵּן עִנְיָנוֹ לוֹמַר שֶׁלְּכָךְ עָשׂוּ דֻּגְמָא וְסִימָן לְפִי שֶׁהָיְתָה גִּיֹּרֶת וְהֵן גֵּרִים וּכְאִלּוּ הֵם לֹא סָבְרוּ לְהַרְחִיקָהּ לְהִסְתַּפְּחָהּ בְּדַת יִשְׂרָאֵל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} <B> חַס וְשָׁלוֹם וְכוּ'.</B> וְאִם תֹּאמַר, וְכִי מַשּׂוֹא פָּנִים יֵשׁ בַּדָּבָר, שֶׁאַף עַל פִּי שֶׁאֵין גָּדוֹל מִמֶּנּוּ בְּדוֹרוֹ אִם יֶחֱטָא וְאָשֵׁם וּפִלְּלוֹ אֱלֹהִים זֶה הַשּׁוֹפֵט אֲשֶׁר יִהְיֶה בַּיָּמִים הָהֵם כוּ'. וְיֵשׁ לוֹמַר, דִּסְבִירָא לֵיהּ דְּהָא דְאָמַר דֻּגְמָא לֹא נִתְכַּוֵּן לְזַלְזֵל בִּכְבוֹדָן כְּלָל. וְעוֹד נִרְאֶה לִי, דְּלֹא פְלִיגֵי בִּלְשׁוֹן דֻּגְמָא, וְאַף לְרַבִּי יְהוּדָה הָיָה זִלְזוּל בַּדָּבָר, וַאֲפִלּוּ הָכִי אָמַר חַס וְשָׁלוֹם שֶׁנִּתְנַדָּה (וְהַיְנוּ שֶׁלֹּא אָמַר חַס וְשָׁלוֹם שֶׁזִּלְזֵל), מִפְּנֵי שֶׁסְּבִירָא לֵיהּ דִּלְאָדָם גָּדוֹל אֵין הַבֵּית דִּין הַקָּטָן מִמֶּנּוּ יְכוֹלִין לְנַדּוֹתוֹ. וְכֵן מוּכָח מִלְּשׁוֹן רַשִׁ"י תַּעֲנִית דַּף כ"ג. וּפְסָחִים דַּף נ"ג. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} וּפֵרוּשׁ מִלַּת פִּקְפּוּק עַל בֻּרְיוֹ, הוּא הַדָּבָר שֶׁאֵין לוֹ קִיּוּם וְהוּא רוֹפֵף. וְהֵבִיא אוֹתוֹ בְּכָאן עַל דֶּרֶךְ הַדִּמְיוֹן בְּאוֹתוֹ שֶׁהֵקֵל בְּטָהֳרַת הַיָּדַיִם שֶׁהִיא מִדְּרַבָּנָן, וְנִתְקַיְּמָה בְּיִשְׂרָאֵל וְנִתְחַזְּקָה. הָרַמְבַּ"ם: {יד} <B> בְּטָהֳרַת יָדָיִם.</B> רְצוֹנוֹ לוֹמַר שֶׁהַיָּדַיִם מִטַּמְּאוֹת מִבְּלִי שְׁאָר הַגּוּף, בְּדָבָר פְּלוֹנִי וּבְדָבָר פְּלוֹנִי, כְּמוֹ שֶׁנִּתְבָּאֵר בְּמַסֶּכֶת יָדַיִם. הָרַמְבַּ"ם:

{טו} <B> אֲנִי כוּ'.</B> נִרְאֶה כִּי הָרִאשׁוֹנִים נִמְנוּ עַל אֵלּוּ אַרְבָּעָה דְּבָרִים וְהָיוּ מֶחֱצָה עַל מֶחֱצָה, הוּא שָׁמַע מִן הַמֶּחֱצָה הָאֶחָד וְהֵן שָׁמְעוּ מִפִּי הַמֶּחֱצָה הָאַחֵר כוּ'. אִי נַמִּי עֲקַבְיָא הָיָה אוֹמֵר כִּי אוֹתָן שֶׁשָּׁמַע מִפִּיהֶם הָיוּ הָרֹב, וַחֲבֵרָיו אָמְרוּ כִּי הָאֲחֵרִים שֶׁכְּנֶגְדָּם הֵם הָיוּ הָרֹב. הָרַאֲבַ"ד: {טז} וְהָיוּ דְּבָרָיו כְּדִבְרֵי רַבִּים. שֶׁאֵין מִתְּנָאֵי הַמְקַבֵּל אֶלָּא שֶׁיִּהְיוּ דְּבָרָיו מֵרַבִּים. הָרַמְבַּ"ם:

{א} מִשּׁוּם קְטַנָּה דְּדַוְקָא יְתוֹמָה הִיא כִּדְפֵרֵשׁ הָרַ"ב, וְאִם כֵּן הוּא הַדִּין דִּגְדוֹלָה הִיא יְתוֹמָה, אֲבָל נִשּׂוּאֵי הַגְּדוֹלָה נִשּׂוּאִין גְּמוּרִין, בֵּין יֵשׁ לָהּ אָב בֵּין שֶׁהִיא יְתוֹמָה: {ב} נִרְאֶה לִי דִּרְצוֹנוֹ לוֹמַר דְּמִשְׁנָה ז' דְּהָתָם, שֶׁהִיא כְּגוֹן הַךְ דְּהָכָא. וְכַיּוֹצֵא בָּהּ הִיא מִשְׁנָה י"א דְּהָתָם, וְכִדְאִיתָא בַּגְּמָרָא, דִּלְהָכִי תָּנֵי קְטַנּוֹת דְּמַשְׁמַע תַּרְתֵּי.
