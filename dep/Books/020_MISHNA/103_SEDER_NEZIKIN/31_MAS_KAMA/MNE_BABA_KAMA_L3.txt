{א} <b>אַרְבָּעָה.</b> מִנְיָנָא לְמִעוּטֵי דְּרַבִּי אוֹשַׁעְיָא וּדְרַבִּי חִיָּא [דְּמָנוּ נַמִּי לְנִזְקֵי גּוּפוֹ]. דְּתַנָּא דִּידָן בְּנִזְקֵי מָמוֹנוֹ קָמַיְרֵי, בְּנִזְקֵי גּוּפוֹ לֹא קָמַיְרֵי. גְּמָרָא: {ב} <B> נְזִיקִין.</B> שֶׁעוֹשִׂין נֶזֶק. וּנְזָקִין שֵׁם הַהֶזֵּק. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} לְהָנָךְ דִּכְתִיבֵי בִּקְרָא בְּהֶדְיָא. רַשִׁ"י: {ד} כִּיחוֹ רֹק הַפֶּה. נִיעוֹ לַחוּת הַחֹטֶם. רַשִׁ"י וְעַיֵּן עוֹד. תּוֹלָדָה דְּבוֹר לְקַמָּן מִשְׁנָה ג', וּפֶרֶק י' דְּבָבָא מְצִיעָא מִשְׁנָה ד': {ה} וְהָא דְּלֹא נָקַט שֵׁן, דַּהֲוָה אֲמֵינָא אֲפִלּוּ הֵיכָא דְּאֵין הֲנָאָה לְהֶזֵּקָה, וּלְפִיכָךְ נָקַט לִישְׁנָא דְּמַבְעֶה, דְּמַשְׁמַע לִישְׁנָא דְּגִלּוּיֵי כוּ' דְּהַיְנוּ לַהֲנָאָתָהּ. שֶׁפְּעָמִים נִגְלִין פְּעָמִים נִכְסִין, דְּהַיְנוּ בִּשְׁעַת אֲכִילָה. נִמּוּקֵי יוֹסֵף: {ו} <B> וְהַהֶבְעֵר.</B> יֵשׁ לְהַקְשׁוֹת אַמַּאי לֹא תָּנֵי אֵשׁ. וְיֵשׁ לוֹמַר דְּאֵשׁ יֵשׁ לְהָבִין מִסְתָּמָא שֶׁהוּא בְּעִנְיָן שֶׁדּוֹלֵק הָאֵשׁ בְּשֶׁל חֲבֵרוֹ, וּקְרָא קָאָמַר כִּי תֵּצֵא אֵשׁ, אֲפִלּוּ מֵעַצְמָהּ, לְכָךְ קָאָמַר הַהֶבְעֵר, וְהוּא לִישְׁנָא דִּקְרָא לֹא תְבַעֲרוּ וְכוּ', וּמַיְרֵי מִסְתָּמָא דִּבְשֶׁלּוֹ הוּא מַדְלִיק, הָכָא נַמִּי דִּכְוָתֵיהּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} <B> לֹא הֲרֵי כוּ'.</B> פֵּרוּשׁ אֵין קֻלָּתוֹ שֶׁל שׁוֹר כְּקֻלָּתוֹ שֶׁל מַבְעֶה, דַּהֲוָה אֲמֵינָא רֶגֶל כוּ', כְּמוֹ שֶׁכָּתַב הָרַ"ב, וּלְפִיכָךְ אִי כָּתַב רַחְמָנָא שׁוֹר לֹא אָתֵי מַבְעֶה מִנֵּיהּ, שֶׁהוּא קִיל מִנֵּיהּ. וְהָא דְּלֹא שָׁנְאָה מִשְׁנָתֵנוּ הַקַּל וָחֹמֶר כְּמוֹ שֶׁאָמַר לֹא זֶה וָזֶה [שֶׁיֵּשׁ בָּהֵם רוּחַ חַיִּים], מִשּׁוּם דְּחַד מֵחַד קַל לִמְצֹא חֹמֶר בְּאֶחָד מִבַּחֲבֵרוֹ. תּוֹסָפוֹת. וְעַיֵּן עוֹד: {ח} אֲבָל מִבּוֹר לְחוּד לֹא נָפֵיק, דְּאִיכָּא לְמִפְרַךְ מַה לְּבוֹר שֶׁתְּחִלַּת עֲשִׂיָּתוֹ לְנֶזֶק כוּ'. וְהַתַּנָּא הֶאֱרִיךְ, לְהַגְדִּיל תּוֹרָה וְיַאֲדִיר. תּוֹסָפוֹת: {ט} דִּכְתִיב וּבִעֵר בִּשְׂדֵה אַחֵר. רַשִׁ"י: {י} כְּדִי נָסֵיב לַהּ, דְּהָא בּוֹר אִצְטְרִיכָא לֵיהּ לְמִכְתַּב, דְּלֹא אָתֵי מִכּוּלְּהוּ, כְּדִתְנַן בְּמַתְנִיתִין. רַשִׁ"י וְתּוֹסָפוֹת: {יא} וּלְרַבִּי יְהוּדָה דְּמִשְׁנָה ה' פֶּרֶק ו' דִּמְחַיֵּב טָמוּן, יֵשׁ לוֹמַר כְּמוֹ שֶׁכָּתְבוּ הַתּוֹסָפוֹת דַּהֲוָה מָצֵי לְמֵימַר דְּכֻלְּהוּ אִצְטְרִיךְ לְמִכְתַּב, דְּאִי הֲוָה נָפְקָא מִבּוֹר וּמְחַד מֵהָנָךְ הֲוָה פָּטְרֵי בְּהוּ כֵּלִים. וּלְרַבִּי יְהוּדָה אִין הָכִי נַמִּי דְּאִצְטְרִיךְ מִשּׁוּם הָכִי: {יב} <B> הָאֵשׁ.</B> וְהָא דְּלֹא הֲדַר וְתָנֵי לֹא הֲרֵי הָאֵשׁ כַּהֲרֵי שְׁנֵיהֶם כִּדְקָא הֲדַר בָּרֵישָׁא, מִשּׁוּם דְּטִרְחָא לֵיהּ מִלְּתָא. נִמּוּקֵי יוֹסֵף. סְבִירָא לֵיהּ דְּיֵשׁ בָּאֵשׁ מַה שֶּׁאֵין בִּשְׁנֵיהֶם לְפִי שֶׁכֹּחַ אַחֵר מְעֹרָב בּוֹ וְאֵין הוֹלֵךְ לְדַעְתּוֹ כְּמוֹ שׁוֹרוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} אַף עַל גַּב דְּלֹא דָּמֵי לְחַד מֵהָנֵי אָבוֹת, אִי יְכוֹלִין לַאֲתוֹיֵי מִתַּרְתֵּי אוֹ מִתְּלַת אוֹ מִטְּפֵי בְּהַצַּד הַשָּׁוֶה לְחִיּוּבָא, מְחַיְּבִינַן לֵיהּ. הָרַ"ן: {יד} <B> וּשְׁמִירָתָן כוּ'.</B> שֶׁנִּתְחַיַּבְתָּ בִּשְׁמִירָתָן מֵחֲמַת שֶׁהֵם מָמוֹנְךָ, אוֹ שֶׁנִּמְסְרוּ לְךָ לְשָׁמְרָן, אוֹ שֶׁעָשִׂיתָ בָּהֶם מַעֲשֶׂה שֶׁתִּתְחַיֵּב בִּשְׁמִירָתָן כוּ'. לַאֲפוֹקֵי הֵיכָא דְּמַסְרִינְהוּ לַאֲחֵרִים לְנָטְרִינְהוּ כְּדַחֲזִי, וּפָשַׁע בָּהֶן הַשּׁוֹמֵר, דְּהַשֵּׁנִי חַיָּב וְהָרִאשׁוֹן פָּטוּר. וּלְאַפּוֹקֵי נַמִּי הֵיכָא דְּנָטְרִינְהוּ כְּאוֹרְחַיְהוּ וְאַזִּיקוּ, דְּפָטוּר. נִמּוּקֵי יוֹסֵף: {טו} <B> חָב.</B> הַאי תָּנָא יְרוּשַׁלְמִי הוּא, דְּתָנֵי לִישָׁנָא קַלִּילָא, וְלֹא תָּנֵי חַיָּב. גְּמָרָא: {טז} <B> תַּשְׁלוּמֵי.</B> לְהָכִי נָקַט לְשׁוֹן הַשְׁלָמָה, דִּפְחַת נְבֵלָה מִשְּׁעַת מִיתָה עַד שְׁעַת הַעֲמָדָה בַּדִּין, דְּנִזָּק הָוֵי. וְהַמּוֹתָר מְשַׁלֵּם לוֹ מַזִּיק. בַּגְּמָרָא דַּף י. וְהוּא הַדִּין דִּשְׁבָחָהּ דְּבֵין מִיתָה לַהַעֲמָדָה בַּדִּין, דְּנִזָּק הָוֵי. נִמּוּקֵי יוֹסֵף. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} דְּהָא מֵיטַב שָׂדֵהוּ דְּהַאִיךְ דְּקָא מְשַׁלֵּם מַשְׁמַע. גְּמָרָא: {יח}  וְיָלְפִינַן מִבּוֹר דִּכְתִיב בֵּיהּ יָשִׁיב, לְרַבּוֹת שָׁוֵה כֶּסֶף. נִמּוּקֵי יוֹסֵף:

{יט} אֲבָל אִם נִזָּק, פְּלִיגֵי תַּנָּאֵי. וּמִכָּל מָקוֹם פָּסְקוּ כְּמַאן דְּאָמַר אַף לִנְזָקִין הָאַחֲרוֹן חַיָּב. וְהַטַּעַם מִפְּנֵי שֶׁהָאַחֲרוֹן הֶעֱתִיקוֹ מִמִּדַּת נְזָקִין לְמִדַּת מִיתָה, וּמִפְּנֵי כָּךְ חַיָּב בַּכֹּל. הַמַּגִּיד מִשְׁנֶה: {כ} <B> מְעִילָה.</B> מְעִילָה הוּא דְּלֵית בְּהוּ, הָא מִקְדַשׁ קַדְּשֵׁי חַיָּב בִּנְזָקִין, כְּגוֹן קָדָשִׁים קַלִּים שֶׁאֵין בָּהֶם מְעִילָה מֵחַיִּים, אֶלָּא אֵמוּרִים לְאַחַר זְרִיקַת דָּם. מָנִי רַבִּי יוֹסֵי הַגְּלִילִי הִיא דְּאָמַר מָמוֹן בְּעָלִים הוּא כוּ': {כא} בַּגְּמָרָא פָּרֵיךְ, מַאן תָּבַע לֵיהּ. אֶלָּא דְּנָגַח תּוֹרָא דְּהֶפְקֵר לְתוֹרָא דִּידָן, וְקָדַם וְזָכָה בּוֹ אַחֵר. אִי נַמִּי נָגַח וְאַחַר כָּךְ הִקְדִּישׁ וְאַחַר כָּךְ הִפְקִיר, שֶׁנֶּאֱמַר וְהוּעַד בִּבְעָלָיו וְהֵמִית אִישׁ וְגוֹ' הַשּׁוֹר יִסָּקֵל וְגַם בְּעָלָיו יוּמָת. וּמַשְׁמַע שֶׁיְּהֵא לוֹ בַּעַל בִּשְׁעַת מִיתָה וְהַעֲמָדָה בַּדִּין וּגְמַר דִּין. וְהַאי קְרָא אַף עַל גַּב דִּבְשׁוֹר שֶׁהֵמִית אָדָם כְּתִיב, אִם אֵינוֹ עִנְיַן לְמִיתָה, דְּשִׁבְעָה שׁוֹר כְּתִיבֵי, תְּנֵהוּ עִנְיָן לִנְזִיקִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} לָאו דַּוְקָא לַחֲבֵרוֹ, דְּהוּא הַדִּין לְשׁוֹר נַמִּי דִּינָא הָכִי, וְאַשִּׁגְרַת לִישְׁנָא דִּגְמָרָא נָקַט. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} דְּבָעִינַן וּבִעֵר בִּשְׂדֵה אַחֵר. וּמִשּׁוּם הָכִי אִם הָיְתָה מְיֻחֶדֶת לְפֵרוֹת וְכוּ' וְהִזִּיק בְּשֵׁן וָרֶגֶל, חַיָּב, דִּשְׂדֵה אַחֵר קָרִינַן בֵּיהּ. רַשִׁ"י: {כד} <B> וּכְשֶׁהִזִּיק כוּ'.</B> לַאֲתוֹיֵי קֶרֶן, דִּכְתִיב כִּי יִגַּח, וְאֵין נְגִיחָה אֶלָּא בְּקֶרֶן כוּ'. גְּמָרָא. וּכְשֶׁהִזִּיק כוּ' דְּרֵישָׁא, לֹא דָּרְשִׁינַן לֵיהּ, דְּלָאו יְתֵרָא הוּא. רַשִׁ"י:

{כט} <B> הַשּׁוֹר.</B> קָא מַשְׁמַע לָן דְּהֻחְלַט הַשּׁוֹר כְּרַבִּי עֲקִיבָא. וּדְלֹא כְּרַבִּי יִשְׁמָעֵאל דְּאָמַר יוּשַׁם הַשּׁוֹר, דְּמַזִּיק בַּעַל חוֹב דְּנִזָּק הוּא, וְיִתֵּן לוֹ הַמַּזִּיק דְּמֵי הַשּׁוֹר. גְּמָרָא: {ל} <B> וְלֹא קִיַּמְתָּ כוּ'.</B> וְרַבִּי מֵאִיר מְקַיְּמוֹ פְּחָת שֶׁפִּחֲתָה מִיתָה מַחֲצִין בַּחַי, שֶׁאִם פָּחֲתָה מִשְּׁעַת מִיתָה וְעַד שְׁעַת הַעֲמָדָה בַּדִּין, פְּחָת דְּנִזָּק. וְרַבִּי יְהוּדָה אִי לְמַחֲצִין בַּחַי לְחוּד קָאָתֵי, אִם כֵּן נִכְתֹּב וְאֶת, מַאי וְגַם, שְׁמַע מִנַּהּ תַּרְתֵּי, דְּשַׁתְּפֵיהּ רַחְמָנָא בִּשְׁבָחָא. וּמִיהוּ אִם עָלָה הַשֶּׁבַח יוֹתֵר מֵהֶזֵּקוֹ לֹא פָּלֵיג בִּשְׁבָחָא דִּבְעָלִים מְשַׁלְּמִין וְאֵין בְּעָלִים נוֹטְלִין. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{לה} <B> לֹא כִי.</B> פֵּרוּשׁ לֹא כֵן הָיָה. וּמָצִינוּ לִישְׁנָא דְּלֹא כִי בְּטַעֲנַת שֶׁמָּא, הִלְכָּךְ מַצִּית לְמֵימַר דְּנִזָּק טוֹעֵן בָּרִי וּמַזִּיק טוֹעֵן שֶׁמָּא, וַאֲפִלּוּ הָכִי תְּנַן הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה. וְאִם כֵּן שָׁמְעִינַן מֵהָא מַתְנִיתִין תַּרְתֵּי, חֲדָא דַּאֲפִלּוּ הֵיכָא דְּיֵשׁ רַגְלַיִם לַדָּבָר דְּהַאי תּוֹרָא אַזְּקֵיהּ מִדְּחָזִינְהוּ דְּקָא רָדִיף {א} וְהֵיכִי דָּמֵי, כְּגוֹן שֶׁרָאָה שׁוֹר וְנָגַח, וְרָאָה שׁוֹר וְלֹא נָגַח, וְרָאָה שׁוֹר וְנָגַח, וְרָאָה שְׁתֵּי פְּעָמִים וְלֹא נָגַח, דַּאֲפִלּוּ מוּעָד לְסֵרוּגִין לֹא הָוֵי. רַשִׁ"י. וּמָצֵי לְפָרוֹשֵׁי נַמִּי כְּגוֹן דְּלֹא אִיַּעַד בְּבֵית דִּין וּכְדִתְנַן בְּמִשְׁנָה ד'. אֶלָּא מִשּׁוּם דְּמִסְתָּמָא מְיַעֲדִין בְּבֵית דִּין לְהָכִי לֹא פֵּרֵשׁ כֵּן: {ב} כְּלוֹמַר וּכְרַבִּי יִשְׁמָעֵאל סְבִירָא לֵיהּ דְּיוּשַׁם הַשּׁוֹר וְלֹא הָוֵי אֶלָּא בַּעַל חוֹב דִּידֵיהּ, וּלְפִיכָךְ כְּשֶׁתָּפַס בְּחוֹבוֹ נַעֲשָׂה שׁוֹמֵר שָׂכָר: {ג} כְּלוֹמַר וּכְרַבִּי עֲקִיבָא סְבִירָא לֵיהּ דְּאָמַר יֻחְלַט הַשּׁוֹר לַחֲצִי נִזְקוֹ. וְנִמְצָא שֶׁאִם כָּחַשׁ אוֹ שִׁבַּח הוּא בִּרְשׁוּת שְׁנֵיהֶם, לֹא נַעֲשָׂה שׁוֹמֵר אֶלָּא מֵחֶלְקוֹ, דְּעַל כָּל אֶחָד מֻטָּל לִשְׁמֹר חֶלְקוֹ. אֲבָל לְרַבִּי יִשְׁמָעֵאל שֶׁאֵין לוֹ חֵלֶק מְבֹרָר בְּגוּף הַשּׁוֹר, לְשִׁעְבּוּדָא בְּעָלְמָא הוּא דְּתָפִיס לֵיהּ, וְהָוֵי כְּמוֹ רְאוּבֵן שֶׁחַיָּב לְשִׁמְעוֹן מָנֶה וְתָפַס שִׁמְעוֹן מִשֶּׁל רְאוּבֵן חֵפֶץ שֶׁשָּׁוֶה מָאתַיִם כְּדֵי לִפְרֹעַ מִמֶּנּוּ, שֶׁחַיָּב בִּשְׁמִירַת כָּל הַחֵפֶץ. אֲבָל לְרַבִּי עֲקִיבָא הָוֵי כְּמוֹ חֵפֶץ שֶׁל שְׁנֵיהֶם שֶׁלִּפְעָמִים זֶה שׁוֹמְרוֹ וְלִפְעָמִים זֶה, וּלְעוֹלָם אַחֲרָיוּתוֹ עַל שְׁנֵיהֶם. הָרֹא"שׁ: {ד} <B> הָאַחֲרוֹן כוּ'.</B> וְאַף עַל גַּב דְּקָיְמָא לָן דְּשֻׁתָּפִין חוֹלְקִין כְּפִי מִסְפַּר הַשֻּׁתָּפִין לֹא לְפִי הַמָּעוֹת כְּמוֹ שֶׁאִיתָא בְּמִשְׁנָה ד' פֶּרֶק י' דִּכְתֻבּוֹת, הַיְנוּ טַעֲמָא לְפִי שֶׁלֹּא נִשְׁתַּתְּפוּ לְדַעַת וְהָוֵי לְהוּ כְּלָקַח זֶה בְּשֶׁלּוֹ וְזֶה בְּשֶׁלּוֹ, דְּקָיְמָא לָן דְּחוֹלְקִין לְפִי הַמָּעוֹת. הָרַ"ן: 

~ פרק ד - משנה ב

{ה} <B> וְאֵינוֹ.</B> מַסְּקִינַן בַּגְּמָרָא דְּלֹא תְּנִינַן בְּוָי"ו אֶלָּא אֵינוֹ מוּעָד, וְדִינָא קָאָמַר, דְּאִיַּעַד לְחַד מִינָא אַף עַל גַּב דְּלֹא מִתְרְמֵי לֵיהּ מִינָא אַחֲרִינָא לְמֶחֱזֵי אִי נָגַח לֵיהּ אִי לֹא, מִסְתָּמָא אֵינוֹ מוּעָד אֶלָּא לְאוֹתוֹ מִין שֶׁהוּעַד לוֹ. נִמּוּקֵי יוֹסֵף: {ו} נִרְאֶה לִי דְּבָעֵי לְפָרֵשׁ קְטַנִּים דִּבְהֵמָה מַה הִיא, דְּלָא תֵימָא דַּקָּה וְגַסָּה, אֶלָּא תַּרְוַיְהוּ בְּחַד מִינָא. וְהוּא הַדִּין בְּאָדָם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} <B> לִגְדוֹלִים.</B> מִסְתָּמָא אֲפִלּוּ לִגְדוֹלִים דְּהַהוּא מִינָא, וְלֹא אָמְרִינַן הוֹאִיל וּפָרַץ בְּהַהוּא מִינָא, לֹא שְׁנָא גְּדוֹלִים דִּידֵיהּ וְלֹא שְׁנָא קְטַנִּים דִּידֵיהּ. גְּמָרָא: {ח} <B> לְשַׁבָּתוֹת.</B> בָּעוּ מִנֵּיהּ כֵּיוָן דִּמְהַנְיָא חֲזָרָה דִּשְׁלֹשָׁה יָמִים לְמִסְתַּר הַעֲדָאָה גְּמוּרָה לְדִידָךְ (בְּפֶרֶק ב' מִשְׁנָה ד'), הֲרֵי כוּ' וְאֵינוֹ מוּעָד לִימוֹת הַחֹל שֶׁבֵּינְתַיִם מַהוּ, מִי אָמְרִינַן כֵּיוָן דַּחֲזָא בִּימוֹת הַחֹל שֶׁבֵּינְתַיִם וְלֹא נָגַח סָתַר לְהוּ לַשַּׁבָּתוֹת [כֵּיוָן דִּמְהַנְיָא חֲזָרָה דִשְׁלֹשָׁה יָמִים] כָּל שֶׁכֵּן דִּמְהַנְיָא לְמִסְתַּר מִקְצָת יְמֵי הַעֲדָאָה. נִמּוּקֵי יוֹסֵף. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} וְלַפֵּרוּשׁ הַשֵּׁנִי צָרִיךְ לוֹמַר שֶׁהֶעֱבִירוּ לְפָנָיו בְּנֵי אָדָם: 

~ פרק ד - משנה ג

{י} <B>שֶׁנֶּאֱמַר כוּ'.</B> וְאַף עַל גַּב דִּלְגַבֵּי תָּם כְּתִיב, בְּהַהִיא אַנְפָּא דְּחַיָּב בְּתָם חַיָּב בְּמוּעָד כִּדְאִיתָא בַּגְּמָרָא. הִלְכָּךְ כֵּיוָן דְּמִעֵט קְרָא רֵעֵהוּ, כָּל הֵיכָא דְּלֹא שָׁווּ נִזָּק וּמַזִּיק לַהֲדָדֵי וַדַּאי לָאו רֵעֵהוּ קָרֵינָא בְּהוּ וּמִפְטְרוּ, נִמּוּקֵי יוֹסֵף: {יא} <B> פָּטוּר.</B> בַּגְּמָרָא פָּרֵיךְ הָא תְּנֵינָא לֵיהּ חֲדָא זִמְנָא (פֶּרֶק קַמָּא מִשְׁנָה ב') וּמְשָׁנֵי תָּנֵי וַהֲדַר מְפָרֵשׁ. וְאָרֵישָׁא לֹא פָּרֵיךְ, דְּהַךְ דְּהָכָא לְגוּפָא, וּלְעֵיל לְפָרוֹשֵׁי בְּאֵיזֶה הֶקְדֵּשׁ: {יב} מִשּׁוּם סֵיפָא אִצְטְרִיךְ, דְּרֵישָׁא מֵרֵעֵהוּ נָפֵיק. מַהַרְשַׁ"ל. וּבַגְּמָרָא מְפָרֵשׁ מַאי וְאוֹמֵר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ד - משנה ד

{יג} <B> חֵרֵשׁ.</B> בְּחֵרֵשׁ שֶׁאֵינוֹ שׁוֹמֵעַ וְאֵינוֹ מְדַבֵּר עָסְקִינַן, דַּהֲוֵי חֵרֵשׁ מֵעִקָּרוֹ. אֲבָל פִּקֵּחַ וְנִתְחָרֵשׁ, בַּר דַּעַת הוּא וְחַיָּב בְּכָל דִּינִין שֶׁבַּתּוֹרָה. נִמּוּקֵי יוֹסֵף: {יד} וּמִמֵּילָא דְּהוּא הַדִּין בְּחֵרֵשׁ וְשׁוֹטֶה. וְהַתּוֹסָפוֹת תָּמְהוּ דְּלֹא דָּמֵי לְהָתָם דְּמַיְרֵי בְּהִזִּיק בְּחַיֵּי הָאָב מִשּׁוּם דְּמִטַּלְטְלֵי דְּיַתְמֵי לֹא מִשְׁתַּעְבְּדֵי לְבַעַל חוֹב. וּמַסְּקֵי דְּהַטַּעַם כֵּיוָן דְּחָס רַחְמָנָא עַל הַתָּם שֶׁלֹּא יְשַׁלֵּם נֶזֶק שָׁלֵם, הִלְכָּךְ לְגַבֵּי יַתְמֵי חָסוּ עֲלַיְהוּ: {טו} <B> אַפּוֹטְרוֹפּוֹס.</B> וְאַף עַל גַּב דִּלְגַבֵּי הֶזֵּק דְּגוּפַיְהוּ קָא פָּסִיק תָּנָא בְּמִשְׁנָה ד' פֶּרֶק ח' פְּגִיעָתָן רָעָה, שָׁאנֵי הָתָם דְּלֹא אֶפְשָׁר לָאַפּוֹטְרוֹפּוֹס לְמֵיקָם בִּנְטִירוּתַיְהוּ וְאִי מוֹקְמָא לְמִיגְבָא מִמָּמוֹנַיְהוּ מְכַלִּינַן לְהוּ לְמָמוֹנַיְהוּ. אֲבָל גַּבֵּי שְׁוָרִין אֶפְשָׁר. נִמּוּקֵי יוֹסֵף: {טז} כְּלוֹמַר אַחַר הַהַעֲדָאָה: {יז} וְאַף עַל גַּב דְּאֵין נִזְקָקִין לְנִכְסֵי יְתוֹמִים אֶלָּא לִשְׁלֹשָׁה דְּבָרִים (בְּרֵישׁ פֶּרֶק ו' דַּעֲרָכִין), הָכָא שָׁאנֵי, דְּאִי אָמַרְתְּ מֵעֲלִיַּת אַפּוֹטְרוֹפּוֹס מִמַּנְעֵי וְלֹא עָבְדֵי. גְּמָרָא. וַאֲפִלּוּ לְאַבָּא שָׁאוּל דְּמִשְׁנָה ד' פֶּרֶק ה' דְּגִטִּין כוּ', דְּהָכָא אֵינוֹ נֶחְשָׁב לְנֶאֱמָן בִּשְׁבִיל זֶה אֶלָּא הוּא כְּשׁוֹמֵר בְּעָלְמָא שֶׁלֹּא יַזִּיק לְעוֹלָם. תּוֹסָפוֹת בְּשֵׁם הַיְרוּשַׁלְמִי: {יח} דִּיכוֹלִין יְתוֹמִים לְמֵימַר אִי הָווּ מְיַעְדֵי לֵיהּ בְּאַנְפָּא דִּילָן הֲוָה עָבְדִינַן לֵיהּ שְׁמִירָה מְעֻלָּה וְלֹא הֲוָה נָגַח. וּלְפִיכָךְ דָּיְנִינַן לֵיהּ כְּתָם עַד שֶׁיָּעִיד בִּפְנֵי בְּעָלִים. וְלֵית הִלְכָתָא כְּרַבִּי מֵאִיר לְגַבֵּי רַבִּי יוֹסֵי. נִמּוּקֵי יוֹסֵף:

{יט} <B> כֹּפֶר.</B> דְּכֹפֶר בְּמוּעָד כְּתִיב. וּמְשַׁלֵּם לַיּוֹרְשִׁים כוּ'. נִמּוּקֵי יוֹסֵף. וְהַיְנוּ דְּמֵי נִזָּק. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} וּבְמוּעָד לְעוֹבְדֵי כּוֹכָבִים הָוֵי מוּעָד לְיִשְׂרָאֵל. דְּלֹא כְּגִרְסַת גְּמָרָא דִּידָן. נִמּוּקֵי יוֹסֵף: {כא} וּכְלוֹמַר שְׁתֵּי פְּעָמִים. וְכֵן הוּא בַּגְּמָרָא: {כב} <B> מִיתָה.</B> דִּבְתָם נַמִּי כְּתִיב סָקֹל יִסָּקֵל. רַשִׁ"י: {כג} בַּגְּמָרָא יָלֵיף לָה. וְדַוְקָא בְּבֶן קְיָמָא וַדַּאי. אֲבָל בֶּן שְׁמֹנָה, וַאֲפִלּוּ בֶּן תִּשְׁעָה פָּחוֹת מִשְּׁלֹשִׁים, פָּטוּר, דְּמִסְּפֵקָא לֹא מַפְקִינַן מָמוֹנָא. נִמּוּקֵי יוֹסֵף. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כד} <B> עֶבֶד כוּ'.</B> בְּעֶבֶד כְּנַעֲנִי הַמֻּטְבָּל לְשֵׁם עַבְדוּת קָאֵי, דְּכֵיוָן דְּשַׁיָּךְ בְּמִצְוֹת חַיְּבֵיהּ רַחְמָנָא לְתוֹרָא קְטָלָא עִלָּוֵיהּ, דְּהָא מָרֵיהּ דְּתוֹרָא הֲוָה מִחַיַּב נַמִּי עִלָּוֵיהּ קְטָלָא, דְּכֵיוָן דְּלָאו עַבְדּוֹ הוּא אֵינוֹ בְּדִין יוֹם אוֹ יוֹמַיִם. אֲבָל אַעֶבֶד נָכְרִי וְהוּא הַדִּין שִׁפְחָה לֹא מִחַיַּב תּוֹרָא קְטָלָא עִלָּוַיְהוּ, דַּאֲפִלּוּ נִתְכַּוֵּן לָהֶם וְהָרַג אֶת יִשְׂרָאֵל פָּטוּר [כִּדְקָתָנֵי בְהֶדְיָא לְנָכְרִי וְהָרַג בֶּן יִשְׂרָאֵל], וּלְעִנְיַן חִיּוּבָא דִּבְעָלִים הֲוָה לֵיהּ כְּמַאן דְּקָטְלֵיהּ לְתוֹרֵיהּ אוֹ לַחֲמָרֵיהּ, דְּאִי תָּם הוּא מְשַׁלֵּם חֲצִי נֶזֶק מִגּוּפוֹ, מוּעָד מְשַׁלֵּם נֶזֶק שָׁלֵם מִן הָעֲלִיָּה. נִמּוּקֵי יוֹסֵף:

{כה} וַעֲרַק לְאַגְמָא כְּדִלְעֵיל. וְנִתְחַכֵּךְ כוּ', כְּלוֹמַר עַכְשָׁיו בִּנְפִילָה אַחֲרוֹנָה. וְהוּא מְשֻׁנֶּה קְצָת, שֶׁהָרַג בִּנְפִילַת הַכֹּתֶל, וּלְכָךְ לֹא נֶחְשָׁב מוּעָד אֶלָּא מֵחֲמַת הָרִאשׁוֹנוֹת. רַשִׁ"י: {כו} וְהוּא הַדִּין לְכוּלְּהוּ הָנָךְ, דְּאִי אִיַּעַד לְמִקְטַל בְּכַוָּנָה אַף עַל גַּב דְּהַשְׁתָּא קָטִיל שֶׁלֹּא בְּכַוָּנָה וּפָטוּר מִמִּיתָה, חַיָּב בְּכֹפֶר. נִמּוּקֵי יוֹסֵף:

{כז} וְאַף עַל גַּב דְּהִשְׁוָה אִשָּׁה לְאִישׁ לְכָל עֳנָשִׁין שֶׁבַּתּוֹרָה, לֹא אָתְיָא אֶלָּא הֵיכָא דְּהַפָּרָשָׁה נֶאֶמְרָה בִּלְשׁוֹן זָכָר, וְהִשְׁוָה הַכָּתוּב, דְּלָא תֵימָא לְשׁוֹן זָכָר דַּוְקָא, אֲבָל הֵיכָא דִּכְתִיב אִישׁ אִצְטְרִיךְ רִבּוּיָא, וְהָכָא אַף עַל גַּב דְּלֹא כְּתִיב אִישׁ אֶלָּא לְשׁוֹן זָכָר הֲוָה יָלְפִינַן מִנְּזָקִין דִּכְתִיב בְּהוּ אִישׁ. תּוֹסָפוֹת. וְשׁוֹר הַיְתוֹמִים, דְּלָא תֵימָא כֵּיוָן דְּלָאו בְּנֵי דֵּעָה נִינְהוּ לֹא מִחַיְּבֵי. וְהָאַפּוֹטְרוֹפּוֹס, דְּלָא תֵימָא כֵּיוָן דְּאִיכָּא שְׁמִירָה כָּל דְּהוּ וְלֹא נָפִישׁ הֶזֵּקָא, אֵימָא חָס רַחְמָנָא עֲלַיְהוּ. וְהַמִּדְבָּר, אַף עַל גַּב דְּלֵית לֵיהּ בְּעָלִים. וְהֶקְדֵּשׁ, אַף עַל גַּב דְּאִיכָּא פְּסֵדָא לַהֶקְדֵּשׁ. נִמּוּקֵי יוֹסֵף. וְשׁוֹר הַגֵּר, דְּלָא תֵימָא דְּהֶפְקֵר וְהֶקְדֵּשׁ דַּוְקָא נָגַח וּלְבַסּוֹף הִקְדִּישׁ וְהִפְקִיר, אֲבָל אִיפְּכָא דְּלֹא הֲוָה לֵיהּ בְּעָלִים מִשְּׁעַת נְגִיחָה לֹא, לְהָכִי אִצְטְרִיךְ. נִרְאֶה לִי: {כח} מִדְּקָתָנֵי תַּרְתֵּי מִדְבָּר וָגֵר, דַּהֲוָה לֵיהּ נַמִּי שׁוֹר הֶפְקֵר, הַיְנוּ רֵישָׁא כוּ', אֶלָּא לָאו הָא קָא מַשְׁמַע לָן דַּאֲפִלּוּ נָגַח וּלְבַסּוֹף הִפְקִיר. גְּמָרָא:

{א} <B> וְנָעַל.</B> פֵּרֵשׁ הָרַ"ב לְקַמָּן, שֶׁהִיא בְּדֶלֶת שֶׁיְּכוֹלָה לַעֲמֹד בְּרוּחַ מְצוּיָה. וְטַעֲמָא מְפָרֵשׁ בַּגְּמָרָא, דְּהַתּוֹרָה מִעֲטָה בִּשְׁמִירַת שֵׁן וָרֶגֶל, דִּכְתִיב בְּהוּ וְשִׁלַּח וּבִעֵר, עַד דַּעֲבֵיד כְּעֵין וְשִׁלַּח, עַד דַּעֲבֵיד כְּעֵין וּבִעֵר, שֶׁיַּאֲכִילֶנָּה בַּיָּדַיִם, כְּלוֹמַר בִּפְשִׁיעָה: {ב} <B> בַּלַּיְלָה.</B> הָא בַּיּוֹם חַיָּב, דְּקָלָא אִית לַהּ לְמִלְּתָא וּמִסְתָּמָא יָדַע שֶׁנִּפְרְצָה. אִי נַמִּי בַּלַּיְלָה אֲפִלּוּ נוֹדַע לוֹ וְיָצְאָה הַבְּהֵמָה אֵין לוֹ לִטְרֹחַ יוֹתֵר מִדַּאי לַחֲזֹר אַחֲרֶיהָ בָּאֲפֵלָה. תּוֹסָפוֹת: {ג} <B> לִסְטִים.</B> וְאִצְטְרִיךְ לְאַשְׁמוֹעִינַן דַּאֲפִלּוּ לִסְטִים פְּטוּרִים כְּשֶׁלֹּא הוֹצִיאוּהָ. תּוֹסָפוֹת: {ד} <B> חַיָּבִים.</B> וְאַף עַל גַּב דִּמְשִׁיכָה לֹא קַנְיָא אֶלָּא בְּמִצְוַת הַמּוֹכֵר, וּמַאן אָמַר לֵיהּ לִמְשֹׁךְ כוּ'. וְנִרְאֶה לִי דִּסְבִירָא לֵיהּ דִּקְנָס הוּא שֶׁקָּנְסוּ חֲכָמִים לַגַּזְלָן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ה} דְּמִדְּקָתָנֵי לְרוֹעֶה וְלֹא קָתָנֵי לְאַחֵר, שְׁמַע מִנֵּיהּ מַאי לְרוֹעֶה, לְרוֹעֶה שֶׁתַּחְתָּיו, כִּדְאִיתָא בַּגְּמָרָא: {ו} <B>לְגִנָּה.</B> בְּאֹנֶס, מִגַּג בְּעָלֶיהָ הַסָּמוּךְ לְגִנַּת חֲבֵרוֹ, אוֹ שֶׁהָיְתָה רְשׁוּת הָרַבִּים גָּבוֹהַּ מִן הַגִּנָּה וְנָפְלָה. רַשִׁ"י: {ז} וְכָךְ פָּסְקוּ הָרִי"ף וְהָרַמְבַּ"ם. אֶלָּא דִּלְפִי זֶה סֵיפָא דִּתְנַן יָרְדָה כוּ', הֲוָה לֵיהּ לְמִתְנֵי דַּאֲפִלּוּ הִפִּילוּהָ חַבְרוֹתֶיהָ מְשַׁלֶּמֶת, וְסִיַּעְתָּא לְהָרֹא"שׁ וְסִיעָתוֹ דִּדְחָפוּהָ כוּ' פָּטוּר: {ח} בַּגְּמָרָא, דְּמִדְּנָקַט בִּשְׂדֵה אַחֵר וְלֹא נָקַט בִּשְׂדֵה חֲבֵרוֹ, אַשְׁמוֹעִינַן עַל גַּב שָׂדֶה אַחֶרֶת. וּמִדְּנָקַט לִקְרָא גַּבֵּי וּבִעֵר וְלֹא נָקַט גַּבֵּי תַּשְׁלוּמִין, וְלֵימָא הָכִי מֵיטַב שָׂדֵהוּ וּמֵיטַב כַּרְמוֹ יְשַׁלֵּם בִּשְׂדֵה אַחֵר, דְּשָׁמִין עַל גַּב שָׂדֶה אַחֶרֶת, דָּרְשִׁינַן נַמִּי דַּוְקָא בִּשְׂדֵה אַחֵר וְלֹא בִּרְשׁוּת הָרַבִּים: {ט} וּלְהָכִי אֵין שָׁמִין עַל גַּב שָׂדֶה שְׁלֵמָה, דְּאִם כֵּן יַפְסִיד הַנִּזָּק, דְּלֹא יְזַלְזֵל בָּהּ בִּשְׁבִיל הֶפְסֵד עֲרוּגָה אַחַת כִּי אִם דָּבָר מֻעָט. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} <B> גְּמוּרִים.</B> דְּכֵיוָן דְּלֹא צְרִיכֵי לַשָּׂדֶה, לֹא עֲלֵיהֶם אָמְרָה תּוֹרָה וּבִעֵר בִּשְׂדֵה אַחֵר אַגַּב שְׂדֵה אַחֵר:

{יא} <B> הֻזְּקָה.</B> וְהָנֵי מִלֵּי שֶׁהֻחְלְקָה, כִּדְפֵרֵשׁ הָרַ"ב בְּמִשְׁנָה ב' פֶּרֶק דִּלְעֵיל: {יב} וַאֲפִלּוּ רַבִּי דְּמִשְׁנָה ג' פֶּרֶק דִּלְעֵיל מוֹדֶה בְּהָא:

{יג} <B> אָדָם.</B> לְפִי שֶׁאֵינָן בְּנֵי שִׁלּוּחִים. תּוֹסָפוֹת: {יד} <B> חַיָּב.</B> וְאַף עַל גַּב דְּיָלְפִינַן דִּשְׁלוּחוֹ שֶׁל אָדָם כְּמוֹתוֹ בְּכָל הַתּוֹרָה כֻּלָּהּ, אָמְרִינַן בַּגְּמָרָא דְּשָׁאנֵי הָכָא, דְּאֵין שָׁלִיחַ לִדְבַר עֲבֵרָה, דְּדִבְרֵי הָרַב וְדִבְרֵי הַתַּלְמִיד דִּבְרֵי מִי שׁוֹמְעִין. אֲבָל בְּחֵרֵשׁ שׁוֹטֶה וְקָטָן כֵּיוָן דְּלָאו בְּנֵי חִיּוּבָא נִינְהוּ לֹא שַׁיָּךְ בְּהוּ דִּבְרֵי הָרַב כוּ', אֶלָּא טַעֲמָא כְּמוֹ שֶׁכָּתוּב בְּסָמוּךְ: {טו} <B> הָעֵצִים כוּ'.</B> דְּאִי לָאו אַחֲרוֹן, קַמָּא לֹא עֲבֵיד מִידֵי. רַשִׁ"י: {טז} <B> הָרוּחַ.</B> וּפֵרֵשׁ בַּיְרוּשַׁלְמִי דַּוְקָא בְּרוּחַ שֶׁאֵינָהּ מְצוּיָה, וְהִיא שֶׁאֵין הָעוֹלָם מִתְנַהֵג בָּהּ תָּמִיד אֶלָּא לִפְעָמִים, אַף עַל פִּי שֶׁהוּא בָּא בְּעִתִּים הַרְבֵּה. וְאֵין צָרִיךְ לוֹמַר רוּחַ סְעָרָה שֶׁאֵינָהּ נוֹשֶׁבֶת אֶלָּא לְעִתִּים רְחוֹקִים. הַמַּגִּיד: {יז} <B> עֵצִים כוּ'.</B> מִדִּכְתִיב הַקָּמָה אַיְּתַר שָׂדֶה לְגוּפוֹ שֶׁל קַרְקַע, דְּהַיְנוּ לִחֲכָה נִירוֹ וְסִכְסְכָה אֲבָנָיו. גְּמָרָא: {יח} <B> גָּבוֹהַּ אַרְבַּע אַמּוֹת.</B> מֵעֲצֵי הַדְּלֵקָה. גְּמָרָא: {יט} <B> אוֹ נָהָר.</B> יֵשׁ אוֹמְרִים דְּבָעִינַן רָחָב שְׁמֹנֶה אַמּוֹת וְיֵשׁ בּוֹ מַיִם. וְיֵשׁ אוֹמְרִים אִם הוּא רָחָב שְׁמֹנֶה אֲפִלּוּ אֵין בּוֹ מַיִם, וְאִם יֵשׁ בּוֹ מַיִם אֲפִלּוּ כָּל שֶׁהוּא: {כ} <B> בְּתוֹךְ שֶׁלּוֹ.</B> אַף עַל גַּב דְּעַד הַשְׁתָּא נַמִּי בְּתוֹךְ שֶׁלּוֹ אָיְרִינַן (דִּבְשֶׁל חֲבֵרוֹ אֲפִלּוּ עָבְרָה גָּדֵר חַיָּב וְאֵין לוֹ שִׁעוּר), מִכָּל מָקוֹם הוֹאִיל וְטַעַם הַפְּטוּר דְּהָכָא מִשּׁוּם דִּבְתוֹךְ שֶׁלּוֹ הוּא, נָקַט מִלְּתָא בְּטַעֲמָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כא} <B> בֵּית כּוֹר.</B> אִם יֵשׁ לוֹ חֲצִי בֵּית כּוֹר לְכָל רוּחַ, פָּטוּר. רַשִׁ"י: {כב} <B> שֵׁשׁ עֶשְׂרֵה אַמּוֹת.</B> לֹא בְּאֶמְצַע שֵׁשׁ עֶשְׂרֵה קָאָמַר, דְּהָא מוֹקְמִינַן לָרֵישָׁא דִּתְנַן דֶּרֶךְ הָרַבִּים כְּוָתֵיהּ, וְהָתָם אֶמְצַע מַאן דָּכַר שְׁמֵיהּ. וּמִדְּרַבִּי אֱלִיעֶזֶר נִשְׁמַע נַמִּי לְרַבִּי עֲקִיבָא דְּבַתְרֵיהּ:

{כג} טַעֲמַיְהוּ יֵשׁ לְפָרֵשׁ, דְּרַבִּי יְהוּדָה לֹא פָּטַר בְּטָמוּן כְּלָל, וְרַבָּנָן בְּמַדְלִיק בְּתוֹךְ שֶׁלּוֹ פָּטְרֵי כִּדְמַשְׁמַע קְרָא כִּי תֵּצֵא אֵשׁ, וּמַשְׁמַע דְּאַיְרֵי בְּמַדְלִיק בְּתוֹךְ שֶׁלּוֹ. אֲבָל בְּמַדְלִיק בְּתוֹךְ שֶׁל חֲבֵרוֹ מְחַיְּבֵי בְּדָבָר שֶׁדַּרְכּוֹ לְהַטְמִין, דְּלֹא אַשְׁכְּחַן דְּפָטַר בֵּיהּ קְרָא טָמוּן, וּבְדָבָר שֶׁאֵין דַּרְכּוֹ לְהַטְמִין פָּטְרֵי מִסְּבָרָא דְּלֹא אִבָּעֵי לֵיהּ לְאַסּוֹקֵי אַדַּעְתֵּיהּ שֶׁיַּנִּיחַ אָדָם אַרְנָקִי בַּגָּדִישׁ. תּוֹסָפוֹת: {כד} <B> חִטִּין כוּ'.</B> אִם חִטִּין חִטִּין. אִם שְׂעֹרִים שְׂעֹרִים. רַשִׁ"י. דְּלֹא שַׁיָּךְ לְמֵימַר בְּהוּ טָמוּן, שֶׁהֲרֵי יוֹדֵעַ שֶׁיֵּשׁ שָׁם חִטִּים אוֹ שְׂעֹרִים. נִמּוּקֵי יוֹסֵף: {כה} <B> גְּדִי כוּ'.</B> שֶׁכֵּן דֶּרֶךְ בְּנֵי אָדָם לַעֲשׂוֹת בַּגָּדִישׁ. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כו} דִּכְתִיב נָקֹם יִנָּקֵם. רַשִׁ"י: {כז} <B> הַבִּירָה.</B> מִגְדָּל גָּדוֹל. רַשִׁ"י:

{א} מַסְּקִינַן בַּגְּמָרָא דְּכָל, רִבּוּיָא הוּא. וּדְלֹא כְּמוֹ שֶׁכָּתַב הָרַ"ב בְּמִשְׁנָה ט' פֶּרֶק ד' דְּבָבָא מְצִיעָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> בְּשׁוֹר וָשֶׂה.</B> בַּגְּמָרָא פָּרֵיךְ, וְנֵילַף שׁוֹר שׁוֹר מִשַּׁבָּת, מַה לְּהַלָּן חַיָּה וָעוֹף כַּיּוֹצֵא בָּהֶן אַף כוּ'. וּמַסִּיק דְּנִכְתּוֹב רַחְמָנָא כִּי יִגְנֹב אִישׁ שׁוֹר וּטְבָחוֹ וּמְכָרוֹ חֲמִשָּׁה בָּקָר יְשַׁלֵּם תַּחְתָּיו וְאַרְבַּע צֹאן תַּחַת הַשֶּׂה, שׁוֹר דְּסֵיפָא וְשֶׂה דְּרֵישָׁא לָמָּה לִי. שְׁמַע מִנַּהּ שׁוֹר וָשֶׂה אִין, מִידֵי אַחֲרִינָא לֹא: 

~ פרק ז - משנה ב

{ג} <B> שְׁנַיִם אֲחֵרִים.</B> עַיֵּן מַה שֶּׁכָּתַבְנוּ בְּמִשְׁנָה ב' פֶּרֶק ו' דְּגִטִּין, דְּלֹא הָוֵי חֲצִי דָּבָר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} כְּלוֹמַר מִשּׁוּם דְּהַךְ סֵיפָא, הוּא דְּתָנֵי לְהַךְ רֵישָׁא. דְּרֵישָׁא לָאו חִדּוּשׁ הוּא, שֶׁהֲרֵי כְּבָר נִתְחַיֵּב קֹדֶם שֶׁמֵּת אָבִיו וְקֹדֶם שֶׁהִקְדִּישׁ כוּ'. הָרֹא"שׁ: {ה} חֻלִּין כוּ'. וְאַף עַל גַּב דְּנֶאֶסְרוּ בַּהֲנָאָה כְּדִתְנַן בְּמִשְׁנָה ט' פֶּרֶק ב' דְּקִדּוּשִׁין, וּמִכִּי שָׁחַט בָּהּ פּוּרְתָא אֲסָרָהּ, אִידָךְ לָאו דְּמָרֵיהּ טָבַח, וְהַהוּא פּוּרְתָא לֹא מְחַיְּבִינַן, דְּוּטְבָחוֹ כֻּלָּהּ בָּעִינַן כְּמוֹ שֶׁכָּתַב הָרַ"ב בְּמִשְׁנָה ד'. וּמַסִּיק כְּגוֹן שֶׁשָּׁחַט מִקְצָת סִימָנִין בַּחוּץ, וּגְמָרָן בִּפְנִים. גְּמָרָא: {ו} דְּגָמַר מִטְּבֹחַ טֶבַח וְהָכֵן, בַּר מִלְּגַבֵּי לָאו דִּשְׁחוּטֵי חוּץ, דִּבְהָכִי חַיְּבֵיהּ רַחְמָנָא. תּוֹסָפוֹת: 

~ פרק ז - משנה ג

{ז} <B> הַכֹּל.</B> הָכִי נַמִּי כְּשֶׁהוּזְמוּ תְּחִלָּה עַל הַטְּבִיחָה. רַשִׁ"י: {ח} כְּלוֹמַר אַהַכְחָשָׁה לֹא מִחַיְּבֵי. וְאַף עַל גַּב דִּבְלָאו הָכִי לֵיכָּא חִיּוּבָא, דְּדִלְמָא בְּעָלִים מְכָרוּהוּ לוֹ, מִכָּל מָקוֹם נָפְקָא מִנַּהּ הֵיכָא שֶׁהֵעִידוּ עֵדִים שֶׁלֹּא נִמְכְּרָה לוֹ, כְּגוֹן שֶׁלֹּא זָזָה יָדָם מִידֵי הַבַּעַל מֵעֵת שֶׁבָּאת בְּהֵמָה זוֹ לְיָדוֹ: 

~ פרק ז - משנה ד

{ט} דִּכְגוֹן שֶׁיֵּשׁ יוֹרְשִׁים אֲחֵרִים עִמּוֹ: {י} <B> וְהִקְדִּישׁ.</B> וּלְאַחַר יֵאוּשׁ אַיְרֵי, דְּחָלָה הֶקְדֵּשׁ עַל יְדֵי יֵאוּשׁ וְשִׁנּוּי רְשׁוּת וְלֹא דְּמָרֵיהּ קָא טָבַח. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} וּסְבִירָא לֵיהּ כָּל הָעוֹמֵד לִזְרֹק כְּזָרוּק דָּמֵי, וְהִלְכָּךְ הָוֵי שְׁחִיטָה רְאוּיָה. אֲבָל אִי לֹא נִשְׁפַּךְ אֶלָּא נִזְרַק, הָא חָזְרָה קֶרֶן לַבְּעָלִים. גְּמָרָא: {יב} וּבְבַעֲלֵי מוּמִין מֵעִקָּרָא עָסְקִינַן, דְּאִי כְּשֶׁקָּדַם הֶקְדֵּשָׁן לְמוּמָן תּוּ לֹא חֲזִי לְפִדְיוֹן, דִּסְבִירָא לֵיהּ לְרַבִּי שִׁמְעוֹן בְּפֶרֶק ז' דִּתְמוּרָה מִשְׁנָה ג' קָדְשֵׁי מִזְבֵּחַ בִּכְלַל הַעֲמָדָה וְהַעֲרָכָה, וּמִשֶּׁנִּשְׁחֲטָה אֵינָהּ יְכוֹלָה לַעֲמֹד. אֲבָל בְּבַעֲלֵי מוּמִין מֵעִקָּרָא מוֹדֶה. רַשִׁ"י: {יג} וְכֵן נַמִּי אִם שְׁחָטָן בַּעַל מוּם בִּפְנִים: 

~ פרק ז - משנה ה

{יד} <B> שֻׁתָּפוּת.</B> קֹדֶם שֶׁגְּנָבוֹ, לֹא קָרִינַן בֵּיהּ וּמְכָרוֹ כֻּלּוֹ בְּאִסּוּר. רַשִׁ"י. וְאַף עַל גַּב דִּכְבָר שָׁמַעְנוּ זֶה לְעֵיל בְּמֵת אָבִיו קֹדֶם שֶׁמְּכָרוֹ. הָתָם הֲוָה אֲמֵינָא מִשּׁוּם דְּיֵשׁ לוֹ שֻׁתָּפוּת בְּכָל אֵבָר וְאֵבָר. אֲבָל הָכָא שָׁמְעִינַן אֲפִלּוּ אֵין לוֹ שֻׁתָּפוּת אֶלָּא בְּאֵבָר אֶחָד. הָרְמָ"ה זַ"ל: {טו} <B> חוּץ כוּ'.</B> דִּכְשֶׁהוֹצִיאוֹ נַעֲשָׂה עָלָיו גַּנָּב, וַאֲפִלּוּ הוֹצִיאוֹ לִרְשׁוּת הָרַבִּים. וְאַף עַל גַּב דְּאֵין מְשִׁיכָה קוֹנָה בִּרְשׁוּת הָרַבִּים, הָנֵי מִלֵּי לְעִנְיַן קְנִיָּה, אֲבָל לְהִתְחַיֵּב בָּאֳנָסִין כֵּיוָן שֶׁהוֹצִיאוֹ מֵרְשׁוּת בְּעָלִים חַיָּב. נִמּוּקֵי יוֹסֵף. וְהַתּוֹסָפוֹת כָּתְבוּ דְּהָכָא בְּהוֹצִיאוֹ לְסִמְטָא דַּוְקָא: {טז} <B> אוֹ כוּ'.</B> נָקַט כֻּלְּהוּ גַּוְנֵי אַף עַל גַּב דְּלֹא צְרִיכֵי: {יז} <B> פָּטוּר.</B> בְּשֶׁלֹּא הִגְבִּיהוֹ עָסְקִינַן, דִּבְהִגְבִּיהַּ הֲוָה קָנֵי [כְּדִלְקַמָּן] וְחַיָּב. תּוֹסָפוֹת: 

~ פרק ז - משנה ו

{יח} הָרַמְבַּ"ם. וּמַשְׁמַע דְּאִלּוּ בְּקֶרֶן נִתְחַיֵּב. וְלֹא יָדַעְתִּי לָמָּה: {יט} <B> חַיָּב.</B> אַגַּנָּב קָאֵי, דְּמִתְחַיֵּב גַּנָּב בְּקִנְיַן שְׁלוּחָיו. רַשִׁ"י. וְאִם תֹּאמַר אַמַּאי חַיָּב בִּמְשִׁיכָה, וְהָא אֵין שָׁלִיחַ לִדְבַר עֲבֵרָה. וְיֵשׁ לוֹמַר דְּהָנֵי לֹא יָדְעֵי דְּאָתֵי לִידֵיהּ בְּאִסּוּרָא, אֶלָּא סְבוּרִין שֶׁהָיוּ שֶׁלּוֹ, וְלֹא שַׁיָּךְ לוֹמַר דִּבְרֵי הָרַב וְדִבְרֵי הַתַּלְמִיד דִּבְרֵי מִי שׁוֹמְעִין. תּוֹסָפוֹת. וְדָחוּ פֵּרוּשׁ זֶה. וְהָרַמְבַּ"ם מְפָרֵשׁ לְמִשְׁנָתֵנוּ כְּשֶׁיָּדַע הַשּׁוֹמֵר אוֹ הַבַּעַל חוֹב שֶׁהוּא גְּנָבוֹ, וְאַף עַל פִּי שֶׁגּוֹנֵב אַחַר הַגַּנָּב פָּטוּר מִכֶּפֶל כוּ', כֵּיוָן שֶׁהַגַּנָּב לֹא קָנָהוּ עֲדַיִן הֲרֵי אֵלּוּ כְּגוֹנְבִין מֵרְשׁוּת הַבְּעָלִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ז - משנה ז

{כ} נִרְאֶה דַּאֲפִלּוּ בַּבַּיִת קָאָמַר דְּאָסוּר. וְהַיְנוּ דְּקָפֵיד אַיִּשּׁוּב וְהַפְסָדַת זְרָעִים, כְּלוֹמַר אֲפִלּוּ שֶׁל עַצְמוֹ: {כא} <B> בִּירוּשָׁלַיִם.</B> אֲפִלּוּ יִשְׂרָאֵל, מִפְּנֵי הַקָּדָשִׁים, שֶׁיִּשְׂרָאֵל אוֹכְלִים שָׁם בְּשַׂר שְׁלָמִים וְתוֹדָה וּמַעֲשַׂר בְּהֵמָה. רַשִׁ"י: {כב} וְאִם תֹּאמַר תֵּיפּוֹק לֵיהּ דְּאָסוּר לַעֲשׂוֹת סְחוֹרָה בְּכָל דְּבָרִים טְמֵאִים, כְּדִתְנַן בְּמַסֶּכֶת שְׁבִיעִית פֶּרֶק ז' מִשְׁנָה ג'. תֵּרֵץ רַבֵּנוּ תָּם זַ"ל, דְּהַיְנוּ דַּוְקָא בִּדְבָרִים הָעוֹמְדִים לַאֲכִילָה, אֲבָל אִם מְגַדְּלָן לִמְשֹׁחַ עוֹרוֹת בְּשֻׁמָּנָן אוֹ לְמָכְרָן לְיִשְׂרָאֵל שֶׁיִּמְשַׁח בָּהֶם שָׁרֵי. וְכָל חֵלֶב מֻתָּר לִמְכֹּר הֵיכָא דְּלֹא קָאֵי לַאֲכִילָה. וּבַיְּרוּשַׁלְמִי נַמִּי מְפָרֵשׁ דְּסוּסִים וַחֲמוֹרִים שָׁרֵי לַעֲשׂוֹת בָּהֶן סְחוֹרָה מִשּׁוּם דִּסְתָמָן לִמְלָאכָה: {כג} <B> הַכֶּלֶב.</B> בְּטוּר וְשֻׁלְחָן עָרוּךְ חֹשֶׁן מִשְׁפָּט הֶעְתִּיקוּ, כֶּלֶב רַע. וְכֵן הוּא בַּגְּמָרָא בִּדְרַבִּי נָתָן כוּ'. וְלִישְׁנָא דְּמַתְנִיתִין [דִּתְנַן] הַכֶּלֶב בְּהֵ"א הַיְדִיעָה דַּיְקָא נַמִּי הָכִי. וְכֵן הוּא עוֹד בַּגְּמָרָא בְּהֶדְיָא מְגַדְּלִין כְּלָבִים כֻּפְרִים: {כד} <B> קָשׁוּר.</B> דְּכֵיוָן דְּקָשׁוּר אֵין מִתְיָרְאִין מִמֶּנּוּ: {כה} כָּךְ תִּרְגֵּם יוֹנָתָן נִמְלְטָה מִפַּח [תְּהִלִּים קכד]. נִמּוּקֵי יוֹסֵף:

~ פרק ח - משנה א

{א} <B> עֶבֶד כוּ'.</B> דְּהָא מַה שֶּׁהוּא חָשׁוּב יוֹתֵר וְנִתְבַּזָּה מֵחֲמַת חֲשִׁיבוּתוֹ זֶהוּ בִּכְלַל בֹּשֶׁת. סְמַ"ע: {ב} <B> כַּמָּה כוּ'.</B> פֵּרוּשׁ עֶבֶד הַנִּמְכָּר בַּשּׁוּק כַּמָּה הָיָה יָפֶה עִם הָאֵבָר, וְכַמָּה הוּא יָפֶה עֶבֶד הַנִּמְכָּר בְּלֹא אוֹתוֹ אֵבָר. וְשׁוּמָא זוֹ לְהָקֵל, דִּלְמֵדִין מִוּבִעֵר בִּשְׂדֵה אַחֵר, דְּאָמְרָה תּוֹרָה לָשׁוּם עֲרוּגָה אַגַּב שָׂדֶה, {א} <B> עֵצִים כוּ'.</B> מַסְּקִינַן בַּגְּמָרָא דְּדַוְקָא בְּשִׁנּוּי דְּלֹא הֲדַר לִבְרִיָּתוֹ כְּבַתְּחִלָּה. וּכְגוֹן שֶׁגָּזַל עֵצִים שֶׁאֵינָם מְשֻׁפִּין וַעֲשָׂאָן כֵּלִים בּוּכָאנֵי, דְּהַיְנוּ שִׁפָּן, דְּלֹא הֲדַר לִבְרִיָּתוֹ, וְנִשְׁתַּנָּה שְׁמוֹ שֶׁנִּקְרָא בּוּכָאנֵי. וְהַיְנוּ דִּתְנַן וַעֲשָׂאָן כֵּלִים: {ב} <B> צֶמֶר כוּ'.</B> הָכִי נַמִּי שֶׁגָּזַל צֶמֶר שֶׁאֵינוֹ טָווּי וְעָשָׂה מֵהֶן לְבָדִים, שֶׁאֵינוֹ חוֹזֵר לִבְרִיָּתוֹ לִהְיוֹת כָּל נִימָא וְנִימָא בִּפְנֵי עַצְמָהּ, וְנִשְׁתַּנָּה שְׁמוֹ שֶׁנִּקְרָא לֶבֶד. וְהַיְנוּ דִּתְנַן וַעֲשָׂאָן בְּגָדִים: {ג} כִּדְיָלֵיף לֵיהּ בַּגְּמָרָא דַּף ס"ז מִקְּרָא [עַיֵּן לְקַמָּן בְּאוֹת ל"ו]. רַשִׁ"י: {ד} <B> הָעוֹמֶדֶת כוּ'.</B> כְּלוֹמַר לַאֲפוֹקֵי שֶׁיָּשׁוּמוּ כְּאִלּוּ יָלְדָה, דְּהָא עַל יְדֵי הַלֵּידָה קְנָאָהּ. וְזֶה לְשׁוֹן הַתּוֹסָפוֹת, וְכָל שֶׁבַח שֶׁמִּשְּׁעַת גְּזֵלָה עַד שֶׁתֵּלֵד, לַגַּזְלָן. עַד כָּאן. וְטַעֲמָא, דִּלְעוֹלָם מְשַׁלֵּם כִּשְׁעַת הַגְּזֵלָה, כִּדְכָיֵל בַּסֵּיפָא, וְקָאֵי אַכּוּלְּהוּ בָּבֵי דְּמַתְנִיתִין, וְלֹא אָזְלִינַן בָּתַר שְׁעַת קְנִיָּה, אֶלָּא מִשֶּׁקְּנָאָהּ חָזְרָה הַשּׁוּמָא לִשְׁעַת הַגְּזֵלָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} רְצוֹנוֹ לוֹמַר דְּדַוְקָא הַוָּלָד וְהַגִּזָּה קוֹנֶה בְּשִׁנּוּי וּמְשַׁלֵּם מַה שֶּׁהָיוּ דְּמֵיהֶם, וְהָעֹדֶף שֶׁשָּׁוֶה עַכְשָׁיו יוֹתֵר הוּא שֶׁלּוֹ. אֲבָל פָּרָה עַצְמָהּ חוֹזֶרֶת בְּעֵינָהּ. דְּלֵידָה וְגִזָּה לָאו שִׁנּוּי הוּא בְּגוּף הַפָּרָה כוּ'. נִמּוּקֵי יוֹסֵף. אֲבָל לְהָרֹא"שׁ, דְּמֵי פָּרָה מַמָּשׁ, דַּהֲוֵי שִׁנּוּי אַף לַפָּרָה עַצְמָהּ וּקְנָאָהּ בּוֹ: {ו} <B> וְיָלְדָה.</B> הוּא הַדִּין אֲפִלּוּ לֹא יָלְדָה, וְאַיְדֵי דְּנָסֵיב רֵישָׁא וְיָלְדָה, נָסֵיב נַמִּי סֵיפָא וְיָלְדָה. גְּמָרָא. וְאַף עַל גַּב דִּבְרֵישָׁא נַמִּי מְשַׁלֵּם כִּשְׁעַת הַגְּזֵלָה, סֵיפָא הָא קָא מַשְׁמַע לָן דְּעִבּוּר וּטְעִינָה הָוֵי שִׁנּוּי כְּמוֹ לֵידָה. וְנִרְאֶה לִי דְּהָכִי נַמִּי דִּינָא דְּמַחֲזִיר הַפָּרָה לְפֵרוּשׁ הַנִּמּוּקֵי יוֹסֵף וּכְדִלְעֵיל. דְּמַאי שְׁנָא שִׁנּוּי הַמִּלּוּי אוֹ שִׁנּוּי הָרֵקוּת. וְקַשְׁיָא לִי, דְּלִתְנֵי נַמִּי בָּרֵישָׁא מְשַׁלֵּם כִּשְׁעַת הַגְּזֵלָה. וְצָרִיךְ עִיּוּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} וְלֹא אָמְרִינַן דְּעַל כָּל פָּנִים יְשַׁלֵּם אַרְבָּעָה וַחֲמִשָּׁה כִּשְׁעַת הַגְּזֵלָה, לְפִי דְּשֶׁלּוֹ הוּא טוֹבֵחַ כוּ'. וּמִמֵּילָא דְּמִכָּל שֶׁכֵּן שֶׁאֵינוֹ מְשַׁלֵּם הַכֶּפֶל אֶלָּא כִּשְׁעַת הַגְּזֵלָה:

{ח} מִשּׁוּם דְּבַגְּמָרָא מְהַפְּכִינַן לַהּ כוּ' וְתָנֵי וַחֲכָמִים אוֹמְרִים בַּעֲבָדִים כוּ': {ט} כֵּיוָן דְּאוֹרְחַיְהוּ בְּהָכִי, כְּהֶזֵּק שֶׁאֵינוֹ נִכָּר חָשְׁבִינַן לֵיהּ, וְאוֹמֵר לוֹ הֲרֵי כוּ', וְהָוֵי כְּחָמֵץ בְּפֶסַח. אֲבָל מַתְנִיתִין בְּשֶׁהִרְקִיבוּ כֻּלָּן שֶׁהוּא הֶזֵּק נִכָּר כְּיַיִן וְהֶחְמִיץ מַיְרֵי. נִמּוּקֵי יוֹסֵף: {י}  שֶׁיָּכוֹל לוֹמַר לוֹ לֵךְ וְהוֹצִיאוֹ שָׁם. רַשִׁ"י: {יא} <B> תְּרוּמָה כוּ'.</B> הַטַּעַם דְּאֵלּוּ אַף שֶׁנִּפְסְלוּ אֵצֶל הַגַּזְלָן לְגַמְרֵי, מִכָּל מָקוֹם תְּבוּאַת תְּרוּמָה וְלֶחֶם חָמֵץ לֹא נִשְׁתַּנָּה מִשְּׁאָר תְּבוּאוֹת וּלְחָמִין, וַהֲרֵי הֵן כְּמוֹ שֶׁהָיוּ בִּשְׁעַת גְּזֵלָה וּכְמוֹ שֶׁהֵן עַכְשָׁיו שְׁאָר תְּבוּאוֹת וּלְחָמִין, מִשּׁוּם הָכִי מָצֵי אָמַר לֵיהּ הֲרֵי כוּ'. מַה שֶּׁאֵין כֵּן בִּפְסַלְתּוֹ מַלְכוּת, כֵּיוָן דְּצוּרַת מַטְבֵּעַ זוֹ אֵינָהּ יוֹצְאָה כְּלָל בְּכָל מְדִינוֹת הַמֶּלֶךְ וְנִטְבְּעָה צוּרָה אַחֶרֶת בִּמְקוֹמָהּ, הֲרֵי נִכָּר הֶפְסֵדָהּ. סְמַ"ע: {יב} <B> וְנֶעֶבְדָה כוּ'.</B> בְּעֵד אֶחָד, שֶׁלֹּא נִסְקְלָה. אֲבָל פְּסוּלָה לְקָרְבָּן. וְטַעֲמָא דְּאָמַר לֵיהּ הֲרֵי כוּ', דְּכָל בְּהֵמוֹת לָאו לְמִזְבֵּחַ קַיְּמֵי. רַשִׁ"י: {יג} <B> לִסָּקֵל.</B> סְבִירָא לֵיהּ לְהַאי תָּנָא דְּגוֹמְרִין דִּינוֹ שֶׁל שׁוֹר שֶׁלֹּא בְּפָנָיו וְאֵין צָרִיךְ לְהָבִיאוֹ לְבֵית דִּין, דְּאָמַר לֵיהּ מַאי עָבְדֵי לֵיהּ, גְּמָרָא:

{יד} דְּאֵין אֻמָּן קוֹנֶה בְּשֶׁבַח כְּלִי. גְּמָרָא. וְלֹא דָּמֵי לְגַזְלָן שֶׁמְּשַׁלֵּם כִּשְׁעַת הַגְּזֵלָה, דְּהָכָא לֹא כִּוֵּן מֵעוֹלָם לִגְזֹל וְלֹא זָכָה בָּהּ. וְעַיֵּן מַה שֶּׁכָּתַבְנוּ לְקַמָּן בְּדִבְרֵי רַבִּי מֵאִיר וְרַבִּי יְהוּדָה: {טו} <B> חַיָּב.</B> שֶׁזֶּה כְּזוֹרֵק חֵץ וְהִזִּיק בּוֹ הוּא. הָרַמְבַּ"ם:

{טז} <B> דְּמֵי כוּ'.</B> וּבְהִקְדִּיחָה בִּשְׁעַת נְפִילָה, שֶׁלֹּא הָיָה בּוֹ שֶׁבַח מֵעוֹלָם. אֲבָל אִם הִקְדִּיחָה לְאַחַר שֶׁקָּלַט וְנַעֲשָׂה שֶׁבַח בַּצֶּמֶר, חַיָּב לִתֵּן לוֹ גַּם דְּמֵי שִׁבְחוֹ, כְּמוֹ נָתַן לָאֻמָּן לְתַקֵּן בְּמַתְנִיתִין דִּלְעֵיל. גְּמָרָא. וְלֹא יָדַעְתִּי לָמָּה לֹא הֶעְתִּיקוּם הַפּוֹסְקִים: {יז} <B> הַיְצִיאָה.</B> פֵּרֵשׁ הָרַמְבַּ"ם מַה שֶּׁהוֹצִיא מִדָּמָיו. וְרַשִׁ"י הוֹסִיף שְׂכַר טָרְחוֹ כִּשְׁאָר שְׂכִיר יוֹם: {יח} רַשִׁ"י. וּלְקַמָּן בְּדִבְרֵי יְהוּדָה בְּהוֹצָאָה יְתֵרָה עַל שֶׁבַח, פֵּרְשׁוּ שֶׁאִם הַשָּׂכָר פָּחוֹת מִן הַשֶּׁבַח שֶׁנּוֹתֵן לוֹ אֶת הַשָּׂכָר, וּמִסְתַּבֵּר וַדַּאי דְּהָכִי נַמִּי אִם הַשָּׂכָר פָּחוֹת מִן הַיְצִיאוֹת דְּנוֹתֵן לוֹ: {יט} וּמַשְׁמַע דְּכִי יָהֵיב לֵיהּ אַגְרָא מִחַיַּב לְמֵיתַב הַצֶּמֶר וְלֹא אָמְרִינַן דְּקָנֵי לֵיהּ לְגַמְרֵי, שֶׁהֲרֵי לֹא נִתְכַּוֵּן הַצּוֹבֵעַ לִזְכּוֹת בּוֹ כְּלָל. מַה שֶּׁאֵין כֵּן בְּגַזְלָן: {כ} רַשִׁ"י. וּמַשְׁמַע דַּוְקָא לִדְשִׁנָּה קָנְסִינַן. וְהִלְכָּךְ אִם בַּעַל הַצֶּמֶר אֵינוֹ חָפֵץ בַּצֶּמֶר אֶלָּא שֶׁיִּתֵּן לוֹ דְּמֵי צַמְרוֹ, הָרְשׁוּת בְּיָדוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כא} <B> אֶת הַיְצִיאָה.</B> לְפִי שֶׁאֵין מִתְכַּוֵּן לִקְנוֹתוֹ. הִלְכָּךְ לֹא דָּמֵי לְגַזְלָן דְּכֻלָּא שְׁבָחָא דִּידֵיהּ הִיא כְּדִתְנַן רֵישׁ פִּרְקִין. תּוֹסָפוֹת:

כְּעֵין שֶׁגָּזַל, יַחֲזִיר, כְּמוֹ שֶׁכָּתַבְתִּי בְּרֵישׁ פִּרְקִין. שְׁמַע מִנַּהּ דִּקְרָא לֹא אַיְרֵי אֶלָּא אִם הוּא כְּעֵין שֶׁגָּזַל שֶׁלֹּא נִשְׁתַּנָּה. נִרְאֶה לִי: {לז} עֵרוּב פֵּרוּשִׁים אֲנִי רוֹאֶה בְּדִבְרֵי הָרַ"ב. דְּזֶה יָצָא לוֹ מֵהָרַמְבַּ"ם, וְהוּא מְפָרֵשׁ שֶׁאֵין לוֹ אַחִים. וְהָרַ"ב כְּבָר פֵּרֵשׁ אֵין לוֹ נְכָסִים, כְּשִׁיטַת שְׁאָר הַמְפָרְשִׁים, וּלְדִידְהוּ יֵשׁ לוֹמַר דְּנָקַט בָּנָיו כוּ' דִּשְׁכִיחֵי, וְהוּא הַדִּין אִם אֵין לוֹ [אַחִים] דְּהַיְרֻשָּׁה מְמַשְׁמֶשֶׁת וְעוֹלָה עַד יַעֲקֹב אָבִינוּ, וַאֲלֵיהֶם [וְלָהֵם] הוּא נוֹתֵן. וְזֶה לְשׁוֹן הָרַמְבַּ"ם, וְאִם לֹא הָיָה אֶצְלוֹ אָח מֵאָבִיו וְלֹא בָּנִים (וְהוּא פֵּרוּשׁ דִּוְאִם אֵין לוֹ) אוֹ לֹא הֶחֱזִיר הוּא אֶת הַגְּזֵלָה לְבָנָיו (וְהוּא פֵּרוּשׁ דִּוְאִם אֵינוֹ רוֹצֶה) יִתֵּן אוֹתוֹ בְּחוֹבוֹ וְכוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לח} <B> וְנִפְרָעִים.</B> דְּמַה שֶּׁמַּחֲזִיר לָהֶם לֹא חָשֵׁיב כְּאִלּוּ נָתַן לָהֶם בְּמַתָּנָה וְשֶׁלֹּא יוּכְלוּ בַּעֲלֵי חוֹב לִטְרֹף, דְּאֵין אַחֲרָיוּת עַל הַמִּטַּלְטְלִים. דְּאֵינוֹ נוֹתֵן לָהֶם לְשֵׁם מַתָּנָה אֶלָּא בִּסְתָמָא בִּשְׁתִיקָה מַחֲזִיר לָהֶם וְאֵין דַּעְתּוֹ אֶלָּא לְקַיֵּם מִצְוַת הֲשָׁבָה בְּעָלְמָא, הִלְכָּךְ לֹא הַוְיָא מַתָּנָה, וְיָכוֹל בַּעַל חוֹב לִטְרֹף מֵהֶם. וְהוּא וַדַּאי אֵינוֹ יָכוֹל לַחֲזֹר וְלִקַּח מֵהֶן, דְּאָז לֹא הָיָה מִתְקַיֵּם הֲשָׁבָה כְּלָל אִם הָיָה יָכוֹל לִקַּח בְּעַצְמוֹ מַה שֶּׁנָּתַן. אֲבָל מַה שֶּׁבַּעַל חוֹב נוֹטֵל מִתְקַיֵּם שַׁפִּיר מִצְוַת הֲשָׁבָה, כֵּיוָן שֶׁבְּעַצְמוֹ אֵינוֹ נוֹטֵל. תּוֹסָפוֹת:

{לט} <B> קוֹנָם.</B> מְבֹאָר בְּרֵישׁ נְדָרִים שֶׁהָאוֹמֵר קוֹנָם תּוֹפֵס הַחֵפֶץ בְּקָרְבָּן, שֶׁקּוֹנָם כִּנּוּי לְקָרְבָּן הוּא, וְיִהְיֶה פֵּרוּשׁ הַלָּשׁוֹן, קוֹנָם יִהְיֶה אִם אַתָּה נֶהֱנֶה כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מ} <B> יִירָשֶׁנּוּ.</B> שֶׁזֶּה כְּאוֹמֵר נְכָסַי עָלֶיךָ אֲסוּרִים (נְדָרִים פ"ה מ"ג). הָרַמְבַּ"ם: {מא} <B> לֹא כוּ'.</B> שֶׁזֶּה כְּמוֹ שֶּׁאָמַר לֵיהּ נְכָסִים אֵלּוּ אֲסוּרִים עָלֶיךָ. שָׁם. וְכָתַב הָרַ"ן לָאו לְמֵימְרָא דְּכֵיוָן דִּנְכָסִים אֲסוּרִים לוֹ בַּהֲנָאָה לֹא יִזְכֶּה בָּהֶן, דְּהָא קָתָנֵי סֵיפָא וְיִתֵּן לְבָנָיו וּלְאֶחָיו וְאִם אֵין לוֹ לֹוֶה וּבַעֲלֵי חוֹב כוּ', וְאִם לֹא זָכָה בְּגוּף הַנְּכָסִים הֵיאַךְ נוֹתְנָם לְבָנָיו אוֹ לְאֶחָיו, וְהֵיאַךְ בַּעֲלֵי חוֹב בָּאִים וְנִפְרָעִים, אֶלָּא וַדַּאי נְכָסִים דִּידֵיהּ נִינְהוּ אֶלָּא שֶׁאֵינוֹ רַשַּׁאי לֵהָנוֹת מֵהֶם. עַד כָּאן. וּמִשּׁוּם הָכִי נִיחָא לְשׁוֹן וְיַחֲזִיר. אֲבָל הָרַ"ן הֶעְתִּיק וְיִתֵּן. וְכֵן הוּא בַּיְרוּשַׁלְמִי: {מב} <B> וְיַחֲזִיר.</B> לָאו לְמֵימְרָא שֶׁיִּתְּנוּ לָהֶם מַמָּשׁ, אֶלָּא שֶׁמַּרְאֶה לָהֶם מָקוֹם וְאוֹמֵר לָהֶם נְכָסַי אֵלּוּ אָסַר עָלַי אַבָּא וְאֵינִי יוֹדֵעַ מַה אֶעֱשֶׂה בָּהֶם, טְלוּ לְעַצְמְכֶם וַעֲשׂוּ מֵהֶם מַה שֶּׁתִּרְצוּ. וּכְהַאי גַּוְנָא לָאו מַתָּנָה הוּא אֶלָּא גַּוְנָא דְּהֶפְקֵר. אֲבָל הַבֵּית יוֹסֵף כָּתַב שֶׁאֲפִלּוּ לִתְּנָם מַמָּשׁ מֻתָּר, כֵּיוָן שֶׁמּוֹדִיעָם שֶׁמִּפְּנֵי שֶׁאֲסוּרִים עָלָיו נוֹתְנָם לָהֶם אֵין כָּאן טוֹבַת הֲנָאָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מג} לָאו דַּוְקָא. וְעַיֵּן פֶּרֶק ד' דִּנְדָרִים מִשְׁנָה ז'. וְלֹא תָּנֵי הָכָא נַמִּי אֵינוֹ רוֹצֶה, מִשּׁוּם דְּוַדַּאי אֵינוֹ רוֹצֶה. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מד} <B> וְנִפְרָעִים.</B> אַחַר מִיתָה. וַאֲפִלּוּ בְּחַיָּיו הֲוָה שָׁרֵי בִּכְהַאי גַּוְנָא, כְּדִתְנַן (שָׁם) הַמֻּדָּר הֲנָאָה מֵחֲבֵרוֹ וְאֵין לוֹ מַה יֹּאכַל יֵלֵךְ אֵצֶל חֶנְוָנִי כוּ'. אֶלָּא לְפִי שֶׁזֶּה אֵין רוֹצֶה שֶׁיֵּהָנֶה מִמֶּנּוּ כְּלָל בְּחַיָּיו וְאֵינוֹ יָכוֹל לַעֲשׂוֹת תַּקָּנָה אֶלָּא מִדַּעְתּוֹ. תּוֹסָפוֹת:

{מה} <B> לַכֹּהֵן.</B> תָּנוּ רַבָּנָן אָשָׁם זֶה קֶרֶן. הַמּוּשָׁב זֶה חֹמֶשׁ. כְּשֶׁהוּא אוֹמֵר מִלְּבַד אֵיל הַכִּפּוּרִים, הֱוֵי אוֹמֵר אָשָׁם זֶה קֶרֶן. וְלָמָּה קוֹרֵהוֹ אָשָׁם, שֶׁאִם הֶחֱזִירוֹ בַּלַּיְלָה דְּלֹא יָצָא, הֶחֱזִירוֹ לַחֲצָאִין לֹא יָצָא, וְאֵין חוֹלְקִים גֶּזֶל הַגֵּר כְּנֶגֶד גֶּזֶל הַגֵּר, כְּדִין אָשָׁם. גְּמָרָא:

{מו} מַשְׁמַע דִּרְצוֹנוֹ לוֹמַר טַעֲמָא שֶׁזָּכוּ בַּמֶּה שֶׁבְּיָדָם כְּדִין נִכְסֵי הַגֵּר כוּ' שֶׁכָּל הַקּוֹדֵם בָּהֶן זָכָה. וּתְמֵהַנִי, דְּאִי הָכִי קְרָא לְמַאי אִצְטְרִיךְ כוּ'. וּבַגְּמָרָא אָמַר אַבַּיֵי שְׁמַע מִנַּהּ כֶּסֶף מְכַפֵּר מֶחֱצָה, דְּאִי לֹא מְכַפֵּר הֲוָה אֲמֵינָא מְהַדַּר לַיּוֹרְשִׁין, מַאי טַעֲמָא, אַדַּעְתָּא דְּהָכִי (דְּלֹא תַּהֲוֵי לֵיהּ כַּפָּרָה) לֹא יָהֵיב לֵיהּ כוּ'. הָא קַמָּן דְּהַטַּעַם מִשּׁוּם דִּמְכַפֵּר מֶחֱצָה. וּלְהַךְ סְבָרָא בָּעִינַן קְרָא דְּלֹא הֲוָה יָדְעִינַן דְּכִפֵּר מֶחֱצָה. אֶלָּא הֲוָה אֲמֵינָא דְּאֵין כַּפָּרָה עַד שֶׁיָּבִיא קָרְבָּן: {מז} <B> יָצָא.</B> יָלְפִינַן לֵיהּ בַּגְּמָרָא מִקְּרָא: {מח} <B> אֵין כוּ'.</B> בַּגְּמָרָא יָלֵיף לֵיהּ: {מט} אֵינוֹ מְדֻקְדָּק, דְּמַשְׁמַע דְּאִי לֹא נָתַן לְבַסּוֹף עִכֵּב, וּכְבָר פֵּרֵשׁ דְּהָעַכָּבָה הַיְנוּ עַל הַהַקְרָבָה, וְכֵיוָן שֶׁכְּבָר קָרְבָה אֵין עוֹד עַכָּבָה כוּ'. וְאוּלַי רְצוֹנוֹ לוֹמַר דְּלָא תֵימָא דִּבְדִיעֲבַד סַגִּי לֵיהּ וְיִפָּטֵר מִן הַחֹמֶשׁ, לְהָכִי קָאָמַר וּלְבַסּוֹף נָתַן, כְּלוֹמַר שֶׁלְּבַסּוֹף יִתֵּן וְלֹא יִפָּטֵר בְּכָךְ. וְקָרֵי בֵּיהּ נוֹתֵן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} דִּרְשׁוּת יוֹרֵשׁ לָאו כִּרְשׁוּת לוֹקֵחַ דָּמֵי, וְיֵאוּשׁ כְּדִי לֹא קָנֵי. רַשִׁ"י. וְהָא דְּדִיֵּק הָרַ"ב לִכְתֹּב וְהִיא בְּעֵינָהּ, מִשּׁוּם וְחַיָּבִים לְהַחֲזִיר דְּנָקַט, כְּלוֹמַר דְּחַיָּבִים לְהַחֲזִיר הַגְּזֵלָה עַצְמָהּ. אֲבָל כְּשֶׁאֵינָהּ בְּעֵינָהּ אֶלָּא נִשְׁתַּנֵּית בְּשִׁנּוּי דְּאָמְרָן בְּרֵישׁ פִּרְקִין דִּלְעֵיל, אֵין חַיָּבִים אֶלָּא דְּמֵי הַגְּזֵלָה, וַהֲרֵי דִּין הַבָּנִים כְּדִין אֲבִיהֶם: {ב} וְאַף עַל גַּב דִּתְבִיעָה בְּעַל פֶּה הִיא: {ג} בִּגְדוֹלִים אַיְרֵי, דִּקְטַנִּים פְּגִיעָתָן רָעָה: {ד} כֶּפֶל לָשׁוֹן הוּא, דְּהָא בְּהָכִי אַיְרִינַן: {ה} דְּאִלּוּ סְלָעִים בְּסֶלַע דְּלֹא מִתְהַנֵּי בַּחִלּוּף לֵיכָּא אִסּוּרָא. וְהָא דְּפֵרֵשׁ דְּאִם חַיָּב אָדָם פְּרוּטוֹת כוּ' דְּשָׁרֵי מִשּׁוּם שֶׁהוּא כְּמַצִּיל מִיָּדוֹ, הַיְנוּ לוֹמַר הֵיכָא דְּמִתְכַּוֵּן לָתֵת סֶלַע כְּדֵי לְקַבֵּל פְּרוּטוֹת, דְּמִתְהַנֵּי. שִׁלְטֵי גִּבּוֹרִים: {ו} <B> אֵין כוּ'.</B> לְשׁוֹן הָרֹא"שׁ, כָּתְבוּ הַתּוֹסָפוֹת דְּאָתֵי כְּרַבָּנָן דְּרַבִּי שִׁמְעוֹן מִשְׁנָה ח' פֶּרֶק כ"ו דְּכֵלִים דְּאָמְרֵי סְתָם גְּזֵלָה לָאו יֵאוּשׁ בְּעָלִים הִיא. וְלִי נִרְאֶה דְּכָל מַה שֶּׁהַמּוֹכֵס וְהַגַּבַּאי נוֹטֵל הָוֵי יֵאוּשׁ בְּעָלִים, וַאֲפִלּוּ אֵין לוֹ קִצְבָּה. וַאֲפִלּוּ הָכִי דָּבָר מְגֻנֶּה הוּא שֶׁיְּקַבֵּל אָדָם מָמוֹן מִמָּקוֹם הַמְיֻחָד לְגָזֵל. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} אַף עַל פִּי שֶׁהוּא עוֹמֵד מֵחֲמַת הַמֶּלֶךְ הֲרֵי הוּא כְּמוֹכֵס שֶׁאֵין לוֹ קִצְבָּה, שֶׁסְּתָמָא יוֹתֵר מִן הַדִּין הוּא נוֹטֵל. הַמַּגִּיד:

{ח} דְּשִׁנּוּי רְשׁוּת הַיְנוּ רְשׁוּת עַצְמוֹ דְּהַאִיךְ דִּמְקַבֵּל מֵהֶם, וּמַהֲנֵי שִׁנּוּי רְשׁוּתוֹ כֵּיוָן שֶׁבָּא לְיָדוֹ אַחַר שֶׁנִּתְיָאֲשׁוּ הַבְּעָלִים כְּבָר. תּוֹסָפוֹת: {ט} <B> נָחִיל כוּ'.</B> הָא קָא מַשְׁמַע לָן דְּאַף עַל גַּב דִּדְבוֹרִים (שֶׁאֵינָן בִּרְשׁוּתוֹ שֶׁל אָדָם כְּמוֹ תַּרְנְגוֹלִים וַאֲוָזִים. הָרַמְבַּ"ם) וְקִנְיָן דִּידְהוּ אֵינוֹ אֶלָּא מִדְּרַבָּנָן, אֲפִלּוּ הָכִי אִם נִתְיָאֲשׁוּ אִין, אִי לֹא לֹא. גְּמָרָא: {י} <B> מִכָּאן כוּ'.</B> כְּגוֹן שֶׁהָיוּ בְּעָלִים מְרַדְּפִים אַחֲרֵיהֶם וְאִשָּׁה וְקָטָן מְסִיחִים לְפִי תֻּמָּם וְאוֹמְרִים מִכָּאן יָצָא נָחִיל זֶה, הוֹאִיל וְקִנְיַן דְּבוֹרִים מִדִּבְרֵיהֶם הֶאֱמִינוּם בּוֹ. גְּמָרָא: {יא} <B> אַף כוּ'.</B> תַּנְיָא רַבִּי יִשְׁמָעֵאל כוּ' תְּנַאי בֵּית דִּין הוּא שֶׁיְּהֵא יוֹרֵד כוּ' וְקוֹצֵץ כוּ', שֶׁעַל מְנָת כֵּן הִנְחִיל יְהוֹשֻׁעַ לְיִשְׂרָאֵל אֶת הָאָרֶץ: 

~ פרק י - משנה ג

{יב} דְּאִי לְאַחַר יֵאוּשׁ הֲרֵי קְנַנְהוּ בְּיֵאוּשׁ וְשִׁנּוּי רְשׁוּת, כְּמוֹ שֶׁכָּתַב הָרַ"ב לְעֵיל. וְאַף עַל גַּב דְּלִפְנֵי יֵאוּשׁ הוּא (וּבְרֵישׁ פִּרְקִין פֵּרֵשׁ הָרַ"ב לִפְנֵי יֵאוּשׁ חַיָּבִים לְשַׁלֵּם), נוֹטֵל הַדָּמִים מִפְּנֵי תַּקָּנַת הַשּׁוּק. פֵּרֵשׁ רַשִׁ"י עַל שֶׁקְּנָאוֹ לוֹקֵחַ בַּשּׁוּק בְּפַרְהֶסְיָא וְלֹא הֵבִין בּוֹ שֶׁגָּנַב, עָשׂוּ לוֹ תַּקָּנָה שֶׁיְּשַׁלֵּם לוֹ בַּעַל הַבַּיִת מְעוֹתָיו. וַאֲפִלּוּ לְמַאן דְּאָמַר סְתָם גְּנֵבָה יֵאוּשׁ בְּעָלִים, מַיְרֵי בִּבְעָלִים מְרַדְּפִים אֶת הַגְּנֵבָה, וּבִסְפָרִים לֵיכָּא יֵאוּשׁ כְּלָל, שֶׁיּוֹדֵעַ שֶׁיָּבֹאוּ לְיַד יִשְׂרָאֵל. תּוֹסָפוֹת. וְנִרְאֶה לִי דִּלְהָכִי תָּנֵי סְפָרִים, לְגִלּוּיֵי דְּכֵלָיו לִפְנֵי יֵאוּשׁ אַיְרֵי, וַהֲווּ דּוּמְיָא דִּסְפָרָיו. וְאִי לֹא תְּנַן כֵּלָיו, הֲוָה אֲמֵינָא דִּסְפָרָיו הַיְנוּ סִפְרֵי תּוֹרָה דְּאָסוּר לְמָכְרָם. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} וְטַעֲמָא דְּהָא מְאַבֵּד אֶת שֶׁלּוֹ לְהַצִּיל אֶת שֶׁל חֲבֵרוֹ, וּבְכִי הָא לֹא מִחַיַּב לְהָשִׁיב אֲבֵדָה, דַּאֲבֵדָתוֹ וַאֲבֵדַת חֲבֵרוֹ שֶׁלּוֹ קוֹדֶמֶת, כְּדִתְנַן סוֹף פֶּרֶק ב' דְּבָבָא מְצִיעָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק י - משנה ד

{יד} <B> חָבִית שֶׁל דְּבַשׁ כוּ' שְׂכָרוֹ.</B> גְּמָרָא, כְּשֶׁעַקַל בֵּית הַבַּד כָּרוּךְ עָלֶיהָ. פֵּרֵשׁ רַשִׁ"י חָבִית שֶׁל דְּבַשׁ קְשׁוּרָה בַּעַקַל וְלֹא יִשָּׁפֵךְ הַכֹּל אֶלָּא מְנַטֵּף מְעַט מְעַט וְלֵיכָּא הֶפְקֵר. עַד כָּאן. וְלֹא יָכוֹל לוֹמַר מֵהֶפְקֵרָא קָא זָכִינָא כְּהָנָךְ דְּמִשְׁנָה ב': {טו} <B> חַיָּב.</B> מִלְּשׁוֹן הָרַ"ב מִשְׁנָה ט' פֶּרֶק ב' דְּבָבָא מְצִיעָא מַשְׁמַע שֶׁצָּרִיךְ שֶׁיֹּאמַר לוֹ שֶׁיַּצִּיל. וְדַוְקָא בְּמָקוֹם שֶׁיֵּשׁ לוֹ הֶפְסֵד כִּי מַתְנִיתִין. אֲבָל כְּשֶׁאֵין לוֹ הֶפְסֵד אִיתָא בַּגְּמָרָא דַּף קט"ז דַּאֲפִלּוּ אָמַר לֵיהּ בַּעַל הָאֲבֵדָה תַּצִּיל וְאֶתֵּן לְךָ כָּךְ וְכָךְ, אֵין לוֹ אֶלָּא שְׂכָרוֹ, דְּיָכוֹל לוֹמַר מְשַׁטֶּה אֲנִי בְּךָ: {טז} <B> מָאתַיִם.</B> אוֹרְחָא דְּמִלְתָּא נָקַט, דְּאִי שְׁנֵיהֶם שָׁוִים לָמָּה יַנִּיחַ שֶׁלּוֹ וְיַצִּיל שֶׁל חֲבֵרוֹ. וְכֵן כָּתַב סְמַ"ע: {יז} <B> אָמַר לוֹ.</B> הוּא הַדִּין בִּפְנֵי בֵּית דִּין, כְּדִתְנַן בְּמִשְׁנָה ט' פֶּרֶק ב' דְּבָבָא מְצִיעָא. אֶלָּא מִשּׁוּם דְּרֵישָׁא מַיְרֵי וַדַּאי כְּשֶׁיֵּשׁ שָׁם הוּא, מַחֵית לְסֵיפָא בְּחַד מַחְתָּא:

~ פרק י - משנה ה

{יח} <B> מֵחֲמַת כוּ'.</B> דָּיְקִינַן בַּגְּמָרָא מִדְּלֹא קָתָנֵי וְאִם לָאו כוּ', מִשּׁוּם דְּהָא מֵרֵישָׁא שָׁמְעִינַן לַהּ, וְהַאי אִם מֵחֲמַת הַגַּזְלָן מִלְּתָא אַחֲרִיתָא אַשְׁמְעִינַן הֵיכָא דַּאֲנָסוּהוּ לְהַרְאוֹת כָּל מַה שֶּׁיֵּשׁ לוֹ וְהֶרְאָה שָׂדֶה זוֹ שֶׁגָּזַל בִּכְלַל שְׂדוֹתָיו, כֵּיוָן דְּלֹא אֲנָסוּהוּ לְהַרְאוֹת אֶלָּא עַל שֶׁלּוֹ לֹא הֲוָה לֵיהּ לְהַרְאוֹת אוֹתָהּ שָׂדֶה שֶׁגָּזַל שֶׁאֵינָהּ שֶׁלּוֹ:  {יט} <B> שְׁטָפָהּ כוּ'.</B> שֶׁהָיְתָה עַל שְׂפַת הַנָּהָר וְהִגְבִּיהַּ שְׂפָתוֹ, וְהַמַּיִם שׁוֹחֲקִין וְכוֹרִין תַּחַת הַשָּׂפָה וּמִתְפַּשֵּׁט בְּתוֹךְ הַשָּׂדֶה. רַשִׁ"י: {כ} <B> הֲרֵי כוּ'.</B> בַּגְּמָרָא יָלֵיף מִכְּלָל וּפְרָט וּכְלָל, מַה הַפְּרָט כוּ', יָצְאוּ קַרְקָעוֹת וַעֲבָדִים וּשְׁטָרוֹת:

~ פרק י - משנה ו

{כא} <B> לֹא כוּ'.</B> אֵין זֶה יָכוֹל לְכוֹפוֹ לְקַבֵּל חוֹבוֹ אוֹ פִּקְדוֹנוֹ בַּמִּדְבָּר, דְּלָאו מְקוֹם שִׁמּוּר הוּא. רַשִׁ"י. כְּלוֹמַר אֲבָל אִם הֶחֱזִיר וְקִבְּלוֹ פָּטוּר:

{כב} נִרְאֶה לִי דְּלָאו לְמֵימְרָא שֶׁצָּרִיךְ שֶׁיִּטְעֹן אוֹתוֹ דְּמַה לִּי לֵיהּ וּלְטַעֲנָתוֹ דְּכֵיוָן דְּהַלָּה מְסֻפָּק בַּפֵּרָעוֹן אֵינוֹ יוֹצֵא יְדֵי שָׁמַיִם עַד שֶׁיֵּצֵא יְדֵי סְפֵקוֹ וְהִלְכָּךְ אֲפִלּוּ אֵינוֹ טוֹעֲנוֹ כְּלוּם. וְכֵן בַּגְּמָרָא אָמְרוּ דְּלֹא תָּבַע לֵיהּ וְלֹא אָמְרוּ דְּטָעַן שֶׁמָּא כוּ' אֶלָּא שֶׁאִם אוֹמֵר בָּרִי לִי שֶׁלֹּא הִלְוֵיתִיךָ הָיָה זֶה פָּטוּר אַף לָצֵאת יְדֵי שָׁמַיִם, מִידֵי דַּהֲוָה טְעָנוֹ חִטִּין וְהוֹדָה לוֹ בִּשְׂעֹרִין דְּפָטוּר אַף מִדְּמֵי שְׂעֹרִים. (וּמוּבָן מִמֵּילָא דְּאַף כְּשֶׁהַנִּתְבָּע יוֹדֵעַ בְּבָרִי שֶׁחַיָּב פָּטוּר כְּמוֹ בִּטְעָנוֹ חִטִּין כוּ'):  {כג} <B>פָּטוּר כוּ'.</B> עַל פִּי בֵּית דִּין. אֲבָל לָצֵאת יְדֵי שָׁמַיִם חַיָּב, כֵּיוָן דְּמִסְתַּפֵּק וְיֵשׁ כָּאן תּוֹבֵעַ בְּבָרִי:

{כד} <B> נִגְנַב.</B> נִרְאֶה לִי דְּמִשּׁוּם סֵיפָא דְּפָטוּר תָּנָא נִגְנַב, דְּלָא תֵימָא דַּוְקָא מֵת מֵעַצְמוֹ פָּטְרִינַן לֵיהּ: {כה} וְהִלְכָּךְ אֲפִלּוּ מֵת חַיָּב כְּדִתְנַן בְּמַתְנִיתִין. וּמַה שֶּׁכָּתַב הָרַ"ב שֶׁבִּגְרָמָתוֹ נֶאֶבְדָה, הוּא הַדִּין בְּמֵת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כו} <B> עֵצִים.</B> מַיְרֵי בְּשׁוֹמְרֵי פֵּרוֹת מְחֻבָּרִין בָּאִילָנוֹת: {כז} וְקָאֵי עַל נָשִׁים. וְזֶה לְשׁוֹן הַמַּגִּיד מִשְׁנֶה, הָעֲגָלִים שָׁם בְּזוֹל, וְהָאִשָּׁה מְגַדַּלְתָּן וּמוֹכַרְתָּן: {כח} <B> אָסוּר.</B> מִפְּנֵי שֶׁהֵן בְּחֶזְקַת גְּנֵבָה. הָרַמְבַּ"ם: {כט} <B> מִכָּל מָקוֹם.</B> מִכָּל אָדָם. וְאִם אָמַר הַטְמֵן, אָסוּר. הָרַמְבַּ"ם:

{ל} שֶׁהַסּוֹרֵק מוֹצִיא עַל יְדֵי הַקּוֹצִים. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לא} <B> יָתֵר כוּ'.</B> כֻּלָּן לְבַעַל הַבַּיִת. וְטַעֲמָא דְּבִקְפֵדָא תַּלְיָא מִלְּתָא, וְכִי הֲוֵי מִידֵי דְּלֹא חֲזִי לֵיהּ לֹא קָפֵיד אֲבָל בְּמִידֵי דַּחֲזִי לֵיהּ קָפֵיד. וְיָתֵר מִכָּאן חֲזִי לֵיהּ וְקָפֵיד אַכֻּלַּהּ דְּהָא הַכֹּל חֲזִי לֵיהּ, כִּדְאַשְׁכְּחַן בְּדִין הָאוֹנָאָה. טוּר: {לב} בַּגְּמָרָא מַסְּקִינַן כִּמְלֹא מַחַט וְחוּץ לְמַחַט כִּמְלֹא מַחַט, דְּהַיְנוּ אֹרֶךְ מְלֹא שְׁנֵי מְחָטִין: {לג} אוֹרְחָא דְּמִלְּתָא נָקַט כוּ'. וְהוּא הַדִּין נַמִּי אִיפְּכָא, שֶׁקַּבְּלָן עָשָׂה אֵצֶל בַּעַל הַבַּיִת וּשְׂכִיר יוֹם בְּבֵיתוֹ, דַּהֲוֵי דִּינָא אִיפְּכָא. דְּכֻלָּהּ מִלְּתָא תַּלְיָא בִּמְקוֹם הַמְּלָאכָה, אִם הִיא אֵצֶל בַּעַל הַבַּיִת קָפֵיד בַּעַל הַבַּיִת עַל הַכֹּל, לְפִי שֶׁכֵּיוָן שֶׁהַיְצִיאוֹת הֵם אֶצְלוֹ קָפֵיד עֲלֵיהֶם. מַה שֶּׁאֵין כֵּן כְּשֶׁאֵינָם אֶצְלוֹ לֹא קָפֵיד עֲלַיְהוּ, דְּלֹא טָרַח בְּהוּ לְאַיְתוּנֵיהּ גַּבֵּיהּ. וְזֶה שֶׁדִּקְדֵּק הָרַ"ב לִכְתֹּב כִּשְׂכִיר יוֹם בְּכַ"ף הַדִּמְיוֹן:
