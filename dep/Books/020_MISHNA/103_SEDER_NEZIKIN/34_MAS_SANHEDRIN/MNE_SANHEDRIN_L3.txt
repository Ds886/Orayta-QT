{א} כְּלוֹמַר דֵּאלֹהִים דִּכְתִיב גַּבֵּי שׁוֹמְרִים הָוֵי כְאִלּוּ כְתוּבִים גַּם כֵּן בְּפָרָשַׁת אִם כֶּסֶף תַּלְוֶה, דְּעֵרוּב פָּרָשִׁיּוֹת כָּתוּב כָּאן כִּדְאִיתָא בַגְּמָרָא: {ב} שֶׁזֶּה טוֹעֲנוֹ שֶׁשָּׁלַח יָד בְּפִקְדוֹנוֹ, דְּהַיְנוּ גְזֵלוֹת. אֲבָל לָוָה וְלֹא שִׁלֵּם לֹא מִקְּרֵי גַזְלָן, דְּמִלְוָה לְהוֹצָאָה נִתְּנָה. וֵאלֹהִים מַשְׁמַע לְשׁוֹן שְׂרָרָה וְרַבָּנוּת. כְּדִמְתַרְגְּמִינָן רְאֵה נְתַתִּיךָ אֱלֹהִים לְפַרְעֹה (שְׁמוֹת ז) לְשׁוֹן רַבָּנוּת. רַשִׁ"י: {ג} <B> וַחֲבָלוֹת.</B> נִרְאֶה לְרַבֵּנוּ תָם דְּכֹפֶר וּשְׁלשִׁים דְּעֶבֶד וְעֶשְׂרִים וְאַרְבָּעָה רָאשֵׁי אֵבָרִים דְּעֶבֶד, בִּכְלַל חֲבָלוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} דְּשׁוֹר הַמּוּעָד דְּאָמְרָן דְּלָאו דְּשֵׁן וָרֶגֶל אֶלָּא דְּקֶרֶן דְּלֹא מַגְבִּין בְּבָבֶל שֶׁאֵין מֻמְחִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ה} דְּאֵין צָרִיךְ לְעַבֵּר הַחֹדֶשׁ אֶלָּא כִּי מִקְדַּשׁ יוֹם שְׁלֹשִׁים וְאֶחָד מִמֵּילָא מִעַבַּר הַחֹדֶשׁ שֶׁעָבַר. דְּיוֹם שְׁלשִׁים בָּתְרֵיהּ שָׁדֵית לֵיהּ. רַשִׁ"י. אֲבָל בְּעִבּוּר שָׁנָה לֹא סַגִּי בְלֹא אֲמִירָה, דְּצָרִיךְ שֶׁיֹּאמְרוּ שֶׁתְּהֵא הַשָּׁנָה מְעֻבֶּרֶת. תּוֹסָפוֹת, וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} הַתְּקוּפָה, בֵּין עַל תְּקוּפַת תַּמּוּז שֶׁמִּתְעַכֶּבֶת וְנִמְשֶׁכֶת, שֶׁאֵין תְּקוּפַת תִּשְׁרֵי נוֹפֶלֶת עַד עֲבֹר הֶחָג, מְעַבְּרִין, דְּנָפְקָא לָן מִקְּרָא וְחַג הָאָסִיף תְּקוּפַת הַשָּׁנָה. בֵּין עַל תְּקוּפַת טֵבֵת שֶׁנִּמְשָׁךְ זְמַנָּהּ עַד טז בְּנִיסָן שֶׁהֲבָאַת עֹמֶר דִּילַהּ בְּתוֹךְ תְּקוּפַת טֵבֵת, מְעַבְּרִין. וְעַל הָאָבִיב, אִם לֹא בָשְׁלָה הַתְּבוּאָה מְעַבְּרִין, דִּקְרָא כְתִיב בְּחֹדֶשׁ הָאָבִיב. וְעַל פֵּרוֹת הָאִילָן [אִם] מִתְאַחֵר בִּשּׁוּלָם יוֹתֵר מִזְּמַן הָעֲצֶרֶת מְעַבְּרִין, שֶׁעֲצֶרֶת זְמַן הַבִּכּוּרִים כוּ'. רַשִׁ"י: {ז} וְלֹא אִתְפָּרֵשׁ טַעֲמָא מַאי. וְאֶפְשָׁר לִי לוֹמַר כְּדֵי לְעוֹרֵר הַבְּרָכוֹת לְהַשְׁלִים הַמְכֻוָּן שֶׁמְּעַבְּרִין לִהְיוֹת הָאָבִיב וְהַפֵּרוֹת וְהָאָסִיף בִּזְמַנָּן. וּמָצִינוּ כַיּוֹצֵא בָזֶה שֶׁמְּעוֹרְרִין זְכוּת הָאָבוֹת בְּהַזְכָּרַת שְׁאֵלָה עַד שֶׁבְּחֶבְרוֹן:

{ח} וְיָהִיב לֵיהּ רְשׁוּתָא [לְמֵידַן דִּינֵי קְנָסוֹת]. גְּמָרָא: {ט} שֶׁאִם לֹא תֹאמַר כֵּן אִי אֶפְשָׁר שֶׁתִּמְצָא בֵית דִּין הַגָּדוֹל לְעוֹלָם, לְפִי שֶׁנִּצְטָרֵךְ שֶׁיִּהְיֶה כָל אֶחָד מֵהֶם סָמוּךְ, וְהַקָּדוֹשׁ בָּרוּךְ הוּא יִעֵד שֶׁיָּשׁוּבוּ, כְּמוֹ שֶׁכָּתוּב וְאָשִׁיבָה שֹׁפְטַיִךְ כְּבָרִאשֹׁנָה וְגוֹ' כוּ'. עַד כָּאן לְשׁוֹן הָרַמְבַּ"ם בְּפֵרוּשׁוֹ. וּבְחִבּוּרוֹ סִיֵּם, וְהַדָּבָר צָרִיךְ הֶכְרֵעַ. וְנִרְאֶה לִי, לְפִי שֶׁאֶפְשָׁר לִדְחוֹת הָרְאָיָה, שֶׁהֲרֵי יִעֵד הַשֵּׁם יִתְבָּרַךְ גַּם כֵּן הִנֵּה אָנֹכִי שֹׁלֵחַ לָכֶם אֵת אֵלִיָּה הַנָּבִיא כוּ', וְהוּא סָמוּךְ בְּלִי סָפֵק וְיָכוֹל לִסְמֹךְ בְּצֵרוּף שְׁנַיִם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} <B> וַעֲרִיפַת.</B> עֲרִיפָה לָאו דַּוְקָא, דִּמְדִידָה הוּא דְהַוְיָא בִשְׁלֹשָׁה. תּוֹסָפוֹת: {יא} תֵּימַהּ, שֶׁזֶּה הַכָּתוּב בִּסְמִיכַת זְקֵנִים לְפַר הֶעְלֵם דָּבָר כָּתוּב, אֲבָל בְּעֶגְלָה נָפְקָא לָן מִדִּכְתִיב זְקֵנֶיךָ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} <B> וְהַמֵּאוּנִין בִּשְׁלשָׁה.</B> דְּכָל דְּתַקּוּן רַבָּנָן כְּעֵין דְּאוֹרַיְתָא תַקּוּן. וּמֵאוּנִין דְּרַבָּנָן, דְּמִדְּאוֹרַיְתָא אֲפִלּוּ מֵאוּן לֹא בַעְיָא, דְּאֵין קִדּוּשֵׁי קְטַנָּה כְלוּם אֶלָּא אִם כֵּן קִדְּשָׁהּ אָבִיהָ. דִּכְתִיב אֶת בִּתִּי נָתַתִּי לָאִישׁ. רַשִׁ"י: {יג} <B> בִּשְׁלשָׁה.</B> דְּמָמוֹנָא הוּא: {יד} סְבָרָא הוּא, מַה לִּי עַיּוֹלֵי מַה לִּי אַפּוֹקֵי, אַפּוֹקֵי מַאי טַעֲמָא, דִּלְמָא טָעֵי, עַיּוֹלֵי נַמִּי דִּלְמָא טָעֵי. גְּמָרָא: {טו} וְהוּא הַדִּין נַמִּי דְּפָלֵיג אַהֶקְדֵּשׁוֹת, דִּבְהוּ נַמִּי כֹּהֵן כְּתִיב: {טז} וּבְפֶרֶק ד' דִּמְגִלָּה מִשְׁנָה ג' פֵּרֵשׁ הָרַ"ב, אִם בָּא לִפְדּוֹתָן. וְכֵן פֵּרֵשׁ רַשִׁ"י. וְהָכִי מִסְתַּבְּרָא: 

~ פרק א - משנה ד

{יז} <B> תַּהֲרֹגוּ.</B> בְּנִרְבָּע כְּתִיב. רַשִׁ"י. וּכְלוֹמַר דְּאִתַּקַּשׁ לְאִישׁ: {יח} דְּלֵיכָּא לְמֵימַר דִּקְרָא כִפְשָׁטֵיהּ לְחַיֵּב מִיתָה לִבְעָלָיו, דְּהָא אָמַר קְרָא מוֹת יוּמַת הַמַּכֶּה רֹצֵחַ הוּא, עַל רְצִיחָתוֹ אַתָּה הוֹרְגוֹ וְאִי אַתָּה הוֹרְגוֹ עַל רְצִיחַת שׁוֹרוֹ. גְּמָרָא: {יט} <B> הַזְּאֵב כוּ'.</B> שֶׁהֵמִיתוּ אָדָם וְנִדּוֹנִין בִּסְקִילָה כְּשׁוֹר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} <B> זָכָה.</B> לַשָּׁמַיִם. (צְדָקָה עָשָׂה. רַשִׁ"י). דְּלֵיכָּא לְמֵימַר לְעוֹרָן, דְּכֵיוָן שֶׁהֵמִיתוּ אָדָם שַׁוִּינְהוּ רַבָּנָן (דְּלֹא הִצְרִיכוּ לְדוּנָן בְּבֵית דִּין) כְּמַאן דְּנִגְמַר דִּינַיְהוּ, וְאִסּוּרֵי הֲנָאָה נִינְהוּ. גְּמָרָא: 

~ פרק א - משנה ה

{כא} פֵּרוּשׁ, עֲלִיָּתוֹ הוּא דַּהֲוֵי בְשִׁבְעִים וְאֶחָד, אֲבָל מִיתָתוֹ בְּעֶשְׂרִים וּשְׁלשָׁה כִּשְׁאָר דִּינֵי נְפָשׁוֹת, כִּדְאִיתָא בַגְּמָרָא. וּפֵרֵשׁ רַשִׁ"י דְּהָא לֹא תָנֵי לֵיהּ בְּמַתְנִיתִין. וּמִשּׁוּם דִּגְמָרוֹ לֹא הָיָה בְשִׁבְעִים וְאֶחָד, לְהָכִי לֹא תָנֵי הַמְרָאָתוֹ וְכֵן מִיתָתוֹ, וְאַף עַל גַּב דְּדַוְקָא בְּשִׁבְעִים וְאֶחָד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} שֶׁהוּא הָיָה דָן קֹדֶם מִנּוּי הַסַּנְהֶדְרִין כָּל מַה שֶּׁהָיוּ הַסַּנְהֶדְרִין דָּנִין אַחַר כָּךְ, וְלֹא נִתְמַנּוּ הַסַּנְהֶדְרִין אֶלָּא לְהָקֵל מֵעָלָיו. וְרַשִׁ"י פֵרֵשׁ שֶׁהָיָה שׁוֹפֵט וְדָן עַל פִּי הַדִּבּוּר וְשָׁקוּל כוּ': {כג} וּבְסַנְהֶדְרִין כָּתַב הָרַ"ב כִּדְאַשְׁכְּחַן בְּמשֶׁה, וְאַף עַל גַּב דְּלֹא כְתִיב וְכֵן תַּעֲשׂוּ, מִשּׁוּם דְּהָכָא מְרַבִּינַן נַמִּי מֶלֶךְ וְנָבִיא כִּדְאִיתָא פֶּרֶק ב' דִּשְׁבוּעוֹת. וְלֹא תָנֵי לְהוּ הָכָא, דְּלֹא חָשֵׁיב אֶלָּא מִלֵּי דְסַנְהֶדְרִין. תּוֹסָפוֹת. וְכֵן צָרִיךְ לוֹמַר בְּמִלְחֶמֶת הָרְשׁוּת דְּלֹא תָנֵי הָכָא מֶלֶךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כד} <B> סַנְהֶדְרִיּוֹת לַשְּׁבָטִים.</B> שֶׁנֶּאֱמַר [דְּבָרִים טז] שֹׁפְטִים וְשֹׁטְרִים תִּתֶּן לְךָ בְּכָל שְׁעָרֶיךָ לִשְׁבָטֶיךָ: {כה} וַאֲפִלּוּ הָכִי אִצְטְרִיךְ לְעֵיל לִמְעוּטֵי שֵׁבֶט. כְּגוֹן אִם יֵשׁ בְּעִיר אַחַת מִשְּׁנֵי שְׁבָטִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כו} הָרַמְבַּ"ם. וְכָתַב הַכֶּסֶף מִשְׁנֶה דְּאַף עַל גַּב דְּלֹא קַיְמָא לָן כְּרַבִּי שִׁמְעוֹן דְּדָרֵישׁ טַעֲמָא דִקְרָא, רָאָה רַבֵּנוּ לְכָתְבוֹ, שֶׁזֶּה טַעַם מִתְקָרֵב אֶל הַשֵּׂכֶל וְאֵין בּוֹ נַפְקוּתָא לְדִינָא: {כז} <B> וְלֹא שָׁלֹשׁ.</B> דִּכְתִיב עָרֶיךָ, הֲרֵי וְרַבִּי מֵאִיר יָהֵיב לְהוּ דִין יוֹשְׁבֵי קְרָנוֹת, וְיָכוֹל לְפָסְלָן. וְרַבָּנָן סָבְרֵי כֵּיוָן שֶׁהִמְחוּם רַבִּים עֲלֵיהֶם לֹא כֹל הֵימֶנּוּ לְפָסְלָן, וַעֲדִיפֵי מִיּוֹשְׁבֵי קְרָנוֹת. אַחֲרוֹנִים: {ה} זֶה הוּא לְשׁוֹן רַשִׁ"י בְּאוֹקִימְתָא קַמַּיְתָא. אֲבָל בַּמַּסְקָנָא כָתַב לֹא אָדוּן לְפָנָיו: {ו} וַאֲפִלּוּ נָשָׂא וְנָתַן בַּיָּד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} וְקַיְמָא לָן כְּמַאן דְּדָאֵין דִּינָא דְגַרְמֵי: {ח} וְלֹא יָדַע שֶׁכְּבָר פָּשַׁט הַמַּעֲשֶׂה בְּכָל הָעוֹלָם כְּדִבְרֵי הָאֶחָד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} וְחוֹזֵר וְנוֹטֵל כוּ' בְּמִכָּל שֶׁכֵּן מֵהָא דִּלְקַמָּן: {י} נָפְקָא מִנֵּיהּ לְעִנְיַן זְמָן. אִי נַמִּי זִמְנִין דְּלֵיכָּא לְבָרוֹרֵי טָעוּתָא לֹא מֵהַמִּשְׁנָה וְלֹא מֵהַסּוּגְיָא דְעָלְמָא, וַאֲפִלּוּ הָכִי אֶפְשָׁר שֶׁאֵינוֹ דִין צֶדֶק וִיכוֹלִים הַדַּיָּנִים לִפְסֹק דְּלֹא כְמוֹ שֶׁפָּסַק זֶה הַדַּיָּן, דְּאַטּוּ כֻלְּהוּ דִינֵי נִשְׁנוּ בַמִּשְׁנָה וְסוּגְיָא דְעָלְמָא: {יא} שֶׁזֶּה מִתְכַּוֵּן לְהַזִּיק הוּא כֵּיוָן דְּלָאו דַּיָּנָא הוּא: {יב} דְּהָכָא בְּבָבֶל שֵׁבֶט, לְשׁוֹן שְׂרָרָה, וְיֵשׁ לָהֶם רְשׁוּת [לְהַפְקִיר, דְּהֶפְקֵר בֵּית דִּין הֶפְקֵר]. וְהָתָם בְּאֶרֶץ יִשְׂרָאֵל מְחוֹקֵק, שְׂרָרָה מֻעֶטֶת. כִּדְאִיתָא שֶׁסָּבוּר הָיָה שֶׁיֵּשׁ לוֹ שְׁתֵּי כִתֵּי עֵדִים וְאָמַר כֵּן וְאַחַר כָּךְ בִּקֵּשׁ וְלֹא מָצָא יַפְסִיד. וְהָרַמְבַּ"ם מְפָרֵשׁ דִּפְלוּגְתַיְהוּ כְּשֶׁהֵבִיא שְׁתֵּי כִתֵּי עֵדִים שֶׁאַחַר שֶׁהֵעִידוּ כַת אַחַת כִּדְבָרָיו הֵבִיא עוֹד אַחֶרֶת וְהֵעִידוּ גַּם כֵּן כְּמוֹ הָרִאשׁוֹנִים, וְטָעַן הַלָּה שֶׁאַרְבָּעָה הָעֵדִים פְּסוּלִים הֵם, רַבִּי מֵאִיר אוֹמֵר שׁוֹמְעִים לוֹ, וְנֹאמַר לְזֶה שֶׁהֱבִיאָם הָבֵא רְאָיָה שֶׁהֵם כְּשֵׁרִים, לְפִי שֶׁאַתָּה לֹא הָיִיתָ מֵבִיא עֵדִים אַחַר עֵדִים אֶלָּא שֶׁאַתָּה יוֹדֵעַ שֶׁיֵּשׁ בָּהֶן מִי שֶׁאֵינוֹ כָשֵׁר, וַאֲנַחְנוּ לֹא נֵדַע אֵיזֶהוּ {טו} וְכָל שֶׁכֵּן לְהָעִיד. עַיֵּן פֶּרֶק ו' דְּנִדָּה מִשְׁנָה ד': {טז} <B> רוֹעֵי בָקָר.</B> הוּא הַדִּין כָּל בֵּי תְלָתָא דְּלֹא גְמִירֵי דִינָא פְּסוּלִים. וְלִרְבוּתָא נָקַט רוֹעֵי בָקָר, דְּלֹא שְׁכִיחֵי בְיִשּׁוּב וְלֹא רָאוּ וְלֹא שָׁמְעוּ עִסְקֵי דִין בֵּין אָדָם לַחֲבֵרוֹ, וַאֲפִלּוּ לְעֵדוּת נַמִּי פַסְלִינְהוּ רַבָּנָן, דְּקָא חָזוּ רַבָּנָן דְּקָא מְכַוְּנֵי וְשָׁרֵי לְהוּ לִרְעוֹת בִּשְׂדוֹת אֲחֵרִים. גְּמָרָא: {יז} וְכֵן דַּעַת נִמּוּקֵי יוֹסֵף דִּלְגַבֵּי עֵדוּת נַמִּי הָוֵי גְמַר דִּין כְּשֶׁפָּסְקוּ בֵית דִּין הַדָּבָר עַל פִּיהֶן. אֲבָל הָרִיבָ"שׁ וּגְאוֹנִים סְבִירָא לְהוּ דִּלְגַבֵּי עֵדוּת הָוֵי גְמַר דִּין כָּל שֶׁנִּתְקַבֵּל עֵדוּתָן בְּבֵית דִּין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יח} וּבְרֵישָׁא נַמִּי בְּאֶתֵּן לְךָ מַחֲלֹקֶת, אֲבָל בְּמָחוּל לְךָ אֲפִלּוּ רַבִּי מֵאִיר מוֹדֶה. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יט} <B> לָאו דַּוְקָא.</B> דְּאִם כֵּן פְּשִׁיטָא, דְּהָא בִלְאַחַר גְּמַר דִּין מַיְרֵי וּמְחִילָה גְמוּרָה הִיא. אֶלָּא מַיְרֵי כְּגוֹן דְּאָמַר לֵיהּ תּוֹבֵעַ לַנִּתְבָּע הֲרֵי עָלַי לַעֲשׂוֹת מִתְּבִיעָה זוֹ כְּכָל מַה שֶּׁיֹּאמְרוּ אַבָּא אוֹ אָבִיךְ הֵן לִפְטוֹר הֵן לְחִיּוּב. שֶׁאֵין בַּלָּשׁוֹן הַזֶּה לְשׁוֹן מְחִילָה. נִמּוּקֵי יוֹסֵף:

{כ} <B> הַפְּסוּלִין.</B> פְּסוּלֵי דְרַבָּנָן קָא חָשֵׁיב. וּמִשּׁוּם הָכִי לֹא גָרְסִינַן וַעֲבָדִים, דְּאִינְהוּ פְסוּלִים מִדְּאוֹרַיְתָא מִקַּל וָחֹמֶר. תּוֹסָפוֹת: {כא} וְהָא דִתְנַן מַלְוֶה בְרִבִּית. מִלְוָה הַבָּאָה בְרִבִּית. גְּמָרָא: {כב} וְאַף עַל גַּב דְּעָרֵב וְעֵדִים עוֹבְרִים נַמִּי בְלֹא תְשִׂימוּן, לֹא תְשִׂימוּן לֶאֱנָשֵׁי, לְמַלְוֶה וְלֹוֶה מַשְׁמַע לְהוּ. הָרֹא"שׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} וְאַף עַל גַּב דְּהַיְנוּ מְשַׂחֵק בְּקֻבְיָא, תָּנָא תוֹלֶה בְדַעַת יוֹנוֹ וְתָנָא תוֹלֶה בְדַעַת עַצְמוֹ. וּצְרִיכֵי כוּ'. בַּגְּמָרָא: {כד} דְּלֹא זָכָה בַעַל הַשּׁוֹבָךְ, דְּמִמֵּילָא קָאָתוּ וְרָבוּ לְהָתָם. רַשִׁ"י: {כה} וְכָל שֶׁכֵּן לָדוּן: {כו} בְּרִבִּית קְצוּצָה. וּמַתְנִיתִין בַּאֲבַק רִבִּית. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כז} דְּלֹא תַחְמֹד לֶאֱנָשֵׁי בְּלֹא דְמֵי מַשְׁמַע לְהוּ: {כח} וּמַיְרֵי שֶׁהֶחֱזִירוּ, דְּאִם לֹא כֵן פְּסוּלִים מִדְּאוֹרַיְתָא, אֶלָּא מִשּׁוּם דְּלֹא יָדְעֵי לְמַאן נְהַדְרֵי, אִי נַמִּי שֶׁמָּא יַחְזְרוּ לְקִלְקוּלָם. תּוֹסָפוֹת: {כט} וְאֶפְשָׁר לֵיהּ לְאִתְזוֹנֵי בְצִנְעָא וּמְבַזֵּי נַפְשֵׁהּ בְּפַרְהֶסְיָא. דְּאִי לֹא אֶפְשָׁר לֵיהּ, חַיּוּתֵיהּ הוּא. גְּמָרָא: {ל} <B> אָמַר רַבִּי יְהוּדָה.</B> אַמְּשַׂחֵק בְּקֻבְיָא קָאֵי. אִי נַמִּי אַמַּפְרִיחֵי יוֹנִים לְפֵרוּשׁ הָרִאשׁוֹן: {לא} <B> כְּשֵׁרִין.</B> וְדַוְקָא כְשֶׁהַמָּעוֹת עַל הַדַּף. אֲבָל הַמְשַׂחֲקִים בַּאֲמָנָה, אֲפִלּוּ הִקְנוּ, אַסְמַכְתָּא גְמוּרָה הִיא וְאֵין קִנְיָן מוֹעִיל בָּהּ. תּוֹסָפוֹת: {לב} שִׁבְרֵי עֵצִים. {יט} <B> סַנְהֶדְרִין.</B> אַף שֶׁל עֶשְׂרִים וּשְׁלשָׁה. רַשִׁ"י: {כ} לְפִי שֶׁאִם הָיוּ יוֹשְׁבִין בְּשׁוּרָה אֵין פְּנֵי הָרִאשׁוֹנִים רוֹאִים זֶה אֶת זֶה. רַשִׁ"י: {כא} שֶׁבֵּית הַמִּקְדָּשׁ בְּאֶמְצַע שֶׁל עוֹלָם. רַשִׁ"י. וְשָׁרְרֵךְ הוּא הַטַּבּוּר. וּכְמוֹ שֶׁהַטַּבּוּר אֵינוֹ מַמָּשׁ בָּאֶמְצַע כֵּן גַּם בֵּית הַמִּקְדָּשׁ אֵינוֹ מַמָּשׁ בָּאֶמְצַע, אֲבָל עַל צַד הַקֵּרוּב הוּא: {כב} <B> הַמְזַכִּין כוּ'.</B> פְּלוֹנִי חִיֵּב וּמִן הַטַּעַם הַזֶּה, וּפְלוֹנִי זִכָּה וּמִן הַטַּעַם הַזֶּה. רַשִׁ"י. וְטַעֲמָא, כְּדֵי שֶׁאִם יֹאמְרוּ שְׁנַיִם טַעַם אֶחָד מִשְּׁנֵי מִקְרָאוֹת לֹא יִמְנוּ אֶלָּא אֶחָד (כִּדְיָלְפִינַן לֵיהּ מִקְּרָא). אִי נַמִּי נָפְקָא מִנַּהּ לַמְחַיְּבִין, דְּאִי לְמָחָר יַחֲזֹר מִטַּעְמוֹ וִיחַיְּבוֹ מִטַּעַם אַחֵר, לִבָּעֵי עוֹד הֲלָנַת דִּין. גְּמָרָא: {כג} הָרַמְבַּ"ם. וּתְמֵהַנִי דְּשַׁפִּיר קָאָמַר רַבִּי יְהוּדָה. וְרַשִׁ"י פֵרֵשׁ בִּדְתַנָּא קַמָּא שֶׁשְּׁנֵיהֶם כּוֹתְבִין דִּבְרֵי הַמְזַכִּין וְגַם דִּבְרֵי הַמְחַיְּבִין. וְרַבִּי יְהוּדָה חָיֵשׁ שֶׁמָּא מֵרֹב הַטֹּרַח יִטְעוּ:

~ פרק ה - משנה א

{א} הָיוּ בוֹדְקִין. לָאו בְּפַעַם אַחַת וְזֶה בִּפְנֵי זֶה: {ב} <B> חֲקִירוֹת.</B> נָקַט חֲדָא מִתַּרְתֵּי דִקְרָא. אַף עַל גַּב דִּבְרֵישׁ פֶּרֶק דִּלְעֵיל נָקַט תַּרְתֵּי, הָכָא סַגִּי לֵיהּ בַּחֲדָא. וְנָקַט חֲקִירָה אַף עַל גַּב דִּכְתִיב אַחַר דְּרִישָׁה, לְפִי שֶׁהַחֲקִירָה הוּא מוֹרֶה יוֹתֵר עַל הָעֲמִידָה עַל אֲמִתַּת הָעִנְיָן מִדְּרִישָׁה, כְּמָה דְאַתְּ אָמַר [אִיּוֹב כה] וּלְכָל תַּכְלִית הוּא חוֹקֵר: {ג} בְּעִיר הַנִּדַּחַת: {ד} דִּוְשָׁאַלְתָּ לֹא מַשְׁמַע חֲקִירָה אֶלָּא עַד דְּאִיכָּא הֵיטֵב בַּהֲדֵיהּ. אֲבָל וְדָרַשְׁתָּ בְּלֹא הֵיטֵב מַשְׁמַע דְּרִישָׁה מְעַלְּיָא. רַשִׁ"י: {ה} בְּיָחִיד הָעוֹבֵד עֲבוֹדָה זָרָה: {ו} בְּעֵדִים זוֹמְמִין. וּמַסְּקִינַן בַּגְּמָרָא דְּכֻלְּהוּ יָלְפִינַן מֵהֲדָדֵי לִתֵּן אֶת הָאָמוּר שֶׁל זֶה בָּזֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} אַף עַל פִּי שֶׁאָמְרוּ בְּכָךְ וְכָךְ בַּחֹדֶשׁ, אַף עַל פִּי כֵן שׁוֹאֲלִין לָהֶם בְּאֵיזֶה יוֹם בְּשַׁבָּת הָיָה, לְפִי שֶׁכָּל הַשֶּׁבַע הַלָּלוּ בָּאוֹת לַהֲבִיאָן לִידֵי הֲזָמָּה, וְאֵין מַעֲלִין עַל לִבָּן חֶשְׁבּוֹן שֶׁל חֹדֶשׁ אִם הָיוּ אֵלּוּ עִמָּהֶן אוֹתוֹ יוֹם בְּמָקוֹם אַחֵר, וּכְשֶׁשּׁוֹאֲלִין אוֹתָן בְּאֵיזֶה יוֹם וְהֵם אוֹמְרִים בְּיוֹם פְּלוֹנִי, נִזְכָּרִים עֵדֵי הֲזָמָּה לַהֲזִמָּם. רַשִׁ"י: {ח} <B> מָקוֹם.</B> שֶׁזּוֹ הִיא הַמּוֹעֶלֶת לְהָבִיא אֶת כָּל הַחֲקִירוֹת לַהֲזָמָּה, דִּבְלָאו הָכִי לֹא מִתַּזְמֵי. רַשִׁ"י: {ט} <B> רַבִּי יוֹסֵי כוּ'.</B> בַּגְּמָרָא מוּכָח דְּרַבִּי יוֹסֵי לֹא פָלֵיג אֶלָּא כְשֶׁמִּתּוֹךְ עֵדוּתָן נִכָּר שֶׁאוֹתוֹ הַיּוֹם שֶׁמְּעִידִין עָלָיו בְּכַמָּה בַחֹדֶשׁ וּבְאֵיזֶה חֹדֶשׁ וְשָׁנָה וְשָׁבוּעַ, דְּאִי לָאו הָכִי אִי אַתָּה יָכוֹל לַהֲזִמָּם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} <B> הָעוֹבֵד כוּ'.</B> אִם מְעִידִין בְּכָךְ, שׁוֹאֲלִין אֶת מִי עָבַד. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ה - משנה ב

{יא} בִּפְנֵי הִלֵּל רַבּוֹ. וְרַבֵּנוּ הַקָּדוֹשׁ כָּתַב אוֹתוֹ בִּשְׁעַת חִבּוּרוֹ כְּמוֹ שֶׁקִּבֵּל אוֹתוֹ. הָרַמְבַּ"ם: {יב} דְּרַחֲמָנָא אָמַר [דְּבָרִים יט] וְדָרְשׁוּ הַשֹּׁפְטִים הֵיטֵב וְהִנֵּה עֵד שֶׁקֶר הָעֵד וְגוֹ' וַעֲשִׂיתֶם לוֹ כַּאֲשֶׁר זָמַם לַעֲשׂוֹת וְגוֹ', עֵדוּת שֶׁאַתָּה יָכוֹל לְקַיֵּם בָּהּ תּוֹרַת הֲזָמָּה הָוֵי עֵדוּת, וְאִי לֹא לֹא. רַשִׁ"י: {יג} <B> וַאֲפִלּוּ שְׁנַיִם.</B> וּבַגְּמָרָא, מַאי אֲפִלּוּ, פְּשִׁיטָא דְּכֵיוָן דְּאֶחָד אוֹמֵר אֵינִי יוֹדֵעַ הַשֵּׁנִי לֹא מַעֲלֶה בִידִיעָתוֹ. וּמַסִּיק דְּהָכִי קָאָמַר, אֲפִלּוּ בַחֲקִירוֹת שְׁנַיִם אוֹמְרִים יָדַעְנוּ וְאֶחָד אוֹמֵר אֵינִי יוֹדֵעַ עֵדוּתָן קַיֶּמֶת, פֵּרוּשׁ אִם יֵשׁ שְׁלשָׁה עֵדִים. וְעַד שֶׁיִּזּוֹמוּ כֻלָּן יֵשׁ לְפָרֵשׁ דְּהַיְנוּ כְּשֶׁכִּוְּנוּ עֵדוּתָן. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} <B> שֶׁמַּכְחִישִׁין.</B> בֵּין שֶׁהֵם שְׁנַיִם, אוֹ שְׁלשָׁה וְהַשְּׁלִישִׁי מַכְחִישׁ. דְּהַכְחָשָׁה עֲדִיפָא מֵאֵינִי יוֹדֵעַ. וְכֵן כָּתַב הָרַמְבַּ"ם: 

~ פרק ה - משנה ג

{טו} <B> אֶחָד אוֹמֵר כוּ'.</B> בָּרְבִיעִי בְשַׁבָּת בִּשְׁנַיִם לַחֹדֶשׁ, וְהַשֵּׁנִי אוֹמֵר בָּרְבִיעִי בְשַׁבָּת בִּשְׁלֹשָׁה לַחֹדֶשׁ: {טז} <B> בְּעִבּוּרוֹ כוּ'.</B> וְהוּא הַדִּין דַּהֲוָה מָצֵי לְמֵימַר זֶה יָדַע בְּחֶסְרוֹנוֹ דְּחֹדֶשׁ שֶׁעָבַר חָסֵר הָיָה. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} <B> בְּטֵלָה.</B> דְּבִתְרֵי עִבּוּרֵי לֹא טָעוּ אֱנָשֵׁי. גְּמָרָא. וַהֲוָה מָצֵי לְמֵימַר אֶחָד אוֹמֵר בְּב' וְאֶחָד בְּד', אֶלָּא דְמַתְחִיל בְּמַאי דְּסִיֵּם: {יח} <B> קַיֶּמֶת.</B> דְּאָדָם טוֹעֶה שְׁתֵּי שָׁעוֹת חָסֵר מַשֶּׁהוּ, דְּכִי נַמִּי אָמַרְתְּ זֶה שֶׁאָמַר שְׁתַּיִם בִּתְחִלַּת שְׁתַּיִם וְזֶה שֶׁאָמַר שָׁלֹשׁ בְּסוֹף שָׁלֹשׁ כָּל רְגָעִים שֶׁבֵּינְתַיִם שֶׁאֵרַע הַמַּעֲשֶׂה בָהֶן יֵשׁ לְאֵלּוּ לִטְעוֹת וְלוֹמַר כְּמוֹ שֶׁאָמְרוּ. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יט} <B> בְּטֵלָה.</B> וְאִם תֹּאמַר וְנִשְׁאָלֵם דִּלְמָא הַאי דְּקָאָמַר שָׁלֹשׁ סוֹף שָׁלֹשׁ, (אִי נַמִּי) הַאי דְּקָאָמַר חָמֵשׁ תְּחִלַּת חָמֵשׁ. וְיֵשׁ לוֹמַר דִּלְבַטֵּל עֵדוּת בָּדְקִינַן לְהוּ מִשּׁוּם דִּכְתִיב וְהִצִּילוּ הָעֵדָה, וְלֹא לְקַיֵּם עֵדוּתָן. תּוֹסָפוֹת: {כ} <B> שֶׁבְּחָמֵשׁ כוּ'.</B> וְאֵין לְהַקְשׁוֹת דִּבְמִשְׁנָה ד' פֶּרֶק קַמָּא דִפְסָחִים גָּזַר רַבִּי יְהוּדָה מִשּׁוּם יוֹם הַמְעֻנָּן דְּטָעֵי עַל הַשְּׁבִיעִית לוֹמַר חֲמִישִׁית וּפָסַק הָרַ"ב כֵּן הֲלָכָה. דְּהָתָם לְחֻמְרָא וְהָכָא לְחֻמְרָא, דְּלֹא גָזְרִינַן וְלֹא קָטְלִינַן מִסְּפֵקָא: 

~ פרק ה - משנה ד

{כא} וְאַף עַל גַּב דִּקְרָא מְסַיֵּם לָמוּת כוּ', מְפָרֵשׁ בַּגְּמָרָא דִּבְעֵד לֵיכָּא לְמֵימַר הָכִי מִשּׁוּם דְּמֵחֲזֵי כְּנוֹגֵעַ בְּעֵדוּתוֹ, פֵּרֵשׁ רַשִׁ"י שֶׁמִּתְחָרֵט בּוֹ וְדוֹאֵג שֶׁמָּא יִגָּמֵר הַדִּין לְחוֹבָה וְיָבִיא זֶה עֵדִים לַהֲזִמָּן. וּמִהְדַּר לֹא מָצֵי הָדְרֵי בְהוּ לְאַחַר כְּדֵי דִבּוּר, דְּכֵיוָן שֶׁהִגִּיד כוּ': {כב} גְּמָרָא. כְּדֵי שֶׁלֹּא תְהֵא עֲלִיָּתוֹ יְרִידָה לוֹ. פֵּרֵשׁ רַשִׁ"י שֶׁאִם יוֹרִידוּהוּ הַיּוֹם לְעֵין כֹּל בּוּשָׁה הוּא לוֹ: {כג} גְּמָרָא. וְנִרְאֶה וַדַּאי דְּאֵינוֹ נִמְנֶה עִמָּהֶן, שֶׁאִם כֵּן יִהְיֶה בֵית דִּין זוּגוֹת. אֶלָּא זוֹכֶה לִהְיוֹת בְּמוֹשָׁב בֵּית דִּין וְעַל הַסַּפְסָל כְּמוֹתָם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כג} <B> בְּשֵׁם עֲבוֹדָה זָרָה.</B> שֶׁנֶּאֱמַר וַאֲשֶׁר יְדַבֵּר בְּשֵׁם אֱלֹהִים אֲחֵרִים. גְּמָרָא: {כד} <B> מַקְדִּימִין.</B> כְּלוֹמַר אֵין לָהֶם לְצַפּוֹת מִיתָה אַחֶרֶת, אֶלָּא מַשְׁכִּימִין לְאוֹתָהּ שֶׁנִּגְמַר לָהּ דִּינוֹ שֶׁל נִדּוֹן. רַשִׁ"י. וְהַתּוֹסָפוֹת פֵּרְשׁוּ דִּלְהָכִי קָאָמַר מַקְדִּימִין, כְּלוֹמַר וַדַּאי אַתָּה צָרִיךְ לַהֲמִיתָם בַּמִּיתָה שֶׁרָצוּ לַהֲמִיתוֹ, אֲבָל וַדַּאי אִם אֵינְךָ יָכוֹל, אֲפִלּוּ הָכִי נַמִּי נְמִיתֵם בְּכָל מִיתָה שֶׁנּוּכַל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כה} <B> וּבוֹעֲלָהּ.</B> נִרְאֶה דְּהָכִי קָאָמַר, חוּץ מִשֶּׁכְּשֶׁהוּזַמּוּ לְבַת כֹּהֵן עִם בּוֹעֲלָהּ, שֶׁגַּם בּוֹעֲלָהּ נִדּוֹן, דְּמִתְקַיֵּם בְּהוּ כַּאֲשֶׁר זָמַם לַעֲשׂוֹת לְאָחִיו. אֲבָל כְּשֶׁלֹּא זָמְמוּ לְאָחִיו, כְּגוֹן שֶׁאֵין מַכִּירִין אוֹתוֹ אוֹ לֹא הִתְרוּ בוֹ כוּ', דְּהַשְׁתָּא אֵין לָנוּ לְהַקִּישָׁם לוֹ, דְּאִין הָכִי נַמִּי דְּמַקְדִּימִין אוֹתָם לַדִּין שֶׁלָּהּ. וְזֶה נִרְאֶה בִּלְשׁוֹן רַשִׁ"י וְהָרַמְבַּ"ם. אֶלָּא שֶׁהַתּוֹסָפוֹת הוֹכִיחוּ דְּאַף עַל פִּי שֶׁאֵין דִּין בַּבּוֹעֵל, נִדּוֹנִים בְּחֶנֶק. וּלְדִבְרֵיהֶם צָרִיךְ לוֹמַר דְּתַנָּא מִסְּרַךְ סְרִיךְ בְּלִישְׁנָא דְרֵישׁ פִּרְקִין, וְנָקַט וּבוֹעֲלָהּ אַף עַל גַּב דְּלֹא אִצְטְרִיךְ לֵיהּ הָכָא:
