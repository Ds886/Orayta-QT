~ פרק א - משנה א

{א} <B> שְׁלֹשָׁה עָשָׂר.</B> וּמִנְיָנָא לְמַעוֹטֵי רָאשֵׁי אֲגַפַּיִם וְרֹאשׁ הַזָּנָב. וְהַיְנוּ לְרַבִּי מֵאִיר, דִּלְרַבִּי יוֹסֵי אִיתַנְהוּ חֲמִשָּׁה עָשָׂר. וּלְרַבִּי יְהוּדָה לֵיתַנְהוּ אֶלָּא שְׁנֵים עָשָׂר: {ב} הַיְנוּ לְטֻמְאַת אֳכָלִין, אֲבָל לְטַמֵּא בְּבֵית הַבְּלִיעָה אֵין צָרִיךְ מַחֲשָׁבָה, שֶׁלֹּא מָצִינוּ שׁוּם טֻמְאָה חֲמוּרָה שֶׁצְּרִיכָה הֶכְשֵׁר, וּלְפִיכָךְ דִּין הוּא שֶׁלֹּא תְּהֵא צְרִיכָה מַחֲשָׁבָה, אֲבָל טֻמְאָה קַלָּה וְהִיא טֻמְאַת אֳכָלִין וּמַשְׁקִים, אַף עַל פִּי שֶׁזּוֹ אֵינָהּ צְרִיכָה הֶכְשֵׁר, הֲרֵי כָּל אֳכָלִין צְרִיכִין הֶכְשֵׁר וּצְרִיכִין מַחֲשָׁבָה, הִלְכָּךְ אַף זוֹ לְטַמֵּא טֻמְאָה קַלָּה אַף עַל פִּי שֶׁאֵינָהּ צְרִיכָה הֶכְשֵׁר צְרִיכָה מַחֲשָׁבָה. גְּמָרָא: {ג} <B> וּמְטַמְּאָה כוּ'.</B> צָרִיךְ עִיּוּן, דְּלֹא תְּנַן נַמִּי וְכַחֲצִי פְּרָס לִפְסֹל הַגְּוִיָּה, כְּדִלְקַמָּן מִשְׁנָה ג'. וּבְפֶרֶק ד' דִּמְעִילָה מִשְׁנָה ה' תְּנַן כָּל הָאֳכָלִין מִצְטָרְפִין לִפְסֹל אֶת הַגְּוִיָּה: {ד} וְכֵן לְהִצְטָרֵף לְפָחוֹת מִכַּבֵּיצָה אֳכָלִים לְקַבֵּל טֻמְאָה, דְּצָרִיךְ שֶׁיִּהְיוּ כַּבֵּיצָה לְשִׁיטַת הָרַ"ב: {ה} דַּהֲוֵי לֵיהּ לְמִכְתַּב וְהָאִישׁ, אֶלָּא לוֹמַר שֶׁבְּבֵית הַנֶּפֶשׁ מְטַמְּאָה. תּוֹרַת כֹּהֲנִים. וְעַיֵּן בְּהָרַ"ב פֶּרֶק ה' דְּזָבִים מִשְׁנָה ט': {ו} דִּכְתִיב וְטָמֵא עַד הָעֶרֶב וְטָהֵר: {ז} בִּשְׁבוּעוֹת דַּף ז'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} צָרִיךְ עִיּוּן, דְּבַגְּמָרָא מַסִּיק דְּצָרִיךְ כַּזַּיִת אֶלָּא דְּגִידִים וַעֲצָמוֹת מִצְטָרְפִים לְמַשֶּׁהוּ שֶׁיֵּשׁ בּוֹ, דְּוַדַּאי דְּאֵין אֲכִילָה פָּחוֹת מִכַּזַּיִת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} וְאַף עַל גַּב דִּבְחֻלִּין פֶּרֶק ד' מִשְׁנָה ד' מַשְׁמַע דְּאֵין צָרִיךְ קְרָא, לְתַנָּא דִּבְרַיְתָא צָרִיךְ קְרָא:

{י} <B>הַכְּנָפַיִם כוּ'. </B>הַנּוֹצָה הַדַּקָּה מֵהֶם. הָרַמְבַּ"ם: {יא} תֵּימַהּ, הָא לְעֵיל תְּנַן שֶׁאֵין צָרִיךְ הֶכְשֵׁר מַיִם וְלֹא הֶכְשֵׁר שֶׁרֶץ. וְהָרַמְבַּ"ם לֹא כָּתַב שֶׁרֶץ: {יב} וְהָרַמְבַּ"ם פֵּרֵשׁ, לְהַשְׁלִים כַּזַּיִת, אִם בָּלַע כַּזַּיִת בָּשָׂר עִם הַנּוֹצָה:

{יג} וְרַשִׁ"י סוֹבֵר שְׁתֵּי בֵּיצִים: {יד} רוֹצֶה לוֹמַר דְּהָא דִּמְטַמְּאָה אָדָם אֵינָהּ מִדְּאוֹרַיְתָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} לֵיכָּא מַאן דְּפָלֵיג, כְּמוֹ שֶׁכָּתַבְתִּי לְעֵיל אוֹת ח': {טז} דְּאֵבָר מִן הַחַי אֵינוֹ נוֹהֵג אֶלָּא בִּטְהוֹרִים. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יז} וְהַדִּבּוּר פֹּה בְּאֳכָלִין שֶׁל תְּרוּמָה. הָרַמְבַּ"ם:

{יח} <B> טוֹפֵחַ.</B> וַאֲפִלּוּ שֶׁלֹּא לְהַטְפִּיחַ. וְלֹא דָּמֵי לְהָא דִּתְנַן בְּסוֹף פֶּרֶק ח' מַשְׁקֶה טוֹפֵחַ אֵינוֹ חִבּוּר לֹא לַטֻּמְאָה וְלֹא לַטָּהֳרָה. אִי נַמִּי, טוֹפֵחַ דְּמַתְנִיתִין לָאו דַּוְקָא. הָרַ"שׁ:

{א} דְּאִי בְּהֻכְשַׁר, אִם כֵּן מִכִּי נְתָנַתּוּ בַּקְּדֵרָה טִמְאַתּוּ. הָרַ"שׁ: {ב} שֶׁסְּתָם יָדַיִם שְׁנִיּוֹת הֵן. הָרַ"שׁ. אֲבָל בְּסֵיפָא תָּנֵי בְּהֶדְיָא בְּיָדַיִם מְסֹאָבוֹת, דְּלֹא תֵימָא מִשּׁוּם טְבוּל יוֹם לְחוּד קָתָנֵי, קָא מַשְׁמַע לָן דִּמְסֹאָבוֹת וּמִשּׁוּם שְׁתֵּי טֻמְאוֹת דְּאִית בָּהּ. נִרְאֶה לִי: {ג} דִּלְמַשְׁקִין אֵין שִׁעוּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} <B> טְמֵא כוּ'.</B> וְקָשֶׁה, דְּלֹא תָּנֵי הָכָא חָזַר לַקְּדֵרָה מַה דִּינוֹ. וְנִרְאֶה לִי, כֵּיוָן דְּעֵרֵב וְתָנֵי בֵּין נָגְעָה בְּמָקוֹם הַנָּגוּב וּבֵין בִּמְקוֹם הַמַּשְׁקֶה, הִלְכָּךְ לֹא תָּנֵי חָזַר כוּ', דְּעַל כָּרְחָךְ אֵינוֹ מְטַמֵּא אֶת הַכֹּל כְּשֶׁחָזַר אֶלָּא אִם כֵּן נָגְעָה בִּמְקוֹם הַמַּשְׁקֶה אוֹ בְּכַבֵּיצָה וַאֲפִלּוּ נָגוּב, אֲבָל בְּפָחוֹת מִכַּבֵּיצָה וְנָגוּב לֹא מַהֲנֵי הַחֲזָרָה כְּלָל, וְכֵיוָן דְּלֹא מָצֵי קָתָנֵי סְתָמָא חָזַר, לֹא חָשׁ לְהַאֲרִיךְ וּלְחַלֵּק: {ה} הָרַ"שׁ. וּכְלוֹמַר לְאַשְׁמְעִינַן דְּאַף עַל גַּב דְּיֵשׁ בָּהּ שְׁתֵּי טֻמְאוֹת, אֲפִלּוּ הָכִי הַקְּדֵרָה טְהוֹרָה: {ו} מִלְּשׁוֹן מְסַלֵּק אֶת הַטַּבְלָה כֻּלָּהּ וּמְנַעֲרָהּ. וּלְלִישָׁנָא אַחֲרִינָא הוּא מִלְּשׁוֹן נִעֵר וְכִסָּה. הָרַ"שׁ: 

~ פרק ב - משנה ב

{ז} וְאַף עַל גַּב דִּלְטַעֲמָא דְּזִמְנִין כוּ' סַגִּי בְּכַבֵּיצָה, שֶׁמָּא אֵין רְגִילוּת שֶׁיְּהֵא אָדָם צָמֵא עַד דְּאָכֵיל לֵיהּ כַּחֲצִי פְּרָס. הָרַ"שׁ: {ח} <B> הָאוֹכֵל כוּ'.</B> שֶׁמָּצִינוּ אוֹכֵל חָמוּר מִן הַמַּאֲכָל, דְּנִבְלַת עוֹף טָהוֹר לֹא מְטַמֵּא בַּחוּץ, וְאִלּוּ אֲכָלָהּ מְטַמֵּא בְּגָדִים, וְאָנוּ אֵיךְ לֹא נַעֲשֶׂה אוֹכֵל כַּמַּאֲכָל. גְּמָרָא. וְאַף עַל גַּב דְּמִטַּעַם הַגְּזֵרָה דְּזִמְנִין אָכִיל כוּ' אֲפִלּוּ אִי נַעֲשֵׂהוּ שֵׁנִי סַגִּי, דְּהָא עוֹשֶׂה מַשְׁקִים תְּחִלָּה, רַבִּי אֱלִיעֶזֶר מַשְׁוֶה מִדּוֹתָיו לַעֲשׂוֹת אוֹכֵל כַּמַּאֲכָל לְגַמְרֵי. תּוֹסָפוֹת. וְכֵן נַמִּי בְּאֹכֶל שְׁלִישִׁי לֹא שָׁיְכָא הַךְ גְּזֵרָה, דְּאֵין שְׁלִישִׁי עוֹשֶׂה רְבִיעִי בַּתְּרוּמָה, לֹא חִלְּקוּ חֲכָמִים בָּאֳכָלִין. הָרַ"שׁ: {ט} וְאִם תֹּאמַר, לֵימָא מִשּׁוּם זִמְנִין דְּאָכֵיל כוּ', וְיֵשׁ לוֹמַר, דְּאִי לָאו דְּאַשְׁכְּחַן הָכִי בְּעָלְמָא, לֵית לָן לְמִגְזַר, דִּלְהָוֵי כְּחוּכָא וְאִטְלוּלָא. תּוֹסָפוֹת: {י} דְּטָהֳרַת תְּרוּמָה טֻמְאָה הִיא אֵצֶל קֹדֶשׁ, וְחָיְשִׁינַן שֶׁמָּא רִאשׁוֹן הוּא וְעוֹשֶׂה שֵׁנִי. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} צָרִיךְ לוֹמַר וְהָכִי מוֹכִיחַ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יב} <B> וּמְטַמְּאִים.</B> לַשְּׁלִישִׁי בַּקֹּדֶשׁ: {יג} <B> נֶאֱכָל כוּ'.</B> וְאֵין לִתְמֹהַּ אֵיךְ נֶאֱכֶלֶת הַתְּרוּמָה כֵּיוָן דִּטְמֵאָה הִיא, מֵאַחַר דְּבַתְּרוּמָה אֵין בָּהּ רְבִיעִי לֵיכָּא אִסּוּרָא. וְכֵן בְּמִשְׁנָה ה' לְעִנְיַן חֲמִישִׁי בַּקֹּדֶשׁ. הָרַ"שׁ וְתוֹסָפוֹת: 

~ פרק ב - משנה ו

{יד} דְּבִגְדֵי אוֹכְלֵי תְּרוּמָה מִדְרָס לַקֹּדֶשׁ, אֲבָל לֹא נוּכַל לַחֲשֹׁב אֳכָלִין לְאָב לֹא מִדִּבְרֵי תּוֹרָה וְלֹא מִדִּבְרֵיהֶם, כִּי אִם כְּנִטְמְאוּ בְּשֶׁרֶץ שֶׁיִּהְיוּ רִאשׁוֹן. הָרַמְבַּ"ם:

{טו} אוֹ עַל טָהֳרַת הַקֹּדֶשׁ. הָרַ"שׁ. וְלִכְאוֹרָה שֶׁבְּכַוָּנָה הִשְׁמִיטוֹ הָרַ"ב, כְּדֵי לְפָרֵשׁ מַתְנִיתִין דְּהָכָא דְּסָבְרָה נַמִּי דְּאֵין חֻלִּין נִתְפָּסִים בִּקְדֻשַּׁת הַקֹּדֶשׁ. וְקָשֶׁה, דִּבְמַתְנִיתִין דִּלְקַמָּן מְפָרֵשׁ לְמַתְנִיתִין דְּהָכָא אִפְּכָא:
