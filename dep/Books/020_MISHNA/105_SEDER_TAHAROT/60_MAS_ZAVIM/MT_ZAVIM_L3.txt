{א} וְקָשֶׁה, דְּלֹא פֵרֵשׁ לְבֵית שַׁמַּאי דִּין רְאִיָּה גוּפַהּ כְּמוֹ שֶׁפֵּרֵשׁ לְבֵית הִלֵּל דְּאֵינָהּ מְטַמְּאָה בְמַשָּׂא. וְהָרַמְבַּ"ם לֹא פֵרֵשׁ בִּדְבֵית הִלֵּל כְּלוּם. וְהָרַ"שׁ פֵּרֵשׁ, דִּלְבֵית שַׁמַּאי מַסִּיט אֶת הָרְאִיָּה, תּוֹלִין. וּפֵרֵשׁ נַמִּי דְּמִשְׁכָּב וּמוֹשָׁב תּוֹלִין, כִּדְתָנֵי בַתּוֹסֶפְתָּא. וּמִנַּהּ אָמְרִינַן בַּגְּמָרָא, דִּכְשׁוֹמֶרֶת יוֹם דְּבֵית שַׁמַּאי, הַיְנוּ כְּבוֹעֵל שׁוֹמֶרֶת יוֹם. וְהָרַ"ב עֵרֵב שְׁנֵי הַפֵּרוּשִׁים: {ב} <B> שְׁתַּיִם.</B> דְּאִלּוּ אַחַת, לְבֵית הִלֵּל אֵינוֹ מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב:

{א} מֵרִבּוּי דְאִישׁ, דִּמְרַבִּינַן קְטַנִּים, כִּדְאִיתָא בַגְּמָרָא. וְנִרְאֶה לִי דַּהֲוֵי כְעֵין מִעוּט אַחַר מִעוּט שֶׁהוּא לְרַבּוֹת. דְּהָא אִישׁ מַשְׁמָעוֹ בֶּן שְׁלֹשׁ עֶשְׂרֵה, וּמְמַעֵט הַפָּחוֹת מִכָּן: {ב} כְּלוֹמַר מַה שֶּׁהוּא רָגִיל שֶׁיָּזוּב מִמֶּנּוּ, וְהוּא לֹבֶן וְדוֹמֶה לְמֵי בָצֵק, וְגַם כְּלֹבֶן בֵּיצָה, דְּכֻלַּהּ חֲדָא מִלְּתָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} אִי הָכִי, כִּי אִטַּמּוּ בִּשְׁאָר טֻמְאוֹת [בְּמֵת אוֹ בְשֶׁרֶץ] לֹא לִשַּׁלְּחוּ. אָמַר קְרָא מִזָּכָר, מִטֻּמְאָה הַפּוֹרֶשֶׁת מִן הַזָּכָר. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} וְאַף עַל גַּב דְּאַנְדְּרוֹגִינוֹס קָמֵי שְׁמַיָּא גַּלְיָא אִם זָכָר הוּא אִם נְקֵבָה, יֵשׁ לוֹמַר דִּמְמַעֵט לֵיהּ מִטַּעַם שֶׁהוּא מְשֻׁנֶּה. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ח} תּוֹסֶפְתָּא. וְחֹלִי אֵין צָרִיךְ לְמִתְנֵי, דְּוַדַּאי דְּכָל זְמַן שֶׁמִּצְטַעֵר: {ט} <B> רַבִּי יוֹסֵי אוֹמֵר.</B> בְּסִפְרִי פְּלִיגֵי בִּדְרָשֵׁי דִקְרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} דְּאַרֵישָׁא מְהַדַּר, דְּרוֹאֶה קֶרִי אֵינוֹ מְטַמֵּא מֵעֵת לְעֵת דְּהַיְנוּ בְּזִיבָה שְׁנִיָּה, וְקָאָמַר דְּנָכְרִי שֶׁרָאָה קֶרִי מְטַמֵּא רְאִיָּה שֶׁהִיא שְׁנִיָּה לָרְאִיָּה שֶׁרָאָה כְּבָר קֹדֶם שֶׁנִּתְגַּיֵּר. וּבִתְנַאי שֶׁתִּהְיֶה רְאִיָּה הַשְּׁנִיָּה אַחַר שֶׁפָּסַק מִן הַדְּאָגָה שֶׁל קַבָּלַת מִצְוֹת: {יא} הָרַ"שׁ. וְקָשֶׁה, דְּאִם כֵּן תַּנָּאָה הוּא וַהֲוָה לֵיהּ לְמִתְנֵי בְּאֳהָלוֹת פֶּרֶק י"א מִשְׁנָה ז'. וְלָכֵן נִרְאֶה כְּגִרְסַת הַסֵּפֶר, וְהִיא גִרְסַת הָרַמְבַּ"ם, וְאֵינוֹ אֶלָּא תּוֹסֶפֶת בֵּאוּר שֶׁכָּךְ הוּא הַהֹוֶה:

{א} דְּלֵיכָּא לְמֵיחַשׁ שֶׁמָּא יָשַׁב עֲלֵיהֶם הַזָּב. מַהֲרַ"ם: {ב} וְנִרְאֶה שֶׁטֻּמְאָתוֹ בְסָפֵק, וְכֵן דַּעַת הָרַמְבַּ"ם. אֶלָּא שֶׁכָּתַב שֶׁמָּא דָרַס עֲלֵיהֶן הַזָּב, וּלְהָרַאֲבַ"ד אֵין כָּאן סָפֵק: {ג} שֶׁכָּל הַמִּתְבַּשֵּׁל אֵין כֻּלּוֹ נוֹגֵעַ בּוֹ, אֲבָל יֵשׁ מִקְצָתוֹ בַּאֲוִירוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} דִּפְשִׁיטָא, מִי גָרַע מִשְּׁאָר טֻמְאוֹת שֶׁמְּטַמְּאוֹת כְּלֵי חֶרֶס מֵאֲוִירָן, כֵּיוָן דִּכְתִיב גַּבֵּי שְׁרָצִים, לָמָּה לֵיהּ לְמִכְתְּבֵיהּ הָכָא. רַשִׁ"י: {ה} דְּגַבֵּי אָדָם נַמִּי כְתִיב [וַיִּקְרָא טו] וְכֹל אֲשֶׁר יִגַּע בּוֹ הַזָּב, וּמַגָּע נַמִּי כְּתִיב [וַיִּקְרָא טו] וְהַנֹּגֵעַ בִּבְשַׂר הַזָּב, אֶלָּא לְהָבִיא הֶסֵּטוֹ. רַשִׁ"י: {ו} וְהוּא שֶׁכָּרַע הַטָּהוֹר. אֲבָל אִם כָּרַע הַזָּב, לֹא נַעֲשָׂה הַטָּהוֹר אֶלָּא רִאשׁוֹן וְאֵינוֹ מְטַמֵּא בְגָדִים: {ז} וּמִכָּל מָקוֹם אֵין טֻמְאָתָן שָׁוָה, וּכְמוֹ שֶׁכָּתַבְתִּי בְּסָמוּךְ: {ח} וְקָשֶׁה, הֲלֹא אֲפִלּוּ אֵינָהּ נֶחְבֵּאת הָוֵי כֹחָהּ רַע לְפִי שֶׁרוֹתֵת תַּחְתָּיו. תּוֹסָפוֹת. וְהַכֶּסֶף מִשְׁנֶה פֵּרֵשׁ לְדִבְרֵי הָרַמְבַּ"ם, כְּשֶׁאוֹחֵז בָּהּ וְהִיא נֶחְבֵּאת תּוֹךְ יָדוֹ וְכוֹפֵף אוֹתָהּ הִיא מֵזִיזָה אוֹתָהּ:

{א} דַּהֲוָה לֵיהּ לְמֵימַר וְרָחַץ בַּמַּיִם בָּעֶרֶב, אֶלָּא לוֹמַר דְּעוֹד טָמֵא גַּם כֵּן כָּל כְּלִי שֶׁיִּגַּע בּוֹ. וּכְלֵי שֶׁטֶף הַיְנוּ כָּל שְׁאָר כֵּלִים חוּץ מִכְּלֵי חֶרֶס. עַל שֵׁם שֶׁיֵּשׁ לָהֶם טָהֳרָה בְּמִקְוֶה. רַשִׁ"י: {ב} דִּכְלִי שֶׁטֶף דּוֹמֶה קְצָת לְבֶגֶד, דַּהֲוֵי נַמִּי כְּלִי, מַה שֶּׁאֵין כֵּן הָאָדָם. וּכְלִי חֶרֶס אֵין דּוֹמֶה לוֹ, שֶׁאֵין לוֹ טָהֳרָה בְּמִקְוֶה: {ג} <B> כָּל הַמְטַמֵּא כוּ'.</B> כָּל אָדָם הַמְטַמֵּא קָאָמַר, דְּהָא מַשְׁכַּחַת כֵּלִים שֶׁאֵין מְטַמְּאִין כֵּלִים, וְאִם נָגְעוּ מְטַמְּאִין אֳכָלִים לִהְיוֹת תְּחִלָּה כִּדְמוּכָח בְּמַתְנִיתִין דִּלְקַמָּן. דְּדַוְקָא בְּאֵין נוֹגְעִים מַיְרֵי סֵיפָא, הָא נוֹגְעִים מְטַמְּאִין שְׁנַיִם. וְכֵן הוּא בְּהֶדְיָא בַתּוֹסֶפְתָּא. וְשָׁם מְפָרֵשׁ הָרַ"ב דְּמַתְנִיתִין כָּל אָדָם קָאָמַר: {ד} כְּלוֹמַר, לְעִנְיַן אֳכָלִים וּמַשְׁקִים. וְאַף עַל פִּי שֶׁאֵינוֹ אַב הַטֻּמְאָה לְעִנְיַן אָדָם וּכְלֵי חֶרֶס:

{כא} כִּדְיָלֵיף לְהוּ בְּסִפְרָא. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} הָרַ"שׁ. וּלְשִׁיטָתֵיהּ אָזֵיל כוּ', אֲבָל הָרַ"ב בְּחִנָּם נַקְטֵיהּ וְאֵין בּוֹ צֹרֶךְ לְפֵרוּשׁ הַמִּשְׁנָה. וְגַם בְּעִקַּר הַדִּין נִרְאֶה לִי דְּלָאו כְּהִלְכָתָא הוּא. וּרְאָיָה מִמִּשְׁנָה ה' ו' דְּפֶרֶק ד':

{כג} <B> שְׁנַיִם.</B> בְּמַגָּע וְלֹא בְמַשָּׂא: {כד} <B> הַמַּסִּיט.</B> כְּלוֹמַר, וְאֵינוֹ נוֹשֵׂא וְנִשָּׂא. וּבֵין שֶׁהַטָּהוֹר הִסִּיט לַטָּמֵא אוֹ אִפְּכָא. וְקָאֵי אַזָּב וְכוּ'. אֲבָל בִּמְצֹרָע וּמִשְׁכָּב וּמוֹשָׁב לֵיכָּא לְפָרוֹשֵׁי הָכִי, דְּהָא כָּל טֻמְאוֹת הַמַּסִּיט טְהוֹרוֹת חוּץ מִשֶּׁל זָב: {כה} בְּלֵידָה יָבֵשְׁתָּא: {כו} <B> הַנּוֹשֵׂא כוּ'.</B> כְּלוֹמַר וַאֲפִלּוּ אֵינוֹ מַסִּיט. מַה שֶּׁאֵין כֵּן נְבֵלָה וּמֵת דְּמִשְׁנָה ג' דְּבָעִינַן דַּוְקָא שֶׁיְּהֵא נִיסָט, עַיֵּן שָׁם:

{לב}<B> שֶׁאֵין כוּ'.</B> דְּאִלּוּ יֵשׁ בָּהֶם, אַף הַנּוֹגֵעַ בָּהֶם מְטַמֵּא בְגָדִים, לְפִי שֶׁאִי אֶפְשָׁר שֶׁיִּגַּע בַּמַּיִם שֶׁלֹּא יַסִּיט אוֹתָם, וַהֲרֵי הוּא מְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד. אֲבָל אִם קָרְשׁוּ וְנַעֲשׂוּ כִגְלִיד דְּאֶפְשָׁר לִגַּע בְּלֹא הֶסֵּט, תָּנֵי בְפֶרֶק קַמָּא דְכֵלִים מִשְׁנָה ב' חֲשׂוּכֵי בְגָדִים בְּמַגָּע. וּכְמוֹ שֶׁכָּתַב הָרַ"ב בְּמִשְׁנָה ח'. וְהַתַּנָּא הָתָם וְהָכָא מִלְּתָא דִפְסִיקָא נָקַט: {לג} וְאֵין הַלָּשׁוֹן מַשְׁמַע כֵּן, דְּהָכִי הֲוָה לֵיהּ לְמִתְנֵי, בַּמֶּה דְבָרִים אֲמוּרִים בְּאָדָם, אֲבָל אֳכָלִים וְכֵלִים הַנּוֹגְעִים בְּאֶחָד כוּ'. וְהָרַמְבַּ"ם פֵּרֵשׁ, כָּל הַנּוֹגֵעַ בִּשְׁאָר אֲבוֹת הַטֻּמְאוֹת חוּץ מֵאֵלּוּ שֶׁמָּנָה עַד הֵנָּה [שֶׁמְּטַמְּאִין שְׁנַיִם כוּ']. וְאָמְרוֹ חוּץ מִן הָאָדָם, רָצָה לוֹמַר הַמֵּת, שֶׁגַּם הוּא לֹא נִמְנָה כָאן, וְנוֹגֵעַ מְטַמֵּא שְׁנַיִם כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
