{א} <B> מֵי רְבִיעִית.</B> וְנֻסַּח אַחֵר מֵרְבִיעִית, כְּלִי הַמַּחֲזִיק רְבִיעִית. רַשִׁ"י: {ב} <B> לַיָּדַיִם.</B> דִּתְנַן (חֲגִיגָה פ"ב מ"ה) נוֹטְלִין לַיָּדַיִם לַחֻלִּין וְלַמַּעֲשֵׂר וְלַתְּרוּמָה, וּסְתָם יָדַיִם אֵינָן אֶלָּא שְׁנִיּוֹת וּפוֹסְלִין תְּרוּמָה וְלֹא חֻלִּין, אֲבָל מִשּׁוּם סְרַךְ תְּרוּמָה תִּקְּנוּ אַף לַחֻלִּין, כְּדֵי שֶׁיְּהֵא רָגִיל בִּנְטִילַת יָדַיִם וְיִטְּלֵן לַתְּרוּמָה. גְּמָרָא: {ג} וְכֵן הָא דְּקָאָמַר לְאַרְבָּעָה, דְּמַשְׁמַע וְלֹא לַחֲמִשָּׁה, שֶׁהֲרֵי לֹא נָתְנוּ חֲכָמִים שִׁעוּר רַק לַמַּיִם שֶׁבַּכְּלִי, וְכִי אִיכָּא שִׁעוּר לֹא חָשׁוּ כַּמָּה יַגִּיעַ לְכָל אֶחָד רַק שֶׁיַּגִּיעַ עַד הַפֶּרֶק. הָרַ"שׁ: {ד} <B> וּלְמֵאָה.</B> גֻּזְמָא הִיא, כְּלוֹמַר כָּל זְמַן שֶׁמַּסְפִּיקִין לִשְׁפֹּךְ כָּל אֶחָד שָׁלֹשׁ פְּעָמִים. טוּר: {ה} <B> רַבִּי יוֹסֵי כוּ'.</B> פָּלֵיג אַף בְּרֵישָׁא דְּלִשְׁנַיִם. הָכִי מוּכָח בַּגְּמָרָא: {ו} וּכְשֶׁנְּטָלָן בְּפָחוֹת מֵרְבִיעִית דַּאֲתוּ מִשְּׁיָרֵי טָהֳרָה. דְּאִלּוּ נְטָלָן מֵרְבִיעִית, הַמַּיִם עַצְמָן טְהוֹרִים, כְּתַנָּא קַמָּא דְּרֵישׁ פֶּרֶק ב', וְעַיֵּן שָׁם בְּהָרַ"ב: {ז} דְּכֵיוָן שֶׁלֹּא טִהֲרוּ אֶת הַיָּדַיִם כָּל מַה שֶּׁמּוֹסִיף עֲלֵיהֶם הֲרֵי הוּא מוֹסִיף טֻמְאָה. הָרַ"שׁ:

{ח} הָרַ"ב הוֹסִיף זֶה אַף עַל פִּי שֶׁלֹּא נִשְׁנָה כֵן, וְגַם בְּתוֹרַת כֹּהֲנִים לֵיתֵיהּ. מִשּׁוּם דְּכָל שֶׁכֵּן הוּא מִכֵּלִים:

{ט} לַקֹּדֶשׁ. וְהַהֶפְרֵשׁ שֶׁבֵּין נְטִילַת יָדַיִם וּבֵין טְבִילָה, כִּי נְטִילַת יָדַיִם הוּא בְּמַיִם שְׁאוּבִין, וְהוּא שֶׁיָּבֹאוּ מִכְּלִי בְּכֹחַ נוֹתֵן, אֲבָל שְׁאוּבִין בְּמִקְוֶה לֹא, וְשִׁנּוּי מַרְאֶה פּוֹסֵל, וּבָעִינַן רְבִיעִית לְכָל אֶחָד. וּטְבִילַת יָדַיִם פָּסוּל בִּשְׁאוּבִין, וְצָרִיךְ אַרְבָּעִים סְאָה. הָרַמְבַּ"ם. וְהָרַאֲבַ"ד סְבִירָא לֵיהּ דִּשְׁאוּבִין בְּמִקְוֶה נַמִּי כְּשֵׁרִין לִטְבִילַת יָדַיִם, כֵּיוָן דִּכְשֵׁרִין לְבַעֲלֵי קְרָיִין: {י} <B> פְּסוּלִין.</B> דּוּמְיָא דְּקִדּוּשׁ יָדַיִם מִן הַכִּיּוֹר דִּכְתִיב הָתָם מַיִם יְתֵרָה. בֵּית יוֹסֵף: {יא} דְּתַנָּא קַמָּא סָתַם דְּבָרָיו וְאָמַר שָׁרָה בָּהֶם פִּתּוֹ כוּ', וְאָתָא אִיהוּ וּמִתַּמֵּהַּ וְכִי כוּ'. וּמַהֲרַ"ם כָּתַב, דְּשָׁרָה דְּתַנָּא קַמָּא מַשְׁמַע דְּנִתְכַּוֵּן לִשְׁרוֹת, הָא נָפַל מִיָּדוֹ שֶׁלֹּא בְכַוָּנַת שְׁרִיָּה כָּשֵׁר. הָא נִתְכַּוֵּן לִשְׁרוֹת בָּזֶה וְנָפַל בָּזֶה פָּסוּל. וְאָתָא רַבִּי שִׁמְעוֹן לְמֵימַר אֲפִלּוּ נִתְכַּוֵּן:

{א} <B> אַחַת.</B> נִתְבָּאֵר בַּחֲגִיגָה כְּשֶׁנִּטְמְאָה יָדוֹ אַחַת, לֹא תִּטַּמֵּא הַשְּׁנִיָּה, וְכֵן כְּשֶׁנִּטְמְאוּ שְׁנֵיהֶם מְטַהֲרָן אַחַת אַחַת. וְכָל זֶה לַתְּרוּמָה וְלֹא לַקֹּדֶשׁ. הָרַמְבַּ"ם: {ב} וְהוּא שֶׁאוֹתָהּ הַשְּׁטִיפָה מְרֻבָּה כִּשְׁתַּיִם. תּוֹסָפוֹת: {ג} לְשִׁיטָתֵיהּ בְּרֵישׁ פֶּרֶק קַמָּא, דְּנוֹתְנִים לַיָּדַיִם, אַמַּיִם שְׁנִיִּים קָאֵי: {ד} הַיְנוּ לַפֵּרוּשׁ הָרִאשׁוֹן. אֲבָל לְהָרַמְבַּ"ם, מְפֹרָשׁ בְּנֻסְחַת אֶרֶץ יִשְׂרָאֵל דְּקָאֵי אַמַּיִם שְׁנִיִּים, שֶׁאִם נָפַל כִּכָּר שֶׁל תְּרוּמָה בְּאֵלּוּ מַיִם שְׁנִיִּים, טָהוֹר לְתַנָּא קַמָּא כְּמוֹ בְּמַיִם שְׁנִיִּים דְּעָלְמָא. וְרַבִּי יוֹסֵי אוֹמֵר שֶׁמְּטַמְּאִין כִּכָּר שֶׁל תְּרוּמָה כְּמוֹ מַיִם רִאשׁוֹנִים שֶׁהֵם תְּחִלָּה, הוֹאִיל וּמַיִם רִאשׁוֹנִים הָיוּ שְׁטִיפָה אַחַת לִשְׁתֵּי הַיָּדַיִם:

{ה} דְּאִלּוּ מֵרְבִיעִית, הָא סְבִירָא לֵיהּ לְתַנָּא קַמָּא דְּרַבִּי יוֹסֵי דְּהַמַּיִם גַּם כֵּן נִטְהֲרוּ. וַהֲלָכָה כְּוָתֵיהּ: {ו} קָשֶׁה, דְּזֶה אֵינוֹ אֶלָּא לְרַבִּי מֵאִיר דְּמַצְרִיךְ מַיִם שְׁנִיִּים אַף לְדַאֲתוּ מִשְּׁיָרֵי טָהֳרָה כְּשֶׁנָּטַל שְׁתֵּי יָדָיו, וּכְבָר כָּתַב דְּאֵין הֲלָכָה כְּמוֹתוֹ. וְיֵשׁ לוֹמַר, דְּמַה שֶּׁכָּתַב אֵין הֲלָכָה כְּרַבִּי מֵאִיר הַיְנוּ לְהָרַמְבַּ"ם, אֲבָל לַפֵּרוּשׁ הָרִאשׁוֹן, הֲלָכָה כְּרַבִּי מֵאִיר. וְכֵן כָּתַב הָרַ"שׁ בְּהֶדְיָא. וְתֵדַע, דְּהָא לְהָרַמְבַּ"ם לֹא אָמְרוּ פָּחוֹת מֵרְבִיעִית אֶלָּא בְּמַיִם שְׁנִיִּים: {ז} <B> תְּרוּמָה.</B> מִזֶּה מוֹכִיחַ הָרַמְבַּ"ם דְּלֹא הִצְרִיכוּ מַיִם שְׁנִיִּים אֶלָּא לִתְרוּמָה, אֲבָל לְחֻלִּין אַף בָּרִאשׁוֹנָה נִטְהֲרוּ הַיָּדַיִם [וּמִמֵּילָא דְּהוּא הַדִּין הַמַּיִם], וְכִי נָפַל עֲלֵיהֶן כִּכָּר שֶׁל חֻלִּין, טָהוֹר. וּשְׁאָר הַפּוֹסְקִים מַצְרִיכִים אַף לְחֻלִּין מַיִם שְׁנִיִּים. וּמִדְּנָקַט בְּמַתְנִיתִין תְּרוּמָה, נִרְאֶה לִי דְּאַף דִּלְדִבְרֵיהֶם טָהוֹר הַכִּכָּר, דְּכֵיוָן דִּנְטִילַת יָדַיִם לְחֻלִּין רַק מִשּׁוּם סְרַךְ תְּרוּמָה הוּא, הֵקֵלּוּ לְעִנְיַן טֻמְאָה, דְּהֵם אָמְרוּ וְהֵם אָמְרוּ. וּמֵהַאי טַעֲמָא נִרְאֶה לִי דְּלֹא קָשֶׁה אֵיךְ הַשְּׁנִיִּים טְהוֹרִים, הָא הָרִאשׁוֹנִים טְמֵאִים וִיטַמְּאוּ הַיָּדַיִם וְהַיָּדַיִם שְׁנִיִּים, דְּהֵם אָמְרוּ כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} שֶׁנִּטְמְאוּ מֵחֲמַת יָדָיו וְחוֹזְרִין וּמְטַמְּאִין אֶת הַיָּד. בֵּית יוֹסֵף: {ט} וְכָל דָּבָר הַחוֹצֵץ בַּגּוּף, חוֹצֵץ בַּיָּדַיִם לְחֻלִּין. גְּמָרָא: {י} וַאֲפִלּוּ לַפֵּרוּשׁ הָרִאשׁוֹן שֶׁל הָרַ"ב בְּרֵישׁ פִּרְקִין, יֵשׁ לוֹמַר דְּלֹא מַצְרִיךְ שְׁתֵּי פְּעָמִים אֶלָּא לְטַהֵר הַמַּיִם לְעִנְיַן טֻמְאַת כִּכָּר. אֲבָל הַיָּדַיִם נִטְהֲרוּ בְּפַעַם אַחַת כְּשֶׁאֵין דָּבָר חוֹצֵץ:

{יא} בֵּאוּר הָעִנְיָן, שֶׁאֵין הַיָּדַיִם מְטַמְּאִין מַשְׁקִין אֶלָּא עַד הַפֶּרֶק. וְכֵן אֵין מַיִם שְׁנִיִּים מְטַהֲרִים הָרִאשׁוֹנִים אֶלָּא בְּעוֹדָן בְּתוֹךְ הַפֶּרֶק. וְחוּץ לַפֶּרֶק דְּרֵישָׁא וְסֵיפָא, הוּא שֶׁהָיוּ בְּתוֹךְ הַפֶּרֶק וְיָצְאוּ אַחַר כָּךְ חוּץ לַפֶּרֶק. בֵּית יוֹסֵף: {יב} וְגַם אֶת הַיָּד. הָרֹא"שׁ: {יג} לְפִי שֶׁהֵם כֻּלָּם בְּמַדְרֵגַת יָד אַחַת אָנוּ חוֹשְׁבִים אוֹתָם. הָרַמְבַּ"ם: {יד} דְּאִם הָיָה חֲצִי לֹג לַחֲמִשָּׁה וּכְשֶׁהַשַּׁמָּשׁ שׁוֹפֵךְ מַיִם עַל יְדֵיהֶם וְנִשְׁאַר בַּכְּלִי רְבִיעִית לַשְּׁנַיִם הָאַחֲרוֹנִים, מַהֲנֵי נְטִילָה, וּסְפֵק יָדַיִם טְהוֹרוֹת. הָרַ"שׁ:

{א} <B> הַמַּכְנִיס כוּ'.</B> כָּל הָנֵי דִּבְפִרְקִין, לֹא מִשְּׁמֹנָה עָשָׂר דָּבָר הֵן, חוּץ מִכִּתְבֵי הַקֹּדֶשׁ. דְּהָנֵי אֲפִלּוּ בְּיָדַיִם טְהוֹרוֹת מַיְרֵי, וְדִשְׁמֹנָה עָשָׂר דָּבָר בִּסְתָם יָדַיִם הוּא: {ב} לְשׁוֹן הָרַ"שׁ, כְּלוֹמַר מַה שֶּׁלֹּא מָצִינוּ הוּא מִשּׁוּם דְּלֹא אֶפְשָׁר: {ג} לְפִי שֶׁאֵין כֵּלִים מְקַבְּלִים טֻמְאָה מִוָּלָד אֶלָּא מִמַּשְׁקִין, וְנָקַט נַמִּי אֳכָלִים אַף עַל פִּי שֶׁמִּטַּמְּאִים מִוָּלָד, שֶׁהֲרֵי מוֹנִין בָּהֶם רִאשׁוֹן וְשֵׁנִי: {ד} מְבֹאָר. כְּלוֹמַר, אֵינְךָ צָרִיךְ לְהִסְתַּפֵּק, כִּי כְּבָר מְפֹרָשׁ הַדָּבָר הֵיטֵב. מַהֲרַ"ם:

{ה} <B> כֹּל כוּ'.</B> הַאי כְּלָלָא לָאו כְּרַבָּנָן דִּלְעֵיל דְּאָמְרֵי בְּוָלָד אֵין מְטַמֵּא אֶלָּא כְּדִמְסַיֵּם דִּבְרֵי רַבִּי יְהוֹשֻׁעַ: {ו} מִדְּסָתַם וְלֹא פֵּרֵשׁ בְּמַאי קָמַיְרֵי, נִרְאֶה דְּבַתְּרוּמָה פְּלִיגֵי. וּמִשּׁוּם הָכִי נַמִּי כָּתַב דְּאֵין הֲלָכָה כְּרַבִּי יְהוֹשֻׁעַ, דְּלַקֹּדֶשׁ קַיְמָא לָן דְּהַיָּד מְטַמָּא הַשְּׁנִיָּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} <B> כִּתְבֵי כוּ'.</B> וְהָא דְּלֹא מַיְתֵי מֵאוֹכֵל אֹכֶל שֵׁנִי שֵׁנִי, דִּפְסוּל גְּוִיָּה שָׁאנֵי. הָרַ"שׁ: {ח} הַיְנוּ לַקֹּדֶשׁ. אֲבָל לַתְּרוּמָה, מִשְּׁמֹנָה עָשָׂר דָּבָר הוּא. וְאַף דְּהָכָא בִּתְרוּמָה קָיְמִינַן, נִיחָא לֵיהּ לְמֵימַר דְּאֵין לָהֶם עִקָּר מִן הַתּוֹרָה: {ט} וְקָשֶׁה, דַּהֲוָה לֵיהּ לִמְנוֹת בִּשְׁמֹנָה עָשָׂר דָּבָר, יָדַיִם הַבָּאִים מֵחֲמַת סֵפֶר. וְאַף דְּלֹא הָווּ יוֹתֵר מִשְּׁמֹנָה עָשָׂר דְּכוּלַּהּ חֲדָא גְּזֵרָה הִיא וְכִדְמַשְׁמַע בְּהָרַ"ב סוֹף פֶּרֶק ט"ו דְּכֵלִים, מִכָּל מָקוֹם הֲוָה לֵיהּ לְפָרֵשׁ כֵּן, דְּנַפְקָא מִנַּהּ דְּהָכָא אֲפִלּוּ טְהוֹרוֹת לְגַמְרֵי וּכְמוֹ שֶׁכָּתַבְתִּי בְּרֵישׁ פִּרְקִין:

{י} <B> שֶׁמִּלְּמַעְלָן כוּ'.</B> וְהוּא הַדִּין לְבֵין הַדַּפִּין. וְלֹא אֲתָא אֶלָּא לַאֲפוֹקֵי כְשֶׁאֵינָן מְחֻבָּרִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} <B> עַמּוּד.</B> כִּי אֲשֶׁר יִשָּׁאֵר בַּסּוֹף, כָּל זְמַן שֶׁלֹּא יַעֲשֶׂה לוֹ עַמּוּד שֶׁיִּגָּלֵל עָלָיו הַסֵּפֶר וְתִכְלֹל אוֹתוֹ קְדֻשַּׁת הַסֵּפֶר, עֲדַיִן הוּא חֹל, לְפִי שֶׁאֶפְשָׁר שֶׁיַּחְתְּכֶנּוּ. הָרַמְבַּ"ם:

{יב} <B> כְּדִבְרֵי בֶן עַזַּאי.</B> דְּמִדְּבֶן עַזַּאי תָּנֵי בַּהֲדָדֵי שִׁיר הַשִּׁירִים וְקֹהֶלֶת, שְׁמַע מִנַּהּ שֶׁקִּבֵּל שֶׁבִּשְׁנֵיהֶם הָיְתָה הַמַּחֲלֹקֶת וְנִמְנוּ וְגָמְרוּ שֶׁיְּטַמְּאוּ. וְהַיְנוּ שֶׁעַל דִּבְרֵי בֶּן עַזַּאי שָׁנָה לָנוּ רַבִּי דִּבְרֵי רַבִּי עֲקִיבָא שֶׁחוֹלֵק עָלָיו. דְּאִי גַּם לְבֶן עַזַּאי לֹא נֶחְלְקוּ עַל שִׁיר הַשִּׁירִים, הָיָה לוֹ לְסַדֵּר דִּבְרֵי רַבִּי עֲקִיבָא קֹדֶם לְבֶן עַזַּאי כוּ'. לְכָךְ הֵעִיד רַבִּי יוֹחָנָן שֶׁכְּדִבְרֵי בֶּן עַזַּאי כֵּן הוּא, בֵּין לְעִנְיַן הַמַּחֲלֹקֶת בֵּין לְעִנְיַן הַגְּמָר:

{א} וְקָשֶׁה, דְּאִם כֵּן גַּם קֹדֶם שֶׁנִּסְדְּקָה תְּהֵא טְמֵאָה מִדְרָס, כְּמוֹ תֵּבָה שֶׁפִּתְחָהּ מִצִּדָּהּ. אֲבָל בְּפֶרֶק כ' דְּכֵלִים מִשְׁנָה ב' כְּתָב הָרַ"ב, שֶׁכְּשֶׁנִּסְדְּקָה אֵינָהּ רְאוּיָה אֶלָּא לִשְׁכִיבָה, וּכְשֶׁלֹּא נִסְדְּקָה אוֹמְרִים לוֹ עֲמֹד. וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם. וְכֵן עִקָּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ב}<B> עַמּוֹן כוּ'.</B> לְשׁוֹן הָרַמְבַּ"ם, וְרָאוּי לְהוֹצִיא מַעֲשֵׂר וּתְרוּמָה בְּחוּץ לָאָרֶץ הַסָּמוּךְ לְאֶרֶץ יִשְׂרָאֵל, מִצְרַיִם וְעַמּוֹן וּמוֹאָב וּבָבֶל. וְכָל זֶה מִדְּרַבָּנָן: {ג} <B> מַעֲשַׂר עָנִי.</B> וּבְדִין הוּא דַּאֲפִלּוּ תְּרוּמָה גְּדוֹלָה לֹא נִתְקַן, אֶלָּא הַכֹּל מַעֲשַׂר עָנִי, אֶלָּא שֶׁלֹּא רָצוּ לְחַלֵּק כָּל כָּךְ מִשְּׁאָר שָׁנִים. תּוֹסָפוֹת: {ד} <B> מַחְמִיר.</B> שֶׁמַּעֲשֵׂר שֵׁנִי קֹדֶשׁ, וּמַעֲשַׂר עָנִי חֹל. הָרַמְבַּ"ם: {ה} <B> וְהַתְּרוּמָה.</B> וְסוֹף הַפָּסוּק, הָבִיאוּ כוּ', אִם לֹא אֶפְתַּח לָכֶם אֵת אֲרֻבּוֹת הַשָּׁמַיִם, וַהֲרִיקֹתִי לָכֶם בְּרָכָה עַד בְּלִי דָי: {ו} <B> אֵצֶל רַבִּי אֱלִיעֶזֶר.</B> לְפִי שֶׁלֹּא הָיָה אֵצֶל הַמִּנְיָן, מִפְּנֵי הַמַּחֲלֹקֶת, שֶׁעַל יְדֵי כֵן בֵּרְכוּהוּ. הָרַמְבַּ"ם: {ז} <B> אַל תָּחֹשּׁוּ.</B> אַל יְהֵא לָכֶם שׁוּם חֲשָׁשׁ וְגִמְגּוּם, שֶׁהֲרֵי הִסְכַּמְתֶּם לַהֲלָכָה. רַשִׁ"י: {ח} וְהָא דְּקָתָנֵי וְהַנִּדּוֹן שֶׁלְּפָנֵינוּ מַעֲשֵׂה זְקֵנִים, הַיְנוּ מִקַּמֵּי דְּיָדְעֵי לְעֵדוּתוֹ שֶׁל רַבִּי אֱלִיעֶזֶר:

{ט}<B> הַכָּתוּב אוֹמֵר כוּ'. </B>וְקָשֶׁה, מַאי סָלְקָא דַּעְתֵּיהּ דְּרַבָּן גַּמְלִיאֵל, מִי לֹא יָדַע בְּכָל אֵלֶּה. וְגַם הַגֵּר שֶׁדָּחַק וְכָנַס בְּתוֹךְ חֲכָמִים מִסְּתָמָא יָדַע הַמִּקְרָא וְלָמָּה שָׁאַל. וְנִרְאֶה לִי, דְּרַבָּן גַּמְלִיאֵל יָדַע מִזֶּה שֶׁסַּנְחֵרִיב בִּלְבֵּל וְשֶׁזֹּאת הָיְתָה שְׁאֵלַת הַגֵּר, אֶלָּא רַבָּן גַּמְלִיאֵל סָבַר שֶׁיֵּשׁ בַּפָּסוּק תְּרֵי לֹא יָבֹא, שֶׁיִּרְצֶה לֶאֱסֹר לָנוּ אַף בִּזְמַן שֶׁיְּהֵא סָפֵק. וְרַבִּי יְהוֹשֻׁעַ חָשַׁב שֶׁאֶפְשָׁר שֶׁרַבָּן גַּמְלִיאֵל סוֹבֵר סָפֵק אַחֵר וּמַיְתֵי מִלֹּא יָבֹא. וְהֵשִׁיב שֶׁיֵּשׁ לְהַתִּיר מֵחֲמַת בִּלְבּוּל. וְאָמַר לוֹ רַבָּן גַּמְלִיאֵל הַכָּתוּב כוּ' אַף אֲנִי יוֹדֵעַ בִּלְבּוּל כוּ' וְעַל זֶה הָיִיתִי דּוֹרֵשׁ הַמִּקְרָא דְּאָתֵי לַאֲתוֹיֵי מִשּׁוּם דְּמִקְרָא מְסַיְּעֵנִי וְאַחַר כָּךְ אָשִׁיב. וְאָתֵי שַׁפִּיר מַאי דְּקָאָמַר וּכְבָר שָׁבוּ וּמְנָלֵיהּ, אֶלָּא דְּלָמַד כֵּן מִיִּתּוּרָא דִּקְרָא. וְהֵשִׁיב לוֹ רַבִּי יְהוֹשֻׁעַ כוּ' וְשַׁבְתִּי כוּ', וְיִתּוּרָא לִדְרָשָׁא אַחֲרִיתִי, כְּמוֹ בְּמַמְזֵר שֶׁכָּתַב שְׁתֵּי פְּעָמִים לֹא יָבֹא וּמֻתָּר הַסָּפֵק מִן הַתּוֹרָה:

{יד} <b>שֶׁאַתֶּם כוּ'.</b> אָמְרוּ לוֹ חֲכָמִים, כִּי אֵין זֶה גֵּרָעוֹן לְמֹשֶׁה וְאֵין בָּזֶה גְּדֻלָּה לַמֶּלֶךְ. שֶׁאָנוּ כּוֹתְבִין גַּם כֵּן ה' עִם פַּרְעֹה וְנַקְדִּים שֵׁם פַּרְעֹה כוּ'. הָרַמְבַּ"ם: {טו} וּבְמַסֶּכֶת פֵּאָה דִּמְסַיֵּם כִּי הַשֹּׁחַד יְעַוֵּר, הָא תָּנֵי וְגוֹ', וְנִתְכַּוֵּן לְסֵיפֵיהּ דִּקְרָא צֶדֶק צֶדֶק תִּרְדֹּף לְמַעַן תִּחְיֶה:
