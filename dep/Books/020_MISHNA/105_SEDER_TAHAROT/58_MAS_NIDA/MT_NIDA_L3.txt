{א} וְדִין תּוֹרָה כָּךְ הוּא דְּלֵיכָּא לְמֵיחַשׁ לְהָכִי, אֶלָּא דְּהִלֵּל עוֹשֶׂה סְיָג לִדְבָרָיו, וְשַׁמַּאי מִשּׁוּם בִּטּוּל פְּרִיָּה וּרְבִיָּה אֵינוֹ עוֹשֶׂה סְיָג. גְּמָרָא: {ב} דַּהֲווּ לְהוּ שָׁלֹשׁ עוֹנוֹת, הַיּוֹם, אֶמֶשׁ, וְאֶתְמוֹל לְמַפְרֵעַ עַד צָהֳרַיִם. וּמִיהוּ אַכַּתִּי לֹא שָׁמַעְנוּ אֶלָּא דְּאִיכָּא לְמַעוֹטֵי וְלֹא לִיטַמּוּ לְמַפְרֵעַ רַק עוֹנָה אַחַת יוֹתֵר, מִצָּהֳרַיִם עַד שַׁחֲרִית. אֲבָל מַה שֶּׁכָּתַב הָרַ"ב לְהוֹסִיף, לֹא יָדַעְנָא מֵהֵיכָא תֵּיתֵי לְהוֹסִיף. וְרַשִׁ"י לֹא כָּתַב לְהוֹסִיף: {ג} <B> וֶסֶת.</B> תֵּימַהּ, לָמָּה הִפְסִיק וְשָׁנָה הַמְשַׁמֶּשֶׁת בְּעִדִּים, הֲוָה לֵיהּ לְמִתְנֵי מִיָּד כֵּיצַד דַּיָּהּ שַׁעְתָּהּ. וְיֵשׁ לוֹמַר, דְּקָמַשְׁמַע לָן אַף עַל גַּב שֶׁיֵּשׁ לָהּ וֶסֶת צְרִיכָה לְשַׁמֵּשׁ בְּעִדִּים. וְאַף עַל גַּב דְּתָנֵי לֵיהּ בְּמִשְׁנָה ז', הָכָא אַשְׁמְעִינַן דְּאוֹתָהּ בְּדִיקָה חֲשִׁיבָה לְמַעֵט מֵעֵת לְעֵת, דַּהֲוֵי חִדּוּשׁ. וּלְקַמָּן בְּאַרְבַּע נָשִׁים דַּיָּן שְׁעָתָן. תּוֹסָפוֹת: {ד} דְּאִי לָאו הָכִי, לֹא: {ה} וְלֹא יַעֲמֹד מִלָּצֵאת חוּצָה. וּלְהָכִי אֲפִלּוּ רַבָּנָן לֹא חָשׁוּ: {ו} <B> לִפְקִידָה.</B> וְאִצְטְרִיךְ, דְּמַהוּ דְּתֵימָא כֵּיוָן דִּזְמַן מֻעָט הוּא לֹא חָשׁוּ רַבָּנָן לִפְסֵידָא דְּטָהֳרוֹת, קָא מַשְׁמַע לָן:

{ז} דִּסְפֵק טֻמְאָה הַבָּאָה בִּידֵי אָדָם, נִשְׁאָלִין עָלֶיהָ אֲפִלּוּ בִּכְלִי מֻנָּח עַל גַּבֵּי קַרְקַע. גְּמָרָא:

{ח} דְּאִי לָאו הָכִי, אֵין לְמֵדִין מִמַּה שֶּׁנֶּאֱמַר הֲלָכָה בַּמִּשְׁנָה. גְּמָרָא. שֶׁהַתַּנָּאִים לֹא דִּקְדְּקוּ אִישׁ בְּדִבְרֵי חֲבֵרוֹ כוּ'. רַשִׁ"י:

{ט} לְרַבָּנָן הוּא דִּמְפָרֵשׁ הָכִי: {י} כְּדִכְתִיב מִי יִתֵּן טָהוֹר מִטָּמֵא (חָלָב מִדָּם) לֹא אֶחָד [אִיּוֹב יד], בִּתְמִיָּה, כְּלוֹמַר יְחִידוֹ שֶׁל עוֹלָם. וְרַבָּנָן, אָמַר רַבִּי יוֹחָנָן זוֹ שִׁכְבַת זֶרַע. וְרַבִּי אֶלְעָזָר אוֹמֵר אֵלּוּ מֵי הַנִּדָּה, שֶׁהַמַּזֶּה וּמַזִּין עָלָיו טָהוֹר, וְנוֹגֵעַ טָמֵא. גְּמָרָא:

{יא} בַּיְרוּשַׁלְמִי פָּרֵיךְ, וְכִי בְּדַעְתָּהּ תָּלוּי הַדָּבָר. וּמְשָׁנֵי, כָּל שֶׁרְאוּיָה לִקְרוֹתָהּ אִימָא. תּוֹסָפוֹת: {יב} אַף עַל פִּי שֶׁלֹּא פָּסְקָה מִלִּרְאוֹת בְּאֵלּוּ שְׁלֹשָׁה חֳדָשִׁים:

{יג} <B>חוּץ כוּ'.</B> וְקָא מַשְׁמַע לָן דְּאֵין אִשָּׁה קוֹבַעַת וֶסֶת בִּימֵי נִדָּה וִימֵי טֹהַר. גְּמָרָא: {יד} וַאֲפִלּוּ אֵין לָהּ וֶסֶת, לְפֵרוּשׁ רַשִׁ"י: {טו} וְלִפְנֵי תַּשְׁמִישׁ לֹא בָּעִינַן בְּדִיקָה, כֵּיוָן דִּלְאַחַר תַּשְׁמִישׁ לֹא בָּעֵי בְּדִיקָה, לִפְנֵי תַּשְׁמִישׁ נַמִּי לֹא בָּעֵי, דִּלְבַעְלָהּ לֹא בָּעֵי בְּדִיקָה אֶלָּא מִגּוֹ דִּלְטָהֳרוֹת, וְהָכָא לֵיכָּא מִגּוֹ. רַשִׁ"י:

{א} כֵּן פֵּרֵשׁ רַשִׁ"י בַּמִּשְׁנָה לְפִי הַסָּלְקָא דַעְתָּךְ, כְּדַרְכּוֹ תָּמִיד. אֲבָל הָרַ"ב הֲוָה לֵיהּ לְפָרֵשׁ כְּמַסְקָנַת הַגְּמָרָא, דְּמַרְבָּה, אַנָּשִׁים קָאֵי. אֲבָל אֲנָשִׁים אֲפִלּוּ חֲדָא זִמְנָא נַמִּי לֹא, שֶׁמָּא יוֹצִיא שִׁכְבַת זֶרַע לְבַטָּלָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} לַאֲפוֹקֵי לְעִנְיַן זוֹב, דְּאַף הוּא מְשֻׁבָּח כַּנָּשִׁים. גְּמָרָא. כְּדֵי לִמְנוֹת רְאִיּוֹתָיו שְׁתַּיִם לַטֻּמְאָה, וְשָׁלֹשׁ לַקָּרְבָּן. רַשִׁ"י: {ג} וּכְאִלּוּ מֵבִיא מַבּוּל לָעוֹלָם. וְאִיכָּא מַאן דְּאָמַר דְּחַיָּב מִיתָה בִּידֵי שָׁמַיִם. וּכְאִלּוּ שׁוֹפֵךְ דָּמִים, וּכְעוֹבֵד עֲבוֹדָה זָרָה. כִּדְאִיתָא בַּגְּמָרָא: {ד} וְזֶהוּ דְּקָא מַשְׁמַע לָן הָכָא. וְכֵן נַמִּי הָא דְּאַף לוֹ מְתַקֶּנֶת עֵד. וְהַךְ דְּהָכָא וְדִלְעֵיל מַיְרֵי לְטָהֳרוֹת, אֲבָל לְבַעְלָהּ אֵין צָרִיךְ בְּדִיקָה כְּלָל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ה} <B> קָרְבָּן.</B> וְאֵין לָחֹשׁ לְדַם מַאֲכֹלֶת, דְּאוֹתוֹ מָקוֹם בָּדוּק אִי נַמִּי דָּחוּק אֵצֶל מַאֲכֹלֶת. גְּמָרָא:

{ו} <B> שֶׁתֵּרֵד כוּ' וְתָדִיחַ כוּ'.</B> בָּעֵד שֶׁבְּיָדָהּ. וְאִי אֵין עֵד בְּיָדָהּ, כְּדֵי שֶׁתּוֹשִׁיט יָדָהּ לְתַחַת הַכַּר אוֹ הַכֶּסֶת וְתִטֹּל עֵד וְתִבְדֹּק בּוֹ. גְּמָרָא: {ז} מִדְּרַבָּנָן. כְּדִין מֵעֵת לְעֵת דִּמְטַמְּאָה מִשְׁכָּב כוּ', כָּל שֶׁכֵּן הִיא עַצְמָהּ. רַשִׁ"י: {ח} <B> שֶׁמְּטַמְּאָה כוּ'.</B> פֵּרוּשׁ, מִכָּאן וּלְהַבָּא. וְטַעֲמָא, הָתָם אֵין שׁוֹר שָׁחוּט לְפָנֶיךָ, הָכָא שׁוֹר שָׁחוּט כוּ' [דְּאֵין לָהּ בַּמֶּה לִתְלוֹת וּמִיּוֹם שֶׁלְּבָשַׁתּוֹ הִיא בְּסָפֵק זֶה. רַשִׁ"י]. גְּמָרָא: 

~ פרק ב - משנה ד

{ט} <B> בְּחֶזְקַת כוּ'.</B> כְּשֶׁלֹּא הִגִּיעַ שְׁעַת וֶסְתָּהּ. וְאִם אֵין לָהּ וֶסֶת, שֶׁלֹּא עָבְרוּ שְׁלֹשִׁים יוֹם לִרְאִיָּתָהּ. דִּוְסָתוֹת דְּאוֹרַיְתָא. וּלְמַאן דְּאָמַר וְסָתוֹת דְּרַבָּנָן, אֲפִלּוּ הִגִּיעַ אֶלָּא שֶׁשָּׁהֲתָה שִׁעוּר שֶׁתִּטְבֹּל נַמִּי שַׁרְיָא, דְּכֵיוָן דִּוְסָתוֹת דְּרַבָּנָן, סָפֵק טְבִילָה מוֹצִיאָה מִידֵי סְפֵק רְאִיָּה. וְכֵן הִיא מַסְקָנַת הַפּוֹסְקִים. אֶלָּא דְּהָרַ"ב לֹא פֵּרֵשׁ כֵּן בְּסוֹף פֶּרֶק ד': {י} <B> עַל כָּל כוּ'.</B> טַעֲמָא מְפָרֵשׁ בַּגְּמָרָא, דְּחָיְשִׁינַן שֶׁמָּא תִּרְאֶה טִפַּת דָּם וְיַעֲמִידוּהוּ כָּתְלֵי בֵּית הָרֶחֶם [לְפִי שָׁעָה] וּתְחַפֶּנּוּ שִׁכְבַת זֶרַע בְּבִיאָה [שְׁנִיָּה], וְכִי בָּדְקָה לְאַחַר תַּשְׁמִישׁ אַחֲרוֹן לֹא מִנְכַּר. וְאַף עַל גַּב דְּהַשְׁתָּא נַמִּי לֹא יָדְעִינַן עַד לְמָחָר, מִיהוּ בָּעִינַן מוֹכִיחָהּ קַיָּם, דְּאִי לָאו הָכִי, מַה הוֹעִילוּ חֲכָמִים בְּתַקָּנָתָן. רַשִׁ"י: {יא} וְקָשֶׁה, דְּכִי בָּדְקָה לְאוֹר הַנֵּר תִּסְגֵּי. וּבַגְּמָרָא מַסִּיק, דְּהַמְשַׁמֵּשׁ לְאוֹר הַנֵּר הֲרֵי זֶה מְגֻנֶּה. אֶלָּא אֵימָא, אוֹ תִּבְדֹּק: {יב} וְהַתּוֹסָפוֹת הִקְשׁוּ עַל פֵּרוּשׁ זֶה. וּמַסְּקֵי דִּלְבֵית הִלֵּל נַמִּי בָּדְקִי אַחַר כָּל תַּשְׁמִישׁ, אֶלָּא דְּלֹא מַצְרְכֵי חָדָשׁ וַאֲפִלּוּ לַצְּנוּעוֹת מִשּׁוּם דְּאָז לֹא נַפְקָא מִנַּהּ בְּחָדָשׁ, דְּאִיכָּא רֹב לִכְלוּךְ מִשִּׁכְבַת זֶרַע. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{י} וְלֹא תָּנֵי סְתָמָא אֵינָהּ נוֹגַעַת, לְאַשְׁמְעִינַן דִּמְעָרָה וְלֹא חָיְשִׁינַן שֶׁמָּא נָגְעָה בַּמַּיִם. תּוֹסְפוֹת יוֹם טוֹב: {יא} נָקַט הַיָּמִים הַמְרֻבִּים. וְהוּא הַדִּין לִמְעֻטִּים דְּשֶׁל זָכָר: {יב} <B> טְמֵא מֵת.</B> שֶׁלֹּא טָבַל. הָרַמְבַּ"ם. וְלֹא יָדַעְתִּי לָמָּה לֹא תָּלִינַן בְּמַגַּע נִדָּה דַּהֲוֵי נַמִּי רִאשׁוֹן. וְהָרַמְבַּ"ם כָּתַב, כְּמוֹ שֶׁנָּגַע בְּנִדָּה אוֹ בִּטְמֵא מֵת: {יג} וְחָזְרוּ בִּשְׁתַּיִם, חֲדָא לְחֻמְרָא וַחֲדָא לְקֻלָּא: {יד} <B> לַקָּדָשִׁים.</B> וּכְאַבָּא שָׁאוּל דְּרֵישׁ פֶּרֶק ב' דִּמְעִילָה. תּוֹסָפוֹת. וְצָרִיךְ עִיּוּן עַל הָרַמְבַּ"ם שֶׁפָּסַק כִּסְתָמָא דְּהָכָא וּכְתַנָּא קַמָּא דְּהָתָם:

{טו} <B>וּמַקֶּפֶת וְקוֹרְאָה כוּ'. </B>סָלְקָא דַעְתָּךְ אֲמֵינָא נִגְזֹר דִּלְמָא אַתְיָא לְמִנְגַּע בָּהּ מֵאַבְּרַאי, קָא מַשְׁמַע לָן. גְּמָרָא: {טז} בְּפֶרֶק ג' דַּחֲגִיגָה מִשְׁנָה ג'. וְהָא דְּכָתַב שָׁם הָרַ"ב דִּמְחֻסַּר כִּפּוּרִים פּוֹסֵל בִּנְגִיעָה, הַיְנוּ קֹדֶם הֲבָאַת קָרְבְּנוֹתָיו. וְלֹא קָאֵי אָרֵישׁ דְּבָרָיו דְּמַיְרֵי לְאַחַר הֲבָאַת קָרְבְּנוֹתָיו דְּאָז אֵין צָרִיךְ טְבִילָה אֶלָּא לַאֲכִילָה, וּכְמוֹ שֶׁכָּתַב כָּאן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק י - משנה ח

{יז} <B> מְטַמְּאִין כוּ'.</B> הוֹאִיל וּכְזָבָה קְטַנָּה הִיא. וְיָלְפִינַן לַהּ מִקְּרָא בַּגְּמָרָא. וְכֵן הָא דְּבוֹעֵל מְטַמֵּא, יָלְפִינַן לַהּ מִקְּרָא. וּבַתּוֹסֶפְתָּא הוּבָא בַּגְּמָרָא [דַּף עב] דִּלְבוֹעֲלָהּ תּוֹלִין, וְאִם תִּרְאֶה לְמָחָר מְטַמֵּא לְמַפְרֵעַ. וְאִם כֵּן, קְרָא אַסְמַכְתָּא בְּעָלְמָא: {יח} וְכָל הֵיכָא דְּבָעֵי שִׁמּוּר, אֵינָהּ טוֹבֶלֶת אֶלָּא בַּיּוֹם: {יט} <B>בְּתוֹךְ אַחַד עָשָׂר.</B> וַאֲפִלּוּ בָּעֲשִׂירִי, אַף עַל גַּב דְּאֵינָהּ רְאוּיָה לָבֹא לִידֵי זָבָה גְּמוּרָה. הָרַמְבַּ"ם:
