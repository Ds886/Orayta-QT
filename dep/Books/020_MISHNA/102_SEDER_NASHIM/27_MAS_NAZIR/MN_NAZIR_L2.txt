~ פרק א - משנה א
<b>כָּל כִּנּוּיֵי נְזִירוּת.</b> דָּבָר שֶׁאֵינוֹ עִקָּרוֹ שֶׁל שֵׁם מִקְּרֵי כִּנּוּי. כְּמוֹ הַמְּכַנֶּה שֵׁם לַחֲבֵרוֹ [בָּבָא מְצִיעָא נח:]: <B>הָאוֹמֵר אֱהֵא.</B> לָאו כִּנּוּי הוּא, אֶלָּא יָד מִקְּרֵי, כְּמוֹ בֵּית יָד שֶׁהַכְּלִי נֶאֱחַז בּוֹ כָּךְ הַנֶּדֶר נִתְפָּס בְּלָשׁוֹן זֶה. וּמַתְנִיתִין חֲסוּרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, כָּל כִּנּוּיֵי נְזִירוּת כִּנְזִירוּת וְכָל יְדוֹת נְזִירוּת כִּנְזִירוּת {א}, אֵלּוּ הֵן יְדוֹת נְזִירוּת, הָאוֹמֵר אֱהֵא, אֱהֵא נָוֶה, וְאֵלּוּ הֵן כִּנּוּיֵי נְזִירוּת, נָזִיק נָזִיחַ פָּזִיחַ: <B>הָאוֹמֵר אֱהֵא.</B> שֶׁרָאָה נָזִיר עוֹבֵר לְפָנָיו וְאָמַר אֱהֵא, אַף עַל פִּי שֶׁלֹּא אָמַר אֱהֵא כָּזֶה, אִם נִתְכַּוֵּן {ב} לִהְיוֹת נָזִיר כְּמוֹתוֹ הֲרֵי זֶה נָזִיר, וְאַף עַל פִּי שֶׁלֹּא הוֹצִיא בְּפִיו כָּזֶה: <B>אֱהֵא נָוֶה.</B> שֶׁהָיָה תּוֹפֵס בִּשְׂעָרוֹ {ג} וְאָמַר אֱהֵא נָוֶה, מַשְׁמַע אֶהֱיֶה נָאֶה בְּגִדּוּל שֵׂעָר זֶה. וְאִם נִתְכַּוֵּן לְשֵׁם כָּךְ הֲרֵי זֶה נָזִיר וְאַף עַל פִּי שֶׁלֹּא פֵּרֵשׁ, שֶׁאֵלּוּ וְכַיּוֹצֵא בָּהֶם יָדוֹת לִנְזִירוּת הֵם וַהֲווּ כִּנְזִירוּת: <B>נָזִיק נָזִיחַ פָּזִיחַ.</B> לְשׁוֹנוֹת שֶׁל גּוֹיִים הֵם שֶׁקּוֹרִין כָּךְ לַנָּזִיר {ד}, וּלְשׁוֹנָם קָרוֹב לַלָּשׁוֹן שֶׁל יִשְׂרָאֵל, וּמְקוֹרוֹ כִּנּוּיֵי נְזִירוּת: <B>הֲרֵינִי כָּזֶה.</B> וְרוֹמֵז לַנָּזִיר שֶׁכְּנֶגְדּוֹ {ה}: <B>הֲרֵינִי מְסַלְסֵל הֲרֵינִי מְכַלְכֵּל הֲרֵי עָלַי לְשַׁלַּח פֶּרַע.</B> וְכֻלְּהוּ דַּוְקָא כְּשֶׁתּוֹפֵס בִּשְׂעָרוֹ {ז} וְדַעְתּוֹ לִנְזִירוּת: <B>מְסַלְסֵל.</B> מַחְלִיק בַּשֵּׂעָר: <B>מְכַלְכֵּל.</B> מְגַדֵּל שֵׂעָר. וְדֻגְמָתוֹ, מִשֶּׁתְּכַלְכֵּל הָעֲטָרָה [נִדָּה נב:], מִשֶּׁיִּגְדַּל שְׂעַר הָעֶרְוָה: <B>הֲרֵי עָלַי צִפֳּרִים.</B> שְׁתֵּי תוֹרִים אוֹ שְׁנֵי בְנֵי יוֹנָה, וְהֵם קָרְבַּן נָזִיר שֶׁנִּטְמָא. וּכְגוֹן שֶׁנָּזִיר עוֹבֵר לְפָנָיו: <B>רַבִּי מֵאִיר אוֹמֵר נָזִיר.</B> הוֹאִיל וְנָזִיר עוֹבֵר לְפָנָיו וְצִפֳּרִים קָרְבַּן נָזִיר שֶׁנִּטְמָא הֵם, מוֹכִיחִין הַדְּבָרִים דַּהֲרֵי עָלַי צִפֳּרִים לִנְזִירוּת קָאָמַר: <B>וַחֲכָמִים אוֹמְרִים אֵינוֹ נָזִיר.</B> וַהֲלָכָה כַּחֲכָמִים, שֶׁאֵינוֹ נָזִיר, אֲבָל מֵבִיא צִפֳּרִים לְשַׁלֵּם נִדְרוֹ:

<B>הֲרֵינִי נָזִיר מִן הַחַרְצַנִּים וּמִן הַזַּגִּים וְכוּ'.</B> אִם הִזְכִּיר אֶחָד מִכָּל אֵלּוּ, הֲרֵי זֶה נָזִיר כְּאִלּוּ אָמַר הֲרֵינִי נָזִיר סְתָם. וּמִשּׁוּם דְּבָעֵי לְמִתְנָא סֵיפָא דְּנָזִיר עוֹלָם וּנְזִיר שִׁמְשׁוֹן אֵין כָּל דִּקְדּוּקֵי נְזִירוּת עֲלֵיהֶם, תָּנָא הָכָא כָּל דִּקְדּוּקֵי נְזִירוּת עָלָיו: <B>מַה בֵּין נְזִיר עוֹלָם וְכוּ'.</B> מַתְנִיתִין חִסּוּרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, וְאִם נָדַר לִהְיוֹת נְזִיר עוֹלָם הֲרֵי זֶה נְזִיר עוֹלָם. וּמַה בֵּין נְזִיר עוֹלָם לִנְזִיר שִׁמְשׁוֹן, נְזִיר עוֹלָם הִכְבִּיד שְׂעָרוֹ מֵקֵל בְּתַעַר מִשְּׁנֵים עָשָׂר חֹדֶשׁ לִשְׁנֵים עָשָׂר חֹדֶשׁ דְּיָלְפִינַן מֵאַבְשָׁלוֹם שֶׁהָיָה נְזִיר עוֹלָם וּכְתִיב בֵּיהּ (שְׁמוּאֵל ב יד) וַיְהִי מִקֵּץ יָמִים לַיָּמִים אֲשֶׁר יְגַלֵּחַ כִּי כָבֵד עָלָיו וְגִלְּחוֹ, וּכְתִיב הָתָם (וַיִּקְרָא כה) יָמִים תִּהְיֶה גְּאֻלָּתוֹ: <B>וְאִם נִטְמָא אֵינוֹ מֵבִיא קָרְבַּן טֻמְאָה.</B> וַאֲפִלּוּ לְכַתְּחִלָּה מֻתָּר לִטָּמֵא, דְּשִׁמְשׁוֹן הָיָה מִטַּמֵּא לְמֵתִים, וְהָכִי גְמִירֵי לַהּ, וְהַאי דְקָתָנֵי אִם נִטְמָא דְּמַשְׁמַע דִּיעֲבַד אִין לְכַתְּחִלָּה לֹא, מִשּׁוּם דְּתָנָא רֵישָׁא גַּבֵּי נְזִיר עוֹלָם וְאִם נִטְמָא תָּנָא סֵיפָא נַמִּי גַּבֵּי נְזִיר שִׁמְשׁוֹן וְאִם נִטְמָא: 

~ פרק א - משנה ג
<B>סְתָם נְזִירוּת שְׁלֹשִׁים יוֹם.</B> דִּכְתִיב (בְּמִדְבָּר ו) קָדוֹשׁ יִהְיֶה. יִהְיֶ"ה בְּגִימַטְרִיָּא שְׁלֹשִׁים, מִכָּאן סָמְכוּ {יד} לוֹמַר שֶׁאֵין נְזִירוּת פָּחוֹת מִשְּׁלֹשִׁים יוֹם: <B>הֲרֵינִי נָזִיר אַחַת גְּדוֹלָה.</B> בֵּין שֶׁאָמַר הֲרֵינִי נָזִיר נְזִירוּת אַחַת גְּדוֹלָה, בֵּין שֶׁאָמַר הֲרֵינִי נָזִיר אַחַת קְטַנָּה, וּבֵין שֶׁאָמַר הֲרֵינִי נָזִיר אַחַת מִכָּאן וְעַד סוֹף הָעוֹלָם, נוֹהֵג נְזִירוּת שְׁלֹשִׁים יוֹם. וּמִכָּאן וְעַד סוֹף הָעוֹלָם דְּקָאָמַר הָכִי מַשְׁמַע, אֲרֻכָּה עָלַי זֹאת הַנְּזִירוּת כְּאִלּוּ הִיא מִכָּאן וְעַד סוֹף הָעוֹלָם {טו}: הָכִי גָּרְסִינַן הֲרֵינִי נָזִיר וְיוֹם אֶחָד הֲרֵינִי נָזִיר וְשָׁעָה אֶחָת הֲרֵינִי נָזִיר אַחַת וּמֶחֱצָה הֲרֵי זֶה נָזִיר שְׁתָּיִם. וְהָכִי פֵּרוּשׁוֹ, אָמַר הֲרֵינִי נָזִיר וְיוֹם אֶחָד כֵּיוָן שֶׁאָמַר הֲרֵינִי נָזִיר קִבֵּל עָלָיו נְזִירוּת אַחַת, וְכִי אָמַר תּוּ וְיוֹם אֶחָד הֲרֵי יֵשׁ כָּאן נְזִירוּת אַחֶרֶת, דְּאֵין נְזִירוּת פָּחוֹת מִשְּׁלֹשִׁים יוֹם. וְכֵן נַמִּי כִּי אָמַר הֲרֵינִי נָזִיר וְשָׁעָה אַחַת וַהֲרֵינִי נָזִיר אַחַת וּמֶחֱצָה, יֵשׁ כָּאן שְׁתֵּי נְזִירוּת, שֶׁאִי אֶפְשָׁר לִהְיוֹת נָזִיר שָׁעָה אַחַת אוֹ מֶחֱצָה נְזִירוּת שֶׁלֹּא יִהְיֶה נָזִיר שְׁלֹשִׁים יוֹם, וּמִשּׁוּם הָכִי הֲרֵי זֶה נָזִיר שְׁתַּיִם וּמְגַלֵּחַ לְסוֹף שְׁלֹשִׁים וְנוֹהֵג נְזִירוּת אַחֶרֶת: הֲרֵינִי נָזִיר שְׁלֹשִׁים יוֹם וְשָׁעָה אֶחָת, הֲרֵי זֶה נָזִיר שְׁלֹשִׁים וְאֶחָד יוֹם, שֶׁאֵין נְזִירוּת לְשָׁעוֹת. שֶׁנֶּאֱמַר (שָׁם) יְמֵי נִזְרוֹ, וַהֲרֵי הוּא כְּאִלּוּ אָמַר שְׁלֹשִׁים וְאֶחָד יוֹם {יט}. וְלֹא אָמְרִינַן דְּשָׁעָה אַחַת דְּאָמַר הַוְיָא נְזִירוּת בִּפְנֵי עַצְמָה, הוֹאִיל וְיֵשׁ לְצָרְפָהּ עִם מִנְיַן שְׁלֹשִׁים שֶׁהִזְכִּיר תְּחִלָּה:

<B>הֲרֵינִי נָזִיר כִּשְׂעַר רֹאשִׁי וְכַעֲפַר הָאָרֶץ וְכוּ' הֲרֵי זֶה נְזִיר עוֹלָם.</B> וְלֹא נְזִיר עוֹלָם מַמָּשׁ, דְּאִלּוּ נְזִיר עוֹלָם מְגַלֵּחַ אֶחָד לִשְׁנֵים עָשָׂר חֹדֶשׁ וְזֶה מְגַלֵּחַ אַחַת לִשְׁלֹשִׁים יוֹם, הוֹאִיל וְתָלָה נְזִירוּתוֹ בִּדְבָרִים חֲלוּקִים {כא}, הָוֵי כְּאִלּוּ נָדַר בְּמִנְיַן נְזִירִיּוֹתָיו כְּמִנְיַן שְׂעַר רֹאשׁוֹ וַעֲפַר הָאָרֶץ, מַה שֶּׁאֵין כֵּן בִּנְזִיר עוֹלָם שֶׁלֹּא חָלַק נְזִירִיּוֹת שֶׁלּוֹ אֶלָּא נַעֲשָׂה הַכֹּל נְזִירוּת אַחַת: <B>רַבִּי אוֹמֵר אֵין זֶה מְגַלֵּחַ אַחַת לִשְׁלֹשִׁים יוֹם.</B> דְּכֵיוָן דְּאָמַר הֲרֵינִי נָזִיר, הַכֹּל נְזִירוּת אַחַת הִיא וְאֵינוֹ מְגַלֵּחַ אֶלָּא לִשְׁנֵים עָשָׂר חֹדֶשׁ כִּנְזִיר עוֹלָם. וְאֵיזֶהוּ מְגַלֵּחַ לִשְׁלֹשִׁים יוֹם, הָאוֹמֵר הֲרֵי עָלַי נְזִירוּת כִּשְׂעַר רֹאשִׁי, דְּאָז מוֹכִיחַ דִּנְזִירִיּוֹת מֻפְסָקוֹת קִבֵּל עָלָיו כְּמִנְיַן שְׂעַר רֹאשׁוֹ. וְאֵין הֲלָכָה כְּרַבִּי:

<B>מוֹנֶה נְזִירוּת כְּמִנְיַן יְמוֹת הַשָּׁנָה.</B> שְׁלֹשׁ מֵאוֹת שִׁשִּׁים וְחָמֵשׁ נְזִירוּת כְּמִנְיַן יְמוֹת הַחַמָּה: <B>אָמַר רַבִּי יְהוּדָה מַעֲשֶׂה הָיָה כֵּיוָן שֶׁהִשְׁלִים מֵת.</B> רַבִּי יְהוּדָה שַׁמְעֵיהּ לְרַבִּי דְּאַפְלִיג אַתַּנָּא קַמָּא וְאָמַר שֶׁהָאוֹמֵר הֲרֵינִי נָזִיר כְּמִנְיַן יְמוֹת הַחַמָּה הָוֵי נְזִיר עוֹלָם. וְהֵבִיא מַעֲשֶׂה לְהוֹכִיחַ שֶׁאֵינוֹ נְזִיר עוֹלָם {כז}, דְּמַעֲשֶׂה הָיָה וְכֵיוָן שֶׁהִשְׁלִים מֵת, וּבִנְזִיר עוֹלָם לֹא שַׁיָּךְ הַשְׁלָמָה, אֶלָּא וַדַּאי מוֹנֶה נְזִירִיּוֹת. וְכֵן הֲלָכָה:

<B>הֲרֵינִי נָזִיר מִן הַגְּרוֹגָרוֹת וּמִן הַדְּבֵלָה בֵּית שַׁמַּאי אוֹמְרִים נָזִיר.</B> דִּסְבִירָא לְהוּ לְבֵית שַׁמַּאי אֵין אָדָם מוֹצִיא דְבָרָיו לְבַטָּלָה, וְכִי אָמַר הֲרֵינִי נָזִיר אַדַּעְתָּא דְלֵהֲוֵי נָזִיר קָאָמַר, וְכִי הָדַר אָמַר מִן הַגְּרוֹגָרוֹת וּמִן הַדְּבֵלָה, מַהֲדַר הוּא דְּבָעֵי הֲדַר בֵּיהּ, וַאֲפִלּוּ בְּתוֹךְ כְּדֵי דִּבּוּר לֹא מָצֵי הֲדַר בֵּיהּ, דִּסְבִירָא לְהוּ הֶקְדֵּשׁ בְּטָעוּת שְׁמֵיהּ הֶקְדֵּשׁ וְלֹא שַׁיָּךְ בֵּיהּ שְׁאֵלָה וְלֹא חֲזָרָה, וְהוּא הַדִּין בְּנָזִיר דִּכְתִיב בֵּיהּ (בְּמִדְבָּר ו) קָדוֹשׁ יִהְיֶה, הִלְכָּךְ הָוֵי נָזִיר. וּבֵית הִלֵּל סָבְרֵי הוֹאִיל וְלֹא נָדַר כְּדֶרֶךְ הַנּוֹדְרִים {א} לֹא הָוֵי נָזִיר, שֶׁאֵין נְזִירוּת מִן הַגְּרוֹגָרוֹת וּמִן הַדְּבֵלָה: אָמַר רַבִּי יְהוּדָה אַף כְּשֶׁאָמְרוּ בֵית שַׁמַּאי לֹא אָמְרוּ אֶלָּא בְאוֹמֵר הֲרֵי הֵן עָלַי קָרְבָּן. דְּלֹא נֶחְלְקוּ בֵּית שַׁמַּאי וּבֵית הִלֵּל לְעִנְיַן נְזִירוּת דְּלֹא הָוֵי נָזִיר. לֹא נֶחְלְקוּ אֶלָּא בְּאוֹמֵר בְּלִבִּי הָיָה שֶׁיִּהְיוּ הַגְּרוֹגָרוֹת עָלַי קָרְבָּן, בֵּית שַׁמַּאי סָבְרֵי דַּהֲוֵי נָדוּר {ב} מִן הַגְּרוֹגָרוֹת, וּבֵית הִלֵּל סָבְרֵי דְּלֹא הָוֵי נָדוּר {ג}:

<B>וְאִם גִּלַּח יוֹם שִׁשִּׁים חָסֵר אֶחָד יָצָא.</B> שֶׁיּוֹם שְׁלֹשִׁים שֶׁל נְזִירוּת רִאשׁוֹנָה עוֹלָה לְכָאן וּלְכָאן, וְכֵיוָן שֶׁיּוֹם שְׁלֹשִׁים שֶׁל הָרִאשׁוֹנָה נִמְנֶה גַּם מִן הַנְּזִירוּת הַשְּׁנִיָּה נִמְצְאוּ שְׁלֹשִׁים יוֹם שֶׁל נְזִירוּת שְׁנִיָּה כָּלִים בְּשִׁשִּׁים חָסֵר אֶחָד:

<B>נִטְמָא יוֹם שְׁלֹשִׁים סוֹתֵר אֶת הַכֹּל.</B> שֶׁהֲרֵי נִטְמָא בְּתוֹךְ יְמֵי הַנְּזִירוּת. וְלֹא אָמְרִינַן דְּיוֹם שְׁלֹשִׁים עוֹלֶה לְכָאן וּלְכָאן דְּמִקְצַת הַיּוֹם כְּכֻלּוֹ אֶלָּא כְּשֶׁגִּלַּח וְהֵבִיא קָרְבְּנוֹתָיו בּוֹ בַיּוֹם: <B>רַבִּי אֱלִיעֶזֶר אוֹמֵר אֵינוֹ סוֹתֵר אֶלָּא שִׁבְעָה.</B> דִּסְבִירָא לֵיהּ אָמְרִינַן מִקְצַת הַיּוֹם כְּכֻלּוֹ וּכְאִלּוּ נִטְמָא לְאַחַר מְלֹאת, וְטֻמְאָה דִּלְאַחַר מְלֹאת לְרַבִּי אֱלִיעֶזֶר אֵינָהּ סוֹתֶרֶת אֶלָּא שִׁבְעָה: <B>הֲרֵינִי נָזִיר שְׁלֹשִׁים יוֹם נִטְמָא יוֹם שְׁלֹשִׁים סוֹתֵר אֶת הַכֹּל.</B> בֵּין לְרַבִּי אֱלִיעֶזֶר בֵּין לְרַבָּנָן. דְּכֵיוָן דְּאָמַר הֲרֵינִי נָזִיר שְׁלֹשִׁים יוֹם, הַכֹּל מוֹדִים דִּשְׁלֹשִׁים יוֹם שְׁלֵמִים בָּעִינַן, וְלֹא אָמְרִינַן בְּהָא מִקְצַת הַיּוֹם כְּכֻלּוֹ. וְרַבִּי אֱלִיעֶזֶר דִּסְבִירָא לֵיהּ שֶׁהַנִּטְמָא בְּיוֹם מְלֹאת מַמָּשׁ סוֹתֵר שְׁלֹשִׁים בִּלְבַד וְאֵינוֹ סוֹתֵר הַכֹּל, יָלֵיף לָהּ מִקְּרָא דִכְתִיב (בְּמִדְבָּר ו) וְזֹאת תּוֹרַת הַנָּזִיר בְּיוֹם מְלֹאת, נִטְמָא בְּיוֹם מְלֹאת תֵּן לוֹ תּוֹרַת הַנָּזִיר, כְּלוֹמַר סְתָם נְזִירוּת שְׁלֹשִׁים יוֹם, הָכָא שֶׁאָמַר הֲרֵינִי נָזִיר שְׁלֹשִׁים יוֹם וְנִטְמָא בְּיוֹם שְׁלֹשִׁים שֶׁהוּא יוֹם מְלֹאת, לְכִי סוֹתֵר כָּל הַשְּׁלֹשִׁים, סוֹתֵר אֶת הַכֹּל. וְרַבָּנָן נַמִּי דִפְלִיגֵי עֲלֵיהּ דְּרַבִּי אֱלִיעֶזֶר וְאָמְרֵי שֶׁהַנִּטְמָא בְּיוֹם מְלֹאת סוֹתֵר אֶת הַכֹּל וַאֲפִלּוּ מָנָה כַּמָּה יָמִים, הָכָא כִּי אָמַר הֲרֵינִי נָזִיר שְׁלֹשִׁים יוֹם וְנִטְמָא בְּיוֹם מְלֹאת, לְכִי נַמִּי סוֹתֵר שְׁלֹשִׁים סוֹתֵר אֶת הַכֹּל, הִלְכָּךְ בֵּין לְרַבִּי אֱלִיעֶזֶר בֵּין לְרַבָּנָן סוֹתֵר אֶת הַכֹּל. וּבְכָל הָא דְּאַפְלִיגוּ רַבִּי אֱלִיעֶזֶר וַחֲכָמִים הֲלָכָה כַּחֲכָמִים: 

~ פרק ג - משנה ד
<B>הֲרֵינִי נָזִיר מֵאָה יוֹם נִטְמָא יוֹם מֵאָה סוֹתֵר אֶת הַכֹּל.</B> רַבָּנָן לְטַעֲמַיְהוּ דְאָמְרֵי נִטְמָא יוֹם מְלֹאת כְּאִלּוּ נִטְמָא בְּתוֹךְ זְמַנּוֹ וְסוֹתֵר אֶת הַכֹּל: <B>רַבִּי אֱלִיעֶזֶר אוֹמֵר לֹא סוֹתֵר אֶלָּא שְׁלֹשִׁים.</B> דִּסְבִירָא לֵיהּ נִטְמָא בְּיוֹם מְלֹאת אֵינוֹ סוֹתֵר אֶלָּא שְׁלֹשִׁים {ד}: <B>נִטְמָא יוֹם מֵאָה וְאֶחָד.</B> לְרַבָּנָן סוֹתֵר שְׁלֹשִׁים, דְּגָזְרוּ יוֹם מֵאָה וְאֶחָד שֶׁהוּא יוֹם תִּגְלַחַת אַטּוּ יוֹם מֵאָה. וּמִכָּל מָקוֹם לֹא הֶחְמִירוּ {ה} עָלָיו לַעֲשׂוֹתוֹ כְּיוֹם מֵאָה שֶׁהוּא יוֹם מְלֹאת שֶׁהוּא סוֹתֵר הַכֹּל, וְגָזְרוּ שֶׁיִּסְתֹּר סְתָם נְזִירוּת שֶׁהוּא שְׁלֹשִׁים יוֹם בִּלְבַד: <B>וְרַבִּי אֱלִיעֶזֶר אוֹמֵר אֵינוֹ סוֹתֵר אֶלָּא שִׁבְעָה.</B> וְאַזְדָא לְטַעְמֵיהּ דַּאֲפִלּוּ בְּיוֹם שְׁלֹשִׁים לֹא גָּזַר בְּאוֹמֵר הֲרֵינִי נָזִיר סְתָם {ו}: 

~ פרק ג - משנה ה
<B>וְאֵינוֹ מֵבִיא קָרְבַּן טֻמְאָה.</B> דְּכִי כְּתִיב קָרְבַּן טֻמְאָה, בְּנָזִיר טָהוֹר שֶׁנִּטְמָא הוּא דִכְתִיב. וּמִכָּל מָקוֹם אִם הִתְרוּ בּוֹ חַיָּב מַלְקוֹת: <B>יָצָא וְנִכְנַס עוֹלִין לוֹ מִן הַמִּנְיָן וּמֵבִיא קָרְבַּן טֻמְאָה.</B> הָא מַתְנִיתִין מִפָּרְשָׁא בַּגְּמָרָא הָכִי, יָצָא מִבֵּית הַקְּבָרוֹת וְהִזָּה שְׁלִישִׁי וּשְׁבִיעִי וְטָבַל וְטָהַר מִטֻּמְאָתוֹ וְהִתְחִיל לִמְנוֹת יְמֵי נְזִירוּתוֹ, אַף עַל פִּי שֶׁחָזַר וְנִכְנַס אַחַר כָּךְ לְבֵית הַקְּבָרוֹת, עוֹלִין לוֹ מִן הַמִּנְיָן אֵלּוּ הַיָּמִים שֶׁמָּנָה אַחַר שֶׁטָּהַר, הוֹאִיל וְהִפְסִיקָה טָהֳרָה בֵּין הַיָּמִים הָרִאשׁוֹנִים שֶׁנָּזַר וְהוּא בְּבֵית הַקְּבָרוֹת וּבֵין אֵלּוּ הַיָּמִים הָאַחֲרוֹנִים. וְאַף עַל פִּי שֶׁחָזַר וְנִכְנַס לְבֵית הַקְּבָרוֹת, טֻמְאַת בֵּית הַקְּבָרוֹת אֵינָהּ סוֹתֶרֶת מִנְיַן הַיָּמִים שֶׁנִּמְנוּ בְּטָהֳרָה, שֶׁאֵין סוֹתֵר בַּנָּזִיר אֶלָּא שְׁתֵּים עֶשְׂרֵה טֻמְאוֹת הָאֲמוּרוֹת בּוֹ {ט}. וְהַאי דְקָאָמַר תּוּ וּמֵבִיא קָרְבַּן טֻמְאָה, הָכִי קָאָמַר, אִם נִטְמָא שׁוּב בְּאַחַת מִן הַטֻּמְאוֹת שֶׁהַנָּזִיר מְגַלֵּחַ, מֵבִיא קָרְבַּן טֻמְאָה וְסוֹתֵר: <B>רַבִּי אֱלִיעֶזֶר אוֹמֵר לֹא בוֹ בַיּוֹם.</B> כְּלוֹמַר אִם בּוֹ בַיּוֹם שֶׁטָּבַל וְטָהַר בּוֹ בַיּוֹם נִטְמָא בְּאַחַת מִן הַטֻּמְאוֹת שֶׁהַנָּזִיר מְגַלֵּחַ, אֵינוֹ סוֹתֵר אוֹתוֹ הַיּוֹם, דִּכְתִיב וְהַיָּמִים הָרִאשׁוֹנִים יִפְּלוּ, אֵין הַטֻּמְאָה סוֹתֶרֶת עַד שֶׁיִּהְיוּ לוֹ שְׁנֵי יָמִים {י} שֶׁל נְזִירוּת מְנוּיִים. וְהוּא הַדִּין בְּנָזִיר בְּעָלְמָא שֶׁנִּטְמָא בְּיוֹם רִאשׁוֹן שֶׁל מִנְיַן נְזִירוּתוֹ, שֶׁאֵין הַטֻּמְאָה סוֹתֶרֶת אוֹתוֹ הַיּוֹם, אֶלָּא מַשְׁלִים עָלָיו מִנְיָן יְמֵי נְזִירוּתוֹ. וַהֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: 

~ פרק ג - משנה ו
<B>וְהִשְׁלִים אֶת נְזִירוּתוֹ וְאַחַר כָּךְ בָּא לָאָרֶץ.</B> שֶׁאֵין נְזִירוּת נוֹהֶגֶת אֶלָּא בָּאָרֶץ, מִשּׁוּם טֻמְאַת אֶרֶץ הָעַמִּים. וּמִי שֶׁנָּדַר נְזִירוּת בְּחוּצָה לָאָרֶץ מְחַיְּבִין אוֹתוֹ לַעֲלוֹת לְאֶרֶץ יִשְׂרָאֵל וְלִהְיוֹת נוֹהֵג שָׁם נְזִירוּתוֹ {יב}: <B>נָזִיר בַּתְּחִלָּה.</B> צָרִיךְ לִנְהֹג בְּאֶרֶץ יִשְׂרָאֵל כְּמִנְיַן הַיָּמִים שֶׁנָּדַר בִּנְזִירוּת, וְהַיָּמִים שֶׁנָּהַג בִּנְזִירוּת בְּחוּצָה לָאָרֶץ כְּאִלּוּ לֹא נָהַג בָּהֶם נְזִירוּת כְּלָל: <B>אָמַר רַבִּי יְהוּדָה לֹא הָיְתָה נְזִירָה אֶלָּא אַרְבַּע עֶשְׂרֵה שָׁנָה.</B> רַבִּי יְהוּדָה סְבִירָא לֵיהּ כְּרַבִּי אֱלִיעֶזֶר דְּאָמַר לְעֵיל בְּפִרְקִין הַנִּטְמָא בְּיוֹם מְלֹאת אֵינוֹ סוֹתֵר אֶלָּא שְׁלֹשִׁים יוֹם, וּמִשּׁוּם הָכִי קָאָמַר דְּהִילְנִי הַמַּלְכָּה שֶׁנִּטְמְאָה בְּסוֹף אַרְבַּע עֶשְׂרֵה שָׁנָה שֶׁהָיָה בְּיוֹם מְלֹאת לֹא סָתְרָה הַכֹּל וְלֹא הֻצְרְכָה לִמְנוֹת עוֹד שֶׁבַע שָׁנִים אֲחֵרוֹת אֶלָּא שְׁלֹשִׁים יוֹם בִּלְבַד. וּלְפִי שֶׁלֹּא הָיְתָה שָׁנָה שְׁלֵמָה לֹא מְנָאָהּ בַּחֶשְׁבּוֹן, וַהֲרֵי הוּא כְאִלּוּ אָמַר לֹא הָיְתָה נְזִירָה אֶלָּא אַרְבַּע עֶשְׂרֵה שָׁנָה וּשְׁלֹשִׁים יוֹם: 

~ פרק ג - משנה ז
<B>אֵלּוּ מְעִידִים שֶׁנָּדַר שְׁתַּיִם.</B> שְׁתֵּי נְזִירִיּוֹת: <B>וְאֵלּוּ מְעִידִים שֶׁנָּדַר חָמֵשׁ.</B> בְּאוֹתָהּ שָׁעָה שֶׁאַתֶּם אוֹמְרִים שֶׁנָּדַר שְׁתַּיִם, אָנוּ מְעִידִים שֶׁנָּדַר חָמֵשׁ נְזִירוּת. וְהוּא אוֹמֵר שֶׁלֹּא נָדַר כְּלָל {יג}: <B>בֵּית שַׁמַּאי אוֹמְרִים נֶחְלְקָה הָעֵדוּת.</B> הוֹאִיל וּמַכְחִישׁוֹת זוֹ אֶת זוֹ נִתְבַּטְּלוּ דִּבְרֵיהֶם וְאֵין כָּאן עֵדוּת כְּלָל: <B>וּבֵית הִלֵּל אוֹמְרִים יֵשׁ בִּכְלַל חָמֵשׁ שְׁתַּיִם.</B> וִיהֵא נָזִיר שְׁתַּיִם:

~ פרק ד - משנה א
<B>מִי שֶׁאָמַר הֲרֵינִי נָזִיר.</B> כֻּלָּם נְזִירִים. וְהוּא שֶׁהִתְפִּיס כָּל אֶחָד מֵהֶן בְּתוֹךְ דִּבְרֵי חֲבֵרוֹ {ב} {ג} בִּכְדֵי שְׁאֵלַת שָׁלוֹם תַּלְמִיד לָרַב, שֶׁהוּא כְּדֵי שֶׁיֹּאמַר שָׁלוֹם עָלֶיךָ רַבִּי: <B>פִּי כְפִיו כוּ'.</B> וְהוּא שֶׁיֹּאמַר פִּי כְּפִיו מִיַּיִן וּשְׂעָרִי כִּשְׂעָרוֹ מִלְּהִגָּזֵז: <B>וְשָׁמַע בַּעְלָהּ וְאָמַר וָאָנִי, אֵינוֹ יָכוֹל לְהָפֵר.</B> שֶׁכְּבָר קִיֵּם נִדְרָהּ כְּשֶׁאָמַר וַאֲנִי. וְאִם אִשָּׁה נָדְרָה בְּנָזִיר וְשָׁמַע אַחֵר וְאָמַר וַאֲנִי, וְאַחַר כָּךְ הֵפֵר לָהּ בַּעְלָהּ, אֵינוֹ מוּפָר לָזֶה שֶׁאָמַר וַאֲנִי, שֶׁאֵין הַבַּעַל עוֹקֵר הַנֶּדֶר מֵעִקָּרוֹ כְּמוֹ הֶחָכָם: 

~ פרק ד - משנה ב
<B>הֲרֵינִי נָזִיר וָאַתְּ וְאָמְרָה אָמֵן מֵפֵר אֶת שֶׁלָּהּ.</B> וְדַוְקָא כִּי אָמַר לָהּ בִּלְשׁוֹן שְׁאֵלָה, כְּלוֹמַר וָאָתְּ מַה תֹּאמְרִי, תֶהֱוִי נְזִירָה כָּמוֹנִי אִם לָאו, אָז יָכוֹל לְהָפֵר. אֲבָל אִם אָמַר הֲרֵינִי נָזִיר וָאָתְּ, בְּנִחוּתָא, וְעָנְתָה אָמֵן, אֵינוֹ יָכוֹל לְהָפֵר, שֶׁהֲרֵי קִיֵּם לָהּ: 

~ פרק ד - משנה ג
<B>וְהָיְתָה שׁוֹתָה יַיִן וּמִטַּמְּאָה לְמֵתִים.</B> וְאַחַר כָּךְ הֵפֵר לָהּ בַּעְלָהּ: <B>הֲרֵי זוֹ סוֹפֶגֶת אֶת הָאַרְבָּעִים.</B> עַל שֶׁעָבְרָה קֹדֶם הֲפָרָה {ה}: <B>תִּסְפֹּג מַכַּת מַרְדּוּת.</B> מִדִּבְרֵי סוֹפְרִים. וּמַכּוֹת מַרְדּוּת {ו} הָאֲמוּרָה בְּכָל מָקוֹם, כְּפִי מַרְאֶה עֵינֵי הַדַּיָּן וּכְפִי צֹרֶךְ הַשָּׁעָה. וְדַוְקָא בַּעֲבֵרָה שֶׁכְּבָר נֶעֶשְׂתָה, אֲבָל גַּבֵּי מִצְוֹת עֲשֵׂה כְּגוֹן עֲשֵׂה סֻכָּה וְאֵינוֹ עוֹשֶׂה, טֹל לוּלָב וְאֵינוֹ נוֹטֵל, מַכִּין אוֹתוֹ עַד שֶׁיַּעֲשֶׂה אוֹ עַד שֶׁתֵּצֵא נַפְשׁוֹ: 

~ פרק ד - משנה ד
<B>וְאִם שֶׁלָּהּ.</B> כְּגוֹן שֶׁנָּתַן לָהּ אַחֵר בְּמַתָּנָה עַל מְנָת שֶׁאֵין לְבַעֲלָהּ רְשׁוּת בָּהֶן {ח}. דְּנִכְסֵי מְלוֹג וְנִכְסֵי צֹאן בַּרְזֶל כֻּלָּן מְשֻׁעְבָּדִים לְבַעֲלָהּ: <B>חַטָּאת תָּמוּת.</B> מַמְתִּינִין לָהּ עַד שֶׁתָּמוּת: <B>וְנֶאֱכָלִין לְיוֹם אֶחָד.</B> כְּשַׁלְמֵי נָזִיר שֶׁאֵין נֶאֱכָלִים אֶלָּא לְיוֹם וְלַיְלָה: <B>וְאֵין טְעוּנִין לָחֶם.</B> שֶׁכָּל שַׁלְמֵי נָזִיר טְעוּנִים לֶחֶם, וְזוֹ הוֹאִיל וְהֵפֵר לָהּ בַּעְלָהּ אֵין שְׁלָמִים הַלָּלוּ טְעוּנִין לֶחֶם: <B>הָיוּ לָהּ מָעוֹת סְתוּמִים.</B> שֶׁהִפְרִישָׁתָן סְתָם לְקָרְבְּנוֹת נָזִיר, וְלֹא פֵּרְשָׁה אֵלּוּ הֵם לַחַטָּאת וְאֵלּוּ הֵם לְעוֹלָה וְאֵלּוּ הֵם לִשְׁלָמִים {י}: <B>יִפְּלוּ לִנְדָבָה.</B> לַתֵּבוֹת שֶׁבַּמִּקְדָּשׁ שֶׁמַּשְׁלִיכִים בָּהֶם שְׁאָר מְעוֹת נְדָבָה וְקוֹנִים מֵהֶן קָרְבָּנוֹת שֶׁכֻּלָּן עוֹלוֹת: <B>לֹא נֶהֱנִין וְלֹא מוֹעֲלִין.</B> לְכַתְּחִלָּה אָסוּר לֵהָנוֹת בָּהֶן, וְאִם נֶהֱנָה אֵינוֹ חַיָּב לְהָבִיא קָרְבַּן מְעִילָה הָאָמוּר בְּכָל מִי שֶׁנֶּהֱנֶה מִן הַהֶקְדֵּשׁ: 

~ פרק ד - משנה ה
<B>נִזְרַק עָלֶיהָ אֶחָד מִן הַדָּמִים אֵינוֹ יָכוֹל לְהָפֵר.</B> דִּלְאַחַר שֶׁנִּזְרַק הַדָּם הִיא יְכוֹלָה לִשְׁתּוֹת יַיִן וְלִטָּמֵא לְמֵתִים {יג}, וְאֵין כָּאן יוֹתֵר נֶדֶר שֶׁל עִנּוּי נֶפֶשׁ: <B>רַבִּי עֲקִיבָא אוֹמֵר אֲפִלּוּ נִשְׁחֲטָה עָלֶיהָ אַחַת מִכָּל הַבְּהֵמוֹת אֵינוֹ יָכוֹל לְהָפֵר.</B> מִשּׁוּם הֶפְסֵד קָדָשִׁים {יד}: <B>תִּגְלַחַת טֻמְאָה יָפֵר.</B> שֶׁצְּרִיכִים לַחְזֹר וְלִמְנוֹת נְזִירוּת טָהֳרָה, וְיָכוֹל לוֹמַר אִי אֶפְשִׁי בְּאִשָּׁה מְנֻוֶּלֶת, כְּלוֹמַר מְעֻנָּה וּמְנוּעָה מִשְּׁתִיַּת יַיִן {טז}: <B>אַף בְּתִגְלַחַת הַטָּהֳרָה יָפֵר.</B> כְּדֵי שֶׁלֹּא תִּצְטָרֵךְ לְהִתְנַּוֵּל בְּגִלּוּחַ, דְּגִלּוּחַ בְּאִשָּׁה נִוּוּל הוּא. וְתַנָּא קַמָּא סָבַר אֵין הַגִּלּוּחַ נִוּוּל, שֶׁהֲרֵי יְכוֹלָה לַעֲשׂוֹת פֵּאָה נָכְרִית {יז}. וְאֵין הֲלָכָה לֹא כְרַבִּי עֲקִיבָא וְלֹא כְרַבִּי: 

~ פרק ד - משנה ו
<B>הָאִישׁ מַדִּיר אֶת בְּנוֹ בְנָזִיר.</B> כְּשֶׁהוּא קָטָן עַד שֶׁיָּבִיא שְׁתֵּי שְׂעָרוֹת אַחַר שֶׁיִּהְיֶה בֶּן שְׁלֹשׁ עֶשְׂרֵה שָׁנָה וְיוֹם אֶחָד, וְכָל דִּין נְזִירוּת עָלָיו, וְאָבִיו מֵבִיא קָרְבְּנוֹתָיו, וְאִם נִטְמָא מֵבִיא קָרְבַּן טֻמְאָה. וְהַאי מַדִּירוֹ, שֶׁאוֹמֵר לוֹ תְּהֵא נָזִיר אוֹ הֲרֵי בְּנִי פְּלוֹנִי נָזִיר, וְהוּא שֶׁלֹּא יִמְחֶה לֹא הַבֵּן וְלֹא הַקְּרוֹבִים. וְדָבָר זֶה הֲלָכָה מִפִּי הַקַּבָּלָה: <B>כֵּיצַד, גִּלַּח אוֹ שֶׁגִּלְּחוּהוּ קְרוֹבִים.</B> כֵּיצַד יַעֲשֶׂה הָאָב בַּקָּרְבָּנוֹת בִּזְמַן שֶׁגִּלַּח הַבֵּן וְלֹא קִבֵּל הַנְּזִירוּת אוֹ שֶׁגִּלְּחוּהוּ קְרוֹבִים, אוֹ מִחָה אוֹ שֶׁמִּחוּהוּ קְרוֹבִים, דְּבָטֵל הַנְּזִירוּת. וְדַוְקָא כְּשֶׁמִּחָה הוּא אוֹ מִחוּ הַקְּרוֹבִים מִיָּד. אֲבָל אִם הִתְחִיל לִנְהֹג נְזִירוּת אוֹ שֶׁקִּבֵּל עָלָיו הַנְּזִירוּת, תּוּ לֹא מָצֵי לְמִמְחֵי לֹא הוּא וְלֹא קְרוֹבִים: 

~ פרק ד - משנה ז
<B>הָאִישׁ מְגַלֵּחַ עַל נְזִירוּת אָבִיו וְאֵין הָאִשָּׁה מְגַלַּחַת עַל נְזִירוּת אָבִיהָ.</B> וַאֲפִלּוּ הִיא בַּת יוֹרֶשֶׁת. וְדָבָר זֶה הֲלָכָה מִפִּי הַקַּבָּלָה: <B>אָמַר רַבִּי יוֹסֵי הֲרֵי אֵלּוּ יִפְּלוּ לִנְדָבָה וְכוּ'.</B> וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי. אֶלָּא בֵּין שֶׁמֵּת אָבִיו וְאָמַר הֲרֵינִי נָזִיר עַל מְנָת שֶׁאֲגַלַּח עַל מְעוֹת אַבָּא, בֵּין שֶׁהָיוּ הוּא וְאָבִיו {יח} נְזִירִים וּמֵת אָבִיו, מְגַלֵּחַ עַל נְזִירוּת אָבִיו. וְאִם הָיוּ בָּנִים רַבִּים וְקָדַם אֶחָד מֵהֶן וְגִלַּח עַל נְזִירוּת אָבִיו, זָכָה:

~ פרק ה - משנה א
<B>בֵּית שַׁמַּאי אוֹמְרִים הֶקְדֵּשׁ טָעוּת הֶקְדֵּשׁ.</B> דְּיָלְפִינַן מִתְּמוּרָה דַּהֲוֵי אֲפִלּוּ בְּטָעוּת, דִּכְתִיב (וַיִּקְרָא כז) וְהָיָה הוּא וּתְמוּרָתוֹ יִהְיֶה קֹדֶשׁ, וְדָרְשִׁינַן יִהְיֶה לְרַבּוֹת שׁוֹגֵג כְּמֵזִיד {א}: <B>וּבֵית הִלֵּל אוֹמְרִים אֵינוֹ הֶקְדֵּשׁ.</B> דְּלֹא גָמְרִינַן תְּחִלַּת הֶקְדֵּשׁ, דָּבָר שֶׁאֵינוֹ בָּא מִכֹּחַ הֶקְדֵּשׁ, מִתְּמוּרָה שֶׁהוּא סוֹף הֶקְדֵּשׁ, שֶׁבָּא מִכֹּחַ דָּבָר אַחֵר שֶׁהָיָה הֶקְדֵּשׁ: 

~ פרק ה - משנה ב
<B>דִּינַר זָהָב.</B> רֵישָׁא אַשְׁמוֹעִינַן בְּמִידֵי דְּקָדוֹשׁ קְדֻשַּׁת הַגּוּף, וְהָכָא אַשְׁמוֹעִינַן בְּמִידֵי דְּקָדוֹשׁ קְדֻשַּׁת דָּמִים:

<B>מִי שֶׁנָּדַר בְּנָזִיר.</B> בִּלְשׁוֹן שֶׁהָיָה דּוֹמֶה לוֹ שֶׁלֹּא הָיָה נָזִיר: <B>וְנִשְׁאַל לְחָכָם וַאֲסָרוֹ.</B> וְאָמַר לוֹ שֶׁיֵּשׁ בְּלָשׁוֹן זֶה לְשׁוֹן נְזִירוּת, וְהוּא לֹא נִזְהַר מִלִּשְׁתּוֹת יַיִן {ג}: <B>מוֹנֶה מִשָּׁעָה שֶׁנָּדַר.</B> וְלֹא קָנְסִינַן לֵיהּ {ד} עַל שֶׁעָבַר וְשָׁתָה יַיִן, אַף עַל גַּב דְּמִסָּפֵק אָסוּר הָיָה לוֹ לִפְרֹשׁ עַד שֶׁיִּשְׁאַל לְחָכָם: <B>נִשְׁאַל לְחָכָם וְהִתִּירוֹ.</B> שֶׁאָמַר לוֹ שֶׁאֵין בְּלָשׁוֹן זֶה לְשׁוֹן נְזִירוּת: <B>תֵּצֵא וְתִרְעֶה בָעֵדֶר.</B> דְּהַפְרָשָׁה בְטָעוּת הֲוַאי וְתֵצֵא לְחֻלִּין. וּבְהָא מוֹדוּ בֵּית שַׁמַּאי, דְּכֵיוָן דְּאֵינוֹ נָזִיר, כִּי קָאָמַר לְקָרְבְּנוֹת נְזִירוּתוֹ וְלֹא מִידֵי קָאָמַר, כְּאָדָם שֶׁאֵינוֹ חַיָּב חַטָּאת וְאָמַר הֲרֵי זוֹ לְחַטָּאתִי: <B>אִי אַתֶּם מוֹדִים בָּזֶה שֶׁהוּא הֶקְדֵּשׁ טָעוּת.</B> וּמַאי שְׁנָא מֵרֵישׁ פִּרְקִין דְּאַמְרִיתּוּ הֶקְדֵּשׁ בְּטָעוּת הָוֵי הֶקְדֵּשׁ: <B>אָמְרוּ לָהֶם בֵּית שַׁמַּאי.</B> לֹא חָשׁוּ לְהָשִׁיבָם עִקָּר טַעֲמָם, אֶלָּא לְדִבְרֵיהֶם קָאָמְרִי לְהוּ, מִתְּשִׁיעִי וְאַחַד עָשָׂר דְּקִדְּשׁוֹ בְטָעוּת, דִּמְרַבִּינַן לֵיהּ מִן וְכָל מַעְשַׂר בָּקָר וָצֹאן (וַיִּקְרָא כז): <B>אָמְרוּ לָהֶם בֵּית הִלֵּל לֹא הַשֵּׁבֶט קִדְּשׁוֹ.</B> בִּתְמִיהַּ {ו}. כְּלוֹמַר גְּזֵרַת הַכָּתוּב הִיא דִתְשִׁיעִי וְאַחַד עָשָׂר הַסְּמוּכִים לָעֲשִׂירִי הַשֵּׁבֶט מְקַדְּשָׁן אִם קָרָא לָהֶם עֲשִׂירִי, וְלֹא יָלְפִינַן מִנֵּיהּ בְּעָלְמָא דְהֶקְדֵּשׁ בְּטָעוּת הָוֵי הֶקְדֵּשׁ, דְּאִלּוּ הָוֵי טַעְמָא מִשּׁוּם הֶקְדֵּשׁ בְּטָעוּת אִם כֵּן אֲפִלּוּ שְׁמִינִי וּשְׁנֵים עָשָׂר נַמִּי, אֶלָּא הַכָּתוּב שֶׁקִּדֵּשׁ עֲשִׂירִי כוּ', וּגְזֵרַת הַכָּתוּב הִיא וְלֹא יָלְפִינַן מִנַּהּ {ז}:

<B>מִי שֶׁנָּדַר בְּנָזִיר.</B> וּבְשָׁעָה שֶׁנָּדַר הָיוּ לוֹ בְהֵמוֹת וְאַדַּעְתָּא דְּהָכִי נָדַר שֶׁיַּקְרִיב קָרְבְּנוֹתָיו מֵאוֹתָם בְּהֵמוֹת. וְהָלַךְ וּמְצָאָן שֶׁנִּגְנְבוּ וּמֵחֲמַת זֶה מִתְחָרֵט עַל שֶׁנָּדַר בְּנָזִיר: <B>אִם עַד שֶׁלֹּא נִגְנְבוּ בְהֵמוֹתָיו נָדַר הֲרֵי זֶה נָזִיר.</B> וְלֹא יַתִּיר לוֹ הֶחָכָם בְּפֶתַח זֶה, דְּנוֹלָד הָוֵי {ט} וְאֵין פּוֹתְחִין בְּנוֹלָד: <B>וְאִם לְאַחַר שֶׁנִּגְנְבוּ נָדַר.</B> וְאָמַר אִלּוּ הָיִיתִי יוֹדֵעַ שֶׁנִּגְנְבוּ לֹא הָיִיתִי נוֹדֵר, הֲרֵי זֶה פֶּתַח וְיַתִּירֶנּוּ הֶחָכָם: וְזוֹ טָעוּת טָעָה נַחוּם הַמָּדִי שֶׁכְּשֶׁעָלוּ נְזִירִים מִן הַגּוֹלָה וּמָצְאוּ בֵית הַמִּקְדָּשׁ חָרֵב. וְהֵם נָדְרוּ קֹדֶם שֶׁחָרַב הַבַּיִת, וְהִתִּיר לָהֶן, וְאָמְרוּ לוֹ חֲכָמִים דְּנוֹלָד הוּא וְאֵין פּוֹתְחִין בְּנוֹלָד. וַהֲלָכָה כַּחֲכָמִים:

<B>הֲרֵינִי נָזִיר שֶׁזֶּה אִישׁ פְּלוֹנִי.</B> אִם זֶה הַהוֹלֵךְ לְנֶגְדִּי הוּא אִישׁ פְּלוֹנִי הֲרֵינִי נָזִיר, וְהַשֵּׁנִי אוֹמֵר אִם אֵין זֶה אִישׁ פְּלוֹנִי הֲרֵינִי נָזִיר: <B>כֻּלָּם נְזִירִים.</B> וַאֲפִלּוּ אוֹתָם שֶׁלֹּא נִתְקַיְּמוּ דִבְרֵיהֶם. דְּכִי הֵיכִי דְהֶקְדֵּשׁ בְּטָעוּת הָוֵי הֶקְדֵּשׁ, הָכִי נַמִּי נְזִירוּת בְּטָעוּת הַוְיָא נְזִירוּת: <B>רַבִּי טַרְפוֹן אוֹמֵר אֵין אֶחָד מֵהֶם נָזִיר.</B> דְּסָבַר רַבִּי טַרְפוֹן אֵין נְזִירוּת אֶלָּא לְהַפְלָאָה, כְּלוֹמַר שֶׁיְּהֵא בָּרוּר וְיָדוּעַ לוֹ בִּשְׁעַת נִדְרוֹ שֶׁיְּהֵא נָזִיר, וְכָל הָנֵי לֹא הָיָה יָדוּעַ לוֹ בִּשְׁעַת נִדְרוֹ שֶׁיִּהְיֶה כִדְבָרָיו. וְאֵין הֲלָכָה כְּרַבִּי טַרְפוֹן:

<B>הִרְתִּיעַ לַאֲחוֹרָיו.</B> זֶה שֶׁבָּא כְּנֶגְדָּן חָזַר לַאֲחוֹרָיו וְלֹא נוֹדַע מִי הוּא: <B>אֵינוֹ נָזִיר.</B> אֵין אֶחָד מֵהֶם נָזִיר, דְּלֹא מַחֵית אֱינַשׁ נַפְשֵׁיהּ אַסְּפֵקָא {יב}, וְדַעְתּוֹ הָיָה בִשְׁעַת הַנֶּדֶר שֶׁאִם לֹא יָבֹא הַדָּבָר לִידֵי בֵּרוּר שֶׁלֹּא יִהְיֶה בִּדְבָרָיו כְּלוּם: <B>רַבִּי שִׁמְעוֹן אוֹמֵר כוּ'.</B> רַבִּי שִׁמְעוֹן לְטַעְמֵיהּ דְּאָמַר סְפֵק נְזִירוּת לְהַחְמִיר. וּמַה תַּקָּנָתָן, שֶׁהֲרֵי אִי אֶפְשָׁר לְהָבִיא קָרְבָּן מִסָּפֵק, אֶלָּא צְרִיכִים לְהַתְנוֹת כָּל אֶחָד וְלוֹמַר אִם אֵינוֹ כִדְבָרָיו שֶׁיְּהֵא נְזִיר נְדָבָה. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן:

<B>רָאָה אֶת הַכּוֹי.</B> שִׁשָּׁה בְּנֵי אָדָם שֶׁרָאוּ אֶת הַכּוֹי, וְאָמַר אֶחָד מֵהֶן הֲרֵינִי נָזִיר אִם זֶה חַיָּה, וְאָמַר הַשֵּׁנִי הֲרֵינִי נָזִיר אִם אֵינוֹ חַיָּה, וְאָמַר הַשְּׁלִישִׁי הֲרֵינִי נָזִיר אִם הוּא בְּהֵמָה, וְאָמַר הָרְבִיעִי הֲרֵינִי נָזִיר אִם אֵינוֹ בְּהֵמָה, וְאָמַר הַחֲמִישִׁי הֲרֵינִי נָזִיר אִם הוּא חַיָּה וּבְהֵמָה, וְאָמַר הַשִּׁשִּׁי הֲרֵינִי נָזִיר אִם אֵינוֹ לֹא חַיָּה וְלֹא בְהֵמָה, וְאָתוּ שְׁלֹשָׁה אַחֲרִינֵי מֵעָלְמָא וְאָמַר חַד מִנַּיְהוּ לְאֵלוּ הַשִּׁשָּׁה, הֲרֵינִי נָזִיר אִם אֶחָד מִכֶּם נָזִיר, וְאָמַר הַשֵּׁנִי הֲרֵינִי נָזִיר אִם אֵין אֶחָד מִכֶּם נָזִיר, וְאָמַר הַשְּׁלִישִׁי הֲרֵינִי נָזִיר אִם כֻּלְּכֶם נְזִירִים, כֻּלָּן נְזִירִים, הַשִּׁשָּׁה הָרִאשׁוֹנִים וְהַשְּׁלֹשָׁה הָאַחֲרוֹנִים. לְבֵית שַׁמַּאי {יג} הָווּ נְזִירִים וַדַּאי, דִּנְזִירוּת בְּטָעוּת הָוֵי נְזִירוּת, וּבֵין אוֹתוֹ שֶׁנִּתְקַיְּמוּ דְבָרָיו וּבֵין אוֹתוֹ שֶׁלֹּא נִתְקַיְּמוּ דְבָרָיו הָוֵי נָזִיר. וּלְבֵית הִלֵּל הָוֵי נְזִירוּת סָפֵק, מִפְּנֵי שֶׁהַכּוֹי סְפֵק בְּהֵמָה סְפֵק חַיָּה סְפֵק בְּרִיָּה בִּפְנֵי עַצְמָהּ {יד}:

<B>וְלֹא פֵרֵשׁ.</B> אַף עַל גַּב דְּכָל הַקָּרְבָּנוֹת צָרִיךְ שֶׁיִּקְרְאוּ לָהֶם הַבְּעָלִים שֵׁם, הָכָא לֹא צָרִיךְ, דִּמְכִי אָמַר אֵלּוּ לִנְזִירוּתִי הָוֵי כְאִלּוּ קָרָא שֵׁם לְכָל אֶחָד וְאֶחָד, דְּכִבְשָׂה אֵינָהּ רְאוּיָה אֶלָּא לְחַטָּאת {כ}, וְכֶבֶשׂ לְעוֹלָה, וְאַיִל לִשְׁלָמִים: <B>וּמְשַׁלֵּחַ תַּחַת הַדּוּד.</B> שֶׁמְּבַשְּׁלִים בּוֹ זֶבַח הַשְּׁלָמִים: <B>הָיָה מְגַלֵּחַ בַּמְּדִינָה.</B> בִּירוּשָׁלַיִם. וְאַף עַל גַּב דִּכְתִיב פֶּתַח אֹהֶל מוֹעֵד, לָאו דַוְקָא, אֶלָּא לְלַמֵּד שֶׁאֵינוֹ מְגַלֵּחַ עַד שֶׁיִּהְיֶה פֶּתַח אֹהֶל מוֹעֵד פָּתוּחַ {כא}: <B>לֹא הָיָה מְשַׁלֵּחַ תַּחַת הַדּוּד.</B> דִּכְתִיב (בְּמִדְבָּר ו) וְלָקַח אֶת שְׂעַר רֹאשׁ נִזְרוֹ וְנָתַן עַל הָאֵשׁ, מִי שֶׁאֵינוֹ מְחֻסָּר אֶלָּא לְקִיחָה וּנְתִינָה, יָצָא זֶה שֶׁמְּחֻסַּר לְקִיחָה וַהֲבָאָה וּנְתִינָה תַּחַת הַדּוּד: <B>בַּמֶּה דְבָרִים אֲמוּרִים.</B> שֶׁבַּמִּקְדָּשׁ נוֹטֵל שֵׂעָר וּמְשַׁלֵּחַ תַּחַת הַדּוּד, בְּתִגְלַחַת טָהֳרָה: <B>אֲבָל בְּתִגְלַחַת טֻמְאָה.</B> אֲפִלּוּ גִלַּח בַּמִּקְדָּשׁ, אֵינוֹ נוֹטֵל שֵׂעָר וּמְשַׁלֵּחַ תַּחַת הַדּוּד שֶׁל אָשָׁם וְחַטַּאת הָעוֹף, דְּלֹא כְתִיב נְתִינַת שֵׂעָר תַּחַת הַדּוּד אֶלָּא בְנָזִיר טָהוֹר: <B>הַכֹּל מְשַׁלְּחִין.</B> טָהוֹר בַּמִּקְדָּשׁ וּבַמְּדִינָה, וְטָמֵא בַמִּקְדָּשׁ {כב}, חוּץ מִטָּמֵא שֶׁגִּלַּח בַּמְּדִינָה, מִפְּנֵי שֶׁשְּׂעָרוֹ נִקְבָּר. וּפְסַק הַהֲלָכָה, שֶׁאֵין מְשַׁלֵּחַ תַּחַת הַדּוּד אֶלָּא נָזִיר טָהוֹר שֶׁגִּלַּח פֶּתַח אֹהֶל מוֹעֵד, שֶׁנַּעֲשָׂה כְמִצְוָתוֹ. וְאִם שִׁלַּח תַּחַת הַדּוּד שֶׁל חַטָּאת, יָצָא {כג}. לֹא נֶאֱמַר אֲשֶׁר תַּחַת זֶבַח הַשְּׁלָמִים אֶלָּא לְמִצְוָה:

~ פרק ו - משנה י
<B>וְנִמְצָא פָסוּל.</B> כְּגוֹן שֶׁנִּשְׁפַּךְ דָּמוֹ, אוֹ יָצָא אוֹ נִטְמָא: <B>תִּגְלַחְתּוֹ פְסוּלָה.</B> הוֹאִיל וְנִפְסַל הַזֶּבַח שֶׁגִּלַּח עָלָיו, הֲוָה לֵיהּ כְּגִלְּחוּהוּ לִסְטִים דְּאָמְרִינַן לְעֵיל לְרַבִּי אֱלִיעֶזֶר סוֹתֵר שִׁבְעָה יָמִים {כט} וּלְרַבָּנָן סוֹתֵר שְׁלֹשִׁים יוֹם: <B>וּזְבָחָיו לֹא עָלוּ לוֹ.</B> שְׁאָר הַזְּבָחִים שֶׁהִקְרִיב אַחַר תִּגְלַחְתּוֹ הַפְּסוּלָה, לֹא עָלוּ לוֹ, דְּכֵיוָן דְּצָרִיךְ לִסְתֹּר עַד שֶׁיִּגְדַּל שְׂעָרוֹ, הָוֵי כְאִלּוּ הִקְרִיבָן קֹדֶם זְמַנָּם: <B>עַל הַחַטָּאת שֶׁלֹּא לִשְׁמָהּ.</B> הַיְנוּ עַל הַזֶּבַח וְנִמְצָא פָּסוּל, דְּחַטָּאת שֶׁלֹּא לִשְׁמָהּ פְּסוּלָה. וְאַיְדֵי דְבָעֵי לְמִתְנֵי עַל הָעוֹלָה וְעַל הַשְּׁלָמִים פְּלֻגְתָּא דְּרַבִּי שִׁמְעוֹן וְרַבָּנָן, תָּנָא נַמִּי עַל הַחַטָּאת: <B>עַל הָעוֹלָה וְעַל הַשְּׁלָמִים שֶׁלֹּא לִשְׁמָן וְכוּ' תִּגְלַחְתּוֹ פְסוּלָה.</B> דְּהוֹאִיל וְלֹא עָלוּ לְשֵׁם חוֹבַת שַׁלְמֵי נָזִיר וְעוֹלַת נָזִיר, הָוֵי לֵיהּ כִּמְגַלֵּחַ עַל עוֹלַת נְדָבָה וְשַׁלְמֵי נְדָבָה. וְרַבִּי שִׁמְעוֹן סָבַר דְּאִם גִּלַּח עַל עוֹלַת נְדָבָה וְשַׁלְמֵי נְדָבָה יָצָא {ל}. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן: <B>אִם גִּלַּח עַל שְׁלָשְׁתָּן.</B> אַחַר שֶׁהִקְרִיבוּ שְׁלָשְׁתָּן: <B>וְנִמְצָא אֶחָד מֵהֶן כָּשֵׁר.</B> וְהַשְּׁאָר פְּסוּלִים: <B>תִּגְלַחְתּוֹ כְשֵׁרָה.</B> לְדִבְרֵי הַכֹּל. דְּהָא תְּנַן וְאִם גִּלַּח עַל אֶחָד מִשְּׁלָשְׁתָּן יָצָא:

<B>עַל הַמֵּת.</B> אַף עַל פִּי שֶׁאֵינוֹ שָׁלֵם {ה} אֶלָּא שֶׁיֵּשׁ בּוֹ רֹב בִּנְיָן, שֶׁהֵם שְׁנֵי שׁוֹקַיִם וְיָרֵךְ אֶחָד, אוֹ רֹב מִנְיַן הָאֵבָרִים שֶׁהֵם מֵאָה עֶשְׂרִים וַחֲמִשָּׁה אֵבָרִים, אֲפִלּוּ אֵין בָּהֶם רֹבַע קַב, מְטַמֵּא בָאֹהֶל, וְנָזִיר מְגַלֵּחַ עָלָיו. וְאִם אֵין בַּמֵּת לֹא רֹב מִנְיָן וְלֹא רֹב בִּנְיָן, אֵין הַנָּזִיר מְגַלֵּחַ עַל אָהֳלוֹ עַד שֶׁיְּהֵא בָּעֲצָמוֹת חֲצִי קַב: <B>וְעַל כַּזַּיִת נֶצֶל.</B> מֹהַל הַיּוֹצֵא מִן הַמֵּת כְּמִין לֵחָה הַיּוֹצֵאת מֵעִפּוּשׁ הַבָּשָׂר: <B>מְלֹא תַרְוָד רָקָב.</B> מְלֹא כָף מֵעֲפַר רִקָּבוֹן שֶׁל מֵת. וְאֵין הָרִקָּבוֹן מְטַמֵּא אֶלָּא כְשֶׁנִּקְבָּר הַמֵּת עָרוֹם בַּאֲרוֹן שֶׁל שַׁיִשׁ וְכַיּוֹצֵא בוֹ שֶׁאֵין שָׁם רִקָּבוֹן אַחֵר אֶלָּא מִגּוּפוֹ שֶׁל מֵת, וְנִקְבַּר הַמֵּת כֻּלּוֹ שָׁלֵם שֶׁלֹּא נֶחְסַר מִמֶּנּוּ אֵבֶר: <B>תַּרְוָד.</B> כַּף גְּדוֹלָה שֶׁמַּחְזֶקֶת מְלֹא חָפְנַיִם {ז}: <B>עַל הַשִּׁדְרָה.</B> אֲפִלּוּ אֵין בָּהּ בָּשָׂר כְּלָל: <B>וְכֵן עַל עֶצֶם הַגֻּלְגֹּלֶת.</B> שֶׁל הָרֹאשׁ, אֲפִלּוּ אֵין בָּהּ בָּשָׂר: <B>וְעַל אֵבֶר מִן הַחַי וְעַל אֵבֶר מִן הַמֵּת שֶׁיֵּשׁ עֲלֵיהֶם בָּשָׂר כָּרָאוּי.</B> כָּל שֶׁאִלּוּ הָיָה הָאֵבָר מְחֻבָּר בְּאָדָם חַי הָיָה יָכוֹל לְהַעֲלוֹת אֲרוּכָה עַל יְדֵי אוֹתוֹ בָשָׂר הוּא נִקְרָא כָרָאוּי. וְהוּא פָּחוֹת מִכַּזַּיִת: <B>וְעַל חֲצִי קַב עֲצָמוֹת.</B> וְאַף עַל גַּב דְּרֹבַע קַב עֲצָמוֹת מְטַמֵּא בָאֹהֶל, הֲלָכָה לְמֹשֶׁה מִסִּינַי שֶׁאֵין הַנָּזִיר מְגַלֵּחַ אֶלָּא עַל חֲצִי קַב, וְכֵן עַל חֲצִי לֹג דָּם. וְאַף עַל גַּב דְּרֹבַע דָּם מְטַמֵּא בָאֹהֶל, אֵין הַנָּזִיר מְגַלֵּחַ אֶלָּא עַל חֲצִי לֹג: <B>וְעַל עֶצֶם כַּשְּׂעֹרָה עַל מַגָּעוֹ וְעַל מַשָּׂאוֹ.</B> אֲבָל עַל אָהֳלוֹ לֹא, שֶׁאֵין עֶצֶם כַּשְּׂעֹרָה מְטַמֵּא בָאֹהֶל: <B>וְסוֹתֵר אֶת הַקּוֹדְמִים.</B> כְּדִכְתִיב (בְּמִדְבָּר ו) וְהַיָּמִים הָרִאשׁוֹנִים יִפְּלוּ: <B>עַד שֶׁיִּטְהַר.</B> שֶׁיִּטְבֹּל וְיַעֲרִיב שִׁמְשׁוֹ אַחַר שֶׁיַּזֶּה שְׁלִישִׁי וּשְׁבִיעִי:

<B>אֲבָל הַסְּכָכוֹת.</B> אִילָן שֶׁיֵּשׁ לוֹ עֲנָפִים מֻבְדָּלִים זֶה מִזֶּה, וְיֵשׁ כַּזַּיִת מִן הַמֵּת תַּחַת אֶחָד מֵהֶן, וְעָבַר הַנָּזִיר וְאֵין יָדוּעַ אִם עָבַר תַּחַת הֶעָנָף הַמַּאֲהִיל עַל הַמֵּת: <B>וְהַפְּרָעוֹת.</B> אֲבָנִים אוֹ עֵצִים בּוֹלְטִים מִן הַכֹּתֶל וְטֻמְאָה תַּחַת אֶחָד מֵהֶם: <B>וּבֵית הַפְּרָס.</B> שָׂדֶה שֶׁנֶּחֱרַשׁ בָּהּ קֶבֶר עוֹשָׂה בֵּית הַפְּרָס מֵאָה אַמָּה, שֶׁכָּךְ שִׁעֲרוּ שֶׁהַמַּחֲרֵשָׁה מוֹלִיכָה אֶת הָעֶצֶם: <B>וְאֶרֶץ הָעַמִּים.</B> שֶׁגָּזְרוּ חֲכָמִים טֻמְאָה עַל אֶרֶץ הָעַמִּים: <B>גּוֹלֵל.</B> כִּסּוּי הָאָרוֹן: <B>דּוֹפֵק.</B> הָאָרוֹן עַצְמוֹ שֶׁהַכִּסּוּי דּוֹפֵק עָלָיו {יג}: <B>וּרְבִיעִית דָּם.</B> אֲפִלּוּ עַל מַשָּׂאוֹ {יד} וּמַגָּעוֹ אֵין הַנָּזִיר מְגַלֵּחַ: <B>וְאֹהֶל וְרֹבַע עֲצָמוֹת.</B> אִם הֶאֱהִיל עַל רֹבַע קַב שֶׁל עַצְמוֹת {טו} הַמֵּת, אֵינוֹ מְגַלֵּחַ, עַד שֶׁיַּאֲהִיל עַל חֲצִי קַב. אֲבָל עַל מַגָּע וּמַשָּׂא {טז} שֶׁל רֹבַע עֲצָמוֹת, מְגַלֵּחַ, וַאֲפִלּוּ הֵם נִמּוֹקִים וְלֹא נִשְׁאַר בָּהֶם עֶצֶם כַּשְּׂעֹרָה שָׁלֵם: <B>וְכֵלִים הַנּוֹגְעִים בְּמֵת.</B> שֶׁהֵם מְטַמְּאִים כַּמֵּת עַצְמוֹ. וְהַנּוֹגֵעַ בָּהֶם טָמֵא טֻמְאַת שִׁבְעָה: <B>וִימֵי סָפְרוֹ.</B> לְאַחַר שֶׁנִּטְהַר מִצָּרַעְתּוֹ, דִּכְתִיב (וַיִּקְרָא יד) וְיָשַׁב מִחוּץ לְאָהֳלוֹ שִׁבְעַת יָמִים: <B>וִימֵי גָמְרוֹ.</B> יְמֵי חִלּוּטוֹ שֶׁל מְצֹרָע: <B>עַל אֵלּוּ אֵין הַנָּזִיר מְגַלֵּחַ.</B> עַל כָּל הָנָךְ דְּמַתְנִיתִין: <B>וּמַזֶּה בַּשְּׁלִישִׁי וּבַשְּׁבִיעִי.</B> הָא לֹא קָאֵי אַיְּמֵי סָפְרוֹ וְאַיְּמֵי גָמְרוֹ, דְּלֹא שַׁיָּךְ בְּהוּ הַזָּאָה שְׁלִישִׁי וּשְׁבִיעִי {כ}: <B>וְאֵינוֹ סוֹתֵר אֶת הַקּוֹדְמִין וּמַתְחִיל וּמוֹנֶה מִיָּד.</B> כָּל אֵלֶּה שֶׁאָמַרְנוּ, אֵינָן עוֹלִין לִימֵי הַנְּזִירוּת וְאֵין סוֹתְרִין, אֶלָּא אַחַר שֶׁיַּזֶּה שְׁלִישִׁי וּשְׁבִיעִי וְיִטְהַר אִם נִטְמָא בְּטֻמְאַת מֵת, וְאַחַר שֶׁיַּשְׁלִים יְמֵי גָמְרוֹ וִימֵי סָפְרוֹ אִם נִצְטָרֵעַ מַשְׁלִים נְזִירָתוֹ עַל מִנְיַן הַיָּמִים שֶׁמָּנָה קֹדֶם שֶׁנִּטְמָא, וְאֵינוֹ מֵבִיא קָרְבַּן טֻמְאָה: <B>בֶּאֱמֶת אָמְרוּ יְמֵי הַזָּב וְהַזָּבָה.</B> כָּל יְמֵי טֻמְאָתָם וִימֵי סְפִירָתָם שִׁבְעָה נְקִיִּים: <B>וִימֵי הֶסְגֵּרוֹ שֶׁל מְצֹרָע.</B> שֶׁנֶּאֱמַר בּוֹ (שָׁם טז) וְהִסְגִּיר אֶת הַנֶּגַע שִׁבְעַת יָמִים, עוֹלִין לְמִנְיַן יְמֵי הַנְּזִירוּת, וְאֵין צָרִיךְ לוֹמַר שֶׁאֵין סוֹתְרִין אֶת הַקּוֹדְמִים:

<B>חַיָּבִין עָלֶיהָ עַל בִּיאַת מִקְדָּשׁ.</B> אִם נִטְמָא בְאוֹתָהּ טֻמְאָה וְנִכְנַס לַמִּקְדָּשׁ, אוֹ אוֹכֵל קָדָשִׁים קֹדֶם שֶׁטָּהַר מִטֻּמְאָתוֹ, חַיָּבִין עָלֶיהָ כָּרֵת אִם הָיָה מֵזִיד, אוֹ קָרְבַּן עוֹלֶה וְיוֹרֵד אִם הָיָה שׁוֹגֵג: <B>לֹא תְהֵא זוֹ קַלָּה מִן הַשֶּׁרֶץ.</B> דְּחַיָּבִין עָלֶיהָ עַל בִּיאַת מִקְדָּשׁ, כְּדִכְתִיב בְּוַיִּקְרָא {ח} אוֹ בְנִבְלַת שֶׁרֶץ טָמֵא. וּמִלְּתָא דְרַבִּי מֵאִיר לֵיתָא, שֶׁהֲרֵי מַגָּע וּמַשָּׂא שֶׁל עֶצֶם כַּשְּׂעֹרָה שֶׁהוּא קַל שֶׁאֵינוֹ מְטַמֵּא בָאֹהֶל, הַנָּזִיר מְגַלֵּחַ עָלֶיהָ כְּדִתְנַן בְּמַתְנִיתִין, וּרְבִיעִית דָּם שֶׁהִיא חֲמוּרָה שֶׁמְּטַמֵּא בָאֹהֶל אֵין הַנָּזִיר מְגַלֵּחַ עָלֶיהָ: <B>אֵין דָּנִין כָּאן קַל וָחֹמֶר.</B> שֶׁאֵין דָּנִין קַל וָחֹמֶר עַל דָּבָר שֶׁהוּא הֲלָכָה לְמֹשֶׁה מִסִּינַי, שֶׁעֶצֶם כַּשְּׂעֹרָה אֵינוֹ כָּתוּב בַּתּוֹרָה אֶלָּא הֲלָכָה הִיא, וְאֵין דָּנִין קַל וָחֹמֶר מֵהֲלָכָה {כג}. וְהֵן הֵן דִּבְרֵי רַבִּי אֱלִיעֶזֶר הֵן הֵן דִּבְרֵי רַבִּי יְהוֹשֻׁעַ שֶׁאָמַר אֶלָּא כָּךְ אָמְרוּ הֲלָכָה:

<B>שְׁנֵי נְזִירִים שֶׁאָמַר לָהֶם אֶחָד כוּ'.</B> וְהֵם שׁוֹתְקִים, דְּאִם הֵם מַכְחִישִׁין אֵינוֹ נֶאֱמָן, דְּעֵד אֶחָד בְּמָקוֹם שֶׁמַּכְחִישִׁין אוֹתוֹ אֵינוֹ נֶאֱמָן. וּמַתְנִיתִין אָיְרֵי כְּגוֹן שֶׁזֶּה הָעֵד לֹא הָיָה עִמָּהֶם, אֶלָּא אָמַר רָאִיתִי מֵרָחוֹק טֻמְאָה שֶׁנִּזְרְקָה בֵינֵיכֶם, שֶׁאִם הָיָה הָעֵד אֶצְלָם בִּמְקוֹמָם {ב} הַוְיָא לָהּ סְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים, וּסְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים סְפֵקוֹ טָהוֹר, דְּכָל סְפֵק טֻמְאָה מִסּוֹטָה יָלְפִינַן לַהּ {ג} דִּכְתִיב בָּהּ (בְּמִדְבָּר ה) וְנִסְתְּרָה וְהִיא נִטְמָאָה, וְאֵינָן אֶלָּא הִיא וְהַבּוֹעֵל, וּסְפֵקָהּ טָמֵא. וְכָל זְמַן שֶׁהֵם יוֹתֵר מִשְּׁנַיִם, אֲפִלּוּ בְתוֹךְ הַבַּיִת סְפֵקוֹ טָהוֹר, דְּהַוְיָא לָהּ כִּסְפֵק טֻמְאָה בִּרְשׁוּת הָרַבִּים, לְכָךְ צָרִיךְ לוֹמַר שֶׁלֹּא הָיָה הָעֵד בִּשְׁעַת סְפֵק טֻמְאָה אֶצְלָם בִּמְקוֹמָם: <B>וְסוֹפְרִים שְׁלֹשִׁים יוֹם.</B> וּכְגוֹן שֶׁשְּׁנֵיהֶם נָדְרוּ נְזִירוּת בְּיַחַד וְנָדְרוּ סְתָם נְזִירוּת. וְהוּא הַדִּין אִם נָדְרוּ שְׁנֵיהֶם בְּיַחַד וּפֵרְשׁוּ נְזִירוּתָם לִזְמַן קָצוּב, שֶׁלְּאַחַר שֶׁהֵבִיאוּ קָרְבַּן טֻמְאָה וְקָרְבַּן טָהֳרָה חוֹזְרִים וּמוֹנִים הַזְּמַן הַקָּצוּב לִנְזִירוּתָם, וּמַפְרִישִׁים קָרְבַּן טָהֳרָה: <B>וְקָרְבַּן טֻמְאָה סָפֵק.</B> חַטַּאת הָעוֹף הַבָּא עַל הַטֻּמְאָה סָפֵק, וְאֵינָהּ נֶאֱכֶלֶת: <B>אֶלָּא מֵבִיא חַטַּאת עוֹף.</B> לִסְפֵק טֻמְאוֹת נְזִירוּת. וְאָשָׁם אֵינוֹ מֵבִיא, כְּרַבָּנָן דְּאָמְרֵי לְעֵיל {ז} הֵבִיא חַטָּאתוֹ וְלֹא הֵבִיא אֲשָׁמוֹ מוֹנֶה: <B>נִמְצָא זֶה מֵבִיא קָרְבְּנוֹתָיו לַחֲצָאִים.</B> אִם נָזִיר טָהוֹר הוּא. דְּעוֹלָה רִאשׁוֹנָה הָיְתָה חוֹבָה, וְעַתָּה הוּא מַקְרִיב חַטָּאת וּשְׁלָמִים: <B>אֲבָל הוֹדוּ חֲכָמִים לְבֶן זוֹמָא.</B> וְלֹא חָשְׁשׁוּ עַל שֶׁמַּקְרִיב קָרְבְּנוֹתָיו לַחֲצָאִים. וַהֲלָכָה כְּבֶן זוֹמָא: 

~ פרק ח - משנה ב
<B>נָזִיר שֶׁהָיָה טָמֵא בְסָפֵק וּמֻחְלָט בְּסָפֵק.</B> סָפֵק אִם נִטְמָא בְמֵת וְסָפֵק אִם הוּא מְצֹרָע מֻחְלָט: <B>אוֹכֵל בַּקָּדָשִׁים אַחַר שִׁשִּׁים יוֹם.</B> אֲבָל קֹדֶם לָכֵן לֹא, דִּסְפֵק מְצֹרָע הוּא וְעַד שֶׁיָּבִיא כַפָּרָתוֹ אָסוּר לֶאֱכֹל בַּקָּדָשִׁים. אֲבָל מִשּׁוּם סְפֵק נָזִיר טָמֵא לֹא מִתְּסַר לֶאֱכֹל בַּקָּדָשִׁים, דְּלֹא מִקְרֵי מְחֻסָּר כַּפָּרָה שֶׁאָסוּר לֶאֱכֹל בַּקָּדָשִׁים אֶלָּא מִי שֶׁטֻּמְאָה יוֹצְאָה עָלָיו מִגּוּפוֹ. כֵּיצַד, נָזַר שְׁלֹשִׁים יוֹם, וּבַיּוֹם הָרִאשׁוֹן אֵרַע לוֹ סְפֵק טֻמְאַת מֵת וּסְפֵק צָרַעַת, מַזֶּה וְשׁוֹנֶה וְטוֹבֵל וְנִטְהַר מִטֻּמְאַת מֵת כְּאִלּוּ בָא לְגַלֵּחַ כְּדִין נָזִיר טָמֵא אוֹ כְּדִין מְצֹרָע שֶׁנִּתְרַפֵּא, וְאֵינוֹ יָכוֹל לְגַלֵּחַ עַד שְׁלֹשִׁים יוֹם מִיּוֹם שֶׁנָּזַר, שֶׁמָּא נָזִיר טָהוֹר הוּא וְאָסוּר לְגַלֵּחַ עַד מְלֹאת יְמֵי נִזְרוֹ וְעַד שֶׁיָּבִיא קָרְבְּנוֹתָיו, וּלְאַחַר שְׁלֹשִׁים יוֹם מְגַלֵּחַ {י}, סְפֵק תִּגְלַחַת מְצֹרָע סְפֵק תִּגְלַחַת טֻמְאָה שֶׁל נָזִיר וּסְפֵק תִּגְלַחַת טָהֳרָה. וּמֵבִיא שְׁתֵּי צִפֳּרִים {יא} כְּדִין תִּגְלַחַת חִלּוּטוֹ שֶׁל מְצֹרָע, וְחַטַּאת הָעוֹף הַבָּאָה עַל הַסָּפֵק מִשּׁוּם סְפֵק נָזִיר טָמֵא, וְעוֹלַת בְּהֵמָה מִשּׁוּם סְפֵק נָזִיר טָהוֹר {יב}. וְאַף עַל פִּי שֶׁאֵינוֹ מֵבִיא שְׁאָר קָרְבָּנוֹת שֶׁל תִּגְלַחַת טָהֳרָה, הָא אָמְרִינַן אִם גִּלַּח עַל אַחַת מִשְּׁלָשְׁתָּן יָצָא. וּמַתְנֶה עָלֶיהָ שֶׁאִם אֵינוֹ נָזִיר טָהוֹר שֶׁתְּהֵא עוֹלַת נְדָבָה. וְדִין מְצֹרָע מֻחְלָט שֶׁצָּרִיךְ שְׁנֵי תִּגְלָחוֹת, אַחַת לְאַחַר גְּמַר חִלּוּטוֹ כְּשֶׁנִּרְפָּא מִצָּרַעְתּוֹ, וְאַחַת לְאַחַר יְמֵי סָפְרוֹ לְאַחַר שֶׁסָּפַר שִׁבְעָה מִן הַתִּגְלַחַת הָרִאשׁוֹנָה, כְּדִכְתִיב (וַיִּקְרָא יד) וְהָיָה בַיּוֹם הַשְּׁבִיעִי יְגַלַּח אֶת כָּל שְׂעָרוֹ, וְזֶה שֶׁהוּא סְפֵק נָזִיר אֵינוֹ יָכוֹל לְגַלֵּחַ, הִלְכָּךְ צָרִיךְ לְהַמְתִּין שְׁלֹשִׁים יוֹם שֶׁל סְפֵק נְזִירוּת טָהֳרָה {יג} וּמְגַלֵּחַ תִּגְלַחַת שֶׁל סְפֵק יְמֵי סָפְרוֹ שֶׁל מְצֹרָע וּסְפֵק נְזִירוּת טָהֳרָה, וּמֵבִיא עוֹלַת בְּהֵמָה שֶׁל סְפֵק נְזִירוּת כְּדֵי לְגַלֵּחַ עַל הַקָּרְבָּנוֹת וּמַתְנֶה עָלֶיהָ. וּלְמָחֳרָת יוֹם תִּגְלַחְתּוֹ מֵבִיא קָרְבַּן מְצֹרָע כְּדֵי לְהַתִּירוֹ בַּקָּדָשִׁים, וּמֵבִיא חַטַּאת עוֹף {יד}, שֶׁחַטַּאת בְּהֵמָה אֵינָהּ בָּאָה עַל הַסָּפֵק, וְאֵינוֹ מֵבִיא אָשָׁם, שֶׁהָאָשָׁם אֵינוֹ מְעַכֵּב מִלֶּאֱכֹל בַּקָּדָשִׁים. וְנִמְצָא שֶׁלְּאַחַר שִׁשִּׁים יוֹם אוֹכֵל בַּקָּדָשִׁים, שֶׁאָז מֵבִיא כַּפָּרָתוֹ עַל סָפֵק חִלּוּטוֹ. וַעֲדַיִן הוּא אָסוּר לִשְׁתּוֹת יַיִן וְלִטַּמֵּא לְמֵתִים, דְּשֶׁמָּא מֻחְלָט הָיָה, וְתִגְלַחַת מְצֹרָע אֵינוֹ עוֹלֶה לֹא לְתִגְלַחַת נָזִיר טָהוֹר וְלֹא לְתִגְלַחַת נָזִיר טָמֵא {טו}, וְטָעוּן עוֹד שְׁתֵּי תִגְלָחוֹת, אֶחָד לְנָזִיר טָמֵא וְאַחַת לְנָזִיר טָהוֹר, וְאִם מְצֹרָע מֻחְלָט הָיָה, כָּל מַה שֶּׁמָּנָה לִימֵי נְזִירָתוֹ אֵין עוֹלִין לוֹ, שֶׁיְּמֵי חִלּוּטוֹ וִימֵי סָפְרוֹ אֵין עוֹלִין לוֹ, וְצָרִיךְ לִמְנוֹת עוֹד שְׁלֹשִׁים יוֹם, וּלְאַחַר שְׁלֹשִׁים יוֹם מְגַלֵּחַ סְפֵק נָזִיר טָמֵא סְפֵק נָזִיר טָהוֹר, וּמֵבִיא חַטַּאת הָעוֹף בִּסְפֵק נָזִיר טָמֵא, וְעוֹלַת בְּהֵמָה בִּסְפֵק נָזִיר טָהוֹר כְּדֵי לְגַלֵּחַ עַל הַקָּרְבָּן, וּמַתְנֶה עָלֶיהָ. וַעֲדַיִן אָסוּר לִשְׁתּוֹת יַיִן וְלִטַּמֵּא לְמֵתִים, דְּשֶׁמָּא מְצֹרָע מֻחְלָט הָיָה, וּלְפִיכָךְ הַשְּׁתֵּי תִּגְלָחוֹת הָרִאשׁוֹנוֹת לֹא עָלוּ לוֹ לְשֵׁם נְזִירוּת, לֹא שֶׁל טֻמְאָה וְלֹא שֶׁל טָהֳרָה, וְשֶׁמָּא נָזִיר טָמֵא הָיָה וְתִגְלַחַת הַשְּׁלִישִׁית תִּגְלַחַת שֶׁל טֻמְאָה הִיא, לְפִיכָךְ צָרִיךְ לִמְנוֹת עוֹד שְׁלֹשִׁים יוֹם שֶׁל נְזִירוּת טָהֳרָה, וּמֵבִיא קָרְבַּן נָזִיר טָהוֹר, וּמַתְנֶה, וְאַחַר כָּךְ מֻתָּר לִשְׁתּוֹת יַיִן וּלְהִטָּמֵא לְמֵתִים. וְכֵן אִם נָדַר נְזִירוּת שָׁנָה וְאֵרַע לוֹ בְּרֹאשׁ שְׁנָתוֹ סְפֵק טֻמְאַת מֵת וּסְפֵק צָרַעַת, מוֹנֶה שָׁנָה וּמְגַלֵּחַ תִּגְלַחַת שֶׁל סְפֵק טֻמְאָה וּסְפֵק טָהֳרָה וּסְפֵק צָרַעַת, וּמוֹנֶה שָׁנָה שְׁנִיָּה וּמְגַלֵּחַ וּמֵבִיא כַפָּרָתוֹ וְאוֹכֵל בַּקָּדָשִׁים, וּמוֹנֶה עוֹד שְׁתֵּי שָׁנִים קֹדֶם שֶׁיִּשְׁתֶּה יַיִן וְיִטַּמָּא לְמֵתִים:

~ פרק ט - משנה א
<B>הַגּוֹיִם אֵין לָהֶם נְזִירוּת.</B> אַף עַל גַּב דְּגוֹיִם מַקְרִיבִין נְדָרִים וּנְדָבוֹת כְּיִשְׂרָאֵל, אִם נָדַר בְּנָזִיר אֵין עָלָיו תּוֹרַת נְזִירוּת וּמֻתָּר לִשְׁתּוֹת יַיִן וְלִטַּמֵּא לְמֵתִים, דִּכְתִיב בְּרֵישׁ פָּרָשַׁת נָזִיר (בְּמִדְבָּר ו) דַּבֵּר אֶל בְּנֵי יִשְׂרָאֵל, בְּנֵי יִשְׂרָאֵל נוֹדְרִים בְּנָזִיר וְאֵין גּוֹיִם נוֹדְרִים בְּנָזִיר: <B>שֶׁהוּא כוֹפֶה אֶת שֶׁיָּצָא לְחֵרוּת. וְרַמְבַּ"ם פֵּרֵשׁ הֵפֵר לְעַבְדּוֹ יָצָא לְחֵרוּת, שֶׁהָאוֹמֵר לְעַבְדּוֹ מוּפָר לְךָ, פָּקַע זְכוּת הָאָדוֹן מִמֶּנּוּ וְיֵצֵא הָעֶבֶד לְחֵרוּת בִּשְׁבִיל כֵּן וְצָרִיךְ שֶׁיַּשְׁלִים נְזִירוּתוֹ. וְלִבִּי מְהַסֵּס בְּפֵרוּשׁ זֶה: <B>עָבַר מִכְּנֶגֶד פָּנָיו.</B> שֶׁבָּרַח מֵרַבּוֹ אַחַר שֶׁנָּדַר בְּנָזִיר: <B>רַבִּי מֵאִיר אוֹמֵר לֹא יִשְׁתֶּה.</B> יַיִן כְּדֵי שֶׁיִּצְטַעֵר וְיָשׁוּב לְרַבּוֹ שֶׁיִּכְפֵּהוּ לַעֲבֹר עַל נִדְרוֹ וְיִהְיֶה מֻתָּר לִשְׁתּוֹת יַיִן: <B>וְרַבִּי יוֹסֵי אוֹמֵר יִשְׁתֶּה.</B> שֶׁלֹּא יֶחֱלֶה וְיָמוּת, שֶׁהֲרֵי עָתִיד הוּא לָשׁוּב לְרַבּוֹ, שֶׁיְּבַקֵּשׁ רַבּוֹ אַחֲרָיו וְיַחְזִירֶנּוּ, וְנִמְצָא כְאִלּוּ הוּא בִּרְשׁוּת רַבּוֹ:

בִקְבֻרָתָם, שֶׁאֵין צָרִיךְ לוֹמַר מִמִּצְרַיִם, אֶלָּא הָכִי קָאָמַר, מֵעָפָר שֶׁל מִצְרַיִם טֹל עִמִּי: <B>מָצָא שְׁלֹשָׁה אִם יֵשׁ בֵּין זֶה לָזֶה מֵאַרְבַּע אַמּוֹת עַד שְׁמֹנֶה.</B> כְּלוֹמַר מִקֶּבֶר רִאשׁוֹן עַד שְׁלִישִׁי אֵין פָּחוֹת מֵאַרְבַּע אַמּוֹת {יח} וְלֹא יוֹתֵר מִשְּׁמֹנֶה: <B>הֲרֵי זוֹ שְׁכוּנַת קְבָרוֹת.</B> וְנִכָּר שֶׁלְּשֵׁם קְבוּרָה נְתָנוּם שָׁם, וְאָסוּר לְפַנּוֹתָם. וַאֲפִלּוּ מֵת אֶחָד אִם נִכָּר שֶׁלְּשֵׁם קְבוּרָה נָתוּן שָׁם אָסוּר לְפַנּוֹתוֹ. אֶלָּא שֶׁבְּאֶחָד אוֹ בִּשְׁנַיִם אָנוּ תוֹלִין שֶׁלֹּא נִקְבְּרוּ שָׁם אֶלָּא לְפִי שָׁעָה וְהָיָה דַּעְתָּם לְפַנּוֹתָם. אֲבָל בִּשְׁלֹשָׁה מוּכָח שֶׁזֶּה מָקוֹם מְיֻחָד לִקְבָרוֹת. וְאֹרֶךְ הַמְּעָרָה דַּרְכָּהּ לִהְיוֹת שֵׁשׁ וְרָחְבָּהּ אַרְבַּע וַאֲלַכְסוֹנָהּ עוֹדֵף שְׁתֵּי אַמּוֹת דְּהַיְנוּ שְׁמֹנֶה. לְהָכִי תְּנַן מֵאַרְבַּע עַד שְׁמֹנֶה. כִּמְלֹא מִטָּה וְקוֹבְרֶיהָ לֹא גָּרְסִינַן לַהּ הָכָא {יט}: <B>וּבוֹדֵק הֵימֶנּוּ וּלְהַלָּן עֶשְׂרִים אַמָּה.</B> שֶׁהַמְּעָרָה אַרְבַּע אַמּוֹת עַל שֵׁשׁ, וְהֶחָצֵר שֶׁהַמְּעָרוֹת פְּתוּחִין לוֹ מִכָּאן וּמִכָּאן שֵׁשׁ עַל שֵׁשׁ, דְּהָכִי סְבִירָא לְהוּ לְרַבָּנָן בְּפֶרֶק הַמּוֹכֵר פֵּרוֹת [קב:], נִמְצָא אֹרֶךְ שְׁתֵּי מְעָרוֹת וְחָצֵר שֶׁבֵּינֵיהֶן שְׁמֹנֶה עֶשְׂרֵה אַמּוֹת {כ}. וּלְפִי שֶׁפְּעָמִים בּוֹדֵק מְעָרָה אַחַת בַּאֲלַכְסוֹנָהּ {כא} וְהָאֲלַכְסוֹן שֶׁל מְעָרָה אַחַת עוֹדֵף שְׁתֵּי אַמּוֹת בְּקֵרוּב {כב}, נִמְצְאוּ עֶשְׂרִים אַמּוֹת, שְׁמֹנֶה דִּמְעָרָה רִאשׁוֹנָה בַּאֲלַכְסוֹנָהּ, וְשֵׁשׁ דְּחָצֵר שֶׁבֵּין שְׁתֵּי הַמְּעָרוֹת, וְשֵׁשׁ דִּמְעָרָה שְׁנִיָּה, דְּחַד אֲלַכְסוֹן אָמְרִינַן, תְּרֵי אֲלַכְסוֹנִי לֹא אָמְרִינַן, וְהַיְנוּ דְקָתָנֵי עֶשְׂרִים. וְעוֹד צָרִיךְ לִבְדֹּק מִלְּמַעְלָה וּמִלְּמַטָּה עֶשְׂרִים אַמָּה, דְּהַיְנוּ אַרְבָּעִים אַמָּה, דְּשֶׁמָּא זוֹ הִיא מְעָרָה שֶׁבְּמִזְרַח הֶחָצֵר וְיֵשׁ עֲדַיִן אַחֶרֶת כְּנֶגְדָּהּ בְּמַעֲרַב הֶחָצֵר. אִי נַמִּי זוֹ הִיא שֶׁבְּמַעֲרַב הֶחָצֵר וְיֵשׁ עֲדַיִן אַחֶרֶת בְּמִזְרַח הֶחָצֵר {כג}: <B>מָצָא אֶחָד בְּסוֹף עֶשְׂרִים בּוֹדֵק הֵימֶנּוּ וּלְהַלָּן עֶשְׂרִים אַמָּה.</B> דְּמִי יֵימַר דְּמִבֵּית הַקְּבָרוֹת הַוְיָא הַהִיא מְעָרָה, שֶׁמָּא קֶבֶר אַחֵר הוּא וְחָצֵר אַחֶרֶת שֶׁל אָדָם אַחֵר, וְצָרִיךְ לַעֲשׂוֹת גַּם לְשָׁם כָּל בְּדִיקוֹת דִּלְעֵיל, דִּכְמוֹ שֶׁיֵּשׁ שָׁם קֶבֶר כְּמוֹ כֵן יֵשׁ אֲחֵרִים: <B>שֶׁרַגְלַיִם לַדָּבָר.</B> שֶׁזּוֹ הַשָּׂדֶה עֲשׂוּיָה לִקְבָרִים וְהָיוּ בָּהּ גַּם מְעָרוֹת אֲחֵרוֹת {כד}. וּמִשּׁוּם רַגְלַיִם לַדָּבָר תָּנֵי כָּל הָנֵי מִשְׁנָיוֹת הָכָא:

<B>כָּל סְפֵק נְגָעִים בַּתְּחִלָּה טָהוֹר.</B> כְּגוֹן שְׁנַיִם שֶׁבָּאוּ אֵצֶל כֹּהֵן, בָּזֶה בַּהֶרֶת כִּגְרִיס וּבָזֶה בַּהֶרֶת כְּסֶלַע, וּבְסוֹף שָׁבוּעַ בָּזֶה כְּסֶלַע וּבָזֶה כְּסֶלַע, וְאֵין יָדוּעַ בְּאֵיזֶה מֵהֶן פָּשָׂה, שְׁנֵיהֶן טְהוֹרִין, וְאַף עַל גַּב דְּוַדַּאי אֶחָד מֵהֶן טָמֵא {כה}: <B>מִשֶּׁנִּזְקַק לַטֻּמְאָה סְפֵקוֹ טָמֵא.</B> כְּגוֹן שְׁנַיִם שֶׁבָּאוּ אֵצֶל כֹּהֵן, בָּזֶה בַּהֶרֶת כִּגְרִיס וּבָזֶה כְּסֶלַע, וּבְסוֹף שָׁבוּעַ בָּזֶה כְּסֶלַע וְעוֹד וּבָזֶה כְּסֶלַע וְעוֹד, חָזְרוּ לִהְיוֹת כְּסֶלַע, אַף עַל פִּי שֶׁהָאֶחָד מֵהֶם וַדַּאי טָהוֹר, שֶׁכְּבָר הָלַךְ הַפִּשְׂיוֹן, שְׁנֵיהֶם טְמֵאִים, כֵּיוָן שֶׁנִּזְקְקוּ לַטֻּמְאָה, עַד שֶׁיַּחְזְרוּ לְכִגְרִיס: <B>בְּשִׁבְעָה דְרָכִים בּוֹדְקִין אֶת הַזָּב.</B> שֶׁאִם רָאָה מֵחֲמַת אֹנֶס, טָהוֹר, דְּדָרְשִׁינַן זָב מִבְּשָׂרוֹ, וְלֹא מֵחֲמַת אָנְסוֹ {כו}: <B>עַד שֶׁלֹּא נִזְקַק לַטֻּמְאָה.</B> דְּהַיְנוּ כְּשֶׁרוֹאֶה רְאִיָּה שְׁנִיָּה שֶׁבָּהּ נַעֲשָׂה זָב גָּמוּר לְטַמֵּא מִשְׁכָּב וּמוֹשָׁב {כז}. אֲבָל רְאִיָּה רִאשׁוֹנָה מְטַמְּאָה בְּאֹנֶס טֻמְאַת עֶרֶב כְּדִין בְּעַל קֶרִי, וּמִצְטָרֶפֶת עִם הַשְּׁנִיָּה {כח} אֲפִלּוּ הָיְתָה בְּאֹנֶס: <B>בְּמַאֲכָל.</B> אִם אוֹכֵל דְּבָרִים הַמְּבִיאִים לִידֵי לֹא הִרְהֵר: <B>וּבְהִרְהוּר.</B> אַף עַל פִּי שֶׁלֹּא רָאָה. אִם אֵרַע לוֹ אֶחָד מֵאֵלּוּ הַשִּׁבְעָה קֹדֶם שֶׁרָאָה רְאִיָּה שְׁנִיָּה, אֵינוֹ נַעֲשֶׂה זָב, וְהַטִּפָּה אֵינָהּ מְטַמְּאָה בַּמַּשָּׂא: <B>מִשֶּׁנִּזְקַק לַטֻּמְאָה.</B> שֶׁרָאָה רְאִיָּה שְׁנִיָּה שֶׁלֹּא בְּאֹנֶס: <B>אֵין בּוֹדְקִין אוֹתוֹ.</B> וְאַף אִם רָאָה שְׁלִישִׁית בְּאֹנֶס נַעֲשֶׂה זָב לְקָרְבָּן {כט}: <B>וּסְפֵקוֹ וְשִׁכְבַת זַרְעוֹ טְמֵאִים.</B> סְפֵקוֹ מֵחֲמַת שִׁכְבַת זַרְעוֹ. כְּגוֹן אִם רָאָה שִׁכְבַת זֶרַע תְּחִלָּה וְטִפַּת זִיבָה הַבָּאָה אַחַר כָּךְ, אֵינָהּ מְטַמְּאָה שֶׁהָרוֹאֶה קֶרִי אֵינוֹ מְטַמֵּא בְּזִיבָה מֵעֵת לְעֵת {לא}. וְעַד שֶׁלֹּא נִזְקַק לַטֻּמְאָה הַקֶּרִי מְטַהֵר אֶת הַזִּיבָה, מִשּׁוּם דַּהֲוֵי אֹנֶס, וּלְאַחַר שֶׁנִּזְקַק לַטֻּמְאָה אֵין הַקֶּרִי מְטַהֵר אֶת הַזִּיבָה, שֶׁאֵין תּוֹלִין שֶׁמֵּחֲמַת הַקֶּרִי בָּאָה הַזִּיבָה: <B>שֶׁרַגְלַיִם לַדָּבָר.</B> שֶׁאֵין אוֹתָהּ רְאִיָּה מֵחֲמַת אֹנֶס, כֵּיוָן שֶׁכְּבָר נַעֲשָׂה זָב {לב}: <B>רַבִּי נְחֶמְיָה פוֹטֵר
