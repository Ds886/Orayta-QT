{א} <b>שֶׁפַּעֲמַיִם כוּ'.</b> אַבְּתוּלָה קָאֵי. רַשִׁ"י. וְכָתַב הָרַ"ן דְּמַקְשׁוּ הָכָא, דְּלִתְנֵי נַמִּי טַעֲמָא דְּשָׁקְדוּ לְמִעוּטֵי יוֹם רִאשׁוֹן בְּשַׁבָּת, וְטַעֲמָא דְּאַלְמָנָה בְּיוֹם חֲמִישִׁי כוּ'. עוֹד יֵשׁ לוֹמַר דְּמִשּׁוּם שָׁקְדוּ לֹא הֲווֹ מְתַקְּנֵי רַבָּנָן לְעִכּוּבֵי מִצְוָה דְנִשּׂוּאִין, אֶלָּא מִשּׁוּם טַעֲמָא דִבְתוּלִים דְּמִלְּתָא דְאִסּוּרָא הִיא הֻצְרְכוּ לְיַחֵד לַבְּתוּלָה יָמִים, וְכֵיוָן שֶׁהֻצְרְכוּ לְיַחֵד יִחֲדוּ לַבְּתוּלָה רְבִיעִי דְּמַעֲלֵי טְפֵי מֵאֶחָד בְּשַׁבָּת מִשּׁוּם שָׁקְדוּ, וְאַגַּב, יִחֲדוּ לָאַלְמָנָה גַּם כֵּן, דְּאִי לֹא, אָתוּ לְמֵימַר דְּנִשּׂוּאֵי אַלְמָנָה זִילוּ לְרַבָּנָן וְלֹא מְתַקְּנֵי בְהוּ מִלְּתָא, וּמִשּׁוּם הָכִי תָּנֵי טַעֲמָא דִבְתוּלִים, לוֹמַר דְּעִקַּר הַתַּקָּנָה הָיְתָה בִּשְׁבִילָהּ. וְעַיֵּן עוֹד: {ב} <B> הַשֵּׁנִי כוּ'.</B> דִּשְׁכִיחֵי, דְּאָתוּ לְמִקְרֵי בְּסִפְרָא. גְּמָרָא: {ג} פֵּרוּשׁ, לְאַחַר שֶׁנִּתְאָרְסָה, וַהֲרֵי הִיא אֲסוּרָה עָלָיו, כְּגוֹן אֵשֶׁת כֹּהֵן, אוֹ שֶׁקִּבֵּל בָּהּ אָבִיהָ קִדּוּשֶׁיהָ פְּחוּתָה מִבַּת שָׁלֹשׁ, דְּלֵיכָּא אֶלָּא חַד סְפֵקָא, בְּכֹהֵן, שֶׁמָּא קֹדֶם שֶׁנִּתְאָרְסָה, וּבִפְחוּתָה מִבַּת שָׁלֹשׁ, שֶׁמָּא בְּאֹנֶס. וּמִשּׁוּם הָכִי אַף בִּשְׁאָר נָשִׁים דְּאִיכָּא סְפֵק סְפֵקָא וְאֵינָהּ אֲסוּרָה עָלָיו לֹא פְּלוּג רַבָּנָן בְּתַקַּנְתָּא. תּוֹסָפוֹת: {ד}  גְּמָרָא. וְהָכִי קָאָמַר, מִצַּד תַּקָּנָה זוֹ דְמִשְׁנָתֵנוּ נִשֵּׂאת בְּכָל יוֹם. אֲבָל וַדַּאי יֵשׁ יָמִים דְּאִסּוּר אַחֵר חָל עֲלַיְהוּ, כְּגוֹן שַׁבָּת וְיוֹם טוֹב וְחֹל הַמּוֹעֵד: {ה} וְאִי נִשֵּׂאת בְּיוֹם אַחֵר לֹא מִקַּיְמָא תַּקַּנְתָּא אֶלָּא מַשְׁכִּים לְמָחָר לִמְלַאכְתּוֹ וְהוֹלֵךְ לוֹ:

{ו} <b>מָאתַיִם.</b> בְּסוֹף פֶּרֶק י"ג פָּסַק הָרַ"ב דִּכְתֻבַּת אִשָּׁה דְּרַבָּנָן, וְהַשְׁתָּא הָנֵי מָאתַיִם בְּכֶסֶף מְדִינָה הֵם, שֶׁכֵּן כָּל שֶׁל דִּבְרֵיהֶם, וְיֵשׁ בְּכָל דִּינָר שִׁבְעָה חֲלָקִים נְחֹשֶׁת וּשְׁמִינִית כֶּסֶף, נִמְצָא מָאתַיִם, עֶשְׂרִים וַחֲמִשָּׁה שֶׁל כֶּסֶף טָהוֹר, וְזוֹ דַעַת הָרַמְבַּ"ם. וְהָרֹא"שׁ סוֹבֵר דְּבִכְתֻבָּה תִּקְּנוּ בְּסֶלַע שֶׁל תּוֹרָה שֶׁכֻּלּוֹ כֶּסֶף טָהוֹר, דְּזֶה הַכְּלָל לֹא נֶאֱמַר אֶלָּא בְּסֶלַע, וְלֹא בְזוּזֵי וּמָנֶה: {ז} <B> וַחֲלוּצָה.</B> לִרְבוּתָא נָקַט, דְּסָלְקָא דַעְתָּךְ אֲמִינָא כֵּיוָן דְּשׁוֹמֶרֶת יָבָם לֹא מִחַסְּרָה כְּנִיסָה לַחֻפָּה שֶׁהֲרֵי יְבָמָהּ יָבֹא עָלֶיהָ בְּעַל כָּרְחָהּ וַהֲוָה אֲמִינָא דְּכִנְשׂוּאָה דַּמְיָא, קָמַשְׁמַע לָן. תּוֹסָפוֹת: {ח} <B> שֶׁנִּפְדּוּ כוּ'.</B> בְּשֵׁם רַבֵּנוּ מוהר"ר לִיוָאֵי ז"ל שָׁמַעְתִּי טַעַם שֶׁשּׁוֹנֶה שֶׁנִּפְדּוּ [וְכוּ'], שֶׁכֵּן מִצְוַת פִּדְיוֹן שְׁבוּיִם קוֹדֶמֶת. וּלְהִתְגַּיֵּר אַף עַל פִּי שֶׁדּוֹחִין אוֹתוֹ בִּתְחִלָּה, אֵין מַרְבִּין עָלָיו וְאֵין מְדַקְדְּקִין עָלָיו, אֲבָל לְשַׁחְרֵר עוֹבֵר בַּעֲשֵׂה דִּוְהִתְנַחַלְתֶּם אוֹתָם: {ט} <B> פְּחוּתוֹת כוּ'.</B> שֶׁאֲפִלּוּ נִבְעֲלוּ, בְּתוּלֵיהֶן חוֹזְרִין. רַ"ן:

{י} וּמִכָּל מָקוֹם לֹא דָמִי לִקְטַנָּה הַנִּבְעֶלֶת, דְּהָתָם בְּתוּלֶיהָ חוֹזְרִין, אֲבָל קָטָן וְהִשִּׁיר הַבְּתוּלִין לֹא עֲדִיפָא מִמֻּכַּת עֵץ, אֶלָּא הָכָא בְּלֹא הִשִּׁיר בְּתוּלִים אַיְרֵי. תּוֹסָפוֹת וְהָרֹא"שׁ: 

~ פרק א - משנה ד

{יא} <B> הַנִּשּׂוּאִין.</B> שֶׁנִּכְנְסָה לַחֻפָּה וְיֵשׁ לָהּ עֵדִים שֶׁלֹּא נִסְתְּרָה. אִי נַמִּי נִסְתְּרָה וְלֹא שָׁהֲתָה כְּדֵי בִיאָה. גְּמָרָא: {יב} <B> וְאֵין כוּ'.</B> אֵין יָכוֹל לְהַפְסִידָהּ אוֹתָהּ מָנֶה וְלוֹמַר בְּחֶזְקַת בְּתוּלָה נְשָׂאתִיךְ וַהֲרֵי מִקְחִי מֶקַּח טָעוּת. רַשִׁ"י: 

~ פרק א - משנה ה

{יג} <B> הָאוֹכֵל.</B> מִדְּקָתָנֵי הָאוֹכֵל, מִכְּלַל דְּאִיכָּא דוּכְתָּא בִּיהוּדָה [נַמִּי] (עַצְמוֹ) דְּלֹא אָכֵיל, וְנַפְקָא מִנָּהּ דְּהֵיכָא דְּלֹא נְהִיגֵי לְיַחֵד יָכוֹל לִטְעֹן. גְּמָרָא וְרַשִׁ"י: {יד} <B> אַלְמְנַת כֹּהֵן.</B> אַלְמָנָה בַת כֹּהֵן. רַשִׁ"י: {טו} <B> לַבְּתוּלָה.</B> בַּת כֹּהֵן כְּשֶׁנִּשֵּׂאת לְכֹהֵן. וְכָל שֶׁכֵּן כְּשֶׁנִּשֵּׂאת לְיִשְׂרָאֵל: {טז} <B> וְלֹא מִחוּ כוּ'.</B> אַף עַל פִּי שֶׁלֹּא כָתַב בִּלְשׁוֹן תּוֹסֶפֶת, אֶלָּא אַרְבַּע מֵאוֹת דְּחָזוּ לִיכִי, וְקָא סָלְקָא דַעְתָּךְ דְּלֹא תִגְבֶּה דְּלֹא חָזוּ [לָהּ], וּבִלְשׁוֹן תּוֹסֶפֶת לֹא כָתַב, קָמַשְׁמַע לָן דְּמִנְהָג טוֹב [הוּא] וְרָאוּי לִהְיוֹת בַּכְּהֻנָּה, וְשַׁיָּךְ בְּהוּ דַּחֲזוּ לִיכִי. תּוֹסָפוֹת: 

~ פרק א - משנה ו

{יז} <B> לֹא כִי.</B> אִי אֶפְשָׁר לְפָרֵשׁ שֶׁמַּכְחִישָׁהּ בְּבָרִי, דְּמֵהֵיכָא יָדַע, וְעוֹד, לָמָּה כְּנָסָהּ וּבְעָלָהּ. אֶלָּא הָכִי קָאָמַר, שֶׁמָּא לֹא כִי: {יח} אֲפִלּוּ מָנֶה.</B> דְּלֹא דָמִי לְמֻכַּת עֵץ דִּלְעֵיל. גְּמָרָא: {יט} <B> נֶאֱמֶנֶת.</B> דְּמוֹקְמִינַן אִתְּתָא אַחֲזָקָה. דִּמְסַפְּקִינַן אַשְּׁעַת אֵרוּסִין אִי בְּתוּלָה הֲוַאי, וְהַעֲמֵד הָאִשָּׁה אַחֶזְקָתָהּ וּבְתוּלָה נוֹלְדָה, וְהַךְ שָׁעָה הוּא דְּאִיתְנִיסָא לְאַחַר אֵרוּסִין. גְּמָרָא: {כ}  הָרַמְבַּ"ם. אֲבָל בְּחִבּוּרוֹ כָּתַב, וְיֵשׁ לוֹ לְהַחֲרִים סְתָם עַל מִי שֶׁטּוֹעֵן שֶׁקֶר כוּ'. וְזֶה עִקָּר, דְּכֵיוָן שֶׁאֵינוֹ טוֹעֵן עָלֶיהָ בְּבָרִי, אֵין כָּאן מָקוֹם לִשְׁבוּעָה אֲפִלּוּ הֶסֵּת, שֶׁלֹּא מְצָאנוּהָ אֶלָּא בְּבָרִי: {כא}  גְּמָרָא. וְכָתַב הָרֹא"שׁ וְאֵין לְהַאֲמִינָהּ בְּמִגּוֹ דְנֶאֱנַסְתִּי, דְּלֹא בַעְיָא לְמִפְסָל נַפְשָׁהּ מִכְּהֻנָּה. וְאֵין לְהַאֲמִין בְּמִגּוֹ דְמֻכַּת עֵץ, דְּלֹא שְׁכִיחַ. אִי נַמִּי, לֹא אָמְרִינַן מִגּוֹ לְאַפּוּקֵי מָמוֹנָא: {כב} פֵּרֵשׁ רַשִׁ"י, וְאֵין לָחוּשׁ שֶׁמָּא יִטְעֹן אָדָם כָּךְ אֶלָּא בֶּאֱמֶת, שֶׁאִם הָיָה שׂוֹנְאָהּ מִתְּחִלָּה לֹא הָיָה טוֹרֵחַ בַּסְּעֻדָּה בְּחִנָּם, אֶלָּא הָיָה מְגָרְשָׁהּ תְּחִלָּה. וְכָתַב הָרַ"ן דְּאֵין לְהַקְשׁוֹת דְּתֵיפּוֹק לֵיהּ דְּהוֹאִיל וְאִיהוּ טוֹעֵן נַמִּי בָּרִי, מֵהֵיכָא תֵּיתֵי שֶׁתְּהֵא הִיא נֶאֱמֶנֶת לְאַפּוּקֵי מָמוֹנָא. דְּלָאו קוּשְׁיָא הִיא כְּלָל, דְּכֵיוָן דְּטַעֲמָא דִכְתֻבָּה כְּדֵי שֶׁלֹּא תְהֵא קַלָּה בְּעֵינָיו לְהוֹצִיאָהּ, אִם אַתָּה אוֹמֵר נֶאֱמָן וַעֲלָהּ דִּידָהּ רָמֵי לְבֵרוּרֵי וְאִי אֶפְשָׁר לָהּ לְהָבִיא רְאָיָה, הֲרֵי הִיא קַלָּה בְּעֵינָיו לְהוֹצִיאָהּ, הִלְכָּךְ צָרִיךְ לְטַעֲמָא דַּחֲזָקָה וְכוּ': {כג} <B> לֹא כוּ'.</B> דְּהָכָא נָמִי אוֹקֵי מָמוֹנָא בְּחֶזְקַת מָרֵיהּ:

{א} <B> לְהוֹסִיף.</B> מַשְׁמַע נוֹסָף עַל הַכְּתֻבָּה שֶׁתִּקְּנוּ חֲכָמִים וְשֵׁם כְּתֻבָּה עָלָיו גַּם הוּא לַמּוֹכֶרֶת וְלַמּוֹחֶלֶת כוּ'. בַּגְּמָרָא: {ב} <B> רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה כוּ'.</B> מַסְקִינַן בַּגְּמָרָא הֲלָכָה לְמַעֲשֶׂה כְּרַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה: {ג} <B> כּוֹתֶבֶת.</B> מְפָרֵשׁ בַּגְּמָרָא דְּאִלּוּ בַּאֲמִירָה בְּעָלְמָא הֲוֵי כְּמַתְנֶה עַל מַה שֶּׁכָּתוּב בַּתּוֹרָה דִּתְנָאוֹ בָּטֵל, וְרַבִּי יְהוּדָה סְבִירָא לֵיהּ נַמִּי דִּכְתֻבָּה דְּרַבָּנָן כִּדְקַיְמָא לָן בְּסוֹף מְכִילְתָּא, אֶלָּא דַּחֲכָמִים עָשׂוּ חִזּוּק לְדִבְרֵיהֶם כְּשֶׁל תּוֹרָה:

{ד} אוֹ עָשׂוֹר, וְלֹא מִשְׁתַּעֵי אֱינִישׁ מֵעִקָּרָא תְּרֵי יוֹמֵי וְאָמַר לֵיהּ לֹא וְהָדָר לִישְׁתַּעֵי אוֹ עָשׂוֹר, אֶלָּא מַאי יָמִים, שָׁנָה כוּ'. גְּמָרָא: {ה} <B> לָאִישׁ.</B> כַּמָּה שֶׁנּוֹתְנִין לָהּ, אִם שְׁנֵים עָשָׂר חֹדֶשׁ שְׁנֵים עָשָׂר חֹדֶשׁ, וְאִם שְׁלֹשִׁים יוֹם שְׁלֹשִׁים יוֹם. וְאֵין מַשְׁגִּיחִין בּוֹ אִם הוּא בָּחוּר אוֹ אַלְמָן. הָרַמְבַּ"ם: {ו} <B> וּלְאַלְמָנָה.</B> פֵּרְשׁוּ הַפּוֹסְקִים מִן הַנִּשּׂוּאִין: {ז} <B> נוֹתְנִין כוּ'.</B> בַּמֶּה דְבָרִים אֲמוּרִים, בְּבַת כֹּהֵן לְכֹהֵן (בְּקִיאָה בִּשְׁמִירַת תְּרוּמָה וּלְמָכְרָהּ בִּימֵי טֻמְאָה). אֲבָל בַּת יִשְׂרָאֵל לְכֹהֵן דִּבְרֵי הַכֹּל מֶחֱצָה חֻלִּין וּמֶחֱצָה תְּרוּמָה. בַּמֶּה דְבָרִים אֲמוּרִים, בַּאֲרוּסָה (וּבַת כֹּהֵן) דְּהוֹאִיל וְהִיא בְּבֵית אָבִיהָ טָרַח אָבִיהָ לְמָכְרָהּ בִּימֵי טֻמְאָה לְפִי שֶׁהוּא רָגִיל בְּכָךְ, אֲבָל בִּנְשׂוּאָה (שֶׁנּוֹתֵן לָהּ בַּעְלָהּ מְזוֹנוֹת בְּבֵיתָהּ), דִּבְרֵי הַכֹּל מֶחֱצָה חֻלִּין וּמֶחֱצָה תְּרוּמָה, דְּכָל כְּבוּדָּה בַּת מֶלֶךְ פְּנִימָה, וְאָבִיהָ וְאַחֶיהָ אֵינָן אֶצְלָהּ שֶׁיִּתְעַסְּקוּ בִצְרָכֶיהָ. גְּמָרָא:

{ח} כֵּן כָּתַב רַשִׁ"י. וְהִקְשָׁה רַבֵּנוּ תַּם, דְּאִם כֵּן אֲפִלּוּ אָכְלָה כַּמָּה שָׁנִים בִּפְנֵי הַבַּעַל אֵינָהּ אוֹכֶלֶת בִּפְנֵי הַיָּבָם, וְהֵיכִי קָתָנֵי חָסֵר יוֹם אֶחָד בִּפְנֵי הַיָּבָם. וְעוֹד, דַּהֲוָה לֵיהּ לְמִתְנֵי הַיְבָמָה אֵינָהּ אוֹכֶלֶת כוּ'. לְפִיכָךְ פֵּרֵשׁ רַבֵּנוּ תַּם דְּדַוְקָא כְּשֶׁלֹּא אָכְלָה בִּפְנֵי הַבַּעַל הוּא דְּלֹא אָכְלָה בִּפְנֵי הַיָּבָם, דְּהַיָּבָם אֵינוֹ מַאֲכִיל, אֲבָל אָכְלָה בִּפְנֵי הַבַּעַל אוֹכֶלֶת נַמִּי בִּפְנֵי הַיָּבָם. וְטַעֲמָא עַיֵּן בְּתוֹסָפוֹת וּבְתוֹסְפוֹת יוֹם טוֹב: {ט} <B> שֶׁתִּכָּנֵס.</B> שֶׁמִּתְיַּחֵד עִמָּהּ וְיַפְרִישֶׁנָּה לוֹ. הָרַמְבַּ"ם. וְהָרַ"ן פֵּרֵשׁ, דְּחֻפָּה לָאו הַיְנוּ יִחוּד, אֶלָּא כָּל שֶׁהֱבִיאָהּ הַבַּעַל מִבֵּית אָבִיהָ לְבֵיתוֹ לְשֵׁם נִשּׂוּאִין: 

~ פרק ה - משנה ד

{י} וְהַמּוֹתָר קָנוּי לוֹ בְּמָעָה כֶסֶף שֶׁתִּקְּנוּ לוֹ חֲכָמִים שֶׁיִּתֵּן לָהּ בְּכָל שַׁבָּת מָעָה כֶסֶף לִצְרָכֶיהָ לְבַד הַמְּזוֹנוֹת. וּלְמַאי דְקַיְמָא לָן דִּיכוֹלָה לוֹמַר אֵינִי נִזּוֹנִית וְאֵינִי עוֹשָׂה, הוּא הַדִּין וְכָל שֶׁכֵּן שֶׁיְּכוֹלָה לוֹמַר אֵינִי נוֹטֶלֶת הַמָּעָה וְאֵינִי נוֹתֶנֶת מוֹתָר כוּ'. וְהָא דְקָדִישׁ, מוֹקֵי רַב וּשְׁמוּאֵל דְּהַיְנוּ לְאַחַר מִיתָה כְּשֶׁיִּירָשֶׁנָּה, וַאֲפִלּוּ אֵינוֹ מַעֲלֶה לָהּ מָעָה. וְנָקַט מוֹתָר דְּאַף עַל גַּב דְּלֹא שְׁכִיחַ דְּמַקְדִּישׁ דָּבָר שֶׁלֹּא בָא לָעוֹלָם, וְכָל שֶׁכֵּן מַעֲשֵׂה יָדֶיהָ עַצְמָן. הָרַ"ן. וּמַה שֶּׁכָּתַב הָרַ"ב וְלֹא הִקְדִּישׁ כוּ', רוֹצֶה לוֹמַר אֲפִלּוּ לֹא הִקְדִּישׁ כוּ', דְּאִלּוּ הִקְדִּישׁ כָּל שֶׁכֵּן הוּא דְקָדִישׁ לְאַחַר מִיתָה: 

~ פרק ה - משנה ה

{יא} רַשִׁ"י. וּפֵרוּשׁוֹ, נוֹתַנְתּוֹ לְאוֹתוֹ דָבָר שֶׁטּוֹחֶנֶת: {יב} <B> וְאוֹפָה.</B> פַּת. רַשִׁ"י: {יג} <B> בַּצֶּמֶר.</B> דְּאִלּוּ בַּפִּשְׁתָּן לֹא פְּסִיקָא לֵיהּ, דְּהַיְנוּ כִּיתָּנָא רוֹמִיתָא, שֶׁאֵינוֹ כּוֹפָהּ לַעֲשׂוֹת בּוֹ לְפִי שֶׁמַּסְרִיחַ אֶת הַפֶּה וּמְשַׁרְבֵּט אֶת הַשְּׂפָתַיִם. פֵּרֵשׁ רַשִׁ"י פּוֹשְׁטָן כְּמִין שַׁרְבִיט עַד שֶׁנַּעֲשׂוֹת גְּדוֹלוֹת, עַל יְדֵי שֶׁהִיא צְרִיכָה לִשְׁרוֹת הַחוּט תָּמִיד בְּרֹק. גְּמָרָא: {יד} <B> לֹא טוֹחֶנֶת.</B> גְּמָרָא, הָא שְׁאָרָא עָבְדָה. וְתֵימָא לֵיהּ עַיְלִית לָךְ אִשָּׁה אַחֶרֶת בִּמְקוֹמִי. מִשּׁוּם דְּאָמַר לָהּ הָא טָרְחֵי לְדִידִי וּלְדִידָהּ, קַמֵּי דִידָךְ מַאן טָרַח. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} לְשׁוֹן רַשִׁ"י, לֵילֵךְ בִּשְׁלִיחוּת לְהָבִיא לוֹ מִבַּיִת לָעֲלִיָּה: {טז} בַּגְּמָרָא. וּפֵרֵשׁ רַשִׁ"י, לִפְרֹס לוֹ סָדִין וּלְבָדִין, דָּבָר שֶׁאֵינוֹ טֹרַח. וּבִכְפִיָּה הָנֵי לֹא כָפֵי לָהּ, אֶלָּא חֲכָמִים הִשִּׂיאוּהָ עֵצָה טוֹבָה לְהַנְהִיג זֹאת בְּיִשְׂרָאֵל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} <B> אַף כוּ'.</B> כְּלוֹמַר, אַף עַל פִּי שֶׁבִּרְצוֹנוֹ אֵינָהּ עוֹשָׂה מְלָאכָה, יוֹצִיא כוּ'. וְאֵין צָרִיךְ לוֹמַר שֶׁאִם אֵין רְצוֹנוֹ שֶׁלֹּא תַעֲשֶׂה מְלָאכָה שֶׁיָּכוֹל לְכוֹפָהּ. אֲבָל לְעִנְיַן הַטַּעַם לֹא קָאֵי עֲלֵיהּ אַף דְּטַעֲמוֹ שֶׁל רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל פּוֹחֵת מִטַּעֲמוֹ שֶׁל רַבִּי אֱלִיעֶזֶר: {יח} <B> הַמַּדִּיר.</B> וְדַע שֶׁבְּכָל מָקוֹם שֶׁתִּרְאֶה הַמַּדִּיר אֶת אִשְׁתּוֹ, רוֹצֶה לוֹמַר שֶׁהוּא אוֹסֵר הֲנָאָתוֹ עָלֶיהָ אִם תַּעֲשֶׂה כָּךְ. אֲבָל אִם אוֹמֵר לָהּ הֲרֵינִי אוֹסֵר עָלַיִךְ שֶׁלֹּא תַעֲשֶׂה כָּךְ אֵין בִּדְבָרָיו כְּלוּם. הַמַּגִּיד: {יט} <B> יוֹצִיא.</B> לֹא נִזְכַּר בַּגְּמָרָא כַּמָּה זְמַן הַהַדָּרָה. וּבַטּוּר כָּתַב שִׁבְעָה יָמִים. וְהַיְנוּ בְּהִדִּירָהּ מֵהֲנָאָתוֹ. אֲבָל מִנְּכָסָיו עַיֵּן רֵישׁ פֶּרֶק ז'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כ}  גְּמָרָא. וְיָהֵיב טַעֲמָא, דְּאֵינוֹ דוֹמֶה מִי שֶׁיֶּשׁ לוֹ פַּת בְּסַלּוֹ לְמִי שֶׁאֵין לוֹ פַּת בְּסַלּוֹ. וּפֵרֵשׁ רַשִׁ"י, אַף כָּאן, בְּלֹא נֶדֶר דַּעְתָּהּ נוֹחָה לְבַעְלָהּ שֶׁמָּא יָבֹא בְּתוֹךְ הַזְּמַן, אֲבָל זוֹ מִשֶּׁהִדִּירָהּ הֲרֵי הִיא כְּמִי שֶׁאֵינָהּ מְצַפָּה לַבַּעַל: {כא} <B> שְׁלֹשִׁים יוֹם.</B> אַף עַל פִּי שֶׁמְּבַטְּלִין בָּזֶה עוֹנָתָן כִּדְמוֹכָח מַתְנִיתִין. וּבַגְּמָרָא, שֶׁעוֹנַת תַּלְמִידֵי חֲכָמִים מֵעֶרֶב שַׁבָּת לְעֶרֶב שַׁבָּת, וּפֵרֵשׁ רַשִׁ"י שֶׁהוּא לֵיל תַּעֲנוּג וְשִׂמְחָה וּשְׁבִיתָה כוּ'. וְדַעַת הָרַ"ן דִּלְכָךְ לֹא הֻזְכַּר בְּמַתְנִיתִין עוֹנַת תַּלְמִידֵי חֲכָמִים, לְפִי דְעוֹנָתָן מִשְׁתַּנֵּית כְּפִי מַה שֶּׁהֵם צְרִיכִים בְּעֵסֶק לִמּוּדָם לְנַדֵּד שֵׁנָה מֵעֵינֵיהֶם, וְאֵין כָּל הַזְּמַנִּים שָׁוִים בָּזֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} <B> הָאֲמוּרָה כוּ'.</B> כְּלוֹמַר מִצְוַת עוֹנָה שֶׁהִיא אֲמוּרָה בַתּוֹרָה, כָּךְ הִיא הַמִּדָּה. אֲבָל אֵין מִלַּת הָאֲמוּרָה מוּסֶבֶת עַל זְמַנָּם שֶׁל כָּל אֶחָד וְאֶחָד, שֶׁזֶּה לֹא מִקְרֵי אֲמוּרָה בַתּוֹרָה, אַף עַל פִּי שֶׁיֵּשׁ לָהֶם סֶמֶךְ מִן הַמִּקְרָאוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} וְזֶה לְשׁוֹן הַתּוֹסָפוֹת, עַל שֵׁם שֶׁדַּרְכָּם לֵישֵׁב בְּטוּלָא דְאַפַּדְנַיְיהוּ:

{כא} דְּמִידֵי דְקָפְדֵי בֵיהּ אֱינָשֵׁי הֲוֵי קְפֵידֵיהּ קְפֵידָא, מִידֵי דְּלֹא קָפְדֵי אֱינָשֵׁי לֹא הֲוֵי קְפֵידֵיהּ קְפֵידָא. גְּמָרָא. וְאִי פֵרֵשׁ, אֲפִלּוּ לֹא קָפְדֵי. תּוֹסָפוֹת. וְכָתַב הָרַ"ן, וְהוּא הַדִּין בִּדְבָרִים שֶׁבֵּינוֹ לְבֵינָהּ אַף עַל גַּב דְּלֹא פֵרֵשׁ: {כב} <B> סְתָם.</B> אַרֵישָׁא קָאֵי, דְּקִדְּשָׁהּ עַל תְּנַאי וּכְנָסָהּ סְתָם, דְּדִלְמָא אַחוּלֵי אַחֲלֵיהּ לִתְנָאָה כֵּיוָן שֶׁכְּנָסָהּ בִּסְתָם. כִּדְרַב בַּגְּמָרָא: {כג} <B> שֶׁלֹּא כוּ'.</B> וְגֵט מִיהָא בַּעְיָא, דִּמְסַפְּקָא לָן אִי אַחוּלֵי אַחֲלֵיהּ לִתְנָאָה, וּלְאִסּוּרָא לְחֻמְרָא, וּלְמָמוֹנָא הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה. הָרַ"ן: {כד} <B> עַל מְנָת כוּ'.</B> אֲפִלּוּ בִסְתָם נַמִּי. דְּהָא בְּפֶרֶק ה' מִשְׁנָה ג' גָּזְרִינַן דַּאֲרוּסָה בַּת יִשְׂרָאֵל לְכֹהֵן לֹא תֹאכַל בַּתְּרוּמָה שֶׁמָּא יִמְצָא בָהּ מוּם כוּ'. וּמִשּׁוּם סֵיפָא דִּכְנָסָהּ סְתָם, נָקַט קִדְּשָׁהּ עַל מְנָת, לְאַשְׁמְעִינַן דְּאַף עַל גַּב דְּקִדְּשָׁהּ עַל מְנָת בַּעְיָא גֵּט. תּוֹסָפוֹת: {כה} דְּאַף דִּלְגַבֵּי כֹּהֲנִים אֶפְשָׁר דְּנָקַט פִּלְפְּלָא בְּפוּמֵיהּ וַעֲבֵיד עֲבוֹדָה. אֲבָל לְגַבֵּי אִשָּׁה שֶׁהוּא מְדַבֵּר עִמָּהּ בְּכָל שָׁעָה לֹא אֶפְשָׁר: {כו} רַשִׁ"י פֵּרֵשׁ שׁוּמָא וורוא"ה בְּלַעַ"ז. וְשָׁמַעְתִּי שֶׁהוּא בִּלְשׁוֹן אַשְׁכְּנַז וָוארְ"ץ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} <B> שֶׁנָּפְלוּ כוּ'.</B> כְּלוֹמַר בִּירֻשָּׁה. וְלָאו דַּוְקָא, אֶלָּא הוּא הַדִּין אֲפִלּוּ נִתְּנוּ לָהּ בְּמַתָּנָה אוֹ שֶׁמָּצְאָה מְצִיאָה, כָּל שֶׁשַּׁיָּךְ הַבַּעַל לִזְכּוֹת בָּהֶן מִדִּין נִכְסֵי מְלוֹג. רַשְׁבָּ"א בְּבֵית יוֹסֵף. וְהוּא הַדִּין בְּכָל מָקוֹם שֶׁשּׁוֹנֶה שֶׁנָּפְלוּ: {ב} <B> לֹא תִמְכֹּר.</B> דְּאֵרוּסִין עוֹשֶׂה סְפֵק נִשּׂוּאִין, שֶׁמָּא תָּבֹא לִידֵי נִשּׂוּאִין שֶׁמָּא לֹא תָבֹא, וּלְפִיכָךְ לְכַתְּחִלָּה לֹא תִמְכֹּר, וְאִם מָכְרָה וְנָתְנָה קַיָּם. גְּמָרָא. וְהִקְשׁוּ בַּתּוֹסָפוֹת, דְּכֵיוָן שֶׁהִיא מֻחְזֶקֶת הֲוֵי סָפֵק וּוַדַּאי וְאֵין סָפֵק מוֹצִיא כוּ', כִּדְאַשְׁכְּחָן בְּשׁוֹמֶרֶת יָבָם בְּמִשְׁנָה ו'. וְתֵרְצוּ, דְּהָכָא וַדַּאי קַיֶּמֶת לִנָּשֵׂא, מַה שֶּׁאֵין כֵּן הָתָם דְּכָךְ עוֹמֶדֶת לַחֲלִיצָה כְּמוֹ לְיִבּוּם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} <B> עַד כוּ'.</B> מַשְׁמַע שֶׁנָּפְלוּ לִפְנֵי נִשּׂוּאִין וְאֵין דָּבָר מַפְסִיק בֵּין נְפִילַת נְכָסִים לַנִּשּׂוּאִין. וְהוּא הַדִּין אֲפִלּוּ נָפְלוּ לָהּ קֹדֶם אֵרוּסִין דְּלֹא תִמְכֹּר מִשֶּׁנִּסֵּת. וְהָא דְנָקַט עַד שֶׁלֹּא נִסֵּת דְּמַשְׁמַע דְּמִשֶּׁנִּתְאָרְסָה, לְאַשְׁמְעִינַן דַּאֲפִלּוּ הָכִי אִם מָכְרָה וְנָתְנָה קַיָּם. תּוֹסָפוֹת:

{ד} שֶׁעַל מְנָת אוֹתָם נְכָסִים נְשָׂאָהּ, לְפִיכָךְ מִכְרָהּ בָּטֵל, שֶׁהָיָה מְצַפֶּה שֶׁתִּפֹּל לָהּ אוֹתָהּ יְרֻשָּׁה. רַשִׁ"י: {ה} <B> וְאִם כוּ'.</B> נִרְאֶה דְּאַכֻּלְּהוּ בָּבֵי דִּלְעֵיל מְהַדַּר וַאֲפִלּוּ אַמִּשֶּׁנִּסֵּת, דְּמִדְּתָלִינַן בְּאֵין יְדוּעִין לַבַּעַל דְּכֵיוָן שֶׁאֵינָן יְדוּעִים לוֹ לֹא זָכָה בָהֶן, אִם כֵּן מַה לִּי שֶׁנָּפְלוּ לָהּ קֹדֶם שֶׁנִּסֵּת אוֹ לְאַחַר שֶׁנִּסֵּת, כֵּיוָן שֶׁלֹּא יָדַע בָּהֶן לֹא זָכָה בְּהוּ:

{ו} וְתָלוּשׁ בִּשְׁעַת יְצִיאָה לֵיכָּא לְמֵימַר דִּפְלִיגֵי עֲלֵיהּ וְנֵימְרוּ שֶׁלָּהּ, דְּכֵיוָן דִּתְלָשָׁן הַכֹּל מוֹדִים שֶׁזָּכָה בָהֶן. רַשִׁ"י: 

~ פרק ח - משנה ה

{ז} <B> יִמָּכְרוּ כוּ'.</B> כְּלוֹמַר שֶׁאִם יִרְצֶה הָאִישׁ אוֹ הָאִשָּׁה לִמְכֹּר אֵין יְכוֹלִין לְעַכֵּב זֶה עַל זֶה. הָרְמָ"ה. וְהָרֹא"שׁ חוֹלֵק וְסוֹבֵר דְּהַכֹּל בִּרְשׁוּת הַבַּעַל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} <B> שְׁבַח כוּ'.</B> דְּהַוְיָא לָהּ וּלְבֵית אָבִיהָ לְשֵׁם וּלְתִפְאֶרֶת. רַשִׁ"י: {ט} <B> רַבִּי יְהוּדָה אוֹמֵר.</B> דְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אָמַר לְמִלְּתֵיהּ בְּרֵישָׁא וְרַבִּי יְהוּדָה אָמַר בְּסֵיפָא כוּ'. וּקְבָעָן רַבִּי בְּמִשְׁנָתֵנוּ כִּשְׁמוּעָתָן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} שֶׁכֵּן כָּל שְׁבוּעוֹת הַמִּשְׁנָה כְּעֵין שֶׁל תּוֹרָה תִּקְּנוּהָ: {יא} לָאו לְמֵימְרָא שֶׁהִיא שְׁבוּעָה מִן הַתּוֹרָה, דְּהָא כָּל הַנִּשְׁבָּעִין שֶׁבַּתּוֹרָה נִשְׁבָּעִין וְלֹא מְשַׁלְּמִין, אֶלָּא כְּעֵין שֶׁל תּוֹרָה קָאָמַר: {יב} דְּאַדַּעְתָּא לְמִשְׁקָל וּלְמֵיפַק לֹא אַקְנֵי [לָהּ]. הָרִי"ף:

{יג} <B> שֶׁמּוֹכֶרֶת.</B> בַּתְּחִלָּה. וְלֹא דָמִי לַאֲרוּסָה דִּבְרֵישׁ פִּרְקִין, לְפִי שֶׁאֲרוּסָה קְנוּיָה עַד שֶׁנִּתְחַיֵּב הַבָּא עָלֶיהָ סְקִילָה הִלְכָּךְ כְּמוֹ כֵן זָכָה בָהּ שֶׁאֵין לָהּ לִמְכֹּר לְכַתְּחִלָּה. אֲבָל שׁוֹמֶרֶת יָבָם שֶׁהַבָּא עָלֶיהָ לֹא נִתְחַיֵּב אֶלָּא מַלְקוֹת, אֵימוּר וַדַּאי דְּבִזְכוּתָהּ נָפְלוּ, הִלְכָּךְ מוֹכֶרֶת לְכַתְּחִלָּה. נִמּוּקֵי יוֹסֵף. וְעַיֵּן מַה שֶּׁכָּתַבְנוּ [בְּאוֹת ב] בְּשֵׁם הַתּוֹסָפוֹת:

{יד} פֵּרֵשׁ רַשִׁ"י, וְהַמְחֻבָּר לַקַּרְקַע כְּקַרְקַע. וְלֹא דָמִי לִמְחֻבָּרִים דְּנִכְסֵי מְלוֹג בְּמִשְׁנָה ג', דְּהָתָם פֵּירָא תַּקִּינוּ לֵיהּ רַבָּנָן הֵיכָא דְּאִיכָּא קֶרֶן קַיָּם, וְהַנֵּי נַמִּי פֵּירֵי נִינְהוּ דְּגָמְרוּ בִרְשׁוּתֵיהּ, אֲבָל הָכָא קַרְנָא וּפֵירֵי דִּידֵיהּ הוּא, וְשִׁעְבּוּדָא הוּא דְּאִית לָהּ עֲלַיְהוּ, וּמַאי דְאִשְׁתַּעְבֵּד לָהּ מֵחַיִּים דְּבַעַל לֹא אָמְרִינַן פֵּירֵי הוּא לְאַפְקוּעֵי שִׁעְבּוּדָא. וּמְשָׁנֵי, תָּנֵי שֶׁלָּהּ, וְלֹא פְּלִיגֵי רַבָּנָן אֶלָּא אַתְּלוּשִׁין וּכְסָפִים, וּלְמֵימְרָא דְּלֹא מִשַּׁעְבְּדֵי מִטַּלְטְלֵי לַכְּתֻבָּה [וְהָכָא לֹא פְּלִיגֵי רַבָּנָן. אֶלָּא מִשּׁוּם דִּלְעֵיל בְּמִשְׁנָה ג' בְּנִכְסֵי מְלוֹג פְּלִיגֵי רַבָּנָן בְּהָא, הִלְכָּךְ נָקְטִינְהוּ נַמִּי הָכָא. וּמִשּׁוּם כָּךְ אִישְׁתַּבַּשׁ תַּנָּא וְתָנָא שֶׁלּוֹ כְּדִלְעֵיל]: {טו} אַף עַל גַּב דִּכְשֶׁיִּבְּמָהּ מֻתֶּרֶת לוֹ בְּכָל עֵת כוּ', דִּסְבָרָא הוּא דְּלֹא אָמַר רַחֲמָנָא וְיִבְּמָהּ כְּדֵי לְגָרְשָׁהּ לְאַלְתַּר. תּוֹסָפוֹת: {טז} וְהַאי וְיִבְּמָהּ, דָּרְשִׁינַן לָהּ בִּיבָמוֹת לִבְעַל כָּרְחָהּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יז} <B> עַל הַשֻּׁלְחָן.</B> הֵן מָעוֹת שֶׁהוּא מִשְׁתַּמֵּשׁ בָּהֶן אִם הוּא שֻׁלְחָנִי. רַשִׁ"י: {יח} <B> וְכֵן לֹא כוּ'.</B> דְּמַהוּ דְתֵימָא דְטַעֲמָא דְּלֹא יְיַחֵד לָהּ לַיְבָמָה כְּדֵי לִמְכֹּר שְׁאָר נְכָסִים דְּאֵינוֹ יָכוֹל לְמָכְרָם, כֵּיוָן דְּלֹא כָתַב לָהּ דְּקַנָאִי, אֲבָל בְּאִשְׁתּוֹ לֹא שַׁיָּךְ לוֹמַר שֶׁלֹּא יוּכַל לִמְכֹּר מֵהַאי טַעֲמָא, דְּהָא כָּתַב לָהּ דְּקַנָאִי, וְסָמְכָה דַּעְתָּהּ כְּשֶׁמְּיַחֵד, קָמַשְׁמַע לָן. גְּמָרָא כְּפִי פֵּרוּשׁ הַתּוֹסָפוֹת: {יט} כְּלוֹמַר וְאֵינוֹ רַשַּׁאי לִמְכֹּר כְּלוּם. גְּמָרָא. דְּאִי מִשּׁוּם גּוּפָהּ פְּשִׁיטָא דְּאֵין לָהּ אֶלָּא כְּתֻבָּתָהּ, אֶלָּא וַדַּאי לְדִיּוּקָא דִּילֵיהּ תַּנְיָא לְמֵידַק הָא לֹא גֵרְשָׁהּ לֹא. וְרֵישָׁא דִּתְנַן כָּל נְכָסָיו אַחֲרָאִין, עֵצָה טוֹבָה קָמַשְׁמַע לָן, דְּהָא בְּאִשְׁתּוֹ דְּיָכוֹל לִמְכֹּר, וּתְנַן דְּלֹא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} כְּלוֹמַר אַף עַל גַּב דְּבִשְׁאָר הַנָּשִׁים תְּנַן לְהָא בְּסוֹף פֶּרֶק דִּלְקַמָּן: {כא} כְּלוֹמַר מִלְּבַד מַה שֶּׁיֵּשׁ לָהּ עַל הָרִאשׁוֹן, לְפִי שֶׁעַכְשָׁיו כְּשֶׁלְּקָחָהּ בְּנִשּׂוּאֵי עַצְמוֹ יִהְיֶה מְחֻיָּב לַעֲשׂוֹת לָהּ כְּתֻבָּה מִמֶּנּוּ, וְהָרִאשׁוֹנָה בִּמְקוֹמָהּ עוֹמֶדֶת. קָמַשְׁמַע לָן דְּאֵין לָהּ אֶלָּא כְּתֻבָּתָהּ, פֵּרוּשׁ, הָאַחַת שֶׁהָיָה לָהּ כְּבָר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} וְהוּא הַדִּין לָאוֹמֵר בִּלְבָד, דְּמַהֲנֵי, כִּדְתָנֵי רַבִּי חִיָּיא הָאוֹמֵר. גְּמָרָא: {ב} פֵּרֵשׁ רַשִׁ"י, כֵּיוָן שֶׁאֵינָהּ יְרֻשַּׁת אֲבוֹתָיו אֶלָּא עַל יְדֵי מַעֲשָׂיו תָּבֹא לוֹ, כְּגוֹן נַחֲלַת אִשְׁתּוֹ הַבָּאָה לוֹ עַל יְדֵי נִשּׂוּאִין שֶׁלּוֹ, דְּהוֹאִיל וּמִשּׁוּם תַּקַּנְתָּא דִידֵיהּ תַּקּוּן רַבָּנָן, וְהוּא בָא לִמְחֹל עָלֶיהָ, מָחוּל. וּמַקְשֶׁה בַגְּמָרָא, אִי הָכִי אֲפִלּוּ נְשׂוּאָה נַמִּי. וּמְשָׁנֵי, דְּכֵיוָן שֶׁנְּשָׂאָהּ הֲרֵי הוּא כְשֻׁתָּף בִּנְכָסֶיהָ, וְכֵיוָן שֶׁהַנְּכָסִים שֶׁלּוֹ צָרִיךְ לְשׁוֹן מַתָּנָה וְאֵין מוֹעִיל לְשׁוֹן דִּין וּדְבָרִים: {ג} וְהוּא הַדִּין נַמִּי לִשְׁאָר תְּנָאִים שֶׁיָּכוֹל לְהַתְנוֹת וּמַהֲנֵי: {ד} דְּמִגּוּפָהּ שֶׁל קַרְקַע קָנוּ מִיָּדוֹ, וּמַתָּנָה גְמוּרָה הִיא. גְּמָרָא: {ה} <B> דִּין מַהֲנֵי אֲפִלּוּ לִנְכָסִים שֶׁנָּפְלוּ לָהּ אַחַר שֶׁנִּשֵּׂאת, דִּלְכֻלֵּי עַלְמָא מִכְרָהּ בָּטֵל, וְאִם כָּתַב לָהּ דִּין וּדְבָרִים בְּעוֹדָהּ אֲרוּסָה מִכְרָהּ קַיָּם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} <B> עַד עוֹלָם.</B> דְּאִי לֹא כָתַב עַד עוֹלָם, הֲוָה אֲמִינָא פֵּרֵי דְפֵרֵי פֵרוֹת אָכֵיל, לְהָכִי אִצְטָרִיךְ עַד עוֹלָם. וְאִי כָתַב לָהּ עַד עוֹלָם וְלֹא כָתַב לָהּ פֵּרֵי פֵרוֹת, הֲוָה אֲמִינָא לְעוֹלָם אַפֵּרוֹתֵיהֶן קָאֵי. גְּמָרָא: {ט} <B> בָּטֵל.</B> מַסְקֵי הַפּוֹסְקִים, דְּהָכָא דִבְמָמוֹן, אֲפִלּוּ בְּשֶׁל תּוֹרָה תְּנָאוֹ קַיָּם:

{טו} <B> הַקּוֹדֵם.</B> וּבַגְּמָרָא אִיפְלִיגֵי אֲמוֹרָאֵי אִי בָעֵי דַּוְקָא שֶׁתָּפְסוּ מֵרְשׁוּת הָרַבִּים, שֶׁאֵינוֹ מְקוֹם קְנִיָּה [כְּלָל] וְלֵיכָּא לְמֵימַר שֶׁזָּכוּ בָהֶן יוֹרְשִׁים, אוֹ אֲפִלּוּ תָּפְסוּ מִסִּמְטָא, וְלֹא אָמְרִינַן דְּכֵיוָן שֶׁסִּמְטָא מָקוֹם רָאוּי לְקִנְיָן זָכוּ בָהֶן יוֹרְשִׁים מִשְּׁעַת מִיתָה: {טז} רוֹצֶה לוֹמַר שְׁטַר חוֹב אוֹ שְׁטַר כְּתֻבָּה, דִּשְׁנֵיהֶן כּוֹשֵׁל הֵן כְּנֶגֶד הַיְתוֹמִים: {יז} <B>יִנָּתְנוּ לַיּוֹרְשִׁים וְכוּ'.</B> מְפָרֵשׁ בַּגְּמָרָא דִּלְרַבִּי עֲקִיבָא אֲפִלּוּ כֻּלְּהוּ נַמִּי דְיַתְמֵי, וְאִי תָּפַס לֹא מַהֲנֵי וּמַפְּקִינַן מִנֵּיהּ. וְאַיְדֵי כוּ':

{ה} <B> לְאַחֵר.</B> וְכָל שֶׁכֵּן כְּשֶׁמָּחֲלְתָּהּ לַיְתוֹמִים, דְּכֵיוָן שֶׁאֵין לָהּ עֲלֵיהֶן כְּתֻבָּה הָא אִסְתַּלְּקָא מִבֵּיתַיְהוּ וְאֵין לָהּ עֲלֵיהֶם מְזוֹנוֹת. הָרַ"ן: {ו} <B> שֶׁלֹּא בְּבֵית דִּין.</B> אַשְׁמְעִינַן שֶׁמּוֹכֶרֶת לִמְזוֹנוֹת וְלֹא אִבְּדָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} <B> לִמְזוֹנוֹת.</B> בַּגְּמָרָא מְפָרְשִׁים דְּעֵצָה טוֹבָה קָמַשְׁמַע לָן, דְּלֹא לִקְּרוֹ לָהּ רַעַבְתָּנוּתָא, שֶׁהָרוֹאֶה שֶׁמּוֹכֶרֶת כָּל קַרְקָעוֹת אֵלּוּ סָבַר שֶׁמּוֹכַרְתָּן לִמְזוֹנוֹת מוֹצִיא עָלֶיהָ שֵׁם רַעַבְתָּנוּת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ח} <B> שָׁוֶה כוּ'.</B> מַחֲלֹקֶת הַפּוֹסְקִים, אִיכָּא דְסָבְרֵי דְּאֵין אוֹנָאָה לְקַרְקָעוֹת כְּלָל, וּלְדִידְהוּ מַתְנִיתִין דְּהָכָא כִּפְשָׁטָהּ. וְאִיכָּא דְסָבְרֵי דְּאֵין אוֹנָאָה עַד פְּלַג אֲבָל בִּפְלַג אִיתָא אוֹנָאָה. וּמְפָרֵשׁ בַּיְרוּשַׁלְמִי, דְּהָכָא שָׁוֶה מָנֶה בְּמָאתַיִם מַיְרֵי שֶׁהוּקַר וְעָמַד בְּמָאתַיִם וְהַלּוֹקֵחַ רוֹצֶה לְקַיֵּם הַמֶּקָּח, וְשָׁוֶה מָאתַיִם בְּמָנֶה כְּשֶׁהוּזַל וְעוֹמֵד עַל מָנֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} דְּהַמּוֹתָר מַתָּנָה הִיא, וְיֵשׁ לוֹמַר לַשָּׁלִיחַ נִתְּנָה, וְיֵשׁ לוֹמַר לַמְשַׁלֵּחַ. רַשִׁ"י: {י} <B> וְדִינָר.</B> יֵשׁ מְפָרְשִׁים דִּינָר דַּוְקָא. וְלֹא נְהִירָא, כֵּיוָן דַּאֲפִלּוּ פָּחוֹת מִכְּדֵי אוֹנָאָה בָּטֵל, אֵין חִלּוּק בֵּין דִּינָר לְפָחוֹת. הָרֹא"שׁ: {יא} מְסַיֵּם רַשִׁ"י, אוֹ הָאוֹנָאָה עַצְמָהּ תִּשְׁעָה קַבִּין. וְהוּא הַדִּין אִם נִשְׁאַר לַיְתוֹמִים שָׁם לְבַד הָאוֹנָאָה תִּשְׁעָה קַבִּין, דְּכֵיוָן דְּיֵשׁ לַיְתוֹמִים קַרְקַע שָׁם כְּשִׁעוּר שָׂדֶה, יֹאמְרוּ אֵין רְצוֹנֵנוּ לִמְכֹּר קַרְקַע הָרָאוּי לָנוּ. אֲבָל אִם אֵין הָאוֹנָאָה רָאוּי לְהִצְטָרֵף לִכְדֵי שָׂדֶה, לֹא הִפְסִידָתָן כְּלָל: {יב} <B> רַבִּי עֲקִיבָא.</B> שֶׁשָּׁמַעְנוּ מִמֶּנּוּ בְּסוֹף פֶּרֶק קַמָּא דְבָבָא בַתְרָא דְּבֵית רֹבַע הוּא שִׁעוּר גִּנָּה. רַשִׁ"י: {יג} <B> אַרְבַּע מֵאוֹת.</B> אִצְטָרִיךְ, דְּלֹא תֵימָא דַּוְקָא בְּרֵישָׁא דְּכִי מָכְרָה מָנֶה בְּמָאתַיִם דְּאִסְתַּלְּקָא מֵהַאי בֵיתָא לְגַמְרֵי, אֲבָל הָכָא נֵימָא דְּאִם טָעֲתָה בְּמָנֶה רִאשׁוֹן יְהֵא מִכְרָהּ בָּטֵל אַטּוּ מָנֶה אַחֲרוֹן, דְּלֹא נֵיתֵי לְמֵימַר מִכְרָהּ קַיָּם, קָמַשְׁמַע לָן. גְּמָרָא, וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כ} <B> הַשְּׁנִיָּה.</B> בֵּין הִכִּיר בָּהּ בֵּין לֹא הִכִּיר בָּהּ: {כא} דְּאִשָּׁה רוֹצָה לְהִנָּשֵׂא יוֹתֵר מֵהָאִישׁ. רַשִׁ"י: {כב} וּבִמְמָאֶנֶת הַיְנוּ טַעֲמָא מִשּׁוּם דְּכֵיוָן דְּתַקּוּן רַבָּנָן נִשּׂוּאִין לִקְטַנָּה מִשּׁוּם תַּקַּנְתָּא כְּדֵי שֶׁלֹּא יִנְהֲגוּ בָהּ מִנְהַג הֶפְקֵר, זִכּוּ פֵּרוֹת נְכָסִים לַבַּעַל. וּבִשְׁנִיָּה מִשּׁוּם קְנָס. וּבְאַיְלוֹנִית הַיְנוּ טַעֲמָא מִשּׁוּם דְּאַף עַל גַּב דְּנִשּׂוּאֵי טָעוּת הוּא כֵּיוָן שֶׁלֹּא הִכִּיר בָּהּ, אֲפִלּוּ הָכִי כֵּיוָן שֶׁרָאֲתָה הָאִשָּׁה בַּעְלָהּ אוֹכֵל פֵּרוֹת וְשָׁתְקָה, מָחֲלָה. הָרַ"ן: {כג}  אַיְלוֹנִית כִּשְׁנִיָּה. הָרַמְבַּ"ם. וְאִי בְּיוֹשֶׁבֶת תַּחְתָּיו לְאַחַר שֶׁנִּמְצֵאת אַיְלוֹנִית, אַמַּאי לֵית לָהּ, וַהֲרֵי הִיא כִּנְשָׂאָהּ מִתְּחִלָּה לְשֵׁם אַיְלוֹנִית, שֶׁיֶּשׁ לָהּ מְזוֹנוֹת. הָרַ"ן: {כד} דְּנִכְסֵי צֹאן בַּרְזֶל, יָכוֹל לוֹמַר שֶׁלִּי הוֹצֵאתִי, שֶׁהֲרֵי בִרְשׁוּת בֵּית דִּין נְשָׂאתִיהָ, וּכְשֶׁאוֹצִיאֶנָּה מִדַּעְתִּי אַחֲזִיר מַה שֶּׁקִּבַּלְתִּי. וְנִכְסֵי מְלוֹג, קַיְמָא לָן דְּעַיְלָא גְלִימָא פֵּרֵי הֲוֵי וּמְכַסֵּי בֵיהּ עַד דְּבָלֵי, וּלְפִיכָךְ כְּשֶׁהוֹצִיאָם בִּרְשׁוּת הוֹצִיאָם. הָרַ"ן: {כה} שֶׁזֶּה הַדִּין אָמְנָם קְנָס הוּא לַשְּׁנִיָּה עַל הָעֲבֵרָה, לְפִי שֶׁהַדַּעַת נוֹטָה אַחַר שֶׁאֵין לָהּ כְּתֻבָּה שֶׁלֹּא יִהְיֶה לַבַּעַל חֻקֵּי הַנִּשּׂוּאִין וְלֹא יֵהָנֶה מִנִּכְסֵי מְלוֹג, אֲבָל מִצַּד הַקְּנָס עָשִׂינוּ נִכְסֵי מְלוֹג שֶׁלָּהּ כְּנִכְסֵי שְׁאָר הַנָּשִׁים, שֶׁהָעִקָּר אֶצְלֵנוּ עַיְלָא גְלִימָא פֵּרֵי הֲוֵי. הָרַמְבַּ"ם: {כו} <B> כְּתֻבָּה.</B> וְהוּא הַדִּין לְכָל תְּנָאֶיהָ, דִּתְנַאי כְתֻבָּה כִּכְתֻבָּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כז} <B> לְכֹהֵן גָּדוֹל.</B> אַהִכִּיר בָּהּ דְּאַיְלוֹנִית קָאֵי, וְקָאָמַר דְּאַלְמָנָה כוּ' נַמִּי כִּי הִכִּיר בָּהּ יֶשׁ לָהּ כְּתֻבָּה. גְּמָרָא:

{א} דְּאִי לֹא אָמַר אַתֶּם עֵדַי פָּטוּר, מִטַּעַם שֶׁלֹּא לְהַשְׁבִּיעַ אֶת עַצְמוֹ כִּדְפֵרֵשׁ הָרַ"ב בְּמִשְׁנָה ו' פֶּרֶק ג' דְּסַנְהֶדְרִין. וְהַתּוֹסָפוֹת כָּתְבוּ דְּפָטוּר גַּם כֵּן מִטַּעַם מְשַׁטֶּה אֲנִי בָךְ. אֲבָל בְּטוּר חֹשֶׁן מִשְׁפָּט סִימָן פ"א כָּתַב בְּלֹא שׁוּם חוֹלֵק דְּלֵיתָא טַעֲנָה זוֹ אֶלָּא כְּשֶׁהָיְתָה הַהוֹדָאָה עַל יְדֵי תְבִיעָה. וּבַנִּידוֹן דִּידָן לֹא מַשְׁמַע שֶׁהָיְתָה שׁוּם תְּבִיעָה, וּמִכִּי אָמַר מִיהַת מִתְחַיֵּב אַף עַל פִּי שֶׁמֵּעוֹלָם לֹא הָיָה חַיָּב לָהּ כְּלוּם. הָרַמְבָּ"ם: {ב} <B> חָמֵשׁ שָׁנִים.</B> דִּבְּרוּ בַּהֹוֶה, שֶׁכֵּן מַקְפֶּדֶת שֶׁתְּהֵא נִזּוֹנֶת עַד שֶׁתְּהֵא בַת שֶׁבַע שָׁנִים, וְאֵין הָאִשָּׁה מִתְקַדֶּשֶׁת עַד אַחַר זְמַן יְנִיקָה דְּהַיְנוּ עֶשְׂרִים וְאַרְבָּעָה חֹדֶשׁ כִּדְאִיתָא בְּאֶבֶן הָעֵזֶר סִימָן י"ג. {ד} <B> הַבַּעַל כוּ'.</B> סָלְקָא דַעְתָּךְ דְּלֹא מִחַיַּב, דְּכֵיוָן דְּיֵשׁ לָהּ מְזוֹנוֹת מֵאַחֵר, כְּאִלּוּ הִתְנָה עִמָּהּ עַל מְנָת שֶׁאֵין לָהּ עָלָיו מְזוֹנוֹת, קָמַשְׁמַע לָן. תּוֹסָפוֹת: {ה} <B> וְהֵן כוּ'.</B> אִצְטָרִיךְ, סָלְקָא דַעְתָּךְ דְּלֹא פָסְקוּ אֶלָּא כָּל זְמַן שֶׁלֹּא נִשֵּׂאת. תּוֹסָפוֹת: {ו} <B> מֵתוּ.</B> מִשּׁוּם בְּנוֹתֵיהֶן דְּאֵין לָהֶם מְזוֹנוֹת אֶלָּא לְאַחַר מִיתָה אִצְטָרִיךְ לְמִתְנֵי אִם מֵתוּ, וּלְאַשְׁמוֹעִינַן דְּאֵין לָהֶם אֶלָּא מִבְּנֵי חוֹרִין וַאֲפִלּוּ קָנוּ מִיָּדוֹ. אֲבָל בְּבַת אִשְׁתּוֹ הוּא הַדִּין דַּאֲפִלּוּ בְּחַיֵּיהֶם הָיָה יָכוֹל לְהַשְׁמִיעֵנוּ שֶׁהִיא נִזּוֹנֶת מֵהַמְשֻׁעְבָּדִין. הַמַּגִּיד: {ז} <B> כְּבַעֲלַת חוֹב.</B> וּמִיהוּ לָאו בַּעֲלַת חוֹב מַמָּשׁ קָאָמַר, שֶׁאִם מֵתָה אֵין לָהּ מְזוֹנוֹת. וְלֹא דַמְיָא לְנִשֵּׂאת, דְּנִשֵּׂאת בַּת מְזוֹנוֹת, וְזוֹ אֵינָהּ כֵּן: {ח} הַיְנוּ כַּמַּסְקָנָא בַּגְּמָרָא דְּמוֹקֵי לְמַתְנִיתִין בְּשֶׁקָּנוּ מִיָּדוֹ אוֹ שֶׁחִיֵּב עַצְמוֹ בִּשְׁטָר, כְּלוֹמַר, אִם צִוָּה לִכְתֹּב שְׁטָר. אֲבָל בְּהַהוּא גַוְנָא שֶׁכָּתַב הָרַ"ב לְעֵיל שֶׁאָמַר לְעֵדִים אַתֶּם עֵדַי, לֹא, דְּמִלְוֶה עַל פֶּה בְּעֵדִים לֹא גָבֵי מִמְּשַׁעְבְּדֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ט} וּדְבַר תֵּימַהּ הוּא לְדַיֵּק הָכִי, וְכִי סַגִּי לֵיהּ לִכְתֹּב אַתְּ תְּהֵא יַתְבָא, אֶלָּא לֹא סַגִּי לֵיהּ בְּלָאו הָכִי וְלֵיכָּא לְמֵידָק מִנֵּיהּ כוּ'. אֲבָל בַּגְּמָרָא הָכִי אִיתָא, נָקְטִינַן מְדוֹר אַלְמָנָה שֶׁנָּפַל כוּ', שָׁמַעְתְּ מִנֵּיהּ דְּלֹא דִיֵּק כְּלָל, אֶלָּא דְּהָכִי נָקְטִינַן מִקַּמָּאי. תּוֹסְפוֹת יוֹם טוֹב. וְעַיֵּן עוֹד: {י} פֵּרֵשׁ רַשִׁ"י, בְּנֵי אָדָם שֶׁעוֹזְרִין זֶה אֶת זֶה וּמִשְׂתַּכְּרִין, וּמַזָּל דְּרַבִּים עָדִיף. וּלְשׁוֹן הָרַמְבַּ"ם, חֲמִשָּׁה שֶׁהָיָה מְזוֹנוֹת כָּל אֶחָד מֵהֶן קַב כְּשֶׁיֹּאכַל לְבַדּוֹ, אִם הָיוּ חֲמִשְׁתָּן בְּבַיִת אֶחָד וְאוֹכְלִין בְּעֵרוּב, מַסְפִּיק לָהֶן אַרְבָּעָה קַבִּין. וְהוּא הַדִּין לִשְׁאָר צָרְכֵי הַבַּיִת: {יא} <B> אֵין לִיךְ כוּ'.</B> כִּי אִם לְפִי בִּרְכַּת הַבַּיִת. אֲבָל בְּפוֹסֵק לְבַת אִשְׁתּוֹ דִּלְעֵיל תְּנַן דְּמוֹלִיךְ לָהּ מְזוֹנוֹתֶיהָ כְּלוֹמַר כָּל הַמְּזוֹנוֹת מִשָּׁלֵם, וְהַיְנוּ טַעֲמָא, דְּהָתָם מְחַיֵּב עַצְמוֹ בְּחוֹב, מַה שֶּׁאֵין כֵּן הָכָא דְּבֵין כָּתַב וּבֵין לֹא כָתַב מִתְחַיֵּב מִכֹּחַ תְּנַאי בֵית דִּין, וּמֵעִקָּרָא כִּי אִתְקּוּן הָכִי תַּקִּינוּ. תּוֹסְפוֹת יוֹם טוֹב:

{טו} וּכְשֶׁהִגִּיעַ זְמַן פִּרְעוֹן הַחוֹב קֹדֶם.</B> הָרַ"ן: {טז} דְּאַף שֶׁצָּרִיךְ זֶה לִמְכֹּר, הָיָה מוֹצֵא הַרְבֵּה, דְּקוֹנִים מְצוּיִים הֵם, דְּאָמְרֵי אֱנָשֵׁי, זָבֵין קָנִית זָבֵינְתְּ אַפְסַדְתְּ:

{יז} <B>זֶה עַל זֶה.</B> רְאוּבֵן עַל שִׁמְעוֹן וְשִׁמְעוֹן עַל רְאוּבֵן. וּשְׁטָר שֶׁהוֹצִיא רְאוּבֵן מֻקְדָּם וְשֶׁל שִׁמְעוֹן מְאֻחָר. אַדְמוֹן אוֹמֵר יָכוֹל לוֹמַר אִלּוּ כוּ' כֵּיצַד כוּ' הָיָה לְךָ לִתְבֹּעַ חוֹבְךָ מִמֶּנִּי. וַחֲכָמִים אוֹמְרִים זֶה גּוֹבֶה כוּ', שֶׁאַף הָרִאשׁוֹן גּוֹבֶה. וּפָרְכִינַן בַּגְּמָרָא, אִי דְמָטֵי זִמְנֵיהּ שֶׁל חוֹב הָרִאשׁוֹן מַאי טַעֲמָא דְרַבָּנָן שֶׁהָרִאשׁוֹן גּוֹבֶה חוֹבוֹ. וְאִי לֹא מָטֵי זִמְנָא מַאי טַעֲמָא דְאַדְמוֹן דְּהָא לֹא מָצֵי אָמַר לֵיהּ אִלּוּ כוּ', דְּמִשּׁוּם דְּלֹא הִגִּיעַ זְמַנּוֹ הֻצְרַךְ לִלְוֹת מִמֶּנּוּ. וּמוֹקְמִינַן לָהּ בַּגְּמָרָא כְּגוֹן שֶׁלָּוָה מִמֶּנּוּ בַּיּוֹם דְּמִשְׁלִים זִמְנֵיהּ. וּפְלִיגֵי אִי עֲבֵיד אֱינַשׁ דְּיַזִּיף לְיוֹמָא אוֹ לֹא. הָרַ"ן:

{יח} <B> שָׁלֹשׁ אֲרָצוֹת כוּ'.</B> הֵן בְּאֶרֶץ יִשְׂרָאֵל חֲלוּקוֹת לְעִנְיַן נִשּׂוּאֵי אִשָּׁה. רַשִׁ"י: {יט} וְהוּא מֵאוֹתָהּ הָאָרֶץ שֶׁנְּשָׂאָהּ:
