~ פרק א - משנה א

{א} <b>גֵּט.</b> לְשׁוֹן אֲרַמִּי. סֵפֶר כְּרִיתוּת (דְּבָרִים כד) מְתֻרְגָּם גֵּט פִּטּוּרִין. וּלְכָךְ נָהֲגוּ לִכְתֹּב בּוֹ שְׁתֵּים עֶשְׂרֵה שׁוּרוֹת כְּמִנְיַן גֵּט. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} וְכֵן פָּסַק רַשִׁ"י. וְהָרַ"ן פֵּרֵשׁ דְּחוּץ לָאָרֶץ שֶׁל צַד מַעֲרָבִי שֶׁל אֶרֶץ יִשְׂרָאֵל נָקַט, לְפִי שֶׁבְּאוֹתוֹ צַד לֵיכָּא עֲיָרוֹת מֻבְלָעוֹת וּסְמוּכוֹת שֶׁהֲרֵי הַיָּם מַפְסִיק, וּבִצְדָדִים אֲחֵרִים לֹא פְּסִיקָא לֵיהּ מִלְּתָא מִשּׁוּם דְּאִיכָּא עֲיָרוֹת סְמוּכוֹת וּמֻבְלָעוֹת (וּלְקַמָּן אָמְרִינַן דְּתַנָּא קַמָּא מַפִּיק אֵלּוּ מִכְּלָל). וּמִשּׁוּם הָכִי אָמַר מִמְּדִינַת הַיָּם שֶׁהִיא מַעֲרַב אֶרֶץ יִשְׂרָאֵל, וְהוּא הַדִּין לְכָל צְדָדִים אֲחֵרִים מֵאוֹתָן עֲיָרוֹת שֶׁהֵן דּוֹמוֹת לְאֵלּוּ שֶׁבִּמְדִינַת הַיָּם, שֶׁאֵינָן לֹא מֻבְלָעוֹת וְלֹא סְמוּכוֹת. וְאֶפְשָׁר שֶׁלָּזֶה כִּוֵּן רַשִׁ"י זִכְרוֹנוֹ לִבְרָכָה בְּמַה שֶּׁסִּיֵּם, כְּלוֹמַר כָּל חוּץ לָאָרֶץ שֶׁהִיא דוֹמָה לְחוּץ לָאָרֶץ שֶׁבִּמְדִינַת הַיָּם: {ג} וּקְרָא כְּתִיב וְכָתַב לָהּ, דְּהַיְנוּ לִשְׁמָהּ. וְאֵין בְּקִיאִין הַיְנוּ דְּאֵין חוֹשְׁשִׁין לְהַךְ דְּרָשָׁה, דְּאִי לָאו הָכִי מַאי שְׁנָא מִשְּׁאָר הִלְכוֹת גִּטִּין דְּלֹא חָיְשִׁינַן לְהוּ דְּלֹא בְּקִיאֵי. תּוֹסָפוֹת: {ד} וְרַשִׁ"י פֵּרֵשׁ דִּסְתָמָא כִּי מַסְהִיד בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם לִשְׁמָהּ קָמַסְהִיד. וְסִיֵּם הָרַ"ן דְּמֵימַר אָמַר הַאי שָׁלִיחַ לָמָּה לְהוּ לְרַבָּנָן לְמִשְׁאָל אִם נִכְתַּב וְנֶחְתַּם בְּפָנַי, שְׁמַע מִנַּהּ שֶׁיֵּשׁ כְּתִיבָה וַחֲתִימָה בְּגֵט שֶׁהִיא פְסוּלָה, וְאִלּוּ לֹא נִכְתַּב לִשְׁמָהּ הָיָה אוֹמֵר הַשָּׁלִיחַ כָּךְ אֵרַע מַעֲשֶׂה דִלְאַחֵר נִכְתַּב. וְהָא דְסַגִּי בְּעֵד אֶחָד, מִשּׁוּם דְּרֹב בְּקִיאִין הֵן. וַאֲפִלּוּ לְרַבִּי מֵאִיר דְּחָיֵישׁ לְמִעוּטָא, הָכָא הָוֵי מִעוּטָא דְמִעוּטָא, וְרַבָּנָן הוּא דְאַצְרוּךְ, וּמִשּׁוּם עִגּוּנָא הֵקֵלּוּ. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} וּפָרֵיךְ בַּגְּמָרָא אִם כֵּן בְּפָנַי נִכְתַּב לָמָּה לִי, וּמְשָׁנֵי דִּלְמָא אָתֵי לְאַחְלוֹפֵי בְּקִיּוּם שְׁטָרוֹת דְּעָלְמָא בְּעֵד אֶחָד: {ו} דְּעֵדִים הַחֲתוּמִים עַל הַשְּׁטָר כְּמוֹ שֶׁנֶּחְקְרָה עֵדוּתָן בְּבֵית דִּין, וְרַבָּנָן הוּא דְאַצְרוּךְ, וּמִשּׁוּם עִגּוּנָא הֵקֵלּוּ. גְּמָרָא: {ז} <B> הַחֶגֶר.</B> בַּגְּמָרָא מְפָרֵשׁ דִּבְהָא פְלִיגֵי, דְתַנָּא קַמָּא סָבַר דְּמֵרֶקֶם וְחֶגֶר אַף עַל פִּי שֶׁהֵן מִחוּץ לָאָרֶץ אֵין צָרִיךְ שֶׁיֹּאמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, לְפִי שֶׁהֵן עֲיָרוֹת הַסְּמוּכוֹת לְאֶרֶץ יִשְׂרָאֵל, וְכָל שֶׁכֵּן מִכְּפַר לוּדִים לְלוֹד שֶׁאֵין צָרִיךְ, לְפִי שֶׁהוּא מֻבְלָע בִּתְחוּם אֶרֶץ יִשְׂרָאֵל כוּ'. וְאָתֵי רַבָּן גַּמְלִיאֵל לָאַפְלוֹגֵי עֲלַהּ וּלְמֵימַר דְּדַוְקָא בְּמֻבְלָעוֹת כְּמוֹ מִכְּפַר לוּדִים גְּמִירֵי וּשְׁכִיחֵי, אֲבָל סְמוּכוֹת כְּגוֹן רֶקֶם לֹא גְמִירֵי וְלֹא שְׁכִיחֵי. וַאֲתָא רַבִּי אֱלִיעֶזֶר לְמֵימַר דַּאֲפִלּוּ מִכְּפַר לוּדִים שֶׁהוּא מֻבְלָע צָרִיךְ לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, דִּנְהִי דְמוֹדֶה לְרַבָּנָן קַמָּאֵי דְמֻבְלָעִים גְּמִירֵי וּשְׁכִיחֵי, אֲפִלּוּ הָכִי מַצְרִיךְ בְּהוּ לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, כְּדֵי שֶׁלֹּא תַחְלֹק בִּמְדִינַת הַיָּם. הָרַ"ן: {ח} <B> מִכְּפַר כוּ'.</B> מִקְרֵי כְּפַר לוּדִים, שֶׁבְּנֵי לוֹד מְצוּיִין בָּהּ תָּמִיד. וּמִשּׁוּם הָכִי נָקַט לְלוֹד לִרְבוּתָא. הָרַ"ן: {ט} <B> וְהַמּוֹלִיךְ.</B> וּבְהָא פְלִיגֵי אַתַּנָּא קַמָּא, דִּלְתַנָּא קַמָּא הַמּוֹלִיךְ מֵאֶרֶץ יִשְׂרָאֵל לְחוּץ לָאָרֶץ אֵין צָרִיךְ, לְפִי שֶׁבְּנֵי אֶרֶץ יִשְׂרָאֵל בְּקִיאִין לִשְׁמָהּ, וַחֲכָמִים גָּזְרוּ מוֹלִיךְ אַטּוּ מֵבִיא. וּלְמַאן דְּאָמַר טַעֲמָא מִשּׁוּם קִיּוּם, חֲכָמִים לִפְרוּשֵׁי מִלְּתָא דְּתַנָּא קַמָּא אָתוּ דְּמֵבִיא לָאו דַּוְקָא, אֶלָּא הוּא הַדִּין מוֹלִיךְ. גְּמָרָא: {י} <B> בִּמְדִינַת הַיָּם.</B> לְטַעֲמָא דְאֵין בְּקִיאִין הוּא הַדִּין בְּאוֹתָהּ מְדִינָה נַמִּי צָרִיךְ שֶׁיֹּאמַר, אֶלָּא דְּנָקַט מִמְּדִינָה לִמְדִינָה לְמֵידַק מִנַּהּ דְּהָא מִמְּדִינָה לִמְדִינָה בְּאֶרֶץ יִשְׂרָאֵל לֹא צָרִיךְ, וַאֲפִלּוּ לְכַתְּחִלָּה. וּלְאִידָךְ טַעֲמָא נַמִּי בְּאֶרֶץ יִשְׂרָאֵל דִּקְבִיעֵי בֵית דִּין שְׁכִיחֵי שַׁיָּרוֹת. גְּמָרָא: {יא} <B> הֶגְמוֹנְיָא.</B> פְּקִידוּת: {יב} גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י שֶׁלֹּא יֵלֵךְ מִזּוֹ לְזוֹ. וּבְאֶרֶץ יִשְׂרָאֵל נַמִּי לְטַעֲמָא דְקִיּוּם, דְּמַקְפִּידִין. וּלְמַאן דְּאָמַר לִשְׁמָהּ נַמִּי, דְּאִיהוּ סְבִירָא לֵיהּ נַמִּי טַעֲמָא דְקִיּוּם, דְּאִלּוּ בְּחוּץ לָאָרֶץ אֲפִלּוּ בְּאוֹתָהּ הֶגְמוֹנְיָא כִדְאִיתָא בַּגְּמָרָא:

{יג} <B> לַמִּזְרָח.</B> אֲבָל מַעֲרַב אֶרֶץ יִשְׂרָאֵל אֵין צָרִיךְ גְּבוּל, שֶׁהַיָּם הַגָּדוֹל גְּבוּלָהּ, דִּכְתִיב וּגְבוּל יָם וְהָיָה לָכֶם הַיָּם הַגָּדוֹל [בְּמִדְבָּר לה]. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} <B> וְאַשְׁקְלוֹן.</B> וּבְסֵפֶר יְהוֹשֻׁעַ מוּכָח דְּמֵאֶרֶץ יִשְׂרָאֵל הִיא. גַּם בַּסִּפְרִי חָשִׁיב פָּרָשַׁת אַשְׁקְלוֹן מִכִּבּוּשׁ עוֹלֵי בָּבֶל. וְהָרַ"ן תֵּרֵץ דְּפָרָשַׁת אַשְׁקְלוֹן אֵינוֹ אֶלָּא פָּרָשַׁת דְּרָכִים שֶׁלָּהּ, וְשֶׁכֵּן הוּא בַּיְרוּשַׁלְמִי:

{טו} <B> אֵינוֹ צָרִיךְ כוּ'.</B> דְּמִשּׁוּם עִגּוּנָא הֵקֵלּוּ וְלֹא הִצְרִיכוּ קִיּוּם כָּל שֶׁלֹּא עִרְעֵר. תּוֹסָפוֹת: {טז} דְּאִי לֹא תֵּימַהּ הָכִי הָא קָיְמָא לָן (כְּתֻבּוֹת פ"ב מ"ח) דְּאֵין עִרְעוּר פָּחוֹת מִשְּׁנַיִם. וְאֶלָּא תְּרֵי, תְּרֵי וּתְרֵי נִינְהוּ, מַה חָזִית דִּסְמוֹךְ אַהָנֵי, סְמוֹךְ אַהָנֵי, וְאוֹקְמַהּ בְּחֶזְקַת אֵשֶׁת אִישׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יז} מִשּׁוּם דְּלֹא קְבִיעֵי בֵית דִּין וְלֹא שְׁכִיחֵי שַׁיָּרוֹת: {יח} מִשּׁוּם דְּקָיְמָא לָן כְּמַאן דְּאָמַר דְּטַעְמָא מִשּׁוּם קִיּוּם הוּא: {יט} דִּלְאַחַר נְתִינָה תּוֹךְ כְּדֵי דִבּוּר מָצֵי לְמֵימַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתָּם. הָרַ"ן: {כ} <B> עֵדִים.</B> דְּקָיְמָא לָן עֵדֵי מְסִירָה כָרְתֵי: {כא} <B> יִתְקַיֵּם.</B> מְפָרֵשׁ בַּגְּמָרָא לְמַאן דְּאָמַר דְּלֹא בְקִיאֵי כוּ' דְּהָכָא לְאַחַר שֶׁלָּמְדוּ, כְּלוֹמַר לְאַחַר שֶׁקִּבְּלוּהָ לַדְּרָשָׁא דְּלִשְׁמָהּ. וַאֲפִלּוּ הָכִי בְּיָכוֹל צָרִיךְ לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתָּם, דְּלֹא עָקְרוּ לְתַקַּנְתַּיְהוּ כוּ'. אֲבָל פִּקֵּחַ וְנִתְחָרֵשׁ מִלְּתָא דְלָא שְׁכִיחַ הִיא וְלֹא גָזְרוּ בָּהּ רַבָּנָן:

{א} וַאֲפִלּוּ לְמַאן דְּאָמַר מִשּׁוּם קִיּוּם, וְצָרִיךְ לוֹמַר בְּפָנַי נִכְתַּב מִשּׁוּם דִּלְמָא אָתֵי לְאַחְלוֹפֵי, בְּחֶצְיוֹ הָרִאשׁוֹן שֶׁיֵּשׁ בּוֹ שֵׁם כוּ' שֶׁהוּא עִקַּר הַגֵּט מִנְכַּר מִלְּתָא וְלֹא אָתֵי לְאַחְלוֹפֵי. תּוֹסָפוֹת: {ב} וְאֵין צָרִיךְ שֶׁיִּהְיוּ שְׁנֵיהֶם אֲדוּקִים וַחֲתוּמִים בַּגֵּט, אֶלָּא שֶׁשְּׁנֵיהֶם אוֹמְרִים שֶׁהַבַּעַל עָשָׂה שְׁנֵיהֶם שְׁלוּחִים עַל כָּךְ וְאָמַר לָהֶם שְׁנֵיכֶם הוֹלִיכוּ גֵט לְאִשְׁתִּי, וּנְתָנוֹ הָאֶחָד בְּמַעֲמַד חֲבֵרוֹ כוּ'. תּוֹסָפוֹת: {ג} לְמַאן דְּאָמַר מִשּׁוּם קִיּוּם הָאִיכָּא שְׁנַיִם שֶׁמְּצוּיִין לְקַיְּמוֹ, וּלְמַאן דְּאָמַר מִשּׁוּם לִשְׁמָהּ, מַיְרֵי לְאַחַר שֶׁלָּמְדוּ, וּבִתְרֵי לֹא שְׁכִיחֵי, הִלְכָּךְ לֹא גָזְרִינַן שֶׁמָּא יַחֲזֹר הַדָּבָר לְקִלְקוּלוֹ: {ד} <B>וְאֶחָד.</B> שְׁלִישִׁי. רַשִׁ"י: {ה} דְּכֵיוָן דְּהַהוּא דְאָמַר בְּפָנַי נֶחְתַּם לֹא אָמַר בְּפָנַי נִכְתַּב, אָתֵי לְאַחְלוֹפֵי בְּקִיּוּם כוּ'. וּמַאן דְּאָמַר לִשְׁמָהּ, הָא סְבִירָא לֵיהּ נַמִּי טַעֲמָא דְּקִיּוּם. גְּמָרָא: {ו} <B> מַכְשִׁיר.</B> כֵּיוָן דְּאִיכָּא תְּרֵי דִמְסַהֲדִי אַכְּתִיבָה, תּוּ לֹא אָתֵי לְאַחְלוֹפֵי. גְּמָרָא: {ז} <B> כָּשֵׁר.</B> לְמַאן דִּסְבִירָא לֵיהּ טַעֲמָא דְקִיּוּם, אֲפִלּוּ אֵין שָׁם עֵד מֵעִיד עַל הַכְּתִיבָה כָּשֵׁר, שֶׁהֲרֵי נִתְקַיֵּם עַל פִּי שְׁנַיִם. וְלֹא נָקַט אֶחָד אוֹמֵר בְּפָנַי נִכְתַּב אֶלָּא אַגַּב רֵישָׁא דְתָנָא אֶחָד אוֹמֵר כוּ'. תּוֹסָפוֹת. וַאֲפִלּוּ יוֹצֵא מִתַּחַת יַד אֶחָד נַמִּי כָּשֵׁר. וּלְטַעֲמָא דְלִשְׁמָהּ דַּוְקָא בְּיוֹצֵא מִתַּחַת יְדֵי שְׁנֵיהֶם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ח} וְנִרְאֶה לִי דְּנָקַט בַּת אֲחוֹתוֹ לְפִי שֶׁהִיא בַּת גִּילוֹ, כִּדְאִיתָא בְּסוֹף פֶּרֶק ה' דִּנְדָרִים. וּבַיְרוּשַׁלְמִי מַעֲשֶׂה בְּאֶחָד שֶׁנָּשָׂא בַת אֲחוֹתוֹ וְזִנְּתָה כוּ'. וּלְפִי זֶה יֵשׁ לוֹמַר דְּמִשּׁוּם מַעֲשֶׂה שֶׁהָיָה נָקַט בַּת אֲחוֹתוֹ: {ט} דְּלֹא אָמְרִינַן אוֹקְמַהּ אַחֶזְקַת אֵשֶׁת אִישׁ וְהַשְׁתָּא הוּא דְאִגָּרְשָׁה, כֵּיוָן שֶׁהִיא גְרוּשָׁה לְפָנֵינוּ. וְעוֹד אַדְּרַבָּה אוֹקְמַהּ בְּחֶזְקַת כְּשֵׁרָה שֶׁלֹּא נִבְעֲלָה כְשֶׁהִיא אֵשֶׁת אִישׁ. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} דְּאִי מִשּׁוּם זְנוּת, לֹא שְׁכִיחַ. וְרַבָּנָן סָבְרֵי דְמִשּׁוּם פֵּירֵי לֹא הֻצְרְכוּ לְתַקֵּן זְמַן, דִּכְשֶׁאֵין בּוֹ זְמַן נַמִּי כְּשֶׁיָּבֹא הַגֵּט לְיַד הָאִשָּׁה יְכוֹלָה לָבֹא לְבֵית דִּין אוֹ לִפְנֵי עֵדִים וְתַרְאֶה לָהֶם גִּטָּהּ וְיִכְתְּבוּ לָהּ שֶׁמֵּאוֹתוֹ הַיּוֹם נִתְגָּרְשָׁה. תּוֹסָפוֹת:

{יא} <B> שֶׁל קְיָמָא.</B> לַאֲתוֹיֵי מֵי טַרְיָא (פֵּרוּשׁ גְּשָׁמִים שֶׁנִּשְׁאָרִים עַל הַגַּגּוֹת. לִישָׁנָא אַחֲרִינָא מַיִם שֶׁשּׁוֹרִין בָּהֶן פֵּירֵי) וְעַפְצָא: {יב} <B> שֶׁל זַיִת.</B> דַּוְקָא עָלֶה שֶׁל זַיִת וְכַיּוֹצֵא בּוֹ דְּמִתְקַיֵּם כוּ', אֲבָל שֶׁל כְּרֵשִׁין וּבְצָלִים וַעֲלֵה וְרָדִין וִירָקוֹת וְכָל דָּבָר שֶׁאֵינוֹ שֶׁל קְיָמָא פָּסוּל. תּוֹסָפוֹת: {יג} אַף עַל גַּב דִּבְמִשְׁנָה ד' פֶּרֶק ב' דְּסוֹטָה לְעִנְיַן מְגִלַּת סוֹטָה דִכְתִיב בַּסֵּפֶר לֹא דָרְשִׁינַן הָכִי, לֹא קָשְׁיָא, דְּהָכָא שָׁאנֵי, דִּכְלָל פְּרָט וּכְלָל דָּרֵישׁ, דִּוְנָתַן חָזַר וְכָלַל, אֲבָל הָתָם כְּלָל וּפְרָט וְאֵין בַּכְּלָל אֶלָּא מַה שֶּׁבַּפְּרָט. אִי נַמִּי הָכָא קְרָא יְתֵרָא קָדָרֵישׁ, וְכָתַב לָהּ תְּרֵי זִמְנֵי. תּוֹסָפוֹת:

{יד} וּלְשׁוֹן (טֹפֶס) [דְּפוּס] הוּא, שֶׁהוּא הֶעְתֵּק וְתֹאַר הַגֵּט. וְתוֹרֵף, הוּא גִלּוּי הַגֵּט, מִלְּשׁוֹן יַנִּיחֶנָּהּ בְּמָקוֹם הַתֻּרְפָּה: {טו} וְלֵיכָּא לְמֵימַר דְּדִיעֲבַד אֲפִלּוּ תוֹרֵף כָּשֵׁר, דְּהָא כֵּיוָן דִּמְחֻבָּר פְּסוּלָא דְאוֹרַיְתָא הוּא אֲפִלּוּ דִיעֲבַד אִית לָן לְמִפְסְלֵיהּ. הָרַ"ן: {טז} יֵשׁ לְעַיֵּן כֵּיוָן דִּכְתִיבַת זְמַן בַּגֵּט לֵיתָא אֶלָּא מִדְּרַבָּנָן, כָּל שֶׁשְּׁאָר הַתּוֹרֵף נִכְתַּב כְּהִלְכָתוֹ, זְמַן לֹא יִהְיֶה לוֹ אֶלָּא דִין טֹפֶס. וְאֶפְשָׁר דְּאַלְּמוּהוּ כְּאִלּוּ הָיָה שֶׁל תּוֹרָה. בֵּית יוֹסֵף: {יז} וְיַעֲבִיר תְּנַאי שֶׁהָיָה בוֹ. רַשִׁ"י: {יח} וְאַף עַל גַּב דְּאֵינוֹ דּוֹמֶה נִמְחַק פַּעַם אֶחָד לְנִמְחָק שְׁתֵּי פְעָמִים, וְיִהְיֶה נִכָּר אֵצֶל מַחַק דְּעֵדִים, סָבַר תַּנָּא קַמָּא דְגָזְרוּ דִּלְמָא דָמִי הַאי מַחְקָא לְהַאי מַחְקָא. וַחֲכָמִים מַכְשִׁירִין, דְּאִי אִיכָּא חֲשָׁשָׁא אָזְלִינַן לְעֵדֵי מְסִירָה. בֵּית יוֹסֵף, וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יט} <B> מַכְשִׁירִין.</B> אֵין לִתְמֹהַּ אֵיךְ מַכְשִׁיר רַבִּי אֱלִיעֶזֶר בְּדָבָר שֶׁיָּכוֹל לְהִזְדַּיֵּף, וְהָא מוֹדֶה רַבִּי אֱלִיעֶזֶר בִּמְזֻיָּף מִתּוֹכוֹ שֶׁהוּא פָּסוּל. דְּהָא מוֹדֶה נַמִּי הֵיכָא דְּלֹא אָתֵי לְמִסְמָךְ עֲלַיְהוּ, כְּגוֹן בְּשֵׁמוֹת מֻבְהָקִים, דְּכָשֵׁר, הָכָא נַמִּי כָּשֵׁר. הַגָּהוֹת אָשְׁרִי: {כ} וְאַף עַל גַּב דְּבִשְׁאָר שְׁטָרוֹת נַמִּי קָיְמָא לָן דְּעַל יְדֵי עֵדֵי מְסִירָה גּוֹבֶה מִמְּשֻׁעְבָּדִים, מִכָּל מָקוֹם כֵּיוָן דִּלְגוֹבַיְנָא קָיְמָא דִלְמָא לֹא מַפִּיק לְהוּ עַד זְמַן אָרֹךְ וְשָׁכְחוּ עֵדֵי מְסִירָה אֶת הַדָּבָר וְסַמְכִינַן אַעֵדֵי חֲתִימָה, וְהוּא מְזַיֵּף כְּתָב הָעֶלְיוֹן וְחוֹתַם הָעֵדִים עוֹמֵד בִּמְקוֹמוֹ. וְכָתְבוּ הַתּוֹסָפוֹת דְּהַיְנוּ דַּוְקָא בִשְׁטָרוֹת הָעוֹמְדִים לִרְאָיָה, אֲבָל שְׁטַר מֶכֶר וּמַתָּנָה דָמִי לְגִטִּין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כא} הַיְנוּ טֹפֶס:  {כב} מִשּׁוּם דְּלֹא שְׁכִיחַ כְּלָל שַׁרְיָא אֲפִלּוּ לְכַתְּחִלָּה. וְלֹא דָמִי לִכְתָבוֹ בִּמְחֻבָּר בְּמַתְנִיתִין דִּלְעֵיל. הָרַ"ן: {כג} <B> אֶת גִּטָּהּ.</B> וּמַקְנָתוֹ לַבַּעַל וְחוֹזֵר וּמוֹסְרוֹ לָהּ לְגֵרוּשִׁין. רַשִׁ"י. וְאַף עַל גַּב דִּמְחֻסַּר הַקְנָאָה בֵין כְּתִיבָה לִנְתִינָה לֹא הָוֵי מְחֻסָּר מַעֲשֶׂה, כֵּיוָן שֶׁאֵינוֹ מַעֲשֶׂה בְגוּפוֹ שֶׁל גֵּט דּוּמְיָא דִקְצִיצָה. הָרַ"ן: {כד} <B> שׁוֹבְרוֹ.</B> שׁוֹבֵר פֵּרָעוֹן כְּתֻבָּה, שֶׁהָאִשָּׁה הָיָה לָהּ לְכָתְבוֹ וּלְמָסְרוֹ לוֹ לִהְיוֹת לוֹ לִרְאָיָה. רַשִׁ"י: {כה} <B> בְּחוֹתְמָיו.</B> הַיְנוּ עֵדֵי מְסִירָה. הָרַ"ן. וְהָא דְנָקַט חוֹתְמָיו, מִשּׁוּם דְּרֹב עֵדֵי מְסִירָה הֵם הַחֲתוּמִים בּוֹ, וְרֹב גִּטִּין חֲתוּמִים הֵם, אֲפִלּוּ לְרַבִּי אֱלִיעֶזֶר, מִפְּנֵי תִקּוּן הָעוֹלָם. רַשִׁ"י: {כו} וְאַף עַל גַּב דְּאָמְרִינַן לְעֵיל דְּסַגִּי בְּקוֹל קֻלְמוֹסָא, הַיְנוּ לְמַאן דְּיָכוֹל לְמִחְזֵי, דְמִסְתְּפֵי סוֹפֵר מִנֵּיהּ דִּלְמָא חָזֵי לֵיהּ, אֲבָל בְּסוּמָא כֵּיוָן דְּלֹא אֶפְשָׁר לְמִחְזֵי לֹא מִסְתְּפֵי מִנֵּיהּ, הִלְכָּךְ לֹא מָצֵי אָמַר. רַ"ן וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כז} הַיְנוּ לְמַאי דְפַסְקִינַן כְּמַאן דְּאָמַר דְּטַעְמָא מִשּׁוּם קִיּוּם:

{ו} <B> אַלְמָנָה כוּ'.</B> וְכָל הַבָּא לִפָּרַע מִנִּכְסֵי יְתוֹמִים. וַהֲוָה מָצֵי לְמִתְנֵי אֵין נִפְרָעִים מִנִּכְסֵי יְתוֹמִים אֶלָּא בִּשְׁבוּעָה, וְנִמְנְעוּ מִלְּהַשְׁבִּיעַ הָאַלְמָנָה, {כד} וְנִרְאֶה לִי שֶׁלֹּא בָא לוֹמַר שֶׁצָּרִיךְ שֶׁיֹּאמַר כֵּן בִּשְׁעַת גֵּרוּשִׁין, דְּאִם כֵּן בְּמוֹצִיא מִשּׁוּם שֵׁם רָע נַמִּי הֲוָה לֵיהּ לְפָרֵשׁ כֵּן. אֶלָּא שֶׁפֵּרוּשׁוֹ מַאי אִכְפַּת לֵיהּ בְּאִם נָדְרָה, וְקָאָמַר שֶׁהוּא אוֹמֵר אִי אֶפְשִׁי כוּ'. תּוֹסְפוֹת יוֹם טוֹב. וְעַיֵּן עוֹד: {כה} לָאו דַּוְקָא, דְּהָא לֹא הִתְנָה בְכָךְ, אֶלָּא לַעַז בְּעָלְמָא. וְכֵן כָּתַב הָרִ"י: {כו} אֲבָל לְקִלְקוּלָא לֹא חָיֵישׁ הָכָא, דִּבְצָרִיךְ חֲקִירַת חָכָם סְבִירָא לֵיהּ כְּרַבִּי אֱלִיעֶזֶר, וּבְאֵינוֹ צָרִיךְ סְבִירָא לֵיהּ כְּרַבִּי מֵאִיר. גְּמָרָא: {כז} נִרְאֶה לִי דְּרַבִּי מֵאִיר לְטַעְמֵיהּ בְּמִשְׁנָה ז' פֶּרֶק י"א דִּנְדָרִים, בְּאוֹמֵר לֹא הָיִיתִי יוֹדֵעַ שֶׁזֶּה נֶדֶר, שֶׁאַף עַל פִּי כֵן הֲוָה לֵיהּ לְהָפֵר, וְרַבִּי אֱלִיעֶזֶר סְבִירָא לֵיהּ כְּרַבָּנָן דְּהָתָם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כח} בַּגְּמָרָא, פְּשִׁיטָא מַאי נֵחוּשׁ לָהּ, מַהוּ דְּתֵימָא לִגְזֹר מִשּׁוּם דְּרַבִּי נָתָן, דְּתַנְיָא רַבִּי נָתָן אוֹמֵר הַנּוֹדֵר כְּאִלּוּ בָּנָה בָּמָה וְהַמְקַיְּמוֹ כְּמוֹ שֶׁהִקְרִיב עָלֶיהָ קָרְבָּן, וְהַאי כֵּיוָן דְּקִיֵּם נִדְרוֹ נִקְנְסֵיהּ מִלְּהַחְזִיר, קָא מַשְׁמַע לָן:

{כט} <B>אַיְלוֹנִית.</B> פֵּרוּשׁ שֶׁחָשַׁב שֶׁהִיא אַיְלוֹנִית גְּמוּרָה, דְּלֹא יָלְדָה, וְגֵרְשָׁהּ בְּגֵט. וְאַף עַל פִּי שֶׁהָאַיְלוֹנִית אֵינָהּ צְרִיכָה גֵט כְּדִתְנַן רֵישׁ פֶּרֶק קַמָּא דִיבָמוֹת כוּ', הָכָא אַיְרֵי כִדְאָמְרָן דְּלֹא נוֹלְדוּ בָּהּ סִימָנֵי אַיְלוֹנִית גְּמוּרִים אֶלָּא שֶׁהוּא חָשַׁב כֵּן, וְהַיְנוּ דְקָתָנֵי כוּ' וְהָיוּ לָהּ בָּנִים. וְאִם נוֹלְדוּ בָּהּ סִימָנֵי אַיְלוֹנִית אִי אֶפְשָׁר בְּשׁוּם צַד שֶׁיִּהְיוּ לָהּ בָּנִים. הָרַ"ן: {ל} לְפִיכָךְ אוֹמְרִים לוֹ קֹדֶם גֵּרוּשִׁין הָוֵי יוֹדֵעַ שֶׁהַמּוֹצִיא אֶת אִשְׁתּוֹ מִשּׁוּם אַיְלוֹנִית לֹא יַחֲזִירֶנָהּ עוֹלָמִית. רַשִׁ"י: {לא} כְּלוֹמַר וְהַשְׁתָּא רַבָּנָן נַמִּי חָיְשֵׁי לְקִלְקוּלָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לב} וְקָשְׁיָא, דְּאִם כֵּן הֵטִיל הַתְּנַאי בַּגֵּט עַצְמוֹ וּכְשֶׁלֹּא נִתְקַיֵּם נִמְצָא שֶׁהַגֵּט בָּטֵל. וּבְמַתְנִיתִין דִּלְעֵיל דְּכַפְלֵיהּ כִּי לֹא נִתְקַיֵּם הַתְּנַאי וְהַגֵּט בָּטֵל מַה הוֹעִילוּ חֲכָמִים בְּתַקָּנָתָם דְּלֹא יַחֲזִיר. וְהַתּוֹסָפוֹת וְהָרַמְבַּ"ם כָּתְבוּ בִּדְכַפְלֵיהּ לְמִלְּתֵיהּ וְלֹא כַפְלֵיהּ לִתְנָאֵיהּ. דְּאִי כַפְלֵיהּ לִתְנָאֵיהּ אֵין מוֹעִיל מַה שֶּׁלֹּא יַחֲזִיר, דִּלְעוֹלָם אִיכָּא קִלְקוּלָא. וְכִי לֹא כַּפְלֵיהּ לְמִלְּתֵיהּ אֲפִלּוּ לַעַז לֵיכָּא, דַּחֲשָׁשׁ קִלְקוּל זֶה אֵינוֹ אֶלָּא לַעַז בְּעָלְמָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {לג} כִּי הוּא הָיָה סָבוּר שֶׁהָיְתָה אַיְלוֹנִית, אֶלָּא שֶׁאָמַרְנוּ לוֹ שֶׁיְּהֵא גּוֹמֵר וּמְגָרֵשׁ בְּכָל צַד, אֲבָל לֹא גֵּרֵשׁ עַל דַּעַת שֶׁתִּתְבַּע כְּתֻבָּה. הָרַ"ן:

{א} <B> בְּעִדִּית.</B> אֶת כָּל נְאוֹת יַעֲקֹב [אֵיכָה ב] תִּרְגּוּם יַת כָּל עִדִּית בֵּית יַעֲקֹב. נְאוֹת אֱלֹהִים [תְּהִלִּים פג] תַּרְגּוּם עִדִּית. עָרוּךְ: {ב} כְּלוֹמַר דְּהָא לְהָכִי תְנַן לָהּ הָכָא בְגִטִּין וְאַף עַל פִּי שֶׁאֵין עִנְיָנוֹ מֵעִנְיָן גִּטִּין, אֶלָּא מִשּׁוּם דְּתָנֵי בְּפֶרֶק דִּלְעֵיל מִלֵּי דְתִקּוּן הָעוֹלָם: {ג} שֶׁנּוֹחַ לוֹ לְאָדָם לִגְבּוֹת מֻעָט מִקַּרְקַע עִדִּית וְלֹא הַרְבֵּה מִזִּבּוּרִית. רַשִׁ"י: {ד} גְּמָרָא. וּפֵרֵשׁ הָרַ"ב אַף עַל פִּי שֶׁהַזִּבּוּרִית שָׁוָה כְנֶגֶד הַחוֹב לֹא קָפִיץ עֲלֵיהּ זְבִינָא: {ה} <B> בְּזִבּוּרִית.</B> הַפָּחוּת וְהָרַע. הָרַמְבַּ"ם. וְלֹא נוֹדַע לִי מֵאֵיזֶה לָשׁוֹן הוּא (נִרְאֶה לִי מִלָּשׁוֹן זִבּוּרָא וְעַקְרַבָּא. בֵּית דָּוִד): {ו} וְקָשֶׁה דִלְמַאן דְּאָמַר כְּתֻבָּה דְאוֹרַיְתָא דְּיָלֵיף לָהּ מִכְּמֹהַר הַבְּתוּלוֹת כוּ', מִדְּאוֹרַיְתָא דִינָהּ בְּעִדִּית, וְאַמַּאי הִפְקִיעוּ חֲכָמִים דִּינָהּ מֵהַאי טַעֲמָא. וְרַבִּי מֵאִיר נַמִּי דְּאָמַר בְּבֵינוֹנִית קָסָבַר כְּתֻבָּה דְאוֹרַיְתָא בְּרֵישׁ פֶּרֶק אַף עַל פִּי. וְיֵשׁ לוֹמַר דְּטַעְמָא דְהִפְקִיעוּ חֲכָמִים דִּינָהּ כִּדְאָמְרִינַן בַּגְּמָרָא גַּבֵּי בַעַל חוֹב, שֶׁלֹּא יִרְאֶה אָדָם שָׂדֶה נָאֶה וַיֹּאמַר אֶקְפֹּץ וְאַלְוֶּנּוּ, הָכִי נַמִּי חָשׁוּ חֲכָמִים בְּאִשָּׁה שֶׁמָּא תַקְנִיטֶנּוּ כְּדֵי שֶׁיְּגָרְשֶׁנָּהּ וְתִגְבֶּה כְּתֻבָּה. וְטַעְמָא דְיוֹתֵר מִשֶּׁהָאִישׁ כוּ' לֹא הָוֵי אֶלָּא מִשּׁוּם דְּלֹא תֵימָא כְמוֹ שֶׁבַּעַל חוֹב בְּבֵינוֹנִית מִשּׁוּם נְעִילַת דֶּלֶת, אִשָּׁה נַמִּי מִשּׁוּם חִנָּא שֶׁיְּהוּ הָאֲנָשִׁים נוֹשְׂאִים חֵן בְּעֵינֵי הַנָּשִׁים וְיִנָּשְׂאוּ לָהֶן תְּהֵא בְבֵינוֹנִית. תּוֹסָפוֹת:

{ז} וְחָשִׁיב תִּקּוּן הָעוֹלָם, דְּאִי לָאו תִּקּוּן הָעוֹלָם הָיָה גוֹבֶה אֲפִלּוּ מִמְּשַׁעְבְּדֵי בְּבֵינוֹנִית מִשּׁוּם נְעִילַת דֶּלֶת. עַד כָּאן. וּמִשּׁוּם הָכִי נָקַט הָרַ"ב בַּעַל חוֹב. וְהוּא הַדִּין לְנִזָּקִין: {ח} <B> יְתוֹמִים.</B> מְפָרֵשׁ טַעְמָא בַגְּמָרָא מִשּׁוּם דְּלֹא מַסִּיק אַדַּעְתֵּיהּ דְּמַלְוֶה דְמִית לֹוֶה וְנָפְלֵי נִכְסֵי קָמֵי יַתְמֵי דַּהֲוָה לֵיהּ נוֹעֵל דֶּלֶת וְהִלְכָּךְ אֲפִלּוּ יְתוֹמִים גְּדוֹלִים. וּבְנִזָּקִין אַף עַל גַּב דְּמִדְּאוֹרַיְתָא בְּעִדִּית, הַיְנוּ דְנִזַּק, וְגַבֵּי יַתְמֵי אוֹקְמוּהָ אַדְּאוֹרַיְתָא. וְעַיֵּן תּוֹסָפוֹת:

{ט} וּלְשׁוֹן לַאֲכִילַת פֵּרוֹת לֹא דָיְקָא, דִלְפֵרוֹת בִּלְחוּד הֲוָה לֵיהּ לְמִתְנֵי. וְהָרַמְבַּ"ם מְפַרֵשׁ כְּשֶׁגָּזַל רְאוּבֵן קַרְקַע מִשִּׁמְעוֹן וְהָיְתָה מְלֵאָה פֵרוֹת וְאָכַל רְאוּבֵן הַפֵּרוֹת וְהִפְסִיד הַקַּרְקַע כוּ', כְּשֶׁיָּבֹא שִׁמְעוֹן לִגְבּוֹת מֵרְאוּבֵן הַפֵּרוֹת שֶׁאָכַל כוּ'. וּשְׁנֵי הַפֵּרוּשִׁים בַּגְּמָרָא בָּבָא מְצִיעָא דַף י"ד. וּפֵרוּשׁ הָרַ"ב נִדְחָה: {י} אֲבָל דְּמֵי פֵרוֹת הֲוָה כְמִלְוֶה עַל פֶּה. דְּאַף עַל פִּי שֶׁכָּתַב מוֹכֵר לַלּוֹקֵחַ אֲנָא אֵיקוּם וַאַשַׁפֵּי אִינוּן וְעַמְלֵיהוֹן וְשִׁבְחֵיהוֹן, אֲפִלּוּ הָכִי אֵין קוֹל יוֹצֵא אֶלָּא עַל הַמִּקָּח, לֹא עַל הַשְּׁאָר, כֵּיוָן שֶׁעֲדַיִן אֵינוֹ בָּעוֹלָם. רַשִׁ"י וְהָרַ"ן: 

~ פרק ה - משנה ד

{יא} <B> אֲבִיהֶן.</B> וְכָל שֶׁכֵּן הַמְּמֻנֶּה מֵהַבֵּית דִּין שֶׁהוּא שְׁלִיחַ בֵּית דִּין: {יב} <B> לְעַשֵּׂר.</B> קוֹרְאִין כָּל הַמַּתָּנוֹת בְּשֵׁם מַעֲשֵׂר בִּלְבַד. וּבַגְּמָרָא רָמִינַן דִּתְנַן (בְּרֵישׁ תְּרוּמוֹת) וְהַתּוֹרֵם אֶת שֶׁאֵינוֹ שֶׁלּוֹ אֵין תְּרוּמָתוֹ תְּרוּמָה, וּמְשָׁנִינַן לֹא קָשְׁיָא כָּאן לְהַאֲכִיל כָּאן לְהַנִּיחַ. פֵּרֵשׁ רַשִׁ"י לְהַאֲכִיל יְתוֹמִים לְאַלְתַּר חַיָּב לְעַשֵּׂר, לְהַנִּיחַ הַתְּבוּאָה בָאוֹצָר עַד שֶׁיִּגְדְּלוּ, לָמָּה לִי עִשּׂוּרֵי, לִכְשֶׁיִּגְדְּלוּ יְעַשְּׂרוּ הֵם: {יג} <B> שֶׁמִּנָּהוּ.</B> וּמִי שֶׁסָּמְכוּ אֶצְלוֹ שַׁבְקֵיהּ מִלָּדוּן בּוֹ. וְיֵשׁ בָּזֶה מַחֲלֹקֶת הַפּוֹסְקִים לְמִי דָמִי יוֹתֵר: {יד} וְאַף עַל גַּב דְּהַיְנוּ מְדַמֵּעַ, שֶׁהוּא נַמִּי מְעָרֵב, אִצְטְרִיךְ לְמִתְנֵי בְּהֶדְיָא, מִשּׁוּם דִּקְנָסָא מִקְּנָסָא לֹא יָלְפִינַן, דְּהָא לָאו דִּינָא הוּא, וְהֵיכָא דְקָנוּס קָנוּס וְהֵיכָא דְלָא קָנוּס לֹא קָנוּס. גְּמָרָא: {טו} וְצָרִיךְ לוֹמַר דְּמַזִּיק שִׁעְבּוּדוֹ שֶׁל חֲבֵרוֹ דִתְנַן בַּפֶּרֶק דִּלְעֵיל מִשְׁנָה ד', דִּלְרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל דְּקָיְמָא לָן כְּוָתֵיהּ מְחָיְבִינַן לֵיהּ, דְּחָשִׁיב הֶזֵּק נִכָּר לְגַבֵּי הָנָךְ דְּהָכָא. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טז} וְהוּא הַדִּין לַעֲבוֹדַת הוֹלָכָה וְקַבָּלָה:

{יז}  <B> הַחֵרֶשֶׁת.</B> מִשּׁוּם תַּרְתֵּי בַּתְרָיָתָא דְּמִשּׁוּם תַּקָּנָה נִינְהוּ, תָּנָא לֵיהּ הָכָא. רַשִׁ"י: {יח} <B> קְטַנָּה כוּ'.</B> אֲבָל חֵרֶשֶׁת דְּאַיְרֵי בַּהּ וְשָׁבְקָה, שְׁמַע מִנַּהּ דְּלֹא אָכְלָה. וְטַעְמָא בַגְּמָרָא, גְזֵרָה שֶׁמָּא חֵרֵשׁ שֶׁנָּשָׂא פִּקַּחַת יַאֲכִילֶנָּהּ, וּגְזֵרָה מִשּׁוּם תְּרוּמָה דְאוֹרַיְתָא: {יט} וְאִם תֹּאמַר וְלוֹקְמָא אֲפִלּוּ בִּדְאוֹרַיְתָא, דְקָטָן אוֹכֵל נְבֵלוֹת הוּא דְאֵין מְצֻוִּין לְהַפְרִישׁוֹ. וְיֵשׁ לוֹמַר דְּבַגְּמָרָא מַשְׁמַע דַּאֲפִלּוּ מַאֲכִילָהּ בְּיָדַיִם. וְעוֹד מַשְׁמַע דַּאֲפִלּוּ לְכִי גָּדְלָה אוֹכֶלֶת מִכֹּחַ זֶה אַף עַל פִּי שֶׁעֲדַיִן לֹא בָּעַל. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} <B> יוֹרְשָׁהּ.</B> דְּהֶפְקֵר בֵּית דִּין הֶפְקֵר כִּדְיָלְפִינַן לֵיהּ בַּגְּמָרָא מִקְּרָאִי: {כא} <B> שֶׁלֹּא נוֹדְעָה.</B> דְּאִי נוֹדְעָה לָרַבִּים, מִלְּתָא דְלָא שְׁכִיחָא הוּא שֶׁיְבִיאֶנָּה הַגַּזְלָן, וְלֹא תִקְּנוּ בְּמִלְּתָא דְלָא שְׁכִיחַ: {כב} <B> לָרַבִּים.</B> אָמְרִינַן בַּיְרוּשַׁלְמִי דְרַבִּים הַיְנוּ שְׁלֹשָׁה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} <B> מְכַפֶּרֶת.</B> וְאִי קָשְׁיָא הָאֵיךְ הִתְנוּ בֵּית דִּין לַעֲקֹר דָּבָר מֵהַתּוֹרָה וְלִפְטֹר אֶת זֶה מֵחַטָּאת שֶׁהוּא חַיָּב, הָא אָמְרִינַן בִּיבָמוֹת דַּף צ' שֵׁב וְאַל תַּעֲשֶׂה שָׁאנֵי דְלָאו מִעַקַּר הוּא. רַשִׁ"י. וְקָשֶׁה דַעֲדַיִן אֵינוֹ מְיֻשָּׁב הָאֵיךְ הַכֹּהֲנִים אוֹכְלִין חֻלִּין בָּעֲזָרָה. וּלְהַמַּסְקָנָא מְתָרֵץ דְּרַבָּנָן אוֹקְמֵיהּ בְּיַד הַגַּזְלָן שֶׁיְּהֵא שֶׁלּוֹ וְשֶׁיּוּכַל לְהַקְדִּישׁוֹ. וּפֵרֵשׁ רַשִׁ"י דְהֶפְקֵר בֵּית דִּין הֶפְקֵר, אִם כֵּן הָוֵי שֶׁלּוֹ נַמִּי לְעִנְיַן כַּפָּרָה, וּמְיֻשַּׁב הַכֹּל. תּוֹסָפוֹת חֲדָשִׁים: 

~ פרק ה - משנה ו

{כד} <B> בִּיהוּדָה.</B> מִלְחֶמֶת טִיטוּס שֶׁהָיָה בִּירוּשָׁלַיִם וּבִיהוּדָה. רַשִׁ"י. וּבַיְרוּשַׁלְמִי, בָּרִאשׁוֹנָה גָזְרוּ גְּזֵרָה עַל יְהוּדָה לְפִי שֶׁמַּסֹּרֶת בְּיָדָם שֶׁיְהוּדָה הָרַג עֵשָׂו דִּכְתִיב יָדְךָ בְּעֹרֶף אוֹיְבֶיךָ [בְּרֵאשִׁית מט]. תּוֹסָפוֹת: {כה} לָשׁוֹן מְסֹרָס הוּא, דְּיִשְׂרָאֵל אַגַּב אוֹנְסֵיהּ הֲוָה כוּ': {כו} אוֹ שֶׁכָּתַב בַּכְּתֻבָּה אוֹ שֶׁהִכְנִיסָה לוֹ {ז} שֶׁכֵּן הַדִּין בְּכָל הַתְּנָאִים, שֶׁהָאוֹמֵר הֲרֵי זֶה גִטֵּךְ אִם אֶעֱשֶׂה דָבָר פְּלוֹנִי אִם לֹא אֶעֱשֶׂה, אֵין הַגֵּט חָל אֶלָּא מֵאוֹתוֹ זְמַן וָאֵילָךְ, כְּדִתְנַן לְקַמָּן מִשְׁנָה ח', הֲרֵי זֶה גִטֵּךְ אִם לֹא בָאתִי מִכָּאן וְעַד שְׁנֵים עָשָׂר חֹדֶשׁ וּמֵת בְּתוֹךְ שְׁנֵים עָשָׂר חֹדֶשׁ אֵינוֹ גֵּט. הָרַ"ן וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} <B> אִם מַתִּי כוּ'.</B> בָּרִיא שֶׁאָמַר אִם מַתִּי. חוֹלֶה שֶׁאָמַר אִם מַתִּי מֵחֹלִי זֶה. הָרַמְבַּ"ם: {ט} <B> לְאַחַר מִיתָה.</B> לֹא הֲוָה זוֹ וְאֵין צָרִיךְ לוֹמַר זוֹ. אֶלָּא הָכִי קָאָמַר, הֲרֵי זֶה גִטֵּךְ אִם מַתִּי כוּ', נַעֲשָׂה כְּאוֹמֵר לְאַחַר מִיתָה וְלֹא אָמַר כְּלוּם: {י} <B> לֹא כוּ'.</B> וּדְלֹא כְּרַבִּי יוֹסֵי דְמִשְׁנָה ז' פֶּרֶק ח' דְּבָבָא בַתְרָא, דְּאָמַר זְמַנּוֹ שֶׁל שְׁטָר מוֹכִיחַ עָלָיו וַאֲפִלּוּ לֹא אָמַר מֵהַיּוֹם כְּמַאן דְּאָמַר דָּמֵי. גְּמָרָא. דִּבְגִטִּין לֹא חָיְשִׁינַן לְדִבְרֵי רַבִּי יוֹסֵי. הַמַּגִּיד מִשְׁנֶה: {יא} <B> מֵהַיּוֹם אִם מַתִּי.</B> דְּאִי מַתִּי שְׁתֵּי לְשׁוֹנוֹת מַשְׁמַע. מַשְׁמַע כְּמֵעַכְשָׁיו, וּמַשְׁמַע כִּלְאַחַר מִיתָה. אָמַר לָהּ מֵהַיּוֹם, כְּמַאן דְּאָמַר לָהּ מֵעַכְשָׁיו דָּמֵי, לֹא אָמַר לָהּ מֵהַיּוֹם כְּמַאן דְּאָמַר לָהּ לְאַחַר מִיתָה דָּמֵי. גְּמָרָא:

{יג} <B> שְׁלֹשָׁה.</B> מִנְיָנָא לְמִעוּטֵי גֵט יָשָׁן דְּתִנָּשֵׂא לְכַתְּחִלָּה. וּלְמִעוּטֵי שְׁלוֹם מַלְכוּת דְּהַוָּלָד מַמְזֵר אַלִּיבָּא דְרַבִּי מֵאִיר, אִי נַמִּי לְרַבָּנָן דְּתֵצֵא. גְּמָרָא: {יד} <B> כָּשֵׁר.</B> וּמַה שֶׁהֵקֵלּוּ בִּשְׁלֹשָׁה גִטִּין הַלָּלוּ מִבִּשְׁאָר גִּטִּין פְּסוּלִין. הַיְנוּ טַעֲמָא מִשּׁוּם דְּדָבָר הַנּוֹהֵג בְּגִטִּין וְלֹא בִּשְׁטָרוֹת הֶחְמִירוּ יוֹתֵר כְּדֵי שֶׁלֹּא יָבֹאוּ לְהַשְׁווֹת גִּטִּין לִשְׁטָרוֹת כוּ'. אֲבָל בְּדָבָר שֶׁלֹּא נִתְקַן בְּגִטִּין בִּלְבַד אֶלָּא גִּטִּין וּשְׁטָרוֹת שָׁוִים בָּהֶם הֵקֵלּוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} <B> וְאֵין כוּ'.</B> טַעֲמָא דְפָסוּל מִשּׁוּם שֶׁיּוּכַל לִכְתֹּב הַזְּמַן כְּמוֹ שֶׁיִּרְצֶה, וְהָוֵי כְאֵין בּוֹ זְמַן דְּאִיכָּא לְמֵיחַשׁ שֶׁמָּא יְחַפֶּה. וְאִי נַמִּי מִשּׁוּם פֵּירֵי וַדַּאי דְּהַבַּעַל אֵינוֹ נֶאֱמָן (הַתּוֹסְפוֹת יוֹם טוֹב קִצֵּר, וְעַיֵּן בַּתּוֹסְפוֹת בַּאֲרִיכוּת), וּלְכָךְ פְּסָלוּהוּ, דְּלִפְעָמִים תַּפְסִיד הָאִשָּׁה שֶׁלֹּא כְדִין כְּשֶׁיִּכְתֹּב יוֹם שֶׁעָמַד בּוֹ וְלֹא יַקְדִּים הַזְּמַן, וְהִיא לֹא תוּכַל לִטְרֹף לְפִי שֶׁאָנוּ חוֹשְׁדִים אוֹתוֹ שֶׁהִקְדִּים הַזְּמַן, וּלְכָךְ פְּסָלוּהוּ דְּהַשְׁתָּא לֹא תִטְרֹף הָאִשָּׁה מִזְּמַן הַכָּתוּב כֵּיוָן שֶׁהַגֵּט פָּסוּל. דְּדַוְקָא בְּגֵט כָּשֵׁר גּוֹבָה מִיּוֹם הַכְּתִיבָה. תּוֹסָפוֹת. וְעַיֵּן עוֹד: {טז} <B> עֵד אֶחָד.</B> נִרְאֶה דְּהוּא הַדִּין דְּאֵין בּוֹ זְמַן, אֶלָּא לְאַשְׁמְעִינַן דְּאַף עַל גַּב דְּיֵשׁ בּוֹ זְמַן לֹא תִנָּשֵׂא לְכַתְּחִלָּה: {יז} <B> הֲרֵי כוּ'.</B> לְמִעוּטֵי הָא דְּתַנְיָא הַמֵּבִיא גֵט מִמְּדִינַת הַיָּם נְתָנוֹ לָהּ וְלֹא אָמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתָּם יוֹצִיא וְהַוָּלָד מַמְזֵר דִּבְרֵי רַבִּי מֵאִיר. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יח} <B> עֵדִים.</B> וְאֵינוֹ כְּתַב יָדוֹ, כָּשֵׁר לִנָּשֵׂא לְכַתְּחִלָּה: {יט} <B> וְגוֹבָה כוּ'.</B> הָכִי קָאָמַר, וְאִם שְׁטָר הוּא גּוֹבָה כוּ', דְּרַבִּי אֶלְעָזָר סְבִירָא לֵיהּ עֵדֵי מְסִירָה כָרְתֵי בֵּין בְּגִטִּין בֵּין בִּשְׁטָרוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כ} <B> לָזוֹ.</B> אֲפִלּוּ לְרַבִּי אֶלְעָזָר דְּאָמַר עֵדֵי מְסִירָה כָרְתֵי, וְעַל כָּרְחֲךָ לָהּ לִשְׁמָהּ דִּכְתִיב בִּקְרָא לֹא אַחֲתִימָה קָאֵי אֶלָּא אַכְּתָב, וּמִיהוּ אַוְנָתַן לֹא קָאֵי, דְלָא בָעִינַן שֶׁיֵּדְעוּ הָעֵדִים בִּשְׁעַת מְסִירָה בְּהֵי מִנַּיְהוּ מִתְגָּרֶשֶׁת. גְּמָרָא: {כא} וְזֶה לְשׁוֹן הָרַ"ן, לֹא בָטֵל מַמָּשׁ, אֶלָּא דְכֵיוָן דְּלֹא יָדְעִינַן דְּמַאן נִיהוּ לֹא יִנָּתֵן לֹא לָזוֹ וְלֹא לָזוֹ. אֲבָל אִם נָתַן לְאַחַת מֵהֶן אוֹ לִשְׁתֵּיהֶן הֲרֵי זֶה סְפֵק גֵּרוּשִׁין: {כב} זְמַן בִּכְלַל תּוֹרֵף הוּא, אֲבָל שֵׁם טֹפֶס הוּנַח עַל כָּל נֻסַּח הַגֵּט כוּ', אֶלָּא שֶׁהוֹצִיאוּ מִן הַכְּלָל וְקָרְאוּ לְמִקְצַת מֵהֶן תּוֹרֵף. וְאִם נֶאֱמַר טֹפֶס בְּלֹא הַזְכָּרַת תּוֹרֵף עִמּוֹ יֵשׁ בַּמַּשְׁמָע טֹפֶס כָּל נֻסַּח הַגֵּט. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} מְפָרֵשׁ בַּגְּמָרָא שֶׁזֶּה הַזְּמַן בּוֹ בַּיּוֹם עַצְמוֹ שֶׁל הָרִאשׁוֹן, דְּאִי לָאו הָכִי הָוֵי רִאשׁוֹן מֻקְדָּם לַחֲתִימַת הָעֵדִים, וְלֹא הָיָה צָרִיךְ לַטַּעַם שֶׁאֵין הָעֵדִים נִקְרִים עִמּוֹ. וְגַם מְפָרֵשׁ בַּגְּמָרָא דְדַוְקָא שֶׁכָּתוּב וּבְכָךְ בְּוָי"ו לְחַבְּרָם. דְּאִי לֹא תֵימָא הָכִי תֵּיפּוֹק לֵיהּ דְּאֵין עֵדִים חֲתוּמִים עַל הַכֹּל: {כד} <B> נִקְרִין עִמּוֹ.</B> הַיְנוּ תַּחְתּוֹן, אֲבָל עֶלְיוֹנִים פְּסוּלִים, שֶׁמָּא לֹא הֵעִידוּ אֶלָּא עַל הָאַחֲרוֹן. רַשִׁ"י: {כה} מְפָרֵשׁ בַּגְּמָרָא דְהָכִי כָּתַב, פְּלוֹנִי בֶּן תַּחַת הָרִאשׁוֹן, וּפְלוֹנִי אַבָּתְרָא. דְּאִי לָאו הָכִי, אֶלָּא בֶּן יַעֲקֹב עֵד אַבַּתְרָא, כְּשֵׁרִים, כְּדִתְנַן בַּמִּשְׁנָה דִלְקַמָּן בֶּן אִישׁ פְּלוֹנִי עֵד, כָּשֵׁר. וְהַשֵּׁנִי כָשֵׁר. וְכֵן נַמִּי כָּשֵׁר קַמָּא, דִפְשִׁיטָא דְקָאֵי אַבֶּן יַעֲקֹב עֵד דְּבָתְרָא. וְאִי נַמִּי כָתַב פְּלוֹנִי בֶּן תַּחַת הָרִאשׁוֹן, וּפְלוֹנִי עֵד אַבָּתְרָא, שְׁנֵיהֶם כְּשֵׁרִים, שֶׁהַבֵּן חָתוּם עַל הָרִאשׁוֹן, וְהָאָב אַשֵּׁנִי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{לט} גְּמָרָא. דְּאִלּוּ לְמֵיסַר אַגַּבְרָא (אִם אֵשֶׁת כֹּהֵן הִיא. רַשִׁ"י) הָא קָיְמָא לָן כָּל קָלָא דְבָתַר נִשּׂוּאִין לֹא חָיְשִׁינַן לֵיהּ: {מ} <B> אֲמַתְלָא.</B> מָשָׁל וְטַעַם. רַשִׁ"י: {מא} <B> עַל תְּנַאי.</B> שֶׁמָּא לֹא נִתְקַיֵּם הַתְּנַאי. רַשִׁ"י: {מב} <B> זוֹ הִיא.</B> אֲמַתְלָא שֶׁל קִדּוּשִׁין. וְהוּא הַדִּין נַמִּי אִיפְּכָא. רַשִׁ"י:

{מג} כְּמוֹ אֵשׁ קָדְחָה בְאַפִּי [דְּבָרִים לב]: {מד} וְאִצְטְרִיךְ לִכְתֹּב עֶרְוָה, לוֹמַר שֶׁאֲפִלּוּ יָצְאָה מִשּׁוּם עֶרְוָה מֻתֶּרֶת לִנָּשֵׂא לְאַחֵר, כְּדִכְתִיב וְיָצְאָה וְהָיְתָה לְאִישׁ אַחֵר. וּלְבֵית שַׁמַּאי אִצְטְרִיךְ דָּבָר לִגְזֵרָה שָׁוָה, דִּכְתִיב הָכָא עֶרְוַת דָּבָר וּכְתִיב לְהַלָּן עַל פִּי שְׁנַיִם עֵדִים יָקוּם דָּבָר, מַה לְּהַלָּן בִּשְׁנַיִם אַף כָּאן בִּשְׁנַיִם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מה} <B> דְּאָמַר רֵישׁ לָקִישׁ כִּי מִשְׁמֵשׁ בְּאַרְבַּע לְשׁוֹנוֹת.</B> אִי, אֶלָּא, דִּלְמָא, דְּהָא. בֵּית שַׁמַּאי סָבְרֵי כִי מָצָא, דְּהָא מָצָא. וְרַבִּי עֲקִיבָא סָבַר כִּי מָצָא, אִי נַמִּי מָצָא: {מו} הַיְנוּ דְּמַסִּיק הַתּוֹסָפוֹת דְּרַבִּי עֲקִיבָא כְבֵית הִלֵּל אָתֵי. דְּאִי לָאו דַּאֲיַתֵּר לֵיהּ דָּבָר לְהִקְדִּיחָה תַּבְשִׁילוֹ לֹא הֲוָה דָרֵישׁ כִּי מָצָא בָּהּ בִּלְשׁוֹן אִי נַמִּי:
