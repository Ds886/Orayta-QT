~ פרק א - משנה א

{א} <b>הַשְּׁקָלִים.</b> הֵם אוֹתָם שֶׁאָמַר הַכָּתוּב עֲלֵיהֶם זֶה יִתְּנוּ כָּל הָעוֹבֵר עַל הַפְּקוּדִים. רַמְבַּ"ם: {ב} וּבָשָׂר בֶּחָלָב שָׁאנֵי, דְּגַלֵּי קְרָא בְּהֶדְיָא דְּבָטֵל בְּאֵינוֹ נוֹתֵן טַעַם, דְּדֶרֶךְ בִּשּׁוּל הַיְנוּ דְּאִיכָּא נְתִינַת טַעַם, נִיחָא דְּלֵיכָּא לְמֵילַף מִשּׁוּם דְּחִדּוּשׁ הוּא דְּלֹא נֶאֱסַר בְּכָבוּשׁ אֶלָּא דַּוְקָא בִּמְבֻשָּׁל: {ג} כְּלוֹמַר לְהָכִי לֹא תָּנֵי דִּבְאַרְבָּעָה עָשָׂר בּוֹ קוֹרִין שְׁאָר עֲיָרוֹת. תּוֹסָפוֹת. וּמִשּׁוּם הָכִי לֹא הֻצְרַךְ לְפָרֵשׁ כֵּן בְּכִלְאַיִם דְּתָנֵי אַיְדֵי דִּשְׁקָלִים דִּפְשִׁיטָא הוּא: {ד} יְרוּשַׁלְמִי.</B> וְהַכַּוָּנָה כִּי כָּל מַה שֶּׁהָיָה מֵאֵלּוּ הַדְּבָרִים תָּלוּי וּמְסֻגָּר וְלֹא נִתְעַסְּקוּ בּוֹ, שֶׁמְּזָרְזִין וּמְמַהֲרִין לְהַשְׁלִימוֹ וּלְהַגִּיעוֹ לְתַכְלִיתוֹ מֵחֲצִי חֹדֶשׁ אֲדָר. הָרַמְבַּ"ם. וּבְוַדַּאי מִי שֶׁהָיָה זָרִיז וְהִקְדִּים לַעֲשׂוֹת הֲרֵי זֶה מְשֻׁבָּח: {ה} וְהַכֶּסֶף מִשְׁנֶה נִסְתַּפֵּק מִלְּשׁוֹן הָרַמְבַּ"ם שֶׁכָּתַב דְּבוֹדְקִים עַל הָרְאוּיָה לִשְׁתּוֹת כוּ', דְּאִלּוּ לְהַשְׁקוֹתָהּ לֹא הָיוּ מְאַחֲרִים אוֹתָהּ עַד חֲמִשָּׁה עָשָׂר בַּאֲדָר אֶלָּא בְּכָל יוֹם מַשְׁקִין אוֹתָהּ, אֲבָל בְּפֵרוּשׁוֹ אֵין נִרְאֶה כֵּן אֶלָּא דְּהַשְׁקָאָה עַצְמָהּ נַמִּי. וְזֶה לְשׁוֹן הָרַמְבַּ"ם, בֵּית דִּין הָיוּ נִפְנִין לְחַפֵּשׂ עַל צָרְכֵי צִבּוּר כוּ' הַהַקְדָּשׁוֹת כוּ' וְגוֹבִים כוּ' כְּדֵי שֶׁיִּהְיֶה כָּל הָעָם עָתִיד לִתֵּן תְּרוּמַת הַשְּׁקָלִים לְחַזֵּק אֶת בֵּית אֱלֹהֵינוּ:

{ו} דִּכְתִיב [עֶזְרָא י] כָּל אֲשֶׁר לֹא יָבֹא לִשְׁלֹשֶׁת יָמִים בַּעֲצַת הַשָּׂרִים וְהַזְּקֵנִים יָחֳרַם כָּל רְכוּשׁוֹ. יְרוּשַׁלְמִי: 

~ פרק א - משנה ג

{ז} <B> שֻׁלְחָנוֹת.</B> שֻׁלְחָן שֵׁם לַשֻּׁלְחָן שֶׁלִּפְנֵי הַשֻּׁלְחָנִי שֶׁמְּקַבֵּל עָלָיו, וְכָךְ קוֹרִין לְחַלְפָנִי שֻׁלְחָנִי בְּכָל הַמִּשְׁנָה: {ח} <B> לְמַשְׁכֵּן.</B> כְּלוֹמַר שֶׁמּוֹצִיאִין הָעֲבוֹט מִבֵּיתוֹ. תַּרְגּוּם אִם חָבוֹל תַּחְבֹּל [שְׁמוֹת כב] אִם מַשְׁכְּנָא תָּסֵב. הָרַמְבַּ"ם. וּמַה שֶּׁכָּתַב מִבֵּיתוֹ הַיְנוּ שֶׁהָיוּ לוֹקְחִים בְּעַל כָּרְחוֹ, וַאֲפִלּוּ כְּסוּתוֹ, וְאֵין כָּאן מִשּׁוּם בַּחוּץ תַּעֲמֹד דְּהָכָא מִצְוַת גָּבוֹהַּ עָלָיו, מַה שֶּׁאֵין כֵּן פְּרִיעַת בַּעַל חוֹב אַף עַל גַּב דְּנַמִּי מִצְוָה הִיא אֵין לְמֵדִין מִמֶּנָּה לְמִצְוַת גָּבוֹהַּ: {ט} <B> שׁוּב אֵינוֹ פוֹסֵק.</B> שֶׁאוֹמְרִין לוֹ מֵאַחַר שֶׁנָּתַתָּ עָלָיו הַשָּׁנָה שֶׁעָבְרָה וְהֶעֱמַדְתָּ עָלָיו זֹאת הַמִּצְוָה תֵּן עָלָיו תָּמִיד עַד שֶׁיַּגִּיעַ לְעֶשְׂרִים וְלֹא תִּפְסֹק. הָרַמְבַּ"ם. וְנִרְאֶה מִדְּבָרָיו דְּאִם מֵת אָבִיו וַעֲדַיִן הוּא קָטָן דְּאֵין חִיּוּב עָלָיו, וּדְלֹא כְּהַר"ב, וְאוּלַי הָרַ"ב רְצוֹנוֹ לוֹמַר שֶׁהַחִיּוּב חָל עַל נִכְסֵי הָאָב וּבְשֶׁיָּרְשׁוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} וְנִרְאֶה לִי דְּהַיְנוּ דַּרְכֵי שָׁלוֹם, שֶׁאִם לֹא נַחְלֹק לָהֶם כָּבוֹד אָתוּ לְאִינְּצוֹיֵי: 

~ פרק א - משנה ד

{יא} כְּדִמְפָרֵשׁ טַעֲמָא בַּיְרוּשַׁלְמִי, דִּכְתִיב כָּל הָעוֹבֵר כוּ', כָּל דְּעָבַר עַל פִּיקוּדָא יִתֵּן, וּכְמוֹ שֶׁכָּתַב הָרַמְבַּ"ם לְעֵיל גַּבֵּי לְוִיִּים, וְאוֹתוֹ מַאן דְּאָמַר שֶׁנִּזְכַּר הוּא נִיהוּ בֶּן בּוּכְרִי דְּהָכָא דְּכֹהֲנִים וּלְוִיִּים שָׁוִין, וְהָכִי נַמִּי הֲוָה מָצֵי לְמֵימָר כָּל לֵוִי, אֶלָּא מִשּׁוּם דְּלֹא מָצֵי לְמֵימָר שֶׁדּוֹרְשִׁים כוּ' לְעַצְמָן. תּוֹסָפוֹת: {יב} וַהֲרֵי בְּנֵי אַהֲרֹן הִתְפָּקְדוּ כוּ' דִּכְתִיב וְאֵלֶּה שְׁמוֹת בְּנֵי אַהֲרֹן.</B> וְהָא דְּהֻצְרַךְ לוֹמַר דְּעוֹבֵר בְּיַם סוּף מִשּׁוּם כוּ' שֶׁלֹּא עָבְרוּ לִפְנֵי מֹשֶׁה אֶלָּא הָיָה הוֹלֵךְ וְעוֹמֵד עַל הַפֶּתַח כְּפֵרוּשׁ רַשִׁ"י בְּחֻמָּשׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ה

{יג} <B> אֲבָל כוּ'.</B> זֶה גִּרְסַת הָרַ"ב. אֲבָל הָרַמְבַּ"ם לֹא גָּרַס אֲבָל, וְהָכִי גָּרַס וְחַטָּאוֹת כוּ' נְדָרִים מְקַבְּלִין מִיָּדָן וְכוּ', וְזֶה לְשׁוֹנוֹ וּלְפִיכָךְ מְקַבְּלִין מֵהֶן חַטָּאוֹת וַאֲשָׁמוֹת לְפִי שֶׁכֵּיוָן שֶׁהוּא נִתְכַּוֵּן בְּכַפָּרָה אוּלַי יָשׁוּב: {יד} <B> כָּל שֶׁאֵין כוּ'.</B> פֵּרוּשׁ לַמִּזְבֵּחַ, אַף עַל גַּב דְּנִדָּר וְנִדָּב לְבֶדֶק הַבַּיִת אֵין מְקַבְּלִין מִיָּדָן, וְכֵן הוּא מְפֹרָשׁ כוּ'. וְכֵן נִרְאֶה בַּיְרוּשַׁלְמִי: 

~ פרק א - משנה ו

{טו} כְּלוֹמַר כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם קָלְבּוֹן כוּ' שֶׁחֲצָיֵי הַשְּׁקָלִים הַכֹּל צְרִיכִין כוּ' לְפִיכָךְ וּמְצָרֵף שֶׁקֶל בִּשְׁנֵי חֲצָאִין יִתֵּן לוֹ תּוֹסֶפֶת עַל הַשֶּׁקֶל. אֲבָל הוּא לֹא הִזְכִּיר לְשׁוֹן הֶכְרֵעַ. וְרַשִׁ"י כָּתַב הֶכְרֵעַ וְיֵשׁ אוֹמְרִים חִלּוּף, שֶׁאֵין שְׁנֵי חֲצָאִין יוֹצְאִין בְּסֶלַע בְּלֹא חִלּוּף מֻעָט. מַשְׁמָע דְּפֵרוּשׁ הֶכְרֵעַ הַיְנוּ תּוֹסֶפֶת בְּעָלְמָא כְּעֵין וְחַיָּב לְהַכְרִיעַ דְּסוֹף פֶּרֶק ו' דְּבָבָא בַּתְרָא. וְכֵן כָּתַב הָרַ"ב סוֹף פֶּרֶק קַמָּא דְּחֻלִּין, וּמִשְׁנָה ג' פֶּרֶק ט' דִּבְכוֹרוֹת. וְכָאן אַרְכְּבֵיהּ אַתְּרֵי רִכְשֵׁי: {טז} וְרֵישָׁא דִּתְנָן וְאֵלּוּ שֶׁחַיָּבִין כוּ' הַיְנוּ כְּשֶׁבָּאִין שְׁנַיִם וְנוֹתְנִין שֶׁקֶל שָׁלֵם, וּמְצִיעָתָא הַשּׁוֹקֵל עַל יְדֵי כֹּהֵן כוּ' הַיְנוּ נַמִּי כְּשֶׁשּׁוֹקֵל עַל יָדוֹ וְעַל יְדֵי כֹּהֵן. הָרַמְבַּ"ם: {יז} וְטַעַם זֶה מִפְּנֵי שֶׁהָעָם קוֹרִין שְׁקָלִים לַחֲצִי הַסְּלָעִים שֶׁהָיוּ שׁוֹקְלִין בְּכָל שָׁנָה, וְשֶׁקֶל שֶׁל מֹשֶׁה קוֹרִין סֶלַע כְּתַרְגּוּמוֹ. רַמְבַּ"ן:

{טו} כָּךְ הוּא גִּרְסָתוֹ.</B> וְגִרְסַת הַסֵּפֶר, בַּמִּזְרָח. וּבְשֶׁל מוּסָפִים לְהֵפֶךְ. וְרַשִׁ"י גָּרַס כְּגִרְסַת הַסֵּפֶר, וְעַיֵּן שָׁם טַעֲמוֹ. וְהַתּוֹסָפוֹת וְהַמְּפָרֵשׁ גָרְסֵי כְּגִרְסַת הָרַ"ב, דִּלְהָכִי נְתָנוּם לְאֶבְרֵי תָּמִיד בַּמַּעֲרָב, לְפִי שֶׁהַמַּעֲרָכָה שֶׁעַל הַמִּזְבֵּחַ הוּא לְמִזְרָחוֹ, לְכָךְ נוֹתְנִים הָאֵבָרִים בְּמַעֲרָבוֹ שֶׁל כֶּבֶשׁ, לְפַנּוֹת מְקוֹם הֲלִיכָתָם כְּשֶׁיּוֹלִיכוּם אַחַר כָּךְ לְרֹאשׁ הַמִּזְבֵּחַ עַל הַמַּעֲרָכָה שֶׁהִיא בַּמִּזְרָח: {טז} <B> מִלְּמַטָּה.</B> הָכִי גָּרַס רַשִׁ"י. וּפֵרֵשׁ שֶׁל רֹאשֵׁי חֳדָשִׁים, אֵבָרִים שֶׁל מוּסְפֵי רֹאשֵׁי חֳדָשִׁים נִתָּנִין הָיוּ בַּכֶּבֶשׁ סָמוּךְ לְכַרְכֹּב הַמִּזְבֵּחַ לְמַטָּה הֵימֶנּוּ מְעַט, דְּהַיְנוּ בַּחֵצִי הָעֶלְיוֹן שֶׁל הַכֶּבֶשׁ, שֶׁכָּל הַכֶּבֶשׁ אָרְכּוֹ שְׁלֹשִׁים וּשְׁתַּיִם, וְאֵינוֹ מְשֻׁפָּע וְעוֹלֶה אֶלָּא תֵּשַׁע אַמּוֹת, שֵׁשׁ אַמּוֹת מִן הָאָרֶץ לַסּוֹבֵב וְשָׁלֹשׁ מִן הַסּוֹבֵב וּלְמַעְלָה, נִמְצְאוּ שְׁנֵי שְׁלִישֵׁי אֹרֶךְ הַכֶּבֶשׁ מִן הָאָרֶץ לַסּוֹבֵב. וְגִרְסַת הָרַ"ב מִלְמַעְלָה, עַיֵּן תּוֹסְפוֹת יוֹם טוֹב. וְלִשְׁתֵּי הַגִּרְסָאוֹת טַעֲמָא דְּעָבְדִינַן שִׁנּוּי בְּרֹאשׁ חֹדֶשׁ מְפֹרָשׁ בַּגְּמָרָא דְּסֻכָּה, לֵידַע שֶׁהֻקְבַּע רֹאשׁ חֹדֶשׁ בִּזְמַנּוֹ. וּפֵרֵשׁ רַשִׁ"י הַאי חֲשִׁיבוּתָא עָבְדִינַן לֵיהּ לְהֶכֵּירָא שֶׁיַּכִּירוּ דִּפְשִׁיטָא לְהוּ לְבֵית דִּין שֶׁקִּדְּשׁוּהוּ כְּהִלְכָתוֹ וְלֹא יְגַמְגֵּם לֵב אָדָם עָלָיו, דְּרֹב בְּנֵי אָדָם לֹא רָאוּ חִדּוּשׁ הַלְּבָנָה: {יז} וַאֲנִי</b> לָמַדְתִּי טַעֲמוֹ מִלְּשׁוֹן הָרַמְבַּ"ם שֶׁסִּיֵּם וְאָמַר, אֲבָל אִם הִקְדִּישָׁן הַכֹּהֵן אַחַר שֶׁבָּא לְיָדוֹ וַהֲרֵי הֵם שֶׁלּוֹ יָכוֹל לְהַקְדִּישָׁן. אֲבָל שְׁקָלִים דְּלַמִּזְבֵּחַ יָכוֹל לְהַקְדִּישׁ, כְּדִתְנַן בְּמִשְׁנָה ז' פֶּרֶק ח' דַּעֲרָכִין דְּמַקְדִּישׁ אָדָם קָדְשֵׁי מִזְבֵּח. תּוֹסְפוֹת יוֹם טוֹב. וְעַיֵּן עוֹד:
