{א} <b>מָבוֹי.</b> פֵּרוּשׁ הָר"ר יוֹנָתָן הַכֹּהֵן שֶׁהוּא מִלְּשׁוֹן מְבוֹא הָעִיר [שׁוֹפְטִים א]: {ב} כְּלוֹמַר כְּדִין הַכַּרְמְלִית שֶׁהִזְכִּיר בְּרֵישׁ פֶּרֶק י"א דְּשַׁבָּת, אֲבָל לֹא שֶׁיִּהְיֶה דִּינוֹ כִּרְשׁוּת הַיָּחִיד גָּמוּר לְחַיֵּב הַזּוֹרֵק מֵרְשׁוּת הָרַבִּים לְתוֹכוֹ, וּכְמוֹ שֶׁאִיתָא בַּמִּשְׁנָה דִּלְקַמָּן דַּאֲפִלּוּ הֻכְשַׁר בְּקוֹרָה לֹא הָוֵי כִּרְשׁוּת הַיָּחִיד גָּמוּר: {ג} וְלֹא יוֹעִיל בָּהּ הֶכֵּירָא דְּקוֹרָה. וּלְרַבִּי יְהוּדָה דְּקוֹרָה מִשּׁוּם מְחִצָּה, אֵין צָרִיךְ לְמַעֵט, כִּדְאִיתָא בַּבְּרַיְתָא. הָרִי"כ: {ד} <B> שֶׁהוּא רָחָב.</B> וַאֲפִלּוּ הָיָה גָּבוֹהַּ יוֹתֵר מֵעֶשְׂרִים אַמָּה. הָרַמְבַּ"ם:

{ה} וּלְעֵיל פֵּרֵשׁ דְּרַבָּנָן גָּזוּר כוּ'.</B> הוֹאִיל וְאֵין הַדָּבָר מְפֹרָשׁ בַּתּוֹרָה, אַף עַל פִּי שֶׁמְּקֻבָּל מִסִּינַי וְהוּא דִּבְרֵי תוֹרָה לְהַעֲנִישׁ עָלָיו בָּעֳנָשִׁים וּמִיתָה, מִקְּרֵי דִּבְרֵי סוֹפְרִים, כְּמוֹ שֶׁכָּתַב בַּכֶּסֶף מִשְׁנֶה רֵישׁ הִלְכוֹת אִישׁוּת:

{ו} <B> לְאָרְכּוֹ.</B> בַּגְּמָרָא הַגִּרְסָא לְרָחְבּוֹ, וְהָכִי מִסְתַּבֵּר, דְּכֵיוָן שֶׁהָאָרִיחַ חֲצִי לְבֵנָה וְהַלְּבֵנָה מְרֻבַּעַת שְׁלֹשָׁה עַל שְׁלֹשָׁה, וְנִמְצָא הָאָרִיחַ שֶׁהוּא הַמֶּחֱצָה לֹא פִּחֲתָה מִן הָאֹרֶךְ רַק מִן הָרֹחַב בִּלְבַד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ז} וּמְפָרֵשׁ בַּגְּמָרָא דְּבָא לְהוֹסִיף שֶׁאַף עַל פִּי שֶׁהִיא מִמִּין שֶׁאֵינָהּ בְּרִיאָה לְקַבֵּל בִּנְיָן כָּזֶה, אֲפִלּוּ הָכִי כָּשֵׁר. וַעֲקוּמָה מְפָרֵשׁ בַּגְּמָרָא שֶׁעַקְמִימוּתָהּ חוּץ לַמָּבוֹי, וְאִלּוּ יִנָּטֵל עַקְמִימוּתָהּ יֵשׁ בֵּין זֶה לָזֶה שְׁלֹשָׁה טְפָחִים, אֲפִלּוּ הָכִי רוֹאִין כְּאִלּוּ כֻּלָּהּ תּוֹךְ הַמָּבוֹי עוֹמֶדֶת. וְכֻלָּהּ מַתְנִיתִין רַבִּי יְהוּדָה הִיא, כֵּן כָּתַב הָרִי"כ. אֲבָל הָרַמְבַּ"ם וְהָרֹא"שׁ כָּתְבוּ דִּכְכֻלֵּי עָלְמָא אָתְיָא, אֶלָּא צָרִיךְ דְּאִלּוּ יִנָּטֵל עַקְמִימוּתָהּ שֶׁלֹּא יְהֵא בֵּין זֶה לָזֶה שְׁלֹשָׁה טְפָחִים:

{ח} וְהוּא הַדִּין בְּקוֹרָה שֶׁצָּרִיךְ שֶׁלֹּא תִּהְיֶה לְמַטָּה מֵעֲשָׂרָה, וְלֹא תָּנֵי בְּקוֹרָה דְּסָמַךְ עַל מִשְׁנָה זוֹ שֶׁל לֶחִי שֶׁשִּׁעוּרוֹ עֲשָׂרָה כְּשִׁעוּר מָבוֹי. הָרִי"כ: {ט} שֶׁאֲפִלּוּ יַפְלִיגֶנּוּ מִן הַכֹּתֶל בִּשְׁלֹשָׁה דְּלֹא לֵיתֵי אֲוִירָא דְּהַאי גִּיסָא וּדְהַאי גִּיסָא וּלְבַטְלֵיהּ כִּדְאַשְׁכְּחַן בִּמְחִצּוֹת הַכִּלְאַיִם כוּ', וּמִשּׁוּם הָכִי מַחְמִיר רַבִּי יְהוּדָה, וַאֲפִלּוּ בְּסוֹמְכוֹ לְגַמְרֵי לַכֹּתֶל בָּעֵי שְׁלֹשָׁה טְפָחִים בְּרָחְבּוֹ. הָרִי"כ:

{י} <B> רוּחַ חַיִּים.</B> וְלֹא חָיְשִׁינַן שֶׁמָּא תָּמוּת, דְּמִיתָה לֹא שְׁכִיחָא. וְשֶׁמָּא תִּבְרַח אוֹ תִּרְבַּץ תַּחְתֶּיהָ לֹא חָיְשִׁינַן, דְּאָיְרֵי בִּמְתוּחָה בְּאַשְׁלֵי מִלְּעֵיל, כְּלוֹמַר קְשׁוּרָה בַּחֲבָלִים לְכָתְלֵי הַמָּבוֹי. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} וְתִכָּוֵּץ: {יב} וְקָשֶׁה, דְּלִפְלוֹג נַמִּי בָּרֵישָׁא.</B> וּבַגְּמָרָא וְרִי"ף וִירוּשַׁלְמִי אִיתָא בָּרֵישָׁא רַבִּי מֵאִיר אוֹסֵר: {יג} כְּדִמְפֹרָשׁ בְּמִשְׁנָה ג' פֶּרֶק ב' דְּגִטִּין:

{יד} <B>שַׁיָּרָא.</B></B> תַּרְגּוּם אֹרְחַת יִשְׁמְעֵאלִים שְׁיָרַת עֲרָבָאֵי. הָרַמְבַּ"ם: {טו} <B> וְהִקִּיפוּהָ.</B> מֵעֶרֶב שַׁבָּת בִּכְלֵי בְּהֵמָה כְּגוֹן מַרְדָּעוֹת וְאֻכָּפִין שֶׁל בְּהֵמָה, וְהִיא מְחִצָּה שֶׁהַרְבֵּה מִמֶּנָּה פָּרוּץ, כְּגוֹן הֶחָלָל שֶׁבֵּין אֻכָּף לְאֻכָּף, אֲפִלּוּ הָכִי הָוֵי מְחִצָּה וּמֻתָּר לְטַלְטֵל בְּכָל הַבִּקְעָה שֶׁהָיְתָה מִתְּחִלָּה כַּרְמְלִית, שֶׁהֲרֵי עַכְשָׁיו הִיא מֻקֶּפֶת לְדִירָה. הָרִי"כ: {טז} <B> עֲשָׂרָה טְפָחִים.</B> מֵהַטַּעַם שֶׁכָּתְבוּ חֲכָמִים בְּסֻכָּה רֵישׁ פֶּרֶק קַמָּא, דְּהָא דְּבָעִינַן בְּשַׁבָּת מְחִצָּה גָּבוֹהַּ עֲשָׂרָה מֵהָתָם יָלְפִינַן: {יז} וְכֵן כָּתַב הָרַמְבַּ"ם, וְשֶׁכֵּן הוּא בְּכָל הַתּוֹרָה:

{יח} <B>וּמִדְּמַאי.</B> דִּתְנַן בְּפֶרֶק ג' דִּדְמַאי מַאֲכִילִין אֶת הָעֲנִיִּים דְּמַאי [וְאֶת הָאַכְסַנְיָא דְּמַאי]. וּפֵרֵשׁ שָׁם הָרַ"ב דְּאִם לָנוּ, חַיָּבִין לְתַקֵּן. וְהִקְשָׁה הָרַ"שׁ מִדְּהָכָא דִּתְנַן גַּבָּהּ וּמִלְּעָרֵב שְׁמַע מִנָּהּ דְּשָׁבְתוּ וְלָנוּ אָיְרִינַן, וַאֲפִלּוּ הָכִי פְּטוּרִים מִדְּמַאי. וְנִרְאֶה לִי דְּלִצְדָדִין קָתָנֵי, דִּפְטוּרִים מִדְּמַאי כְּשֶׁלֹּא לָנוּ, וְאִם לָנוּ וְשָׁבְתוּ חַיָּבִין בִּדְמַאי, אֲבָל פְּטוּרִין מִלְּעָרֵב. תּוֹסְפוֹת יוֹם טוֹב: {יט} דְּיֵשׁ לָהֵן סְמָךְ מִקְּרָאֵי:

{א} <B>גָּבְהָן עֲשָׂרָה כוּ'.</B> גָּבְהָן עֲשָׂרָה שֶׁאֵין נִקְרֵאת מְחִצָּה בְּפָחוֹת מֵעֲשָׂרָה, וְרָחְבָּן שִׁשָּׁה טְפָחִים דְּזֶהוּ אַמָּה, שֶׁאַף עַל פִּי שֶׁבִּמְחִצַּת חָצֵר סַגִּי בְּפַס אַרְבָּעָה, לֹא דָּמִי, דְּהָתָם בְּחָצֵר אִיכָּא שָׁלֹשׁ מְחִצּוֹת שְׁלֵמוֹת אֲבָל בְּפַסֵּי בֵּירָאוֹת שֶׁרֻבּוֹ פָּרוּץ מִכָּל צַד מַצְרִיכִין פַּס שֶׁל שִׁשָּׁה טְפָחִים. רִי"כ: {ב} <B> קְשׁוּרוֹת.</B> וְאַף עַל פִּי שֶׁאֵין הַפָּרוֹת מוֹשְׁכוֹת בָּעֹל בְּשַׁבָּת אֲפִלּוּ הָכִי מְשַׁעֲרִין בְּהוּ, לְפִי שֶׁגַּם בִּימוֹת הַחוֹל נִכְנָסוֹת שָׁם לִשְׁתּוֹת, וּפְעָמִים שֶׁכְּשֶׁהַצֶּמֶד הָאֶחָד נִכְנָס לִשְׁתּוֹת מוֹצֵא הַשֵּׁנִי שֶׁשָּׁתָה מִשָּׁם וְרוֹצֶה לָצֵאת, וְאִם יִהְיוּ בֵּין הַדְּיוּמְדִין פָּחוֹת מִזֶּה הַשִּׁעוּר יָבֹאוּ הַפָּרוֹת וְיַפִּילוּ הַדְּיוּמְדִין לָאָרֶץ. הָרִי"כ: {ג} הָכִי תָּנוּ רַבָּנָן.</B> וּפֵרֵשׁ רַשִׁ"י וְלֹא פָרָה נִכְנֶסֶת וּפָרָה יוֹצֵאת, דְּאִם כֵּן לֹא הֲוָה קְשׁוּרוֹת: {ד} רְצוֹנוֹ לוֹמַר דְּאִלּוּ עֶשֶׂר אַמּוֹת לְרַבִּי מֵאִיר וְכֵן הַשְּׁלֹשׁ עֶשְׂרֵה וּשְׁלִישׁ לְרַבִּי יְהוּדָה לֹא בָּעִינַן מְצֻמְצָמוֹת. וְכֵן הוּא בְּרַשִׁ"י וְרַמְבַּ"ם:

{א} <B> נְתָנוּהוּ.</B> לְפִי שֶׁדֶּרֶךְ הוּא כְּשֶׁמּוֹלִיכִין הָאָדָם מוֹלִיכִין אוֹתוֹ אֶל עִיר, הִלְכָּךְ קָתָנֵי הוֹלִיכוּהוּ גַּבֵּי עִיר. אֶלָּא שֶׁאִם אֵרַע לִפְעָמִים שֶׁלֹּא יוּכְלוּ לָבֹא אֶל עִיר אוֹ מִפְּנֵי סִבָּה וּמִקְרֶה שֶׁיִּתְּנוּהוּ לְמָקוֹם אַחֵר הַמִּשְׁתַּמֵּר כְּמוֹ בְּדִיר אוֹ סַהַר, לְכָךְ תָּנֵי גַּבַּיְהוּ נְתָנוּהוּ: {ב} וְאֵין הָכִי נַמִּי דְּבָעִיר נַמִּי גָּזְרוּ דְּהָא וַדַּאי מַשְׁמַע דְּאַכָּל מַה שֶּׁהֻזְכַּר בְּדִבְרֵי רַבָּן גַּמְלִיאֵל וְרַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה קָמִפְלְגֵי, וְנָקְטֵי חֲדָא מִתַּרְתֵּי בָּבֵי דְּתַנָּא קַמָּא, וְהוּא הַדִּין לְאִידָךְ. תּוֹסְפוֹת יוֹם טוֹב: {ג} לְפִי שֶׁאֵינוֹ דּוֹמֶה לְדִיר וְסַהַר אֶלָּא כְּשֶׁהִיא קְבוּעָה בַּנָּמָל כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם. אִי נַמִּי כְּפֵרוּשׁ רַשִׁ"י דִּכְשֶׁהַסְּפִינָה מְהַלֶּכֶת כְּשֶׁעוֹקֵר רַגְלוֹ קֹדֶם שֶׁיַּנִּיחֵהוּ מוֹצִיאָתוֹ הַסְּפִינָה מֵאַרְבַּע אַמּוֹת שֶׁלּוֹ וְנִכְנָס בְּאַרְבַּע אַמּוֹת אֲחֵרוֹת עַל כָּרְחוֹ, וְהָוֵי כְּמוֹ שֶׁהוֹצִיאוּהוּ לִסְטִים וְנוֹתְנִין אוֹתוֹ תָדִיר בְּאַרְבַּע אַמּוֹת אֲחֵרוֹת, שֶׁנּוֹתְנִין לוֹ אַרְבַּע אַמּוֹת, וְכֵן לְעוֹלָם: {ד} [וְטַעֲמָא] כִּדְאִיתָא בַּגְּמָרָא דִּלְמָא עָמְדָה פִּתְאוֹם וְלָאו אַדַּעְתַּיְהוּ.</B> וּלְטַעֲמָא דְּשָׁבְתוּ בַּאֲוִיר כוּ' נִרְאֶה דְּהוֹאִיל וְיָצְאוּ חוּץ לַתְּחוּם רָצוּ לְהַחְמִיר, שֶׁלֹּא תִטְעֶה וְלֹא תְחַלֵּק בֵּין יוֹצֵא לְחוּץ לַתְּחוּם וְשָׁבַת בַּמְּחִצּוֹת הָהֵן מִבְּעוֹד יוֹם כְּמוֹ בִּסְפִינָה לְהֵיכָא דְּלֹא שָׁבַת כְּמוֹ בְּדִיר וְסַהַר וְתַתִּיר הַכֹּל: 

~ פרק ד - משנה ג

{ה} גְּמָרָא.</B> וְטַעֲמָא שֶׁמָּא יַרְגִּישׁוּ בָּהֶם הָאוֹיְבִים שֶׁיָּצְאוּ וְיִרְדְּפוּ אַחֲרֵיהֶן וְאֵין כְּלֵי זַיִן בְּיָדָם לְהַצִּיל. וְלִמְקוֹמָן הַיְנוּ לַמָּקוֹם שֶׁנָּתְנוּ לָהֶם חֲכָמִים, שֶׁהוּא עַד אַלְפַּיִם אַמָּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ד - משנה ד

{ו} וְהַמְּדִידָה הוּא מַהֲלַךְ אַלְפַּיִם פְּסִיעוֹת בֵּינוֹנִית שֶׁלֹּא יַרְחִיק רַגְלָיו, וְזֶה שִׁעוּר אַמָּה אַחַת בֵּין רֶגֶל לְרֶגֶל. הָרִי"כ: {ז} <B> יִכָּנֵס.</B> כֵּיוָן שֶׁאִלּוּ הָיָה יוֹדֵעַ שֶׁהָעִיר תּוֹךְ תְּחוּמוֹ לֹא הָיָה קוֹנֶה שְׁבִיתָה אֶלָּא עִם בְּנֵי הָעִיר, הֲרֵי הוּא כְּמִי שֶׁקָּנָה שְׁבִיתָה עִמָּהֶם. בֵּית יוֹסֵף: 

~ פרק ד - משנה ה

{ח} וּמִטַּעֲמָא אַחֲרִינָא, הוֹאִיל וְנֵעוֹר קוֹנֶה יָשֵׁן נַמִּי קוֹנֶה, כִּדְפֵרֵשׁ רַשִׁ"י: {ט} הַיְנוּ בְּלֹא רֹאשׁוֹ, וּבְפִשּׁוּט יָדַיִם (וְרַגְלַיִם) נִכְנָס הָרֹאשׁ. תּוֹסָפוֹת: {י} <B>וְהוּא בְאֶמְצָעָן.</B> נָטַר עַד הָכָא וְהוּא הַדִּין לְבִתְרֵי בָּבֵי דִּבְמִשְׁנָה קַמַּיְתָא: {יא} <B>וּמוֹדֶה רַבִּי יְהוּדָה.</B> וְהַיְנוּ [דְּאִיכָּא בֵּינַיְהוּ דְּרַבִּי יְהוּדָה וַחֲכָמִים] דְּרַבָּנָן סָבְרֵי אַרְבַּע לְכָל צְדָדָיו הַיְנוּ שְׁמֹנֶה עַל שְׁמֹנֶה, וְרַבִּי יְהוּדָה סָבַר לְחַד רוּחַ וְתוּ לֹא: 

~ פרק ד - משנה ו

{יב} דַּחֲפָצָיו כְּרַגְלָיו כְּדִתְנַן [בֵּיצָה פ"ה מ"ג] הַבְּהֵמָה וְהַכֵּלִים כְּרַגְלֵי הַבְּעָלִים: 

~ פרק ד - משנה ז

{יג} וְחָשְׁכָה לוֹ הַיְנוּ אִם יִרְצֶה לֵילֵךְ לְבֵיתוֹ, אֲבָל לְעִקָּרוֹ שֶׁל אִילָן אָזֵיל. וְאִיכָּא דְּאָמְרֵי בַּגְּמָרָא דְּחָשְׁכָה לוֹ אִם יֵלֵךְ מְעַט מְעַט אֲבָל כִּי רָהֵיט מָטֵי לְהָתָם, וְכֵן הַסְכָּמַת הַפּוֹסְקִים: {יד} וְהָא דִּבְפֶרֶק ג' מִשְׁנָה ד' אֲפִלּוּ לְמַאן דְּאָמַר דִּסְפֵק עֵרוּב אֵינוֹ עֵרוּב מִכָּל מָקוֹם יֵשׁ לוֹ אַלְפַּיִם שֶׁמִּבֵּיתוֹ כְּמוֹ שֶׁכָּתַב הָרַ"ב שָׁם, וְלֹא אָמְרִינַן שֶׁלֹּא יָזוּז מִמְּקוֹמוֹ, הָתָם כְּעוֹמֵד בְּעִירוֹ לֹא עָקַר דַּעְתּוֹ מֵעִירוֹ, וְדַעְתּוֹ שֶׁאִם לֹא יִקְנֶה לוֹ שָׁם שְׁבִיתָה שֶׁיִּהְיֶה לוֹ תְּחוּם בֵּיתוֹ. תּוֹסָפוֹת: {טו} <B> אַלְפַּיִם אַמָּה.</B> דְּהָא עִקַּר עֵרוּב בָּרֶגֶל כִּדְמַסְקִינַן בְּמִשְׁנָה ט', וְהָכָא דְּדָחוּק הוּא דְּחָשְׁכָה לוֹ שָׁרִינַן לֵיהּ בְּאוֹמֵר שְׁבִיתָתִי כוּ': 

~ פרק ד - משנה ח

{טז} <B> זָכָה לוֹ.</B> וְקָא מַשְׁמַע לָן שֶׁאֵין נִקְרָא זֶה שְׁבִיתָה בְּטָעוּת כִּדְאָמַר רַבִּי יְהוּדָה בְּמִשְׁנָה ד', דְּלֹא דָּמִי מִי שֶׁהוּא סָמוּךְ לָעִיר וְלֹא יָדַע, דְּנִכָּר לַכֹּל וּמְפֻרְסָם שֶׁשְּׁבִיתָתוֹ בְּטָעוּת, מַה שֶּׁאֵינוֹ נִכָּר [לוֹמָר] דְּאִלּוּ הָיָה יוֹדֵעַ הָאִילָן אוֹ הַגָּדֵר לֹא הָיָה אוֹמֵר תְּהֵא שְׁבִיתָתִי בִּמְקוֹמִי. הָרִי"כ: {יז} וְלֹא קַשְׁיָא אַמַּה שֶּׁכָּתַב בְּפֶרֶק ג' מִשְׁנָה ה' דִּתְחוּמִין דְּרַבָּנָן.</B> שֶׁהֲרֵי כָּתַב הַטּוּר סִימָן שצ"ז דִּקְרָאֵי אַסְמַכְתָּא בְּעָלְמָא: 

~ פרק ד - משנה ט

{יח} כְּלוֹמַר וּפְלוּגְתַּיְהוּ בְּאוֹמֵר שְׁבִיתָתִי בִּמְקוֹמִי [וְאֵינוֹ] עָנִי, שֶׁיֵּשׁ לוֹ פַּת: {יט} הָכִי קָאָמַר, אַף עַל פִּי שֶׁיֵּשׁ עִמּוֹ פַּת הָוֵי כְּמִי שֶׁאֵין עִמּוֹ פַּת, הוֹאִיל וּבַדֶּרֶךְ הוּא וְחָשְׁכָה לוֹ, כִּדְמַסִּיק שֶׁהָיָה יָכוֹל לָרוּץ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} <B> הֶעָנִי כוּ'.</B> אֲתָאָן לְרַבִּי מֵאִיר, וּכְלוֹמַר בְּזוֹ אָמְרוּ הֶעָנִי כוּ', אֲבָל עִקַּר עֵרוּב בְּפַת גְּמִירָא וְטַעֲמֵיהּ דְּכֵיוָן שֶׁמֵּבִיא פַּת גִּלָּה דַּעְתּוֹ וְנִכָּר שֶׁרְצוֹנוֹ לִשְׁבֹּת שָׁם. אֲבָל כְּשֶׁמְּעָרֵב בְּרַגְלָיו אָמְרוּ אֱינָשֵׁי אִם תַּלְמִיד חָכָם הוּא שְׁמַעְתֵּיהּ מְשַׁכְתֵּיהּ, וְאִם עַם הָאָרֶץ חֲמוֹרוֹ אִתְאַבִיד לֵיהּ וְהוֹלֵךְ לְבַקְּרוֹ שָׁם. וְרַבִּי יְהוּדָה סָבַר כֵּיוָן שֶׁטּוֹרֵחַ הוּא בְּעַצְמוֹ יֵשׁ הֶכֵּירָא יוֹתֵר מִשֶּׁשּׁוֹלֵחַ שָׁם [שָׁלִיחַ] וּבְיָדוֹ פַּת:

{א} דַּהֲוֵי רְשׁוּת הַיָּחִיד כִּדְפֵּרֵשׁ רֵישׁ פֶּרֶק י"א דְּשַׁבָּת:  {ב} גָּבוֹהַּ עֲשָׂרָה אוֹ נָמוֹךְ. אַף עַל גַּב דְּהַיְנוּ הָךְ, דְּאִם הָאֶחָד גָּבוֹהַּ עֲשָׂרָה בְּהֶכְרֵחַ הַשֵּׁנִי נָמוּךְ מִמֶּנּוּ עֲשָׂרָה, נָקַט הָכִי מִשּׁוּם דְּלִפְעָמִים אֵין בַּיִת אֶחָד גָּבוֹהַּ יוֹתֵר מֵחֲבֵרוֹ, אֶלָּא שֶׁהָאֶחָד נִבְנָה בָּעֵמֶק שֶׁהַקַּרְקַע נָמוּךְ, וְאָז לֹא שַׁיָּךְ לוֹמַר שֶׁזֶּה גָּבוֹהַּ אֶלָּא שֶׁזֶּה נָמוּךְ: {ג} <B> וּבְלֹא הֻקַּף לְדִירָה, וּכְרַבִּי עֲקִיבָא בְּפֶרֶק ב' מִשְׁנָה ה' דַּהֲלָכָה כְּמוֹתוֹ.</B> וְאִי הֻקַּף לְדִירָה אֲפִלּוּ בְּיָתֵר נַמִּי, טוּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} <B> וּבִלְבַד שֶׁלֹּא יִכָּנֵס בִּכְלִי זֶה לְבַיִת מִן הַבָּתִּים, אֶלָּא אִם כֵּן עֵרְבוּ.</B> הָרַמְבַּ"ם: {ה} וְאַף עַל גַּב דְּהַשְׁתָּא שְׁכִיחָא מָאנֵי דְּבָתִּים בֶּחָצֵר, אֲפִלּוּ הָכִי כֵּלִים שֶׁשָּׁבְתוּ לְתוֹכָן שָׁרֵי לְטַלְטוֹלֵי, וְלֹא גָּזְרִינַן דִּלְמָא אָתֵי לַאֲפוֹקֵי מָאנֵי דְּבָתִּים לֶחָצֵר:

{ו} גַּג גָּדוֹל. פֵּרֵשׁ הָרַמְבַּ"ם כְּשֶׁיֵּשׁ כֹּתֶל מַקִּיף בַּגַּגּוֹת, וְכֵן נִרְאֶה מִדִּבְרֵי הָרַ"ב (אֶלָּא שֶׁלֹּא פֵּרֵשׁ בְּהֶדְיָא לְפִי שֶׁאֵין צָרִיךְ כְּתָלִים מַמָּשׁ אֶלָּא כְּשֶׁאֵין הַגַּגִּין בּוֹלְטוֹת מֵהַמְּחִצּוֹת אָמְרִינַן גּוּד אַסִּיק), וְכִדְרַב בַּגְּמָרָא. וּמְפָרֵשׁ בַּגְּמָרָא דִּלְהָכִי תָּנֵי גַּג וְחָצֵר, דְּגַג דּוּמְיָא דְּחָצֵר, מֶה חָצֵר מִנְכְּרָא מְחִצּוֹתֶיהָ, אַף גַּג מִינְכְּרָא מְחִצּוֹתֶיהָ. דְּאִי לֹא, לֹא אָמְרִינַן גּוּד אַסִּיק, וְגָדוֹל נַמִּי אָסוּר:  {ז} דְּאִלּוּ כֵּלִים שֶׁשָּׁבְתוּ בְּתוֹכָן מֻתָּרִין בִּשְׁתֵּיהֶן כִּדְקָיְמָא לָן לְרַבִּי שִׁמְעוֹן: {ח} כְּלוֹמַר בִּשְׁבִיל אוֹתָן גִּפּוּפִין שָׁרֵי בְּכָל הַגַּג וּכְלִישָׁנֵיהּ [לְקַמָּן] גַּבֵּי חָצֵר: {ט} גְּדוֹלָה מֻתֶּרֶת. אִם עֵרְבָה לְעַצְמָהּ. רַשִׁ"י: {י} <B> לִרְשׁוּת הָרַבִּים.</B> אֲפִלּוּ בְּשַׁבָּת. טוּר סִימָן שע"ד: {יא} <B> מִתּוֹכָהּ לִרְשׁוּת הָרַבִּים.</B> וְכָתַב רַשִׁ"י וְהוּא הַדִּין מִתּוֹכָהּ לִרְשׁוּת הַיָּחִיד, וְקָשֶׁה דְהֲווּ לֵיהּ לְמֵימַר בְּלִישְׁנֵיהּ דְּרַבִּי אֱלִיעֶזֶר מִתּוֹכָהּ לִרְשׁוּת הַיָּחִיד. וּלְמַסְקָנָא דִּגְמָרָא נִיחָא, דִּבְתוֹכָהּ לְכֻלֵּי עָלְמָא כַּרְמְלִית הוּא. כִּי פְלִיגֵי בִּמְקוֹם מְחִצָּה, דְּרַבִּי אֱלִיעֶזֶר סָבַר צִדֵּי רְשׁוּת הָרַבִּים כִּרְשׁוּת הָרַבִּים, וּמְקוֹם מְחִצָּה זוֹ צִדֵּי רְשׁוּת הָרַבִּים הִיא, וְרַבָּנָן סָבְרֵי דְּצִדֵּי רְשׁוּת הָרַבִּים לָאו כִּרְשׁוּת הָרַבִּים. וּמִתּוֹכָהּ דְּרַבִּי אֱלִיעֶזֶר מִתּוֹךְ מְקוֹם מְחִצָּתָהּ קָאָמַר, וּמִשּׁוּם דְּרַבָּנָן נָקְטֵי מִתּוֹכָהּ נָקַט אִיהוּ נַמִּי מִתּוֹכָהּ. וְרַבָּנָן הָכִי קָאָמְרֵי מִי לֹא קָא מוֹדֵית לָן דְּמִתּוֹכָהּ לִרְשׁוּת הָרַבִּים פָּטוּר דְּכַרְמְלִית הִיא, צִדֵּי נַמִּי לֹא שְׁנָא. וְרַבִּי אֱלִיעֶזֶר, הָתָם לֹא קָא דָּרְסוּ לַהּ רַבִּים, וְהָכָא דָּרְסוּ לַהּ רַבִּים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יב} אַף עַל גַּב דְּלִפְנִים בִּמְקוֹם שֶׁכָּלְתָה הַפִּרְצָה מֵהַכְּתָלִים מִשָּׁם וָאֵילָךְ יֵשׁ תִּקְרָה, לֹא אָמְרִינַן יוֹרֵד וְסוֹתֵם הוֹאִיל וּבָאֲלַכְסוֹן הוּא. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:  {יג} כְּלוֹמַר דִּלְעֵיל בִּקְטַנָּה שֶׁנִּפְרֶצֶת לִגְדוֹלָה אָמְרִינַן דְּהֻתְּרָה כוּ', וְעַל זֶה קָאָמַר וְלֹא דָּמִי לְדִלְעֵיל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יד} <B> וַחֲכָמִים אוֹסְרִין.</B> הוֹאִיל וּמְפֻלָּשׁ לֹא אָמְרִינַן פִּי תִּקְרָה יוֹרֵד וְסוֹתֵם. וּבְמַתְנִיתִין דִּלְעֵיל הֲוָה מָצֵי לְאוֹקְמַהּ בִּשְׁתֵּי רוּחוֹת זוֹ כְּנֶגֶד זוֹ וּכְרַבָּנָן, כָּתְבוּ הַתּוֹסָפוֹת דְּנִיחָא לֵיהּ לְאוֹקְמֵיהּ בְּכָל עִנְיָן:  {טו} <B> וְעוֹד אָמַר רַבִּי יְהוּדָה.</B> כְּלוֹמַר וְעוֹד זוֹ מַתִּיר אַף עַל גַּב דְּלֵיכָּא פִּי תִּקְרָה. גְּמָרָא. מִכְּלַל הָא כָּל שֶׁכֵּן הַךְ לְעֵיל דְּאִיכָּא נַמִּי פִּי תִּקְרָה, אֶלָּא דְּאָמַר רַבִּי יְהוּדָה לְהָא דִּלְעֵיל דְּקָסָבַר דְּרַבָּנָן יוֹדוּ לוֹ בְּהַהוּא: {טז} <B> מְעָרְבִין.</B> מִשְׁתַּתְּפִין. הָרִי"כ:

{א} רְצוֹנוֹ לוֹמַר וְיֵשׁ לָחוּשׁ שֶׁיִּתְבַּזּוּ, וּכְמוֹ שֶׁכָּתַב [לְקַמָּן] בְּדִבּוּר הַמַּתְחִיל בַּיְשָׁנוֹת, אֲבָל מֵחֲשַׁשׁ גַּנָּבֵי לֹא הִתִּירוּ:  {ב} וְכֵן כָּתְבוּ הַפּוֹסְקִים, וְאִם כֵּן סְבִירָא לְהוּ דְּשַׁבָּת זְמַן תְּפִלִּין הִיא, וְתֵימַהּ דִּבְמִשְׁנָה ג' פֶּרֶק ג' דִּבְרָכוֹת מְפָרֵשׁ הָרַ"ב דְּלָאו זְמַן תְּפִלִּין הוּא, וְכֵן הַסְכָּמַת כָּל הַפּוֹסְקִים, וְצָרִיךְ עִיּוּן, תּוֹסְפוֹת יוֹם טוֹב. וּמַהַרְשָׁ"א מְיַשֵּׁב עַל זֶה דַּאֲפִלּוּ לְמַאן דְּאָמַר לָאו זְמַן תְּפִלִּין הוּא אָסוּר לִלְבֹּשׁ יוֹתֵר מִזּוּג אֶחָד, דְּלֹא שָׁרֵי לֵיהּ יוֹתֵר מִמַּלְבּוּשׁוֹ בַּחֹל. וְהָא דְּאוֹקִימְנָא כְּמַאן דְּאָמַר זְמַן כוּ' מִשּׁוּם דִּסְתָמָא דְּמַתְנִיתִין כְּרַבִּי מֵאִיר וְרַבִּי מֵאִיר סְבִירָא לֵיהּ זְמַן תְּפִלִּין הוּא. וְהַמָּגֵן אַבְרָהָם תֵּרֵץ כֵּיוָן דַּאֲנַן לֹא בְּקִיאִין לְכַוֵּן מְקוֹם הֲנָחָתָן וְאִם יָנִיחַ שֶׁלֹּא בִּמְקוֹמוֹ יִהְיֶה מַשְּׂאוֹי, לָכֵן לֹא יָבִיא אֶלָּא זוּג זוּג: {ג} וְאַף עַל גַּב דְּהַשְׁתָּא אֵין בּוֹ קֶשֶׁר רַשַּׁאי לְלָבְשׁוֹ, דַּאֲפִלּוּ הָכִי הָוֵי דֶּרֶךְ מַלְבּוּשׁ, אִי נַמִּי דְּעָנְבוֹ. תּוֹסָפוֹת.: {ד} דְּאִסּוּר נַמִּי אִיכָּא: {ה} אַכֻּלָּהּ מַתְנִיתִין קָאֵי, בֵּין מְעַט בֵּין צְבָתִים, מְכַסָּן בְּטַלִּיתוֹ שֶׁלֹּא יֻטְּלוּ כָּל כָּךְ בְּבִזָּיוֹן, וּמַנִּיחָן שָׁם וְהוֹלֵךְ וְלֹא יְזִיזֵם מִשָּׁם שֶׁמָּא יַרְגִּישׁוּ בּוֹ וְיִתְחַיֵּב לַמַּלְכוּת. רַשִׁ"י: {ו} וְנִרְאֶה לִי דְּרַבֵּנוּ הַקָּדוֹשׁ הֻכְרַח לְהַחְסִיר לְמִשְׁנָתֵנוּ, דְּהָא חַכְמֵי הַמִּשְׁנָה לֹא רָצוּ לְגַלּוֹת הָא דְּפָחוֹת פָּחוֹת מֵאַרְבַּע אַמּוֹת, כְּמוֹ שֶׁכָּתַבְנוּ בְּרֵישׁ פֶּרֶק כ"ד דְּשַׁבָּת:

{ז} <B>לַחֲבֵרוֹ.</B> וְתַנָּא קַמָּא סָבַר פָּחוֹת מֵאַרְבַּע אַמּוֹת עָדִיף, דְּאִי אָמַרְתְּ נוֹתְנוֹ לַחֲבֵרוֹ כוּ' אַוְשָׁא מִלְּתָא דְּשַׁבָּת. גְּמָרָא: {ח} <B> בְּנוֹ.</B> דְּאִיכָּא דְּבַר מִצְוָה, דְּצָרִיךְ הוּא לְהַרְחִיצוֹ וּלְסוּכוֹ בְּיוֹם הִוָּלְדוֹ. וְאַלִּיבָּא דְּתַנָּא קַמָּא שָׁרֵי פָּחוֹת [פָּחוֹת] מֵאַרְבַּע אַמּוֹת. רִי"כ: {ט} <B> אֲפִלּוּ מֵאָה.</B> דְּאַף עַל גַּב דְּקַשְׁיָא לֵיהּ יָדָא, כְּלוֹמַר מַה שֶׁמְּטַלְטְלִין אוֹתוֹ מִיָּד לְיָד, אֲפִלּוּ הָכִי הָא עֲדִיפָא מִשֶּׁיּוֹלִיכֶנּוּ פָּחוֹת כוּ' גְּזֵרָה כוּ'. גְּמָרָא: {י} וּפֵרְשׁוּ הַתּוֹסָפוֹת חוּץ לַתְּחוּם דְּחָבִית, אֲבָל לֹא חוּץ לַתְּחוּם שֶׁל הַזּוֹכֶה [בּוֹ], כְּדִתְנַן מִשְׁנָה ה' פֶּרֶק ה' דְּבֵיצָה דִכְרַגְלֵי הַמְמַלֵּא. אִי נַמִּי שֶׁלֹּא נִתְכַּוֵּן שׁוּם אֶחָד שֶׁיִּזְכֶּה לֹא לוֹ וְלֹא לַאֲחֵרִים, אֶלָּא כָּל אֶחָד לְמַה שֶּׁשּׁוֹתֶה: {יא} לְעֵיל בְּפֶרֶק ד' מִשְׁנָה ה':

{טז} הָכִי אִיתָא בַּגְּמָרָא, דְּלֹא תִקְשֶׁה מִמִּשְׁנָה ד' דְּמֻתָּר לַעֲמֹד בִּרְשׁוּת הַיָּחִיד וּלְטַלְטֵל בִּרְשׁוּת הָרַבִּים, אֶלָּא דִּלְעֵיל בְּכֵלִים שֶׁאֵינָן צְרִיכִין לוֹ וְהָכָא בִּצְרִיכִין לוֹ: {יז} וְסָרְסֵהוּ וּפָרְשֵׁהוּ כָּךְ. וְקוֹלֵט בַּכְּלִי מִן הַמַּיִם הַיּוֹרְדִים מִן הַמַּזְחֵלָה שֶׁהִיא לְמַטָּה מֵעֲשָׂרָה טְפָחִים וְכָתַב רַשִׁ"י דְּהוּא הַדִּין [כְּשֶׁהַמַּזְחֵלָה] לְמַעְלָה מֵעֲשָׂרָה שָׁרֵי וְנָקַט לְמַטָּה מֵעֲשָׂרָה מִשּׁוּם שַׁרְיוּתָא דְּצִנּוֹר בִּמְצָרֵף: {יח} כְּלוֹמַר דִּבְהָכִי עָסְקִינַן, דִּסְתָם מַזְחֵלָה הִיא בְּפָחוֹת מִשְּׁלֹשָׁה: {יט} כְּלוֹמַר דְּאִלּוּ לְמַעְלָה מֵעֲשָׂרָה לֹא מַשְׁכַּחַת שׁוּם צַד הֶתֵּר בְּצֵרוּף: {כ} כְּלוֹמַר שְׁלֹשָׁה טְפָחִים, דְּלֹא הָוֵי כְּלָבוּד לַגַּג. וּמִשּׁוּם דִּסְתָם צִנּוֹר כֵּן הוּא הִלְכָּךְ נָקַט בְּצִנּוֹר. וְהוּא הַדִּין נַמִּי לְמַזְחֵלָה הַבּוֹלֶטֶת שְׁלֹשָׁה טְפָחִים דְּדִינָא הָכִי: {כא} דְּאַף עַל גַּב דְּמַתְנִיתִין בִּלְמַטָּה מֵעֲשָׂרָה וְלָבוּד נַמִּי לֵיכָּא כִּדְאָמְרִינַן, אֲפִלּוּ הָכִי אַרְבָּעָה עַל אַרְבָּעָה לְמַטָּה מֵעֲשָׂרָה כַּרְמְלִית הָוֵי וּמַפִּיק מִכַּרְמְלִית לִרְשׁוּת הָרַבִּים. רַשִׁ"י. וּמִשּׁוּם הָכִי בִּלְמַעְלָה מֵעֲשָׂרָה אָסוּר גַּם בְּצִנּוֹר לְצָרֵף אַף עַל גַּב דְּאֵין בּוֹ אַרְבָּעָה עַל אַרְבָּעָה, גְּזֵרָה מִשּׁוּם יֵשׁ בָּהּ אַרְבָּעָה עַל אַרְבָּעָה דַּהֲוֵי רְשׁוּת הַיָּחִיד דְּאוֹרַיְתָא:

{מא} דַּחֲצִיצָה פּוֹסֶלֶת בָּהּ, דְּבָעִינַן וְלָקַח הַכֹּהֵן, וְכֵיוָן דְּחָיֵץ לֹא הָוֵי לְקִיחָה בְּכֹהֵן. רַשִׁ"י: {מב} הָכִי אִיתָא בַּגְּמָרָא, דְּטַעֲמָא לָאו מִשּׁוּם שֶׁאֵין שְׁבוּת בַּמִּקְדָּשׁ וַאֲפִלּוּ כֹּהֵן דְּלָאו בַּר עֲבוֹדָה, אֶלָּא דַּוְקָא כֹּהֵן דְּבַר עֲבוֹדָה הוּא, שֶׁהִתִּירוּ לוֹ סוֹפוֹ מִשּׁוּם תְּחִלָּתוֹ: {מג} פֵּרוּשׁ שֶׁיַּחְלִיק גֻּמּוֹת שֶׁבָּרְטִיָּה, אֲבָל לִשְׁחִיקַת סַמָּנִים לֹא חָיְשִׁינַן, כֵּיוָן דַּהֲוֵי עֲלֵיהּ מֵאֶתְמוֹל: {מד} רַשִׁ"י. דִּסְבִירָא לֵיהּ דְּאִי סִלְּקָהּ מִשּׁוּם עֲבוֹדָה אַף עַל גַּב דְּמֵאֶתְמוֹל הוּא דְּסִלְּקָהּ, מִכָּל מָקוֹם הוֹאִיל מִשּׁוּם עֲבוֹדָה הוּא שֶׁסִּלְּקָהּ אִי לֹא שָׁרֵית לֵיהּ כוּ': {מה} וְתֵימַהּ דְּאִם כֵּן מַה לִּי שֶׁאֵינוֹ שֶׁל קְיָמָא מַה לִּי שֶׁל קְיָמָא, דְּהָא הָכִי אִיתָא לְרַבִּי אֱלִיעֶזֶר [שַׁבָּת פֶּרֶק יט] דִּסְבִירָא לֵיהּ הָכִי. וְנִרְאֶה דְּטַעֲמָא דְּאֵין עִקַּר שִׁירָה אֶלָּא בַּפֶּה, דְּהָכִי סְבִירָא לֵיהּ לְהָרַ"ב מִשְׁנָה ד' פֶּרֶק ב' דַּעֲרָכִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {מו} <B> יַבֶּלֶת.</B> בְּרֵישׁ פֶּרֶק ו' דִּפְסָחִים מְפָרֵשׁ הָרַ"ב דְּדַוְקָא בִּיְבֵשָׁה דְּמִפְרַךְ פְּרִיכָא. וּבִכְלִי אָסוּר, מִשּׁוּם שֶׁאֶפְשָׁר לַעֲשׂוֹת מֵעֶרֶב שַׁבָּת:

{מז} <B> כּוֹרֵךְ.</B> וְאִם תֹּאמַר הָא הַוְיָא חֲצִיצָה, מוֹקֵי בַּגְּמָרָא דְּהָכָא בִּשְׂמֹאל, דַּעֲבוֹדַת כֹּהֵן בְּיָמִין. וְאִי נַמִּי בְּיָמִין, שֶׁלֹּא בִּמְקוֹם עֲבוֹדָה, וּפֵרֵשׁ רַשִׁ"י כְּגוֹן עַל גַּב הָאֶצְבַּע: {מח} קָשֶׁה, דְּתֵפּוֹק לֵיהּ דְּאֵין זֶה צֹרֶךְ עֲבוֹדָה.</B> וְזֶה לְשׁוֹן רַשִׁ"י, אָסוּר שֶׁאֵין זֶה צֹרֶךְ עֲבוֹדָה, וְעוֹד דַּהֲוָה לֵיהּ חוֹבֵל כוּ' (עַד כָּאן). וּבְהָכִי נִיחָא נַמִּי מֵהַאי דְּפֶרֶק י"ט דְּשַׁבָּת: {מט} פֵּרוּשׁ וְחַיָּב מִשּׁוּם זוֹרֵעַ, שֶׁכְּמוֹ שֶׁהַזּוֹרֵעַ כַּוָּנָתוֹ לְהַצְמִיחַ הַפְּרִי, כֵּן כַּוָּנַת הַמַּשְׁקֶה צְמָחִים וְאִילָנוֹת: {נ} לְהָרַמְבַּ"ם אֲפִלּוּ בְּחָצֵר, כָּל שֶׁאֵין שָׁם גִּנָּה. וּלְרַבֵּנוּ תָּם דַּוְקָא בְּגַלְגַּל קָטָן כְּאוֹתָן שֶׁבַּבָּתִּים. טוּר אֹרַח חַיִּים סִימָן של"ח: {נא} וְכֻלְּהוּ בּוֹרוֹת הַנּוֹבְעִים מַיִם חַיִּים נַמִּי נִקְרָאִים כָּךְ.</B> וְלֹא זֶהוּ הֵתֵּרוֹ מִשּׁוּם שֶׁהָיָה קַר, אֶלָּא הֵתֵּרוֹ מִפְּנֵי הַצֹּרֶךְ הָיָה:

{נב} <B> שֶׁרֶץ שֶׁנִּמְצָא.</B> כְּלוֹמַר שֶׁמְּצָאוּהוּ מֵת. הָרִי"כ: {נג} תַּרְגּוּם אַבְנֵט הֶמְיָנָא.</B> הָרַמְבַּ"ם:  {נד}  <b>הָאוּלָם וְלַמִּזְבֵּחַ.</b> דְּמִשָּׁם וּלְהַלָּן קְדֻשָּׁה יְתֵרָה כְּדִתְנַן לַהּ סוֹף פֶּרֶק קַמָּא דְּכֵלִים, עַל שְׁבוּת כָּזֶה גָּזְרוּ אַף בַּמִּקְדָּשׁ:  {נה} וְהָא דְּנָטַר עַד הָכָא, מִשּׁוּם דְּבָעֵי לְאַסּוֹקֵי מִלֵּי דְּמִקְדָּשׁ אֲבָל לֹא בַּמְּדִינָה. תּוֹסָפוֹת:  {נו} כְּפֵרוּשׁוֹ דִּלְעֵיל דִּלְתַּנָּא קַמָּא עַצְמוֹ לֹא הִתִּירוּ אֶלָּא שֶׁאֵינוֹ שֶׁל קְיָמָא:  {נז} דִּתְחוּמִין אִיכָּא מַאן דְּאָמַר דְּאִית בֵּיהּ דְּרָרָא דְּאִסּוּר דְּאוֹרַיְתָא, וְכָאן שֶׁאָנוּ מַחְמִירִין דַּאֲפִלּוּ שְׁבוּת הַקַּל אֵין מַתִּירִין בַּמְּדִינָה, רַבִּי שִׁמְעוֹן מַחְמִיר יוֹתֵר מֵהֶם, שֶׁאֲפִלּוּ בַּמִּקְדָּשׁ אֵינוֹ מַתִּיר שְׁבוּת שֶׁאֶפְשָׁר לָבֹא לִידֵי חִיּוּב. רִי"כ:
