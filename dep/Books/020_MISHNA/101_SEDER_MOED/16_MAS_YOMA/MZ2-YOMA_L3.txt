{א} <b>מִבֵּיתוֹ.</b> פֵּרֵשׁ הָרַמְבַּ"ם מֵאִשְׁתּוֹ, כְּדֵי שֶׁלֹּא תִּמָּצֵא נִדָּה וְיִהְיֶה הוּא טָמֵא שִׁבְעָה יָמִים, כְּמוֹ שֶׁכָּתוּב [וַיִּקְרָא טו] וּתְהִי נִדָּתָהּ עָלָיו וְטָמֵא שִׁבְעָה יָמִים. וְהָכִי פֵּרוּשׁוֹ, שֶׁלֹּא תִּמְצָא אַחַר בִּיאָה דָּם עַל הַסָּדִין שֶׁקִּנְּחָה בּוֹ וְאֵין יָדוּעַ אִם בִּשְׁעַת בִּיאָה הָיָה הַדָּם אוֹ אַחַר כֵּן, וְיִהְיֶה בְּתוֹךְ כְּדֵי שֶׁתֵּרֵד מִן הַמִּטָּה וְתָדִיחַ, דְּבִכְהַאי גַּוְנָא אַף לְתַנָּא קַמָּא דְּמִשְׁנָה ג' פֶּרֶק ב' דְּנִדָּה מְטַמְּאָה אֶת בּוֹעֲלָהּ. גְּמָרָא: {ב} <B> אֵין לַדָּבָר סוֹף.</B> אֲבָל בְּטֻמְאָה, כֹּהֵן גָּדוֹל זָרִיז הוּא וְנִשְׁמָר מֵהַטֻּמְאָה, וְלֹא מַתְקִינִין לוֹ כֹּהֵן אַחֵר אֶלָּא מִשּׁוּם דְּכִי עָבְדִינַן לֵיהּ צָרָה כָּל שֶׁכֵּן דְּמִזָּרִיז טְפֵי. גְּמָרָא:

{ג} <B> הַנֵּרוֹת.</B> פֵּרוּשׁ הֲטָבַת שְׁתֵּי נֵרוֹת. אֲבָל הֲטָבַת חָמֵשׁ נֵרוֹת קוֹדְמִין לִקְטֹרֶת, שֶׁכֵּן בְּפֶרֶק ב' מִשְׁנָה ג' דְּפַיִס דְּמִי מְדַשֵּׁן הַמְּנוֹרָה קֹדֶם לַקְּטֹרֶת, אֶלָּא הָתָם בְּחָמֵשׁ נֵרוֹת וְהָכָא בִּשְׁתֵּי נֵרוֹת. גְּמָרָא. וְהָא דְּלֹא תְנַן הָכָא הֲטָבַת חָמֵשׁ נֵרוֹת, מִשּׁוּם דְּלֹא הָוְיָן בְּכֹהֵן גָּדוֹל. תּוֹסָפוֹת: {ד} <B> הָרֹאשׁ כוּ'.</B> נָקַט הָאֵבָרִים הָרִאשׁוֹנִים בְּהַעֲלָאָה [וְהוּא הַדִּין לְכֻלְּהוּ]:

{ה} <B> פָּרִים כוּ'.</B> אֲבָל שְׂעִירִים לֹא, דְּכֵיוָן דְּעַל חֵטְא דְּיִשְׂרָאֵל קָא אָתוּ חָלְשָׁא דַּעְתֵּיהּ. וּפַר דְּעַל חֵטְא דִּידֵיהּ וּדְאֶחָיו הַכֹּהֲנִים קָאָתֵי, אִי אִיכָּא אֱינַשׁ בַּכֹּהֲנִים דְּאִית בֵּיהּ מִלְּתָא מֵידַע יָדַע לֵיהּ וּמְהַדֵּר לֵיהּ בִּתְשׁוּבָה. גְּמָרָא:

{ו} שֶׁבְּאוֹתוֹ שָׁעָה לֹא יוּכַל שׁוּם אָדָם לָדַעַת מַה שֶּׁהוּא עוֹשֶׂה, כְּמוֹ שֶׁנֶּאֱמַר [וַיִּקְרָא טז] וְכָל אָדָם לֹא יִהְיֶה בְּאֹהֶל מוֹעֵד בְּבֹאוֹ לְכַפֵּר בַּקֹּדֶשׁ עַד צֵאתוֹ. הָרַמְבַּ"ם: {ז} <B> בֵּית דִּין.</B> הָכִי קָאָמְרוּ לֵיהּ, מַשְׁבִּיעִין אָנוּ עָלֶיךָ עַל דַּעְתֵּנוּ וְעַל דַּעַת בֵּית דִּין. גְּמָרָא. וְלֹא שֶׁהוּא שְׁלוּחָם וּשְׁלִיחַ בֵּית דִּין לְעִנְיַן עֲבוֹדָה, דְּקָיְמָא לָן דְּהָנֵי כֹּהֲנֵי שְׁלוּחֵי דְּרַחְמָנָא נִינְהוּ וְלֹא שְׁלוּחֵי בֵּית דִּין, כְּמוֹ שֶׁכָּתַבְתִּי בְּמִשְׁנָה ג' פֶּרֶק ד' דִּנְדָרִים: {ח} פֵּרֵשׁ רַשִׁ"י דִּכְתִיב [שְׁמוֹת ד] וְהֵן לֹא יַאֲמִינוּ לִי, וּכְתִיב וְהִנֵּה יָדוֹ מְצֹרַעַת כַּשָּׁלֶג. וּפֵרוּשׁ כְּשֵׁרִים מִי שֶׁמַּעֲשָׂיו סְתוּמִים וְשֶׁמָּא אֵין בְּלִבּוֹ כְּלוּם. הָרַמְבַּ"ם:

{א} <B> כָּל מִי כוּ'.</B> דְּמֵעִקָּרָא סָבְרֵי כֵּיוָן דְּאִיכָּא אֹנֶס שֵׁנָה לֹא אָתוּ, הִלְכָּךְ אַף עַל גַּב דְּתִקְּנוּ פַּיִס לִשְׁאָר עֲבוֹדוֹת לֹא חָזוּ לְתַקֵּן לְהַךְ עֲבוֹדָה. כֵּיוָן דְּחָזוּ דְּאָתוּ וְקָאָתוּ לִידֵי סַכָּנָה תְּקִינוּ לַהּ רַבָּנָן נַמִּי פַּיְסָא. גְּמָרָא: {ב} שֶׁהַמִּשְׁמָר נֶחְלַק לְשִׁבְעָה בָּתֵּי אָבוֹת: {ג} <B>  אוֹמֵר לָהֶם.</B> פֵּרוּשׁ לְכָל הַכֹּהֲנִים שֶׁיֵּשׁ שָׁם, דְּכֵיוָן דְּזֶה הַגּוֹרָל שֶׁל הַמְּרוּצָה לֹא עָלְתָה, חָזְרוּ לְהָטִיל גּוֹרָל בֵּין כֻּלָּם: {ד} לִכְאוֹרָה שֶׁכָּךְ הָיָה עוֹמֵד בְּלֹא מִצְנֶפֶת עַד שֶׁפִּיְּסוּ.</B> וְדַעַת הָרַמְבַּ"ם דְּמַחְזִירָהּ, וְעַל יְדֵי הַנְּטִילָה יַדְּעֵי מִי הוּא שֶׁמִּמֶּנּוּ מַתְחִילִין: {ה} מַשְׁמַע דִּנְטִילַת הַמִּצְנֶפֶת וַאֲמִירַת הַסָּךְ עַל יְדֵי מְמֻנֶּה אֶחָד נַעֲשֵׂית.</B> וְיֵשׁ אוֹמְרִים שֶׁנּוֹטֵל הַמִּצְנֶפֶת לֹא הָיָה יוֹדֵעַ סַךְ הַמִּנְיָן, וְהָאוֹמֵר הַמִּנְיָן לֹא הָיָה יוֹדֵעַ מִמִּי נָטַל הַמִּצְנֶפֶת:

{ז} <B> וְהַחֲבִתִּין.</B> פֵּרֵשׁ הָרַ"ב מִנְחָתוֹ שֶׁל כֹּהֵן גָּדוֹל שֶׁנֶּאֱמַר מַחֲצִיתָהּ [וְגוֹ'], וְסָמִיךְ לֵיהּ עַל מַחֲבַת בַּשֶּׁמֶן תֵּעָשֶׂה, לְכָךְ קְרוּיָה חֲבִתִּין. רַשִׁ"י: {ח} כְּלוֹמַר גְּדוֹל הָאֵבָרִים, וְאִיתָא נַמִּי דְּמִקְּרָא יָלְפִינַן דְּרֹאשׁ קוֹדֵם, דִּכְתִיב [וַיִּקְרָא א] אֶת רֹאשׁוֹ וְאֶת פִּדְרוֹ וְעָרַךְ: {ט} וְהַקְּרָבַיִם לֹא שְׁנָאוּם, דִּלְפִי שֶׁאֵין לָהֶם תְּנוּעָה מֵעַצְמָם מוֹדֶה בֶּן עֲזַאי דִּקְרֵבִים בָּאַחֲרוֹנָה:

{א} <B> לִקְרוֹת.</B> פֵּרֵשׁ רַשִׁ"י דְּיָלְפִינַן מִמִּלּוּאִים. וְאָמְרִינַן בְּפֶרֶק קַמָּא דַּף ה' מִנַּיִן שֶׁאַף מִקְרָא פָּרָשָׁה שֶׁנִּצְטַוָּה בִּוְאַתָּה תְּצַוֶּה מְעַכֵּב, תַּלְמוּד לוֹמַר זֶה הַדָּבָר אֲשֶׁר צִוָּה ה' לַעֲשׂוֹת, בַּעֲשִׂיַּת הַמִּלּוּאִים בְּצַו אֶת אַהֲרֹן, אֲפִלּוּ דִּבּוּר מְעַכֵּב: {ב} <B> הַכְּנֶסֶת.</B> בֵּית הַכְּנֶסֶת הָיָה סָמוּךְ לָעֲזָרָה, בְּהַר הַבַּיִת. רַשִׁ"י: {ג} וְנִקְרָא חַזָּן שֶׁהוּא לְשׁוֹן חוֹזֶה שֶׁצָּרִיךְ לִרְאוֹת מַה הֵם צָרְכֵי צִבּוּר: {ד} דְּלֹא תֵּימָא שֶׁזֶּה שֶׁהַסְּגָן אֵינוֹ מְקַבְּלוֹ מֵהַחַזָּן לְפִי שֶׁהַחַזָּן חוֹלֵק כָּבוֹד לַסְּגָן שֶׁיְּקַבְּלוֹ מֵרֹאשׁ הַכְּנֶסֶת, וְקַיְמָא לָן דְּאֵין חוֹלְקִים כָּבוֹד לַתַּלְמִיד בִּמְקוֹם רַב אֶלָּא הֵיכָא דְּרַבֵּיהּ פָּלֵיג לֵיהּ יְקָרָא, וַאֲנַן סְתָמָא תְּנַן, דִּלְעוֹלָם הַמִּנְהָג כֵּן. תּוֹסָפוֹת: {ה} גְּמָרָא. וְנִרְאֶה טַעֲמָא, אִי מִשּׁוּם יְקָרָא דְּכֹהֵן גָּדוֹל, שֶׁלֹּא יִהְיֶה עוֹמֵד כָּל כָּךְ זְמַן רַב בְּלֹא עֲבוֹדָה מִתְּחִלַּת שִׁלּוּחַ הַשָּׂעִיר עַד הַגִּיעוֹ לִמְקוֹמוֹ, וְאִי נַמִּי מִשּׁוּם חוּלְשָׁא דְּכֹהֵן גָּדוֹל. תּוֹסְפוֹת יוֹם טוֹב: {ו} <B> שֶׁבְּחֻמַּשׁ הַפְּקוּדִים.</B> סֵפֶר וַיְדַבֵּר, שֶׁמַּתְחִיל בְּמִנְיָנָן שֶׁל יִשְׂרָאֵל. רַשִׁ"י: {ז} וְהָרַמְבַּ"ם כָּתַב דְּהָנֵי שְׁמֹנֶה בְּרָכוֹת כֻּלָּן לְאַחֲרֵיהֶם הֵן, שֶׁכָּתַב וּמוֹסִיף לְאַחֲרֶיהָ שֶׁבַע בְּרָכוֹת: {ח} <B>וְעַל הָעֲבוֹדָה כוּ'.</B> מְבָרֵךְ בִּשְׁבִיל הָעֲבוֹדָה שֶׁעָבַד. וְהַהוֹדָאָה סְמוּכָה לָעֲבוֹדָה, דִּכְתִיב (תְּהִלִּים נ) זוֹבֵחַ תּוֹדָה יְכַבְּדָנְנִי, אַחַר הָעֲבוֹדָה תּוֹדָה, וְהִיא הוֹדָאָה. רַשִׁ"י: {ט} כְּמָה דְאַת אָמַר [מְלָכִים א ח] אֶל הָרִנָּה וְאֶל הַתְּפִלָּה, וְתִרְגֵּם יוֹנָתָן צְלוֹתָא וּבָעוּתָא. וּבַגְּמָרָא דִּבְרָכוֹת (לא.) אֶל הָרִנָּה וְאֶל הַתְּפִלָּה, רִנָּה זוֹ תְּפִלָּה, פֵּרוּשׁ שֶׁבַח. וּבְיִרְמְיָה ז' מְתַרְגֵּם רִנָּה בָּעוּתָא. וּבְמִדְרָשׁ רַבָּה [רִנָּה] זוֹ קִלּוּסוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא: 

~ פרק ז - משנה ב

{י} <B> כְּאֶחָת.</B> וְאַף עַל גַּב דִּכְתִיב יוֹצִיא אֶל מִחוּץ לַמַּחֲנֶה וּשְׂרָפוֹ בָּתַר אֵילוֹ וְאֵיל הָעָם וְאֵמוּרֵי חַטָּאת, יָלֵיף בַּגְּמָרָא דִּכְתִיב נַמִּי וְהַמְשַׁלֵּחַ בָּתַר הָכִי, וּמָה הַמְשַׁלֵּחַ דְּמֵעִקָּרָא דְּאֵין עֲמִידָה שֶׁלּוֹ אֶלָּא עַד מַתַּן דָּמִים דַּחֲבֵרוֹ, כְּמוֹ שֶׁכָּתַב [הָרַ"ב] בְּרֵישׁ פֶּרֶק ו', הָכִי נַמִּי וְהַשּׂוֹרֵף יְכַבֵּס דְּמֵעִקָּרָא וּקְרָאֵי דְּלֹא כְּסִדְרָן כְּדִלְקַמָּן. תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק ז - משנה ג

{יא} <B> קִדֵּשׁ כוּ'.</B> דְּאִלּוּ קָרָא בְּאִצְטְלִית כְּבָר קִדֵּשׁ קֹדֶם שֶׁפָּשַׁט בִּגְדֵי בּוּץ, וְאֵינוֹ צָרִיךְ עַתָּה אֶלָּא טְבִילָה וְקִדּוּשׁ. הָרֹא"שׁ: {יב} דְּאָמְרִינַן בַּגְּמָרָא דְּמַתְנִיתִין לָאו מִתַּקַּנְתָּא הוּא, דְּהָא לֹא קָאָמַר פַּר וְשָׂעִיר וְאֵמוּרֵי חַטָּאת אֵימָתַי עֲבֵיד לְהוּ, אֶלָּא כִּדְקָתָנֵי בַּבְּרַיְתָא כְּמוֹ שֶׁכָּתַב הָרַ"ב. וּפֵרְשׁוּ הַתּוֹסָפוֹת דְּמִכָּל מָקוֹם לָאו מְשַׁבַּשְׁתָּא הוּא, וְשִׁיֵּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} <B> עִם תָּמִיד כוּ'.</B> קָאֵי אַשִּׁבְעָה כְּבָשִׂים, וּפַר הָעוֹלָה נַמִּי עִם שֶׁל שַׁחַר, וְכִדְּכָתַב הָרַ"ב מִטַּעַם הַגְּמָרָא: {יד} לְפִי שֶׁהוּא מִן הַמּוּסָפִין הָיָה רָאוּי לְהַקְדִּימוֹ, אֶלָּא דִּקְרָא קָאָמַר מִלְּבַד כוּ'. וּמִיהוּ בַּתְרֵיהּ לֹא מְאַחֲרִינַן לֵיהּ: {טו} וְהָא דְּלֹא קָתָנֵי לְהוּ מִשּׁוּם דְּרַבִּי אֱלִיעֶזֶר מוֹדֶה לוֹ דְּבִטְבִילָה שְׁלִישִׁית הָווּ: {טז} וְנִרְאֶה לִי דְּטַעֲמוֹ מִשּׁוּם דִּוְאֶת חֵלֶב הַחַטָּאת כְּתִיב בָּתַר וְיָצָא וְעָשָׂה אֶת עוֹלָתוֹ: {יז} וּרְצוֹנוֹ לוֹמַר בִּטְבִילָה רְבִיעִית וְטַעֲמָא מְפֹרָשׁ בַּגְּמָרָא דִּכְתִיב וּבָא אַהֲרֹן כוּ' וּפָשַׁט אֶת בִּגְדֵי הַבַּד אֲשֶׁר לָבַשׁ. כְּלוּם אָדָם פּוֹשֵׁט אֶלָּא מַה שֶּׁלָּבַשׁ, וּמַה תַּלְמוּד לוֹמַר אֲשֶׁר לָבַשׁ, שֶׁלָּבַשׁ כְּבָר. אַלְמָא זוֹ הִיא פְּשִׁיטָה שְׁנִיָּה. שְׁמַע מִינַהּ דְּיֵשׁ הַפְסָקַת חִלּוּף בְּגָדִים בֵּין עֲבוֹדַת הַיּוֹם לְהַךְ בִּיאָה כוּ', וּבְהוֹצָאַת כַּף וּמַחְתָּה אַיְרֵי. וְכֵיוָן דִּשְׁמַע מִינַהּ דְּאִיכָּא הֶפְסֵק, שְׁמַע מִינַהּ דְּהַךְ קְרָא שֶׁלֹּא כְּסִדְרוֹ, וְדִכְתִיב בַּתְרֵיהּ וְיָצָא וְעָשָׂה אֶת עוֹלָתוֹ כוּ' מֻקְדָּם לְשֶׁל זֶה: {יח} בִּטְבִילָה חֲמִשִּׁית.</B> רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב בַּאֲרִיכוּת: 

~ פרק ז - משנה ד

{יט} דְּהָא בַּשַּׁחַר מֵטִיב אֶת הַנֵּרוֹת, דְּהַיְנוּ דִּשּׁוּנָן וְתִקּוּן הַפְּתִילוֹת הַחֲדָשׁוֹת, אֶלָּא הֲטָבָה זוֹ הַדְלָקָה הִיא. אֲבָל הָרַמְבַּ"ם סְבִירָא לֵיהּ דְּבַשַׁחַר נַמִּי מַדְלִיקָן אַחַר שֶׁמֵּטִיבָן, וְאִם כֵּן בָּעֶרֶב הָיָה צָרִיךְ לַחֲזֹר וּלְתַקְּנָן וּלְהַדְלִיקָן, וּלְדִידֵיהּ שַׁפִּיר תְּנַן לְהֵטִיב: 

~ פרק ז - משנה ה

{כ} <B> וּמִצְנֶפֶת.</B> בַּתּוֹרָה נִקְרָא דְּשֶׁל הֶדְיוֹטוֹת מִגְבַּעַת, וּלְשׁוֹן תּוֹרָה לְחוּד וּלְשׁוֹן חֲכָמִים לְחוּד. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כא} וְאַף עַל גַּב דְּלִישְׁנָא דִּבְרַיְתָא פָּנָיו כְּלַפֵּי הַכֹּהֵן, הָכִי פֵּרוּשׁוֹ, פָּנָיו פּוֹנוֹת לַמָּקוֹם שֶׁפּוֹנִים פְּנֵי הַנִּשְׁאָל: {כב} כְּמִי שֶׁמִּתְפַּלֵּל בֵּינוֹ לְבֵין עַצְמוֹ. הָרַמְבַּ"ם. וְהַיְנוּ דְּמִתְּפִלַּת חַנָּה יָלְפִינַן:

{א} דִּכְתִיב וְהַאֲבַדְתִּי אֶת הַנֶּפֶשׁ הַהִיא, עִנּוּי שֶׁהוּא אֲבֵדַת נֶפֶשׁ, וְאֵיזֶה זֶה אֲכִילָה וּשְׁתִיָּה. גְּמָרָא דַּף ע"ד: {ב} דְּכִי כָּל הַנֶּפֶשׁ אֲשֶׁר לֹא תְּעֻנֶּה לֹא קָחָשֵׁיב, דְּלֹא חָשֵׁיב אֶלָּא צִוּוּיִין, אֲבָל הַאי דְּלֹא צִוּוּי לֹא קָחָשֵׁיב. תּוֹסָפוֹת: {ג} <B> בַּאֲכִילָה וּשְׁתִיָּה.</B> אֲכִילָה וּשְׁתִיָּה דְּאִקְּרוּ עִנּוּי מְנָלָן, כְּתִיב הָכָא תְּעַנּוּ, וּכְתִיב [דְּבָרִים ח] וַיְעַנְּךָ וַיַּרְעִיבֶךָ, שְׁמַע מִינַהּ דִּרְעָבוֹן אִקְּרֵי עִנּוּי. רְחִיצָה, דִּכְתִיב [מְלָכִים א ב] כִּי נָשָׂאתָ כוּ' וְכִי הִתְעַנֵּיתָ בְּכָל אֲשֶׁר הִתְעַנָּה אָבִי, וּכְתִיב הָתָם (שְׁמוּאֵל ב יז) כִּי אָמְרוּ הָעָם רָעֵב וְעָיֵף וְצָמֵא, וּכְתִיב (מִשְׁלֵי כה) מַיִם קָרִים עַל נֶפֶשׁ עֲיֵפָה, וְלֹא כְתִיב בְּנֶפֶשׁ, שְׁמַע מִינַהּ עָיֵף מֵרְחִיצָה הוּא. סִיכָה, דִּכְתִיב (דָּנִיֵּאל י) לֶחֶם וְגוֹ' וְסוֹך לֹא סָכְתִּי, וּכְתִיב (שָׁם) כִּי מִן הַיּוֹם וְגוֹ' וּלְהִתְעַנּוֹת. תַּשְׁמִישׁ הַמִּטָּה, דִּכְתִיב (בְּרֵאשִׁית לא) אִם תְּעַנֶּה אֶת בְּנוֹתַי וְאִם תִּקַּח נָשִׁים, תְּעַנֶּה מִתַּשְׁמִישׁ, תִּקַּח צָרוֹת, גְּמָרָא: {ד} <B> וְהַמֶּלֶךְ כוּ'.</B> דְּאִסּוּרָא דִּידְהוּ לָאו דְּאוֹרָיְתָא, וּקְרָאֵי אַסְמַכְתָּא בְּעָלְמָא, כְּמוֹ שֶׁכָּתְבוּ הָרַמְבַּ"ם וְהָרֹא"שׁ, וְהִלְכָּךְ הֵקֵלּוּ בּוֹ, דְּהֵם אָמְרוּ וְהֵם אָמְרוּ. וְאִי נַמִּי דְּאוֹרַיְתָא, כֵּיוָן דְּלֹא פֵּרְשׁוּ בְּהֶדְיָא בִּקְרָא קִלֵּי טְפֵי, דִּמְסָרָן לַחֲכָמִים וְהֵם יְכוֹלִין לְהָקֵל וּלְהַתִּיר כְּפִי שֶׁנִּרְאֶה לָהֶם. הָרַ"ן:

{ה} <B> כָּמוֹהָ וּכְגַרְעִינָתָהּ.</B> פֵּרוּשׁ הַכּוֹתֶבֶת אֵינָהּ דְּבוּקָה לְגַרְעִינָתָהּ וְיֵשׁ בֵּינֵיהֶם חָלָל, וְקָא מַשְׁמַע לָן דְּצָרִיךְ לִמְעֹךְ אוֹתוֹ חָלָל לְשַׁעֵר בְּכָמוֹהָ וּכְגַרְעִינָתָהּ בִּלְבַד. תּוֹסָפוֹת וְהָרַ"ן: {ו} יֵשׁ לְכַוֵּן שֶׁאֵינָהּ פְּחוּתָה מִמֶּנָּה אֶלָּא מְעָט, דְּכֵיוָן דְּחַד אָמוֹרָא אָמַר בַּגְּמָרָא יְתֵירָה מִכְּבֵיצָה לֹא מַפְלְגִינַן מִנֵּיהּ מַאן דְּפָלֵיג עֲלֵיהּ. הָרַ"ן: {ז} הַשִּׁעוּר בְּאָדָם בֵּינוֹנִי הוּא פָּחוֹת מֵרְבִיעִית, אֲבָל בִּשְׁאָר אָדָם הַכֹּל לְפִי גָּדְלוֹ וּלְפִי קָטְנוֹ. וּמִיהוּ בַּאֲכִילָה לֹא מְפַלְגִינַן אֶלָּא אֲפִלּוּ לְעוֹג מֶלֶךְ הַבָּשָׁן מְחַיְּבִים בִּכְכוֹתֶבֶת, וְטַעֲמָא אָמְרִינַן בַּגְּמָרָא דְּבַאֲכִילָה קִים לְהוּ לְרַבָּנָן דִּבְהָכִי מִיַּתְבָא דַּעְתֵּיהּ. וּמִיהוּ כֻּלֵּי עָלְמָא טוּבָא וְעוֹג מֶלֶךְ הַבָּשָׁן פּוּרְתָא. וְהַשְּׁתִיָּה בְּדִידֵיהּ מִיַּתְבָא דַּעְתֵּיהּ, בִּדְחַבְרֵיהּ לֹא מִיַּתְבָא דַּעְתֵּיהּ. וּמִשּׁוּם הָכִי לְעִנְיַן דְּלָאו בִּיתוּבָא דַּעְתָּא הוּא, בְּשֶׁל עוֹלָם הֵן שָׁמִין בְּבֵינוֹנִית. הָרַ"נ: {ח} לֹא קָאִימְנָא שַׁפִּיר בְּהַאי פָּחוֹת כַּמָּה הֲוָה בָּצִיר מִנֵּיהּ.</B> הָרַ"ן: {ט} <B> אֵין מִצְטָרְפִין.</B> דְּאַף עַל גַּב דְּחַד שְׁמָא נִינְהוּ כְּדִלְקַמָּן, הוֹאִיל וּמִשּׁוּם יִתּוּבָא דַּעְתָּא הוּא, הַאי לֹא קָמִיַּתְבָא דַּעְתֵּיהּ, כִּדְאָמַר בַּגְּמָרָא:

{י} שֶׁנֶּאֱמַר וְנָתַתָּ הַכֶּסֶף בְּכֹל אֲשֶׁר תְּאַוֶּה נַפְשְׁךָ כוּ' בַּיַּיִן וּבַשֵּׁכָר.</B> שֵׁכָר שְׁתִיָּה הוּא, דְּיָלִיף שֵׁכָר שֵׁכָר מִנָּזִיר, וְקַרְיֵהּ רַחְמָנָא וְאָכַלְתָּ, דְּאִלּוּ מִיַּיִן אִיכָּא לְמֵימַר עַל יְדֵי אֲנִיגְרוֹן שֶׁהוּא מִין מַאֲכָל שֶׁנּוֹתְנִין בּוֹ יַיִן: {יא} <B> שֶׁאֵינָן רְאוּיִין.</B> כְּגוֹן עֲשָׂבִים הַמָּרִים אוֹ שְׂרָפִים הַבְּאוּשִׁין. הָרַמְבַּ"ם: {יב} <B> צִיר.</B> כְּגוֹן צִיר כוּ': 

~ פרק ח - משנה ה

{יג} כָּךְ כָּתַב רַשִׁ"י.</B> וְנִרְאֶה לִי שֶׁדִּקְדְּקוּ לְפָרֵשׁ כֵּן מֵהָנֵי עֻבְדֵּי דְּאִיתְּנְהוּ בַּגְּמָרָא. הַהִיא עֻבָּרָה דְּאַרְחָא, לְחִישׁוּ לְהוּ וְאִלְּחִישָׁה, קָרֵי עֲלֵיהּ בְּטֶרֶם אֱצָרְךָ בַּבֶּטֶן וְגוֹ' וְנָפֵיק מִינַהּ רַבִּי יוֹחָנָן. וְהַהִיא עֻבָּרָהּ דְּאַרְחָה וּלְחִישׁוּ לַהּ וְלֹא אִלְּחִישָׁא, וְקָרֵי עֲלֵיהּ זֹרוּ רְשָׁעִים מֵרָחֶם, וְנָפֵיק מִינַהּ שַׁבְּתַאי אוֹצֵר פְּרִי. הָא קַמָּן דְּמִצַּד הָעֻבָּר בָּא הַתַּאֲוָה, שֶׁהוּא מֵרִיחַ וּמִתְאַוֶּה, שֶׁהֲרֵי בְּיָדוֹ לַחֲזֹר בּוֹ. תּוֹסְפוֹת יוֹם טוֹב: {יד} וְאָמְרִינַן תּוּנְבָּא, פֵּרוּשׁ שְׁטוּת נָקֵיט לֵיהּ מֵחֲמַת חָלְיוֹ, אֲבָל אָמַר צָרִיךְ כוּ', דְּלֵב יוֹדֵעַ מָרַת נַפְשׁוֹ, וּמִשּׁוּם דִּסְפֵק נְפָשׁוֹת לְהָקֵל הִלְכָּךְ כִּי אָמַר צָרִיךְ תָּלִינַן בְּלֵב יוֹדֵעַ, וְלָאו מִשּׁוּם תּוּנְבָּא, אֶלָּא קִים לֵיהּ בְּגַוֵּיהּ טְפֵי, וְכִי אָמַר אֵינוֹ צָרִיךְ תָּלִינַן בְּתוּנְבָּא. תּוֹסְפוֹת יוֹם טוֹב:

{טו} <B> דְּבָרִים טְמֵאִים.</B> הֵם הַדְּבָרִים הָאֲסוּרִים כִּשְׁקָצִים וּרְמָשִׂים וּבְהֵמָה טְמֵאָה וְזוּלָתָם. הָרַמְבַּ"ם: {טז} זֶה אֵינוֹ מוֹעִיל אֶלָּא בְּדֶרֶךְ סְגֻלָּה, וְאֵין עוֹבְרִין עַל הַמִּצְוֹת אֶלָּא בִּרְפוּאָה בִּלְבַד, רְצוֹנוֹ לוֹמַר בִּדְבָרִים הַמְרַפְּאִים בְּטֶבַע, וְהוּא דָּבָר אֲמִתִּי הוֹצִיאוּ מִדַּעַת וְהַנִּסָּיוֹן הַקָּרוֹב לָאֱמֶת. אֲבָל לְהִתְרַפְּאוֹת בִּדְבָרִים שֶׁהֵם מְרַפְּאִים בִּסְגֻלָּתָן אָסוּר, כִּי כֹּחָם חַלָּשׁ, אֵינוֹ מִצַּד הַדַּעַת, וְנִסְיוֹנוֹ רָחוֹק, וְהִיא טַעֲנָה חֲלוּשָׁה מִן הַטּוֹעֵן: {יז} <B> סְפֵק נְפָשׁוֹת כוּ'.</B> דִּכְתִיב (וַיִּקְרָא יח) אֲשֶׁר יַעֲשֶׂה אוֹתָם הָאָדָם וָחַי בָּהֶם, וְלֹא שֶׁיָּמוּת בָּהֶם. וּפֵרֵשׁ רַשִׁ"י אֲשֶׁר יַעֲשֶׂה הָאָדָם הַמִּצְוֹת שֶׁיִּחְיֶה בָּהֶם וַדַּאי וְלֹא שָׁיְכָא בַּעֲשִׂיָּתָם לָבֹא לִידֵי סְפֵק מִיתָה, אַלְמָא מְחַלְּלִין עַל הַסָּפֵק. גְּמָרָא דַּף פ"ה:

לְהִלָּחֵם נֶגֶד הַצַּר הַצֹּרֵר, וּבְכֹחִי לִמְחֹל גַּם עַל אֲשֶׁר קִלְקֵל נֶגְדְּכֶם, דַּהֲרֵי כֻּלְכֶם חַיָּבִים בִּכְבֹדִי הַנֹּגֵעַ לִכְבֹדְכֶם, וּבִדְבָרֵינוּ מְבֹאָר בַּאֲרִיכוּת בְּעֶזְרַת הַשֵּׁם יִתְבָּרַךְ. מֵעַתָּה עוֹלֶה שַׁפִּיר נֹפֶת דִּבְרֵי הַמִּשְׁנָה, דְּבַתְּחִלָּה קָאָמַר אֶת זוֹ דָּרַשׁ רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה, שֶׁדּוֹרֵשׁ הַכְרָעַת הַמִּקְרָא מִכָּל חַטֹּאתֵיכֶם לִפְנֵי ה', דְּנִדְרָשׁ לְאַחֲרָיו, וּמְיַשֵּׁב רַבִּי עֲקִיבָא לְהַכְרִיעַ דְּכֵן הַהֶכְרַח שֶׁלֹּא לִדְרֹשׁ לְפָנָיו כַּמְבֹאָר. וְזֶהוּ אַשְׁרֵיכֶם יִשְׂרָאֵל, דַּאֲשֶׁר תְּהִלָּה שֶׁהוּא לָכֶם לִפְנֵי מִי אַתֶּם מִטַּהֲרִים, דְּמַעֲלֶה עֲלֵיכֶם כִּבְיָכוֹל כְּאִלּוּ קִדַּשְׁתֶּם אוֹתִי הוּא רַק מִי מְטַהֵר אֶתְכֶם, דְּטָהֳרָה הוּא רַק לְעַצְמְכֶם, וְאִם קִדַּשְׁתֶּם עַצְמְכֶם מַעֲלֶה אֲנִי עֲלֵיכֶם כְּאִלּוּ קִדַּשְׁתֶּם אוֹתִי כוּ', וְהוּא רַק בְּמַדְרֵגַת אַהֲבַת הַבֵּן, וְכִי קָדוֹשׁ ה' בֵּין אַתֶּם כוּ'. וְהַשֹּׁרֶשׁ פּוֹרֶה כוּ' שֶׁאִם אֵין אַתֶּם מַקְדִּישִׁים אֵינִי מְקֻדָּשׁ, הוּא רֹאשׁ וְלַעֲנָה, וְאִם חָטָאתָ מַה תִּפְעַל לְיוֹצֵר הָאָדָם. וְעוֹלֶה שַׁפִּיר אֲבִיכֶם שֶׁבַּשָּׁמַיִם, דְּהוּא בְּמַדְרֵגַת אָב כִּבְיָכוֹל וְהַשָּׁמַיִם שָׁמַיִם לַה' כְּמוֹ שֶׁנֶּאֱמַר וְזָרַקְתִּי כוּ' וּטְהַרְתֶּם, שֶׁאַתֶּם תִּטָּהֲרוּ. וּרְאָיָה מְפֹרֶשֶׁת עוֹד מִפָּסוּק מִקְוֵה יִשְׂרָאֵל, דְּקָאֵי לְרֵישָׁא דִּקְרָא בְּיִרְמְיָה, אִם עֲוֹנֵינוּ עָנוּ בָּנוּ ה' עֲשֵׂה לְמַעַן שְׁמֶךָ כִּי רַבּוּ מְשׁוּבוֹתֵינוּ לְךָ חָטָאנוּ. דְּבָזֶה שֶׁהָלַכְנוּ שׁוֹבָב בְּדַרְכֵי לְבָבֵנוּ לְךָ חָטָאנוּ, אֲנַחְנוּ הַחוֹטְאִים וְהַחֲסֵרִים, וּלְנַפְשֵׁנוּ הֲרֵעֹנוּ וְלֹא לְךָ יוֹצֵר הָאָדָם. וְעַל זֶה מְיַעֵד אוֹתָנוּ מִקְוֵה יִשְׂרָאֵל כוּ' לְטַהֵר אוֹתָנוּ. וְכֵיוָן שֶׁכֵּן מֻכְרָח הַכְרָעַת הַפָּסוּק מִכָּל חַטֹּאתֵיכֶם לִפְנֵי ה', דַּעֲבֵרוֹת שֶׁבֵּין אָדָם לַחֲבֵרוֹ אֵין יוֹם כִּפּוּר מְכַפֵּר עַד שֶׁיְּרַצֶּה אֶת חֲבֵרוֹ. לְשׁוֹן תּוֹסְפוֹת יוֹם טוֹב. לְפִי שֶׁהַדֶּרֶךְ הַיְּשָׁרָה הִיא שֶׁיָּבוֹר הָאָדָם בְּעַצְמוֹ לְהִתְקַדֵּשׁ וְלִטָּהֵר, וְאָז כְּשֶׁבָּא לִטָּהֵר מְסַיְּעִין לוֹ, וְעַל זֶה אָמַר לִפְנֵי מִי אַתֶּם מִטַּהֲרִים. וּמִפְּנֵי שֶׁלִּפְעָמִים לְפִי שֶׁיֵּצֶר לֵב הָאָדָם רַע עָלָיו לֹא יוּכַל לְהִתְגַּבֵּר עָלָיו לְכוּפוֹ לִטָּהֵר בִּתְשׁוּבָה, וְהוּא רַחוּם יְכַפֵּר עָוֹן יְעוֹרְרֵהוּ וּבִימִין צִדְקוֹ יִתְמְכֵהוּ לְבִלְתִּי יִדַּח מִמֶּנּוּ נִדָּח. וְזֶהוּ מִי מְטַהֵר אֶתְכֶם. וְנֶאֱמַר וְזָרַקְתִּי, וְזֶהוּ בְּלִי הִתְעוֹרְרוּת הָאָדָם, אֶלָּא בְּרַחֲמָיו מְטַהֲרָם. וְהַשֵּׁם יִתְבָּרַךְ בְּרַחֲמָיו יְטַהֲרֵנוּ מִכָּל טֻמְאוֹתֵינוּ, וּלְבֵית קָדְשׁוֹ יְקַבֵּץ נְפוּצוֹתֵינוּ, בִּמְהֵרָה בְּיָמֵינוּ אָמֵן וְאָמֵן:
