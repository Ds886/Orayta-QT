~ פרק א - משנה א

{א} וְהֵ"א מִתְחַלֵּף בְּחֵי"ת. רַשִׁ"י. שֶׁשְּׁנֵיהֶם מִמּוֹצָא הַגָּרוֹן. וְעָיֵף הַיְנוּ לְמַיִם, כְּדִכְתִיב [מִשְׁלֵי כה] מַיִם קָרִים עַל נֶפֶשׁ עֲיֵפָה: {ב} <B> בַּמּוֹעֵד.</B> הַמִּשְׁנָה תִּקְרָא [מוֹעֵד, מַה שֶּׁאָנוּ קוֹרִים] חֻלּוֹ שֶׁל מוֹעֵד, [וְהֵם] הַיָּמִים שֶׁבֵּין מוֹעֵד לְמוֹעֵד. הָרַמְבַּ"ם: {ג} רַשִׁ"י.</B> וְהָרִי"ף וְהָרַמְבַּ"ם וְהָרֹא"שׁ וְהַטּוּר פֶּה אֶחָד הִסְכִּימוּ דְּאֵין לְחֻלּוֹ שֶׁל מוֹעֵד עִקָּר מִן הַתּוֹרָה. וּבְרַיְתָא דַּחֲגִיגָה דְּלַמְּדוּהָ מִקְּרָא אֵינוֹ אֶלָּא אַסְמַכְתָּא. וְהַבֵּית יוֹסֵף כָּתַב שֶׁנִּרְאֶה לוֹ מִזֹּאת הַבְּרַיְתָא שֶׁיֵּשׁ לוֹ עִקָּר מִן הַתּוֹרָה, וְהַכָּתוּב מְסָרָהּ לַחֲכָמִים לֶאֱסֹר וּלְהַתִּיר כְּפִי שֶׁנִּרְאֶה לָהֶם: {ד} אֲבָל הָרַמְבַּ"ם נִרְאֶה שֶׁאֵין סוֹבֵר כָּךְ, אֶלָּא דְּבֵית הַבַּעַל נַמִּי אָסוּר בִּשְׁבִיעִית. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} כְּלוֹמַר שֶׁאֲפִלּוּ דָּבָר הָאָבֵד לֹא הִתִּירוּ כְּשֶׁיֵּשׁ טִרְחָא: {ו} <B>  שֶׁלֹּא.</B> וְאִצְטְרִיךְ לְמִתְנֵי לְמִעוּטֵי שְׂדֵה הַבַּעַל, שֶׁאֲפִלּוּ מִמַּעְיָן שֶׁלֹּא יָצָא בִּתְחִלָּה אָסוּר. גְּמָרָא: {ז} <B> אֵין מַשְׁקִין.</B> אַמּוֹעֵד קָאֵי, דְּגַבֵּי שְׁבִיעִית לֹא חָיְישִׁינַן לְטִרְחָא יְתֵרָא, וְטַעְמָא מְפָרֵשׁ בַּיְרוּשַׁלְמִי דִּשְׁבִיעִית עַל יְדֵי שֶׁהוּא מֻתָּר בִּמְלָאכָה שֶׁאֵינָהּ דְּקַרְקַע הִתִּירוּ אֲפִלּוּ בְּדָבָר שֶׁל טֹרַח, אִי נַמִּי שְׁבִיעִית עַל יְדֵי שֶׁזְּמַנָּהּ מְרֻבֶּה כוּ' וְלֹא מִסְתַּבֵּר לַעֲשׂוֹת שִׁבְעָה יָמִים הָאַחֲרוֹנִים כְּשִׁבְעָה דְּרֶגֶל: {ח} וּמִלְּשׁוֹן קוּלְתָּא שֶׁהוּא תִּרְגּוּם כַּדָּהּ עַל שִׁכְמָה, נִקְרֵאת קִילוֹן.</B> הָרַ"ן: {ט} <B> הַקִּילוֹן.</B> כֵּיוָן דְּמֵי הַגְּשָׁמִים לֹא נֶאֶסְרוּ אֶלָּא מִשּׁוּם גְּזֵרָה דְּמֵי הַקִּילוֹן, הִלְכָּךְ שַׁפִּיר נַקְטֵיהּ אַחַר מֵי גְּשָׁמִים. הָרַ"ן: {י} <B> וְאֵין עוֹשִׂין כוּ'.</B> מִדְּרֵישָׁא דְּהַךְ אֲבָל אֵין מַשְׁקִין לֹא קָאֵי אַשְּׁבִיעִית, הַךְ סֵיפָא נַמִּי לֹא קָאֵי אַשְּׁבִיעִית. הָרַמְבַּ"ם: {יא} כְּדֵי שֶׁיִּתְעַכְּבוּ שָׁם הַמַּיִם, מִלְּשׁוֹן [רוּת א] הֲלָהֶן תֵּעָגֵנָה, שֶׁהוּא לְשׁוֹן עִכּוּב. הָרַ"ן: {יב} לִישְׁנָא דִּבְרַיְתָא, וּבְאַתְרֵיהּ דְּתַנָּא דְּמַתְנִיתִין שְׁכִיחֵי גְּפָנִים וְלֹא זֵיתִים: 

~ פרק א - משנה ב

{יג} <B>וַחֲכָמִים אוֹמְרִים כוּ'.</B> דְּכָל שֶׁנִּרְאֶה כְּחוֹפֵר לֹא הֲוֵי עֲבוֹדָה, דְּעוֹדֵר לֹא הֲוֵי כֵּיוָן דְּמַשְׁלִיךְ אֶת הֶעָפָר לְמֵרָחוֹק, וְהָעוֹדֵר מֵנִיחַ בִּמְקוֹמוֹ. הָרַ"ן. וּבָרַמְבַּ"ם מַשְׁמָע דְּמַתִּיר אֲפִלּוּ אֵינוֹ מַשְׁלִיךְ לְמֵרָחוֹק, וְעַיֵּן שָׁם הַטַּעַם: {יד} <B> הַמְקֻלְקָלוֹת.</B> סִיּוּמָא דְּמִלְּתָא דְּרַבָּנָן הוּא: {טו} וְלֹא דָּמִי לְמַעְיָן שֶׁיָּצָא בַּתְּחִלָּה כְּדִלְעֵיל דְּחַיְּשִׁינַן שֶׁיִּפְּלוּ גְּדוֹתָיו וְאָתֵי לְתַקֵּן, דְּתִקּוּנָא דְּהָתָם עֲשִׂיָּה בַּתְּחִלָּה הִיא כִּדְקָתָנֵי שֶׁיָּצָא בַּתְּחִלָּה: {טז} <B>שֶׁבִּרְשׁוּת הָרַבִּים.</B> אֲפִלּוּ אֵין רַבִּים צְרִיכִים לָהֶם. וְטַעְמָא כָּתַב הָרַ"ן בְּשֵׁם הָרַאבַ"ד לְפִי שֶׁצָּרְכֵי רַבִּים אֵינָם נִגְמָרִים אֶלָּא בִּשְׁעַת הַבַּטָּלָה, מִשּׁוּם דְּדָמִי לִקְדֵרָה דְּבֵי שֻׁתְּפֵי כוּ', וּמַסִּיק דְּכִי הִתִּירוּ בִּמְלֶאכֶת הֶדְיוֹט [וּכְדִלְקָמָּן]. וְנִרְאֶה לִי דְּהַיְנוּ דְּתָנֵי וְחוֹטְטִין [דְּהַיְנוּ שֶׁהוּא מְלֶאכֶת הֶדְיוֹט] וּפֵרוּשֵׁי קָמְפָרֵשׁ דְּאִי לֹא תֵּימָא הָכִי לִישָׁנָא יְתֵרָא הוּא דְּהַיְנוּ מְתַקְּנִין: {יז} <B> וּמְתַקְּנִין כוּ'.</B> בְּרֵישׁ שְׁקָלִים תְּנַן דִּבְט"ו בַּאֲדָר מְתַקְּנִין. וּלְפִי שֶׁהַקִּלְקוּלִים מְצוּיִין בַּדְּרָכִים לְכָךְ הֻצְרַךְ תּוּ לְתַקֵּן בְּחֹל הַמּוֹעֵד: {יח} <B> וְעוֹשִׂין כוּ'.</B> לַאֲתוֹיֵי בּוֹרוֹת שִׁיחִין וּמְעָרוֹת כְּשֶׁרַבִּים צְרִיכִין לָהֶם עַתָּה. אֲבָל שֶׁלֹּא לְצֹרֶךְ הַמּוֹעֵד לֹא הִתִּירוּ, מִפְּנֵי שֶׁהֵן מְלֶאכֶת אֻמָּן, וְאֵינוֹ יָכֹל לְשַׁנּוֹת בָּהֶן, וְאִם יְשַׁנֶּה בָּהֶן יַפְסִיד. הָרֹא"שׁ. וְעַיֵּן פֵּרוּשׁ הָרַ"ב בְּרֵישׁ שְׁקָלִים, וְהוּא מֵהַיְרוּשַׁלְמִי. וּגְמָרָא דִּידָן לֹא סְבִירָא לֵיהּ הַאי דִּיּוּקָא, מִשּׁוּם דְּלֹא דָּמִי לְהָנָךְ פְּרָטֵי דְּמַתְנִיתִין: {יט} <B> וְיוֹצְאִין כוּ'.</B> בַּגְּמָרָא רָמֵי וְהָתְנַן בְּרֵישׁ שְׁקָלִים בְּא' בַּאֲדָר כוּ', וּמְשָׁנֵי כָּאן בְּבָכִיר כָּאן בְּאָפִיל: {כ} וְלֹא הֲוֵי מְעִילָה, דְּלֵב בֵּית דִּין מַתְנֶה עֲלֵיהֶן: 

~ פרק א - משנה ג

{כא} דְּאִי בְּבֵית הַשְּׁלָחִין הָתְנַן בְּרֵישׁ פִּרְקִין דְּמַשְׁקִין, וְהַיְינוּ כָּל הַשָּׂדֶה. אֲבָל תֵּימַהּ מַה דּוֹחְקֵיהּ לְאוֹקְמֵיהּ בִּשְׂדֵה הַבַּעַל, לוֹקְמֵיהּ בִּשְׁלָחִין, וְאִלּוּ בֵּית הַבַּעַל מַאן דְּכַר שְׁמֵיהּ כוּ', וְאִי קַשְׁיָא מַשְׁקִין דִּלְעֵיל הֲרֵי כָּתַב בְּעַצְמוֹ לְעֵיל דְּשֶׁל אִילָן לֹא שָׁרֵי בַּמּוֹעֵד דְּלֹא פָּסְדֵי, וּמַתְנִיתִין הָכִי דַּיְיקָא דִּמְשִׁיכָה מֵאִילָן לְאִילָן הוּא דְּמֻתָּר הָא הַשְׁקָאָה לֹא. וְכֵן דִּבְרֵי הָרַ"ן לְחַלֵּק בְּכָךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כב} סְבָרָתוֹ בָּזֶה כְּפֵרוּשׁ רַשִׁ"י דִּמְפָרֵשׁ זְרָעִים שֶׁלֹּא שָׁתוּ לִפְנֵי הַמּוֹעֵד שֶׁלֹּא הָיוּ מְלֻמָּדִים לִשְׁתּוֹת תָּמִיד לִפְנֵי הַמּוֹעֵד לֹא יַשְׁקֶה בַּמּוֹעֵד דְּהוֹאִיל וְלֹא מַשְׁקֶה לְהוּ תָּדִיר לִפְנֵי הַמּוֹעֵד לֹא הֲוֵי לְהוּ פְּסִידָא אִי לֹא מַשְׁקֶה לְהוּ בַּמּוֹעֵד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ד

{כג} וְהַגֻּמָּא אֵינוֹ אֶלָּא כְּדֵי שֶׁיִּתְקַבְּצוּ שָׁמָּה אִישׁוּת הַרְבֵּה, וְאֶפְשָׁר שֶׁנּוֹתֵן שָׁם אֳכָלִין כְּדֵי שֶׁיָּבֹאוּ וְיֵאָחְזוּ בַּמְּצוּדָה. הָרַ"ן: {כד} <B> וּבַשְּׁבִיעִית.</B> לְשׁוֹן רַשִׁ"י אַף עַל גַּב דִּמְתַקֵּן אֶת הַשָּׂדֶה. וּלְשׁוֹן הָרַ"ן דְּלֹא גָּזְרִינַן דִּלְמָא מֵחֲזֵי כַּחֲרִישָׁה כְּשֶׁאָדָם חוֹפֵר וְסוֹתֵם אֶת חֹרֵיהֶם: {כה} <B> וּמְקָרִין.</B> נִרְאֶה לִי שֶׁהוּא נִגְזַר מִשֵּׁם קִיר. וְלֹא תָּנֵי מְקִירִין, שֶׁלֹּא יִתְעָרֵב בְּמִלַּת מֵקִיר שֶׁעִנְיָנוֹ קְרִירוּת:
