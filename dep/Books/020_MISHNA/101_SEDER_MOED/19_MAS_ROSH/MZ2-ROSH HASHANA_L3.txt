~ פרק א - משנה א

{א} שֶׁהֲרֵי מָנָה שָׁנָה זוֹ שְׁנַת אַרְבַּע מֵאוֹת וּשְׁמוֹנִים לִיצִיאַת מִצְרַיִם, וְשָׁנָה רְבִיעִית לִמְלֹךְ שְׁלֹמֹה. רַשִׁ"י: {ב} כִּדְּיָלֵיף לַהּ מִקְּרָאֵי בַּגְּמָרָא: {ג} לְמַלְכֵי יִשְׂרָאֵל אַסְבְּרַא לֵיהּ, דִּבְהוּ קַיְמִינַן הַשְׁתָּא: {ד} דְּאִלּוּ בְּנִיסָן שֶׁלְּאַחַר אֲדָר וְכָתַב שֶׁל שָׁנָה רִאשׁוֹנָה, אַף עַל פִּי שֶׁלְּפִי הָאֱמֶת הִיא שָׁנָה הַשְּׁנִיָּה, לֵיכָּא לְמֵיחַשׁ כוּ', שֶׁכֵּיוָן שֶׁזֶּה הַמֶּלֶךְ לֹא מָלַךְ שׁוּם נִיסָן, וְשֵׁם הַמֶּלֶךְ הֲרֵי כָּתוּב בַּשְּׁטָר, וְנִמְצָא שֶׁלֹּא יוּכַל לִטְרֹף לָקוֹחוֹת אֶלָּא שֶׁלָּקְחוּ אַחַר הַנִּיסָן שֶׁמָּלַךְ זֶה הַמֶּלֶךְ, וּלְאוֹתוֹ כְּדִין הוּא טוֹרֵף: {ה} עַיֵּן בְּהָרַ"ב מִשְׁנָה ה' פֶּרֶק י' דִּשְׁבִיעִית: {ו} כְּלוֹמַר וְרַבִּי שִׁמְעוֹן הִיא, גְּמָרָא. וְטַעֲמַיְיהוּ דְּרַבִּי שִׁמְעוֹן וְרַבָּנָן מִקְּרָאֵי יָלְפִינַן לְהוּ בַּגְּמָרָא: {ז} <B> בְּתִשְׁרֵי.</B> וְאַרְבָּעָה רָאשֵׁי שָׁנִים דְּחָשְׁבֵי לַחֲמִשָּׁה עָשָׂר בְּנִיסָן שֶׁהוּא רֹאשׁ הַשָּׁנָה לָרְגָלִים, הָכִי קָאָמַר, אַרְבָּעָה רָאשֵׁי שָׁנִים לְדִבְרֵי הַכֹּל. לְרַבִּי שִׁמְעוֹן דְּבָעֵי שְׁלֹשָׁה רְגָלִים כְּסִדְרָן דַּל מַעֲשַׂר בְּהֵמָה, וּלְדִפְלִיגֵי עֲלֵיהּ דַּל רְגָלִים. גְּמָרָא: {ח} וְהַאי גָּמְרוּ הַיְנוּ שֶׁנִּגְמַר לְהִתְחַיֵּב בְּמַעֲשֵׂר, שֶׁאֵין אוֹכְלִים אֲפִלּוּ עֲרַאי, וְהוּא מִשֶּׁיִרְאֶה פְּנֵי הַבַּיִת: {ט} גְּמָרָא. וּזְמַן לֵידַת בְּהֵמָה גַּסָּה אַף עַל גַּב דְּיוֹלֶדֶת לְתִשְׁעָה חֳדָשִׁים, נַמִּי בְּאָב הוּא, אֶלָּא שֶׁזְּמַן עִבּוּרָן קָדִים אַרְבָּעָה חֳדָשִׁים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} <B> לַשָּׁנִים כוּ'.</B> מִקְּרָאֵי יָלְפִינַן לְהוּ בַּגְּמָרָא: {יא} שֶׁיּוֹם הַנְּטִיעָה אֵינוֹ מִכְּלָל יְמֵי הַקְּלִיטָה: {יב} בַּגְּמָרָא יָלֵיף לַהּ: {יג} הָכִי יָלְפִינַן לְהוּ בַּגְּמָרָא: {יד} כִּדְמְפָרֵשׁ בַּגְּמָרָא דְתַנָּא תָּנֵי יְרָקוֹת שֶׁמַּעַשְׂרוֹתֵיהֶן מִדְּרַבָּנָן. וְכָל שֶׁכֵּן מַה שֶּׁחַיָּבִים מִדְּאוֹרַיְתָא. וּדְתָנֵי יְרָקוֹת לְשׁוֹן רַבִּים, מִשּׁוּם דִּתְרֵי גַּוְנֵי יָרָק הֵן דְּאָזְלִינַן בָּתַר לְקִיטָה כִּדְאָמְרִינַן לְעִנְיַן גָּרְנוֹ מִשֶּׁנֶּאֱגַד [וְכוּ' בְּמַסֶּכֶת מַעַשְׂרוֹת פֶּרֶק א' מִשְׁנָה ה']: {טו} וּבַגְּמָרָא אִיתָא נַמִּי וְאִם הָיְתָה שְׁנִיָּה נִכְנֶסֶת לַשְּׁלִישִׁית כוּ' כִּדְבַסָּמוּךְ בְּהָרַ"ב: {טז} <B>בִּשְׁבָט.</B> הוֹאִיל וְיָצְאוּ רֹב גִּשְׁמֵי שָׁנָה וְעָלָה הַשָּׂרָף בָּאִילָנוֹת, וְנִמְצָא הַפֵּרוֹת חוֹנְטִין מֵעַתָּה. גְּמָרָא: {יז} דִּלְעִנְיַן עָרְלָה וּרְבָעִי כְּבָר נִתְבָּאֵר לְעֵיל: 

~ פרק א - משנה ב

{יח} זֶה לְשׁוֹן הַבְּרַיְתָא, מִפְּנֵי מָה אָמְרָה הַתּוֹרָה הָבִיאוּ עֹמֶר בְּפֶסַח, מִפְּנֵי שֶׁהַפֶּסַח זְמַן תְּבוּאָה הוּא, אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא הָבִיאוּ כוּ' כְּדֵי כוּ', וְכֵן בְּכֻלָּם: {יט} <B>בְּרֹאשׁ הַשָּׁנָה.</B> בַּגְּמָרָא (דַּף ח) יָלְפִינַן לַהּ מִקְּרָאֵי. וְאַף עַל גַּב דְּוַדַּאי עַל כָּל הַמְּאֹרָעוֹת נִדּוֹנִין וּבִכְלָלָם אִינָךְ שְׁלֹשָׁה דִּינִים, מִכָּל מָקוֹם בִּשְׁלֹשָׁה פְּרָקִים הַלָּלוּ הוּא גְּזַר דִּין שֶׁל כָּל הָעוֹלָם, וּבְרֹאשׁ הַשָּׁנָה כָּל בָּאֵי כוּ' וְגוֹזְרִין לְכָל אֶחָד חֶלְקוֹ מִדְּבָרִים אֵלּוּ. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} גְּמָרָא.</B> וּפֵרֵשׁ רַשִׁ"י וְאַקְּרָא דִּלְעֵיל מִנֵּיהּ קָמְהַדַּר, הִשְׁגִּיחַ אֶל כָּל כוּ', הַיּוֹצֵר אוֹתָם הִשְׁגִּיחַ יַחַד אֶת לִבָּם. וּלְהָכִי נָסֵיב הַאי תַּנָּא לְהַךְ קְרָא, לוֹמַר שֶׁאַף עַל פִּי שֶׁעוֹבְרִין כִּבְנֵי מָרוֹן בְּדִקְדּוּק גָּדוֹל, מִכָּל מָקוֹם כֻּלָּן נִסְקָרִין בִּסְקִירָה אַחַת. וְזֶהוּ שֶׁכָּתַב הָרַמְבַּ"ם, וְהַנִּגְלֶה מִזֶּה הַמַּאֲמָר מְבֹאָר כַּאֲשֶׁר תִּרְאֶה, אֲבָל הַנִּסְתָּר עִנְיָנוֹ קָשֶׁה בְּלִי סָפֵק, עַד כָּאן. כִּי הָאָדָם אִי אֶפְשָׁר לְצַיֵּר שְׁנֵי הֲפָכִים בְּנוֹשֵׂא אֶחָד בִּזְמַן אֶחָד שֶׁיַּעַבְרוּ אֶחָד אֶחָד וְיִהְיוּ נִסְקָרִים בִּסְקִירָה אַחַת, אֲבָל כְּמוֹ שֶׁאִי אֶפְשָׁר לְהַשִּׂיג מַהוּתוֹ יִתְבָּרֵךְ וְיִתְעַלֶּה זִכְרוֹ לָנֵצַח, כֵּן אִי אֶפְשָׁר לְהַשִּׂיג הַשְׁגָּחָתוֹ וּסְקִירָתוֹ. תּוֹסְפוֹת יוֹם טוֹב: {כא} <B> וּבֶחָג.</B> וְאַף עַל פִּי שֶׁשְּׁלָשְׁתָּן בְּשֵׁם חַגִּים נִקְרְאוּ בַּכָּתוּב, מִכָּל מָקוֹם מִלַּת חָג הֲנָחָתוֹ עַל הַמָּחוֹל וְהָרִקּוּד שֶׁהוּא בְּחַג הַסֻּכּוֹת, וְהָיְתָה שִׂמְחָה יְתֵירָה שֶׁל בֵּית הַשּׁוֹאֵבָה: {כב} כְּדֵי שֶׁיִּתְבָּרְכוּ לָכֶם גִּשְׁמֵי הַשָּׁנָה.</B> בְּרַיְתָא: 

~ פרק א - משנה ג

{כג} <B> נִיסָן.</B> אֲבָל סִיוָן מִפְּנֵי הָעֲצֶרֶת לֹא צָרִיךְ, דְּבִסְפִירַת עֹמֶר הַדָּבָר תָּלוּי. תּוֹסָפוֹת: {כד} וְאַף עַל גַּב דִּבְשִׁבְעָה עָשָׂר בְּתַמּוּז נַמִּי אֵרְעוּ חֲמִשָּׁה דְּבָרִים חֻרְבַּן בֵּית הַמִּקְדָּשׁ תַּקִיפָא טוּבָא. וְעוֹד דִּבְתִשְׁעָה בְּאָב צָרָה אַחַת הֻכְפְּלָה בּוֹ. תּוֹסָפוֹת: {כה} <B> הַפּוּרִים.</B> וְאִלּוּ עַל אֲדָר הַשֵּׁנִי מִפְּנֵי פּוּרִים לֹא קָתָנֵי, דִּסְבִירָא לֵיהּ דִּלְעוֹלָם אֲדָר הָרִאשׁוֹן מְעֻבָּר וְיָדְעוּ בְּנֵי הַגּוֹלָה דִּבְיוֹם שְׁלֹשִׁים וְאֶחָד נִתְקַדֵּשׁ. גְּמָרָא: {כו} <B> פֶּסַח קָטָן.</B> לְפִיכָךְ קָרֵי לֵיהּ פֶּסַח קָטָן, דְּלֹא הָיָה רַק יוֹם אֶחָד. מְפָרֵשׁ הָרַמְבַּ"ם בְּהִלְכוֹת קִדּוּשׁ הַחֹדֶשׁ: 

~ פרק א - משנה ד

{כז} הַיְנוּ לְאַחַר שֶׁיִּשְׁמְעוּ מִפִּי בֵּית דִּין מְקֻדָּשׁ, דְּאַף שֶׁנִּרְאֶה בַּעֲלִיל [בְּיוֹם כ"ט] אוֹ בְּלֵיל שְׁלֹשִׁים צְרִיכִין לְהַמְתִּין עַד שֶׁיִּשְׁמְעוּ כוּ', דְּדִלְמָא מִמְלְכֵי בֵית דִּין וּמְעַבְּרֵי לֵיהּ וְאָתֵי לְקִלְקוּלֵי מוֹעֲדוֹת, וְאִם כֵּן אִם לֹא יָבֹאוּ עֵדִים וְיִתְעַבֵּר הַחֹדֶשׁ לֹא יֵצְאוּ הַשְּׁלוּחִין עַד לְמָחָר, שֶׁאֵין מְקַדְּשִׁין בַּלַּיְלָה (וְאַף דְּאָז אִי אֶפְשָׁר לִדְחוֹתוֹ יוֹתֵר צְרִיכִין לְהַמְתִּין, דְּגָזְרֵי הַאי אַטּוּ הַאי), נִמְצְאוּ מְמַעֲטִין הִלּוּכָן מַהֲלַךְ לַיְלָה וְיוֹתֵר. אֲבָל בִּשְׁאָר חֳדָשִׁים אֵינָם צְרִיכִין לְהַמְתִּין עַד שֶׁיִּשְׁמְעוּ מְקֻדָּשׁ, דְּאֵין הַמּוֹעֲדוֹת תְּלוּיִן בָּהֶם, וְאִם כֵּן אֵין חִלּוּל שַׁבָּת [שֶׁל עֵדִים] מְמַהֵר יְצִיאַת הַשְּׁלוּחִים. גְּמָרָא וּפֵרוּשׁ רַשִׁ"י וְתוֹסָפוֹת: {כח} <B> וּבָהֶן כוּ'.</B> אֲבָל אֱלוּל אַף עַל גַּב דְּמִנֵּיהּ חַשְׁבִינַן לַעֲשׂוֹת בְּיוֹם שְׁלֹשִׁים רֹאשׁ הַשָּׁנָה, לֹא נָפְקָא מִינָהּ מִידֵי, לְפִי שֶׁעֲדַיִן בְּיוֹם שְׁלֹשִׁים יוֹשְׁבִים וּמְצַפִּים כוּ', וְאִם לֹא נוֹדַע לָהֶם נוֹהֲגִין הַיּוֹם וְיוֹם שְׁלֹשִׁים וְאֶחָד קֹדֶשׁ עַד שֶׁיָּבֹאוּ לָהֶם שְׁלוּחֵי תִּשְׁרֵי. הָרַמְבַּ"ם:

{יא} <B> מַתְקִיעַ.</B> מִדְּלֹא תְּנַן תּוֹקֵעַ, דִּקְדֵּק רַבֵּנוּ הַאי שֶׁאֵין שְׁלִיחַ צִבּוּר תּוֹקֵעַ אֶלָּא אַחֵר. וְטַעְמָא כְּדִתְנַן בְּפֶרֶק ה' דִּבְרָכוֹת לְעִנְיַן נְשִׂיאַת כַּפַּיִם, וְכָל חִלּוּקֵי דִּינִין דְּהָתָם שַׁיְּכִי נַמִּי הָכָא. הָרֹא"שׁ: {יב} וּמִנְהָגָם הָיָה שֶׁשְּׁלִיחַ צִבּוּר הַמִּתְפַּלֵּל בְּיוֹצֵר אֵינוֹ מִתְפַּלֵּל בְּמוּסָף אֶלָּא שְׁלִיחַ צִבּוּר אַחֵר. הָרַ"ן: {יג} וְאַף עַל פִּי שֶׁבָּטְלָה הַגְּזֵרָה לֹא עָבְדִינַן כִּדְמֵעִקָּרָא, אַף עַל גַּב דִּזְרִיזִין מַקְדִימִין כוּ', דְּחַיְישִׁינַן שֶׁמָּא יַחֲזֹר הַדָּבָר לְקִלְקוּלוֹ. תּוֹסָפוֹת. וְכָתַב הָרַ"ן דְּתִקְנוּ תְּקִיעוֹת וְתֵשַׁע בְּרָכוֹת בְּמוּסָפִין, דְּהָא עַל סֵדֶר בְּרָכוֹת תּוֹקְעִים כִּדִלְעֵיל, וְקֹדֶם לָכֵן הָיָה הַכֹּל בַּשַּׁחֲרִית: {יד} דְּאֶפְשָׁר מֶלֶךְ יוֹשֵׁב עַל כִּסֵּא דִּין וְסִפְרֵי חַיִּים וְסִפְרֵי מֵתִים פְּתוּחִין לְפָנָיו וְיִשְׂרָאֵל אוֹמְרִים שִׁירָה. גְּמָרָא: {טו} <B> רִאשׁוֹן כוּ'.</B> מִשּׁוּם דִּזְרִיזִין מַקְדִימִין לְמִצְוֹת. גְּמָרָא: {טז} <B>מַקְרֵא.</B> וְלֹא קָתָנֵי קוֹרֵא, הַיְנוּ טַעְמָא לְפִי שֶׁהוּא מַקְרֵא לַצִּבּוּר וּכִדְאָמְרִינַן הוּא אוֹמֵר הַלְלוּיָהּ וְהֵן אוֹמְרִים הַלְלוּיָהּ. הָרַ"ן:

{יז} <B> אֵין מַעֲבִירִין כוּ'.</B> מַאי טַעְמָא, שׁוֹפָר עֲשֵׂה וְיוֹם טוֹב עֲשֵׂה וְלֹא תַעֲשֶׂה, וְאֵין עֲשֵׂה דּוֹחֶה לֹא תַעֲשֶׂה וַעֲשֵׂה. גְּמָרָא. וַחֲכָמִים עָשׂוּ חִזּוּק לְדִבְרֵיהֶם הָכָא כְּשֶׁל תּוֹרָה. הָרַ"ן: {יח} <B> לֹא תַעֲשֶׂה.</B> זוֹ וְאֵין צָרִיךְ לוֹמַר זוֹ קָתָנֵי. גְּמָרָא: {יט} כִּדְפֵרֵשׁ רַשִׁ"י.</B> וְטַעְמָא, שֶׁהוֹאִיל וְהִגִּיעַ לְחִנּוּךְ מִתְעַסְּקִים עִמּוֹ כְּדֵי שֶׁיִּלְמֹד, שֶׁיֵּדַע לִתְקֹעַ לָצֵאת יְדֵי חוֹבָה כְּשֶׁיָּבֹא רֹאשׁ הַשָּׁנָה, אֲבָל כְּשֶׁלֹּא הִגִּיעַ לְחִנּוּךְ לֹא. רַשִׁ"י. וְאַף דְּקָיְמָא לָן קָטָן הָאוֹכֵל אִסּוּרִים אֵין בֵּית דִּין מְצֻוִּין לְהַפְרִישׁוֹ, הָנֵי מִלֵּי בְּמִלְתָּא דְּלֹא מִפַרְסְמָא אֲבָל בְּמִפַרְסְמָא כִּי הָא שֶׁהַקּוֹל נִשְׁמַע לָרַבִּים, מְעַכְּבִין, דִּלְמָא נָפֵיק מִינָהּ חוּרְבָא לִתְקֹעַ בְּשַׁבָּת שֶׁלֹּא בִּפְנֵי בֵּית דִּין. הָרַ"ן: {כ} <B> מִתְעַסְּקִין.</B> שֶׁאוֹמְרִים לָהֶם תִּקְעוּ וּמִשְׁתַּדְּלִין עִמָּהֶם. הַמַּגִּיד: {כא} <B> לֹא יָצָא.</B> לְמַה שֶּׁפֵּרֵשׁ הָרַ"ב פֶּרֶק דִּלְעֵיל מִשְׁנָה ז' דְּבָעֵי כַּוָּנָה לָצֵאת, מִקְרֵי מִתְעַסֵּק כָּל שֶׁאֵינוֹ מִתְכַּוֵּן לִתְקֹעַ תְּקִיעָה שֶׁל מִצְוָה, וּמִמַּתְנִיתִין דִּלְעֵיל לֹא שָׁמְעִינַן אֶלָּא דְּהַשּׁוֹמֵעַ צָּרִיךְ כַּוָּנָה, אֲבָל הַתּוֹקֵעַ עַצְמוֹ הֲוָה אֲמִינָא דְּיֵצֵא בּוֹ אֲפִלּוּ בְּמִתְעַסֵּק וְלֹא לִבָּעֵי כַּוָּנָה לָצֵאת, קָא מַשְׁמַע לָן מַתְנִיתִין דְּהָכָא. וְתָנֵי תוּ וְהַשּׁוֹמֵעַ, לְאַשְׁמְעִינַן דְּאַף דְּנִתְכַּוֵּן לָצֵאת אִם הַתּוֹקֵעַ מִתְעַסֵּק לֹא יָצָא:

{כב} דְּסֵיפֵיהּ</b> דִּקְרָא דִּוְהַעֲבַרְתָּ שׁוֹפַר תְּרוּעָה אִיהוּ: {כג} מִשּׁוּם דִּמְרָמֵי לַהּ רָמֵי מַתְנִיתִין וּבְרַיְתָא אַהֲדָדֵי, מְשָׁנֵי לַהּ בַּגְּמָרָא דְּתַנָּא דִּידָן קָא חָשִׁיב תְּקִיעוֹת דְּכֻלְּהוּ בָּבֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כד} <B> כְּשָׁלֹשׁ.</B> רַ"ח גּוֹרֵס שָׁלֹשׁ יְבָבוֹת, וְלֹא כְּשָׁלֹשׁ, דְּהָא תְּרוּעָה עַצְמָה שָׁלֹשׁ יְבָבוֹת, וְהֵיכִי שַׁיָּךְ לְמִתְנֵי כְּשָׁלֹשׁ. אֶלָּא שֶׁאֶפְשָׁר שֶׁלְּפִי שֶׁעִקַּר יְבָבוֹת בְּפִיו, קָאָמַר דְּשִׁעוּר תְּרוּעָה כְּשָׁלֹשׁ יְבָבוֹת שֶׁאָדָם מְיַלֵּל בְּפִיו: {כה} <B> יְבָבוֹת.</B> בַּגְּמָרָא, דִּתְרוּעָה מְתַרְגְמִינַן יְבָבָא, וּכְתִיב בְּאִמֵּיהּ דְּסִיסְרָא [שֹׁפְטִים ה] בְּעַד הַחַלּוֹן נִשְׁקְפָה וַתְּיַבֵּב. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כו} <B> בָּרִאשׁוֹנָה כוּ'.</B> שֶׁאֵין לַתְּקִיעוֹת שִׁעוּר לְמַעְלָה, שֶׁאִם רָצָה לְהַאֲרִיךְ מַאֲרִיךְ, וְאֵין הַדָּבָר נִכָּר כְּשֶׁהוּא מַאֲרִיךְ שֶׁיִּתְכַּוֵּן לִשְׁתֵּי תְּקִיעוֹת, אֶלָּא מֵחֲמַת כְּשֶׁהִיא כִּפְלַיִם כְּשִׁעוּר תְּקִיעָה רִאשׁוֹנָה, מִשּׁוּם הָכִי תָּנֵי תָּקַע בָּרִאשׁוֹנָה. הָרֹא"שׁ וְהָרַ"ן: {כז} וְאַף עַל פִּי שֶׁיִּמְשֹׁךְ עַד שֶׁיֹּאמַר עֲשָׂרָה זִכְרוֹנוֹת יֵשׁ לַעֲשׂוֹת כָּךְ.</B> וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כח} <B> אֶלָּא אֶחָת.</B> הַיְנוּ תְּקִיעָה שֶׁלַּאֲחָרֶיהָ, וְלֹא כַּיֵּשׁ מְפָרְשִׁים דְּהַיְנוּ הָרִאשׁוֹנָה וְשֶׁמִּשּׁוּם הָכִי תָּנֵי תָּקַע בָּרִאשׁוֹנָה: {כט} <B> שָׁלֹשׁ פְּעָמִים.</B> בִּשְׁבִיל מַלְכוּיוֹת, וְכֵן בִּשְׁבִיל זִכְרוֹנוֹת, וְכֵן בִּשְׁבִיל שׁוֹפָרוֹת. רַשִׁ"י: {ל} דְּאַף עַל גַּב דְּכָל יִשְׂרָאֵל עֲרֵבִים הֵם זֶה לָזֶה, וּמֵהַאי טַעְמָא מוֹצִיא זֶה אֶת זֶה בְּבִרְכוֹת הַמִּצְוֹת, הָא תַּנְיָא בַּגְּמָרָא [אַמַּתְנִיתִין דְּהָכָא] יָחִיד שֶׁלֹּא בֵּרַךְ אֵין חֲבֵרוֹ מְבָרֵךְ עָלָיו, וְטַעְמָא כָּתַב הָרַ"ן בְּשֵׁם יְרוּשַׁלְמִי מִשּׁוּם דִּתְפִלָּה מִסְתַּבֵּר שֶׁיִּהְיֶה כָּל אֶחָד מְבַקֵּשׁ רַחֲמִים עַל עַצְמוֹ:
