~ פרק א - משנה א

{א} דְּוַדַּאי לֹא אָתָא לְאַשְׁמְעִינַן אֶלָּא אַמַּאי דִמְרַבִּינַן. דִּבְמִי דִפְשִׁיטָא לָן שֶׁשּׁוֹחֵט, מַאי אָתֵי לְמֵימְרָא, קְרָא כְּתִיב וְזָבַחְתָּ: {ב} הַנְהוּ דְכָתַב בְּרֵישׁ פֶּרֶק ב' דְּהִלְכָתָא גְּמִירֵי לְהוּ: {ג} לְשׁוֹן עֲיֵפוּת וְחַלִּישׁוּת הַלֵּב, שֶׁאֵין יָכוֹל לִרְאוֹת מַכַּת חֶרֶב וְסַכִּין אֲפִלּוּ בִּבְהֵמָה. כְּמוֹ וְתִתְעַלַּפְנָה הַבְּתוּלֹת [עָמוֹס ח]. רַשִׁ"י: {ד} וּמִיהוּ אַדִּידְהוּ נַמִּי קָאֵי, דְּדַוְקָא בְּדִיעֲבַד מַכְשִׁירִין שְׁחִיטָתָן, אֲבָל לְכַתְּחִלָּה לֹא אֲפִלּוּ בַאֲחֵרִים עוֹמְדִים עַל גַּבֵּיהֶם. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} רַשִׁ"י. וּמְסַיֵּם הָרַ"ן, שֶׁאֵין שְׁחִיטָה אֶלָּא בְּיִשְׂרָאֵל. אֲבָל נָכְרִי כֵּיוָן דְּלָאו בְּתוֹרַת שְׁחִיטָה הוּא, כְּשֶׁשְּׁחָטָהּ הַוְיָא לָהּ כְּמֵתָה מֵאֵלֶיהָ. הָרַ"ן. וּבְבֵית יוֹסֵף בְּשֵׁם רִ"י, מִשּׁוּם דִּכְתִיב וְזָבַחְתָּ וְאָכַלְתָּ אוֹתוֹ, שֶׁהוּא בַּר זְבִיחָה אֱכֹל מִזִּבְחוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} כְּלוֹמַר כְּתִקְרֹבֶת עֲבוֹדָה זָרָה דַּאֲסוּרָה בַהֲנָאָה, כְּדִתְנַן בְּמִשְׁנָה ג' פֶּרֶק ב' דַּעֲבוֹדָה זָרָה: {ז} וְזֶה לְשׁוֹן הָרַמְבַּ"ם, לְפִי שֶׁעוֹבְדֵי עֲבוֹדָה זָרָה חֲלוּקִים לִשְׁנֵי חֲלָקִים, הָאֶחָד מֵהֶן הַיּוֹדְעִין לַעֲשׂוֹת אוֹתָהּ, רְצוֹנוֹ לוֹמַר לְהַבִּיט אֶל הַמַּזָּלוֹת הַצּוֹמְחִים לְצֹרֶךְ מְלַאכְתָּם, וּלְהוֹרִיד הָרוּחָנִיּוּת בָּהֶן וּשְׁאָר אוֹתָן שְׁטֻיּוֹת הַמְטַנְּפִים הַשֵּׂכֶל כוּ'. וְהַחֵלֶק הַשֵּׁנִי, הֵן הָעוֹבְדִים לְאוֹתָן הַצּוּרוֹת הַיְדוּעוֹת כְּפִי מַה שֶּׁלָּמְדוּ מֵחַכְמֵיהֶם בִּלְבָד. וְכֵן הֵם רֹב עוֹבְדֵי עֲבוֹדָה זָרָה. וְהַחֵלֶק הַזֶּה הָאַחֲרוֹן עָלָיו אָמְרוּ חֲכָמִים זַ"ל בַּלָּשׁוֹן הַזֶּה, נָכְרִים שֶׁבְּחוּץ לָאָרֶץ לָאו עוֹבְדֵי עֲבוֹדָה זָרָה אֶלָּא מַעֲשֵׂה אֲבוֹתֵיהֶם בִּידֵיהֶם. הָרַמְבַּ"ם: {ח} וּכְרַבִּי יְהוּדָה בֶּן בְּתֵירָא דְּתַנְיָא כוּ' מִנַיִן לְתִקְרֹבֶת עֲבוֹדָה זָרָה שֶׁהִיא מְטַמְּאָה בְאֹהֶל, שֶׁנֶאֱמַר [תְּהִלִּים קו] וַיִּצָּמְדוּ לְבַעַל פְּעוֹר וַיֹּאכְלוּ זִבְחֵי מֵתִים, מַה מֵּת מְטַמֵּא בְאֹהֶל אַף תִּקְרֹבֶת עֲבוֹדָה זָרָה מְטַמְּאָה בְאֹהֶל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} דְּאִלּוּ בְּמֵזִיד, לֹא יֵאָכֵל עוֹלָמִית. כִּדְאַשְׁכְּחָן בִּמְבַשֵּׁל בְּשַׁבָּת בְּמֵזִיד בְּמִשְׁנָה ג' פֶּרֶק ב' דִּתְרוּמוֹת, וּבְפֵרוּשׁ הָרַ"ב שָׁם, דְּלַאֲחֵרִים מֻתָּר לְמוֹצָאֵי שַׁבָּת. וְכֵן כָּתְבוּ הָרַמְבַּ"ם וְהַטּוּר. וְלִכְאוֹרָה בִּשְׁחִיטָה נַמִּי דִּינָא הָכִי. וּמִכָּל מָקוֹם מַתְנִיתִין לֹא מִתּוֹקְמָא בְּמֵזִיד, דִּשְׁחִיטָתוֹ כְשֵׁרָה קָתָנֵי לֹא שְׁנָא לוֹ וְלֹא שְׁנָא לַאֲחֵרִים כוּ'. וְכִי תֵימָא אַמַּאי כָּשֵׁר לַאֲחֵרִים הָא הֲוֵי מְחַלֵּל שַׁבָּת, דְּפֵרֵשׁ הָרַ"ב בַּמִּשְׁנָה דִלְקַמָּן דִּשְׁחִיטָתוֹ פְסוּלָה, יֵשׁ לוֹמַר דִּלְקַמָּן מַיְרֵי בְּפַרְהֶסְיָא וְהָכָא בְּצִנְעָא. וְעוֹד יֵשׁ לוֹמַר דְּמִשּׁוּם פַּעַם אַחַת לָא חֲשִׁיב מוּמָר. תּוֹסָפוֹת. אֲבָל בְּהָרַמְבַּ"ם וְהַטּוּר מַשְׁמַע דִּבְמֵזִיד אָסוּר אַף לַאֲחֵרִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} גְּמָרָא, מַאי טַעֲמָא, כֵּיוָן דְּאִי אֶפְשָׁר לְכַזַּיִת בָּשָׂר בְּלֹא שְׁחִיטָה, אַדַּעְתָּא דְחוֹלֶה קָא שָׁחַט. פֵּרֵשׁ רַשִׁ"י, וְלֵיכָּא לְמִגְזַר שֶׁמָּא יַרְבֶּה בַּשְּׁחִיטָה בִּשְׁבִיל בָּרִיא, דְּהָא מִשּׁוּם הַהוּא זַיִת דְּחוֹלֶה בָּעֵי לְמִשְׁחָט כֻּלָּהּ: {יא} <B> בְּנַפְשׁוֹ.</B> בְּשַׁבָּת סְקִילָה. וּבְיוֹם הַכִּפּוּרִים כָּרֵת. הָרַ"ן: 

~ פרק א - משנה ב

{יב} מִשּׁוּם צוֹר אִצְטְרִיךְ לְאוֹקְמֵיהּ הָכִי. אֲבָל קָנֶה הֲוָה מָצֵי לְמֵימַר דְּמַיְרֵי בְּקָנֶה שֶׁאֵינוֹ גָדֵל בַּאֲגָם כְּדִלְקַמָּן. תּוֹסָפוֹת: {יג} בַּגְּמָרָא יָלְפִינַן מִקְּרָא דִּשְׁחִיטָה בְּתָלוּשׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} שֶׁאָמְרוּ מוּמָר לַעֲבוֹדָה זָרָה מוּמָר לְכָל הַתּוֹרָה כֻלָּהּ. הָרַמְבַּ"ם: {טו} לְפִי שֶׁעִקָּר בְּיָדֵינוּ גַּם כֵּן הַמְחַלֵּל שַׁבָּת בְּפַרְהֶסְיָא הֲרֵי הוּא כְּעוֹבֵד עֲבוֹדָה זָרָה לְכָל דְּבָרָיו. הָרַמְבַּ"ם: {טז} דִּשְׁחִיטָתוֹ פְסוּלָה, כְּדִתְנַן פֶּרֶק ב' מִשְׁנָה ח': {יז} לְשׁוֹן רַשִׁ"י אַף עַל גַּב דִּתְנַן בְּפֶרֶק ב' מִשְׁנָה ט' אֵין שׁוֹחֲטִין לְתוֹךְ יַמִּים, בְּרֹאשׁ הַסְּפִינָה שָׁרֵי דְּמוֹכְחָא מִלְּתָא דִּלְנַקֵּר כוּ': {יח} דְּאִלּוּ אֶחָד לֹא מִקְרֵי מְחֻבָּר בְּמַה שֶּׁהוּא בַּלֶּחִי, דְּלֶחִי הֲוֵי כְּבֵית יָד. הָרַמְבַּ"ם: {יט} בִּמְחֻבֶּרֶת לַגּוּף הוּא. דְּאִלּוּ לַיָּד וְהַיָּד תְּלוּשָׁה מִן הַגּוּף, לֹא מִקְרֵי מְחֻבָּר. כְּמוֹ שֵׁן בַּלֶּחִי: 

~ פרק א - משנה ג

{כ} וְהָא דְקָרֵי לָהּ טַבַּעַת סְתָמָא, מִשּׁוּם דְּאֵין בְּכָל הַקָּנֶה טַבַּעַת גְּמוּרָה אֶלָּא הִיא. שֶׁאֵינָן שְׁלֵמוֹת, אֶלָּא רְצוּעַת בָּשָׂר מְהַלֶּכֶת לְאָרְכּוֹ שֶׁל קָנֶה וּמְחַבֶּרֶת רָאשֵׁי הַטַּבָּעוֹת. רַשִׁ"י: {כא} וְהַיְנוּ דִּבְסָמוּךְ פֵּרֵשׁ מְלֹא חוּט הַשַּׂעֲרָה. וּמַה שֶּׁחָסֵר בָּזֶה גִּלָּה בָזֶה: 

~ פרק א - משנה ד

{כב} וְהָרַ"ן כָּתַב, דְּמִצְוַת שְׁחִיטָה לְכַתְּחִלָּה בְּאֶמְצַע צַוָּאר הוּא. דְּמִן הַצְּדָדִים אִיכָּא לְמֵיחַשׁ שֶׁמָּא תִּשָּׁבֵר מִפְרֶקֶת קֹדֶם שְׁחִיטַת הַסִּימָנִין. וְתוּ, כֵּיוָן דִּשְׁחִיטָה בַּסִּימָנִין הִיא, כִּי שָׁחֵיט מִן הַצְּדָדִין וְחוֹתֵךְ מִן הַמִּפְרֶקֶת אִיכָּא צַעַר בַּעֲלֵי חַיִּים. לְפִיכָךְ נִרְאֶה, דְּדִיעֲבַד הוּא דִּשְׁחִיטָתוֹ כְשֵׁרָה, אֲבָל לְכַתְּחִלָּה מִצְוָה לִשְׁחֹט מִן אֶמְצַע הַצַּוָּאר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} כְּדִכְתִיב [יִרְמְיָה ב] כִּי פָנוּ אֵלַי עֹרֶף. וְלֹא פָּנִים. גְּמָרָא: {כד} <B> פָּסוּל כוּ'.</B> מְפָרֵשׁ בַּגְּמָרָא דַּאֲתָא לְמִעוּטֵי מַחֲזִיר סִימָנִים לַאֲחוֹרֵי הָעֹרֶף, דְּפָסוּל בַּמְּלִיקָה. וְכָתַב רַשִׁ"י וְאַיְדֵי דְתָנֵי [רֵישָׁא] כָּשֵׁר בַּשְּׁחִיטָה כוּ', תָּנֵי נַמִּי [סֵיפָא] כָּשֵׁר בַּמְּלִיקָה כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} <B> גִּיד.</B> הוּא הַפְּנִימִי. רַשִׁ"י: {ב} <B> בָּאָרֶץ כוּ'.</B> כֻּלְּהוּ תְּנַנְהוּ מִשּׁוּם דְּאִצְטְרִיךְ לְמִתְנֵי בְּחֻלִּין וּבְמֻקְדָּשִׁים. הָרַ"ן: {ג} מִשּׁוּם מִמַּשְׁקֵה יִשְׂרָאֵל. וּלְעִנְיַן הַקָּרְבָּנוֹת וְנִסְכֵּיהֶם כְּתִיב, וְדָרְשִׁינַן מִן הַמֻּתָּר לְיִשְׂרָאֵל. גְּמָרָא: {ד} מִשּׁוּם הַקְרִיבֵהוּ נָא לְפֶחָתֶךָ אִי אֶפְשָׁר לְהַעֲלוֹת הַיָּרֵךְ כְּשֶׁהִיא מְפֹרַעַת. הִלְכָּךְ מַעֲלֵהוּ כְּמוֹת שֶׁהִיא שְׁלֵמָה וְאַחַר כָּךְ חוֹלְצוֹ בְּרֹאשׁ הַמִּזְבֵּחַ. גְּמָרָא: {ה} דְּבָעֵי רַבִּי יִרְמְיָה אִי בָּתַר דִּידֵיהּ אָזְלִינַן אִי בָּתַר מִינֵיהּ, וְקָאֵי בְּתֵיקוּ. וּמִבַּעְיָא לֵיהּ נַמִּי בִּבְהֵמָה וְלֹא עָגֹל, וְקָאֵי נַמִּי בְּתֵיקוּ: {ו} רַבִּי מֵאִיר לְטַעֲמֵיהּ דִּסְבִירָא לֵיהּ דְּאֵינוֹ נִתָּר בִּשְׁחִיטַת אִמּוֹ כִּדְאִיתָא בְּפֶרֶק ד' מִשְׁנָה ה', הִלְכָּךְ חָשׁוּב בְּהֵמָה בִּפְנֵי עַצְמוֹ וְחֶלְבּוֹ וְגִידוֹ אֲסוּרִים. וְרַבִּי יְהוּדָה לְטַעֲמֵיהּ דְּאָמַר נִתָּר בִּשְׁחִיטַת אִמּוֹ, הִלְכָּךְ קָרִינַן בֵּיהּ בְּהֵמָה בַּבְּהֵמָה תֹּאכֵלוּ. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} וְרַבִּי יְהוּדָה הִיא. דִּלְרַבִּי מֵאִיר כִּי הֵיכִי דְאוֹסֵר בְּגִיד אוֹסֵר נַמִּי בְּחֵלֶב. דְּבַחֲדָא שַׁיְּטָא שַׁיְּטֵי. וְהַיְנוּ דְּבַפֵּרוּשׁ הָאַחֵר כָּתַב הָרַ"ב לְדִבְרֵי הַכֹּל. אֲבָל בַּגְּמָרָא לֵיתָא רַק אוֹקִימְתָּא דִלְדִבְרֵי הַכֹּל וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} <B> הַחֵלֶב.</B> חֵלֶב מַאן דְּכַר שְׁמֵיהּ. הָכִי קָאָמַר [רַבִּי מֵאִיר] אֵין נֶאֱמָנִין עָלָיו וְעַל הַחֵלֶב. וַחֲכָמִים אוֹמְרִים כוּ'. גְּמָרָא. וּבִסְתָם חֵלֶב אַיְרֵי, דִּבְחֶלְבּוֹ שֶׁל גִּיד שָׁרֵי רַבִּי יְהוּדָה לְגַמְרֵי. תּוֹסָפוֹת:

{ט} <B> שׁוֹלֵחַ כוּ'.</B> שְׁמַע מִנַּהּ שֶׁמֻּתָּר בַּהֲנָאָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} לְשׁוֹן חֲתִיכָה. כְּמוֹ שֶׁכָּתַב הָרַ"ב בְּמִשְׁנָה ד' פֶּרֶק ב' דְּכִלְאַיִם: 

~ פרק ז - משנה ג

{יא} וְרַבָּנָן, הַהוּא דְּפָשֵׁיט אִסּוּרֵיהּ בְּכֻלֵּיהּ יָרֵךְ, [הוּא וּקְנוֹקְנוֹתָיו וְשָׁרָשָׁיו נִפְשָׁטִין בְּכָל הַיָּרֵךְ, וְהַיְנוּ גִּיד הַנָּשֶׁה הַפְּנִימִי הַנִּמְצָא בִּתְחִלַּת פְּרִיעַת הַיָּרֵךְ. רַשִׁ"י]. לַאֲפוֹקֵי חִיצוֹן דְּלֹא. גְּמָרָא:

{יב} וְאַף עַל פִּי שֶׁשֻּׁמַּן הַגִּיד בְּשִׁשִּׁים, אַף עַל פִּי שֶׁהוּא מִדִּבְרֵיהֶם, הוֹאִיל וְגִיד הַנָּשֶׁה בְּרִיָּה בִּפְנֵי עַצְמוֹ הֶחֱמִירוּ בוֹ כְּאִסּוּרֵי תוֹרָה. בֵּית יוֹסֵף: {יג} מִפְּנֵי שֶׁהִיא בְּרִיָּה בִּפְנֵי עַצְמָהּ עָשׂוּ הֶכֵּר בָּהּ וְהוֹסִיפוּ בְּשִׁעוּרֶיהָ. הָרַמְבַּ"ם. וְרָצָה לוֹמַר שֶׁיֵּשׁ בָּהּ אֶפְרוֹחַ שֶׁהִיא בְּרִיָּה. כֶּסֶף מִשְׁנֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

~ פרק ח - משנה א

{א} אֲבָל בְּשַׂר חַיָּה פְּלֻגְתָּא דְתַנָּאֵי בְּמִשְׁנָה ד': {ב} דִּיֵּק לְמִנְקָט אֲכִילָה וְאַף עַל גַּב דְּמַתְנִיתִין בִּשּׁוּל שָׁנְאָה, כְּשִׁיטַת הָרַמְבַּ"ם דְּלֹא אָסְרוּ חֲכָמִים אֶלָּא אֲכִילָה אֲבָל לֹא בִּשּׁוּל וַהֲנָיָה. וּפֵרֵשׁ הַמַּגִּיד שֶׁלֹּא יָפוּ כֹחָם מִבְּשַׂר טְהוֹרָה עִם חֲלֵב טְמֵאָה כוּ' שֶׁמֻּתָּרִין בְּבִשּׁוּל וַהֲנָיָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} גְּמָרָא. דְּאִי מִשּׁוּם אֲכִילָה לֵיכָּא לְמִגְזָר, דַּאֲכִילַת עוֹף גּוּפָהּ גְּזֵרָה, וַאֲנַן נֵיקוֹם וְנִגְזֹר גְּזֵרָה לִגְזֵרָה כוּ'. וְאִלְפָס רוֹתֵחַ הַיְנוּ בִּכְלִי רִאשׁוֹן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} <B> לְהַעֲלוֹתוֹ כוּ'.</B> חוּץ כוּ'. וְאִין הָכִי נַמִּי דִּבְשַׂר דָּגִים אֲפִלּוּ לְאָכְלוֹ בְּחָלָב שָׁרֵי, אֶלָּא נָקַט הַעֲלָאָה לְאַשְׁמְעִינַן דִּשְׁאָר בָּשָׂר אֲפִלּוּ בְּהַעֲלָאָה נַמִּי אָסוּר כוּ'. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} <B> וּבֵית הִלֵּל אוֹמְרִים כוּ'.</B> אַמִּלְּתַיְהוּ דְבֵית שַׁמַּאי מִיהַדְרוּ דְּקָאָמְרֵי עוֹלֶה וְאֵינוֹ נֶאֱכָל: {ו} <B> וְאֵינוֹ כוּ'.</B> וּבִלְבַד שֶׁלֹּא יִהְיוּ נוֹגְעִים זֶה בָזֶה. הָרַ"ן: 

~ פרק ח - משנה ג

{ז} כְּגוֹן שֶׁחֲתִיכָה שֶׁתַּחְתֶּיהָ קְצָתָהּ חוּץ לָרֹטֶב וְזוֹ מֻנַּחַת עָלֶיהָ. טוּר: {ח} בַּגְּמָרָא. וּפֵרֵשׁ רַשִׁ"י, מִפְּנֵי שֶׁמֵּי הַשּׁוּלַיִם עוֹלִין עַד פִּיהָ וְיוֹרְדִין: {ט} דַּחֲתִיכָה עַצְמָהּ נַעֲשֵׂית נְבֵלָה, וּבְנִעֵר אוֹ כִּסָּה לִבְסוֹף. כִּדְאָמְרִינַן בַּגְּמָרָא: {י} דַּחֲלֵב שְׁחוּטָה מִדְּאוֹרַיְתָא שָׁרֵי, אֶלָּא דְּרַבָּנָן גָּזְרוּ דִּלְמָא אָתֵי לְמֵיכָל בָּשָׂר בְּחָלָב. וְכֵיוָן שֶׁהוֹצִיא אֶת הֶחָלָב כָּל מַה שֶּׁיּוּכַל, תּוּ לֵיכָּא לְמִגְזָר. הָרֹא"שׁ: {יא} וְטַעֲמָא מִשּׁוּם טַעַם בָּשָׂר הַנִּבְלָע בֶּחָלָב שֶׁבַּכְּחָל. בֵּית יוֹסֵף. וְהָרַ"ן פֵּרֵשׁ מִשּׁוּם מַרְאִית הָעַיִן, שֶׁאַף עַל פִּי שֶׁרָאוּי לְהַתִּיר כְּחָל זֶה מִפְּנֵי שֶׁהֶחָלָב יָצָא מִמֶּנּוּ וְנִתְבַּטֵּל, אָתֵי לְמִשְׁרֵי חֲתִיכַת נְבֵלָה. וְנִרְאֶה לִי דְּרָצָה לוֹמַר הַנַּעֲשֵׂית נְבֵלָה עַל יְדֵי בְּלִיעַת חָלָב אוֹ שְׁאָר אִסּוּר שֶׁלֹּא הָיָה בוֹ שִׁשִּׁים לְבַטֵּל, דְּהַשְׁתָּא הַוְיָא דוּמְיָא דִכְחָל שֶׁאִסּוּרוֹ גַם כֵּן מִשּׁוּם בְּלִיעָה: 

~ פרק ח - משנה ד

{יב} <B> בַּהֲנָאָה.</B> דִּתְלָתָא קְרָאֵי כְּתִיבֵי. אֶחָד לְבִשּׁוּל וְאֶחָד לַאֲכִילָה וְאֶחָד לַהֲנָאָה: {יג} <B> רַבִּי עֲקִיבָא כוּ'.</B> אֵינוֹ חוֹלֵק אֶלָּא מְפָרֵשׁ לְדִבְרֵי תַּנָּא קַמָּא. הָרֹא"שׁ: {יד} כִּדְיָלֵיף לֵיהּ בַּגְּמָרָא: {טו} אַרֵישָׁא מְהַדַּר, בְּשַׂר בְּהֵמָה טְהוֹרָה וְכוּ'. דְּכָל בְּהֵמָה בְּמַשְׁמַע: {טז} צְרִיכֵי תְּרֵי קְרָאֵי, כְּדֵי שֶׁיִּהְיוּ שְׁנֵי כְתוּבִין הַבָּאִים כְּאֶחָד דְּאֵין מְלַמְּדִין. דְּאִי לֵיכָּא אֶלָּא חַד קְרָא, לֵילֵיף מִנֵּיהּ. כִּדְאִיתָא בַּגְּמָרָא:

{יז} <B> אֲסוּרָה.</B> בַּגְּמָרָא מַקְשִׁינַן אַהָא דְאָמְרִינַן בְּפֶרֶק ב' דַּעֲבוֹדָה זָרָה מִשְׁנָה ה' דְּטַעֲמָא דְּאָסְרוּ גְּבִינָה שֶׁל נָכְרִים מִפְּנֵי שֶׁמַּעֲמִידִין אוֹתָהּ בְּעוֹר קֵבָה שֶׁל נְבֵלָה. וּמַשְׁמַע הָא קֵבָה גוּפָא שָׁרְיָא. וּמְשָׁנִינַן, לֹא קַשְׁיָא, הָכָא קֹדֶם חֲזָרָה, וְהָתָם לְאַחַר חֲזָרָה דְּרַבִּי יְהוֹשֻׁע דְּהָתָם בַּמִּשְׁנָה: {יח} לֹא דָק, דִּלְהָרַמְבַּ"ם אֵין חִלּוּק בֵּין קָרוּשׁ לְצָלוּל. וְכָתַב הַכֶּסֶף מִשְׁנֶה, שֶׁצָּרִיךְ לוֹמַר דִּכְשֵׁרָה שֶׁיָּנְקָה מִן הַטְּרֵפָה קֵבָתָהּ אֲסוּרָה, קֹדֶם חֲזָרָה נִשְׁנֵית. {כ} <B> חֹמֶר כוּ'.</B> מִשּׁוּם סֵיפָא שֶׁהַדָּם נוֹהֵג כוּ' תָּנָא לָהּ הָכָא: {כא} <B> וְחַיָּבִין כוּ'.</B> בְּעִנְיַן אִסּוּר חָל עַל אִסּוּר עַיֵּן בְּפֶרֶק ג' דִּכְרֵתוֹת מִשְׁנָה ד': {כב} רַשִׁ"י. וּפְלִיאָה בְּעֵינַי, דְּהַךְ קְרָא בְּסֵדֶר אַחֲרֵי בְּפָרָשַׁת שְׁחוּטֵי חוּץ לְעִנְיַן שְׁלָמִים כְּתִיב, וַהֲוָה לֵיהּ לִכְתֹּב קְרָא דִּבְפָרָשַׁת עוֹלָה בְּסֵפֶר וַיִּקְרָא וְזָרְקוּ אֶת הַדָּם וְגוֹ' וְעָרְכוּ אֶת הַנְּתָחִים וְגוֹ'. וְכֵן כָּתְבוּ רַשִׁ"י וְהָרַ"ב בְּמִשְׁנָה ד' פֶּרֶק ד' דִּזְבָחִים: {כג} וּלְשׁוֹן רַשִׁ"י, דָּבָר שֶׁאֵין אַחֵר מַתִּירוֹ אֶלָּא הוּא מַתִּיר עַצְמוֹ. וְהַיְנוּ דְּחֵלֶב אַף עַל פִּי שֶׁהוּא מַתִּיר הַבָּשָׂר לַכֹּהֲנִים, כְּדִכְתִיב בִּשְׁמוּאֵל א' ב' קַטֵּר יַקְטִירוּן כַּיּוֹם הַחֵלֶב וְקַח לָךְ, מִכָּל מָקוֹם יֵשׁ לָהֶם מַתִּיר דְּהַיְנוּ הַדָּם. וְהַתּוֹסָפוֹת כָּתְבוּ דְּלָא חֲשִׁיבֵי מַתִּיר כֵּיוָן דְּאִם נִטְמְאוּ אוֹ אָבְדוּ שָׁרֵי בָּשָׂר בַּאֲכִילָה:

{א} וּכְלוֹמַר דְּתָנֵי נַמִּי בֵּיהּ שְׁחִיטָה שֶׁאֵינָהּ רְאוּיָה: {ב} דַּאֲשֶׁר יִזָּרֵעַ לֹא אִצְטְרִיךְ, וּכְמוֹ שֶׁכָּתַבְתִּי בְּרֵישׁ עֻקְצִין. וְאַף עַל גַּב דְּלֹא שַׁיְּכָא אֶלָּא בִזְרָעִים, מַסְקִינַן בַּגְּמָרָא דִּתְלָתָא קְרָאֵי כְּתִיבֵי [וּפֵרֵשׁ רַשִׁ"י, זֶרַע זָרוּעַ יִזָּרֵעַ], חַד לְשׁוֹמֵר דִּזְרָעִים, וְחַד לְשׁוֹמֵר דְּאִילָנוֹת כְּלוֹמַר פֵּרוֹת דְּאִילָנוֹת, וְחַד לְשׁוֹמֵר בָּשָׂר בֵּיצִים וְדָגִים כוּ': {ג} פֵּרֵשׁ רַשִׁ"י, לֵיחָה הַיּוֹצֵא מִן הַבָּשָׂר שֶׁקּוֹרִין גלייר"א: {ד} פֵּרֵשׁ רַשִׁ"י, תְּבָלִין בְּאַנְפֵּי נַפְשַׁיְהוּ לֹא אָכְלֵי לְהוּ אֱינָשֵׁי. וּלְעֵיל פֶּרֶק ז' מִשְׁנָה ה' פֵּרֵשׁ דַּק דַּק: {ה} <B> וְהָעֲצָמוֹת.</B> שֶׁיֵּשׁ בָּהֶן מֹחַ וְהוּא אֹכֶל וְהָעֶצֶם שׁוֹמֵר לוֹ לְפִיכָךְ מִצְטָרֵף עִמּוֹ. רַשִׁ"י: {ו} <B> וְהַגִּידִין.</B> שֵׁם נוֹפֵל עַל הַגִּידִים הַדּוֹפְקִים, וְעַל שֶׁאֵין דּוֹפְקִים, וְעַל הַקְּשָׁרִים, וְעַל הַקְּרוּמִים, וּמֵתָרִים, וְהָעֲצַבִּים. הָרַמְבַּ"ם: {ז} וּמַשְׁמַע דְּאִלּוּ לְקַבֵּל טֻמְאָה מִתְטַמְּאִים אַף בְּפָחוֹת מִכַּבֵּיצָה. וְלֹא פֵרֵשׁ כֵּן בְּרֵישׁ פֶּרֶק ה' דִּתְרוּמוֹת: {ח} רַשִׁ"י. וּמְסַיֵּם וּבִטְהוֹרָה לְיִשְׂרָאֵל לֹא אִצְטְרִיךְ לְמִתְנֵי, דְּכָל שֶׁכֵּן כֵּלִים פֶּרֶק ד' מִשְׁנָה ב': {י} כְּלוֹמַר בָּאֱלָל דְּרֵישָׁא קָאָמְרִינַן דִּבְכֻלְּהוּ אֱלָל קָאָמַר דְּמִצְטָרְפִין, וּבְאַפֵּי נַפְשַׁיְהוּ לֹא מְטַמְּאוּ. וְאָתָא רַבִּי יְהוּדָה לְמֵימַר מֵאֵלָיו, כְּגוֹן שֶׁכִּנְּסוּהוּ תִּינוֹקוֹת אַחַר הֶפְשֵׁט, לֹא, הוֹאִיל וּבָטֵל מִתְּחִלָּה וְנַעֲשָׂה עוֹר, תּוּ לֹא הָדַר הֲוֵי בָּשָׂר: {יב} רַשִׁ"י. דְּאִי אַאֲכִילָה מַאי בָּעֵי מְכֻנָּס {א} שֶׁכֵּן רַבִּי אִלָּעִאי פּוֹטֵר נַמִּי בְּהוּ מֵהַאי טַעֲמָא דְּיָלֵיף לֵיהּ נְתִינָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} מִשּׁוּם דְּאַשְׁכְּחָן אֲמוֹרָאֵי טוּבָא דְּנָהֲגוּ: {ג} בַּגְּמָרָא פָּרְכִינַן עֲלָהּ מֵהָא דְאָמְרִינַן דְּקָדָשִׁים אֲפִלּוּ שֶׁל בֶּדֶק הַבַּיִת אֲסוּרִים בְּגִיזָה כְּמוֹ שֶׁכָּתוּב פֶּרֶק י' מִשְׁנָה ב'. וּמַסִּיק, בַּמַּקְדִּישׁ חוּץ מִגִּיזָה, וּכְחָשָׁה דְגִיזָה. וּפֵרוּשׁ כְּחָשָׁה דְגִיזָה, שֶׁהַגִּיזָה הַנִּטֶּלֶת מַכְחֶשֶׁת הַבָּשָׂר, וְאִי לָאו דְּאָמַר חוּץ הָיָה אָסוּר לְגָזְזָהּ [שֶׁלֹּא לְהַכְחִישָׁהּ]. וְהָנֵי חוּץ מִשּׁוּם קָדְשֵׁי בֶדֶק הַבַּיִת נָקְטֵיהּ, דְּאִי קָדְשֵׁי מִזְבֵּחַ אֲפִלּוּ כִּי אָמַר חוּץ לֹא מַהֲנֵי, דַּאֲפִלּוּ לֹא קַדִּישׁ אֶלָּא אֵבָר מִמֶּנָּה פָּשְׁטָה בְּכֻלָּהּ: {ד} <B> בִּרְחֵלוֹת.</B> וְלֹא בָעִזִּים. אַף עַל גַּב דְּעִזִּים אִקְּרֵי נַמִּי צֹאן דִּכְתִיב [בְּרֵאשִׁית כז] לֶךְ נָּא אֶל הַצֹּאן וְקַח לִי מִשָּׁם שְׁנֵי גְדָיֵי עִזִּים. תּוֹסָפוֹת: {ה} בַּגְּמָרָא, וְנֵילַף גִּיזָה גִּיזָה מִבְּכוֹר כוּ'. וּמַסְקִינַן, אֶלָּא כִּדְאָמַר רַבִּי יְהוֹשֻׁעַ בֶּן לֵוִי, לַעֲמֹד לְשָׁרֵת [דְּסָמוּךְ לְגֵז כְּתִיב], דָּבָר הָרָאוּי לְשָׁרֵת, הָכִי נַמִּי דָּבָר הָרָאוּי לְשָׁרֵת. פֵּרֵשׁ רַשִׁ"י לִתְכֵלֶת דְּבִגְדֵי שְׂרָד, וְהָתָם צֶמֶר בָּעִינַן, וְאֵין צֶמֶר אֶלָּא שֶׁל רְחֵלִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
