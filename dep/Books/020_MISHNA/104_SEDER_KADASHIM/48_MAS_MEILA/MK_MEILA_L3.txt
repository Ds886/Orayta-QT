~ פרק א - משנה א

{א} <B>קָדְשֵׁי קָדָשִׁים.</B> כְּגוֹן חַטָּאת וְאָשָׁם וְשַׁלְמֵי צִבּוּר שֶׁיֶּשׁ בָּהֶן מְעִילָה עַד שֶׁיְּהֵא בָהֶן שְׁעַת הֶתֵּר לַכֹּהֲנִים, דְּהַיְנוּ לְאַחַר זְרִיקָה, דְּעַד שְׁעַת זְרִיקָה מִקְרֵי קָדְשֵׁי ה', וּבָאֵמוּרִין אַף לְאַחַר זְרִיקָה. אֲבָל בְּקָדָשִׁים קַלִּים אֵין בָּהֶן מְעִילָה עַד לְאַחַר זְרִיקָה, שֶׁאָז הֻבְרַר חֵלֶק גָּבוֹהַּ, וְאָז יֵשׁ מְעִילָה בְאֵמוּרֵיהֶם. תּוֹסָפוֹת: {ב}  גְּמָרָא. דְּלֵיכָּא לְמֵימַר דְּהָא קָא מַשְׁמַע לָן דִּזְרִיקָה לֹא מַהֲנֵי בְּהוּ לַאֲפוֹקֵי מִידֵי מְעִילָה. דְּהָא פְּשִׁיטָא הִיא. כֵּיוָן שֶׁהַזְּרִיקָה בְפָסוּל דְּלֹא מַהֲנֵי וְלֹא מִידִי. תּוֹסָפוֹת: {ג} לֹא לַגָּבוֹהַּ וְלֹא לַהֶדְיוֹט. דְּאֵין פּוֹדִים אֶת הַקֳּדָשִׁים לְהַאֲכִילָן לַכְּלָבִים. רַשִׁ"י. וְזֶה לְשׁוֹן הַתּוֹסָפוֹת, דְּכֵיוָן שֶׁהוֹלְכִין לְאִבּוּד לֹא אִקְרוּ קָדְשֵׁי ה': {ד}  גְּמָרָא. וּלְפִי זֶה מַשְׁמַע דְּמוֹעֲלִין דְּבַר תּוֹרָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} אֲבָל שְׁחָטָן בַּדָּרוֹם וְקִבֵּל דָּמָן בַּדָּרוֹם, בְּהָא לֹא מַיְרֵי מַתְנִיתִין, וְאַף עַל גַּב דַּהֲוֵי רְבוּתָא טְפֵי, מִשּׁוּם דְּמִסְּתָמָא כֵּיוָן דְּשָׁמְעִינַן חֲדָא לְרֵיעוּתָא הוּא הַדִּין בְּתַרְתֵּי. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} <B> חוּץ כוּ'.</B> דִּלְכֻלֵּי עָלְמָא אִם עָלוּ לֹא יֵרְדוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} <B> חוּץ כוּ'.</B> וְהוּא הַדִּין לְכָל הָנֵי דְתָנֵי בָרֵישָׁא, שְׁחָטָהּ בַּדָּרוֹם אוֹ בַלַּיְלָה שֶׁלֹּא הָיָה לָהּ שְׁעַת הֶתֵּר לַכֹּהֲנִים. וְהַיְנוּ טַעֲמָא דְּנָקַט הָנֵי תְרֵין דְּחוּץ כוּ' טְפֵי מִכָּל הָנֵי דְתָנֵי בָרֵישָׁא דְּמוֹעֲלִין בָּהֶן, דְּהָנֵי תְרֵי אִתַּקּוּשׁ אַהֲדָדֵי וּמֵחַד קְרָא נָפְקֵי מִוְּאִם הֵאָכֹל יֵאָכֵל. רַשִׁ"י: {ח} <B> וְשֶׁקִּבְּלוּ כוּ'.</B> וְהוּא הַדִּין נַמִּי לְהָא דְתָנֵי בָרֵישָׁא: 

~ פרק א - משנה ב

{ט} דַּהֲוֵי כְּמוֹ שֶׁלֹּא נִזְרַק עֲלֵיהֶם הַדָּם וְלֹא קָרְבוּ כָל מַתִּירָיו. וּמִשּׁוּם נוֹתָר, דְּאֵין נוֹתָר אֶלָּא בַבָּשָׂר הָרָאוּי לַאֲכִילָה בְתוֹךְ זְמַנּוֹ. וּמִשּׁוּם טֻמְאַת הַגּוּף כִּדְאָמְרִינַן בִּמְנָחוֹת הַנִּתָּר לַטְּהוֹרִין חַיָּבִין עָלָיו מִשּׁוּם טֻמְאָה וְשֶׁאֵינוֹ כוּ' אֵין חַיָּבִין עָלָיו מִשּׁוּם טֻמְאָה. וְיָלְפִינַן לְהוּ מִקְּרָא. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} וְלֹא לְמִשְׁרְיֵהּ לַאֲכִילָה, כְּמוֹ שֶּׁכָּתַבְנוּ בְסוֹף פִּרְקִין: {יא} כְּגוֹן שֶׁשָּׁחֲטוּ שְׁנֵי כֹהֲנִים כְּאֶחָד. רַשִׁ"י: 

~ פרק א - משנה ג

{יב} וּצְרִיכֵי, דְּאִי אִתְמַר בְּקָדְשֵׁי קָדָשִׁים הֲוָה אֲמֵינָא בְּהָא קָאָמַר רַבִּי אֱלִיעֶזֶר כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} וְנִרְאֶה מֵהָרַמְבַּ"ם דְּאֵין צָרִיךְ כָּאן יָצָא מִקְצָת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} <B> מוֹעֲלִין כוּ'.</B> וְאַף עַל גַּב דְּאִיתַנְהוּ אַבְּרָאֵי. דְּאִי הֲדַר עַיְּלִינְהוּ, כֻּלֵּי עָלְמָא לֹא פְלִיגֵי. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} <B> וְחַיָּבִין כוּ'.</B> נִרְאֶה לִי דְלָא עֲדִיף מִבָּשָׂר הַיּוֹצֵא, דְּלֹא אָמַר רַבִּי עֲקִיבָא דִזְרִיקָה מוֹעֶלֶת לַאֲכִילָה. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ד

{טז} לְפִי שֶׁקָּדְשֵׁי קָדָשִׁים הַנֶּאֱמָרִים כָּאן הֵן הַחַטָּאת וְהָאָשָׁם שֶׁאוֹכְלִים אוֹתָן הַכֹּהֲנִים. אֲבָל הָעוֹלָה מוֹעֲלִין בִּבְשָׂרָהּ וַאֲפִלּוּ לְאַחַר זְרִיקָה. הָרַמְבַּ"ם: {יז} <B> וְאֵין כוּ'.</B> וְאַף עַל גַּב דְּאַף קֹדֶם זְרִיקָה אֵין בּוֹ מְעִילָה, קָתָנֵי לְמֵידַק הָא אִסּוּרָא אִיכָּא וּמַיְרֵי בְיוֹצֵא. וּלְמֵימְרָא דְלָא אָמַר רַבִּי עֲקִיבָא דִזְרִיקָה מוֹעֶלֶת לַיּוֹצֵא לְהַתִּירוֹ בַּאֲכִילָה. גְמָרָא:

~ פרק ב - משנה א

{א} <B> חַטַּאת כוּ'.</B> עַד הַשְׁתָּא אַיְרֵי בְדִין פְּסוּלֵי קָדָשִׁים מָתַי יִמְעֲלוּ בָהֶם. וּמִכָּאן וָאֵילָךְ אַיְרֵי בִּכְשֵׁרִים. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> מִשֶּׁהֻקְדְּשָׁה.</B> בַּפֶּה, אֲפִלּוּ אִם נֶהֱנָה מִמֶּנָּה מֵחַיִּים, כְּגוֹן שֶׁמְּכָרָהּ דְּהוֹצִיאָהּ לְחֻלִּין. רַשִׁ"י: {ג} <B> הֻכְשְׁרָה.</B> כָּל מַה שֶּׁאָמַר בְּזֶה הַפֶּרֶק הֶכְשֵׁר, רָצָה לוֹמַר הֲכָנָה לַדְּבָרִים שֶׁעֲתִידִים לִזְכֹּר אוֹתָם, לֹא עִנְיַן הֶכְשֵׁר טֻמְאָה הַנִּזְכָּר בְּכָל מָקוֹם. וְדַע זֶה, וְלֹא יַטְעֲךָ שִׁתּוּף הָעִנְיָן. הָרַמְבַּ"ם: {ד} <B> וּבְלִינָה.</B> בְּלִינַת דָּם בִּשְׁקִיעַת הַחַמָּה, אִי נַמִּי בְלִינַת בָּשָׂר לְיוֹם וָלַיְלָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב. וְלֹא קָתָנֵי וּבַיּוֹצֵא, דְּמַתְנִיתִין רַבִּי עֲקִיבָא הִיא וּלְדִידֵיהּ הַזָּיָה מַהֲנֵי לֵיהּ לַיּוֹצֵא, וּמִדְּלֹא פָסֵיק לֵיהּ לֹא קָתָנֵי: {ה} <B> וְאֵין כוּ'.</B> לְפִי שֶׁאֵין בְּחַטַּאת הָעוֹף אֵמוּרִים רַק כֻּלָּהּ לַכֹּהֲנִים כִּבְשַׂר חַטָּאת. הָרַמְבַּ"ם. וְעַיֵּן סוֹף פִּרְקִין דִּלְעֵיל: 

~ פרק ב - משנה ב

{ו} וְשׁוּב כֵּיוָן שֶׁנַּעֲשֵׂית מִצְוָתוֹ אֵין מוֹעֲלִין בּוֹ. רַשִׁ"י. כְּלוֹמַר דְּלֹא אִקְרוּ עוֹד קָדְשֵׁי ה'. וְעַד שֶׁתֵּצֵא, עַד שֶׁתֵּרָאֶה. גְּמָרָא. פֵּרֵשׁ רַשִׁ"י אַחַר שֶׁהֵרִים כוּ' אוֹתוֹ הַנִּשְׁאָר רָאוּי וְעוֹמֵד לְהוֹצָאַת הַדֶּשֶׁן: 

~ פרק ב - משנה ג

{ז} וְהוּא הַדִּין בִּשְׁאָר הָעֲבוֹדוֹת: {ח} <B> שֶׁיֻּתַּךְ.</B> גָּזוּר מִן תּוֹךְ. רְצוֹנוֹ לוֹמַר עַד שֶׁיֵּעָשֶׂה הַבָּשָׂר חָלוּל וְיִתְהַפְּכוּ חֲלָקָיו בָּאֵשׁ עַד שֶׁיִּדְמֶה לִסְפוֹג הַיָּם. וְעִנְיָן זֶה יִהְיֶה בַבָּשָׂר אַחַר שֶׁיִּשָּׂרֵף לְגַמְרֵי. הָרַמְבַּ"ם. וְעַיֵּן סוֹף פֶּרֶק י"ב דִּזְבָחִים. וְצָרִיךְ לוֹמַר דְּכָאן גָּרַס שֶׁיִּתֹּךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} וַהֲרֵי תְּרוּמַת הַדֶּשֶׁן שֶׁנַּעֲשֵׂית מִצְוָתוֹ וְיֶשׁ בּוֹ מְעִילָה כוּ', מִשּׁוּם דַּהֲווּ תְּרוּמַת הַדֶּשֶׁן וְעֶגְלָה עֲרוּפָה שְׁנֵי כְתוּבִין הַבָּאִין כְּאֶחָד וְאֵין מְלַמְּדִין. וּלְמַאן דְּאָמַר מְלַמְּדִים, תְּרֵי מִעוּטֵי הָעֲרוּפָה (הֵ"א קַמַּיְתָא) וְשָׂמוֹ (וָי"ו בַּתְרָאֵי), הָנֵי אִין מִידִי אַחֲרִינֵי לֹא. גְּמָרָא: 

~ פרק ב - משנה ד
