{א} <B>מַעֲרִיכִין כוּ'.</B> וְהָעֲרָכִין כֻּלָּן וְהַדָּמִים, הַכֹּל סְתָמָן לְבֶדֶק הַבַּיִת, וְיִפְּלוּ הַכֹּל לַלִּשְׁכָּה שֶׁהָיְתָה בַמִּקְדָּשׁ מוּכֶנֶת לְקָדְשֵׁי בֶּדֶק הַבַּיִת. הָרַמְבַּ"ם: {ב} וְנִרְאֶה שֶׁצָּרִיךְ לוֹמַר בֶּן שְׁתֵּים עֶשְׂרֵה וְיוֹם אֶחָד. וְכֵן הוּא בַּתּוֹסָפוֹת. וּכְדִתְנַן בְּמִשְׁנָה ו' פֶּרֶק ה' דְּנִדָּה. וְהָתָם בַּגְּמָרָא דָרְשִׁינַן לַהּ בְּמֻפְלָא הַסָּמוּךְ לְאִישׁ: {ג} נְפָשׁוֹת כָּל דְּהוּ. גְּמָרָא: {ד} וְתָנֵי וַהֲדַר מְפָרֵשׁ. גְּמָרָא. וְהוּא הַדִּין נַמִּי דְּמָצֵי לְמִנְקַט טֻמְטוּם כוּ' חֵרֵשׁ שׁוֹטֶה וְקָטָן וְנָכְרִי, אֶלָּא חֲדָא מִתַּרְתֵּי וּתְלַת נָקַט כוּ'. רַשִׁ"י. וְעַיֵּן עוֹד: {ה} הַאי קְרָא בְּמַעֲרִיךְ כְּתִיב, וּלְפִיכָךְ הָיָה נִרְאֶה לִי לְהַגִּיהַּ לְאַשְׁמְעִינַן דְּמַעֲרִיכִין. וְיֵשׁ לוֹמַר דְּלִישְׁנָא דְמִשְׁנָה ג' דִּלְקַמָּן נָקַט. וְעַיֵּן שָׁם: {ו} <B> וַעֲבָדִים.</B> אַף עַל גַּב דְּגַבֵּי מִצְוֹת יֵשׁ לוֹ דִּין אִשָּׁה, לְפִי עֶרֶךְ זָכָר הוּא נֶעֱרָךְ, דְּלֹא אִשְׁתְּמִיט תַּנָּא דְלֵימָא אִם הָיָה נֶעֱרָךְ כְּאִשָּׁה. תּוֹסָפוֹת: {ז} אִי נַמִּי כְגוֹן שֶׁנָּתְנוּ לָהֶם עַל מְנָת שֶׁאֵין לָרַב אוֹ לַבַּעַל רְשׁוּת בָּהֶן. תּוֹסָפוֹת:

{ח} וְאַף עַל גַּב דְּאִישׁ בְּמַעֲרִיךְ כְּתִיב, אִם אֵינוֹ עִנְיָן לַמַּעֲרִיךְ תְּנֵהוּ עִנְיָן לַנֶּעֱרָךְ. תּוֹסָפוֹת: {ט} וּתְמֵהַנִי דִּבְמִשְׁנָה ה' פֶּרֶק קַמָּא דִּשְׁקָלִים תְּנַן סְתָמָא, וְאֵין מְקַבְּלִין מִנָּכְרִי לְבֶדֶק הַבַּיִת, שֶׁכֵּן מְפֹרָשׁ עַל יְדֵי עֶזְרָא לֹא לָכֶם וְלָנוּ לִבְנוֹת בַּיִת לֵאלֹקֵינוּ [עֶזְרָא ד]. וּבַגְּמָרָא דְּהָכָא אָמְרִינַן דְּהִלְכָתָא כְרַבִּי מֵאִיר מִסְתַּבְּרָא דִּכְתִיב לֹא לָכֶם וְלָנוּ. וּבְנֻסְחַת אֶרֶץ יִשְׂרָאֵל שֶׁבְּפֵרוּשׁ הָרַמְבַּ"ם מָצָאתִי, וַהֲלָכָה כְרַבִּי מֵאִיר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} <B> שֶׁנּוֹדְרִין.</B> כָּתְבוּ הַתּוֹסָפוֹת, וְצָרִיךְ עִיּוּן דְּהָא לְרַבִּי מֵאִיר אֵין נָכְרִים מִתְנַדְּבִים לְבֶדֶק הַבַּיִת כוּ'. וְשֶׁמָּא הָדַר בֵּיהּ רָבָא מֵהָא דְאָמַר דַּהֲלָכָה כְּרַבִּי מֵאִיר מִקְּרָא דְלֹא לָכֶם וְגוֹ'. אֲבָל הָרַמְבַּ"ם כָּתַב, נָכְרִי שֶׁאָמַר דָּמַי כוּ' נוֹתֵן כְּפִי נִדְרוֹ וְאֵינוֹ נוֹפֵל לַלִּשְׁכָּה, שֶׁאֵין מְקַבְּלִין [מִן הַנָּכְרִי נְדָבָה אוֹ נֶדֶר] לְחַזֵּק אֶת בֶּדֶק הַבַּיִת אוֹ בֶּדֶק יְרוּשָׁלַיִם שֶׁנֶּאֱמַר לֹא לָכֶם וְגוֹ'. וּמַה יֵּעָשֶׂה בָּהֶן, יִבָּדֵק הַנָּכְרִי עַל דַּעַת מִי נָדַר, אִם נָדַר עַל דַּעַת יִשְׂרָאֵל יוֹצִיאוּהוּ [בֵּית דִּין] בַּמֶּה שֶׁיֵּרָאֶה לָהֶן חוּץ מִבֶּדֶק הַבַּיִת וִירוּשָׁלַיִם, וְאִם אָמַר לַשָּׁמַיִם נָדַרְתִּי, יִגָּנְזוּ:

{יא} <B> הַגּוֹסֵס.</B> יָדוּעַ. וְהוּא שֶׁקּוֹל גְּרוֹנוֹ נִשְׁמָע בִּשְׁעַת הַמִּיתָה. הָרַמְבַּ"ם. וְנִרְאֶה לִי, שֶׁהוּא מִלְּשׁוֹן מְגִיסָה בִקְדֵרָה, שֶׁהַלֵּחָה מִתְהַפֵּךְ בִּגְרוֹנוֹ, כְּמוֹ הַמֵּגִיס בִּקְדֵרָה שֶׁמְּהַפֵּךְ מַה שֶּׁבַּקְּדֵרָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} אַף עַל גַּב דְּאָמַר בְּסַנְהֶדְרִין פֶּרֶק ו' שֶׁהָיָה אֶחָד רוֹכֵב עַל סוּס וְסוּדָרִין בְּיָדוֹ, מָצְאוּ לוֹ זְכוּת הָיָה הַלָּה מֵנִיף בַּסּוּדָרִין, מִכָּל מָקוֹם לֹא שְׁכִיחַ. תּוֹסָפוֹת: {יג} דְּהַיּוֹצֵא לֵהָרֵג עַל פִּי בֵּית דִּין הוּא עִנְיָן שֶׁאֵינוֹ תָלוּי בִּרְצוֹנֵנוּ אֶלָּא הַתּוֹרָה מְמִיתָה אוֹתוֹ, אֲבָל הַמֶּלֶךְ לִפְעָמִים מִתְמַלֵּא רַחֲמִים וּמוֹחֵל. הָרַמְבַּ"ם: {יד} וְצָרִיךְ עִיּוּן מְנָלָן לוֹמַר מִקְּרָא גַּבֵּי מַעֲרִיךְ שֶׁלֹּא יְהֵא נֶעֱרָךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {טו} מִי שֶׁהוּא רָאוּי לֵחָרֵם וְלֵהָרֵג אֵינוֹ בִכְלַל פִּדְיוֹן. רַשִׁ"י: {טז} אֲבָל בְּגוֹסֵס מוֹדֶה רַבִּי חֲנִינָא. רַשִׁ"י: {יז} וּבְאִם הִקְדִּישׁ לֹא פְלִיגֵי שֶׁמִּיָּד שֶׁהִקְדִּישׁ חָל הַהֶקְדֵּשׁ, וְלֹא דָמְיָא לְמִלְוָה. אֲבָל יֵשׁ לְהַקְשׁוֹת אַמַּאי לֹא פְלִיגֵי בְּאִם נָדַר וְהֶעֱרִיךְ. וְשֶׁמָּא בְּנוֹדֵר וּמַעֲרִיךְ יָחוּל מִיָּד. תּוֹסָפוֹת: {יח} תְּמֵהַנִי מַאי תַּקָּנָה אִיכָּא, הָא קַיְמָא לָן כְּרַבִּי מֵאִיר וְרַבִּי יְהוּדָה בְּמִשְׁנָה ג' פֶּרֶק ח' דִּבְכוֹרוֹת דְּמִלְוָה עַל פֶּה גוֹבֶה מִן הַיּוֹרְשִׁים, וְאִם כֵּן הֲוָה לֵיהּ לְמֵימַר אֲבָל הָאִידְנָא קַיְמָא לָן הֲלָכָה כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: 

~ פרק א - משנה ד

{יט} <B> יָשְׁבָה כוּ'.</B> פֵּרֵשׁ רַשִׁ"י, קֹדֶם שֶׁנִּגְמַר דִּינָהּ. וְלֵיתָא, דְּאַיּוֹצֵא לֵהָרֵג קָאֵי, וְכָל הַיּוֹצֵא לֵהָרֵג לְאַחַר גְּמַר דִּין. וְהַיְנוּ טַעֲמָא דְרֵישָׁא דְאֵין מַמְתִּינִין מִשּׁוּם עִנּוּי הַדִּין, דְּלֵיכָּא עִנּוּי הַדִּין אֶלָּא לְאַחַר גְּמַר דִּין. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כ} גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י מֵת אָסוּר בַּהֲנָאָה, דִּכְתִיב וַתָּמָת שָׁם מִרְיָם [בְּמִדְבָּר כ], וְגָמַר שָׁם שָׁם מֵעֶגְלָה עֲרוּפָה. וְכֵיוָן שֶׁהַמֵּת אָסוּר בַּהֲנָאָה, כָּל נוֹיָיו מִתַּסְרֵי נַמִּי בַהֲנָאָה. וְהָא דִתְנַן שֶׁנֶּהֶרְגָה, רְבוּתָא קָא מַשְׁמַע לָן דִּגְמַר דִּינָהּ אוֹסַרְתָּהּ קֹדֶם שֶׁתָּמוּת. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כא} <B> אֲסוּרָה כוּ'.</B> כִּדְקַיְמָא לָן וּבַעַל הַשּׁוֹר נָקִי, כְּאָדָם שֶׁאוֹמֵר לַחֲבֵרוֹ יָצָא פְלוֹנִי נָקִי מִנְּכָסָיו וְאֵין לוֹ בָהֶן הֲנָיָה שֶׁל כְּלוּם. רַשִׁ"י: {כב} וְתֵימַהּ, מַאי אֲפִלּוּ, דְּהָא בְאִשָּׁה נַמִּי שְׂעָרָהּ מִתְּסַר. וְהָרַמְבַּ"ם לֹא סִיֵּם אֲפִלּוּ שְׂעָרָהּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} וְכֵן לְקַמָּן כָּתַב עָנִי בֶּן עֶשְׂרִים שֶׁהֶעֱרִיךְ עַצְמוֹ כוּ'. וְלָאו דַּוְקָא, דְּהוּא הַדִּין הֶעֱרִיךְ לְאַחֵר, אֶלָּא חֲדָא מִתַּרְתֵּי נָקַט: {ב} <B> פָּחוֹת כוּ'.</B> לָאו דַּוְקָא פָּחוֹת מִסֶּלַע, אֶלָּא אֲפִלּוּ נָתַן כָּל הָעֵרֶךְ חוּץ מִסֶּלַע אֶחָד מְגַלְגְּלִין עָלָיו אֶת הַכֹּל. תּוֹסָפוֹת. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ג} <B> אֵין כוּ'.</B> הָא תוּ לָמָּה לִי, הָא קָא מַשְׁמַע לָן פָּחוּת מִסֶּלַע הוּא דְלֵיכָּא, הָא יָתֵר עַל סֶלַע אִיכָּא [אַף עַל גַּב דְּלֵיכָּא חֲמִשִּׁים סְלָעִים, דְּאִם הָיוּ בְיָדוֹ חֲמִשָּׁה סְלָעִים נוֹתֵן כֻּלָּם, דְּהַיְנוּ יָתֵר עַל סֶלַע וּפָחוּת מֵחֲמִשִּׁים]. יָתֵר עַל חֲמִשִּׁים הוּא דְלֵיכָּא, הָא פָּחוּת מֵחֲמִשִּׁים אִיכָּא [כְּדַאֲמַרַן]. וּסְתָמָא כְּרַבָּנָן. גְּמָרָא: {ד} [וּמְסַיֵּם] וּצְרִיכָה שִׁבְעָה נְקִיִּים וְקָרְבָּן, פֵּרוּשׁ עוֹלָה בָּאָה עַל תְּנַאי, וְהַחַטָּאת אֵינָהּ נֶאֱכֶלֶת. וְזֶה לְשׁוֹן הָרַמְבַּ"ם, וְאִם אָמְרָה שְׁנֵי יָמִים רָאִיתִי, נֹאמַר כָּאן פִּתְחָהּ שִׁבְעָה עָשָׂר, לְפִי שֶׁאָנוּ {א} וְנִקְרָא בִּלְשׁוֹן אַשְׁכְּנַז עלינבוגי"ן: {ב} וְאַף עַל פִּי שֶׁאֵין רְאָיָה לַדָּבָר זֵכֶר לַדָּבָר, שֶׁנֶּאֱמַר [יְחֶזְקֵאל כג] אֲשֶׁר בְּשַׂר חֲמוֹרִים בְּשָׂרָם. בַּגְּמָרָא: {ג} פֵּרוּשׁ לְפֵרוּשׁוֹ, שֶׁבַּתְּחִלָּה אוֹמְדִין לִקַּח כְּשִׁעוּר הַיָּד כוּ', וּמְמַלְּאִין בּוֹ הֶחָבִית דְּשֶׁמָּא לֹא כִוְּנוּ הָאֹמֶד וְיִצְטָרֵךְ לְהוֹסִיף אוֹ לִגְרֹעַ, וּלְפִיכָךְ שׁוֹקְלִין אַחַר כָּךְ כְּשֶׁכְּבָר נִתְמַלֵּא הֶחָבִית מִמַּה שֶּׁהָיָה חָסֵר עַל יְדֵי הַיָּד, שֶׁעַכְשָׁיו אָנוּ יוֹדְעִים שֶׁלָּקַחְנוּ הַשִּׁעוּר בְּשָׁוֶה. וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם, שֶׁיִּטֹּל כוּ' וְיִתֵּן לְתוֹךְ הֶחָבִית כוּ' וְאַחַר כָּךְ שׁוֹקֵל. וְיִתָּכֵן שֶׁמְּפָרֵשׁ וְשׁוֹקֵל דְּמַתְנִיתִין מִלְּשׁוֹן שְׁקַל קָלָא פְּתַק בֵּיהּ שֶׁעִנְיָנוֹ נְטִילָה. אֲבָל הַמִּשְׁקָל לֹא תְנֵינָא לֵיהּ בְּמַתְנִיתִין אֶלָּא דְמִמֵּילָא הָכִי עָבְדִינַן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ד} הָרַמְבַּ"ם. וְקָשֶׁה לִי, דְּהַךְ קְרָא לֹא כְתִיב אֶלָּא בְּאִם מָךְ הוּא מֵעֶרְכֶּךָ, וְאִם כֵּן כְּשֶׁאֵינוֹ מָךְ לֹא בָעֵי הַעֲמָדָה לִפְנֵי הַכֹּהֵן כְּלָל. וּבַגְּמָרָא, יִתְּנוּ הַיּוֹרְשִׁים, שְׁמַע מִנַּהּ מִלְוָה עַל פֶּה גּוֹבֶה מִן הַיּוֹרְשִׁים כוּ'. הָכָא בְמַאי עָסְקִינַן כְּשֶׁעָמַד בַּדִּין. פֵּרֵשׁ רַשִׁ"י, וְחִיְּבוּהוּ קֹדֶם מוֹתוֹ דַּהֲוָה לֵיהּ כְּמִלְוָה בִּשְׁטָר, וְכָתַב הָרַאֲבַ"ד, עֲמִידָה בַדִּין שֶׁהֻזְכְּרָה בַּגְּמָרָא לֹא בָאוּ לָהּ מִדֶּרֶךְ הַעֲמָדָה וְהַעֲרָכָה, שֶׁהֲרֵי רָאוּי הָיָה לָזֶה בִּשְׁעַת הַנֶּדֶר כוּ', אֲבָל הֻזְכְּרָה לְחִיּוּב הַיּוֹרְשִׁים מִפְּנֵי שֶׁהוּא מִלְוָה עַל פֶּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} <B> שֶׁאֵין כוּ'.</B> [נִרְאֶה לִי דְּטַעֲמָא לְפִי שֶׁהַמֵּת אָסוּר בַּהֲנָאָה, וּמִשּׁוּם הָכִי אֵין אוֹמְרִים כָּךְ בִּבְהֵמָה]. וְהָא דְּתַנְיָא בְּבָבָא קַמָּא וְנָתַן פִּדְיוֹן נַפְשׁוֹ דְמֵי נִזָּק [פֵּרוּשׁ שֶׁשָּׁמִין אוֹתוֹ לְאַחַר מִיתָה כַּמָּה הָיָה שָׁוֶה מֵחַיִּים. תּוֹסָפוֹת], הָתָם הוּא טַעֲמָא מִשּׁוּם דְּמִשַּׁעְתָּא דְאַזִּיק רָמֵי עָלֵיהּ לְשִׁלּוּמֵי כִשְׁעַת הַהֶזֵּק, וְהַהִיא שַׁעְתֵּיהּ דְּאַזְּקֵיהּ חַי הוּא, אֲבָל דָּמַי עָלָי שֶׁאֵינוֹ נוֹדֵר אֶלָּא מַה שֶּׁיְּשׁוּמוּהוּ בֵית דִּין, וַהֲרֵי לֹא בָא לִידֵי כָךְ שֶׁהֲרֵי מֵת, לֹא יִתְּנוּ הַיּוֹרְשִׁין. רַשִׁ"י: {ו} <B> לֹא כוּ'.</B> בַּגְּמָרָא יָלֵיף דַּהֲוָה לֵיהּ לִכְתֹּב בְּעֵרֶךְ וּכְתִיב בְּעֶרְכְּךָ, עֵרֶךְ כֻּלּוֹ הוּא נוֹתֵן וְלֹא עֵרֶךְ אֵבָרִים. וְקֻשְׁיַת רַשִׁ"י בַּחֻמָּשׁ לָמָּה הֻכְפַּל כַּמָּה זִמְנֵי בַּלָּשׁוֹן הַזֶּה, עַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ז} וְאִפְּכָא לֵיכָּא לְאַקוּשֵׁי לְחַיֵּב לַמַּעֲרִיךְ אֲפִלּוּ אוֹמֵר עֵרֶךְ יָדִי עָלָי, דְּהָא מְמַעֲטִינַן מִדִּכְתִיב עֶרְכְּךָ וְלֹא כְתִיב עֵרֶךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} וְכֵן כָּתַב הָרַמְבַּ"ם. וְלֹא אֵדַע לָמָּה שִׁנָּה מִלְּשׁוֹן הַמִּשְׁנָה דִּלְעֵיל עֵרֶךְ רֹאשִׁי כוּ': {ט} <B> עֵרֶךְ כֻּלּוֹ.</B> הָרַמְבַּ"ם. דִּכְמוֹ דְזֶה הַכְּלָל לַאֲתוֹיֵי אוֹמֵר דְּמֵי חֲצִי אֵבָר, הָכִי נַמִּי מַיְתִינַן לְאוֹמֵר עֵרֶךְ חֲצִי אֵבָר כוּ'. אֲבָל הָרַ"ב דְּלֹא כָתַב אֶלָּא אוֹמֵר דְּמֵי חֶצְיִי גָּרַס דְּמֵי כֻלּוֹ. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{יא} וְתָנֵי קָדְשֵׁי מִזְבֵּחַ וְתָנֵי קָדְשֵׁי בֶדֶק הַבַּיִת: {יב} אַסֵּיפָא קָאֵי. וּבְדִין הֲוָה לֵיהּ לְהַקְדִּים הַדִּבּוּר שֶׁאַחֲרָיו מֵת הַשּׁוֹר כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יג} <B> עָלָי.</B> בַּגְּמָרָא לֹא שָׁנוּ אֶלָּא דְאָמַר דְּמֵי כוּ', אֲבָל אָמַר שׁוֹר זֶה כוּ', אַף עַל גַּב דְּאָמַר עָלָי, כֵּיוָן דְּאָמַר זֶה וָמֵת, אֵינוֹ חַיָּב בְּאַחֲרָיוּתוֹ, עָלָי לְהָבִיאוֹ קָאָמַר, פֵּרֵשׁ רַשִׁ"י, עָלָי לְהָבִיאוֹ, לִטְרֹחַ עַד שֶׁיִּתְקָרֵב. אֲבָל כִּי אָמַר דְּמֵי, אִם אָבַד הַשּׁוֹר הַדָּמִים לֹא אָבְדוּ, דְּהָא לֹא אָמַר דָּמִים הַלָּלוּ וְלֹא הָיוּ בְעֵין. וְרֵישָׁא נַמִּי דְאָמַר דְּמֵי. דְּאִם לֹא כֵן, אֲפִלּוּ אָמַר עָלָי. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{כב} נִרְאֶה דִמְפָרֵשׁ לְתַנָּא קַמָּא דְאַיְרֵי בַלְוִיִּם, וְרַבִּי אֱלִיעֶזֶר לְפָרוֹשֵׁי אֲתָא. אֲבָל הָרַמְבַּ"ם פֵּרֵשׁ דְּהַתַּנָּא קַמָּא אַיְרֵי בְעָרֵי יִשְׂרָאֵל. וְכֵן פֵּרֵשׁ רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כג} <B> עוֹשִׂין כוּ'.</B> כְּשֶׁיֵּשׁ נוֹי לָעִיר הַרְבֵּה קוֹפְצִים לָדוּר בְּתוֹכָהּ וּמִתְיַשֶּׁבֶת: {כד} אֲבָל לֹא בִשְׁנַת הַיּוֹבֵל עַצְמָהּ. דְּהָא אִיכָּא קַל וָחֹמֶר, וּמַה מְּכוּרָה כְבָר יוֹצְאָה עַכְשָׁיו שֶׁאֵינָהּ מְכוּרָה אֵינוֹ דִין שֶׁלֹּא תִמָּכֵר. כֶּסֶף מִשְׁנֶה: {כה} <B> לַלְוִיִּם.</B> וְכֹהֲנִים נַמִּי בְנֵי לֵוִי הֵם. כְּדִכְתִיב וְהַכֹּהֲנִים הַלְוִיִּם בְּנֵי צָדוֹק [יְחֶזְקֵאל מד]:
