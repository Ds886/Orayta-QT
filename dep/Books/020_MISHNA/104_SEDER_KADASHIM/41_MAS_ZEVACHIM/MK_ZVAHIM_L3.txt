{א} <b>הַזְּבָחִים.</b> שֵׁם אֶחָד כּוֹלֵל כָּל הַקָּרְבָּנוֹת הַקְּרֵבִים מִבַּעֲלֵי חַיִּים שֶׁנּוֹהֲגִים בּוֹ לְדוֹרוֹת כַּאֲשֶׁר צִוָּנוּ הַשֵּׁם. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> שֶׁנִּזְבְּחוּ.</B> עֲבוֹדָה קַמַּיְתָא נָקַט, וְהוּא הַדִּין קַבָּלָה וְהוֹלָכָה וּזְרִיקָה כִּדְתָנֵי סֵיפָא. תּוֹסָפוֹת: {ג} וְלֶאֱכֹל אֶת הַנֶּאֱכָלִים. רַשִׁ"י: {ד} וּלְשׁוֹן הָרַמְבַּ"ם, שֶׁשָּׁחַט שֶׁלֹּא לִשְׁמָן אוֹ שֶׁלֹּא לְשֵׁם בְּעָלִים: {ה} דָּיֵק לֵיהּ הָרַמְבַּ"ם מִדִּתְנַן לַבְּעָלִים: {ו} בַּגְּמָרָא, וּמִשּׁוּם דְּלֵב בֵּית דִּין מַתְנֶה עֲלֵיהֶן: {ז} לָאו דְּמַשְׁמַע לֵיהּ מִוְּלָקַח. אֶלָּא מַיְתֵי לֵיהּ בַּגְּמָרָא מִוְּכִפֵּר עָלָיו הַכֹּהֵן מֵחַטָּאתוֹ, שֶׁתְּהֵא כַפָּרָה לְשֵׁם חַטָּאת: {ח} הָכִי דָּרֵישׁ לֵיהּ, אֵין דָּבָר אַחֵר מְחַלֵּל אֶת קָדְשֵׁי בְנֵי יִשְׂרָאֵל. רַשִׁ"י: {ט} וּלְבֶן בְּתֵירָא דְּמִשְׁנָה ג' דְּפָסְקוּ הָרַמְבַּ"ם וְהָרַ"ב כְּמוֹתוֹ אֲפִלּוּ קֹדֶם חֲצוֹת נַמִּי. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {י} וּלְרַבָּנָן לְמַאי הִלְכָתָא אִתְקַשׁ אָשָׁם לְחַטָּאת, לוֹמַר לְךָ, מַה חַטָּאת טְעוּנָה סְמִיכָה, דִּכְתִיב בָּהּ וְסָמַךְ יָדוֹ עַל רֹאשׁ הַחַטָּאת, אַף אָשָׁם טָעוּן סְמִיכָה. גְּמָרָא: {יא} <B> עַל חֵטְא.</B> אִיכָּא חַטָּאת וְאָשָׁם נַמִּי דְּאֵין בָּאִין עַל חֵטְא, חַטַּאת יוֹלֶדֶת וַאֲשַׁם מְצֹרָע. אֶלָּא מִשּׁוּם דְּרֻבָּא אָתוּ עַל חֵטְא, מַשְׁוֶה לְהוּ אַהֲדָדֵי. תּוֹסָפוֹת:

{יב} <B> לְשֵׁם פֶּסַח כוּ'.</B> דְּמָצִינוּ מוֹתַר פֶּסַח בָּא שְׁלָמִים, וְאֵין מוֹתַר שְׁלָמִים בָּא פֶסַח, מַה פֶּסַח שֶׁמּוֹתָרָהּ בָּא שְׁלָמִים, שְׁחָטָהּ לְשֵׁם שְׁלָמִים פָּסוּל, שְׁלָמִים שֶׁאֵין מוֹתָרָן בָּא פֶסַח, אִם שְׁחָטָן לְשֵׁם פֶּסַח אֵינוֹ דִין שֶׁיְּהוּ פְסוּלִים. וְכֵן מָצִינוּ בְּחַטָּאת שֶׁמּוֹתָרָהּ בָּא עוֹלָה, וְאֵין מוֹתַר עוֹלָה בָּא חַטָּאת. גְּמָרָא: {יג} <B> כְּשֵׁרִים.</B> בַּגְּמָרָא בָּעֵי רַבִּי זֵירָא אִי כְשֵׁרִים וְעוֹלִין קָאָמַר, וּבְתַרְתֵּי פְלִיגֵי, אִי כְשֵׁרִים וְלֹא עוֹלִין קָאָמַר, וּבְגָבוֹהַּ מוֹדֶה לְרַבָּנָן וּבְנָמוּךְ פָּלֵיג, דִּבְנָמוּךְ פְּסוּלִים לְגַמְרֵי:

{יד} וְלֹא דִסְבִירָא לֵיהּ דְּכָל הַיּוֹם כָּשֵׁר וּמִשּׁוּם הָכִי פּוֹסֵל שֶׁלֹּא לִשְׁמוֹ, אֶלָּא אִיהוּ נַמִּי פּוֹסֵל פֶּסַח שֶׁנִּשְׁחַט קֹדֶם חֲצוֹת, כְּדִתְנַן בְּמִשְׁנָה ג' פֶּרֶק ה' דִפְסָחִים. גְּמָרָא: {טו} <B> מְקֻבָּל.</B> גְּמָרָא גְּמִירֵי לֵיהּ וְלֵית לֵיהּ טַעֲמָא בְגַוֵּיהּ. גְּמָרָא:

{טז} שְׁמַע מִנַּהּ דְּכָל הַכָּשֵׁר בְּזָר לָאו עֲבוֹדָה הִיא. וְקַשְׁיָא שְׁחִיטָה דִּכְשֵׁרָה בְזָר כְּדִתְנַן בְּסוֹף פֶּרֶק ג' כוּ'. וְיֵשׁ לוֹמַר דְּשָׁאנֵי שְׁחִיטָה, דִּבְהַנְהוּ קְרָאֵי דְּמִעֲטֵיהּ רַחֲמָנָא שֶׁלֹּא לִשְׁמָהּ בְּפֶסַח וְחַטָּאת, זְבִיחָה וּשְׁחִיטָה כְּתִיב בְּהוּ בְּהֶדְיָא:

{א} <B> זָר.</B> הָא דְּלֹא תָנֵי בַעַל מוּם וּשְׁתוּיֵי יַיִן וּפְרוּעֵי רֹאשׁ, מִשּׁוּם דִּכְתִיב בְּהוּ חִלּוּל בְּהֶדְיָא. תּוֹסָפוֹת: {ב} דִּבְנֵי יִשְׂרָאֵל לְמַעוֹטֵי מַאי כוּ', אֶלָּא הָכִי קָאָמַר וְיִנָּזְרוּ הַכֹּהֲנִים, יִתְרַחֲקוּ מִטֻּמְאָה. וּבְנֵי יִשְׂרָאֵל אַף הֵם יִנָּזְרוּ [לִמֵּד עַל הַזָּר] כוּ'. גְּמָרָא: {ג} שֶׁכָּךְ כָּתוּב בַּתּוֹרָה וַתִּקְרֶאנָה אוֹתִי כָּאֵלֶה וְאָכַלְתִּי חַטָּאת הַיּוֹם. הָרַמְבַּ"ם. וּמִדְּאָסוּר בַּאֲכִילָה יָלְפִינַן בְּקַל וָחֹמֶר שֶׁאִם עָבַד חִלֵּל: {ד} <B> טְבוּל יוֹם.</B> דְּשֶׁרֶץ וּדְמֵת וּשְׁאָר טֻמְאוֹת שֶׁלֹּא הֻצְרְכוּ לְקָרְבָּן. רַשִׁ"י: {ה} וּכְלוֹמַר וְטָהֵר, מִכְּלָל שֶׁהוּא טָמֵא. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ו} <B> מְחֻסַּר בְּגָדִים.</B> דְּאָמַר קְרָא וְחָגַרְתָּ אוֹתָם אַבְנֵט וְגוֹ' וְהָיְתָה לָהֶם כְּהֻנָּה לְחֻקַּת עוֹלָם, בִּזְמַן שֶׁבִּגְדֵיהֶם עֲלֵיהֶם כְּהֻנָּתָם עֲלֵיהֶם. גְּמָרָא: {ז} דִּכְתִיב וְעָרְכוּ בְנֵי אַהֲרֹן הַכֹּהֲנִים אֶת הַנְּתָחִים, הַכֹּהֲנִים בְּכִהֻנָּן, מִכָּאן לְכֹהֵן הֶדְיוֹט שֶׁלָּבַשׁ בִּגְדֵי כֹהֵן גָּדוֹל וְעָבַד, עֲבוֹדָתוֹ פְסוּלָה. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ח} כְּלוֹמַר כָּפוּף, שֶׁמֵּנִיחַ יָדָיו כוּ' וּלְפִיכָךְ אֵינוֹ עוֹמֵד. אֲבָל יוֹשֵׁב לֹא, כִּדְאִיתָא בַּגְּמָרָא, דִּכְתִיב אוֹ בְגִשְׁתָּם אֶל הַמִּזְבֵּחַ לְשָׁרֵת, פֵּרֵשׁ רַשִׁ"י אִתְקוּשׁ קִדּוּשׁ לְשֵׁרוּת: {ט} <B> עָרֵל.</B> כֹּהֵן שֶׁמֵּתוּ אֶחָיו מֵחֲמַת מִילָה. רַשִׁ"י: {י} דְּמִלְּתָא דִפְשִׁיטָא הִיא דְּבֶן נֵכָר שֶׁעָבַד עֲבוֹדָתוֹ פְסוּלָה, וְאֵין צָרִיךְ לַקְּרָא. וְכֵן כָּתַב הָרַמְבַּ"ם. אֲבָל בַּגְּמָרָא יָלֵיף לְהוּ מִקְּרָאֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} <B> טָמֵא.</B> זוֹ וְאֵין צָרִיךְ לוֹמַר זוֹ קָתָנֵי, דְּהָא כְּבָר תָּנֵי טְבוּל יוֹם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} לַעֲמֹד לְשָׁרֵת, שֶׁהֲרֵי דָּוִד קִדְּשָׁהּ, כְּדִכְתִיב כִּי קִדֵּשׁ דָּוִד וְכוּ', וּכְלֵי שָׁרֵת מְקַדְּשִׁין, דִּכְתִיב וַיִּמְשָׁחֵם וַיְקַדֵּשׁ אוֹתָם [בְּמִדְבָּר ז]. רַשִׁ"י: {יג} בֵּין כֹּהֵן לַכְּלִי שָׁרֵת בְּקַבָּלַת הַדָּם, כְּדִכְתִיב וְלָקַח הַכֹּהֵן, שֶׁתְּהֵא לְקִיחָה בְּעַצְמוֹ שֶׁל כֹּהֵן. רַשִׁ"י: {יד} <B> רַגְלֵי כוּ'.</B> דְּעַל גַּבֵּי אָדָם לֹא שְׁכִיחַ כְּמוֹ עַל גַּבֵּי בְּהֵמָה: {טו} וַחֲכָמִים לְמֵדִין יָד מֵאֶצְבַּע. הָרַמְבַּ"ם: {טז} <B> הַכֶּבֶשׁ.</B> שֶׁאֵינוֹ מְקוֹם מַתַּן דָּמִים. אוֹ בַּמִּזְבֵּחַ שֶׁלֹּא כְנֶגֶד הַיְסוֹד. רַשִׁ"י: {יז} אִי הָכִי בָשָׂר נַמִּי. אָמַר קְרָא לְכַפֵּר, לְכַפָּרָה נְתַתִּיו וְלֹא לְדָבָר אַחֵר. גְּמָרָא. וְעַיֵּן מִשְׁנָה ב' פֶּרֶק ג', דְּאַסֵּיפָא הַנִּתָּנִין לְמַטָּה כוּ' וְנִתָּנִים בִּפְנִים נַמִּי קָאֵי: {יח} אוֹ אֶל מִבֵּית קֹדֶשׁ הַקֳּדָשִׁים, כְּמוֹ שֶׁיִּתְבָּאֵר בְּפֶרֶק ה' מִן הַמַּסֶּכְתָּא הַזֹּאת. הָרַמְבַּ"ם: {יט} וּבַגְּמָרָא, פְּשִׁיטָא דְּאֵין בּוֹ כָּרֵת, דְּכִי כְתִיב כָּרֵת בְּאוֹכֵל מִפִּגּוּל שֶׁנִּתְפַּגֵּל בְּמַחֲשָׁבָה, כְּדִלְקַמָּן, וּמְשָׁנִינַן דְּהָכִי קָאָמַר, אִם נָתַן בְּמַחֲשָׁבָה דְחוּץ לִזְמַנּוֹ פָּסוּל וְאֵין בּוֹ כָּרֵת, דְּמַחֲשֶׁבֶת פִּגּוּל לֹא חָיְלָא, הוֹאִיל וְלֹא שָׁרִינַן בָּשָׂר בַּאֲכִילָה:

{יח} <B> הַזּוֹרֵק כוּ'.</B> נַפְקָא לֵיהּ בַּגְּמָרָא [דַּף קז] מִקְּרָא: {יט} <B> חַיָּב.</B> וְלֹא פָלֵיג רַבִּי אֱלִיעֶזֶר, דַּאֲפִלּוּ בְּחַטָּאוֹת הַפְּנִימִיּוֹת אִם נִשְׁפַּךְ הַדָּם אַחַר שֶׁנָּתַן מַתָּנָה אַחַת, מְהַנְּיָא הַךְ דְּיָהֲבָא, וּמֵבִיא פַּר אַחֵר וּמַתְחִיל. רַשִׁ"י: {כ} וְכֵן פָּסַק הָרַמְבַּ"ם. וּתְמֵהַנִי, דְּמַה לִּי מְפֹרָשׁ מִדְּאוֹרַיְתָא מַה לִּי הֲלָכָה לְמֹשֶׁה מִסִּינַי, שֶׁהֲרֵי שְׁתַּיִם זוּ שָׁמַעְנוּ מִפִּי הַגְּבוּרָה וְלָמָּה לֹא יִהְיֶה חַיָּב עַל מַה שֶּׁהוּא הֲלָכָה לְמֹשֶׁה מִסִּינַי כְּמוֹ עַל מַה שֶּׁהוּא מִדְּאוֹרַיְתָא. וְתוּ, דְּבַגְּמָרָא מַסִּיק דְּטַעְמֵיהּ דְּרַבִּי אֶלְעָזָר דַּהֲלָכָה לְמֹשֶׁה מִסִּינַי הִיא. וְהַשְׁתָּא קָשֶׁה לִי דִּבְמִשְׁנָה ט' פֶּרֶק ד' דְּסֻכָּה כָּתַבְתִּי דְּמַסְקָנָא דִגְמָרָא דַּהֲלָכָה לְמֹשֶׁה מִסִּינַי הִיא, וְאִם כֵּן אַמַּאי אֵין הֲלָכָה כְּרַבִּי אֶלְעָזָר כוּ'. וּלְפִי זֶה יִרְאֶה לִי דִּבְשִׁעוּרָא פָלֵיג רַבִּי אֶלְעָזָר דִּסְבִירָא לֵיהּ דְּאֵין צָרִיךְ שִׁעוּר כְּלָל, וְכִדְאִיתָא בַגְּמָרָא, דְּמֵי הֶחָג מַשְׁמַע אֲפִלּוּ פָּחוֹת מִשְּׁלֹשָׁה לֻגִּין. וּלְפִי דְקַיְמָא לָן כְּמַאן דְּאָמַר בְּסֻכָּה דְּבָעִינַן שְׁלֹשָׁה לֻגִּין, לְכָךְ פָּסַק הָרַמְבַּ"ם דְּלֹא כְרַבִּי אֶלְעָזָר דְּהָכָא. וְכֵן פָּסַק בְּחִבּוּרוֹ, הַמְנַסֵּךְ שְׁלֹשָׁה לֻגִּין חַיָּב, פָּחוֹת מִכֵּן פָּטוּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כא} כְּלוֹמַר דְּלֹא כִסְתָמָא דְמַתְנִיתִין רֵישׁ פֶּרֶק ה':

{יח} <B> נַחֲלָה.</B> שֶׁנִּתְקַיְּמָה קְדֻשָּׁתָהּ וַעֲמִידָתָהּ לְעוֹלָם, וְעָלֶיהָ אָמַר הַנָּבִיא [תְּהִלִּים צד] וְנַחֲלָתוֹ לֹא יַעֲזֹב, לְפִי שֶׁאָמַר בִּתְחִלַּת דְּבָרָיו שֶׁהַשֵּׁם בָּחַר בִּירוּשָׁלַיִם לִשְׁכִינָתוֹ וּבָחַר בְּיִשְׂרָאֵל לִסְגֻלָּתוֹ. אַחַר כָּךְ אָמַר שֶׁהַקָּדוֹשׁ בָּרוּךְ הוּא יִשְׁמֹר הָעָם הַזֶּה אֲשֶׁר בָּחַר בּוֹ בַּמָּקוֹם הַהוּא אֲשֶׁר בָּחַר בּוֹ. שֶׁנֶּאֱמַר [שָׁם קלב] כִּי בָחַר ה' בְּצִיּוֹן. הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יט} <B> הַקְּלָעִים.</B> לִפְנִים לְכָתְלֵי הָעֲזָרָה. הָרַמְבַּ"ם: {כ} <B> הַחוֹמָה.</B> שֶׁל יְרוּשָׁלַיִם, שֶׁהִיא בִּמְקוֹם מַחֲנֵה יִשְׂרָאֵל שֶׁבַּמִּדְבָּר, כְּדִכְתִיב [דְּבָרִים יד] וְאָכַלְתָּ שָׁם לִפְנֵי ה' אֱלֹהֶיךָ. רַשִׁ"י:

{כא} וּכְלוֹמַר דְּעִקַּר פָּרָשַׁת שְׁחוּטֵי חוּץ עַל אוֹתָן שֶׁבַּמִּדְבָּר נֶאֶמְרָה: {כב} הָכִי פֵּרְשׁוּ בַּבְּרַיְתָא, דְּכָל הָעִנְיָן שֶׁל אוֹתָהּ פָּרָשָׁה בִּתְחִלָּתָהּ עַד פָּסוּק וְלֹא יִזְבְּחוּ וְגוֹ' שֶׁעָנַשׁ כָּרֵת עַל הַשּׁוֹחֵט בַּחוּץ וְהַמַּעֲלֶה בַחוּץ, מְדַבֵּר בְּקָדָשִׁים שֶׁהִקְדִּישָׁן בִּשְׁעַת אִסּוּר בָּמוֹת וְהִקְרִיבָן בִּשְׁעַת אִסּוּר בָּמוֹת, דְּהַיְנוּ מִשֶּׁהוּקַם הַמִּשְׁכָּן. וּמִפָּסוּק וְלֹא יִזְבְּחוּ וּלְהַלָּן מְדַבֵּר בְּקָדָשִׁים שֶׁהִקְדִּישָׁן בִּשְׁעַת הֶתֵּר בָּמוֹת וּכְדִכְתִיב אֲשֶׁר הֵם זוֹבְחִים עַל פְּנֵי הַשָּׂדֶה, וְהָכִי מַשְׁמַע אֲשֶׁר הֵם זוֹבְחִים שֶׁהָיוּ עוֹמְדִים לִשָּׁחֵט בַּבָּמָה אִם הִקְרִיבָהּ בִּשְׁעַת אִסּוּר בָּמוֹת כְּאִלּוּ הִקְרִיב עַל פְּנֵי הַשָּׂדֶה. וְשָׁם נֶאֱמַר וֶהֱבִיאוּם לַה' מִצְוַת עֲשֵׂה. וְלֹא יִזְבְּחוּ עוֹד מִצְוַת לֹא תַעֲשֶׂה. יָכוֹל יְהֵא עָנוּשׁ כָּרֵת, תַּלְמוּד לוֹמַר זֹאת לָהֶם וְכוּ', זֹאת לָהֶם עֹנֶשׁ עֲשֵׂה וְלֹא תַעֲשֶׂה, וְלֹא אַחֶרֶת לָהֶם כָּרֵת: 

~ פרק יד - משנה י

{כג} פֵּרֵשׁ רַשִׁ"י וְזָבְחוּ זִבְחֵי שְׁלָמִים לַה' אוֹתָם, וְאוֹקְמִינָא לְהַאי קְרָא בְקָדָשִׁים שֶׁהִקְדִּישָׁן בִּשְׁעַת הֶתֵּר בָּמוֹת, וּכְתִיב אוֹתָם מִעוּטָא, אַלְמָא זְבָחִים הוּא דַּהֲווּ נְהִיגֵי בְהוּ, וְלֹא עוֹפוֹת וּמְנָחוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {כד} <B> הַזְּמָן.</B> דְּיָלֵיף בַּבְּרַיְתָא מִוְּזֹאת תּוֹרַת זֶבַח הַשְּׁלָמִים, לַעֲשׂוֹת זְמַן בְּבָמָה קְטַנָּה כִּזְמַן בָּמָה גְדוֹלָה: {כה} <B> וְהַנּוֹתָר.</B> נַמִּי מִפִּגּוּל יָלֵיף מִגְּזֵרָה שָׁוָה דְעָוֹן עָוֹן. רַשִׁ"י: {כו} וּתְמֵהַנִי לָמָּה לֹא פֵרְשׁוֹ כְּמוֹ בְּכָל מָקוֹם דְּמַתְנִיתִין פִּגּוּל נוֹתָר טָמֵא, דְּטָמֵא לְעִנְיַן אֲכִילָה. וְכֵן פֵּרֵשׁ רַשִׁ"י, הָכָא וְטָמֵא נַמִּי בְּהַהִיא פַּרְשְׁתָא כְתִיב, וְהַנֶּפֶשׁ אֲשֶׁר תֹּאכַל וְגוֹ':
