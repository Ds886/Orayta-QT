~ פרק א - משנה א

{א} <b>שׁוֹמְרִים כוּ'.</b> מִשּׁוּם דְּבָעֵי לְאוֹרוֹיֵי בְּמַעֲשֵׂה דְתָמִיד עֲבוֹדַת כֹּהֲנִים, נָקַט תְּחִלָּה שְׁמִירָתָן וּמְקוֹם שְׁכִיבָתָן עַד שֶׁהוּא מְסַדֵּר כָּל הַסֵּדֶר שֶׁתְּחִלָּה טוֹבְלִין כוּ'. (מְפָרֵשׁ הַמַּסֶּכְתָּא. וְאֵינוֹ פֵּרוּשׁ רַשִׁ"י): {ב} וְאַף עַל גַּב דִּקְרָא קָאָמַר וְהַחוֹנִים וְגוֹ' מֹשֶׁה וְאַהֲרֹן וּבָנָיו, הָכִי קָאָמַר וְהַחוֹנִים וְגוֹ' מֹשֶׁה, וַהֲדַר אַהֲרֹן וּבָנָיו. מִדִּכְתִיב וְהַחוֹנִים וּכְתִיב שׁוֹמְרֵי, חוֹנִים לְחוּד וְשׁוֹמְרִים לְחוּד. גְּמָרָא: {ג} <B> הַנִּיצוֹץ.</B> לֹא שָׁמַעְתִּי טַעַם לָמָּה נִקְרָא כֵן. מְפָרֵשׁ: {ד} וַאֲפִלּוּ הָיוּ אוֹתָן עֲלִיּוֹת מִקְצָתָן בָּעֲזָרָה, יְכוֹלִין כֹּהֲנִים לֵישֵׁב כְּשֶׁהָיוּ עֲיֵפִים, דְּגַגִּים וַעֲלִיּוֹת לֹא נִתְקַדְּשׁוּ. וּבָעֲזָרָה אֵין אָדָם יָכוֹל לֵישֵׁב, דְּאֵין יְשִׁיבָה בָעֲזָרָה אֶלָּא לְמַלְכֵי בֵּית דָּוִד. מְפָרֵשׁ: {ה} וְהֵם כֹּהֲנִים פְּחוּתִים מִשְּׁלֹשׁ עֶשְׂרֵה, וּלְפִי שֶׁעֲדַיִן לֹא הִגִּיעוּ לַעֲשׂוֹת עֲבוֹדָה עָבְדִינַן לְהוּ שׁוֹמְרִים, שֶׁאוֹתָן שֶׁיְּכוֹלִין לַעֲבֹד אֵינָן רוֹצִין לִשְׁמֹר. וְגַם אֵין מַנִּיחִין אוֹתָן לִשְׁמֹר. מְפָרֵשׁ: {ו} וּבִלְשׁוֹן אַשְׁכְּנַז גיוועל"ב: {ז} <B> יְשֵׁנִים כוּ'.</B> בְּחֶצְיָהּ שֶׁבָּחֹל: {ח} <B> בְּיָדָם.</B> כְּלוֹמַר, בִּידֵי אוֹתָם הַכֹּהֲנִים הַיְשֵׁנִים לְשָׁם הָיוּ מֻנָּחוֹת כָּל מַפְתְּחוֹת הָעֲזָרָה. וּמְפָרֵשׁ בְּמַסֶּכֶת מִדּוֹת סוֹף פֶּרֶק קַמָּא הֵיכָן הָיְתָה הַנָּחָתָן, מָקוֹם הָיָה שָׁם אַמָּה עַל אַמָּה. מְפָרֵשׁ: {ט} וּבַגְּמָרָא אָמְרִינַן, הָתָם קָרֵי לְהוּ רוֹבִים הָכָא קָרֵי לְהוּ פִרְחֵי כְהֻנָּה. אָמְרֵי, אִין, הָתָם דְּלֹא מְטוֹ לְמֶעֱבַד עֲבוֹדָה קָרֵי לְהוּ רוֹבִים, הָכָא דִמְטוֹ לְהוּ לְמֶעֱבַד עֲבוֹדָה קָרֵי לְהוּ פִּרְחֵי: {י} אֲבָל לֵהָנוֹת נִתְּנוּ אֲפִלּוּ שֶׁלֹּא בִּשְׁעַת עֲבוֹדָה. כְּמוֹ שֶׁכָּתַב הָרַ"ב בְּרֵישׁ פֶּרֶק ז' דְּיוֹמָא: {יא} <B> וְטָבַל.</B> וְהָא דִתְנַן בְּמַסֶּכֶת מְגִלָּה פֶּרֶק ב' מִשְׁנָה ד' לֹא מַזִּין וְלֹא טוֹבְלִין אֶלָּא בַיּוֹם, הַיְנוּ בְּחַיָּבֵי טְבִילָה שֶׁצְּרִיכִין סְפִירָה כְּמוֹ טְמֵא מֵת אוֹ זָב אוֹ מְצֹרָע כוּ'. וְנִדָּה וְיוֹלֶדֶת טְבִילָתָן בְּלֵיל שְׁמִינִי. מְפָרֵשׁ: {יב} עַיֵּן פֵּרוּשׁ הָרַ"ב בְּסוֹף פֶּרֶק קַמָּא דְּמִדּוֹת: 

~ פרק א - משנה ב

{יג} <B>וְדוֹפֵק.</B> מוֹרֶה עַל הַמַּקִּישׁ בַּפֶּתַח בְּנַחַת. כְּמוֹ קוֹל דּוֹדִי דוֹפֵק [שִׁיר הַשִּׁירִים ה]. הָרַמְבַּ"ם: {יד} <B> וְיָפִיס.</B> בְּכָאן בַּחֲצִי בֵּית הַמּוֹקֵד הַבָּנוּי בְּקֹדֶשׁ. דְּאֵין מַטִּילִין פַּיִס אֶלָּא בָעֲזָרָה בַּמָּקוֹם הַקָּדוֹשׁ, דִּכְתִיב בְּבֵית אֱלֹהִים נְהַלֵּךְ בְּרָגֶשׁ כוּ' [תְּהִלִּים נה]. וְאַף עַל גַּב דִּשְׁאָר פְּיָסוֹת הָיוּ בְּלִשְׁכַּת הַגָּזִית כוּ', הָתָם הַיְנוּ טַעֲמָא מִפְּנֵי שֶׁאוֹתָם הַכֹּהֲנִים שֶׁהָיוּ בָעֲזָרָה הָיוּ קְרוֹבִים לְלִשְׁכַּת הַגָּזִית וְהָתָם הָיוּ מְפַיְּסִים, אֲבָל הָכָא כֵּיוָן דְּהָיוּ בְּבֵית הַמּוֹקֵד שֶׁעֲדַיִן לֹא יָרְדוּ לָעֲזָרָה וְאִיכָּא נַמִּי בְּבֵית הַמּוֹקֵד מָקוֹם מְקֻדָּשׁ הָיוּ מְפַיְּסִים הָתָם, וּבָעֲזָרָה אִי אֶפְשָׁר לְפַיֵּס כוּ'. מְפָרֵשׁ. אֲבָל הָרַמְבַּ"ם כָּתַב, וְיָפִיסוּ בְּלִשְׁכַּת הַגָּזִית פַּיִס רִאשׁוֹן וְשֵׁנִי:

{טו} וְהַךְ סְתָמָא רַבִּי יְהוּדָה דְּמִדּוֹת פֶּרֶק קַמָּא מִשְׁנָה ז': {טז} <B> בְּיָדָם.</B> וּבְלֵילֵי שַׁבָּת וְיוֹם טוֹב כָּתַב הָרַמְבַּ"ם דְּאֵין בְּיָדָם אוּר, אֶלָּא בּוֹדְקִין בַּנֵּרוֹת הַדּוֹלְקִים שָׁם מֵעֶרֶב שַׁבָּת. וְאַף עַל גַּב דְּאֵין שְׁבוּת בַּמִּקְדָּשׁ, הָכָא גָזְרוּ, דְּלִשָּׂא אֲבוּקָה שֶׁל נֵר בְּיָדוֹ קָרוֹב לְוַדַּאי לְהַטּוֹת הַנֵּר שֶׁהוּא מַבְעִיר מַמָּשׁ: {יז} כְּלוֹמַר, רָחוֹק מִכָּתְלֵי הָעֲזָרָה. וְאָמְרִינַן בַּגְּמָרָא דְּשֶׁל בִּנְיָן הָיוּ, כְּלוֹמַר שֶׁל אֲבָנִים. דְּהָא כְתִיב לֹא תִטַּע לְךָ אֲשֵׁרָה כָּל עֵץ אֵצֶל מִזְבַּח ה' אֱלֹהֶיךָ [דְּבָרִים טז], וְהָכִי קָאָמַר, לֹא תִטַּע אֲשֵׁרָה וְכָל עֵץ אֵצֶל וְגוֹ': {יח} שֶׁבֵּית הַמּוֹקֵד הָיָה בַּצָּפוֹן, כְּדִתְנַן בְּפֶרֶק ג' מִשְׁנָה ג': {יט} <B>שָׁלוֹם כוּ'.</B> כֵּפֶל לָשׁוֹן. מְפָרֵשׁ: {כ} <B> הֶעֱמִידוּ כוּ'.</B> לָאו דַּחֲבִתִּין קְדִימֵי, שֶׁאֵין הַסֵּדֶר שָׁנוּי כָּךְ לְקַמָּן. אֶלָּא מַעֲמִיד אוֹתָם לְהָחֵם חַמִּין לִרְבוּכָה. גְּמָרָא:

{כא} <B> הִזָּהֵר.</B> אַף עַל פִּי שֶׁהוּא יוֹדֵעַ בַּדָּבָר צָרִיךְ לְהַזְהִירוֹ, שֶׁהוּא הוֹלֵךְ יְחִידִי וְשֶׁמָּא יִשְׁכַּח וְיִגַּע, וּמִתּוֹךְ שֶׁמַּזְהִירִין אוֹתוֹ נִזְהָר. מְפָרֵשׁ: {כב} <B> עַד כוּ'.</B> דְּלִינָה פּוֹסֶלֶת מִן קִדּוּשׁ יָדַיִם וְרַגְלַיִם, וְצָרִיךְ לַחֲזֹר וּלְקַדֵּשׁ. מְפָרֵשׁ: {כג} <B> וַהֲרֵי כוּ'.</B> כָּל זֶה הָיָה אוֹמֵר לוֹ, הֲרֵי הַמַּחְתָּה נְתוּנָה כוּ'. מְפָרֵשׁ: {כד} <B> בְּמַעֲרָבוֹ.</B> כְּלוֹמַר בַּאֲלַכְסוֹן [הִיא נְתוּנָה] מְמֻצַּעַת בֵּין כֶּבֶשׁ לַמִּזְבֵּחַ בְּזָוִית הַכֶּבֶשׁ הִיא נְתוּנָה אֵצֶל מָקוֹם שֶׁכָּלֶה הַכֶּבֶשׁ, דְּהַיְנוּ בֵּין הַכֶּבֶשׁ לַמִּזְבֵּחַ. מְפָרֵשׁ: {כה} <B> אֵין כוּ'.</B> שֶׁבֵּין אוּלָם לַמִּזְבֵּחַ אֵין אָדָם יָכוֹל לִכָּנֵס בָּהֶן אֶלָּא בִּשְׁעַת עֲבוֹדָה, וְכֶבֶשׁ הוּא כְּמִזְבֵּחַ לְכָל דָּבָר. מְפָרֵשׁ: {כו} <B> וְלֹא נֵר כוּ'.</B> לְפִי שֶׁהוֹלֵךְ יְחִידִי. מְפָרֵשׁ. וְנִרְאֶה לִי שֶׁרְצוֹנוֹ לוֹמַר דְּמִשּׁוּם שֶׁהוֹלֵךְ יְחִידִי הֲוָה אֲמֵינָא שֶׁיִּהְיֶה נֵר בְּיָדוֹ, קָא מַשְׁמַע לָן מַתְנִיתִין דְּלֹא: {כז} <B> לֹא כוּ'.</B> לְפִי שֶׁהָיוּ מִצַּד אַחֵר שֶׁל כֶּבֶשׁ בְּמִזְרָחוֹ, דְּהָא גֹּבַהּ הַכֶּבֶשׁ שֶׁהוּא גָבוֹהַּ הַרְבֵּה אַמּוֹת, הָיָה מַפְסִיק. מְפָרֵשׁ: {כח} <B> וְלֹא כוּ'.</B> מִפְּנֵי שֶׁרוֹצֶה לוֹמַר עַד שֶׁיִּהְיוּ שׁוֹמְעִין כוּ' תָּנֵי וְלֹא הָיוּ כוּ'. מְפָרֵשׁ: {כט} <B> עֵת.</B> לְקַדֵּשׁ. [עַיֵּן פ"ג מ"ב. בֵּית דָּוִד]. מְפָרֵשׁ: {ל} דַּאֲשֶׁר תֹּאכַל הָאֵשׁ כְּתִיב: {לא} <B> לַצָּפוֹן.</B> כְּלוֹמַר, לֹא הָיָה מְהַלֵּךְ כְּדַרְכּוֹ דֶּרֶךְ מִזְרָח וְיוֹצֵא חוּץ לָעֲזָרָה אֶלָּא הוֹפֵךְ פָּנָיו לִצְפוֹן הָעֲזָרָה דְּהַיְנוּ לִדְרוֹם הַמִּזְבֵּחַ: {לב} <B> הָלַךְ כוּ'.</B> וְאִם תֹּאמַר הָא בָעִינַן וְשָׂמוֹ אֵצֶל הַמִּזְבֵּחַ, וּנְהִי דְּכֶבֶשׁ כְּמִזְבֵּחַ לְכָל מִלֵּי, מִכָּל מָקוֹם הָיָה לָנוּ לְקָרְבוֹ אֵצֶל הַמִּזְבֵּחַ וַעֲדַיִן הוּא רָחוֹק מִן הַמִּזְבֵּחַ עֶשְׂרִים אַמָּה. וְאָמַר לִי רַבִּי, דְּמִוְּהִשְׁלִיךְ כוּ' דָּיְקִינַן דְּקִים לְהוּ לְרַבָּנָן דְּלֹא מִקְּרֵי הַשְׁלָכָה פָּחוֹת מֵעֶשְׂרִים אַמָּה, וְעוֹלַת הָעוֹף הִיא נַעֲשֵׂית עַל קֶרֶן דְּרוֹמִית מִזְרָחִית, הִלְכָּךְ מַנִּיחַ אֶת הַדֶּשֶׁן רָחוֹק מִן הַמִּזְבֵּחַ עֶשְׂרִים אַמָּה מְקוֹם שֶׁנּוֹתְנִים שָׁם מֻרְאָה וְנוֹצָה כוּ'. מְפָרֵשׁ: {לג} <B> מֻרְאוֹת כוּ'.</B> וְכָל אֵלּוּ נִבְלָעִים בִּמְקוֹמָן. אֲבָל תְּרוּמַת הַדֶּשֶׁן נִרְאֶה לִי דְּלֹא הָיְתָה נִבְלַעַת, כִּדְמוּכָח בְּיוֹמָא דַּף כ"א וּבִכְרֵתוֹת דַּף ו'. מְפָרֵשׁ. אֲבָל הַתּוֹסָפוֹת בְּיוֹמָא שָׁם וּבִזְבָחִים דַּף ס"ד כָּתְבוּ דְּאֵין רְאָיָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} הוֹאִיל וְרוֹצִין לְקַדֵּשׁ יְדֵיהֶם וְרַגְלֵיהֶם לְצֹרֶךְ עֲבוֹדָה רַשָּׁאִים לָלֶכֶת בֵּין הָאוּלָם לַמִּזְבֵּחַ. מְפָרֵשׁ: {ב} <B> הַמַּגְרֵפוֹת.</B> יָעִים לְהָסִיר אֶת הַדֶּשֶׁן וּלְפַנּוֹתוֹ לַתַּפּוּחַ: {ג} <B> הַצִּנּוֹרוֹת.</B> מַזְלְגוֹת לְהָסִיר אֶת הָאֵבָרִים. מְפָרֵשׁ: {ד} <B> וְעָלוּ כוּ'.</B> וְקֹדֶם תְּרוּמַת הַדֶּשֶׁן לֹא הָיוּ יְכוֹלִים לַעֲשׂוֹת, לְפִי שֶׁצָּרִיךְ לְהָרִים תְּרוּמַת הַדֶּשֶׁן קֹדֶם שֶׁיְּפַנּוּ אֶת הַדֶּשֶׁן לַתַּפּוּחַ. מְפָרֵשׁ: {ה} <B> וְהַפְּדָרִים.</B> הַחֲלָבִים. וְכֻלָּם נִקְרְאוּ כֵן עַל שֵׁם הַפֶּדֶר שֶׁבַּכָּתוּב: {ו} <B> סוֹדְרִין כוּ'.</B> כְּדֵי שֶׁלֹּא יִפָּסְלוּ בְּלִינָה, שֶׁאֵין יְכוֹלִין לְהַקְטִירָן עַד [לְמָחָר] לְאַחַר שֶׁיַּקְטִירוּ אֵבְרֵי תָמִיד, וְאִם לֹא סִדְּרָן עַל הַמִּזְבֵּחַ הָיוּ נִפְסָלִים בְּלִינָה כוּ'. וְאָמְרִינַן בִּזְבָחִים דַּף פ"ז דְּאֵין לִינָה בַּמִּזְבֵּחַ, וְכֶבֶשׁ הֲרֵי הוּא כְמִזְבֵּחַ לְכָל דָּבָר. וּלְאַחַר שֶׁיְּסַדְּרוּ אֶת הַמַּעֲרָכוֹת יַחֲזִירוּ אֶת הָאֵבָרִים עַל הַמַּעֲרָכָה וְיָצִיתוּ בָהּ הָאוּר:

{ז} <B> הֵחֵלּוּ.</B> הִתְחִילוּ. מְפָרֵשׁ: {ח} וְאַחַר כָּךְ גּוֹרְפִין מֵהַכְּרִי הַהוּא וּמוֹצִיאִין אֶל שֶׁפֶךְ הַדֶּשֶׁן עַד שֶׁמְּבַעֲרִין הָאֵפֶר מִן הַמִּזְבֵּחַ, וְהוּא הַנִּקְרָא דִשּׁוּן הַמִּזְבֵּחַ עַל הָאֱמֶת, זוּלָתִי בַמּוֹעֲדִים כוּ'. הָרַמְבַּ"ם: {ט} <B> הַכֹּהֵן.</B> מִי שֶׁמֻּטָּל עָלָיו לַעֲשׂוֹת. מְפָרֵשׁ. וְהָרַמְבַּ"ם כָּתַב, כָּל מִי שֶׁיִּרְצֶה מִן הַכֹּהֲנִים כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{י} וְקַשְׁיָא לִי לִישְׁנָא דְּהֵחֵלּוּ מַעֲלִין שֶׁהוּא לְשׁוֹן רַבִּים, וּבְמִשְׁנָה ה' פֶּרֶק ב' דְּיוֹמָא כָּתַבְתִּי דְּמִדִּכְתִיב וּבִעֵר יָלְפִינַן דְּבַבֹּקֶר כֹּהֵן אֶחָד מַעֲלֶה אוֹתָם. וּבְדִבְרֵי הָרַמְבַּ"ם מַשְׁמַע דְּגִזְרִין דְּהָכָא לָאו שְׁנֵי גִזְרִים הֵם, אֶלָּא בְּכָל עֲצֵי מַעֲרָכָה [דְּמַעֲרָכָה גְדוֹלָה] מַיְרֵי. וְזֶה לְשׁוֹן הַמְפָרֵשׁ, הַזּוֹכֶה בִּתְרוּמַת הַדֶּשֶׁן זָכָה בְּסִדּוּר הַמַּעֲרָכָה וּבִשְׁנֵי גִזְרִין. וְהָא דְקָתָנֵי הֵחֵלּוּ מַעֲלִין, רָצָה לוֹמַר שֶׁהָיוּ מְסַיְּעִין לוֹ, שֶׁהֵם הָיוּ מַעֲלִין הָעֵצִים וְהוּא סִדְּרָן. וְסִדּוּר הַשְּׁנֵי גִזְרִין הָיָה אַחַר מַעֲרָכָה שְׁנִיָּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יא} אַלִּיבָּא דְּמַאן דְּאָמַר דְּטַעֲמָא מִשּׁוּם יִשּׁוּב אֶרֶץ יִשְׂרָאֵל. מְפָרֵשׁ:

מִדֹּפֶן אֵצֶל חָזֶה, וַעֲדַיִן לֹא חָתַךְ דֹּפֶן הַשְּׂמָאלִי. לְכָךְ שַׁיָּךְ לוֹמַר וְהִנִּיחַ בָּהּ, שֶׁעֲדַיִן הֵן מְחֻבָּרִין לַדֹּפֶן. מְפָרֵשׁ: {כד} <B> בָּהּ.</B> כָּתַב הַמְפָרֵשׁ דְּלֹא כוּ'.</B> וּבִשְׁבִיל שֶׁהַכָּבֵד לָאו מִן הַדֹּפֶן הוּא וְאֵינוֹ בִכְלָלָהּ אֶלָּא תָלוּי בָהּ, לְכָךְ נִקְרֵאת גְּדוֹלָה. מְפָרֵשׁ: {כו} <B> חוֹתְכוֹ.</B> אֵצֶל רֶגֶל הַשְּׂמָאלִי שֶׁעֲדַיִן {א} עִקַּר הַדָּת וְרֵאשִׁיתוֹ. הָרַמְבַּ"ם: {ב} <B> אֶת הָעָם.</B> עִם הָעָם. וַעֲבוֹדָה, בִּשְׁבִיל עֲבוֹדָה שֶׁעָשׂוּ הָיוּ מְבָרְכִין אַחֲרֶיהָ רְצֵה כוּ' בָּרוּךְ הַמְקַבֵּל עֲבוֹדַת עַמּוֹ יִשְׂרָאֵל בְּרָצוֹן. אִי נַמִּי שֶׁאוֹתְךָ לְבַדְּךָ בְּיִרְאָה נַעֲבֹד. וּבִרְכַּת כֹּהֲנִים, לְבָרֵךְ אֶת הָעָם. רַשִׁ"י: {ג} וְהָרַמְבַּ"ם בִּמְקוֹם בִּרְכַּת כֹּהֲנִים כָּתַב שִׂים שָׁלוֹם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} הַתּוֹסָפוֹת הִקְשׁוּ, מַאי בְּרָכָה הִיא הוֹאִיל וְלֵית בָּהּ לֹא שֵׁם וְלֹא מַלְכוּת, כְּלוֹמַר דְּבַגְּמָרָא אָמְרִינַן דְּכָל בְּרָכָה שֶׁאֵין בָּהּ שֵׁם וּמַלְכוּת אֵינָהּ בְּרָכָה:

{ז} <B>הַכַּף.</B> כִּדְאַשְׁכְּחַן בַּנְּשִׂיאִים כַּף אַחַת מְלֵאָה קְטֹרֶת. גְּמָרָא: {ח} <B> גָּדוֹל.</B> וְאַף עַל פִּי שֶׁבַּתַּרְקָב עַצְמוֹ לֹא שַׁיָךְ לוֹמַר גָּדוֹל דְּהָא עַל שֵׁם מִדָּתוֹ נִקְרָא תַרְקָב, הָכָא קוֹרֵהוּ גָדוֹל בְּעֵרֶךְ הַדִּמְיוֹן שֶׁאֵלָיו נִדְמֶה הַכַּף, וְהַכַּף מַחֲזִיק שְׁלֹשָׁה קַבִּין, וְהַטֶּנִי שֶׁנִּדְמֶה גַם כֵּן לַתַּרְקָב אֵינוֹ מַחֲזִיק אֶלָּא קַבַּיִם וָחֵצִי: {ט} <B> בְּתוֹכוֹ.</B> וּבַמִּקְרָא כַּף אַחַת. וְאָמַר הָרַב אַבְרָהָם בֶּן עֶזְרָא, כָּל שֶׁאֵין בּוֹ רוּחַ חַיִּים, זָכְרֵהוּ וְנָקְבֵהוּ: {י} וְקַשְׁיָא לִי מִדִּתְנַן בְּמִשְׁנָה ב' פֶּרֶק ז' וְכַף וְכִסּוּיָהּ. וְיֶשׁ לוֹמַר שֶׁכְּשֶׁנָּתַן הַכַּף לַחֲבֵרוֹ כְּדִתְנַן בְּפֶרֶק ו' מִשְׁנָה ג' שֶׁהָיָה נוֹתֵן לוֹ גַם כֵּן כְּסוּי הַבָּזָךְ:

{יא} הָכִי תָנֵי רַבִּי יְהוּדָה בְּיוֹמָא דַּף כ"ה עָמוּד ב', וּכְלוֹמַר שֶׁכָּךְ רָגִיל לוֹמַר לָזֶה שֶׁעִמּוֹ. אֲבָל אִם יִרְצֶה לוֹמַר לְאַחֵר אוֹ שֶׁיִּרְצֶה לַעֲשׂוֹת הַכֹּל לְבַדּוֹ כִּדְאַשְׁכְּחָן בְּכֹהֵן גָּדוֹל בְּיוֹם הַכִּפּוּרִים, הָרְשׁוּת בְּיָדוֹ. הָכִי פֵּרְשׁוּ בַתּוֹסָפוֹת. אֶלָּא שֶׁדַּעְתָּם דְּרַבָּנָן פְּלִיגֵי עֲלֵיהּ בְּהָא דְלָא סְבִירָא לְהוּ דִּבְדַעַת כֹּהֵן תַּלְיָא מִלְּתָא. וּמִיהוּ מוֹדוּ דְּאֵין חוֹזְרִין וּמְפִיסִין אֶלָּא דְּהָעוֹמֵד לִימִינוֹ זוֹכֶה מִכֹּחַ פַּיִס. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} כְּתַנָּא קַמָּא דְּמִשְׁנָה ד' פֶּרֶק ד' דְּיוֹמָא: {יג} כְּלוֹמַר, לְאַחַר שֶׁיָּרַד וְלֹא בְּעוֹדוֹ עַל הַמִּזְבֵּחַ אַחַר שֶׁחָתָה כוּ' כְּדֵי שֶׁתְּהֵא מְלֵאָה וְלֹא יִתְפַּזֵּר בִּירִידָתוֹ בְּשִׁפּוּע הַכֶּבֶשׁ: {יד} וְאַף עַל פִּי שֶׁמִּן הַתּוֹרָה אֵין אִסּוּר בְּכִבּוּי אֶלָּא כְשֶׁמִּתְכַּוֵּן לַעֲשׂוֹת פֶּחָם, יֵשׁ לוֹמַר דְּהָכָא כֵיוָן דְּאֶפְשָׁר בְּלֹא כִבּוּי לֹא הִתִּירוּ הַשְּׁבוּת: {טו} לֹא דָק לְמִכְתַּב אַלִּיבָּא דְהִלְכָתָא, דַּהֲלָכָה כְּרַבִּי עֲקִיבָא שֶׁאוֹמֵר בַּמִּשְׁנָה דְּסוֹף עֵרוּבִין שֶׁאַף מִן הָעֲזָרָה מוֹצִיאִין בְּמָקוֹם שֶׁחַיָּבִים עַל זְדוֹנוֹ כָרֵת כוּ': {טז} לֹא דָק כְּלָל, שֶׁזֶּה אַף דְּלֹא כְרַבִּי שִׁמְעוֹן בֶּן נַנָּס דְּפָלֵיג אַדְּרַבִּי עֲקִיבָא, דְּהָא תְנַן הָתָם דְּמוֹדֶה אַף בְּמִבֵּין הָאוּלָם וְלַמִּזְבֵּחַ:

{יז} <B>הַמַּגְרֵפָה.</B> וְהָיָה לָהּ קוֹל גָּדוֹל. וְלֹא יִתָּכֵן לוֹמַר שֶׁהִיא הַמַּגְרֵפָה דְּמִשְׁנָה ח' פֶּרֶק ג' שֶׁהָיָה כְּלִי שִׁיר חָשׁוּב וְיִזְרְקֵהוּ, שֶׁאִי אֶפְשָׁר שֶׁלֹּא יִשָּׁבְרוּ קָנֶיהָ. וְגַם מַגְרֵפָה שֶׁגּוֹרְפִין בָּהּ הַדֶּשֶׁן אִי אֶפְשָׁר לוֹמַר, שֶׁהֲרֵי אֵין לָהּ קוֹל. אֲבָל הָיָה זֶה בְּלִי סָפֵק כְּלִי שְׁלִישִׁי שֶׁמַּשְׁמִיעַ קוֹל וְהָיָה גַם צוּרָתוֹ כְּצוּרַת מַגְרֵפָה: {יח} פְּלֻגְתָּא דְאָמוֹרָאֵי בִּפְסָחִים דַּף פ"ב. וְשָׁם פֵּרֵשׁ רַשִׁ"י טְמֵאִים שֶׁל בֵּית אָב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} <B> בְּדִשּׁוּן כוּ'.</B> וּכְפִי הַסֵּדֶר הַשָּׁנוּי בְּמִשְׁנָתֵנוּ שְׁנֵיהֶם הִקְדִּימוּ לִכָּנֵס וְלָצֵאת קֹדֶם שֶׁיַּקְטִיר הַמַּקְטִיר. וּכְבָר כָּתַבְתִּי בְּסוֹף פֶּרֶק ג' דְּאָתְיָא כְּאַבָּא שָׁאוּל. אֲבָל לְרַבָּנָן, לְאַחַר שֶׁהִקְטִיר נִכְנָס זֶה שֶׁזָּכָה בְּדִשּׁוּן הַמְּנוֹרָה וּמֵיטִיב שְׁתֵּי הַנֵּרוֹת. וְכֵן כָּתַב הָרַמְבַּ"ם, וּמְסַיֵּם וְיוֹצֵא זֶה שֶׁהִקְטִיר עִם מְדַשֵּׁן הַמְּנוֹרָה: {ב} וְלֹא דָיֵק, דְּאַחַר זְרִיקָה הוּא שֶׁהָיְתָה כְּנִיסָה זוֹ: {ג} לָאו דַּוְקָא, דְּהָא בְּסוֹף פֶּרֶק ג' תְּנַן וּמַדְלִיקָן מִן הַדּוֹלְקִים, וּפֵרֵשׁ הָרַ"ב וְאִם אֵין הַנֵּרוֹת דּוֹלְקִין מַדְלִיקָן מִמִּזְבַּח הָעוֹלָה, שְׁמַע מִנַּהּ דְּכִי יֵשׁ דּוֹלְקִין שַׁפִּיר טְפֵי לְמִדְלְקִינְהוּ מִנַּיְהוּ. וְלֹא בָא לוֹמַר אֶלָּא שֶׁאֵין מַדְלִיקִין בְּשׁוּם אֵשׁ אַחֵר שֶׁבַּמִּקְדָּשׁ, וְלֹא לְמַעוֹטֵי נֵרוֹת דִּמְנוֹרָה אִם יֵשׁ בָּהֶם דּוֹלְקִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{ד} <B> בְּחָפְנָיו.</B> וּמִשְׁתַּחֲוֶה וְיוֹצֵא. הָרַמְבַּ"ם: {ה} פֵּרוּשׁ לְפֵרוּשׁוֹ, שֶׁהָיָה שׁוֹפְכוֹ לְצַד מַעֲרָב, וְאַף עַל פִּי שֶׁרֹב מִן הַקְּטֹרֶת נִשְׁפָּךְ בִּשְׁפִיכָה שֶׁל צַד מַעֲרָב, מִכָּל מָקוֹם אֵרַע שֶׁנִּתְפַּזֵּר לְצַד עַצְמוֹ, וְהָיָה גוֹרְרוֹ מַה שֶּׁנִּתְפַּזֵּר לְצִדּוֹ וְצוֹבְרוֹ גַם כֵּן לְצִבּוּר הַשְּׁפִיכָה, וְעַל יְדֵי שֶׁנַּעֲשָׂה הַכֹּל צָבוּר נִשְׁתַּהָה הֶעָשָׁן מִלָּבֹא. וְלִישְׁנָא דִמְרַדֵּד לֹא מַשְׁמַע כֵן, דְּלֹא דָמִי לִמְרַדְּדָן דִּלְעֵיל. וּבְיוֹמָא פֵּרֵשׁ רַשִׁ"י, צוֹבֵר חוּצָה לוֹ לְצַד מַעֲרָב וְהוֹלֵךְ וּמוֹשֵׁךְ יָדוֹ אֵלָיו וְצוֹבֵר תָּמִיד בִּמְשִׁיכָה, וְלֹא יַתְחִיל לִצְבֹּר שֶׁלְּפָנָיו שֶׁמָּא יִכָּוֶה כְּשֶׁיִּפְשֹׁט זְרוֹעוֹ וְיִמְשֹׁךְ חוּצָה לוֹ: {ו} <B> פָּרְשׁוּ.</B> לְשׁוֹן הָרַמְבַּ"ם, וְאַחַר שֶׁאוֹמֵר, יִפְרְשׁוּ כָּל הָעָם:

{א} דְּהָא דִכְתִיב וְאַל יָבֹא בְּכָל עֵת אֶל הַקֹּדֶשׁ [וַיִּקְרָא טז], דָּרְשִׁינַן בִּמְנָחוֹת דְּקֹדֶשׁ הוּא הַהֵיכָל בְּאַל יָבֹא, וְאֶל מִבֵּית לַפָּרֹכֶת כוּ' בְּמִיתָה. וּמִלְּשׁוֹן הָרַ"ב מַשְׁמַע דְּכָל הִשְׁתַּחֲוָיָה שָׁרְיָא וְלֹא הַוְיָא שֶׁלֹּא לְצֹרֶךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> שֶׁשָּׁמַע כוּ'.</B> עַל יְדֵי פַעֲמוֹנֵי זָהָב שֶׁבְּשׁוּלֵי הַמְּעִיל: {ג} <B> יוֹצֵא.</B> פֵּרוּשׁ, יוֹצֵא מִמְּקוֹמוֹ בַּאֲשֶׁר הוּא שָׁם וְרוֹצֶה לִכָּנֵס לַהֵיכָל, קָדַם הוּא וְהִגְבִּיהַּ כוּ'. אֲבָל הָרַמְבַּ"ם הֶעְתִּיק כוּ' וְיִכָּנֵס לַהֵיכָל וְיִשְׁתַּחֲוֶה וְכֵיוָן שֶׁיִּשְׁמַע הַסְּגָן קוֹל רַגְלָיו שֶׁל כֹּהֵן גָּדוֹל שֶׁהוּא יוֹצֵא מַגְבִּיהַּ לוֹ אֶת הַפָּרֹכֶת. נִרְאֶה דְּשֶׁהוּא יוֹצֵא הַיְנוּ מֵהַהֵיכָל כוּ'. וְקָשֶׁה, דְּלָמָּה לֹא מַגְבִּיהַּ נַמִּי בִּכְנִיסָתוֹ לְשָׁם. וְנִרְאֶה לִי שֶׁהָיָה הַפָּרֹכֶת כָּל הַיּוֹם מֻגְבָּהּ וְעוֹמֵד עַל יְדֵי קְשָׁרִים וִיתֵרוֹת לִכָּנֵס וְלָצֵאת הַכֹּהֲנִים הַצְּרִיכִים לַעֲבוֹדַת מְנוֹרָה וּמִזְבֵּחַ פְּנִימִי, אֶלָּא שֶׁכְּשֶׁנִּכְנָס הַכֹּהֵן גָּדוֹל הָיוּ מְסִירִים הַפָּרֹכֶת מֵהַקְּשָׁרִים וִיתֵרוֹת כְּדֵי שֶׁיְּהֵא נִתְלֶה עַל עָמְדוֹ, וְכָבוֹד גָּדוֹל נַעֲשָׂה לוֹ לַכֹּהֵן גָּדוֹל בָּזֶה לְהַצְנִיעַ אַחֲרָיו הַקֹּדֶשׁ כְּאִלּוּ מִתְיַחֵד שָׁם עִם הַשּׁוֹכֵן בַּבַּיִת בָּרוּךְ הוּא, וּלְפִיכָךְ אַחַר שֶׁחָזַר וְרָצָה לָצֵאת הֻצְרָךְ לְהַגְבִּיהַּ, וְלֹא כֵן בִּשְׁעַת כְּנִיסָתוֹ. כֵּן נִרְאֶה לִי:

{ד} לְשׁוֹן הָרַמְבַּ"ם, רוֹמֵז אֶל כָּל הַכֹּהֲנִים שֶׁקָּרְאוּ בְּלִשְׁכַּת הַגָּזִית וְאַחַר כָּךְ עָשׂוּ מִן הָעֲבוֹדוֹת מַה שֶּׁעָשׂוּ: {ה} שֶׁזָּכַר אוֹתָם וְאָמַר שֶׁגָּמְרוּ עֲבוֹדָתָן תְּחִלָּה וְעָמְדוּ עַל מַעֲלוֹת הָאוּלָם קֹדֶם שֶׁיַּגִּיעוּ שְׁאָר אֲחֵיהֶם הָעוֹסְקִים בְּהַעֲלָאַת אֵבְרֵי הַתָּמִיד. הָרַמְבַּ"ם. וְכֵן כָּתַב בְּחִבּוּרוֹ שֶׁאַחַר שֶׁמַּעֲלִין הָאֵבָרִין לַמִּזְבֵּחַ מַתְחִילִין אֵלּוּ שֶׁעַל מַעֲלוֹת הָאוּלָם וּמְבָרְכִין בִּרְכַּת כֹּהֲנִים וְכוּ':

{ו} <B> לְהַקְטִיר.</B> עַל שֵׁם שֶׁעוֹלֶה קִטּוּר וְתִמְרוֹת עָשָׁן מִן הַחֲלָבִים וְהָאֵבָרִים קָרֵי לְהוּ בִּלְשׁוֹן הַקְטָרָה. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ז} דְּאִלּוּ סְמִיכָה דִקְרָא לֹא אַשְׁכְּחַן אֶלָּא בְּחַיִּים וְלֹא בִשְׁחוּטִין: {ח} <B>הַשֵּׁנִי כוּ'.</B> הַאי נַמִּי מִשּׁוּם יְקָרָא דְכֹהֵן גָּדוֹל, וּכְעֵין קְרִיאַת כֹּהֵן גָּדוֹל בְּיוֹם הַכִּפּוּרִים: {ט} <B> לְנַסֵּךְ.</B> הוּא הַדִּין לַהַעֲלָאַת סֹלֶת וְהַקְטָרַת הַחֲבִתִּין, אֶלָּא דְּלֹא תָנֵי אֶלָּא אֵלּוּ שֶׁנִּתְחַדֵּשׁ בָּהֶם דָּבָר בְּכֹהֵן גָּדוֹל: {י} שֶׁאֵין אוֹמְרִים שִׁירָה אֶלָּא עַל הַיַּיִן, כִּדְיָלֵיף בַּגְּמָרָא מִקְּרָא. וּפֵרְשׁוּ הַתּוֹסָפוֹת, אֵין אוֹמְרִים שִׁירָה עַל שׁוּם אֲכִילַת מִזְבֵּחַ, כְּגוֹן זְרִיקַת דָּמִים וְנִסּוּךְ הַמַּיִם, כִּי אִם עַל הַיַּיִן. אֲבָל וַדַּאי מָצִינוּ שִׁירָה בְּלֹא יַיִן, כְּגוֹן הַלֵּל בִּשְׁחִיטַת פְּסָחִים: {יא} <B> תָּקְעוּ.</B> לְפִי שֶׁאֶפְשָׁר לָאָדָם שֶׁלֹּא יִרְאֶה הֲנָפַת הַסּוּדָרִים כְּשֶׁיִּפְנֶה לִבּוֹ לְדָבָר אַחֵר, וְאִי נַמִּי כְּשֶׁהָיוּ בְתוֹךְ הַלְּשָׁכוֹת אוֹ בְהַר הַבַּיִת לֹא בָעֲזָרָה, אֲבָל קוֹל הַתְּקִיעוֹת אִי אֶפְשָׁר שֶׁלֹּא יִשְׁמָעֵם, וּלְפִיכָךְ תָּקְעוּ כְּדֵי שֶׁיִּפְנוּ לִרְאוֹת בַּהֲנָפַת הַסּוּדָרִים. נִרְאֶה לִי: {יב} <B> בַּצֶּלְצָל.</B> כְּלֵי זֶמֶר אֲחֵרִים הָיוּ גַם כֵּן שָׁם, אֶלָּא שֶׁזֶּה הָיָה כְּלִי זֶמֶר מְיֻחָד שֶׁאֵין שָׁם אֶלָּא הוּא, וְקוֹל גָּדוֹל הָיָה לוֹ וְהוּא הַמַּתְחִיל, וּמְמֻנֶּה מְיֻחָד הָיָה לוֹ: {יג} וּבְמִשְׁנָה ה' פֶּרֶק ה' דְּסֻכָּה כָּתַב הָרַ"ב, שֶׁבַּשִּׁיר הָיוּ שְׁלֹשָׁה פְּרָקִים שֶׁמַּפְסִיקִים בּוֹ, וּמַשְׁמַע דְּלֹא קָחֲשִׁיב לְהָנָךְ דְּהָכָא כוּ', יֵשׁ לוֹמַר מִשּׁוּם דְּאֶפְשָׁר בְּלֹא הֵם עַל יְדֵי הַסּוּדָרִים כוּ'. אִי נַמִּי, תַּנָּאֵי נִינְהוּ. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} <B> הָעָם.</B> כָּל הָעָם שֶׁבָּעֲזָרָה:

{טו} <B> לַה' כוּ'.</B> כָּל הַמִּזְמוֹר. רַשִׁ"י: {טז} וְנִתְעַלָּה וְיָשַׁב בַּמָּרוֹם דֻּגְמַת שָׁכְנוֹ בְּעִירוֹ וְהַר קָדְשׁוֹ. רַשִׁ"י: {יז} וּלְשׁוֹן הָרַמְבַּ"ם, שֶׁעָלֶיהָ יִהְיֶה הַדִּין וְהַדַּיָּנִין. עַד כָּאן. וְאַסֵּיפָא דִקְרָא קָיְמֵי, בְּקֶרֶב אֱלֹהִים יִשְׁפֹּט. וּבַגְּמָרָא דְרֹאשׁ הַשָּׁנָה שָׁם, עַל שׁוּם שֶׁגִּלָּה אֶרֶץ בְּחָכְמָתוֹ וְהֵכִין תֵּבֵל לַעֲדָתוֹ. פֵּרֵשׁ רַשִׁ"י שֶׁגִּלָּה אֶרֶץ מְקוֹם מַצַּב עֲדָתוֹ: {יח} <B> לְחַיֵּי.</B> וְהָרַ"ב הֶעְתִּיק לְחֵי הָעוֹלָמִים. וּשְׁנֵיהֶם צָדְקוּ יַחְדָּיו. שֶׁאִם אַתָּה אוֹמֵר לְחַיֵּי הָעוֹלָמִים, הַכַּוָּנָה לַהַשֵּׁם יִתְבָּרַךְ שֶׁהוּא חַיֵּי הָעוֹלָמִים כֻּלָּם, כָּעִנְיָן שֶׁנֶּאֱמַר וְאַתָּה מְחַיֶּה אֶת כֻּלָּם. וַאֲפִלּוּ אִם תֹּאמַר חַיֵּי הָעוֹלָם הוּא נָכוֹן, כִּי הוּא סָמוּךְ מִשֵּׁם חַיִּים שֶׁלֹּא נִמְצָא נִפְרָד בִּלְשׁוֹן הַקֹּדֶשׁ. וּכְשֶׁאַתָּה אוֹמֵר חֵי הָעוֹלָמִים גַּם כֵּן נָכוֹן, וְתִהְיֶה הַקְּרִיאָה חֵי בְּצֵירֵ"י, כָּעִנְיָן שֶׁנֶּאֱמַר וַיִּשָּׁבַע בְּחֵי הָעוֹלָם [דָּנִיאֵל יב], שֶׁהוּא סָמוּךְ, וְיִהְיֶה עִנְיָנוֹ חַיִּים שֶׁל הָעוֹלָמִים וְהוּא הַשֵּׁם יִתְבָּרַךְ שֶׁהוּא חִיּוּתוֹ שֶׁל עוֹלָם כָּאָמוּר. מַה שֶּׁאֵין כֵּן חַי בְפַתָּ"ח, שֶׁאֵינוֹ סָמוּךְ, יִהְיֶה עִנְיָנוֹ הָעוֹלָם שֶׁהוּא חַי, וְאֵין הָעוֹלָם עַצְמוֹ חַי עַל צַד הָאֱמֶת. כֵּן כָּתַב הָרַמְבַּ"ם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יט} לַאֲפוֹקֵי מִמַּאן דְּאָמַר תְּרֵי חֲרוּב, שֶׁנֶּאֱמַר [הוֹשֵׁעַ ו] יְחַיֵּינוּ מִיֹּמַיִם. גְּמָרָא:
