{א} בַּגְּמָרָא יָלֵיף לֵיהּ מִקְּרָא: {ב} <B> נָשִׁים.</B> אַף עַל גַּב דִּתְנַן בְּמִשְׁנָה ז' פֶּרֶק קַמָּא דְּקִדּוּשִׁין כָּל מִצְוֹת לֹא תַעֲשֶׂה כוּ' אֶחָד אֲנָשִׁים וְאֶחָד נָשִׁים חַיָּבִים כוּ', אִצְטְרִיךְ לְמִתְנֵי הָכִי, דְּמַהוּ דְּתֵימָא הָנֵי מִלֵּי עֹנֶשׁ דְּשָׁוֶה בֵּין בְּיָחִיד בֵּין בְּצִבּוּר, אֲבָל הָכָא כֵּיוָן דְּעֹנֶשׁ שֶׁאֵינוֹ שָׁוֶה בַּכֹּל הוּא, דִּתְנַן אֵין הַצִּבּוּר וְהַשֻּׁתָּפִים עוֹשִׂין תְּמוּרָה, אִשָּׁה נַמִּי כִּי עֲבַדָה לֹא לָקְיָא, קָא מַשְׁמַע לָן, מִדִּכְתִיב וְאִם הָמֵר, וַהֲוָה מָצֵי לְמִכְתַּב אִם, אָתֵי וָי"ו לְרַבּוֹת הָאִשָּׁה. גְּמָרָא: {ג} <B> הָאַרְבָּעִים.</B> וְאִם תֹּאמַר וְאַמַּאי לֹא תָּנֵי שְׁמוֹנִים, דְּהָא כְּתִיב לֹא יַחֲלִיפֶנּוּ וְלֹא יָמִיר אֹתוֹ. וְיֵשׁ לוֹמַר דְּלֹא נָחֵית לְמִנְיַן מַלְקִיּוֹת אֶלָּא כְּלוֹמַר דְּלָקֵי עֲלֵיהּ. עוֹד יֵשׁ לוֹמַר דִּלְעוֹלָם לֹא לָקֵי אֶלָּא אַרְבָּעִים, וּתְרֵי לָאוֵי צְרִיכֵי, חַד בְּקָרְבָּן שֶׁלּוֹ וְחַד בְּשֶׁל חֲבֵרוֹ, וּכְגוֹן דְּאָמַר כָּל הָרוֹצֶה לְהָמִיר יָבֹא וְיָמִיר. תּוֹסָפוֹת: {ד} וְכֵן כָּתַב הָרַמְבַּ"ם. וּבַגְּמָרָא יָלְפִינַן לְנִשְׁבַּע וּמְקַלֵּל מִקְּרָאֵי. וּמֵימֵר, אָמַר רַבִּי יוֹחָנָן לֹא תִתְנֵי וּמֵימֵר מִשּׁוּם דִּבְדִבּוּרוֹ עֲבִיד מַעֲשֶׂה. פֵּרֵשׁ רַשִׁ"י שֶׁעָשָׂה מֵחֻלִּין קָדָשִׁים. וְאִם כֵּן קֻשְׁיַת הָרַ"ב לֹא קָשֶׁה מִידִי, וְכֵן הִקְשָׁה הַכֶּסֶף מִשְׁנֶה וְהַמַּגִּיד מִשְׁנֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} וְכֵן כֹּהֵן שֶׁהֵמִיר בִּבְכוֹר שֶׁנּוֹלַד לוֹ, לֹא בִּבְכוֹר שֶׁלָּקַח מִיִּשְׂרָאֵל, הֲרֵי זוֹ תְּמוּרָה. הָרַמְבַּ"ם:

{ו} <B> אֶחָד כוּ'.</B> יָלֵיף לְהוּ בַּגְּמָרָא מִקְּרָא: 

~ פרק א - משנה ג

{ז} <B> וַהֲלֹא כוּ'.</B> וּבְרֵישׁ פֶּרֶק י"א דְּחֻלִּין כָּתַבְתִּי דְּאַף לְתַנָּא קַמָּא כָּךְ הוּא בְּמַקְדִּישׁ אֵבָר שֶׁהַנְּשָׁמָה תְּלוּיָה בּוֹ: 

~ פרק א - משנה ד

{ח} רַשִׁ"י וְהָרַ"מ. וּלְפֵרוּשׁ זֶה אֵינוֹ דּוֹמֶה הַךְ לְפִי חֶשְׁבּוֹן לְאִינָךְ, דְּאִינְהוּ מַיְרֵי לְפִי חֶשְׁבּוֹן הַדָּבָר הַפּוֹסֵל. לְכָךְ נִרְאֶה פֵּרוּשׁ הַתּוֹסָפוֹת שֶׁפֵּרְשׁוּ, חֶשְׁבּוֹן הַלֻּגִּין שֶׁפּוֹסְלִין הַמִּקְוֶה, דְּהַיְנוּ שְׁלֹשָׁה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} וְכֵן כָּתַב הָרַמְבַּ"ם. וְתֵימַהּ, דְּהָא סָתַם לָן כְּוָתֵיהּ בְּסוֹף פֶּרֶק ג' דְּמִקְוָאוֹת [וְהָא דְּנָקַט בִּדְבָרָיו רַבִּי יוֹסֵי בֶּן חוֹנִי, דְּהָכִי קָתָנֵי לַהּ בְּהֶדְיָא בַּבְּרַיְתָא. הָרַ"שׁ]. וּבֶאֱמֶת הָרַמְבַּ"ם בְּחִבּוּרוֹ חָזַר בּוֹ: 

~ פרק א - משנה ה

{י} גְּמָרָא, מַאי חָזֵית דְּאָמַרְתְּ סֵיפֵיהּ דִּקְרָא דַּוְקָא, דִּלְמָא רֵישָׁא דִּקְרָא דַּוְקָא [וְאֶל כֶּלִי לְהָכִי אָתָא שֶׁתְּהֵא חִיּוּתוֹ בַּכְּלִי שֶׁלֹּא יְמַלְּאֵם בִּכְלִי אַחֵר וְיִעָרֵם לִכְלִי זֶה, אֶלָּא זֶה יִתֵּן לְתוֹךְ הַמַּעְיָן וְיִקַּח הַמַּיִם לְתוֹכוֹ. רַשִׁ"י] לֹא מָצֵית אָמַרְתְּ. מַאי מָצִינוּ בְּכָל מָקוֹם מַכְשִׁיר לְמַעְלָה [כְּגוֹן בְּסוֹטָה, עָפָר לְמַעְלָה שֶׁהוּא הַמַּכְשִׁיר אֶת הַמַּיִם לְבַדּוֹ וְהָוֵי לְמַעְלָה, דְּמִצְוָה לְהַקְדִּים הַמַּיִם כְּדִכְתִיב וּמִן הֶעָפָר וְגוֹ'], אַף כָּאן מַכְשִׁיר לְמַעְלָה: {יא} וּפְלִיאָה הוּא בְּעֵינַי, דְּהָתְנַן בְּמִשְׁנָה ג' פֶּרֶק י"ז דְּאֳהָלוֹת, הַחוֹרֵשׁ מִשָּׂדֶה שֶׁאָבַד הַקֶּבֶר בְּתוֹכָהּ אֵינוֹ עוֹשֶׂה בֵּית הַפְּרָס, וּפֵרֵשׁ הָרַ"ב דַּהֲוֵי לֵיהּ סְפֵק סְפֵקָא, שֶׁמָּא לֹא חָרַשׁ בִּמְקוֹם הַקֶּבֶר [וַאֲפִלּוּ חָרַשׁ כָּל הַשָּׂדֶה יֵשׁ לְכָל צַד סָפֵק דְּשֶׁמָּא הָיָה הַקֶּבֶר בַּצַּד הָאַחֵר], וְאִם תִּמְצָא לוֹמַר חָרַשׁ, שֶׁמָּא לֹא הוֹלִיךְ מִמֶּנָּה עֶצֶם. וְלִי נִרְאֶה לְפָרֵשׁ דְּמַאי דְּאָמַר בַּגְּמָרָא שָׁלֹשׁ שָׂדוֹת וּשְׁתֵּי מַעֲנוֹת, פֵּרוּשׁ, בֵּין כָּל הַשָּׁלֹשׁ שָׂדוֹת שְׁתֵּי מַעֲנוֹת וְהַקֶּבֶר הָיָה בְּאֶמְצָעוֹ שֶׁל אֶמְצָעִי. וּכְפֵרוּשׁ הָרַ"ב הָתָם רֵישׁ הַפֶּרֶק. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יב} וְכֵן פֵּרֵשׁ רַשִׁ"י, דְּהָא סְתָמָא קָאָמַר אֵינָהּ תְּרוּמָה. וְהַתּוֹסָפוֹת הִקְשׁוּ, דְּאֵין זֶה כְּמוֹ וְלֹא תְּמוּרָה אַחַר תְּמוּרָה, דְּהָתָם הָרִאשׁוֹן תְּמוּרָה וְהַשֵּׁנִי אֵינָהּ תְּמוּרָה. לָכֵן נִרְאֶה לְפָרֵשׁ הָכִי נַמִּי דְּהָרִאשׁוֹן תְּרוּמָה וְלֹא הַשֵּׁנִי: {יג} דְּפֶרֶק ג' דִּתְרוּמוֹת מִשְׁנָה ג' דְּקָאָמַר אֵין תְּרוּמַת שְׁנֵיהֶם תְּרוּמָה. וְהָא דְּאִיתָא הָתָם תְּרוּמַת שְׁנֵיהֶם תְּרוּמָה, כָּתַב הָרַ"שׁ, דְּחִלּוּף גִּרְסָאוֹת הֵן. וּלְפֵרוּשׁ הַתּוֹסָפוֹת דִּבְסָמוּךְ, צָרִיךְ לוֹמַר דְּגָרְסִינַן דְּתַנְיָא, וּמַיְרֵי בְּשֶׁנָּתְנוּ רְשׁוּת זֶה לָזֶה לִתְרֹם. וְהָתָם בִּתְרוּמוֹת מַיְרֵי בְּלֹא נָתְנוּ רְשׁוּת זֶה לָזֶה רְשׁוּת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {יד} וְהוּא הַדִּין אִפְּכָא נַמִּי. וְעַיֵּן שָׁם בְּפֵרוּשׁ הָרַ"ב: {טו} כְּלוֹמַר, דְּאַף תְּרוּמַת הַשֵּׁנִי תְּרוּמָה, דְּאֵין לוֹמַר דִּתְרוּמַת הַשֵּׁנִי דַּוְקָא תְּרוּמָה וְלֹא הָרִאשׁוֹן, דְּמַאי אוּלְמֵיהּ דְּהַאי מֵהַאי. כֶּסֶף מִשְׁנֶה: 

~ פרק א - משנה ו

{טז} <B> קָרְבְּנוֹת כוּ'.</B> וְבַגְּמָרָא גָּרַס קָדְשֵׁי. וְכֵן בְּרַשִׁ"י וְהָרַ"ב]. וְיֵשׁ לְיַשֵּׁב גִּרְסַת הַסֵּפֶר, דְּהָא מוֹדוּ רַבָּנָן דְּקָרְבַּן ה' אִקְּרוּ כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב שֶׁהֶאֱרִיךְ דְּרַבִּי שִׁמְעוֹן וְרַבָּנָן לֹא פְּלִיגֵי אֶלָּא בְּמַשְׁמָעוּת דּוֹרְשִׁין וְלֹא בְּעִקַּר הַדִּין, דְּכִי אִמַּעֲטוּ קָרְבְּנוֹת צִבּוּר מִלָּאו, אִמַּעֲטוּ נַמִּי דְּאֵין מְמִירִין, דְּהָא אֵין מְמִירִין אֶלָּא בְּשֶׁלָּהֶם. וְדֹחַק לוֹמַר דְּמַיְרֵי כְּשֶׁאָמְרוּ הַצִּבּוּר כָּל הָרוֹצֶה לְהָמִיר יָמִיר: {יז} <B> לְהָקִישׁ כוּ'.</B> וּמַאן דְּפָלֵיג אַדְּרַבִּי שִׁמְעוֹן אִצְטְרִיכָא לֵיהּ בַּגְּמָרָא לִדְרָשָׁא אַחֲרִיתִי:

{א} <B> עוֹשִׂין כוּ'.</B> מְפָרֵשׁ בַּגְּמָרָא בְּעִקַּר זְבִיחָה [הֶקְדֵּשׁ רִאשׁוֹן. רַשִׁ"י] קָתָנֵי. דְּהָא תְּמוּרָה, דְּקָרְבַּן יָחִיד, וְאֵינָהּ עוֹשָׂה תְּמוּרָה. וּוָלָד, דְּקָרְבַּן יָחִיד, וְאֵינוֹ כוּ'. וְעוֹף, לָאו בַּר זְבִיחָה הוּא, דְּנִמְלָק וְלֹא נִשְׁחָט: {ב} וְכֵן לְשׁוֹן רַשִׁ"י. וְטַעֲמַיְהוּ, דְּהָא יֵשׁ שֶׁקָּבוּעַ לָהֶם זְמַן וְאִם עָבַר זְמַנָּן אֵין חַיָּבִים בְּאַחֲרָיוּתָן. חֲבִתֵּי כֹּהֵן גָּדוֹל וּפַר יוֹם הַכִּפּוּרִים. אֶלָּא יֵשׁ בְּקָרְבְּנוֹת הַיָּחִיד קָאָמְרִינַן, וְלֹא כָּל קָרְבְּנוֹת הַיָּחִיד: {ג} וּלְאַפּוֹקֵי פַּר הַבָּא עַל כָּל הַמִּצְוֹת, וּשְׂעִירֵי עֲבוֹדָה זָרָה: {ד} <B> נִסְכֵּיהֶם.</B> כְּלוֹמַר הַיַּיִן וְהַסֹּלֶת: {ה} <B> לֹא בְאַחֲרָיוּתָן כוּ'.</B> בַּגְּמָרָא יָלֵיף לֵיהּ מִקְּרָא: {ו} <B>חֲבִתֵּי כוּ'.</B> וְאֵין בָּהֶם אֲכִילָה. וּלְכָךְ לֹא תְּנַן לְהוּ בַּהֲדֵי חֲמִשָּׁה דְּבָרִים הַבָּאִים בְּטֻמְאָה דִּתְנַן בִּפְסָחִים פֶּרֶק ז' מִשְׁנָה ד':

{ז} דְּכָל הֶחָמֵשׁ חַטָּאוֹת נִשְׁנוּ בַּחֲדָא שִׁיטָה, וְנִשְׁתַּכְּחוּ בִּימֵי אֶבְלוֹ שֶׁל מֹשֶׁה אִי בְּיָחִיד אִי בְּצִבּוּר. וְכֵיוָן דְּאִיכָּא בְּהוּ דְּלֹא אֶפְשָׁר לִהְיוֹת בְּצִבּוּר, כֻּלְּהוּ נַמִּי הָכִי נִשְׁנוּ דְּאֵינָן בְּצִבּוּר. וְלֵיכָּא לְמִפְרַךְ דְּאֵין דָּנִין אֶפְשָׁר מִשֶּׁאִי אֶפְשָׁר כִּדְאִיתָא בַּגְּמָרָא: {ח} יָלְפִינַן לַהּ בַּגְּמָרָא מִשְּׂעִירֵי רְגָלִים וְרָאשֵׁי חֳדָשִׁים דְּאָמַר רַחֲמָנָא אַיְתִינְהוּ מִתְּרוּמַת הַלִּשְׁכָּה, וְדִלְמָא מֵתוּ מָרַיְהוּ דְּהָנֵי זוּזֵי, אֶלָּא לָאו שְׁמַע מִנַּהּ אֵין הַצִּבּוּר מֵתִים:

{א} <B> וְלַד כוּ'.</B> יָלֵיף לְהוּ בַּגְּמָרָא מִקְּרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> וּוְלָדָן.</B> רוֹצֶה לוֹמַר וַלְדֵי שְׁלָמִים וַלְדֵי תְּמוּרַת שְׁלָמִים. הָרַמְבַּ"ם: {ג} וְאִם תֹּאמַר וְלִתְנֵי עַד סוֹף הָעוֹלָם וְלֹא לִתְנֵי וְלָדָן כוּ'. וְיֵשׁ לוֹמַר דְּזוֹ אַף זוֹ קָתָנֵי, שֶׁכֵּן דֶּרֶךְ מִשְׁנָה לִשְׁנוֹיֵי הָכִי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ד} וְהָרַמְבַּ"ם מְפָרֵשׁ, עוֹבֵר עַל מַה שֶּׁאָמַר רַחֲמָנָא לֹא תְאַחֵר לְשַׁלְּמוֹ. עַד כָּאן. וַהֲוָה לֵיהּ לְמֵימַר נַמִּי עֲשֵׂה: {ה} דְּבִשְׁלָמָא בְּפֶסַח לֹא אַקְרְבֵיהּ דְּאֵמוּר מְחֻסַּר זְמַן הָוֵי דְּנוֹלַד בְּפֶסַח וְעָבַר הָרֶגֶל קֹדֶם שֶׁיִּמְלְאוּ לוֹ שִׁבְעָה יָמִים. רַשִׁ"י:

{ח} <B> תִּרְעֶה.</B> וּבְעוֹלָה מִלְּתָא דִּפְשִׁיטָא הִיא. וּרְבוּתָא אַשְׁמְעִינַן לְעֵיל דַּאֲפִלּוּ יָלְדָה זָכָר הָרָאוּי לְעוֹלָה דְּיִרְעֶה. וְהָכִי נַמִּי אָשָׁם שֶׁיָּלְדָה זָכָר שֶׁיִּרְעֶה. וּבַגְּמָרָא, דְּגַם רַבִּי אֱלִיעֶזֶר מוֹדֶה בָּהּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ט} בְּרֵישׁ מַסֶּכֶת זְבָחִים. רַשִׁ"י: {י} <B> יָבִיא כוּ'.</B> דְּעַד כָּאן לֹא אָמַר רַבִּי אֱלִיעֶזֶר הוּא עַצְמוֹ יַקְרִיב עוֹלָה [אֶלָּא גַּבֵּי מַפְרִישׁ נְקֵבָה לְעוֹלָה], דְּאִיכָּא שֵׁם עוֹלָה עַל אִמּוֹ. אֲבָל גַּבֵּי תְּמוּרַת וְלַד אָשָׁם, דְּלֵיכָּא שֵׁם אָשָׁם עַל אִמּוֹ, מוֹדֶה רַבִּי אֱלִיעֶזֶר דִּבְדָמָיו אִין, הוּא עַצְמוֹ לֹא קָרֵב. גְּמָרָא: {יא} בַּגְּמָרָא, וְאָמְרוּ לֵיהּ לְנִדְבַת צִבּוּר אָזְלֵי. וְנִרְאֶה, דִּפְלִיגֵי בְּמוֹתָרוֹת דְּסוֹף פֶּרֶק ב' דִּשְׁקָלִים, דְּרַבִּי אֱלִיעֶזֶר לֹא סְבִירָא לֵיהּ דִּלְצִבּוּר: {יב} <B> אָשָׁם כוּ'.</B> וּצְרִיכָא, דְּאִי אַשְׁמְעִינַן אָשָׁם שֶׁכִּפְּרוּ בְּעָלָיו, בְּהָא קָאָמַר רַבִּי אֱלִיעֶזֶר יָמוּתוּ, מִשּׁוּם דְּגָזַר לְאַחַר כַּפָּרָה [שֶׁנִּתְכַּפֵּר בָּרִאשׁוֹן] אַטּוּ לִפְנֵי כַּפָּרָה יָבִיא בְּדָמֶיהָ גַּם כֵּן עוֹלָה [וְקָעֲבִיד אִסּוּרָא, דְּכָל זְמַן שֶׁלֹּא נִתְכַּפֵּר קַיְמֵי דָּמֶיהָ לְאָשָׁם], אֲבָל גַּבֵּי תְּמוּרַת אָשָׁם וְלַד תְּמוּרָתָהּ [דְּלָאו לְהַקְרָבָה קַיְמָא דְּהָכִי גְּמִירֵי לַהּ] אֵימָא מוֹדֶה לְהוּ לְרַבָּנָן. וְאִי אַשְׁמְעִינַן הָתָם, בְּהָא קָאָמְרֵי רַבָּנָן, צְרִיכֵי:

{א} <B> וְלַד כוּ'.</B> בִּפְסִיקְתָּא יָהֵיב סִימָנַיְהוּ. ותמנ"ע: {ב} <B> יָמוּתוּ.</B> לֹא שֶׁיַּהַרְגֵם בִּכְלִי אוֹ בַּיָּדַיִם, אֶלָּא מַכְנִיסִין אוֹתָן לְבַיִת אֶחָד וּמַנִּיחִין אוֹתָן שָׁם עַד שֶׁתָּמוּת, כְּלוֹמַר שֶׁאֵין נוֹתְנִין לָהֶם מְזוֹנוֹת. רַשִׁ"י: {ג} וְצָרִיךְ לְמִתְנֵי אִם אָבְדָה גַּבֵּי בַּעַל מוּם, דְּאִי תָּנֵי גַּבֵּי שֶׁעָבְרָה שְׁנָתָהּ הֲוָה אֲמֵינָא הָתָם הוּא דִמְהַנְּיָא לֵיהּ לְאַבְּדָהּ [לְהָבִיא לִידֵי מִיתָה], מִשּׁוּם דְּלֹא חַזְיָא לְמִלְּתָא [אֲפִלּוּ קֹדֶם אָבְדָה], אֲבָל בַּעַל מוּם דְּאִי לֹא מוּמָא, חַזְיָא, אֵימָא לֹא תְּהַנֵּי לֵיהּ אָבְדָה. וְאִי תָּנָא גַּבֵּי בַּעַל מוּם, הָתָם הוּא כוּ', מִשּׁוּם דְּלֹא חַזְיָא לְהַקְרָבָה [בְּשׁוּם קָרְבָּן], אֲבָל עָבְרָה שְׁנָתָהּ דְּחַזְיָא לְהַקְרָבָה [בִּשְׁאָר קָרְבָּנוֹת] אֵימָא לֹא תְּהַנֵּי לֵיהּ אָבְדָה, צְרִיכֵי. גְּמָרָא: {ד} וְכֵן כָּתַב הָרַמְבַּ"ם. וְתֵימַהּ, דְּבַגְּמָרָא מְפָרְשֵׁי דִּבְרֵי הַכֹּל בְּמִתְכַּפֵּר בְּשֶׁאֵינָהּ אֲבוּדָה דְּמֵתָה כוּ'. אֲבָל לְשׁוֹן הָרַמְבַּ"ם שֶׁבְּנֻסְחַת אֶרֶץ יִשְׂרָאֵל יֵשׁ לְפָרֵשׁ כְּשִׁיטַת הַתּוֹסָפוֹת דְּאַדְּרַבָּה בְּרִבּוּי רִיעוּתָא יוֹתֵר מִסְתַּבֵּר לְהָבִיא לִרְעִיָּה וְלֹא לְמִיתָה, וּדְנָקַט בַּעֲלַת מוּם לִרְבוּתָא דְּסָלְקָא דַעְתָּךְ הוֹאִיל וְאִתְרַבֵּי רִיעוּתָא כָּל כָּךְ דְּאָבְדָה וּבַעַל מוּם כוּ' תִּרְעֶה, קָמַשְׁמַע לָן דַּאֲפִלּוּ הָכִי מֵתָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

{א} <B> מַעֲרִימִין.</B> דְּאִי אַתָּה מַקְדִּישׁוֹ קְדֻשַּׁת מִזְבֵּחַ. וְתַחְבּוּלוֹת הַהֶתֵּר תִּקָּרֵא עָרְמָה, וְשֶׁאֵינָהּ לְהֶתֵּר תִּקָּרֵא מִרְמָה. הָרַמְבַּ"ם. וְאַף עַל פִּי שֶׁמָּצִינוּ מִרְמָה אֵצֶל יַעֲקֹב וּבָנָיו וְתִרְגֵּם אוּנְקְלוּס בְּחָכְמָא, וּמָצִינוּ לְהָרְגוֹ בְעָרְמָה. לְשׁוֹן תּוֹרָה לְחוּד וּלְשׁוֹן חֲכָמִים לְחוּד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ב} <B> מְבַכֶּרֶת כוּ'.</B> בְּסִפְרָא, אֲשֶׁר יְבֻכַּר כוּ' לֹא יַקְדִּישׁ, מִי שֶׁיְּבֻכַּר אִי אַתָּה מַקְדִּישׁוֹ, אֲבָל אַתָּה מַקְדִּישׁוֹ בַּבֶּטֶן. הָרַמְבַּ"ם: {ג} <B> עוֹלָה.</B> מְפָרֵשׁ בַּגְּמָרָא דְּדַוְקָא עוֹלָה דְּלֹא נַחְתֵּיהּ מִקְּדֻשָּׁתֵיהּ. אֲבָל שְׁלָמִים לֹא, דְּנַחְתַּיהּ מִקְּדֻשָּׁתֵיהּ שֶׁיֶּהֱנֶה בּוֹ. הָרַמְבַּ"ם: {ד} וּבְהוּ נַמִּי תְּנַן בְּמִשְׁנָה ג' פֶּרֶק ז' דְּאֵין מְשַׁנִּין מִקְּדֻשָּׁה לִקְדֻשָּׁה, וְהָכָא דִּלְמִיתָה אָזֵיל רַשַּׁאי לְשַׁנּוֹתוֹ אַף לִשְׁלָמִים. וְעַיֵּן בַּמִּשְׁנָה דִּלְקַמָּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: {ה} <B> עוֹלָה.</B> אַף עַל גַּב דְּיָכוֹל לַעֲשׂוֹתוֹ שְׁלָמִים, אוֹרְחָא דְמִלְּתָא נָקַט, שֶׁכְּשֶׁיֵּשׁ לוֹ שְׁתֵּי בְּהֵמוֹת וְאֶחָד מֵהֶן רָאוּי לְעוֹלָה, עוֹשֵׂהוּ עוֹלָה, כֵּיוָן שֶׁיֵּשׁ לוֹ גַּם כֵּן אַחֶרֶת לִשְׁלָמִים:

{יז} <B> יִקָּבֵר.</B> וְהָכִי נַמִּי פָּלֵיג בְּחַטַּאת הָעוֹף. וּבְתוֹרַת כֹּהֲנִים דָּרֵישׁ מִכִּי קֹדֶשׁ הֵם, הֵם בִּשְׂרֵפָה וְלֹא חַטַּאת הָעוֹף וַאֲשַׁם תָּלוּי בִּשְׂרֵפָה. תּוֹסָפוֹת: {יח} אֲבָל בְּנִקְבָּרִין אַף עַל גַּב דְּאִיכָּא נַמִּי לְמֵיחַשׁ לְהָכִי, מִכָּל מָקוֹם אִי אֶפְשָׁר לְשָׂרְפָן, דַּאֲפָרָן אָסוּר בַּהֲנָאָה. וְהוֹאִיל וְהַנִּשְׂרָפִין אֲפָרָן מֻתָּר אָתֵי לְמִתְהַנֵּי מִנֵּיהּ, וְאִי אֶפְשָׁר לוֹמַר שֶׁאֵפֶר אֵלּוּ יִקָּבֵר, לְפִי שֶׁאֵין בְּיַד כָּל אָדָם לְחַלֵּק בֵּין אֵפֶר לְאֵפֶר אֵיזֶהוּ בִּקְבוּרָה וְאֵיזֶהוּ בִּשְׂרֵפָה, וְאָתֵי לְמִטְעֵי. הִלְכָּךְ אָמְרוּ אוֹתָן שֶׁאֲפָרָן אָסוּר יִקְבְּרֶנּוּ בְּעַצְמָן. נִרְאֶה לִי:   {יט} צָרִיךְ עִיּוּן טַעֲמָא מַאי. וְאוֹמֵר מוֹרִי הָרמ"ר דְּנִשְׂרָפִין כֵּיוָן שֶׁצִּוָּה הַכָּתוּב לְשָׂרְפָן אַחַר שֶׁעָשָׂה כְּאִלּוּ נַעֲשֵׂית מִצְוָתָן, וְאֵין לְךָ דָּבָר שֶׁנַּעֲשֵׂית מִצְוָתוֹ וּמוֹעֲלִין בּוֹ, הָכִי נַמִּי כֵּיוָן שֶׁנַּעֲשֵׂית מִצְוָתוֹ הָלַךְ אִסּוּרֵיהּ. אֲבָל הַנִּקְבָּרִים דְּלֹא הִטְעִין הַכָּתוּב לְשָׂרְפָן מָשׁוּךְ אִסּוּרַיְהוּ לְעוֹלָם. תּוֹסָפוֹת: {כ} וְכֵיוָן דְּטָעוּן גְּנִיזָה אָסוּר בַּהֲנָאָה. רַשִׁ"י:  {כא} <B> אֵינוֹ כוּ'.</B> שֶׁאַף עַל פִּי שֶׁהוּא מַחֲמִיר בִּשְׂרֵפָתָהּ. הֲרֵי הֵקֵל בַּאֲפָרָן. שֶׁאֵפֶר הַנִּקְבָּרִים אָסוּר. הָרַמְבַּ"ם: {ט} תֵּימַהּ, דִּלְמַאי נַפְקָא מִנַּהּ מִתְחַשְּׁבִין בְּאַפֵּי נַפְשַׁיְהוּ כוּ'. אֲבָל הָרַמְבַּ"ם כָּתַב וּמְחַשְּׁבִין לָהּ מַה שֶּׁרָאוּי לָהֶם כוּ'. וּלְפִיכָךְ נִרְאֶה לִי לְהַגִּיהַּ בְּדִבְרֵי הָרַ"ב וְנִפְדִּין בְּשָׁוְיָן וְהַשְּׁאָר לְפִי חֶשְׁבּוֹן. וְעוֹד יֵשׁ לוֹמַר, דְּדַעְתּוֹ כְדַעַת הַתּוֹסָפוֹת דְּהַנָּפְקָא מִנַּהּ שֶׁלֹּא חָשְׁבִינַן לְהוּ הַמִּדְרוֹן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
